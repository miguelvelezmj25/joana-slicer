804     /**
805      * Initialize the environment, including running recovery, if it is not
806      * already initialized.
807      *
808      * Note that this method should be called even when opening additional
809      * handles for an already initialized environment.  If initialization is
810      * still in progress then this method will block until it is finished.
811      *
812      * @return true if we are opening the first handle for this environment and
813      * recovery is run (when ENV_RECOVERY is configured to true); false if we
814      * are opening an additional handle and recovery is not run.
815      */
816     public synchronized boolean finishInit(EnvironmentConfig envConfig)
817         throws DatabaseException {
818 
819         if (initializedSuccessfully) {
820             return false;
821         }
822 
823         boolean success = false;
824         try {
825 
826             /*
827              * Do not do recovery if this environment is for a utility that
828              * reads the log directly.
829              */
830             final boolean doRecovery =
831                 configManager.getBoolean(EnvironmentParams.ENV_RECOVERY);
832             if (doRecovery) {
833 
834                 /*
835                  * Run recovery.  Note that debug logging to the database log
836                  * is disabled until recovery is finished.
837                  */
838                 boolean recoverySuccess = false;
839                 try {
840                     RecoveryManager recoveryManager =
841                         new RecoveryManager(this);
842                     recoveryManager.recover(isReadOnly);
843 
844                     postRecoveryConversion();
845                     recoverySuccess = true;
846                 } finally {
847                     try {
848 
849                         /*
850                          * Flush to get all exception tracing out to the log.
851                          */
852                         logManager.flushSync();
853                         fileManager.clear();
854                     } catch (IOException e) {
855                         /* Ignore second order exceptions. */
856                         if (recoverySuccess) {
857                             throw new EnvironmentFailureException
858                                 (this, EnvironmentFailureReason.LOG_INTEGRITY,
859                                  e);
860                         }
861                     } catch (Exception e) {
862                         if (recoverySuccess) {
863                             throw EnvironmentFailureException.
864                                 unexpectedException(this, e);
865                         }
866                     }
867                 }
868             } else {
869                 isReadOnly = true;
870 
871                 /*
872                  * Normally when recovery is skipped, we don't need to
873                  * instantiate comparators.  But even without recovery, some
874                  * utilities such as DbScavenger need comparators.
875                  */
876                 if (!configManager.getBoolean
877                         (EnvironmentParams.ENV_COMPARATORS_REQUIRED)) {
878                     noComparators = true;
879                 }
880             }
881 
882             /*
883              * Cache a few critical values. We keep our timeout in millis
884              * because Object.wait takes millis.
885              */
886             lockTimeout =
887                 configManager.getDuration(EnvironmentParams.LOCK_TIMEOUT);
888             txnTimeout =
889                 configManager.getDuration(EnvironmentParams.TXN_TIMEOUT);
890 
891             /*
892              * Initialize the environment memory usage number. Must be called
893              * after recovery, because recovery determines the starting size of
894              * the in-memory tree.
895              */
896             memoryBudget.initCacheMemoryUsage
897                 (dbMapTree.getTreeAdminMemory());
898 
899             /*
900              * Call config observer and start daemons last after everything
901              * else is initialized. Note that all config parameters, both
902              * mutable and non-mutable, needed by the memoryBudget have already
903              * been initialized when the configManager was instantiated.
904              */
905             envConfigUpdate(configManager, envConfig);
906 
907             /*
908              * Mark initialized before creating the internal env, since
909              * otherwise a we'll recurse and attempt to create another
910              * EnvironmentImpl.
911              */
912             initializedSuccessfully = true;
913 
914             if (doRecovery) {
915 
916                 /*
917                  * Perform dup database conversion after recovery and other
918                  * initialization is complete, but before running daemons.
919                  */
920                 convertDupDatabases();
921 
922                 /* Create internal env before SyncCleanerBarrier. */
923                 envInternal = createInternalEnvironment();
924             }
925 
926             /*
927              * Mark as open before starting daemons. Note that this will allow
928              * background eviction threads to run, so it should not be done
929              * until we are ready for multi-threaded access.
930              */
931             open();
932 
933             runOrPauseDaemons(configManager);
934             success = true;
935             return true;
936         } finally {
937             if (!success) {
938                 /* Release any environment locks if there was a problem. */
939                 clearFileManager();
940                 closeHandlers();
941             }
942 
943             /*
944              * DbEnvPool.addEnvironment is called by RecoveryManager.buildTree
945              * during recovery above, to enable eviction during recovery.  If
946              * we fail to create the environment, we must remove it.
947              */
948             if (!success && sharedCache) {
949                 evictor.removeSharedCacheEnv(this);
950             }
951 
952             startupTracker.stop(Phase.TOTAL_ENV_OPEN);
953             startupTracker.setProgress(RecoveryProgress.RECOVERY_FINISHED);
954         }
955     }

1300      * exceeded when the given amount written is added to the cumulative total,

1302      * background activities such as the checkpointer and evictor after
1303      * performing a file write operation.
1304      *
1305      * <p>The number of writes is estimated by dividing the bytes written by
1306      * the log buffer size.  Since the log write buffer is shared by all

1308      *

1310      */
1311     public void updateBackgroundWrites(int writeSize, int logBufferSize) {
1312

1315          * between the time we check it and the time we use it below.
1316          */
1317         int limit = backgroundWriteLimit;
1318         if (limit > 0) {
1319             synchronized (backgroundTrackingMutex) {
1320                 backgroundWriteBytes += writeSize;
1321                 int writeCount = (int) (backgroundWriteBytes / logBufferSize);
1322                 if (writeCount >= limit) {
1323                     backgroundSleepBacklog += 1;
1324                     /* Remainder is rolled forward. */
1325                     backgroundWriteBytes -= (limit * logBufferSize);
1326                     assert backgroundWriteBytes >= 0;

1332     /**
1333      * If the sleep backlog is non-zero (set by updateBackgroundReads or
1334      * updateBackgroundWrites), sleep for the configured interval and decrement
1335      * the backlog.
1336      *
1337      * <p>If two threads call this method and the first call causes a sleep,
1338      * the call by the second thread will block until the first thread's sleep
1339      * interval is over.  When the call by the second thread is unblocked, if
1340      * another sleep is needed then the second thread will sleep again.  In
1341      * other words, when lots of sleeps are needed, background threads may
1342      * backup.  This is intended to give foreground threads a chance to "catch
1343      * up" when background threads are doing a lot of IO.</p>
1344      */
1345     public void sleepAfterBackgroundIO() {
1346         if (backgroundSleepBacklog > 0) {
1347             synchronized (backgroundSleepMutex) {
1348                 /* Sleep. Rethrow interrupts if they occur. */
1349                 try {
1350                     /* FindBugs: OK that we're sleeping with a mutex held. */
1351                     Thread.sleep(backgroundSleepInterval);
1352                 } catch (InterruptedException e) {
1353                     Thread.currentThread().interrupt();
1354                 }
1355                 /* Assert has intentional side effect for unit testing. */
1356                 assert TestHookExecute.doHookIfSet(backgroundSleepHook);
1357             }
1358             synchronized (backgroundTrackingMutex) {
1359                 /* Decrement backlog last to make other threads wait. */
1360                 if (backgroundSleepBacklog > 0) {
1361                     backgroundSleepBacklog -= 1;
1362                 }
1363             }
1364         }
1365     }
