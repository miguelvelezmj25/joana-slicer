31 /**
32  * This class is used to maintain two pieces of persistent state. The
33  * replication group node identifier of the Arbiter and a VLSN value that
34  * represents the highest commit record VLSN the Arbiter has acknowledged.
35  */
36 class ArbiterVLSNTracker {
37     private static final int VERSION = 1;

39     private RandomAccessFile raf;
40     private final File dataFile;
41     private VLSN currentVLSN = VLSN.NULL_VLSN;
42     private volatile VLSN dtvlsn = VLSN.NULL_VLSN;
43     private final int VERSION_OFFSET = 0;
44     private final int NODEID_OFFSET = Integer.SIZE + VERSION_OFFSET;
45     private final int DATA_OFFSET = Integer.SIZE + NODEID_OFFSET;
46     private int nodeId = NameIdPair.NULL_NODE_ID;
47     private final StatGroup stats;
48     private final LongStat nWrites;
49     private final LongStat nFSyncs;
50     private final LongStat vlsnStat;
51     private final LongStat dtVlsnStat;

53     ArbiterVLSNTracker(File file) {
54         dataFile = file;
55         boolean fileExists = dataFile.exists();
56 
57         stats = new StatGroup(ArbiterStatDefinition.ARBIO_GROUP_NAME,
58                               ArbiterStatDefinition.ARBIO_GROUP_DESC);
59         nFSyncs = new LongStat(stats, ARB_N_FSYNCS);
60         nWrites = new LongStat(stats, ARB_N_WRITES);
61         vlsnStat = new LongStat(stats, ARB_VLSN);
62         dtVlsnStat = new LongStat(stats, ARB_DTVLSN);
63         try {
64             raf = new RandomAccessFile(dataFile, "rw");
65             if (fileExists) {
66                 final int readVersion = readVersion();
67                 if (readVersion > VERSION) {
68                     throw new RuntimeException(
69                         "Arbiter data file does not have a supported " +
70                         "version field " +
71                         dataFile.getAbsolutePath());
72                 }
73                 nodeId = readNodeId();
74                 if (raf.length() > DATA_OFFSET) {
75                     raf.seek(DATA_OFFSET);
76                     currentVLSN = new VLSN(raf.readLong());
77                     dtvlsn =  new VLSN(raf.readLong());
78                 }
79             } else {
80                 writeVersion(VERSION);
81                 writeNodeIdInternal(nodeId);
82             }
83         } catch (IOException e) {
84             throw new RuntimeException(
85                 "Unable to read the Arbiter data file " +
86                 dataFile.getAbsolutePath());
87         }
88         catch (Exception e) {
89             throw new RuntimeException(
90                 "Unable to open the Arbiter data file " +
91                 dataFile.getAbsolutePath() + " exception " + e.getMessage());
92         }
93     }

188     public synchronized void write(VLSN nextCurrentVLSN,
189                                    VLSN nextDTVLSN,
190                                    boolean doFSync) {
191         if (raf == null) {
192             throw new RuntimeException(
193                 "Internal error: Unable to write the Arbiter data file " +
194                 " because the file is not open." +
195                 dataFile.getAbsolutePath());
196         }
197         if (nextCurrentVLSN.compareTo(currentVLSN) > 0) {
198             this.currentVLSN = nextCurrentVLSN;
199             this.dtvlsn = nextDTVLSN;
200             try {
201                 raf.seek(DATA_OFFSET);
202                 raf.writeLong(nextCurrentVLSN.getSequence());
203                 raf.writeLong(nextDTVLSN.getSequence());
204                 nWrites.add(2);
205                 if (doFSync) {
206                     doFSync();
207                 }
208             } catch (IOException e) {
209                 throw new RuntimeException(
210                     "Unable to write the Arbiter data file " +
211                     dataFile.getAbsolutePath());
212             }
213         }
214     }
