367             (length < 0) ||
368             (offset > toSend.length - length)) {
369             throw new IndexOutOfBoundsException();
370         }
371

373         for (int i = offset; i < offset + length; ++i) {
374             toSendRemaining += toSend[i].remaining();
375         }
376         if (toSendRemaining == 0) {
377             return 0;
378         }
379         final int toSendTotal = toSendRemaining;
380 
381         /*

383          * a flush to clear out the network transmit buffer.
384          */
385         flush_internal();
386

399                      * still receive this because flush_internal isn't

431      * Attempt to flush any pending writes to the underlying socket buffer.
432      * The caller should ensure that it is the only thread accessing the
433      * DataChannel in order that the return value be meaningful.
434      *
435      * @return flush status
436      */
437     @Override
438     public FlushStatus flush()
439         throws IOException {
440 
441         int n = flush_internal();
442         if (writeLock.tryLock()) {
443             try {
444                 HandshakeStatus hstatus =
445                     sslEngine.getHandshakeStatus();
446                 switch (hstatus) {
447                 case NEED_TASK:
448                     return FlushStatus.NEED_TASK;
449                 case NEED_UNWRAP:
450                     return FlushStatus.NEED_READ;
451                 case NEED_WRAP:
452                     /*
453                      * We should not be here if we are the only thread doing
454                      * handshake, so there must be another thread, they will
455                      * flush after they wrap, our job is done here.
456                      */
457                     return FlushStatus.DONE;
458                 case FINISHED:
459                 case NOT_HANDSHAKING:
460                     break;
461                 default:
462                     assert false : "Unexpected handshake status.";
463                 }
464 
465                 if (n == 0) {
466                     /*
467                      * It is possible that there was nothing to flush last time
468                      * we flushed but someone wrote something before we
469                      * acquired the lock, so we flush again here
470                      */
471                     n = flush_internal();
472                 }
473 
474                 final int pos = netXmitBuffer.position();
475 
476                 if (pos == 0) {
477                     return FlushStatus.DONE;
478                 }
479 
480                 if (n != 0) {
481                     return FlushStatus.AGAIN;
482                 }
483 
484                 /* Here n == 0 and pos != 0, i.e., socket write busy. */
485                 return FlushStatus.WRITE_BUSY;
486             } finally {
487                 writeLock.unlock();
488             }
489         }
490 
491         /*
492          * If we weren't able to acquire the write lock, we can't be sure that
493          * everything has been flushed, and there's a good chance that someone
494          * else is writing (which the caller should have protected against in
495          * order to get a reliable answer). Just ask the caller to flush again.
496          */
497         return FlushStatus.AGAIN;
498     }

505 
506         int count = 0;
507 
508         /*
509          * Don't insist on getting a lock.  If someone else has it, they will
510          * probably flush it for us.
511          */
512         if (writeLock.tryLock()) {
513             try {
514                 if (netXmitBuffer.position() == 0) {
515                     return 0;
516                 }
517                 netXmitBuffer.flip();
518 
519                 /*
520                  * try/finally to keep things clean, in case the socket channel
521                  * gets closed
522                  */
523                 try {
524                     count = socketChannel.write(netXmitBuffer);
525                 } finally {
526                     netXmitBuffer.compact();
527                 }
528             } finally {
529                 writeLock.unlock();
530             }
531         }

570      * buffers.
571      *
572      * @param src the source ByteBuffer - it is expected to be ready for a get.
573      * @param dsts the destination array of ByteBuffers, each of which is
574      * expected to be ready for a put.
575      * @param offset the offset within the buffer array of the first buffer
576      * into which bytes are to be transferred.
577      * @param length the maximum number of buffers to be accessed
578      * @return The number of bytes transfered from src to dst
579      */
580     private int transfer(ByteBuffer src,
581                          ByteBuffer[] dsts,
582                          int offset,
583                          int length) {
584 
585         int transferred = 0;
586         for (int i = offset; i < offset + length; ++i) {
587             final ByteBuffer dst = dsts[i];
588             final int space = dst.remaining();
589 
590             if (src.remaining() > space) {
591                 /* not enough room for it all */
592                 final ByteBuffer slice = src.slice();
593                 slice.limit(space);
594                 dst.put(slice);
595                 src.position(src.position() + space);
596                 transferred += space;
597             } else {
598                 transferred += src.remaining();
599                 dst.put(src);
600                 break;
601             }
602         }
603         return transferred;
604     }

617     /*
618      * Attempt a handshake step.
619      *
620      * @return true if it is appropriate to call this again immediately.
621      */
622     private boolean processOneHandshake() throws IOException {
623 
624         int readCount = 0;
625         int flushCount = 0;
626         SSLEngineResult engineResult = null;
627 
628         switch (sslEngine.getHandshakeStatus()) {
629         case FINISHED:
630             /*
631              * Just finished handshaking. We shouldn't actually see this here
632              * as it is only supposed to be produced by a wrap or unwrap.
633              */
634             return false;
635 
636         case NEED_TASK:
637             /*
638              * Need results from delegated tasks before handshaking can
639              * continue, so do them now.  We assume that the tasks are done
640              * inline, and so we can return true here.
641              */
642             runDelegatedTasks();
643             return true;
644 
645         case NEED_UNWRAP:
646             {
647                 boolean unwrapped = false;
648 
649                 /* Attempt to flush anything that is pending */
650                 try {
651                     flush_internal();
652                 } catch (SocketException se) {
653                 }
654 
655                 /*
656                  * Attempt to process anything that is pending in the
657                  * netRecvBuffer.
658                  */
659                 readLock.lock();
660                 try {
661                     if (netRecvBuffer.position() > 0) {
662                         netRecvBuffer.flip();
663                         engineResult =
664                             sslEngine.unwrap(netRecvBuffer, appRecvBuffer);
665                         netRecvBuffer.compact();
666                         if (engineResult.getStatus() == Status.OK) {
667                             unwrapped = true;
668                         }
669                     }
670 
671                     if (!unwrapped && !sslEngine.isInboundDone()) {
672                         /*
673                          * Either we had nothing in the netRecvBuffer or there
674                          * was not enough data to unwrap, so let's try getting
675                          * some more.
676                          *
677                          * If a re-negotiation is happening and the
678                          * appRecvBuffer was full, we could have received a
679                          * BUFFER_OVERFLOW engineResult, in which case a read()
680                          * is not really helpful here, but it's harmless and is
681                          * a rare occurrence, so we won't worry about it.
682                          */
683                         readCount = socketChannel.read(netRecvBuffer);
684                         if (readCount < 0) {
685                             try {
686                                 sslEngine.closeInbound();
687                                 sslInboundClosed = true;
688                             } catch (SSLException ssle) {
689                                 // ignore
690                             }
691                         }
692 
693                         netRecvBuffer.flip();
694                         engineResult =
695                             sslEngine.unwrap(netRecvBuffer, appRecvBuffer);
696                         netRecvBuffer.compact();
697                     }
698                 } finally {
699                     readLock.unlock();
700                 }
701             }
702 
703             break;
704 
705         case NEED_WRAP:
706             /*
707              * Must send data to the remote side before handshaking can
708              * continue, so wrap() must be called.
709              */
710             writeLock.lock();
711             try {
712                 engineResult = sslEngine.wrap(emptyXmitBuffer, netXmitBuffer);
713             } finally {
714                 writeLock.unlock();
715             }
716 
717             if (engineResult.getStatus() == Status.CLOSED) {
718                 /*
719                  * If the engine is already closed, flush may fail, and that's
720                  * ok, so squash any exceptions that happen
721                  */
722                 try {
723                     /* ignore the flush count */
724                     flush_internal();
725                 } catch (SocketException se) {
726                 }
727             } else {
728                 flushCount = flush_internal();
729             }
730             break;
731 
732         case NOT_HANDSHAKING:
733             /* Not currently handshaking */
734             return false;
735         }
736 
737         /*
738          * We may have done a wrap or unwrap above.  Check the engineResult
739          */
740 
741         if (engineResult != null) {
742             if (engineResult.getHandshakeStatus() == HandshakeStatus.FINISHED) {
743                 /*
744                  * Handshaking just completed.   Here is our chance to do any
745                  * session validation that might be required.
746                  */
747                 if (sslEngine.getUseClientMode()) {
748                     if (hostVerifier != null) {
749                         peerTrusted =
750                             hostVerifier.verify(targetHost,
751                                                 sslEngine.getSession());
752                         if (peerTrusted) {
753                             logger.log(FINE,
754                                           "SSL host verifier reports that " +
755                                           "connection target is valid");
756                         } else {
757                             logger.log(INFO,
758                                        "SSL host verifier reports that " +
759                                        "connection target is NOT valid");
760                             throw new IOException(
761                                 "Server identity could not be verified");
762                         }
763                     }
764                 } else {
765                     if (authenticator != null) {
766                         peerTrusted =
767                             authenticator.isTrusted(sslEngine.getSession());
768                         if (peerTrusted) {
769                             logger.log(FINE,
770                                        "SSL authenticator reports that " +
771                                        "channel is trusted");
772                         } else {
773                             logger.log(INFO,
774                                        "SSL authenticator reports that " +
775                                        "channel is NOT trusted");
776                         }
777                     }
778                 }
779             }
780 
781             switch (engineResult.getStatus()) {
782             case BUFFER_UNDERFLOW:
783                 /*
784                  * This must have resulted from an unwrap, meaning we need to
785                  * do another read.  If the last read did something useful,
786                  * tell the caller to call us again.
787                  */
788                 return readCount > 0;
789 
790             case BUFFER_OVERFLOW:
791                 /*
792                  * Either we were processing an unwrap and the appRecvBuffer is
793                  * full or we were processing a wrap and the netXmitBuffer is
794                  * full.  For the unwrap case, the only way we can make progress
795                  * is for the application to receive control.  For the wrap
796                  * case, we may be able to make progress if the flush
797                  * did something useful.
798                  */
799                 if ((sslEngine.getHandshakeStatus() ==
800                      HandshakeStatus.NEED_WRAP) &&
801                     flushCount > 0) {
802                     return true;
803                 }
804                 return false;
805 
806             case CLOSED:
807                 if (sslEngine.isOutboundDone()) {
808                     try {
809                         socketChannel.socket().shutdownOutput();
810                     } catch (Exception e) {
811                     }
812                 }
813                 return false;
814 
815             case OK:
816                 break;
817             }
818         }
819 
820         /*
821          * Tell the caller to try again.  Cases where no handshake progress
822          * can be made should return false above.
823          */
824         return true;
825     }
