112      * whichever of this map or the argument is newer, with those entries
113      * updated with any values from both maps.  Treats this map as newest if
114      * both have the same timestamp.  This method does not compute negative
115      * intervals, since negation does not work properly for this non-additive
116      * stat.  The base argument must be a LongAvgRateMapStat.
117      */
118     @Override
119     public LongAvgRateMapStat computeInterval(Stat<String> base) {
120         assert base instanceof LongAvgRateMapStat;
121         final LongAvgRateMapStat copy = copy();
122         final LongAvgRateMapStat baseCopy =
123             (LongAvgRateMapStat) base.copy();
124         if (copy.getLatestTime() < baseCopy.getLatestTime()) {
125             return copy.updateLatest(baseCopy);
126         }
127         return baseCopy.updateLatest(copy);
128     }

130     /**
131      * Update this map to reflect changes from the argument, including merging
132      * latest changes, removing entries not in the argument, and adding ones
133      * not in this instance.
134      */
135     private synchronized LongAvgRateMapStat updateLatest(
136         final LongAvgRateMapStat latest) {
137 
138         synchronized (latest) {
139             for (final Iterator<Entry<String, LongAvgRate>> i =
140                      statMap.entrySet().iterator();
141                  i.hasNext(); ) {
142                 final Entry<String, LongAvgRate> e = i.next();
143                 final LongAvgRate latestStat =
144                     latest.statMap.get(e.getKey());
145                 if (latestStat != null) {
146                     e.getValue().add(latestStat);
147                 } else {
148                     i.remove();
149                 }
150             }
151 
152             for (final Entry<String, LongAvgRate> e :
153                      latest.statMap.entrySet()) {
154                 final String key = e.getKey();
155                 if (!statMap.containsKey(key)) {
156                     statMap.put(key, e.getValue());
157                 }
158             }
159         }
160         return this;
161     }

163     /**

165      * the time of the latest stat removal.
166      */
167     private synchronized long getLatestTime() {
168         long latestTime = removeStatTimestamp;
169         for (final LongAvgRate stat : statMap.values()) {
170             latestTime = Math.max(latestTime, stat.getPrevTime());
171         }
