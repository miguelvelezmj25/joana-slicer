588     // Inherit javadoc
589     public int size() {
590 
591         boolean countDups = iterateDuplicates();
592         if (DbCompat.DATABASE_COUNT && countDups && !view.range.hasBound()) {
593             try {
594                 return (int) DbCompat.getDatabaseCount(view.db);
595             } catch (Exception e) {
596                 throw StoredContainer.convertException(e);
597             }
598         } else {
599             int count = 0;
600             CursorConfig cursorConfig = view.currentTxn.isLockingMode() ?
601                 CursorConfig.READ_UNCOMMITTED : null;
602             DataCursor cursor = null;
603             /* Auto-commit is not needed because ReadUncommitted is used. */
604             try {
605                 cursor = new DataCursor(view, false, cursorConfig);
606                 OperationStatus status = cursor.getFirst(false);
607                 while (status == OperationStatus.SUCCESS) {
608                     if (countDups) {
609                         count += cursor.count();
610                     } else {
611                         count += 1;
612                     }
613                     status = cursor.getNextNoDup(false);
614                 }
615             } catch (Exception e) {
616                 throw StoredContainer.convertException(e);
617             } finally {
618                 closeCursor(cursor);
619             }
620             return count;
621         }
622     }
