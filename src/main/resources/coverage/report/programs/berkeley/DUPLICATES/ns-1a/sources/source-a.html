72 /**
73  * This class is used to encapsulate all access to the rep group data that is
74  * present in every replicated JE environment. The rep group data exists
75  * primarily to support dynamic group membership. Both read and update access
76  * must be done through the APIs provided by this class.
77  *
78  * The database is simply a representation of the RepGroup. Each entry in the
79  * database represents a node in RepGroup; the key is the String node name, and
80  * the data is the serialized ReplicationNode.  There is a special entry keyed
81  * by GROUP_KEY that holds the contents of the RepGroup (excluding the nodes)
82  * itself.
83  *
84  * The database may be modified concurrently by multiple transactions as a
85  * master processes requests to update it. It may also be accessed by multiple
86  * overlapping transactions as a Replica replays the rep stream. These updates
87  * need to be interleaved with operations like getGroup() that create copies of
88  * the RepGroup instance. To avoid deadlocks, entries in the database are
89  * accessed in order of ascending key. GROUP_KEY in particular is associated
90  * with the lowest key value so that it's locked first implicitly as part of
91  * any iteration and any other modifications to the database must first lock it
92  * before making changes to the group itself.
93  *
94  * An instance of this class is created as part of a replication node and is
95  * retained for the entire lifetime of that node.
96  */

342      * database, using the txn and handles that were passed in.

345                                            DatabaseImpl dbImpl,
346                                            Txn txn)
347         throws DatabaseException {
348 
349         final DatabaseEntry keyEntry = new DatabaseEntry();
350         final DatabaseEntry value = new DatabaseEntry();
351         NodeBinding nodeBinding = null;
352         final GroupBinding groupBinding = new GroupBinding();
353 
354         RepGroupImpl group = null;

356             new HashMap<Integer, RepNodeImpl>();

358         cursorConfig.setReadCommitted(true);
359

361

364             while (mcursor.getNext(keyEntry, value, LockMode.DEFAULT) ==
365                    OperationStatus.SUCCESS) {

490     void ensureMember(RepNodeImpl ensureNode)
491         throws DatabaseException {
492 
493         if (ensureNode.getType().hasTransientId()) {
494             throw new IllegalArgumentException(
495                 "Attempt to call ensureMember on " + ensureNode.getType() +
496                 " node: " + ensureNode);
497         }
498         DatabaseImpl groupDbImpl;
499         try {
500             groupDbImpl = repImpl.getGroupDb();
501         } catch (DatabaseNotFoundException e) {
502             /* Should never happen. */
503             throw EnvironmentFailureException.unexpectedException(e);
504         }
505 
506         DatabaseEntry nodeNameKey = new DatabaseEntry();
507         StringBinding.stringToEntry(ensureNode.getName(), nodeNameKey);
508 
509         DatabaseEntry value = new DatabaseEntry();
510         NodeBinding mib = null;
511 
512         Txn txn = null;
513         Cursor cursor = null;
514         try {
515             txn = new ReadonlyTxn(repImpl, NO_ACK);
516 
517             /*
518              * Fetch the group so we know the group format version.  Read the
519              * group before reading the node entry in each case to avoid the
520              * potential of deadlocks caused by reversing the order of lock
521              * acquisition.
522              */
523             final RepGroupImpl repGroup =
524                 fetchGroupObject(txn, groupDbImpl, LockMode.DEFAULT);
525             mib = new NodeBinding(repGroup.getFormatVersion());
526 
527             CursorConfig config = new CursorConfig();
528             config.setReadCommitted(true);
529             cursor = makeCursor(groupDbImpl, txn, config);
530 
531             OperationStatus status =
532                 cursor.getSearchKey(nodeNameKey, value, null);
533             if (status == OperationStatus.SUCCESS) {
534                 /* Let's see if the entry needs updating. */
535                 RepNodeImpl miInDb = mib.entryToObject(value);
536                 if (miInDb.equivalent(ensureNode)) {
537                     if (miInDb.isQuorumAck()) {
538                         /* Present, matched and acknowledged. */
539                         return;
540                     }
541                     ensureNode.getNameIdPair().update(miInDb.getNameIdPair());
542                     /* Not acknowledged, retry the update. */
543                 } else {
544                     /* Present but not equivalent. */
545                     LoggerUtils.warning(logger, repImpl,
546                                         "Incompatible node descriptions. " +
547                                         "Membership database definition: " +
548                                         miInDb.toString() +
549                                         " Transient definition: " +
550                                         ensureNode.toString());
551                     if (ensureNode.getType() != miInDb.getType()) {
552                         throw EnvironmentFailureException.unexpectedState(
553                             "Conflicting node types for node " +
554                             ensureNode.getName() +
555                             ": expected " + ensureNode.getType() +
556                             ", found " + miInDb.getType());
557                     }
558                     throw EnvironmentFailureException.unexpectedState(
559                         "Incompatible node descriptions for node: " +
560                         ensureNode.getName() + ", node ID: " +
561                         ensureNode.getNodeId());
562                 }
563                 LoggerUtils.info(logger, repImpl,
564                                  "Present but not ack'd node: " +
565                                  ensureNode.getNodeId() +
566                                  " ack status: " + miInDb.isQuorumAck());
567             }
568             cursor.close();
569             cursor = null;
570             txn.commit();
571             txn = null;
572         } finally {
573             if (cursor != null) {
574                 cursor.close();
575             }
576 
577             if (txn != null) {
578                 txn.abort();
579             }
580 
581         }
582         createMember(ensureNode);
583 
584         /* Refresh group and Fire an ADD GroupChangeEvent. */
585         refreshGroupAndNotifyGroupChange
586             (ensureNode.getName(), GroupChangeType.ADD);
587     }

623                 node.setRemoved(true);
624                 saveNodeObject(txn, node, groupDbImpl, repGroup);
625             }
626             /** Override phase 2 to delete the node entry if delete is true. */
627             @Override
628             void phase2Body() {
629                 if (!delete) {
630                     super.phase2Body();
631                     return;
632                 }
633                 final DatabaseEntry nodeNameKey = new DatabaseEntry();
634                 StringBinding.stringToEntry(removeNode.getName(), nodeNameKey);
635                 final Cursor cursor =
636                     makeCursor(groupDbImpl, txn, CursorConfig.DEFAULT);
637                 try {
638                     final OperationStatus status = cursor.getSearchKey(
639                         nodeNameKey, new DatabaseEntry(), LockMode.RMW);
640                     if (status != OperationStatus.SUCCESS) {
641                         throw EnvironmentFailureException.unexpectedState(
642                             "Node ID: " + removeNode.getNameIdPair() +
643                             " not present in group db");
644                     }
645                     cursor.delete();
646                 } finally {
647                     cursor.close();
648                 }
649             }
650         };
651 
652         twoPhaseUpdate.execute();
653 
654         /* Refresh group and fire a REMOVE GroupChangeEvent. */
655         refreshGroupAndNotifyGroupChange
656             (removeNode.getName(), GroupChangeType.REMOVE);
657

659                          "Successfully deleted node: " + removeNode.getName());

721     /**
722      * Implements the two phase update of membership information.
723      *
724      * In the first phase the master repeatedly tries to commit the "put"
725      * operation until it gets a Quorum of acks, ensuring that the operation
726      * has been made durable. Nodes that obtain this entry will start using it
727      * in elections. However, the node itself will not participate in elections
728      * until it has successfully completed phase 2.
729      *
730      * In the second phase, the entry for the member is updated to note
731      * that a quorum of acks was received.
732      *
733      * Failure leaves the database with the member info absent, or
734      * present but without the update to quorumAcks indicating that a
735      * quorum has acknowledged the change.
736      *
737      * @param node the member info for the node.
738      * @param quorumAck whether to require acknowledgments from a quorum
739      *
740      * @throws InsufficientReplicasException upon failure of 2p member update
741      * @throws InsufficientAcksException upon failure of 2p member update
742      * @throws DatabaseException upon failure.
743      */
744     private void twoPhaseMemberUpdate(final RepNodeImpl node,
745                                       final boolean quorumAck)
746         throws DatabaseException {
747 
748         TwoPhaseUpdate twoPhaseUpdate = new TwoPhaseUpdate(node, quorumAck) {
749             int saveOrigId = node.getNameIdPair().getId();
750 
751             @Override
752             void phase1Body() {
753                 RepGroupImpl repGroup =
754                     fetchGroupObject(txn, groupDbImpl, LockMode.RMW);
755                 repGroup = fetchGroup(repGroup.getName(), groupDbImpl, txn);
756                 int changeVersion = repGroup.incrementChangeVersion();
757                 if (node.getNameIdPair().hasNullId()) {
758                     node.getNameIdPair().setId(repGroup.getNextNodeId());
759                 }
760                 repGroup.checkForConflicts(node);
761                 saveGroupObject(txn, repGroup, groupDbImpl);
762                 node.setChangeVersion(changeVersion);
763                 final RepNodeImpl existingNode =
764                     repGroup.getNode(node.getName());
765                 if ((existingNode != null) && (node.getJEVersion() == null)) {
766                     node.updateJEVersion(existingNode.getJEVersion());
767                 }
768                 saveNodeObject(txn, node, groupDbImpl, repGroup);
769             }
770 
771             @Override
772             void deadlockHandler() {
773                 node.getNameIdPair().setId(saveOrigId, false);
774             }
775 
776             @Override
777             void insufficientReplicasHandler() {
778                 node.getNameIdPair().setId(saveOrigId, false);
779             }
780         };
781 
782         twoPhaseUpdate.execute();
783     }

785     /**
786      * This method is not used when the CBVLSN is defunct -- see GlobalCBVLSN.
787      * This method was not moved to GlobalCBVLSN to avoid modularity problems.
788      *
789      * Updates the database entry associated with the node with the new local
790      * CBVLSN, if it can do so without encountering lock contention, and unless
791      * the node is a secondary, arbiter, or external node.  Also updates the
792      * rep node's transient  group information about the global CBVLSN. If it
793      * encounters contention, it returns false, and the caller must retry at
794      * some later point in time.
795      *
796      * Note that changes to the local CBVLSN do not update the group version
797      * number since they do not impact group membership.
798      *
799      * @param nameIdPair identifies the node being updated
800      * @param newCBVLSN the new local CBVLSN to be associated with the node.
801      * @param nodeType the node type of the RepNode
802      * @return true if the update succeeded.
803      * @throws DatabaseException
804      */
805     public boolean updateLocalCBVLSN(final NameIdPair nameIdPair,
806                                      final VLSN newCBVLSN,
807                                      final NodeType nodeType)
808         throws DatabaseException {
809 
810         DatabaseImpl groupDbImpl = null;
811         try {
812             groupDbImpl = repImpl.probeGroupDb();
813         } catch (DatabaseException e) {
814             /* Contention on the groupDbImpl, try later. */
815             return false;
816         }
817 
818         if (groupDbImpl == null) {
819             /* Contention on the groupDbImpl, try later. */
820             return false;
821         }
822 
823         DatabaseEntry nodeNameKey = new DatabaseEntry();
824         StringBinding.stringToEntry(nameIdPair.getName(), nodeNameKey);
825         DatabaseEntry value = new DatabaseEntry();
826         final CleanerBarrierState barrierState =
827             new CleanerBarrierState(newCBVLSN, System.currentTimeMillis());
828         Txn txn = null;
829         Cursor cursor = null;
830         boolean ok = false;
831         try {
832 
833             /*
834              * No database update for secondary, arbiter, or external nodes,
835              * but set ok to true so that the rep node's group information is
836              * updated.
837              */
838             if (nodeType.isSecondary() || nodeType.isArbiter() ||
839                 nodeType.isExternal()) {
840                 ok = true;
841                 return true;
842             }
843 
844             final TransactionConfig txnConfig = new TransactionConfig();
845 
846             txnConfig.setDurability(NO_ACK_NO_SYNC_DURABILITY);
847             /*
848              * Don't wait for locks. It's ok to miss an update because we could
849              * not acquire the lock, since the operation will be retried later
850              * by a subsequent heartbeat message.
851              */
852             txnConfig.setNoWait(true);
853             txn = new MasterTxn(repImpl,
854                                 txnConfig,
855                                 repImpl.getNameIdPair());
856 
857             /* Read the group first to avoid deadlocks */
858             final RepGroupImpl repGroup =
859                 fetchGroupObject(txn, groupDbImpl, LockMode.DEFAULT);
860             cursor = makeCursor(groupDbImpl, txn, CursorConfig.DEFAULT);
861 
862             OperationStatus status =
863                     cursor.getSearchKey(nodeNameKey, value, LockMode.RMW);
864             if (status != OperationStatus.SUCCESS) {
865                 throw EnvironmentFailureException.unexpectedState
866                     ("Node ID: " + nameIdPair + " not present in group db");
867             }
868 
869             /* Let's see if the entry needs updating. */
870             final NodeBinding nodeBinding =
871                 new NodeBinding(repGroup.getFormatVersion());
872             final RepNodeImpl node = nodeBinding.entryToObject(value);
873             final VLSN lastCBVLSN = node.getBarrierState().getLastCBVLSN();
874             if (lastCBVLSN.equals(newCBVLSN)) {
875                 ok = true;
876                 return true;
877             }
878 
879             node.setBarrierState(barrierState);
880             nodeBinding.objectToEntry(node, value);
881             status = cursor.putCurrent(value);
882             if (status != OperationStatus.SUCCESS) {
883                 throw EnvironmentFailureException.unexpectedState
884                     ("Node ID: " + nameIdPair +
885                      " stored localCBVLSN could not be updated. Status: " +
886                      status);
887             }
888             LoggerUtils.fine(logger, repImpl,
889                              "Local CBVLSN updated to " + newCBVLSN +
890                              " for node " + nameIdPair);
891             ok = true;
892         } finally {
893             if (cursor != null) {
894                 cursor.close();
895             }
896 
897             if (txn != null) {
898                 if (ok) {
899                     txn.commit(NO_ACK_NO_SYNC_DURABILITY);
900                 } else {
901                     txn.abort();
902                 }
903                 txn = null;
904             }
905             if (ok) {
906                 /* RepNode may be null during shutdown. [#17424] */
907                 RepNode repNode = repImpl.getRepNode();
908                 if (repNode != null) {
909                     repNode.updateGroupInfo(nameIdPair, barrierState);
910                 }
911             }
912         }
913 
914         return true;
915     }

1102      * Supports the serialization/deserialization of node info into and out of
1103      * the database.  Nodes are always saved using the current group format
1104      * version, and the node's format version is checked on reading to make
1105      * sure it is not newer than the current group format version, although
1106      * they could have an older format version if they have not been saved
1107      * recently.
1108      *
1109      * <p>Prior to RepGroupImpl version 3, the second field was always the
1110      * ordinal value of the node type, which was either 0 or 1.  Starting with
1111      * version 3, values greater than 1 are treated as the rep group version of

1113      * in the next field, and the jeVersion field added at the end.

1118         static final int APPROX_MAX_SIZE =
1119             40 +                /* node name (guess) */
1120             4 +                 /* node ID */
1121             1 +                 /* group version */
1122             1 +                 /* NodeType */
1123             1 +                 /* quorumAck */
1124             1 +                 /* isRemoved */
1125             40 +                /* hostName (guess) */
1126             4 +                 /* port */
1127             8 +                 /* lastCBVLSN */
1128             8 +                 /* barrierTime */
1129             4 +                 /* changeVersion */
1130             10;                 /* jeVersion (approx) */
1131 
1132         /** The maximum node type value for version 2. */
1133         private static final int V2_MAX_NODE_TYPE = 1;
1134

1205             output.writeInt(mi.getPort());
1206             output.writeLong(syncState.getLastCBVLSN().getSequence());
1207             output.writeLong(syncState.getBarrierTime());
1208             output.writeInt(mi.getChangeVersion());

1210                 output.writeString(jeVersionString(mi.getJEVersion()));
1211             }

1215     /**
1216      * Implements two phase updates for membership changes to the group
1217      * database. It compartmentalizes the retry operations and exception
1218      * handling so that it's independent of the core logic.
1219      */
1220     private abstract class TwoPhaseUpdate {
1221 
1222         final RepNodeImpl node;
1223         final boolean quorumAck;
1224         final DatabaseImpl groupDbImpl;
1225 
1226         protected Txn txn;
1227         private DatabaseException phase1Exception = null;
1228 
1229         TwoPhaseUpdate(final RepNodeImpl node, final boolean quorumAck) {
1230             this.node = node;
1231             this.quorumAck = quorumAck;
1232             try {
1233                 groupDbImpl = repImpl.getGroupDb();
1234             } catch (DatabaseNotFoundException e) {
1235                 /* Should never happen. */
1236                 throw EnvironmentFailureException.unexpectedException(e);
1237             }
1238         }
1239 
1240         /* Phase1 exception handlers for phase1Body-specific cleanup */
1241         void insufficientReplicasHandler() {}
1242 
1243         void deadlockHandler() {}
1244 
1245         /* The changes to be made in phase1 */
1246         abstract void phase1Body();
1247 
1248         /* The changes to be made in phase2. */
1249         void phase2Body() {
1250             node.setQuorumAck(true);
1251             final RepGroupImpl repGroup =
1252                 fetchGroupObject(txn, groupDbImpl, LockMode.DEFAULT);
1253             saveNodeObject(txn, node, groupDbImpl, repGroup);
1254         }
1255 
1256         private void phase1()
1257             throws DatabaseException {
1258 
1259             for (int i = 0; i < QUORUM_ACK_RETRIES; i++ ) {
1260                 txn = null;
1261                 try {
1262                     txn = new MasterTxn(repImpl,
