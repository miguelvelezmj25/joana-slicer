21 /**
22  * Encapsulates the functionality around dealing with HostPort string pairs
23  * having the format:
24  *
25  *  host[:port]
26  */
27 
28 public class HostPortPair {
29 
30     static public final String SEPARATOR = ":";
31 
32     /**
33      * Parses a hostPort pair into the socket it represents.
34      * @param hostPortPair
35      * @return socket address for this host pair
36      *
37      * @throws IllegalArgumentException via ReplicatedEnvironment and Monitor
38      * ctors.
39      */
40     public static InetSocketAddress getSocket(String hostPortPair) {
41         if ("".equals(hostPortPair)) {
42             throw new IllegalArgumentException
43                 ("Host and port pair was missing");
44         }
45         int portStartIndex = hostPortPair.indexOf(SEPARATOR);
46         String hostName = hostPortPair;
47         int port = -1;
48         if (portStartIndex < 0) {
49             port = Integer.parseInt(RepParams.DEFAULT_PORT.getDefault());
50         } else {
51             hostName = hostPortPair.substring(0, portStartIndex);
52             port =
53                 Integer.parseInt(hostPortPair.substring(portStartIndex+1));
54         }
55         return new InetSocketAddress(hostName, port);
56     }
57 
58     /**
59      * Parses hostPort pairs into sockets it represents.
60      *
61      * @param hostPortPairs
62      *
63      * @return a set of socket addresses for these host pairs
64      */
65     public static Set<InetSocketAddress> getSockets(String hostPortPairs) {
66         Set<InetSocketAddress> helpers = new HashSet<InetSocketAddress>();
67         if (hostPortPairs != null) {
68             for (String hostPortPair : hostPortPairs.split(",")) {
69                 final String hpp = hostPortPair.trim();
70                 if (hpp.length() > 0) {
71                     helpers.add(getSocket(hpp));
72                 }
73             }
74         }
75 
76         return helpers;
77     }
78 
79     public static String getString(String host, int port) {
80         return host + SEPARATOR + port;
81     }

83     /**

87         int portStartIndex = hostPortPair.indexOf(SEPARATOR);

89                 hostPortPair :

91     }
92 
93     /**

96     public static int getPort(String hostPortPair) {
97         int portStartIndex = hostPortPair.indexOf(SEPARATOR);
98         return Integer.parseInt((portStartIndex < 0) ?
99                                 RepParams.DEFAULT_PORT.getDefault() :
100                                 hostPortPair.substring(portStartIndex+1));
101     }
