219     /*
220      * We can assume that deleteStart.getPrev() is either NULL_VLSN or is
221      * on a sync-able boundary. We can also assume that lastTxnEnd has not
222      * been changed. And lastly, we can assume that this range is not empty,
223      * since that was checked earlier on.

225     VLSNRange shortenFromEnd(final VLSN deleteStart) {
226         VLSN newLast = deleteStart.getPrev();
227 
228         assert newLast.compareTo(lastTxnEnd) >= 0 :
229         "Can't truncate at " + newLast +
230             " because it overwrites a commit at " +  lastTxnEnd;
231 
232         if (newLast.equals(NULL_VLSN)) {
233             return new VLSNRange(NULL_VLSN, NULL_VLSN, NULL_VLSN, NULL_VLSN);
234         }
235         return new VLSNRange(first, newLast, newLast, lastTxnEnd);
236     }

241     VLSNRange shortenFromHead(final VLSN deleteEnd) {
242

245         if (deleteEnd.compareTo(last) == 0) {
246             newFirst = NULL_VLSN;

251 
252         /* We shouldn't be truncating the last sync */
253         assert (lastSync.equals(NULL_VLSN) ||
254                 lastSync.compareTo(newFirst) >= 0 ) :
255             "Can't truncate lastSync= " + lastSync + " deleteEnd=" + deleteEnd;
256

258             lastTxnEnd : NULL_VLSN;

263     /**
264      * Compare two VLSNs, normalizing for NULL_VLSN. If one of them is
265      * NULL_VLSN, return the other one. If neither are NULL_VLSN, use the
266      * result of the comparison, expressed as the value of "better" to decide
267      * which one to return. If "better" is true, return "otherVLSN".
268      */
269     private VLSN getComparison(final VLSN thisVLSN,
270                                final VLSN otherVLSN,
271                                final boolean better) {
272         if (thisVLSN.equals(NULL_VLSN)) {
273             return otherVLSN;
274         }
275 
276         if (otherVLSN.equals(NULL_VLSN)) {
277             return thisVLSN;
278         }
279 
280         if (better) {
281             return otherVLSN;
282         }
283 
284         return thisVLSN;
285     }

338         if (first.equals(NULL_VLSN)) {

340                   (lastSync.equals(NULL_VLSN)) &&
341                   (lastTxnEnd.equals(NULL_VLSN)))) {

348         } else {
