73 /**
74  * The ArbiterAcker is used to acknowledge transactions. A feeder
75  * connection is established with the current master. Commit and Heartbeat
76  * messages are sent by the master. The ArbiterAcker responds and persistently
77  * tracks the high VLSN of the commit messages that it acknowledges.
78  *
79  * There are configuration parameters that are used.
80  * RepParams.REPLICA_MESSAGE_QUEUE_SIZE used for the replay queue size and
81  *                                      in the computation of the output
82  *                                      queue size.
83  * RepParams.REPLICA_TIMEOUT used for the Arbiter feeder channel timeout.
84  * RepParams.PRE_HEARTBEAT_TIMEOUT used for the Arbiter feeder channel timeout
85  *                                 before the first heartbeat is sent.
86  * RepParams.REPLICA_RECEIVE_BUFFER_SIZE used for the datachannel buffer size.
87  * RepParams.REPSTREAM_OPEN_TIMEOUT used for the datachannel open timeout.
88  * RepParams.MAX_CLOCK_DELTA - used for ReplicaFeederHandshake maximum clock
89  *                             delta.
90  * RepParams.HEARTBEAT_INTERVAL heartbeat interval in millis.
91  * RepParams.ENABLE_GROUP_ACKS enables output thread ack grouping.
92  *
93  * The main Arbiter thread reads messages from the feeder channel and queues
94  * the message on the request queue. The request thread reads entries
95  * from the request queue. The request thread may queue an entry on the
96  * output queue. The ArbiterOutputThread reads from the output queue and
97  * writes to the network channel.
98  *  read from network -> ArbiterAcker main thread  -> requestQueue
99  *  requestQueue      -> RequestThread             -> outputQueue
100  *  outputQueue       -> ArbiterOutputThread       -> writes to network
101  */

239     /**
240      * The core Arbiter control loop. The loop exits when it
241      * encounters one of the following possible conditions:
242      *
243      * 1) The connection to the master can no longer be maintained, due to
244      * connectivity issues, or because the master has explicitly shutdown its
245      * connections due to an election.
246      *
247      * 2) The node becomes aware of a change in master, that is, assertSync()
248      * fails.
249      *
250      * 3) The loop is interrupted, which is interpreted as a request to
251      * shutdown the Arbiter node as a whole.
252      *
253      * 4) It fails to establish its node information in the master as it
254      * attempts to join the replication group for the first time.
255      *
256      * Normal exit from this run loop results in the Arbiter node retrying
257      * finding the group master.
258      * A thrown exception, on the other hand, results in the Arbiter
259      * node as a whole terminating its operation and no longer participating in
260      * the replication group, that is, it enters the DETACHED state.
261      *
262      * @throws InterruptedException
263      * @throws DatabaseException if the environment cannot be closed/for a
264      * re-init
265      * @throws GroupShutdownException
266      */
267     void runArbiterAckLoop()
268         throws InterruptedException,
269                DatabaseException,
270                GroupShutdownException {
271 
272         Class<? extends RetryException> retryExceptionClass = null;
273         int retryCount = 0;
274         try {
275 
276             while (true) {
277                 try {
278                     runArbiterAckLoopInternal();
279                     /* Normal exit */
280                     break;
281                 } catch (RetryException e) {
282                     if (!arbiterImpl.getMasterStatus().inSync()) {
283                         LoggerUtils.fine(logger, repImpl,
284                                          "Retry terminated, out of sync.");
285                         break;
286                     }
287                     if ((e.getClass() == retryExceptionClass) ||
288                         (e.retries == 0)) {
289                         if (++retryCount >= e.retries) {
290                             /* Exit replica retry elections */
291                             LoggerUtils.info
292                                 (logger, repImpl,
293                                  "Failed to recover from exception: " +
294                                  e.getMessage() + ", despite " + e.retries +
295                                  " retries.\n" +
296                                  LoggerUtils.getStackTrace(e));
297                             break;
298                         }
299                     } else {
300                         retryCount = 0;
301                         retryExceptionClass = e.getClass();
302                     }
303                     LoggerUtils.fine(logger, repImpl, "Retry #: " +
304                                      retryCount + "/" + e.retries +
305                                      " Will retry Arbiter loop after " +
306                                      e.retrySleepMs + "ms. ");
307                     Thread.sleep(e.retrySleepMs);
308                     if (!arbiterImpl.getMasterStatus().inSync()) {
309                         break;
310                     }
311                 }
312             }
313         } finally {
314             arbiterImpl.resetReadyLatch(shutdownException);
315         }
316         /* Exit use elections to try a different master. */
317     }

340     private void runArbiterAckLoopInternal()
341         throws InterruptedException,
342                RetryException {
343 
344         shutdownException = null;
345         LoggerUtils.info(logger, repImpl,
346                          "Arbiter loop started with master: " +
347                          arbiterImpl.getMasterStatus().getNodeMasterNameId());
348         try {
349             initializeConnection();
350             arbiterImpl.setState(ReplicatedEnvironment.State.REPLICA);
351             doRunArbiterLoopInternalWork();
352             arbiterImpl.setState(ReplicatedEnvironment.State.UNKNOWN);
353         } catch (ClosedByInterruptException closedByInterruptException) {
354             if (arbiterImpl.isShutdown()) {
355                 LoggerUtils.info(logger, repImpl,
356                                  "Arbiter loop interrupted for shutdown.");
357                 return;
358             }
359             LoggerUtils.warning(logger, repImpl,
360                                 "Arbiter loop unexpected interrupt.");
361             throw new InterruptedException
362                 (closedByInterruptException.getMessage());
363         } catch (IOException e) {
364 
365             /*
366              * Master may have changed with the master shutting down its
367              * connection as a result. Normal course of events, log it and
368              * return to the outer node level loop.
369              */
370             LoggerUtils.fine(logger, repImpl,
371                              "Arbiter IO exception: " + e.getMessage() +
372                              "\n" + LoggerUtils.getStackTrace(e));
373         } catch (RetryException e) {
374             /* Propagate it outwards. Node does not need to shutdown. */
375             throw e;
376         } catch (GroupShutdownException e) {
377             shutdownException = e;
378             throw e;
379         } catch (RuntimeException e) {
380             shutdownException = e;
381             LoggerUtils.severe(logger, repImpl,
382                                "Arbiter unexpected exception " + e +
383                                 " " + LoggerUtils.getStackTrace(e));
384             throw e;
385         } catch (MasterSyncException e) {
386             /* expected change in masters from an election. */
387             LoggerUtils.fine(logger, repImpl, e.getMessage());
388         } catch (Exception e) {
389             shutdownException = e;
390             LoggerUtils.severe(logger, repImpl,
391                                "Arbiter unexpected exception " + e +
392                                " " + LoggerUtils.getStackTrace(e));
393             throw EnvironmentFailureException.unexpectedException(e);
394         } finally {
395             loopExitCleanup();
396         }
397     }

399     protected void doRunArbiterLoopInternalWork()
400        throws Exception {
401 
402         final int timeoutMs = repImpl.getConfigManager().
403                 getDuration(RepParams.REPLICA_TIMEOUT);
404         arbiterFeederChannel.setTimeoutMs(timeoutMs);
405 
406         requestQueue.clear();
407         outputQueue.clear();
408 
409         arbiterOutputThread =
410                 new ArbiterOutputThread(repImpl,
411                                         outputQueue,
412                                         protocol,
413                                         arbiterFeederChannel.getChannel(),
414                                         arbiterImpl.getArbiterVLSNTracker());
415         arbiterOutputThread.start();
416 
417         requestThread = new RequestThread();
418         requestThread.start();
419 
420         long maxPending = 0;
421 
422         try {
423             while (true) {
424                 Message message = protocol.read(arbiterFeederChannel);
425 
426                 if (arbiterImpl.isShutdownOrInvalid() || (message == null)) {
427                     return;
428                 }
429 
430                 while (!requestQueue.
431                         offer(message,
432                               QUEUE_POLL_INTERVAL_NS,
433                               TimeUnit.NANOSECONDS)) {
434                     /* Offer timed out. */
435                     if (!requestThread.isAlive()) {
436                         return;
437                     }
438                     /* Retry the offer */
439                     nReplayQueueOverflow.increment();
440                 }
441 
442                 final int pending = requestQueue.size();
443                 if (pending > maxPending) {
444                     maxPending = pending;
445                     LoggerUtils.fine(logger, repImpl,
446                                      "Max pending request log items:" +
447                                       maxPending);
448                 }
449             }
450         } catch (IOException ioe) {
451 
452             /*
453              * Make sure messages in the queue are processed. Ensure, in
454              * particular, that shutdown requests are processed and not ignored
455              * due to the IOEException resulting from a closed connection.
456              */
457             requestThread.exitRequest = RequestExitType.SOFT;
458         } finally {
459 
460             if (requestThread.exitRequest == RequestExitType.SOFT) {
461 
462                 /*
463                  * Drain all queued messages, exceptions may be generated
464                  * in the process. They logically precede IO exceptions.
465                  */
466                 requestThread.join();
467             }
468 
469             try {
470 
471                 if (requestThread.exception != null) {
472                     /* request thread is dead or exiting. */
473                     throw requestThread.exception;
474                 }
475 
476                 if (arbiterOutputThread.getException() != null) {
477                     throw arbiterOutputThread.getException();
478                 }
479             } finally {
480 
481                 /* Ensure thread has exited in all circumstances */
482                 requestThread.exitRequest = RequestExitType.IMMEDIATE;
483                 requestThread.join();
484                 arbiterOutputThread.shutdownThread(logger);
485             }
486         }
487     }

528      * socket is configured with a read timeout that's a multiple of the

530      *

534     private void createArbiterFeederChannel()
535         throws IOException, ConnectRetryException {
536 
537         DataChannel dataChannel = null;
538 
539         final DbConfigManager configManager = repImpl.getConfigManager();
540         final int timeoutMs = configManager.
541             getDuration(RepParams.PRE_HEARTBEAT_TIMEOUT);
542 
543         final int receiveBufferSize =
544                 configManager.getInt(RepParams.REPLICA_RECEIVE_BUFFER_SIZE);

548                 getDuration(RepParams.REPSTREAM_OPEN_TIMEOUT);
549 
550             /*
551              * Note that soTimeout is not set since it's a blocking channel and
552              * setSoTimeout has no effect on a blocking nio channel.
553              *
554              * Push responses out rapidly, they are small (heart beat or commit
555              * response) and need timely delivery to the master.
556              * (tcpNoDelay = true)
557              */
558 
559             final ConnectOptions connectOpts = new ConnectOptions().
560                 setTcpNoDelay(true).
561                 setReceiveBufferSize(receiveBufferSize).
562                 setOpenTimeout(openTimeout).
563                 setBlocking(true);
564 
565             dataChannel =
566                 repImpl.getChannelFactory().
567                 connect(arbiterImpl.getMasterStatus().getNodeMaster(),
568                         connectOpts);
569 
570             arbiterFeederChannel =
571                 new NamedChannelWithTimeout(repImpl,
572                                             logger,
573                                             arbiterImpl.getChannelTimeoutTask(),
574                                             dataChannel,
575                                             timeoutMs);
576 
577             ServiceDispatcher.doServiceHandshake
578                 (dataChannel, FeederManager.FEEDER_SERVICE);
579         } catch (ConnectException e) {
580 
581             /*
582              * A network problem, or the node went down between the time we
583              * learned it was the master and we tried to connect.
584              */
585             throw new ConnectRetryException(e.getMessage(),
586                                             NETWORK_RETRIES,
587                                             CONNECT_RETRY_SLEEP_MS);
588         } catch (ServiceConnectFailedException e) {
589 
590             /*
591              * The feeder may not have established the Feeder Service
592              * as yet. For example, the transition to the master may not have
593              * been completed. Wait longer.
594              */
595            if (e.getResponse() == Response.UNKNOWN_SERVICE) {
596                throw new ConnectRetryException(e.getMessage(),
597                                                SERVICE_UNAVAILABLE_RETRIES,
598                                                CONNECT_RETRY_SLEEP_MS);
599            }
600            throw EnvironmentFailureException.unexpectedException(e);
601         }
602     }

626      * Process the shutdown message from the master and return the
627      * GroupShutdownException that must be thrown to exit the Replica loop.
628      *
629      * @return the GroupShutdownException
630      */
631     private GroupShutdownException processShutdown(ShutdownRequest shutdown)
632         throws IOException {
633 
634         /*
635          * Acknowledge the shutdown message right away, since the checkpoint
636          * operation can take a long time to complete. Long enough to exceed
637          * the feeder timeout on the master. The master only needs to know that
638          * the replica has received the message.
639          */
640         queueAck(ReplicaOutputThreadBase.SHUTDOWN_ACK);
641 
642         /*
643          * Turn off network timeouts on the replica, since we don't want the
644          * replica to timeout the connection. The connection itself is no
645          * longer used past this point and will be reclaimed as part of normal
646          * replica exit cleanup.
647          */
648         arbiterFeederChannel.setTimeoutMs(Integer.MAX_VALUE);
649         final String masterHostName =
650             arbiterImpl.getMasterStatus().getGroupMaster().getHostName();
651         return new GroupShutdownException(
652             logger,
653             repImpl,
654             masterHostName,
655             arbiterImpl.getArbiterVLSNTracker().get(),
656             shutdown.getShutdownTimeMs());
657     }

679     /**
680      * Apply the operation represented by this log entry on this Arbiter node.
681      */
682     private Message replayEntries(Message firstMessage) throws IOException {
683         boolean doSync = false;
684         long highVLSN = 0;
685         Message shutdownMessage = null;
686         groupXact.clear();
687         groupMessages.clear();
688         groupMessages.add(firstMessage);
689         requestQueue.drainTo(groupMessages, N_MAX_GROUP_XACT);
690         for (int i = 0;i < groupMessages.size(); i++) {
691             final Message message = groupMessages.get(i) ;
692 
693             final MessageOp messageOp = message.getOp();
694 
695             if (messageOp == Protocol.SHUTDOWN_REQUEST) {
696                 shutdownMessage = message;
697             } else if (messageOp == Protocol.HEARTBEAT) {
698                 groupXact.add(ReplicaOutputThreadBase.HEARTBEAT_ACK);
699             } else {
700                 InputWireRecord wireRecord =
701                     ((Protocol.Entry) message).getWireRecord();
702                 final byte entryType = wireRecord.getEntryType();
703                 lastReplayedVLSN = wireRecord.getVLSN();
704 
705                 if (LOG_TXN_COMMIT.equalsType(entryType)) {
706                     Protocol.Commit commitEntry = (Protocol.Commit) message;
707                     if (commitEntry.getReplicaSyncPolicy() ==
708                         SyncPolicy.SYNC) {
709                         doSync = true;
710                     }
711 
712                     LogEntry logEntry = wireRecord.getLogEntry();
713                     if (lastReplayedVLSN.getSequence() > highVLSN) {
714                         highVLSN = lastReplayedVLSN.getSequence();
715                     }
716                     final TxnCommit masterCommit =
717                         (TxnCommit) logEntry.getMainItem();
718 
719                     long nextDTVLSN = masterCommit.getDTVLSN();
720                     if (nextDTVLSN == VLSN.UNINITIALIZED_VLSN_SEQUENCE) {
721                         /* Pre-DTVLSN log commit record. */
722                         nextDTVLSN = wireRecord.getVLSN().getSequence();
723                     }
724                     /*
725                      * The Arbiter, unlike Replicas, does not receive commits
726                      * in ascending VLSN order, so discard lower DTVLSNs.
727                      */
728                     dtvlsn = nextDTVLSN > dtvlsn ? nextDTVLSN : dtvlsn;
729                     groupXact.add(logEntry.getTransactionId());
730                     nAcks.increment();
731                     if (logger.isLoggable(Level.FINEST)) {
732                         LoggerUtils.finest(logger, repImpl,
733                                            "Arbiter ack commit record " +
734                                             wireRecord);
735                     }
736                 } else {
737                     String errMsg = "Illegal message type recieved by " +
738                             " Arbiter. [" + wireRecord + "]";
739                     throw new IllegalStateException(errMsg);
740                 }
741             }
742         }
743 
744         if (doSync ||
745             (lastFSyncTime + FSYNC_INTERVAL) <= System.currentTimeMillis()) {
746             doSync = true;
747             lastFSyncTime = System.currentTimeMillis();
748         }
749 
750         arbiterImpl.getArbiterVLSNTracker().write(new VLSN(highVLSN),
751                                                   new VLSN(dtvlsn),
752                                                   doSync);
753         for (int i = 0; i < groupXact.size(); i++) {
754             queueAck(groupXact.get(i));
755         }
756         return shutdownMessage;
757     }

769     class RequestThread extends StoppableThread {
770 
771         volatile private Exception exception;
772 
773         /*
774          * Set asynchronously when a shutdown is being requested.
775          */
776         volatile RequestExitType exitRequest = null;
777 
778         /* The queue poll interval, 1 second */
779         private final static long REQUEST_QUEUE_POLL_INTERVAL_NS = 1000000000l;
780 
781         protected RequestThread() {
782             super(repImpl, "RequestThread");
783         }
784 
785         @Override
786         protected int initiateSoftShutdown() {
787            /* Use immediate, since the stream will continue to be read. */
788            exitRequest = RequestExitType.IMMEDIATE;
789            return 0;
790         }
791 
792         @Override
793         public void run() {
794 
795             LoggerUtils.fine(logger, repImpl,
796                              "Request thread started. Message queue size:" +
797                               requestQueue.remainingCapacity());
798             try {
799                 while (true) {
800                     final Message message =
801                         requestQueue.poll(REQUEST_QUEUE_POLL_INTERVAL_NS,
802                                           TimeUnit.NANOSECONDS);
803 
804                     if ((exitRequest == RequestExitType.IMMEDIATE) ||
805                         ((exitRequest == RequestExitType.SOFT) &&
806                          (message == null)) ||
807                          arbiterImpl.isShutdownOrInvalid()) {
808                         return;
809                     }
810                     arbiterImpl.getMasterStatus().assertSync();
811                     if (message == null) {
812                         /* Timeout on poll. */
813                         continue;
814                     }
815                     Message shutdownMessage = replayEntries(message);
816                     if (shutdownMessage != null) {
817                         throw processShutdown(
818                             (ShutdownRequest) shutdownMessage);
819                     }
820                 }
821             } catch (Exception e) {
822                 exception = e;
823 
824                 /*
825                  * Bring it to the attention of the main thread by freeing
826                  * up the "offer" wait right away.
827                  */
828                 requestQueue.clear();
829 
830                 /*
831                  * Get the attention of the main arbiter thread in case it's
832                  * waiting in a read on the socket channel.
833                  */
834                 LoggerUtils.fine(logger, repImpl,
835                                  "closing arbiterFeederChannel = " +
836                                  arbiterFeederChannel);
837                 RepUtils.shutdownChannel(arbiterFeederChannel);
838 
839                 LoggerUtils.info(logger, repImpl,
840                                  "ArbiterAcker thread exiting with exception:" +
841                                   e.getMessage());
842             }
843         }
844 
845         @Override
846         protected Logger getLogger() {
847             return logger;
848         }
849     }
