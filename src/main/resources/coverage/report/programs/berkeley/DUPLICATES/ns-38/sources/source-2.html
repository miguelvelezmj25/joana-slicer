186     /** 
187      * Add the measurements from "other" and recalculate the min, max, and
188      * average values. The 95th and 99th percentile are not recalculated, 
189      * because the histogram from LatencyStatis not available, and those values
190      * can't be generated.
191      */
192     public void rollup(Latency other) {
193         if (other == null || other.totalOps == 0 || other.totalRequests == 0) {
194             throw new IllegalStateException
195                 ("Can't rollup a Latency that doesn't have any data");
196         }
197 
198         if (maxTrackedLatencyMillis != other.maxTrackedLatencyMillis) {
199             throw new IllegalStateException
200                 ("Can't rollup a Latency whose maxTrackedLatencyMillis is " +
201                  "different");
202         }
203 
204         if (min > other.min) {
205             min = other.min;
206         }
207 
208         if (max < other.max) {
209             max = other.max;
210         }
211 
212         avg = ((totalRequests * avg) + (other.totalRequests * other.avg)) / 
213               (totalRequests + other.totalRequests);
214 
215         /* Clear out 95th and 99th. They have become invalid. */
216         percent95 = 0;
217         percent99 = 0;
218 
219         totalOps += other.totalOps;
220         totalRequests += other.totalRequests;
221         opsOverflow += other.opsOverflow;
222     }
