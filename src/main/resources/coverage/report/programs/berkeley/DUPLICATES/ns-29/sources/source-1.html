23 /**
24  * A ghost bucket stands in as a placeholder for a set of vlsns that are 
25  * unknown. This kind of bucket can only be present at the very beginning of 
26  * the vlsn range. 
27  *
28  * This fulfills an edge case that can arise when vlsns are inserted out of 
29  * order, and log cleaner truncation lops off the leading edge of the index. 
30  * For example, suppose vlsns were inserted in this order:
31 
32  * vlsnIndex.put(vlsn=2, lsn=1/2)
33  * vlsnIndex.put(vlsn=1, lsn=1/0)
34  * vlsnIndex.put(vlsn=3, lsn=1/3)
35  * ...
36  * vlsnIndex.put(vlsn=5, lsn=2/9)
37  * vlsnIndex.put(vlsn=4, lsn=2/0)
38  * vlsnIndex.put(vlsn=6, lsn=2/10)
39  * ..
40  * This results in an index that has two buckets. Bucket 1 = {vlsn 2,3} and
41  * bucket 2 = {vlsn 5,6}. If we log clean file 1, we will truncate log at vlsn
42  * 3, and the new range will be vlsn 4-> vlsn 6. But the beginning and end of
43  * each range needs to have a valid bucket, and there is no bucket to represent
44  * vlsn 4. A GhostBucket is added to the head of the bucket set.
45  */
46 class GhostBucket extends VLSNBucket {
47     private long firstPossibleLsn;
48     private long lastPossibleLsn;
49     
50     GhostBucket(VLSN ghostVLSN,
51                 long firstPossibleLsn,
52                 long lastPossibleLsn) {
53         /* 
54          * Use ghostVLSN for the firstVLSN, which will make the own(),
55          * getFirst, getLast() methods work.
56          */
57         super(DbLsn.getFileNumber(firstPossibleLsn), // fileNumber
58               0, // stride
59               1, // maxMappings
60               1, // maxDistance,
61               ghostVLSN); // firstVLSN
62         this.firstPossibleLsn = firstPossibleLsn;
63         this.lastPossibleLsn = lastPossibleLsn;
64         dirty = true;
65     }
66 
67     /**
68      * Ideally, this would be a constructor, but we have to read several
69      * items off the tuple input first before calling super();
70      */
71     static GhostBucket makeNewInstance(TupleInput ti) {
72         VLSN ghostVLSN = new VLSN(ti.readPackedLong());
73         long firstLsn = ti.readPackedLong();
74         long lastLsn = ti.readPackedLong();
75         return new GhostBucket(ghostVLSN, firstLsn, lastLsn);
76     }

88     }
