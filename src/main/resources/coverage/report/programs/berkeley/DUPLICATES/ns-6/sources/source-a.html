339     final boolean removeKey(final Object key, final Object[] oldVal) {
340 
341         DataCursor cursor = null;
342         boolean doAutoCommit = beginAutoCommit();
343         try {
344             cursor = new DataCursor(view, true);
345             boolean found = false;
346             OperationStatus status = cursor.getSearchKey(key, null, true);
347             while (status == OperationStatus.SUCCESS) {
348                 cursor.delete();
349                 found = true;
350                 if (oldVal != null && oldVal[0] == null) {
351                     oldVal[0] = cursor.getCurrentValue();
352                 }
353                 status = areDuplicatesAllowed() ?
354                     cursor.getNextDup(true): OperationStatus.NOTFOUND;
355             }
356             closeCursor(cursor);
357             commitAutoCommit(doAutoCommit);
358             return found;
359         } catch (Exception e) {
360             closeCursor(cursor);
361             throw handleException(e, doAutoCommit);
362         }
363     }

437     final boolean beginAutoCommit() {
438         if (view.transactional) {
439             final CurrentTransaction currentTxn = view.getCurrentTxn();
440             try {
441                 if (currentTxn.isAutoCommitAllowed()) {
442                     currentTxn.beginTransaction(null);
443                     return true;
444                 }
445             } catch (DatabaseException e) {
446                 throw RuntimeExceptionWrapper.wrapIfNeeded(e);
447             }
448         }
449         return false;
450     }
