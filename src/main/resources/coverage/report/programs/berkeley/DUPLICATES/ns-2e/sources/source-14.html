16 import java.util.ArrayList;
17 import java.util.List;

19 /**
20  * Splitter is used to split a string based on a delimiter.
21  * Support includes double quoted strings, and the escape character.
22  * Raw tokens are returned that include the double quotes, white space,
23  * and escape characters.
24  *
25  */
26 public class Splitter {
27     private static final char QUOTECHAR = '"';
28     private static final char ESCAPECHAR = '\\';
29     private final char delimiter;
30     private final List<String> tokens = new ArrayList<String>();
31     private enum StateType {COLLECT, COLLECTANY, QUOTE};
32     private StateType prevState;
33     private StateType state;
34     private int startIndex;
35     private int curIndex;
36     private String row;
37 
38     public Splitter(char delimiter) {
39         this.delimiter = delimiter;
40     }
41 
42     public String[] tokenize(String inrow) {
43         row = inrow;
44         state = StateType.COLLECT;
45         tokens.clear();
46         startIndex = 0;
47         curIndex = 0;
48         for (int cur = 0; cur < row.length(); cur++) {
49             char c = row.charAt(cur);
50             switch (state) {
51                 case COLLECT :
52                     if (isDelimiter(c)) {
53                         outputToken();
54                         startIndex = cur + 1;
55                         curIndex = startIndex;
56                     } else {
57                         if (isQuote(c) && isQuoteState()) {
58                             state = StateType.QUOTE;
59                         } else if (isEscape(c)) {
60                             prevState = state;
61                             state = StateType.COLLECTANY;
62                         }
63                         curIndex++;
64                     }
65                     break;
66                 case COLLECTANY:
67                     curIndex++;
68                     state = prevState;
69                     break;
70                 case QUOTE:
71                     if (isEscape(c)) {
72                         prevState = state;
73                         state = StateType.COLLECTANY;
74                     } else if (isQuote(c)) {
75                         state = StateType.COLLECT;
76                     }
77                     curIndex++;
78                     break;
79             }
80         }
81         outputToken();
82         String[] retvals = new String[tokens.size()];
83         tokens.toArray(retvals);
84         return retvals;
85     }
86 
87     private boolean isQuote(char c) {
88         return (c == QUOTECHAR) ? true : false;
89     }
90 
91     private boolean isEscape(char c) {
92         return (c == ESCAPECHAR) ? true : false;
93     }
94 
95     private boolean isDelimiter(char c) {
96         return (c == delimiter) ? true : false;
97     }
98 
99     private void outputToken() {
100         if (startIndex < curIndex) {
101             tokens.add(row.substring(startIndex, curIndex));
102         } else {
103             tokens.add("");
104         }
105     }
106 
107     private boolean isQuoteState() {
108         for (int i = startIndex; i < curIndex; i++) {
109             if (!Character.isWhitespace(row.charAt(i))) {
110                 return false;
111             }
112         }
113         return true;
114     }
