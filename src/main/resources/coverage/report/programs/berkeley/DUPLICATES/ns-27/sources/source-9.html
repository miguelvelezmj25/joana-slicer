35 /**
36  * DbSync is a utility for ensuring that a group of replication nodes have
37  * fully caught up on the replication stream. The target use case is
38  * testing. If a replication group has crashed abruptly, nodes may have closed
39  * without finishing the full replay of the replication stream and the
40  * environments might not have the same contents. This makes it impossible to
41  * compare the contents of the environments for correctness.
42  * <p>
43  * DbSync assumes that all nodes are down. The utility is invoked for each node
44  * in the group. The node will come up and rejoin the group, causing the whole
45  * group to reach the same point in the replication stream. If the node becomes
46  * the master, it will issue a shutdown request. Otherwise, a node is a
47  * replica, and will wait for the shutdown message to come, and will then
48  * close.
49  */

240     /**
241      * Open this replication node. Block until the node has opened, synced up,
242      * and closed.
243      */
244     public void sync()
245         throws Exception {
246 
247         /*
248          * Set the ReplicaAckPolicy to ALL, so that all the replicas can get
249          * into a same sync point.
250          */
251         Durability durability =
252             new Durability(Durability.SyncPolicy.WRITE_NO_SYNC,
253                            Durability.SyncPolicy.WRITE_NO_SYNC,
254                            Durability.ReplicaAckPolicy.ALL);
255 
256         envConfig.setDurability(durability);
257         repConfig.setHelperHosts(helperHost);
258 
259         /* Exit if can't create a replicated Environment successfully. */
260         StatusListener listener = new StatusListener();
261         ReplicatedEnvironment repEnv = null;
262         try {
263             repEnv = new ReplicatedEnvironment(new File(envHome),
264                                                repConfig,
265                                                envConfig,
266                                                null,
267                                                QuorumPolicy.ALL);
268             repEnv.setStateChangeListener(listener);
269         } catch (Exception e) {
270             System.err.println("Can't successfully initialize " +
271                                repConfig.getNodeName() + " because of " + e);
272             System.exit(-1);
273         }
274 
275         /* Wait until the node becomes active. */
276         listener.awaitActiveState();
277 
278         if (repEnv.getState().isMaster()) {
279 
280             /*
281              * If master, start a transaction as a way of ascertaining whether
282              * all nodes are up. Since the ReplicaAckPolicy is ALL, the
283              * transaction will only begin when all the nodes are available.
284              */
285             Transaction txn = repEnv.beginTransaction(null, null);
286             txn.abort();
287 
288             /* Invoke the group shutdown API. */
289             repEnv.shutdownGroup(timeout, TimeUnit.SECONDS);
290         } else if (repEnv.getState().isReplica()) {
291             for (long i = 0; i < timeout; i++) {
292                 try {
293 
294                     /*
295                      * The replica will throw a GroupShutdownException if it
296                      * has received and processed the group close command from
297                      * the master.
298                      */
299                     repEnv.getState();
300                     Thread.sleep(1000);
301                 } catch (GroupShutdownException e) {
302                     break;
303                 }
304             }
305         }
306 
307         /* Shutdown the rep node. */
308         repEnv.close();
309     }

316

320
