56     /**
57      * Convert a string that is either 0xabc or 0xabc/0x123 into an lsn.
58      */
59     public static long readLsn(String lsnVal) {
60         int slashOff = lsnVal.indexOf("/");
61         if (slashOff < 0) {
62             long fileNum = readLongNumber(lsnVal);
63             return DbLsn.makeLsn(fileNum, 0);
64         } else {
65             long fileNum = readLongNumber(lsnVal.substring(0, slashOff));
66             long offset = CmdUtil.readLongNumber
67                 (lsnVal.substring(slashOff + 1));
68             return DbLsn.makeLsn(fileNum, offset);
69         }
70     }

76     public static void formatEntry(StringBuilder sb,
77                                    byte[] entryData,
78                                    boolean formatUsingPrintable) {
79         for (byte element : entryData) {
80             int b = element & 0xff;
81             if (formatUsingPrintable) {
82                 if (isPrint(b)) {
83                     if (b == 0134) {  /* backslash */
84                         sb.append('\\');
85                     }
86                     sb.append(printableChars.charAt(b - 33));
87                 } else {
88                     sb.append('\\');
89                     String hex = Integer.toHexString(b);
90                     if (b < 16) {
91                         sb.append('0');
92                     }
93                     sb.append(hex);
94                 }
95             } else {
96                 String hex = Integer.toHexString(b);
97                 if (b < 16) {
98                     sb.append('0');
99                 }
100                 sb.append(hex);
101             }
102         }
103     }
