34 /**
35  * A ReadonlyTxn represents
36  *  - a user initiated Txn executed on the Master node, when local-write or
37  *    read-only is configured, or
38  *  - a user initiated Txn executed on the Replica node, whether or not
39  *    local-write is configured, or
40  *  - an auto-commit Txn on a Replica node for a replicated DB.
41  *
42  * As its name implies it is used to implement the read-only semantics for
43  * access to replicated DBs on the Replica. It is not replicated txn, i.e.,
44  * it is not part of the rep stream.
45  *
46  * In addition, it uses the transaction hooks defined on Txn to implement the
47  * ReplicaConsistencyPolicy.  This must be done for all access to replicated
48  * DBs, including when local-write is configured.
49  */
50 public class ReadonlyTxn extends Txn {
51 
52     private final boolean localWrite;
53 
54     public ReadonlyTxn(EnvironmentImpl envImpl, TransactionConfig config)
55         throws DatabaseException {
56 
57         super(envImpl, config, ReplicationContext.NO_REPLICATE);
58 
59         localWrite = config.getLocalWrite();
60     }
61 
62     @Override
63     public boolean isLocalWrite() {
64         return localWrite;
65     }
66 
67     /**
68      * Provides a wrapper to screen for write locks. The use of write locks is
69      * used to infer that an attempt is being made to modify a replicated
70      * database. Note that this technique misses "conditional" updates, for
71      * example a delete operation using a non-existent key, but we are ok with
72      * that since the primary intent here is to ensure the integrity of the
73      * replicated stream that is being played back at that replica and these
74      * checks prevent such mishaps.
75      */
76     @Override
77     public LockResult lockInternal(long lsn,
78                                    LockType lockType,
79                                    boolean noWait,
80                                    boolean jumpAheadOfWaiters,
81                                    DatabaseImpl database)
82         throws LockNotAvailableException, LockConflictException,
83                DatabaseException {
84 
85         if (lockType.isWriteLock() && !database.allowReplicaWrite()) {
86             disallowReplicaWrite();
87         }
88         return super.lockInternal
89             (lsn, lockType, noWait, jumpAheadOfWaiters, database);
90     }
91 
92     /**
93      * If logging occurs before locking, we must screen out write locks here.
94      *
95      * If we allow the operation (e.g., for a NameLN), then be sure to call the
96      * base class method to prepare to undo in the (very unlikely) event that
97      * logging succeeds but locking fails. [#22875]
98      */
99     @Override
100     public void preLogWithoutLock(DatabaseImpl database) {
101         if (!database.allowReplicaWrite()) {
102             disallowReplicaWrite();
103         }
104         super.preLogWithoutLock(database);
105     }
106 
107     /**
108      * Unconditionally throws ReplicaWriteException because this locker was
109      * created on a replica.
110      */
111     @Override
112     public void disallowReplicaWrite() {
113         throw new ReplicaWriteException
114             (this, ((RepImpl) envImpl).getStateChangeEvent());
115     }
116 
117     /**
118      * Verifies that consistency requirements are met before allowing the
119      * transaction to proceed.
120      */
121     @Override
122     protected void txnBeginHook(TransactionConfig config)
123         throws ReplicaConsistencyException, DatabaseException {
124 
125         checkConsistency((RepImpl) envImpl, config.getConsistencyPolicy());
126     }
127 
128     /**
129      * Utility method used here and by ReplicaThreadLocker.
130      */
131     static void checkConsistency(final RepImpl repImpl,
132                                  final ReplicaConsistencyPolicy policy) {
133         if (State.DETACHED.equals(repImpl.getState()) ||
134             State.MASTER.equals(repImpl.getState())) {
135             /* Detached state, permit read-only access to the environment. */
136             return;
137         }
138         assert (policy != null) : "Missing default consistency policy";
139         try {
140             policy.ensureConsistency(repImpl);
141         } catch (InterruptedException e) {
142             throw new ThreadInterruptedException(repImpl, e);
143         } catch (MasterStateException e) {
144             /*
145              * Transitioned to master, while waiting for consistency, so the
146              * txn is free to go ahead on the master.
147              */
148             return;
149         }
150     }
151 }
