32 /**
33  * Contains methods for calculating utilization and for selecting files to
34  * clean.
35  *
36  * Note that we do clean files that are protected from deletion by HA/DataSync.
37  * If we did not clean them and a large number of files were to become
38  * unprotected at once, a large amount of log cleaning may suddenly be
39  * necessary.  Cleaning the files avoids this.  Better still would be to delete
40  * the metadata, but that would require writing a log entry to indicate the
41  * file is ready to be deleted, to avoid cleaning from scratch after a crash.
42  * [#16643] [#19221]
43  *
44  * Historical note: Prior to JE 6.0, LN utilization adjustments were needed
45  * because the LN last logged size was not stored in the BIN [#18633].
46  * Originally in JE 5, the corrected average LN size was used to adjust
47  * utilization. This was changed later in JE 5 to a correction factor since
48  * different log files may have different average LN sizes [#21106]. Then in
49  * JE 6.0 the last logged size was added to the BIN, the default for
50  * {@link com.sleepycat.je.EnvironmentConfig#CLEANER_ADJUST_UTILIZATION} was
51  * changed to false and a warning was added that the feature will be removed in
52  * the future [#22275]. Finally in JE 6.3 the LN adjustment code and data in
53  * CheckpointEnd were removed and the parameter was deprecated [#24090].
54  *
55  * Unlike with LNs, we do not store the last logged size of INs, so their
56  * obsolete size is computed as an average and this has the potential to cause
57  * over/under-cleaning. This problem is not known to occur, but if there are
58  * over/under-cleaning problems we should examine the recalculated info that is
59  * output as part of the CleanerRun INFO message.
60  *
61  * === Expired Data and Utilization ===
62  *
63  * Per-file histograms are calculated by the {@link ExpirationTracker} and
64  * stored in an internal database and cache by the {@link ExpirationProfile}.
65  * The histograms are used to calculate the expired bytes for a file at a
66  * particular time. Since obsolete (not expired) data can overlap with expired
67  * data, upper and lower bounds for overall utilization are determined. When
68  * the lower bound is below minUtilization, cleaning occurs.
69  *
70  * The file that has the lowest average utilization (midway between its upper
71  * and lower bounds) is selected for cleaning. If the file's upper and lower
72  * bounds are not close together (or the same), and the upper bound is above a
73  * threshold, then two-pass cleaning is performed. See
74  * {@link EnvironmentParams#CLEANER_TWO_PASS_GAP} and
75  * {@link EnvironmentParams#CLEANER_TWO_PASS_THRESHOLD}.
76  *
77  * The first pass of two-pass cleaning reads the file but doesn't do any real
78  * cleaning (no side effects). If this pass finds that the true utilization of
79  * the file is above the threshold (the same threshold as above), then cleaning
80  * does not take place and instead the histogram is updated. In this case the
81  * obsolete and expired data (in the old histogram) overlap, and it is the
82  * overlap that caused utilization to be estimated incorrectly. The new/updated
83  * histogram is calculated such that there is no overlap, improving utilization
84  * accuracy. If the first pass finds that true utilization is below the
85  * threshold, then normal cleaning (pass two) occurs. Two-pass cleaning
86  * protects against "over cleaning".
87  *
88  * The use of the overall utilization lower bound to drive cleaning is
89  * considered sufficient to protect against "under cleaning". Therefore, a disk
90  * space threshold is unnecessary.
91  *
92  * Gradual expiration is used to prevent cleaning spikes on day or hour
93  * boundaries. For purposes of driving cleaning, the utilization lower bound is
94  * calculated by distributing the bytes that expired in the current day/hour
95  * period evenly over that day/hour.
96  */

132     /**
133      * Returns the best file that qualifies for cleaning or probing, or null
134      * if no file qualifies.
135      *
136      * This method is called by FileSelector and synchronization order is:
137      * 1-FileSelector, 2-UtilizationCalculator, 3-ExpirationProfile.
138      *
139      * @param fileSummaryMap the map containing file summary info.
140      *
141      * @param forceCleaning is true to always select a file, even if its
142      * utilization is above the minimum utilization threshold.
143      *
144      * @return {file number, required utilization for 2-pass cleaning},
145      * or null if no file qualifies for cleaning.
146      */
147     synchronized Pair<Long, Integer> getBestFile(
148         final SortedMap<Long, FileSummary> fileSummaryMap,
149         final boolean forceCleaning) {
150 
151         /* Paranoia.  There should always be 1 file. */
152         if (fileSummaryMap.size() == 0) {
153             LoggerUtils.logMsg(logger, env, Level.SEVERE,
154                                "Can't clean, map is empty.");
155             return null;
156         }
157 
158         final FileSelector fileSelector = cleaner.getFileSelector();
159         final Set<Long> inProgressFiles = fileSelector.getInProgressFiles();
160 
161         /* Refresh expiration info so it reflects the current time. */
162         final ExpirationProfile expProfile = cleaner.getExpirationProfile();
163         final long currentTime = TTL.currentSystemTime();
164         expProfile.refresh(currentTime);
165 
166         /*
167          * Use local variables for mutable properties.  Using values that are
168          * changing during a single file selection pass would not produce a
169          * well defined result.
170          *
171          * Note that age is a distance between files not a number of files,
172          * that is, deleted files are counted in the age.
173          */
174         final int totalThreshold = cleaner.minUtilization;
175         final int fileThreshold = cleaner.minFileUtilization;
176         final int twoPassThreshold = cleaner.twoPassThreshold;
177         final int twoPassGap = cleaner.twoPassGap;
178         final int minAge = cleaner.minAge;
179         final boolean gradualExpiration = cleaner.gradualExpiration;
180         final boolean expirationEnabled = env.isExpirationEnabled();
181 
182         /*
183          * Cleaning must refrain from rearranging the portion of log processed
184          * as recovery time. Do not clean a file greater or equal to the first
185          * active file used in recovery, which is either the last log file or
186          * the file of the first active LSN in an active transaction, whichever
187          * is earlier.
188          *
189          * TxnManager.getFirstActiveLsn() (firstActiveTxnLsn below) is
190          * guaranteed to be earlier or equal to the first active LSN of the
191          * checkpoint that will be performed before deleting the selected log
192          * file. By selecting a file prior to this point we ensure that will
193          * not clean any entry that may be replayed by recovery.
194          *
195          * For example:
196          * 200 ckptA start, determines that ckpt's firstActiveLsn = 100
197          * 400 ckptA end
198          * 600 ckptB start, determines that ckpt's firstActiveLsn = 300
199          * 800 ckptB end
200          *
201          * Any cleaning that executes before ckpt A start will be constrained
202          * to files <= lsn 100, because it will have checked the TxnManager.
203          * If cleaning executes after ckptA start, it may indeed clean after
204          * ckptA's firstActiveLsn, but the cleaning run will wait to ckptB end
205          * to delete files.
206          */
207         long firstActiveFile = fileSummaryMap.lastKey();
208         final long firstActiveTxnLsn = env.getTxnManager().getFirstActiveLsn();
209 
210         if (firstActiveTxnLsn != DbLsn.NULL_LSN) {
211 
212             long firstActiveTxnFile =
213                 DbLsn.getFileNumber(firstActiveTxnLsn);
214 
215             if (firstActiveFile > firstActiveTxnFile) {
216                 firstActiveFile = firstActiveTxnFile;
217             }
218         }
219 
220         /*
221          * Note that minAge is at least one and may be configured to a higher
222          * value to prevent cleaning recently active files.
223          */
224         final long lastFileToClean = firstActiveFile - minAge;
225 
226         /*
227          * Given the total, obsolete and expired bytes:
228          *
229          * min utilization: 100 * ((obsolete + expired) / total)
230          * max utilization: 100 * (max(obsolete, expired) / total)
231          * avg utilization: (max - min) / 2
232          *
233          * Calculate min/max utilization in several categories:
234          *
235          * + Current total utilization, not counting the results of cleaning
236          *   that has not yet occurred, and using the current expired sizes.
237          *
238          * + Predicted total utilization, estimating the results of cleaning
239          *   yet to occur, and using a gradual expired size to prevent cleaning
240          *   spikes after hour/day boundaries.
241          *
242          * + Utilization for the "best" file to use when cleaning normally. The
243           *  file with the lowest avg utilization is selected.
244          *
245          * + Also determine the "best gradual" file with the lowest gradual
246          *   max utilization. Note that when a file is selected due to the
247          *   minFileUtilization threshold, gradual utilization must be used.
248          *   Otherwise, cleaning due to this threshold would defeat gradual
249          *   expiration in general.
250          */
251         Long bestFile = null;
252         int bestFileAvgUtil = 101;
253         int bestFileMinUtil = 0;
254         int bestFileMaxUtil = 0;
255         Long bestGradualFile = null;
256         int bestGradualFileMaxUtil = 101;
257         long currentTotalSize = 0;
258         long currentMinObsoleteSize = 0;
259         long currentMaxObsoleteSize = 0;
260         long predictedTotalSize = 0;
261         long predictedMinObsoleteSize = 0;
262         long predictedMaxObsoleteSize = 0;
263 
264         for (final Map.Entry<Long, FileSummary> entry :
265              fileSummaryMap.entrySet()) {
266 
267             final Long file = entry.getKey();
268             final long fileNum = file;
269 
270             if (!env.getFileProtector().isActiveOrNewFile(file)) {
271                 continue;
272             }
273 
274             final FileSummary summary = entry.getValue();
275 
276             final int expiredSize;
277             final int expiredGradualSize;
278 
279             if (expirationEnabled) {
280 
281                 final Pair<Integer, Integer> expiredSizes =
282                     expProfile.getExpiredBytes(fileNum, currentTime);
283 
284                 expiredSize = Math.min(
285                     expiredSizes.first(), summary.totalSize);
286 
287                 expiredGradualSize = gradualExpiration ?
288                     Math.min(expiredSizes.second(), summary.totalSize) :
289                     expiredSize;
290 
291             } else {
292                 expiredSize = 0;
293                 expiredGradualSize = 0;
294             }
295 
296             /*
297              * If the file is safe-to-delete, it is entirely obsolete by
298              * definition. This is more accurate than using getObsoleteSize.
299              */
300             final int obsoleteSize = summary.getObsoleteSize();
301 
302             final int minObsoleteSize = Math.max(
303                 obsoleteSize, expiredSize);
304 
305             final int maxObsoleteSize = Math.min(
306                 obsoleteSize + expiredSize,
307                 summary.totalSize);
308 
309             final int minGradualObsoleteSize = Math.max(
310                 obsoleteSize, expiredGradualSize);
311 
312             final int maxGradualObsoleteSize = Math.min(
313                 obsoleteSize + expiredGradualSize,
314                 summary.totalSize);
315 
316             currentTotalSize += summary.totalSize;
317             currentMinObsoleteSize += minObsoleteSize;
318             currentMaxObsoleteSize += maxObsoleteSize;
319 
320             /*
321              * If the file has been clean or is being cleaned, assume the
322              * file's data will occupy only its currently utilized bytes, after
323              * cleaning and deletion, based on the min obsolete size. This is
324              * an intentionally overly optimistic prediction of the results of
325              * cleaning, and is used to prevent over-cleaning, especially one
326              * due to a backlog created using inaccurate predictions.
327              */
328             if (inProgressFiles.contains(file)) {
329                 final int utilizedSize = summary.totalSize - minObsoleteSize;
330                 predictedTotalSize += utilizedSize;
331                 continue;
332             }
333 
334             predictedTotalSize += summary.totalSize;
335             predictedMinObsoleteSize += minGradualObsoleteSize;
336             predictedMaxObsoleteSize += maxGradualObsoleteSize;
337 
338             /* Skip files that are too young to be cleaned. */
339             if (fileNum > lastFileToClean) {
340                 continue;
341             }
342 
343             /*
344              * Pick the "best" file -- the one having the lowest avg
345              * utilization so far.
346              */
347             final int thisMinUtil = FileSummary.utilization(
348                 maxObsoleteSize, summary.totalSize);
349 
350             final int thisMaxUtil = FileSummary.utilization(
351                 minObsoleteSize, summary.totalSize);
352 
353             final int thisAvgUtil = (thisMinUtil + thisMaxUtil) / 2;
354 
355             if (bestFile == null || thisAvgUtil < bestFileAvgUtil) {
356                 bestFile = file;
357                 bestFileAvgUtil = thisAvgUtil;
358                 bestFileMinUtil = thisMinUtil;
359                 bestFileMaxUtil = thisMaxUtil;
360             }
361 
362             /*
363              * Pick the "best gradual" file -- the one having the lowest max
364              * gradual utilization so far.
365              */
366             final int thisGradualMaxUtil = FileSummary.utilization(
367                 minGradualObsoleteSize, summary.totalSize);
368 
369             if (bestGradualFile == null ||
370                 thisGradualMaxUtil < bestGradualFileMaxUtil) {
371 
372                 bestGradualFile = file;
373                 bestGradualFileMaxUtil = thisGradualMaxUtil;
374             }
375         }
376 
377         final int currentMinUtil = FileSummary.utilization(
378             currentMaxObsoleteSize, currentTotalSize);
379 
380         final int currentMaxUtil = FileSummary.utilization(
381             currentMinObsoleteSize, currentTotalSize);
382 
383         final int predictedMinUtil = FileSummary.utilization(
384             predictedMaxObsoleteSize, predictedTotalSize);
385 
386         final int predictedMaxUtil = FileSummary.utilization(
387             predictedMinObsoleteSize, predictedTotalSize);
388 
389         currentMinUtilization = currentMinUtil;
390         currentMaxUtilization = currentMaxUtil;
391         predictedMinUtilization = predictedMinUtil;
392         predictedMaxUtilization = predictedMaxUtil;
393 
394         /*
395          * 1. If total min utilization is below the threshold, clean the
396          *    "best" file, which is the one with the lowest avg utilization.
397          *
398          * 2. Else if the "best gradual" file has a max gradual utilization
399          *    that is below the threshold for a single file, clean it.
400          *
401          * 3. Else if there are more files to migrate, clean the next file to
402          *    be migrated.
403          *
404          * 4. Else if cleaning is forced for unit testing, clean the best file.
405          */
406         final Long fileChosen;
407         final String reason;
408 
409         if (predictedMinUtil < totalThreshold) {
410 
411             fileChosen = bestFile;
412             reason = "predicted min util is below minUtilization";
413 
414         } else if (bestGradualFileMaxUtil < fileThreshold) {
415 
416             fileChosen = bestGradualFile;
417             reason = "file has avg util below minFileUtilization";
418 
419         } else if (filesToMigrate.hasNext(fileSummaryMap, inProgressFiles)) {
420 
421             fileChosen = filesToMigrate.next(fileSummaryMap, inProgressFiles);
422             reason = "there are more forceCleanFiles";
423 
424         } else if (forceCleaning) {
425 
426             fileChosen = bestFile;
427             reason = "forced for testing";
428 
429         } else {
430             fileChosen = null;
431             reason = "no file selected";
432         }
433 
434         String bestFileMsg = "";
435         String twoPassMsg = "";
436         int pass1RequiredUtil = -1;
437 
438         if (fileChosen != null && fileChosen.equals(bestFile)) {
439 
440             bestFileMsg =
441                 ", chose file with util min: " + bestFileMinUtil +
442                 " max: " + bestFileMaxUtil +
443                 " avg: " + bestFileAvgUtil;
444 
445             /*
446              * If the difference between the file's min and max utilization is
447              * more than twoPassGap, and its max utilization is more then
448              * twoPassThreshold, use two pass cleaning. When using two-pass
449              * cleaning, skip the second pass if its recalculated utilization
450              * is higher than twoPassThreshold. In other words, if the benefit
451              * of cleaning is low, don't actually clean it.
452              */
453             if (bestFileMaxUtil > twoPassThreshold &&
454                 bestFileMaxUtil - bestFileMinUtil >= twoPassGap) {
455 
456                 pass1RequiredUtil = twoPassThreshold;
457                 twoPassMsg = ", 2-pass cleaning";
458             }
459         }
460 
461         final Level logLevel = (fileChosen != null) ? Level.INFO : Level.FINE;
462 
463         if (logger.isLoggable(logLevel)) {
464 
465             LoggerUtils.logMsg(
466                 logger, env, logLevel,
467                 "Clean file " +
468                 ((fileChosen != null) ?
469                     ("0x" + Long.toHexString(fileChosen)) : "none") +
470                 ": " + reason + twoPassMsg +
471                 ", current util min: " + currentMinUtil +
472                 " max: " + currentMaxUtil +
473                 ", predicted util min: " + predictedMinUtil +
474                 " max: " + predictedMaxUtil +
475                 bestFileMsg);
476         }
477 
478         return (fileChosen != null) ?
479             new Pair<>(fileChosen, pass1RequiredUtil) :
480             null;
481     }
