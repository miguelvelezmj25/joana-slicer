26 /**
27  * Format for messages received at across the wire for replication. Instead of
28  * sending a direct copy of the log entry as it is stored on the JE log files
29  * (LogEntryHeader + LogEntry), select parts of the header are sent.
30  *
31  * An InputWireRecord de-serializes the logEntry from the message bytes and
32  * releases any claim on the backing ByteBuffer.
33  */
34 public class InputWireRecord extends WireRecord {

44                     final ByteBuffer msgBuffer,
45                     final BaseProtocol protocol)
46         throws DatabaseException {

49 
50         logEntry = instantiateEntry(envImpl, msgBuffer);
51     }

53     private static LogEntryHeader createLogEntryHeader(
54         final ByteBuffer msgBuffer, final BaseProtocol protocol) {
55 
56         final byte entryType = msgBuffer.get();
57         int entryVersion = LogUtils.readInt(msgBuffer);
58         final int itemSize = LogUtils.readInt(msgBuffer);
59         final VLSN vlsn = new VLSN(LogUtils.readLong(msgBuffer));
60 
61         /*
62          * Check to see if we need to fix the entry's log version to work
63          * around [#25222].
64          */
65         if ((entryVersion > LogEntryType.LOG_VERSION_EXPIRE_INFO)
66             && protocol.getFixLogVersion12Entries()) {
67             entryVersion = LogEntryType.LOG_VERSION_EXPIRE_INFO;
68         }
69 
70         return new LogEntryHeader(entryType, entryVersion, itemSize, vlsn);
71     }
72
