119     @Override
120     public void visit(final String name, final Object value) {
121         ++size;
122         if (named) {
123             bv.putShort(cw.newUTF8(name));
124         }
125         if (value instanceof String) {
126             bv.put12('s', cw.newUTF8((String) value));
127         } else if (value instanceof Byte) {
128             bv.put12('B', cw.newInteger(((Byte) value).byteValue()).index);
129         } else if (value instanceof Boolean) {
130             int v = ((Boolean) value).booleanValue() ? 1 : 0;
131             bv.put12('Z', cw.newInteger(v).index);
132         } else if (value instanceof Character) {
133             bv.put12('C', cw.newInteger(((Character) value).charValue()).index);
134         } else if (value instanceof Short) {
135             bv.put12('S', cw.newInteger(((Short) value).shortValue()).index);
136         } else if (value instanceof Type) {
137             bv.put12('c', cw.newUTF8(((Type) value).getDescriptor()));
138         } else if (value instanceof byte[]) {
139             byte[] v = (byte[]) value;
140             bv.put12('[', v.length);
141             for (int i = 0; i < v.length; i++) {
142                 bv.put12('B', cw.newInteger(v[i]).index);
143             }
144         } else if (value instanceof boolean[]) {
145             boolean[] v = (boolean[]) value;
146             bv.put12('[', v.length);
147             for (int i = 0; i < v.length; i++) {
148                 bv.put12('Z', cw.newInteger(v[i] ? 1 : 0).index);
149             }
150         } else if (value instanceof short[]) {
151             short[] v = (short[]) value;
152             bv.put12('[', v.length);
153             for (int i = 0; i < v.length; i++) {
154                 bv.put12('S', cw.newInteger(v[i]).index);
155             }
156         } else if (value instanceof char[]) {
157             char[] v = (char[]) value;
158             bv.put12('[', v.length);
159             for (int i = 0; i < v.length; i++) {
160                 bv.put12('C', cw.newInteger(v[i]).index);
161             }
162         } else if (value instanceof int[]) {
163             int[] v = (int[]) value;
164             bv.put12('[', v.length);
165             for (int i = 0; i < v.length; i++) {
166                 bv.put12('I', cw.newInteger(v[i]).index);
167             }
168         } else if (value instanceof long[]) {
169             long[] v = (long[]) value;
170             bv.put12('[', v.length);
171             for (int i = 0; i < v.length; i++) {
172                 bv.put12('J', cw.newLong(v[i]).index);
173             }
174         } else if (value instanceof float[]) {
175             float[] v = (float[]) value;
176             bv.put12('[', v.length);
177             for (int i = 0; i < v.length; i++) {
178                 bv.put12('F', cw.newFloat(v[i]).index);
179             }
180         } else if (value instanceof double[]) {
181             double[] v = (double[]) value;
182             bv.put12('[', v.length);
183             for (int i = 0; i < v.length; i++) {
184                 bv.put12('D', cw.newDouble(v[i]).index);
185             }
186         } else {
187             Item i = cw.newConstItem(value);
188             bv.put12(".s.IFJDCS".charAt(i.type), i.index);
189         }
190     }

195         ++size;
196         if (named) {
197             bv.putShort(cw.newUTF8(name));

202     @Override
203     public AnnotationVisitor visitAnnotation(final String name,
204             final String desc) {
205         ++size;
206         if (named) {
207             bv.putShort(cw.newUTF8(name));
208         }
209         // write tag and type, and reserve space for values count
210         bv.put12('@', cw.newUTF8(desc)).putShort(0);
211         return new AnnotationWriter(cw, true, bv, bv, bv.length - 2);
212     }

214     @Override
215     public AnnotationVisitor visitArray(final String name) {
216         ++size;
217         if (named) {
218             bv.putShort(cw.newUTF8(name));
219         }
220         // write tag, and reserve space for array size
221         bv.put12('[', 0);
222         return new AnnotationWriter(cw, false, bv, bv, bv.length - 2);

225     @Override
226     public void visitEnd() {
227         if (parent != null) {
228             byte[] data = parent.data;
229             data[offset] = (byte) (size >>> 8);
230             data[offset + 1] = (byte) size;
231         }

253     /**
254      * Puts the annotations of this annotation writer list into the given byte
255      * vector.
256      * 
257      * @param out
258      *            where the annotations must be put.
259      */
260     void put(final ByteVector out) {
261         int n = 0;
262         int size = 2;
263         AnnotationWriter aw = this;
264         AnnotationWriter last = null;
265         while (aw != null) {
266             ++n;
267             size += aw.bv.length;
268             aw.visitEnd(); // in case user forgot to call visitEnd
269             aw.prev = last;
270             last = aw;
271             aw = aw.next;
272         }
273         out.putInt(size);
274         out.putShort(n);
275         aw = last;
276         while (aw != null) {
277             out.putByteArray(aw.bv.data, 0, aw.bv.length);
278             aw = aw.prev;
279         }
280     }

283      * Puts the given annotation lists into the given byte vector.
284      * 
285      * @param panns
286      *            an array of annotation writer lists.
287      * @param off
288      *            index of the first annotation to be written.
289      * @param out
290      *            where the annotations must be put.
291      */
292     static void put(final AnnotationWriter[] panns, final int off,
293             final ByteVector out) {
294         int size = 1 + 2 * (panns.length - off);
295         for (int i = off; i < panns.length; ++i) {
296             size += panns[i] == null ? 0 : panns[i].getSize();
297         }
298         out.putInt(size).putByte(panns.length - off);
299         for (int i = off; i < panns.length; ++i) {
300             AnnotationWriter aw = panns[i];
301             AnnotationWriter last = null;
302             int n = 0;
303             while (aw != null) {
304                 ++n;
305                 aw.visitEnd(); // in case user forgot to call visitEnd
306                 aw.prev = last;
307                 last = aw;
308                 aw = aw.next;
309             }
310             out.putShort(n);
311             aw = last;
312             while (aw != null) {
313                 out.putByteArray(aw.bv.data, 0, aw.bv.length);
314                 aw = aw.prev;
315             }
316         }
317     }

319     /**
320      * Puts the given type reference and type path into the given bytevector.
321      * LOCAL_VARIABLE and RESOURCE_VARIABLE target types are not supported.
322      * 
323      * @param typeRef
324      *            a reference to the annotated type. See {@link TypeReference}.
325      * @param typePath
326      *            the path to the annotated type argument, wildcard bound, array
327      *            element type, or static inner type within 'typeRef'. May be
328      *            <tt>null</tt> if the annotation targets 'typeRef' as a whole.
329      * @param out
330      *            where the type reference and type path must be put.
331      */
332     static void putTarget(int typeRef, TypePath typePath, ByteVector out) {
333         switch (typeRef >>> 24) {
334         case 0x00: // CLASS_TYPE_PARAMETER
335         case 0x01: // METHOD_TYPE_PARAMETER
336         case 0x16: // METHOD_FORMAL_PARAMETER
337             out.putShort(typeRef >>> 16);
338             break;
339         case 0x13: // FIELD
340         case 0x14: // METHOD_RETURN
341         case 0x15: // METHOD_RECEIVER
342             out.putByte(typeRef >>> 24);
343             break;
344         case 0x47: // CAST
345         case 0x48: // CONSTRUCTOR_INVOCATION_TYPE_ARGUMENT
346         case 0x49: // METHOD_INVOCATION_TYPE_ARGUMENT
347         case 0x4A: // CONSTRUCTOR_REFERENCE_TYPE_ARGUMENT
348         case 0x4B: // METHOD_REFERENCE_TYPE_ARGUMENT
349             out.putInt(typeRef);
350             break;
351         // case 0x10: // CLASS_EXTENDS
352         // case 0x11: // CLASS_TYPE_PARAMETER_BOUND
353         // case 0x12: // METHOD_TYPE_PARAMETER_BOUND
354         // case 0x17: // THROWS
355         // case 0x42: // EXCEPTION_PARAMETER
356         // case 0x43: // INSTANCEOF
357         // case 0x44: // NEW
358         // case 0x45: // CONSTRUCTOR_REFERENCE
359         // case 0x46: // METHOD_REFERENCE
360         default:
361             out.put12(typeRef >>> 24, (typeRef & 0xFFFF00) >> 8);
362             break;
363         }
364         if (typePath == null) {
365             out.putByte(0);
366         } else {
367             int length = typePath.b[typePath.offset] * 2 + 1;
368             out.putByteArray(typePath.b, typePath.offset, length);
369         }
370     }
