234     /**
235      * Repositions to a given raw key/data pair, or just past it if that record
236      * has been deleted.
237      *
238      * @return REPOS_EXACT, REPOS_NEXT or REPOS_EOF.
239      */
240     int repositionRange(byte[] lastKeyBytes,
241                         byte[] lastPriKeyBytes,
242                         byte[] lastValueBytes,
243                         boolean lockForWrite)
244         throws DatabaseException {
245 
246         OpReadOptions options = OpReadOptions.make(getLockMode(lockForWrite));
247         OpResult result;
248 
249         /* Use the given key/data byte arrays. */
250         setThangs(lastKeyBytes, lastPriKeyBytes, lastValueBytes);
251 
252         /* Position on or after the given key/data pair. */
253 
254         if (!view.dupsAllowed) {
255             /*
256              * No-dups is the simple case. Search for key >= lastKey, and then
257              * compare to see if we found lastKey again.
258              */
259             result = cursor.getSearchKeyRange(
260                 keyThang, primaryKeyThang, valueThang, options);
261 
262             if (!result.isSuccess()) {
263                 return REPOS_EOF;
264             }
265 
266             return KeyRange.equalBytes(
267                 lastKeyBytes, 0, lastKeyBytes.length,
268                 keyThang.getData(), keyThang.getOffset(), keyThang.getSize()) ?
269                 REPOS_EXACT : REPOS_NEXT;
270         }
271 
272         /*
273          * Duplicates are more complex.
274          *
275          * Search for key == lastKey && data >= lastData, and then compare to
276          * see if we found lastData again.
277          */
278         result = cursor.getSearchBothRange(
279             keyThang, primaryKeyThang, valueThang, options);
280 
281         if (result.isSuccess()) {
282 
283             DatabaseEntry thang =
284                 view.isSecondary() ? primaryKeyThang : valueThang;
285 
286             byte[] bytes =
287                 view.isSecondary() ? lastPriKeyBytes : lastValueBytes;
288 
289             return KeyRange.equalBytes(
290                 bytes, 0, bytes.length,
291                 thang.getData(), thang.getOffset(), thang.getSize()) ?
292                 REPOS_EXACT : REPOS_NEXT;
293         }
294 
295         /*
296          * The record with lastKey/lastData must have been deleted and there
297          * are no records with key == lastkey && data > lastData.
298          *
299          * Search for key >= lastKey, but keep in mind that we will probably
300          * land on the first record with lastKey.
301          */
302         result = cursor.getSearchKeyRange(
303             keyThang, primaryKeyThang, valueThang, options);
304 
305         if (!result.isSuccess()) {
306             return REPOS_EOF;
307         }
308 
309         /*
310          * If we are positioned on the first dup of lastKey, skip over its
311          * records to the first record with the next key.
312          */
313         if (KeyRange.equalBytes(lastKeyBytes, 0, lastKeyBytes.length,
314             keyThang.getData(), keyThang.getOffset(), keyThang.getSize())) {
315 
316             result = cursor.getNextNoDup(keyThang, primaryKeyThang,
317                 valueThang, options);
318 
319             if (!result.isSuccess()) {
320                 return REPOS_EOF;
321             }
322         }
323 
324         return REPOS_NEXT;
325     }

329      *
330      * @throws IllegalStateException when the database has unordered keys or
331      * unordered duplicates.
332      *
333      * @return whether the search succeeded.

336                             byte[] priKeyBytes,
337                             byte[] valueBytes,

650     /**
651      * Find the given key and value using getSearchBoth if possible or a
652      * sequential scan otherwise, no join cursor allowed.
653      */
654     OperationStatus findBoth(Object key, Object value, boolean lockForWrite)
655         throws DatabaseException {
656 
657         checkNoJoinCursor();
658         OpReadOptions options = OpReadOptions.make(getLockMode(lockForWrite));
659         view.useValue(value, valueThang, null);
660         if (view.dupsView) {
661             if (view.useKey(key, value, primaryKeyThang, view.dupsRange)) {
662                 KeyRange.copy(view.dupsKey, keyThang);
663                 if (otherThang == null) {
664                     otherThang = new DatabaseEntry();
665                 }
666                 OperationStatus status = cursor.getSearchBoth(
667                     keyThang, primaryKeyThang, otherThang, options).status();
668                 if (status == OperationStatus.SUCCESS &&
669                     KeyRange.equalBytes(otherThang, valueThang)) {
670                     return status;
671                 }
672             }
673         } else if (view.useKey(key, value, keyThang, range)) {
674             if (view.isSecondary()) {
675                 if (otherThang == null) {
676                     otherThang = new DatabaseEntry();
677                 }
678                 OperationStatus status = cursor.getSearchKey(
679                     keyThang, primaryKeyThang, otherThang, options).status();
680                 while (status == OperationStatus.SUCCESS) {
681                     if (KeyRange.equalBytes(otherThang, valueThang)) {
682                         return status;
683                     }
684                     status = cursor.getNextDup(
685                         keyThang, primaryKeyThang, otherThang,
686                         options).status();
687                 }
688                 /* if status != SUCCESS set range cursor to invalid? */
689             } else {
690                 return cursor.getSearchBoth(
691                     keyThang, null, valueThang, options).status();
692             }
693         }
694         return OperationStatus.NOTFOUND;
695     }

697     /**
698      * Find the given value using getSearchBoth if possible or a sequential
699      * scan otherwise, no join cursor allowed.
700      */
701     OperationStatus findValue(Object value, boolean findFirst)
702         throws DatabaseException {
703 
704         checkNoJoinCursor();
705 
706         if (view.entityBinding != null && !view.isSecondary() &&
707             (findFirst || !view.dupsAllowed)) {
708             return findBoth(null, value, false);
709         } else {
710             if (otherThang == null) {
711                 otherThang = new DatabaseEntry();
712             }
713             view.useValue(value, otherThang, null);
714             OperationStatus status = findFirst ? getFirst(false)
715                                                : getLast(false);
716             while (status == OperationStatus.SUCCESS) {
717                 if (KeyRange.equalBytes(valueThang, otherThang)) {
