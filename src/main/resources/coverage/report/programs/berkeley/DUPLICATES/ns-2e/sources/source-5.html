43 /**
44  * Dump the contents of a database. This utility may be used programmatically
45  * or from the command line.
46  *
47  * <p>When using this utility as a command line program, and the
48  * application uses custom key comparators, be sure to add the jars or
49  * classes to the classpath that contain the application's comparator
50  * classes.</p>
51  *
52  *<pre>
53  * java { com.sleepycat.je.util.DbDump |
54  *        -jar je-&lt;version&gt;.jar DbDump }
55  *   -h &lt;dir&gt;           # environment home directory
56  *  [-f &lt;fileName&gt;]     # output file, for non -rR dumps
57  *  [-l]                # list databases in the environment
58  *  [-p]                # output printable characters
59  *  [-r]                # salvage mode
60  *  [-R]                # aggressive salvage mode
61  *  [-d] &lt;directory&gt;    # directory for *.dump files (salvage mode)
62  *  [-s &lt;databaseName&gt;] # database to dump
63  *  [-v]                # verbose in salvage mode
64  *  [-V]                # print JE version number
65  *</pre>
66  * See {@link DbDump#main} for a full description of the
67  * command line arguments.
68  * <p>
69  * To dump a database to a stream from code:
70  * <pre>
71  *    DbDump dump = new DbDump(env, databaseName, outputStream, boolean);

73  * </pre>
74  *
75  *<p>
76  * Because a <code>DATA=END</code> marker is used to terminate the dump of
77  * each database, multiple databases can be dumped and loaded using a single
78  * stream. The {@link DbDump#dump} method leaves the stream positioned after
79  * the last line written and the {@link DbLoad#load} method leaves the stream
80  * positioned after the last line read.</p>

153      * The main used by the DbDump utility.
154      *
155      * @param argv The arguments accepted by the DbDump utility.
156      *
157      * <pre>
158      * usage: java { com.sleepycat.je.util.DbDump | -jar
159      * je-&lt;version&gt;.jar DbDump }
160      *             [-f output-file] [-l] [-p] [-V]
161      *             [-s database] -h dbEnvHome [-rR] [-v]
162      *             [-d directory]
163      * </pre>
164      *
165      * <dl>
166      * <dt>
167      * -f - the file to dump to. If omitted, output is to System.out.
168      * Does not apply when -r or -R is used.
169      * <dt>
170      * -l - list the databases in the environment.
171      * <dt>
172      * -p - output printable characters.
173      * <dd>If characters in either the key or data items are printing
174      * characters (as defined by isprint(3)), use printing characters in file
175      * to represent them. This option permits users to use standard text
176      * editors and tools to modify the contents of databases.</dd>
177      * <dt>
178      * -V - display the version of the JE library.
179      * <dt>
180      * -s database - the database to dump. Does not apply when -r or -R is
181      * used.
182      * <dt>
183      * -h dbEnvHome - the directory containing the database environment.
184      * <dt>
185      * -d directory - the output directory for *.dump files. Applies only when
186      * -r or -R is used.
187      * <dt>
188      * -v - print progress information to stdout for -r or -R mode.
189      * <dt>
190      * -r - Salvage data from possibly corrupt data files.
191      * <dd>
192      * The records for all databases are output. The records for each database
193      * are saved into &lt;databaseName&gt;.dump files in the current directory.
194      * <p>
195      * This option recreates the Btree structure in memory, so as large a heap
196      * size as possible should be specified. If -r cannot be used due to
197      * insufficient memory, use -R instead.
198      * <p>
199      * When used on uncorrupted data files, this option should return
200      * equivalent data to a normal dump, but most likely in a different order;
201      * in other words, it should output a transactionally correct data set.
202      * However, there is one exception where not all committed records will be
203      * output:
204      * <ul>
205      *     <li>When a committed transaction spans more than one .jdb file, and
206      *     the last file in this set of files has been deleted by the log
207      *     cleaner but earlier files have not, records for that transaction
208      *     that appear in the earlier files will not be output. This is because
209      *     the Commit entry in the last file is missing, and DbDump believes
210      *     that the transaction was not committed. Such missing output should
211      *     be relatively rare. Note that records in deleted files will be
212      *     output, because they were migrated forward by the log cleaner and
213      *     are no longer associated with a transaction.</li>
214      * </ul>
215      * </dd>
216      * <dt>
217      * -R - Aggressively salvage data from a possibly corrupt file.
218      * <dd>
219      * <p>
220      * The records for all databases are output. The records for each database
221      * are saved into &lt;databaseName&gt;.dump files in the current directory.
222      * <p>
223      * Unlike -r, the -R option does not recreate the Btree structure in
224      * memory. However, it does use a bit set to track all committed
225      * transactions, so as large a heap size as possible should be specified.
226      * <p>
227      * -R also differs from -r in that -R does not return a transactionally
228      * correct data set. This is because the Btree information is not
229      * reconstructed in memory. Therefore, data dumped in this fashion will
230      * almost certainly have to be edited by hand or other means before or
231      * after the data is reloaded. Be aware of the following abnormalities.
232      * <ul>
233      *     <li>Deleted records are often output. An application specific
234      *     technique should normally be used to correct for this.</li>
235      *     <li>Multiple versions of the same record are sometimes output. When
236      *     this happens, the more recent version of a record is output first.
237      *     Therefore, the -n option should normally be used when running
238      *     DbLoad.</li>
239      *     <li>When a committed transaction spans more than one .jdb file, and
240      *     the last file in this set of files has been deleted by the log
241      *     cleaner but earlier files have not, records for that transaction
242      *     that appear in the earlier files will not be output. This is because
243      *     the Commit entry in the last file is missing, and DbDump believes
244      *     that the transaction was not committed. Such missing output should
245      *     be relatively rare. Note that records in deleted files will be
246      *     output, because they were migrated forward by the log cleaner and
247      *     are no longer associated with a transaction. (This abnormality also
248      *     occurs with -r.)</li>
249      * </ul>
250      * </dd>
251      * </dl>
252      *
253      * @throws EnvironmentFailureException if an unexpected, internal or

256     public static void main(String argv[])
257         throws Exception {
258 
259         DbDump dumper = new DbDump();
260         boolean listDbs = dumper.parseArgs(argv);
261         if (dumper.doScavengerRun) {
262             dumper.openEnv(false);
263             dumper = new DbScavenger(dumper.env,
264                                      dumper.outputDirectory,
265                                      dumper.formatUsingPrintable,
266                                      dumper.doAggressiveScavengerRun,
267                                      dumper.verbose);
268             ((DbScavenger) dumper).setDumpCorruptedBounds(true);
269         }
270 
271         if (listDbs) {
272             dumper.listDbs();
273             System.exit(0);
274         }
275 
276         try {
277             dumper.dump();
278         } catch (Throwable T) {
279             T.printStackTrace();
280         } finally {
281             dumper.env.close();
282             if (dumper.outputFile != null &&
283                 dumper.outputFile != System.out) {
284                 dumper.outputFile.close();
285             }
286         }
287     }

308     protected boolean parseArgs(String argv[])
309         throws IOException {
310 
311         int argc = 0;
312         int nArgs = argv.length;
313         boolean listDbs = false;
314         while (argc < nArgs) {
315             String thisArg = argv[argc++];
316             if (thisArg.equals("-p")) {
317                 formatUsingPrintable = true;
318             } else if (thisArg.equals("-V")) {
319                 System.out.println(JEVersion.CURRENT_VERSION);
320                 System.exit(0);
321             } else if (thisArg.equals("-l")) {
322                 listDbs = true;
323             } else if (thisArg.equals("-r")) {
324                 doScavengerRun = true;
325             } else if (thisArg.equals("-R")) {
326                 doScavengerRun = true;
327                 doAggressiveScavengerRun = true;
328             } else if (thisArg.equals("-f")) {
329                 if (argc < nArgs) {
330                     outputFileName = argv[argc++];
331                 } else {
332                     printUsage("-f requires an argument");
333                 }
334             } else if (thisArg.equals("-h")) {
335                 if (argc < nArgs) {
336                     String envDir = argv[argc++];
337                     envHome = new File(envDir);
338                 } else {
339                     printUsage("-h requires an argument");
340                 }
341             } else if (thisArg.equals("-d")) {
342                 if (argc < nArgs) {
343                     outputDirectory = argv[argc++];
344                 } else {
345                     printUsage("-d requires an argument");
346                 }
347             } else if (thisArg.equals("-s")) {
348                 if (argc < nArgs) {
349                     dbName = argv[argc++];
350                 } else {
351                     printUsage("-s requires an argument");
352                 }
353             } else if (thisArg.equals("-v")) {
354                 verbose = true;
355             } else {
356                 printUsage(thisArg + " is not a valid option.");
357             }
358         }
359 
360         if (envHome == null) {
361             printUsage("-h is a required argument");
362         }
363 
364         if (!listDbs &&
365             !doScavengerRun) {
366             if (dbName == null) {
367                 printUsage("Must supply a database name if -l not supplied.");
368             }
369         }
370 
371         if (outputFileName == null) {
372             outputFile = System.out;
373         } else {
374             outputFile = new PrintStream(new FileOutputStream(outputFileName));
375         }
376 
377         return listDbs;
378     }

403      * Perform the dump.
404      *
405      * @throws EnvironmentFailureException if an unexpected, internal or
406      * environment-wide failure occurs.
407      *
408      * @throws IOException in subclasses.
409      */
410     public void dump()
411         throws EnvironmentNotFoundException,
412                EnvironmentLockedException,
413                DatabaseNotFoundException,
414                IOException {
415 
416         openEnv(true);
417 
418         LoggerUtils.envLogMsg(Level.INFO, DbInternal.getNonNullEnvImpl(env),
419                               "DbDump.dump of " + dbName + " starting");
420 
421         DatabaseEntry foundKey = new DatabaseEntry();
422         DatabaseEntry foundData = new DatabaseEntry();
423 
424         DatabaseConfig dbConfig = new DatabaseConfig();
425         dbConfig.setReadOnly(true);
426         DbInternal.setUseExistingConfig(dbConfig, true);
427         Database db;
428         try {
429             db = env.openDatabase(null, dbName, dbConfig);
430         } catch (DatabaseExistsException e) {
431             /* Should never happen, ExclusiveCreate is false. */
432             throw EnvironmentFailureException.unexpectedException(e);
433         }
434         dupSort = db.getConfig().getSortedDuplicates();
435 
436         printHeader(outputFile, dupSort, formatUsingPrintable);
437 
438         Cursor cursor = db.openCursor(null, null);
439         while (cursor.getNext(foundKey, foundData, LockMode.DEFAULT) ==
440                OperationStatus.SUCCESS) {
441             dumpOne(outputFile, foundKey.getData(), formatUsingPrintable);
442             dumpOne(outputFile, foundData.getData(), formatUsingPrintable);
443         }
444         cursor.close();
445         db.close();
446         outputFile.println("DATA=END");
447 
448         LoggerUtils.envLogMsg(Level.INFO, DbInternal.getNonNullEnvImpl(env),
449                               "DbDump.dump of " + dbName + " ending");
450     }
