749     /*
750      * a default filter that filters out entries from internal db and db that
751      * supports duplicates, and subscribes to all tables. It allows subscriber
752      * to receive updates from all tables since no table id is set.
753      */
754     private static class DefaultFeederFilter
755             implements FeederFilter, Serializable {
756         private static final long serialVersionUID = 1L;
757 
758         DefaultFeederFilter() {
759             super();
760         }
761 
762         @Override
763         public String[] getTableIds() {
764             return null;
765         }
766 
767         @Override
768         public OutputWireRecord execute(final OutputWireRecord record,
769                                         final RepImpl repImpl) {
770 
771             /* keep record if db id is null */
772             final DatabaseId dbId = record.getReplicableDBId();
773             if (dbId == null) {
774                 return record;
775             }
776 
777             final DbTree dbTree = repImpl.getDbTree();
778             final DatabaseImpl impl = dbTree.getDb(dbId);
779             try {
780                 /* keep record if db impl is not available */
781                 if (impl == null) {
782                     return record;
783                 }
784 
785                 /* filter out if from an db supporting duplicates */
786                 if (impl.getSortedDuplicates()) {
787                     return null;
788                 }
789 
790                 /* filter out if from an internal db */
791                 if (impl.isInternalDb()) {
792                     return null;
793                 }
794 
795                 return record;
796             } finally {
797                 if (impl != null) {
798                     dbTree.releaseDb(impl);
799                 }
800             }
801         }
802     }
