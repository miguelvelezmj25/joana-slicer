19 /**
20  * Class used by a node to track changes in Master Status. It's updated by
21  * the Listener. It represents the abstract notion that the notion of the
22  * current Replica Group is definitive and is always in advance of the notion
23  * of a master at each node. A node is typically playing catch up as it tries
24  * to bring its view in line with that of the group.
25  */

35     private NameIdPair groupMasterNameId = NameIdPair.NULL;

39      * as the node tries to catch up.

41     private String nodeMasterHostName = null;
42     private int nodeMasterPort = 0;
43     private NameIdPair nodeMasterNameId = NameIdPair.NULL;
