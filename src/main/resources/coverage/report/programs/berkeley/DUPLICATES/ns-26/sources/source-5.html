45 /**
46  * LDiff provides a mechanism for efficiently comparing two quiescent
47  * databases, typically residing on different machines connected by a
48  * network. The comparison is done at the logical level rather than the
49  * physical level, so that we can compare the contents of replicated databases
50  * where the logical contents may be identical, but the physical logs may be
51  * very different. If the databases are found to be different, it provides
52  * information that would help identify the specific nature of the differences.
53  *
54  * This class provides the external API used to initiate a comparison.
55  *
56  * For details, please review the document at:
57  *
58  * @see <a href="https://sleepycat.oracle.com/trac/wiki/JEDatabaseLogicalDiff">
59  *      LDiff </a>
60  */

533     /**
534      * A mechanism for efficiently comparing two quiescent databases, typically
535      * residing on different machines connected by a network.
536      *
537      * @param db a valid, open Database handle
538      * @param blkBag a bag of blocks to diff against db.
539      *
540      * @return true if the two comparing databases are identical.
541      * @throws Exception
542      */
543     public boolean diff(Database db, BlockBag blkBag)
544         throws Exception {
545 
546         /* Suppose the two comparing databases are identical, by default. */
547         boolean identical = true;
548 
549         /*
550          * window represents a BlockSize window into db. Initialize it to
551          * represent the block starting at the first key in db.
552          */
553         Cursor cursor = db.openCursor(null, null);
554 
555         long pos = 1;
556         int numKeys = cfg.getBlockSize();
557         Window window = new Window(cursor, numKeys);
558 
559         int errors = 0;
560         int maxerrors = cfg.getMaxErrors();
561         tracker = new DiffTracker(numKeys);
562         while (window.getChecksum() != 0 && blkBag.size() > 0) {
563 
564             /*
565              * Find the block in the bag whose checksum and md5 match the
566              * current window. This block, if it exists, is match.
567              */
568             Block match = findMatch(db.getEnvironment(), blkBag, window);
569             if (match != null) {
570                 tracker.setBlockDiffBegin
571                     (blkBag.getBlock(), blkBag.getBlockIndex());
572                 /* Remove match and any earlier blocks from the bag. */
573                 List<Block> removed = blkBag.remove(match);
574                 if (removed != null) {
575                     identical = false;
576                     errors += removed.size();
577                     tracker.calBlockDiffSize(blkBag.getBlockIndex());
578                     if (maxerrors > 0 && errors >= maxerrors) {
579                         break;
580                     }
581                 }
582 
583                 tracker.addDiffRegion(window);
584 
585                 /* Advance the window beyond the just matched block. */
586                 window.nextWindow();
587                 pos += window.size();
588 
589                 continue;
590             }
591 
592             identical = false;
593             LoggerUtils.envLogMsg
594                 (Level.FINE,
595                  DbInternal.getNonNullEnvImpl(db.getEnvironment()),
596                  "Unmatched block at position " + pos);
597             errors++;
598             if (maxerrors > 0 && errors >= maxerrors) {
599                 break;
600             }
601 
602             /* Roll the window forward by one key. */
603             window.rollWindow();
604 
605             if (window.getChecksum() != 0) {
606                 pos++;
607             }
608         }
609 
610         cursor.close();
611 
612         if (window.getChecksum() != 0) {
613 
614             /*
615              * We ran out of blocks in blkBag before we got to the end of db.
616              * Update the unmatched key range.
617              */
618             LoggerUtils.envLogMsg
619                 (Level.FINE,
620                  DbInternal.getNonNullEnvImpl(db.getEnvironment()),
621                  "Local Db has addtional records starting at " + pos + ".");
622             identical = false;
623             tracker.addWindowAdditionalDiffs(window);
624         }
625 
626         if (blkBag.size() > 0) {
627             /* All remaining blocks in the bag are unmatched. */
628             for (Block b : blkBag) {
629                 LoggerUtils.envLogMsg
630                     (Level.FINE,
631                      DbInternal.getNonNullEnvImpl(db.getEnvironment()),
632                      "Unmatched remote block: " + b);
633             }
634             identical = false;
635             tracker.addBlockBagAdditionalDiffs(window, blkBag);
636         }
637 
638         return identical;
639     }

650     /**
651      * Find the block in the bag whose rolling checksum and md5 match the given
652      * window. The md5 for the window is computed lazily, since it's more
653      * expensive to compute. There is a slight chance that multiple blocks in
654      * the bag will match the checksum and the md5 hash, return the first such
655      * block added to the bag in that case.
656      *
657      * @param blkBag a bag of blocks to search for a match
658      * @param window the block sized window of the db we're diffing
659      * @return A block which matches the window's checksum and the window's md5
660      * hash, or null if no block matches.
661      */
662     private Block findMatch(Environment env, BlockBag blkBag, Window window) {
663         List<Block> matches = blkBag.get(window.getChecksum());
664         if (matches == null) {
665             return null;
666         }
667 
668         /* Delay the computation of the hash until we know we need it. */
669         byte[] md5 = window.getMd5Hash();
670         for (Block b : matches) {
671             if (Arrays.equals(b.getMd5Hash(), md5)) {
672                 return b;

712     /**
713      * Connect to addr and perform a service handshake.  Retry as specified by
714      * the config object.
715      *
716      * @param addr the remote address to connect to
717      * @param dcFactory the channel factory for connection creation
718      * @return an open DataChannel
719      * @throws IOException if an exception occurs with the DataChannel
720      * @throws ServiceConnectFailedException if the remote service is busy
721      */
722     private DataChannel connect(InetSocketAddress addr,
723                                 DataChannelFactory dcFactory)
724         throws IOException, ServiceConnectFailedException {
725 
726         int triesLeft = cfg.getMaxConnectionAttempts();
727         DataChannel ret = null;
728         while (true) {
729             try {
730                 ret = dcFactory.connect(addr,
731                                         new ConnectOptions().
732                                         setBlocking(true).
733                                         setTcpNoDelay(true).
734                                         setOpenTimeout(SOCKET_TIMEOUT_MS).
735                                         setReadTimeout(SOCKET_TIMEOUT_MS));
736                 ServiceDispatcher.doServiceHandshake(ret, LDiffService.NAME);
737                 break;
738             } catch (ServiceConnectFailedException scfe) {
739                 if ((ret != null) && ret.isOpen()) {
740                     ret.close();
741                 }
742 
743                 /*
744                  * Unable to connect because the remote service is busy.  If
745                  * the user requested it, keep re-trying.  triesLeft == -1
746                  * means never abort.
747                  */
748                 if (triesLeft > 0) {
749                     triesLeft--;
750                 }
751 
752                 if (!cfg.getWaitIfBusy() || triesLeft == 0) {
753                     throw scfe;
754                 }
755             }
756         }
757 
758         return ret;
759     }

783     private class LDiffIterator implements Iterator<Block> {
784         private Block cached;
785         private Cursor cursor;
786         private final Database db;
787         private DatabaseEntry lastKey, lastData;
788         private boolean more;
789         private int i;
790         private final int numKeys;
791 
792         public LDiffIterator(Database db) {
793             i = 0;
794             numKeys = LDiff.this.cfg.getBlockSize();
795             cached = null;
796             more = true;
797             this.db = db;
798 
799             /* Prime the pump, get the first block in cached and set more. */
800             next();
801         }
802 
803         @Override
804         public boolean hasNext() {
805             return more;
806         }
807 
808         @Override
809         public void remove() {
810 
811         }
812 
813         @Override
814         public Block next() {
815             if (!more) {
816                 throw new NoSuchElementException();
817             }
818 
819             /*
820              * We don't want to return a block with 0 keys, but we can't know
821              * ahead of time whether the block will have any keys and the user
822              * likely called hasNext() already.  So when asked for block i, we
823              * cache block i+1, check whether it's empty and return the
824              * previously cached block i.  If block i+1 is empty, the next call
825              * to hasNext() will return false and the empty block won't be
826              * returned.
827              */
828             cursor = db.openCursor(null, null);
829             if (lastKey == null) {
830                 lastKey = new DatabaseEntry();
831                 lastData = new DatabaseEntry();
832             } else {
833                 cursor.getSearchBoth(lastKey, lastData, null);
834             }
835             Block ret = cached;
836             cached = LDiffUtil.readBlock(i++, cursor, numKeys);
837             if (cached.numRecords == 0) {
838                 more = false;
839             } else {
840                 cursor.getCurrent(lastKey, lastData, null);
841             }
842             cursor.close();
843             return ret;
844         }
845 
846         @Override
847         protected void finalize() throws Throwable {
848             try {
849                 cursor.close();
850             } finally {
851                 super.finalize();
852             }
853         }
854     }
