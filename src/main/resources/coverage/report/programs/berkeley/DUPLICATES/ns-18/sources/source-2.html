24 /**
25  * Designated Primary arbitration relies on the configuration parameter
26  * je.rep.designatedPrimary. This form of arbitration is only effective when
27  * the electable group size is 2. When one of the two electable nodes goes
28  * down, the remaining node is permitted to win elections, retain authoritative
29  * mastership, and commit transactions without any participation from its dead
30  * sibling, if and only if it has been configured as designated primary.
31  * <p>
32  * The user is responsible for ensuring that only one node at any time is
33  * annointed as the designated primary. There is some sanity checking that
34  * designated primary is only set by one node by master/replica syncups.  The
35  * parameter is mutable.
36  */
37 public class DesignatedPrimaryProvider implements ArbiterProvider {
38 
39     private final RepImpl repImpl;
40 
41     private final Logger logger;
42 
43     DesignatedPrimaryProvider(RepImpl repImpl) {
44         this.repImpl = repImpl;
45         logger = LoggerUtils.getLogger(getClass());
46     }
47 
48     /**
49      * Try to activate this node as a Primary, if it has been configured as
50      * such and if the group size is two. This method is invoked when an
51      * operation falls short of quorum requirements and is ready to trade
52      * durability for availability. More specifically it's invoked when an
53      * election fails, or there is an insufficient number of replicas during
54      * a begin transaction or a transaction commit.
55      *
56      * Active arbitration ends when the Non-Primary contacts it.
57      *
58      * @return true if the primary was activated.
59      */
60     @Override
61     public boolean attemptActivation() {
62         if (checkDesignatedPrimary()) {
63             LoggerUtils.info(logger, repImpl,
64                              "Primary activated; quorum is one.");
65             return true;
66         }
67 
68         LoggerUtils.fine(logger, repImpl,
69                          "Attempted unsuccessfully to activate designated " +
70                          "primary");
71         return false;
72     }
73 
74     /**
75      * Return true if this node is in a 2-node group and is configured as the
76      * Designated Primary, and is therefore capable of entering active
77      * arbitration.
78      */
79     @Override
80     public boolean activationPossible() {
81         return checkDesignatedPrimary();
82     }
83 
84     /**
85      * No cleannup is needed when exiting designated primary arbitration.
86      */
87     @Override
88     public void endArbitration() {
89         /* nothing to do. */
90     }
91 
92     /**
93      * Check the electable group size and the designated primary configuration
94      * to see if this node has the option of becoming the primary.
95      */
96     private boolean checkDesignatedPrimary() {
97         return (repImpl != null) && /* repImpl can be null in unit tests */
98             repImpl.isDesignatedPrimary() &&
99             repImpl.getRepNode().getGroup().getElectableGroupSize() == 2;
100     }
101 
102     /**
103      * When operating under designated primary arbitration, the election quorum
104      * is 1 for a group with electable group size of 2,
105      */
106     @Override
107     public int getElectionQuorumSize(QuorumPolicy quorumPolicy) {
108         return 1;
109     }
110 
111     /**
112      * Always returns 0, no replica acks are needed when acting under
113      * designated primary arbitration.
114      * TODO: is this still true with non voting nodes?
115      */
116     @Override
117     public int getAckCount(ReplicaAckPolicy ackPolicy) {
118         return 0;
119     }
120 
121     /**
122      * Return true if this node is no longer configured as the designated
123      * primary under the new configuration.
124      */
125     @Override
126     public boolean shouldEndArbitration(ReplicationMutableConfig newConfig) {
127         return (!newConfig.getDesignatedPrimary());
128     }
129 }
