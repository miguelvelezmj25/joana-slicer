447      * Sets the nodes associated with the Rep group. Note that both nodesById
448      * and nodesByIndex are initialized.
449      */
450     public void setNodes(final Map<Integer, RepNodeImpl> nodes) {
451 
452         synchronized (nodesById) {
453 
454             /* Remove nodes with persistent id */
455             for (final Iterator<RepNodeImpl> iter =
456                      nodesById.values().iterator();
457                  iter.hasNext(); ) {
458                 final RepNodeImpl node = iter.next();
459                 if (!node.getType().hasTransientId()) {
460                     iter.remove();
461                     nodesByName.remove(node.getName());
462                 }
463             }
464 
465             /* Add specified nodes */
466             if (nodes != null) {
467                 for (final RepNodeImpl node : nodes.values()) {
468                     final RepNodeImpl prevById =
469                         nodesById.put(node.getNodeId(), node);
470                     final RepNodeImpl prevByName =
471                         nodesByName.put(node.getName(), node);
472 
473                     /*
474                      * Also remove entries for any previous nodes if the
475                      * mapping between names and IDs was changed.
476                      */
477                     if ((prevById != null) &&
478                         !node.getName().equals(prevById.getName())) {
479                         nodesByName.remove(prevById.getName());
480                     }
481                     if ((prevByName != null) &&
482                         node.getNodeId() != prevByName.getNodeId()) {
483                         nodesById.remove(prevByName.getNodeId());
484                     }
485                 }
486             }
487 
488             assert new HashSet<>(nodesById.values()).equals(
489                 new HashSet<>(nodesByName.values()))
490                 : "Node maps indexed by ID and name differ: " +
491                 "IDs: " + nodesById + ", Names: " + nodesByName;
492         }
493     }

680      * Sets the minimum JE version that a node must be running in order to join
681      * the replication group.  The group object should have had its nodes
682      * fetched using the {@link RepGroupDB#fetchGroup} method and should be
683      * stored to the group database after making this change.  Throws a {@link
684      * MinJEVersionUnsupportedException} if the requested version is not
685      * supported.  Updates the group format version as needed to match the JE
686      * version.  Has no effect if the current minimum value is already as high
687      * or higher than the requested one.
688      *
689      * @param newMinJEVersion the new minimum JE version
690      * @throws MinJEVersionUnsupportedException if the requested version is not
691      * supported by the group's electable nodes
692      */
693     public void setMinJEVersion(final JEVersion newMinJEVersion)
694         throws MinJEVersionUnsupportedException {
695 
696         if (newMinJEVersion == null) {
697             throw new IllegalArgumentException(
698                 "The newMinJEVersion argument must not be null");
699         }
700         if (newMinJEVersion.compareTo(minJEVersion) <= 0) {
701             return;
702         }
703         final int newFormatVersion = getMaxFormatVersion(newMinJEVersion);
704 
705         /* Minimum JE version is not stored before format version 3 */
706         if (newFormatVersion < FORMAT_VERSION_3) {
707             return;
708         }
709 
710         for (final RepNodeImpl node : getElectableMembers()) {
711             final JEVersion nodeJEVersion = node.getJEVersion();
712             if ((nodeJEVersion != null) &&
713                 nodeJEVersion.compareTo(newMinJEVersion) < 0) {
714                 throw new MinJEVersionUnsupportedException(
715                     newMinJEVersion, node.getName(), nodeJEVersion);
716             }
717         }
718         minJEVersion = newMinJEVersion;
719         formatVersion = newFormatVersion;
720     }

724      * has been fetched via a readUncommitted access to the rep group database.
725      * It does so by ensuring that the summarized values match the nodes that
726      * were actually read.
727      */
728     public void makeConsistent() {
729         synchronized (nodesById) {
730             if (nodesById.isEmpty()) {
731                 return;
732             }
733             int computedNodeId = NODE_SEQUENCE_START-1;
734             int computedChangeVersion = -1;
735             for (RepNodeImpl mi : nodesById.values()) {
736                 /* Get the highest node ID */

738                     computedNodeId = mi.getNodeId();
739                 }
740                 /* Get the highest change version. */
741                 if (computedChangeVersion < mi.getChangeVersion()) {
742                     computedChangeVersion = mi.getChangeVersion();
743                 }
744             }
745             setNodeIdSequence(computedNodeId);
746             changeVersion = computedChangeVersion;

755      * Serializes an object by converting its TupleBinding byte based
756      * representation into the hex characters denoting the bytes.
757      *
758      * @param <T> the type of the object being serialized
759      * @param binding the tuble binding used to convert it into its byte form
760      * @param object the object being serialized
761      * @return the hex string containing the serialized hex form of the object
762      */
763     static <T> String objectToHex(TupleBinding<T> binding, T object) {
764         StringBuilder buffer = new StringBuilder();
765         TupleOutput tuple = new TupleOutput(new byte[100]);
766         binding.objectToEntry(object, tuple);
767         byte[] bytes = tuple.getBufferBytes();
768         int size = tuple.getBufferLength();
769 
770         for (int i = 0; i < size; i++) {
771             int lowNibble = (bytes[i] & 0xf);
772             int highNibble = ((bytes[i]>>4) & 0xf);
773             buffer.append(Character.forDigit(lowNibble, 16));
774             buffer.append(Character.forDigit(highNibble, 16));
775         }
776         return buffer.toString();
777     }

781      * in subclasses of SimpleProtocol. The serialized form is a multi-token
782      * string. The first token represents the RepGroup object itself with each
783      * subsequent node representing a node in the group. Tokens are separated
784      * by '|', the protocol separator character. The number of tokens is thus
785      * equal to the number of nodes in the group + 1. Each token is itself a
786      * hex character based representation of the binding used to serialize a

797         synchronized (nodesById) {

799

802                  * format version
803                  */

805                         mi, groupFormatVersion)) {

807                     buffer.append(serializeHex(mi, groupFormatVersion));

875      * Carries out the two step de-serialization from hex string into a byte
876      * buffer and subsequently into its object representation.
877      *
878      * @return the object representation
879      */
880     private static <T> T hexToObject(TupleBinding<T> binding, String hex) {
881         byte buffer[] = new byte[(hex.length() / 2)];
882         for (int i = 0; i < hex.length(); i += 2) {
883             int value = Character.digit(hex.charAt(i), 16);
884             value |= Character.digit(hex.charAt(i + 1), 16) << 4;
885             buffer[i >> 1] = (byte)value;
886         }
887         TupleInput tuple = new TupleInput(buffer);
888         return binding.entryToObject(tuple);
889     }

893      * the token at <code>start</code>represents the group object and each
894      * subsequent token represents a node in the group.
895      *
896      * @param tokens the array representing the group and its nodes
897      * @param start the position in the array at which to start the
898      * de-serialization.
899      *
900      * @return the de-serialized RepGroup
901      */
902     static public RepGroupImpl deserializeHex(String[] tokens, int start) {
903         final RepGroupDB.GroupBinding groupBinding =
904             new RepGroupDB.GroupBinding();
905         RepGroupImpl group = hexToObject(groupBinding, tokens[start++]);
906         Map<Integer, RepNodeImpl> nodeMap =
907             new HashMap<Integer, RepNodeImpl>();

909             RepNodeImpl n =
910                 hexDeserializeNode(tokens[start++], group.getFormatVersion());
911             RepNodeImpl old = nodeMap.put(n.getNameIdPair().getId(), n);
912             assert(old == null);
913         }
914         group.setNodes(nodeMap);
915         return group;
916     }
