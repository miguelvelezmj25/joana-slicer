25 /**
26  * Verifies the checksums in an {@code InputStream} for a log file in a JE
27  * {@code Environment}.
28  *
29  * <p>This {@code InputStream} reads input from some other given {@code
30  * InputStream}, and verifies checksums while reading.  Its primary intended
31  * use is to verify log files that are being copied as part of a programmatic
32  * backup.  It is critical that invalid files are not added to a backup set,
33  * since then both the live environment and the backup will be invalid.</p>
34  *
35  * <p>The following example verifies log files as they are being copied.  The
36  * {@link DbBackup} class should normally be used to obtain the array of files
37  * to be copied.</p>
38  *
39  * <!-- NOTE: Whenever the method below is changed, update the copy in
40  * VerifyLogTest to match, so that it will be tested. -->
41  *
42  * <pre>
43  *  void copyFiles(final Environment env,
44  *                 final String[] fileNames,
45  *                 final File destDir,
46  *                 final int bufSize)
47  *      throws IOException, DatabaseException {
48  *
49  *      final File srcDir = env.getHome();
50  *
51  *      for (final String fileName : fileNames) {
52  *
53  *          final File destFile = new File(destDir, fileName);
54  *          final FileOutputStream fos = new FileOutputStream(destFile);
55  *
56  *          final File srcFile = new File(srcDir, fileName);
57  *          final FileInputStream fis = new FileInputStream(srcFile);
58  *          final LogVerificationInputStream vis =
59  *              new LogVerificationInputStream(env, fis, fileName);
60  *
61  *          final byte[] buf = new byte[bufSize];
62  *
63  *          try {
64  *              while (true) {
65  *                  final int len = vis.read(buf);
66  *                  if (len &lt; 0) {
67  *                      break;
68  *                  }
69  *                  fos.write(buf, 0, len);
70  *              }
71  *          } finally {
72  *              fos.close();
73  *              vis.close();
74  *          }
75  *      }
76  *  }
77  * </pre>
78  *
79  * <p>It is important to read the entire underlying input stream until the
80  * end-of-file is reached to detect incomplete entries at the end of the log
81  * file.</p>
82  *
83  * <p>Note that {@code mark} and {@code reset} are not supported and {@code
84  * markSupported} returns false.  The default {@link InputStream}
85  * implementation of these methods is used.</p>
86  *
87  * @see DbBackup
88  * @see DbVerifyLog
89  */

146

194             if (lenRead < 0) {
195                 verifier.verifyAtEof();
196             }

199

201

203     }

205     /**
206      * {@inheritDoc}
207      *
208      * <p>This method reads the underlying {@code InputStream} in order to
209      * skip the required number of bytes and verifies the contents of the
210      * stream.  A temporary buffer is allocated lazily for reading.</p>
211      *
212      * @throws LogVerificationException if a checksum cannot be verified or a
213      * log entry is determined to be invalid by examining its contents.
214      *
215      * @throws EnvironmentFailureException if an unexpected, internal or
216      * environment-wide failure occurs.
217      */
218     @Override
219     public long skip(final long bytesToSkip)
220         throws IOException {
221 
222         if (bytesToSkip <= 0) {
223             return 0;
224         }
225 
226         /*
227          * Like InputStream.skip, we lazily allocate a skip buffer.  We must
228          * read the data in order to validate the checksum.  Unlike the
229          * InputStream.skip implementation, we cannot use a static buffer
230          * because we do process the data and cannot allow multiple threads to
231          * share the same buffer.
232          */
233         if (skipBuf == null) {
234             skipBuf = new byte[SKIP_BUF_SIZE];
235         }
236 
237         long remaining = bytesToSkip;
238         while (remaining > 0) {
239             final int lenRead = read
240                 (skipBuf, 0, (int) Math.min(SKIP_BUF_SIZE, remaining));
241             if (lenRead < 0) {
242                 break;
243             }
244             remaining -= lenRead;
245         }
246 
247         return bytesToSkip - remaining;
248     }
