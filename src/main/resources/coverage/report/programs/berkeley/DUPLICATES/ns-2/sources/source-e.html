71     /**
72      * Removes the range between start and end from the given exception
73      * handlers.
74      * 
75      * @param h
76      *            an exception handler list.
77      * @param start
78      *            the start of the range to be removed.
79      * @param end
80      *            the end of the range to be removed. Maybe null.
81      * @return the exception handler list with the start-end range removed.
82      */
83     static Handler remove(Handler h, Label start, Label end) {
84         if (h == null) {
85             return null;
86         } else {
87             h.next = remove(h.next, start, end);
88         }
89         int hstart = h.start.position;
90         int hend = h.end.position;
91         int s = start.position;
92         int e = end == null ? Integer.MAX_VALUE : end.position;
93         // if [hstart,hend[ and [s,e[ intervals intersect...
94         if (s < hend && e > hstart) {
95             if (s <= hstart) {
96                 if (e >= hend) {
97                     // [hstart,hend[ fully included in [s,e[, h removed
98                     h = h.next;
99                 } else {
100                     // [hstart,hend[ minus [s,e[ = [e,hend[
101                     h.start = end;
102                 }
103             } else if (e >= hend) {
104                 // [hstart,hend[ minus [s,e[ = [hstart,s[
105                 h.end = start;
106             } else {
107                 // [hstart,hend[ minus [s,e[ = [hstart,s[ + [e,hend[
108                 Handler g = new Handler();
109                 g.start = end;
110                 g.end = h.end;
111                 g.handler = h.handler;
112                 g.desc = h.desc;
113                 g.type = h.type;
114                 g.next = h.next;
115                 h.end = start;
116                 h.next = g;
117             }
118         }
119         return h;
120     }
