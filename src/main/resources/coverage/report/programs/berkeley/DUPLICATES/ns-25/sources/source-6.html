545     /*
546      * Check that the specified node is an appropriate target. For example,
547      * make sure it's a valid node in the group, and it's not the same as
548      * the original node.
549      */
550     private RepNodeImpl checkMember(String nodeName,
551                                     String masterErrorMessage,
552                                     boolean electableOnly)
553         throws MasterStateException,
554                MemberNotFoundException {
555 
556         final RepGroupImpl group = getGroup().getRepGroupImpl();
557         final RepNodeImpl node = group.getNode(nodeName);
558 
559         /* Check the membership. */
560         if (node == null ||
561             /* Creation is not yet acknowledged */
562             (!node.isRemoved() && !node.isQuorumAck())) {
563             throw new MemberNotFoundException("Node: " + nodeName +
564                                               " is not a member of the " +
565                                               "group: " + groupName);
566         }
567 
568         if (electableOnly && !node.getType().isElectable()) {
569             throw new IllegalArgumentException("Node: " + nodeName +
570                                                " must have node type" +
571                                                " ELECTABLE, was " +
572                                                node.getType());
573         }
574 
575         if (node.isRemoved() && node.isQuorumAck()) {
576             throw new MemberNotFoundException("Node: " + nodeName +
577                                               " is not currently a member " +
578                                               "of the group: " + groupName +
579                                               ", it has been removed.");
580         }
581 
582         /* Check if the node itself is the master. */
583         if (masterErrorMessage != null) {
584             final InetSocketAddress masterAddress = getMasterSocket();
585             if (masterAddress.equals(node.getSocketAddress())) {
586                 throw new MasterStateException(masterErrorMessage);
587             }
588         }
589 
590         return node;
591     }

593     /* Do a message exchange with the targeted master. */
594     private ResponseMessage doMessageExchange(RequestMessage request,
595                                               Class<?> respClass)
596         throws EnvironmentFailureException,
597                MasterStateException,
598                MemberNotFoundException,
599                UnknownMasterException {
600 
601         /* Do the communication. */
602         final InetSocketAddress masterAddress = getMasterSocket();
603         final MessageExchange me = groupProtocol.new MessageExchange
604             (masterAddress, GroupService.SERVICE_NAME, request);
605         me.run();
606 
607         ResponseMessage resp = me.getResponseMessage();
608 
609         if (resp == null) {
610             if (me.getException() != null) {
611                 throw new UnknownMasterException
612                     ("Problem communicating with master.", me.getException());
613             }
614 
615             /*
616              * Returning null on success is part of the message protocol, the
617              * caller expects it.
618              */
619             return null;
620         }
621 
622         if (respClass == null && resp instanceof Fail) {
623             throw getException(resp);
624         }
625 
626         if (respClass != null &&
627             !(resp.getClass().getName().equals(respClass.getName()))) {
628             throw getException(resp);
629         }
630 
631         return resp;
632     }

634     /**
635      * Examines the response and generates a meaningful error exception.
636      */
637     private DatabaseException getException(ResponseMessage resp) {
638         if (resp == null) {
639             return EnvironmentFailureException.unexpectedState
640                 ("No response to request");
641         }
642 
643         if (resp instanceof Fail) {
644             Fail fail = (Fail) resp;
645             switch (fail.getReason()) {
646                 case MEMBER_NOT_FOUND:
647                     return new MemberNotFoundException(fail.getMessage());
648                 case IS_MASTER:

650                 case IS_REPLICA:
651                     return new ReplicaStateException(fail.getMessage());
652                 case TRANSFER_FAIL:

656                         (fail.getMessage());
657                 default:

659                         unexpectedState(fail.getMessage());

661         }
662 
663         if (resp instanceof ProtocolError) {
664             return EnvironmentFailureException.unexpectedState
665                 (((ProtocolError)resp).getMessage());
666         }
667

669             ("Response not recognized: " + resp);
