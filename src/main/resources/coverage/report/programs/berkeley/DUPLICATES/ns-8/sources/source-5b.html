97             checkOpen();
98             Txn txn = getXATransactionInternal(xid);

100                 throw new XAException

102             }
103             removeReferringHandle(new Transaction(this, txn));
104             if (txn.isOnlyAbortable()) {
105                 throw new XAException(XAException.XA_RBROLLBACK);

116     @Override
117     public void end(Xid xid, int flags)
118         throws XAException {
119 
120         if (DEBUG) {
121             System.out.println("*** end called " + xid + "/" + flags);
122         }
123 
124         /* flags - One of TMSUCCESS, TMFAIL, or TMSUSPEND. */
125 
126         boolean tmFail = (flags & XAResource.TMFAIL) != 0;
127         boolean tmSuccess = (flags & XAResource.TMSUCCESS) != 0;
128         boolean tmSuspend = (flags & XAResource.TMSUSPEND) != 0;
129         if ((!tmFail && !tmSuccess && !tmSuspend) ||
130             (tmFail && tmSuccess) ||
131             ((tmFail || tmSuccess) && tmSuspend)) {
132             throw new XAException(XAException.XAER_INVAL);
133         }
134 
135         final EnvironmentImpl envImpl = getNonNullEnvImpl();
136 
137         if (DEBUG) {
138             System.out.println
139                 ("Transaction for " + Thread.currentThread() + " is " +
140                  envImpl.getTxnManager().getTxnForThread());
141         }
142 
143         Transaction transaction = envImpl.getTxnManager().unsetTxnForThread();
144         if (transaction == null) {
145             transaction = getXATransaction(xid);
146         }
147         Txn txn = (transaction != null) ? transaction.getTxn() : null;
148         if (txn == null) {
149             throw new XAException(XAException.XAER_NOTA);
150         }
151 
152         if (tmFail) {
153 
154             /*
155              * Creating the XAFailureException will set the txn to abort-only.
156              * This exception stack trace will provide more "cause" information
157              * when it is wrapped in an exception that is thrown later, which
158              * occurs when an attempt is made to use the txn.
159              */
160             new XAFailureException(txn);
161         }
162 
163         if (tmSuspend) {
164             if (txn.isSuspended()) {
165                 throw new XAException(XAException.XAER_PROTO);
166             }
167             txn.setSuspended(true);
168         }
169 
170         /* For tmSuccess, do nothing further. */
171     }

234              * commit it here because the user doesn't need to (and isn't
235              * allowed to either).
236              */
237             if (ret == XAResource.XA_RDONLY) {
238                 commit(xid, true);
239             }
240

243             throwNewXAException(e);

326     @Override
327     public void start(Xid xid, int flags)
328         throws XAException {
329 
330         if (DEBUG) {
331             System.out.println("*** start called " + xid + "/" + flags);
332         }
333 
334         boolean tmJoin = (flags & XAResource.TMJOIN) != 0;
335         boolean tmResume = (flags & XAResource.TMRESUME) != 0;
336 
337         /* Check flags - only one of TMNOFLAGS, TMJOIN, or TMRESUME. */
338         if (xid == null ||
339             (tmJoin && tmResume) ||
340             (!tmJoin &&
341              !tmResume &&
342              flags != XAResource.TMNOFLAGS)) {
343             throw new XAException(XAException.XAER_INVAL);
344         }
345 
346         try {
347             Transaction txn = getXATransaction(xid);
348             TxnManager txnMgr = getNonNullEnvImpl().getTxnManager();
349 
350             if (flags == XAResource.TMNOFLAGS) {
351 
352                 /*
353                  * If neither RESUME nor JOIN was set, make sure xid doesn't
354                  * exist in allXATxns.  Throw XAER_DUPID if it does.
355                  */
356                 if (txn == null) {
357                     if (DEBUG) {
358                         System.out.println
359                             ("Transaction for XID " + xid + " being created");
360                     }
361 
362                     txn = beginTransaction(null, null);
363                     setXATransaction(xid, txn);
364 
365                 } else {
366                     throw new XAException(XAException.XAER_DUPID);
367                 }
368             } else if (tmJoin) {
369                 if (txn == null) {
370                     throw new XAException(XAException.XAER_NOTA);
371                 }
372 
373                 if (txnMgr.getTxnForThread() != null ||
374                     txn.getPrepared()) {
375                     throw new XAException(XAException.XAER_PROTO);
376                 }
377             } else if (tmResume) {
378                 if (txn == null) {
379                     throw new XAException(XAException.XAER_NOTA);
380                 }
381 
382                 if (!txn.getTxn().isSuspended()) {
383                     throw new XAException(XAException.XAER_PROTO);
384                 }
385                 txn.getTxn().setSuspended(false);
386             }
387 
388             if (DEBUG) {
389                 System.out.println
390                     ("Setting Transaction for " + Thread.currentThread());
391             }
392             txnMgr.setTxnForThread(txn);
393         } catch (DatabaseException DE) {
394             if (DEBUG) {
395                 System.out.println("*** start exception");
396             }
397             throwNewXAException(DE);
398         }
399 
400         if (DEBUG) {
401             System.out.println("*** start finished");
402         }
403     }
