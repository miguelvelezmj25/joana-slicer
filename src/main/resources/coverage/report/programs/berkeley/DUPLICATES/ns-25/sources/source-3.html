150     private void parseArgs(String argv[]) {
151         int argc = 0;
152         int nArgs = argv.length;
153         String netPropsName = null;
154 
155         if (nArgs == 0) {
156             printUsage(null);
157             System.exit(0);
158         }
159 
160         while (argc < nArgs) {
161             String thisArg = argv[argc++];
162             if (thisArg.equals("-groupName")) {
163                 if (argc < nArgs) {
164                     groupName = argv[argc++];
165                 } else {
166                     printUsage("-groupName requires an argument");
167                 }
168             } else if (thisArg.equals("-helperHosts")) {
169                 if (argc < nArgs) {
170                     helperSockets = HostPortPair.getSockets(argv[argc++]);
171                 } else {
172                     printUsage("-helperHosts requires an argument");
173                 }
174             } else if (thisArg.equals("-dumpGroup")) {
175                 actions.add(Command.DUMP);
176             } else if (thisArg.equals("-removeMember")) {
177                 if (argc < nArgs) {
178                     nodeName = argv[argc++];
179                     actions.add(Command.REMOVE);
180                 } else {
181                     printUsage("-removeMember requires an argument");
182                 }
183             } else if (thisArg.equals("-updateAddress")) {
184                 if (argc < nArgs) {
185                     nodeName = argv[argc++];
186 
187                     if (argc < nArgs) {
188                         String hostPort = argv[argc++];
189                         int index = hostPort.indexOf(":");
190                         if (index < 0) {
191                             printUsage("Host port pair format must be " +
192                                        "<host name>:<port number>");
193                         }
194 
195                         newHostName = hostPort.substring(0, index);
196                         newPort = Integer.parseInt
197                             (hostPort.substring(index + 1, hostPort.length()));
198                     } else {
199                         printUsage("-updateAddress requires a " +
200                                    "<host name>:<port number> argument");
201                     }
202 
203                     actions.add(Command.UPDATE_ADDRESS);
204                 } else {
205                     printUsage
206                         ("-updateAddress requires the node name argument");
207                 }
208             } else if (thisArg.equals("-transferMaster")) {
209 
210                 // TODO: it wouldn't be too hard to allow "-force" as a
211                 // node name.
212                 //
213                 if (argc < nArgs && "-force".equals(argv[argc])) {
214                     forceFlag = true;
215                     argc++;
216                 }
217                 if (argc + 1 < nArgs) {
218                     nodeName = argv[argc++];
219 
220                     /*
221                      * Allow either
222                      *     -transferMaster mercury,venus 900 ms
223                      * or
224                      *     -transferMaster mercury,venus "900 ms"
225                      */
226                     if (argc + 1 < nArgs && argv[argc + 1].charAt(0) != '-') {
227                         timeout = argv[argc] + " " + argv[argc + 1];
228                         argc += 2;
229                     } else {
230                         timeout = argv[argc++];
231                     }
232 
233                     actions.add(Command.TRANSFER_MASTER);
234                 } else {
235                     printUsage
236                         ("-transferMaster requires at least two arguments");
237                 }
238             } else if (thisArg.equals("-netProps")) {
239                 if (argc < nArgs) {
240                     netPropsName = argv[argc++];
241                 } else {
242                     printUsage("-netProps requires an argument");
243                 }
244             } else if (thisArg.equals("-deleteMember")) {
245                 if (argc < nArgs) {
246                     nodeName = argv[argc++];
247                     actions.add(Command.DELETE);
248                 } else {
249                     printUsage("-deleteMember requires an argument");
250                 }
251             } else {
252                 printUsage(thisArg + " is not a valid argument");
253             }
254         }
255 
256         ReplicationNetworkConfig repNetConfig =
257         		ReplicationNetworkConfig.createDefault();
258         if (netPropsName != null) {
259             try {
260                 repNetConfig =
261                     ReplicationNetworkConfig.create(new File(netPropsName));
262             } catch (FileNotFoundException fnfe) {
263                 printUsage("The net properties file " + netPropsName +
264                            " does not exist: " + fnfe.getMessage());
265             } catch (IllegalArgumentException iae) {
266                 printUsage("The net properties file " + netPropsName +
267                            " is not valid: " + iae.getMessage());
268             }
269         }
270 
271         this.channelFactory = initializeFactory(repNetConfig, groupName);
272     }

274     /* Execute commands */
275     private void run()
276         throws Exception {
277 
278         createGroupAdmin();
279 
280         if (actions.size() == 0) {
281             return;
282         }
283 
284         for (Command action : actions) {
285             switch (action) {
286 
287                 /* Dump the group information. */
288             case DUMP:
289                 dumpGroup();
290                 break;
291 
292                 /* Remove a member. */
293             case REMOVE:
294                 removeMember(nodeName);
295                 break;
296 
297                 /* Transfer the current mastership to a specified node. */
298             case TRANSFER_MASTER:
299                 transferMaster(nodeName, timeout);
300                 break;
301 
302                 /* Update the network address of a specified node. */
303             case UPDATE_ADDRESS:
304                 updateAddress(nodeName, newHostName, newPort);
305                 break;
306 
307                 /* Delete a member */
308             case DELETE:
309                 deleteMember(nodeName);
310                 break;
311 
312             default:
313                 throw new AssertionError();
314             }
315         }
316     }

516     private String getFormattedOutput() {
517         StringBuilder sb = new StringBuilder();
518         RepGroupImpl repGroupImpl = groupAdmin.getGroup().getRepGroupImpl();
519 
520         /* Get the master node name. */
521         String masterName = groupAdmin.getMasterNodeName();
522 
523         /* Get the electable nodes information. */
524         sb.append("\nGroup: " + repGroupImpl.getName() + "\n");
525         sb.append("Electable Members:\n");
526         Set<RepNodeImpl> nodes = repGroupImpl.getElectableMembers();
527         if (nodes.size() == 0) {
528             sb.append("    No electable members\n");
529         } else {
530             for (RepNodeImpl node : nodes) {
531                 String type =
532                     masterName.equals(node.getName()) ? "master, " : "";
533                 sb.append("    " + node.getName() + " (" + type +
534                           node.getHostName() + ":" + node.getPort() + ", " +
535                           node.getBarrierState() + ")\n");
536             }
537         }
538 
539         /* Get the monitors information. */
540         sb.append("\nMonitor Members:\n");
541         nodes = repGroupImpl.getMonitorMembers();
542         if (nodes.size() == 0) {
543             sb.append("    No monitors\n");
544         } else {
545             for (RepNodeImpl node : nodes) {
546                 sb.append("    " + node.getName() + " (" + node.getHostName() +
547                           ":" + node.getPort() + ")\n");
548             }
549         }
550 
551         /* Get information about secondary nodes */
552         sb.append("\nSecondary Members:\n");
553         nodes = repGroupImpl.getSecondaryMembers();
554         if (nodes.isEmpty()) {
555             sb.append("    No secondary members\n");
556         } else {
557             for (final RepNodeImpl node : nodes) {
558                 sb.append("    " + node.getName() + " (" + node.getHostName() +
559                           ":" + node.getPort() + ", " +
560                           node.getBarrierState() + ")\n");
561             }
562         }
563 
564         /* Get information about external nodes */
565         sb.append("\nExternal Members:\n");
566         nodes = repGroupImpl.getExternalMembers();
567         if (nodes.isEmpty()) {
568             sb.append("    No external members\n");
569         } else {
570             for (final RepNodeImpl node : nodes) {
571                 sb.append("    " + node.getName() + " (" + node.getHostName() +
572                           ":" + node.getPort() + ", " +
573                           node.getBarrierState() + ")\n");
574             }
575         }
576 
577         return sb.toString();
578     }
