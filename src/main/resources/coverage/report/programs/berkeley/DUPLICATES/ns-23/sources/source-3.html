70     /**
71      * Start subscription main thread, subscribe from the very first VLSN
72      * from the feeder. The subscriber will stay alive and consume all entries
73      * until it shuts down.
74      *
75      * @throws InsufficientLogException if feeder is unable to stream from
76      *                                  start VLSN
77      * @throws GroupShutdownException   if subscription receives group shutdown
78      * @throws InternalException        if internal exception
79      * @throws TimeoutException         if subscription initialization timeout
80     */
81     public void start()
82         throws IllegalArgumentException, InsufficientLogException,
83         GroupShutdownException, InternalException, TimeoutException {
84 
85         start(VLSN.FIRST_VLSN);
86     }

88     /**
89      * Start subscription main thread, subscribe from a specific VLSN
90      * from the feeder. The subscriber will stay alive and consume all entries
91      * until it shuts down.
92      *
93      * @param vlsn the start VLSN of subscription. It cannot be NULL_VLSN
94      *             otherwise an IllegalArgumentException will be raised.
95      *
96      * @throws InsufficientLogException if feeder is unable to stream from
97      *                                  start VLSN
98      * @throws GroupShutdownException   if subscription receives group shutdown
99      * @throws InternalException        if internal exception
100      * @throws TimeoutException         if subscription initialization timeout
101      */
102     public void start(VLSN vlsn)
103         throws IllegalArgumentException, InsufficientLogException,
104         GroupShutdownException, InternalException, TimeoutException {
105 
106         if (vlsn.equals(VLSN.NULL_VLSN)) {
107             throw new IllegalArgumentException("Start VLSN cannot be null");
108         }
109 
110         subscriptionThread =
111             new SubscriptionThread(dummyRepEnv, vlsn,
112                                    configuration, statistics,
113                                    logger);
114         /* fire the subscription thread */
115         subscriptionThread.start();
116 
117         if (!waitForSubscriptionInitDone(subscriptionThread)) {
118             LoggerUtils.warning(logger,
119                                 RepInternal.getNonNullRepImpl(dummyRepEnv),
120                                 "Timeout in initialization, shut down " +
121                                 "subscription.");
122             shutdown();
123             throw new TimeoutException("Subscription initialization timeout " +
124                                        "after " +
125                                        configuration.getPollTimeoutMs() +
126                                        " ms");
127         }
128 
129         /* if not success, throw exception to caller */
130         final Exception exp = subscriptionThread.getStoredException();
131         switch (subscriptionThread.getStatus()) {
132             case SUCCESS:
133                 break;
134 
135             case VLSN_NOT_AVAILABLE:
136                 /* shutdown and close env before throw exception to client */
137                 shutdown();
138                 throw (InsufficientLogException) exp;
139 
140             case GRP_SHUTDOWN:
141                 /* shutdown and close env before throw exception to client */
142                 shutdown();
143                 throw (GroupShutdownException) exp;
144 
145             case UNKNOWN_ERROR:
146             case CONNECTION_ERROR:
147             default:
148                 /* shutdown and close env before throw exception to client */
149                 shutdown();
150                 throw new InternalException("internal exception from " +
151                                             "subscription thread, err:" +
152                                             exp.getMessage(), exp);
153         }
154     }
155 
156     /**
157      * Shutdown a subscription completely
158      */
159     public void shutdown() {
160         if (subscriptionThread != null && subscriptionThread.isAlive()) {
161             subscriptionThread.shutdown();
162         }
163         subscriptionThread = null;
164 
165         if (dummyRepEnv != null) {
166             final NodeType nodeType = configuration.getNodeType();
167             if (nodeType.hasTransientId() && !dummyRepEnv.isClosed()) {
168                 RepInternal.getNonNullRepImpl(dummyRepEnv)
169                            .getNameIdPair()
170                            .revertToNull();
171             }
172             dummyRepEnv.close();
173             logger.fine("Closed env " + dummyRepEnv.getNodeName() +
174                         "(forget transient id? " +
175                         nodeType.hasTransientId() + ")");
176         }
177     }

222     /**
223      * Create a dummy replicated env used by subscription. The dummy env will
224      * be used in the SubscriptionThread, SubscriptionProcessMessageThread and
225      * SubscriptionOutputThread to connect to feeder.
226      *
227      * @param conf   subscription configuration
228      * @param logger logger
229      * @return a replicated environment
230      * @throws IllegalArgumentException if env directory does not exist
231      */
232     private static ReplicatedEnvironment
233     createDummyRepEnv(SubscriptionConfig conf, Logger logger)
234         throws IllegalArgumentException {
235 
236         final ReplicatedEnvironment ret;
237         final File envHome = new File(conf.getSubscriberHome());
238         if (!envHome.exists()) {
239             throw new IllegalArgumentException("Env directory " +
240                                                envHome.getAbsolutePath() +
241                                                " does not exist.");
242         }
243 
244         ret =
245             RepInternal.createInternalEnvHandle(envHome,
246                                                 conf.createReplicationConfig(),
247                                                 conf.createEnvConfig());
248 
249         /*
250          * A safety check and clear id if necessary, to prevent env with
251          * existing id from failing the subscription
252          */
253         final NameIdPair pair = RepInternal.getNonNullRepImpl(ret)
254                                            .getNameIdPair();
255         if (conf.getNodeType().hasTransientId() && !pair.hasNullId()) {
256             logger.fine("Env has a non-null id, clear its id(name id: " +
257                         pair + ")");
258             pair.revertToNull();
259         }
260         logger.fine("Env created with name id pair " + pair);
261         return ret;
262     }

264     /**
265      * Wait for subscription thread to finish initialization
266      *
267      * @param t thread of subscription
268      * @return true if init done successfully, false if timeout
269      */
270     private boolean waitForSubscriptionInitDone(final SubscriptionThread t) {
271         return new PollCondition(configuration.getPollIntervalMs(),
272                                  configuration.getPollTimeoutMs()) {
273             @Override
274             protected boolean condition() {
275                 return t.getStatus() != SubscriptionStatus.INIT;
276             }
277 
278         }.await();
279     }
