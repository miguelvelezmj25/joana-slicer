230     private K getFirstOrLastKey(boolean doGetFirst) {
231 
232         DataCursor cursor = null;
233         try {
234             cursor = new DataCursor(view, false);
235             OperationStatus status;
236             if (doGetFirst) {
237                 status = cursor.getFirst(false);
238             } else {
239                 status = cursor.getLast(false);
240             }
241             return (K) ((status == OperationStatus.SUCCESS) ?
242                 cursor.getCurrentKey() :
243                 null);
244         } catch (Exception e) {
245             throw StoredContainer.convertException(e);
246         } finally {
247             closeCursor(cursor);
248         }
249     }
