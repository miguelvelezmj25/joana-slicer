74 /**
75  * This class implements a hot network backup that permits it to obtain a
76  * consistent set of log files from any running environment that provides a
77  * LogFileFeeder service. This class thus plays the role of a client, and the
78  * running environment that of a server.
79  * <p>
80  * The log files that are retrieved over the network are placed in a directory
81  * that can serve as an environment directory for a JE stand alone or HA
82  * environment. If log files are already present in the target directory, it
83  * will try reuse them, if they are really consistent with those on the server.
84  * Extant log files that are no longer part of the current backup file set are
85  * deleted or are renamed, depending on how the backup operation was
86  * configured.
87  * <p>
88  * Renamed backup files have the following syntax:
89  *
90  * NNNNNNNN.bup.<backup number>
91  *
92  * where the backup number is the number associated with the backup attempt,
93  * rather than with an individual file. That is, the backup number is increased
94  * by one each time a backup is repeated in the same directory and log files
95  * actually needed to be renamed.
96  * <p>
97  * The implementation tries to be resilient in the face of network failures and
98  * minimizes the amount of work that might need to be done if the client or
99  * server were to fail and had to be restarted. Users of this API must be
100  * careful to ensure that the execute() completes successfully before accessing
101  * the environment. The user fails to do this, the InsufficientLogException
102  * will be thrown again when the user attempts to open the environment. This
103  * safeguard is implemented using the {@link RestoreMarker} mechanism.
104  */

463     /**
464      * Retrieves all the files in the list, that are not already in the envDir.
465      * @throws DatabaseException
466      */
467     private void getFiles(String[] fileNames)
468         throws IOException, DatabaseException,
469                 RestoreMarker.FileCreationException {
470 
471         LoggerUtils.info(logger, repImpl,
472             fileNames.length + " files in backup set");
473 
474         /* Get all file transfer lengths first, so we can track progress */
475         final List<FileAndLength> fileTransferLengths =
476             getFileTransferLengths(fileNames);
477 
478         for (final FileAndLength entry : fileTransferLengths) {
479             if (testBarrier != null) {
480                 try {
481                     testBarrier.await();
482                 } catch (InterruptedException e) {
483                     // Ignore just a test mechanism
484                 } catch (BrokenBarrierException e) {
485                     throw EnvironmentFailureException.unexpectedException(e);
486                 }
487             }
488 
489             for (int i = 0; i < DIGEST_RETRIES; i++) {
490                 try {
491                     getFile(entry.file);
492                     fetchCount.increment();
493                     break;
494                 } catch (DigestException e) {
495                     if ((i + 1) == DIGEST_RETRIES) {
496                         throw new IOException("Digest mismatch despite "
497                             + DIGEST_RETRIES + " attempts");
498                     }
499 
500                     /* Account for the additional transfer */
501                     expectedBytes.add(entry.length);
502                     continue;
503                 }
504             }
505         }
506         /* We've finished transferring all files, remove the marker file. */
507         restoreMarker.removeMarkerFile(fileManager);
508 
509         /* All done, shutdown conversation with the server. */
510         protocol.write(protocol.new Done(), channel);
511     }

523     /**
524      * Returns information about files that need to be transferred, and updates
525      * expectedBytes and skipCount accordingly.  This method tries to avoid
526      * requesting the SHA1 if the file lengths are not equal, since computing
527      * the SHA1 if it's not already cached requires a pass over the log
528      * file. Note that the server will always send back the SHA1 value if it
529      * has it cached.
530      */
531     private List<FileAndLength> getFileTransferLengths(String[] fileNames)
532         throws IOException, DatabaseException {
533 
534         final List<FileAndLength> fileTransferLengths = new ArrayList<>();
535         for (final String fileName : fileNames) {
536 
537             /*
538              * Use the full path of this file in case the environment uses
539              * multiple data directories.
540              */
541             final File file = new File(fileManager.getFullFileName(fileName));
542             protocol.write(protocol.new FileInfoReq(fileName, false), channel);
543             FileInfoResp statResp =
544                 protocol.read(channel, FileInfoResp.class);
545             final long fileLength = statResp.getFileLength();
546 
547             /*
548              * See if we can skip the file if it is present with correct length
549              */
550             if (file.exists() && (fileLength == file.length())) {
551 
552                 /* Make sure we have the message digest */
553                 if (statResp.getDigestSHA1().length == 0) {
554                     protocol.write(
555                         protocol.new FileInfoReq(fileName, true), channel);
556                     statResp =
557                         protocol.read(channel, FileInfoResp.class);
558                 }
559                 final byte digest[] =
560                     LogFileFeeder.getSHA1Digest(file, fileLength).digest();
561                 if (Arrays.equals(digest, statResp.getDigestSHA1())) {
562                     LoggerUtils.info(logger, repImpl,
563                         "File: " + file.getCanonicalPath() +
564                         " length: " + fileLength +
565                         " available with matching SHA1, copy skipped");
566                     skipCount.increment();
567                     continue;
568                 }
569             }
570             fileTransferLengths.add(new FileAndLength(file, fileLength));
571             expectedBytes.add(fileLength);
572         }
573         return fileTransferLengths;
574     }

576     /**
577      * Requests and obtains the specific log file from the server. The file is
578      * first created under a name with the .tmp suffix and is renamed to its
579      * true name only after its digest has been verified.
580      *
581      * This method is protected to facilitate error testing.
582      */
583     protected void getFile(File file)
584         throws IOException, ProtocolException, DigestException,
585                 RestoreMarker.FileCreationException {
586 
587         LoggerUtils.fine(logger, repImpl, "Requesting file: " + file);
588         protocol.write(protocol.new FileReq(file.getName()), channel);
589         FileStart fileResp = protocol.read(channel, FileStart.class);
590 
591         /*
592          * Delete the tmp file if it already exists.
593          *
594          * Use the full path of this file in case the environment uses multiple
595          * data directories.
596          */
597         File tmpFile = new File(fileManager.getFullFileName(file.getName()) +
598                                 FileManager.TMP_SUFFIX);
599         if (tmpFile.exists()) {
600             boolean deleted = tmpFile.delete();
601             if (!deleted) {
602                 throw EnvironmentFailureException.unexpectedState
603                     ("Could not delete file: " + tmpFile);
604             }
605         }
606 
607         /*
608          * Use a direct buffer to avoid an unnecessary copies into and out of
609          * native buffers.
610          */
611         final ByteBuffer buffer =
612                 ByteBuffer.allocateDirect(LogFileFeeder.TRANSFER_BYTES);
613         messageDigest.reset();
614 
615         /* Write the tmp file. */
616         final FileOutputStream fileStream = new FileOutputStream(tmpFile);
617         final FileChannel fileChannel = fileStream.getChannel();
618 
619         try {
620             /* Compute the transfer rate roughly once each MB */
621             final int rateInterval = 0x100000 / LogFileFeeder.TRANSFER_BYTES;
622             int count = 0;
623 
624             /* Copy over the file contents. */
625             for (long bytes = fileResp.getFileLength(); bytes > 0;) {
626                 int readSize =
627                     (int) Math.min(LogFileFeeder.TRANSFER_BYTES, bytes);
628                 buffer.clear();
629                 buffer.limit(readSize);
630                 int actualBytes = channel.read(buffer);
631                 if (actualBytes == -1) {
632                     throw new IOException("Premature EOF. Was expecting:"
633                                           + readSize);
634                 }
635                 bytes -= actualBytes;
636 
637                 buffer.flip();
638                 fileChannel.write(buffer);
639 
640                 buffer.rewind();
641                 messageDigest.update(buffer);
642                 transferredBytes.add(actualBytes);
643 
644                 /* Update the transfer rate at interval and last time */
645                 if (((++count % rateInterval) == 0) || (bytes <= 0)) {
646                     transferRate.add(
647                         transferredBytes.get(), System.currentTimeMillis());
648                 }
649             }
650 
651             if (logger.isLoggable(Level.INFO)) {
652                 LoggerUtils.info(logger, repImpl,
653                     String.format(
654                         "Fetched log file: %s, size: %,d bytes," +
655                         " %s bytes," +
656                         " %s bytes," +
657                         " %s bytes/second",
658                         file.getName(),
659                         fileResp.getFileLength(),
660                         transferredBytes,
661                         expectedBytes,
662                         transferRate));
663             }
664         } finally {
665             fileStream.close();
666         }
667 
668         final FileEnd fileEnd = protocol.read(channel, FileEnd.class);
669 
670         /* Check that the read is successful. */
671         if (!Arrays.equals(messageDigest.digest(), fileEnd.getDigestSHA1())) {
672             LoggerUtils.warning(logger, repImpl,
673                 "digest mismatch on file: " + file);
674             throw new DigestException();
675         }
676 
677         /*
678          * We're about to alter the files that exist in the log, either by
679          * deleting file N.jdb, or by renaming N.jdb.tmp -> N, and thereby
680          * adding a file to the set in the directory. Create the marker that
681          * says this log is no longer coherent and can't be recovered. Marker
682          * file creation can safely be called multiple times; the file will
683          * only be created the first time.
684          */
685         restoreMarker.createMarkerFile
686             (RestoreRequired.FailureType.NETWORK_RESTORE,
687              exceptionProperties);
688 
689         assert TestHookExecute.doHookIfSet(interruptHook, file);
690 
691         /* Now that we know it's good, move the file into place. */
692         if (file.exists()) {
693             /*
694              * Delete or back up this and all subsequent obsolete files,
695              * excluding the marker file. The marker file will be explicitly
696              * cleaned up when the entire backup finishes.
697              */
698             disposeObsoleteFiles(file);
699         }
700 
701         /* Rename the tmp file. */
702         LoggerUtils.fine(logger, repImpl, "Renamed " + tmpFile + " to " + file);
703         boolean renamed = tmpFile.renameTo(file);
704         if (!renamed) {
705             throw EnvironmentFailureException.unexpectedState
706                 ("Rename from: " + tmpFile + " to " + file + " failed");
707         }
708 
709         /* Retain last modified time, to leave an audit trail. */
710         if (!file.setLastModified(fileResp.getLastModifiedTime())) {
711             throw EnvironmentFailureException.unexpectedState
712                 ("File.setlastModifiedTime() for:" + file + " and time " +
713                  new Date(fileResp.getLastModifiedTime()) + " failed.");
714         }
715     }

717     /**
718      * Renames (or deletes) this log file, and all other files following it in
719      * the log sequence. The operation is done from the highest file down to
720      * this one, to ensure the integrity of the log files in the directory is
721      * always preserved. Exclude the marker file because that is meant to serve
722      * as an indicator that the backup is in progress. It will be explicitly
723      * removed only when the entire backup is finished.
724      *
725      * @param startFile the lowest numbered log file that must be renamed or
726      * deleted
727      * @throws IOException
728      */
729     private void disposeObsoleteFiles(File startFile) throws IOException {
730         File[] dirFiles = fileManager.listJDBFiles();
731         Arrays.sort(dirFiles); // sorts in ascending order
732 
733         /* Start with highest numbered file to be robust in case of failure. */
734         for (int i = dirFiles.length - 1; i >= 0; i--) {
735             File file = dirFiles[i];
736 
737             /* Skip the marker file, wait until the whole backup is done */
738             if (file.getName().equals(RestoreMarker.getMarkerFileName())) {
739                 continue;
740             }
741             disposeFile(file);
742             if (startFile.equals(file)) {
743                 break;
744             }
745         }
746     }

748     /**
749      * Remove the file from the current set of log files in the directory.
750      * @param file
751      */
752     private void disposeFile(File file) {
753         disposedCount.increment();
754         final long fileNumber = fileManager.getNumFromName(file.getName());
755         if (retainLogfiles) {
756             boolean renamed = false;
757             try {
758                 renamed =
759                     fileManager.renameFile(fileNumber, FileManager.BUP_SUFFIX);
760             } catch (IOException e) {
761                 throw EnvironmentFailureException.unexpectedState
762                     ("Could not rename log file " + file.getPath() +
763                      " because of exception: " + e.getMessage());
764             }
765 
766             if (!renamed) {
767                 throw EnvironmentFailureException.unexpectedState
768                     ("Could not rename log file " +  file.getPath());
769             }
770             LoggerUtils.fine(logger, repImpl,
771                 "Renamed log file: " + file.getPath());
772         } else {
773             boolean deleted = false;
774             try {
775                 deleted = fileManager.deleteFile(fileNumber);
776             } catch (IOException e) {
777                 throw EnvironmentFailureException.unexpectedException
778                     ("Could not delete log file " + file.getPath() +
779                      " during network restore.", e);
780             }
781             if (!deleted) {
782                 throw EnvironmentFailureException.unexpectedState
783                     ("Could not delete log file " +  file.getPath());
784             }
785             LoggerUtils.fine(logger, repImpl,
786                 "deleted log file: " + file.getPath());
787         }
788     }
