309     /**
310      * Finds the key according to the Search param. If dups are configured, GT
311      * and GTE will land on the first dup for the matching key, while LT and
312      * LTE will land on the last dup for the matching key.
313      *
314      * search() and searchBoth() in this class may eventually be exposed as
315      * public JE Cursor methods, but this isn't practical now for the following
316      * reasons:
317      *
318      *  + The API design needs more thought. Perhaps Search.EQ should be added.
319      *    Perhaps existing Cursor methods should be deprecated.
320      *
321      *  + This implementation moves the cursor multiple times and does not
322      *    release locks on the intermediate records.
323      *
324      *  + This could be implemented more efficiently using lower level cursor
325      *    code. For example, an LTE search would actually more efficient than
326      *    the existing GTE search (getSearchKeyRange and getSearchBothRange).
327      *
328      * These methods are used by KVStore.
329      */
330     public static OperationResult search(
331         final Cursor cursor,
332         final DatabaseEntry key,
333         final DatabaseEntry pKey,
334         final DatabaseEntry data,
335         final Search searchMode,
336         final ReadOptions options) {
337 
338         final DatabaseImpl dbImpl = cursor.getDatabaseImpl();
339         KeyRange range = new KeyRange(dbImpl.getBtreeComparator());
340         final boolean first;
341 
342         switch (searchMode) {
343         case GT:
344         case GTE:
345             range = range.subRange(
346                 key, searchMode == Search.GTE, null, false);
347             first = true;
348             break;
349         case LT:
350         case LTE:
351             range = range.subRange(
352                 null, false, key, searchMode == Search.LTE);
353             first = false;
354             break;
355         default:
356             throw EnvironmentFailureException.unexpectedState();
357         }
358 
359         final RangeCursor rangeCursor = new RangeCursor(
360             range, null, dbImpl.getSortedDuplicates(), cursor);
361 
362         final OpReadOptions opReadOptions = OpReadOptions.make(options);
363 
364         final OpResult result = (first) ?
365             rangeCursor.getFirst(key, pKey, data, opReadOptions) :
366             rangeCursor.getLast(key, pKey, data, opReadOptions);
367 
368         /* RangeCursor should not have dup'd the cursor. */
369         assert cursor == rangeCursor.getCursor();
370 
371         return result.jeResult;
372     }

393      * Searches with the dups for the given key and finds the dup matching the
394      * pKey value, according to the Search param.
395      *
396      * See search() for more discussion.
397      */
398     public static OperationResult searchBoth(
399         final Cursor cursor,
400         final DatabaseEntry key,
401         final DatabaseEntry pKey,
402         final DatabaseEntry data,
403         final Search searchMode,
404         final ReadOptions options) {
405 
406         final LockMode lockMode =
407             options != null ? options.getLockMode() : null;
408 
409         final DatabaseImpl dbImpl = cursor.getDatabaseImpl();
410         KeyRange range = new KeyRange(dbImpl.getBtreeComparator());
411         range = range.subRange(key);
412         KeyRange pKeyRange = new KeyRange(dbImpl.getDuplicateComparator());
413         final boolean first;
414 
415         switch (searchMode) {
416         case GT:
417         case GTE:
418             pKeyRange = pKeyRange.subRange(
419                 pKey, searchMode == Search.GTE, null, false);
420             first = true;
421             break;
422         case LT:
423         case LTE:
424             pKeyRange = pKeyRange.subRange(
425                 null, false, pKey, searchMode == Search.LTE);
426             first = false;
427             break;
428         default:
429             throw EnvironmentFailureException.unexpectedState();
430         }
431 
432         final RangeCursor rangeCursor = new RangeCursor(
433             range, pKeyRange, dbImpl.getSortedDuplicates(), cursor);
434 
435         final OpReadOptions opReadOptions = OpReadOptions.make(options);
436 
437         final OpResult result = (first) ?
