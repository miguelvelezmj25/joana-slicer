26 /**
27  * This log entry is used to indicate that the environment's log files are not
28  * recoverable and that some sort of curative action should happen first. It's
29  * a general purpose mechanism that can be used for many types of errors.
30  */
31 public class RestoreRequired implements Loggable {

47     public RestoreRequired(FailureType failureType,

49         this.failureType = failureType;
