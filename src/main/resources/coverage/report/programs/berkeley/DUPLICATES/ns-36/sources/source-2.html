25  * If someone add a property in some FooConfig.java,
26  *   (1) If the setter/getter methods are setFoo/getFoo, the name of the
27  *       property should be "foo", which means the first letter of the property
28  *       name should be lower case.
29  *   (2) The setter method for this property setProperty should return "this",
30  *       and setPropertyVoid method which returns void value must be added.
31  *       The return type of the getter method should be the same as the
32  *       parameter of the setter method.
33  *   (3) The setter method and getter method must be added into
34  *       FooConfigBeanInfo;
35  *   (4) If for some of the setter methods in the FooConfig.java, setterVoid
36  *       methods are not necessary, then add the name of such setter methods
37  *       into the ArrayList ignoreMethods within the corresponding
38  *       FooConfigBeanInfo.getPropertyDescriptors method. For example,
39  *       setMaxSeedTestHook method in DiskOrderedCursorConfig.java is only used
40  *       for unit tests, so "setMaxSeedTestHook" is added into ignoreMethods
41  *       list within DiskOrderedCursorConfigBeanInfo.getPropertyDescriptors.
42  *
43  *
44  * If someone adds a new FooConfig.java,
45  *   (1) The definition of setter/getter mehods and the names of the properties
46  *       should follow the rules described above.
47  *   (2) There must be FooConfigBeanInfo.java. You can write it according to
48  *       the current beaninfo classes.

75      * Get the propertis' infomation, including all the properties's names
76      * and their getter/setter methods.
77      */
78     protected static void getPropertiesInfo(Class cls) {
79         propertiesName.clear();
80         getterAndSetterMethods.clear();
81         try {
82 
83             /* Get all of the public methods. */
84             ArrayList<String> allMethodNames = new ArrayList<String>();
85             Method[] methods = cls.getMethods();
86             for (int i = 0; i < methods.length; i++) {
87                 allMethodNames.add(methods[i].getName());
88             }
89             for (int i = 0; i < allMethodNames.size(); i++) {
90                 String name = allMethodNames.get(i);
91                 String subName = name.substring(0, 3);
92 
93                 /* If it is a setter method. */
94                 if (subName.equals("set")) {
95                     if (isIgnoreMethods(name)) {
96                         continue;
97                     }
98                     String propertyName = name.substring(3);
99                     Method getterMethod = null;
100                     try {
101                         getterMethod = cls.getMethod("get" + propertyName);
102                     } catch (NoSuchMethodException e) {
103                         getterMethod = null;
104                     }
105                     if (getterMethod != null) {
106                         getterAndSetterMethods.add("get" + propertyName);
107                         getterAndSetterMethods.add(name + "Void");
108 
109                         /*
110                          * Add the real property name into propertiesName.
111                          * if the names of setter/getter methods are
112                          * setFoo/getFoo, the name of the property should be
113                          * "foo".
114                          */
115                         propertiesName.add
116                             (propertyName.substring(0, 1).toLowerCase() +
117                              propertyName.substring(1));
118                     }
119                 }
120             }
121         } catch (SecurityException e) {
122             e.printStackTrace();
123         }
124     }

126     private static boolean isIgnoreMethods(String methodName) {
127         for (int i = 0; i < ignoreMethods.size(); i++) {
128             if (ignoreMethods.get(i).equals(methodName)) {
129                 return true;
130             }
131         }
132         return false;
133     }

135     protected static PropertyDescriptor[] getPdescriptor(Class cls) {
136         getPropertiesInfo(cls);
137         final int propertyNum = propertiesName.size();
138         assert propertyNum * 2 == getterAndSetterMethods.size();
139         PropertyDescriptor[] properties = new PropertyDescriptor[propertyNum];
140         try {
141             for (int i = 0, j = 0; i < propertyNum; i += 1, j += 2) {
142                 properties[i] = new PropertyDescriptor
143                     (propertiesName.get(i), cls, getterAndSetterMethods.get(j),
144                      getterAndSetterMethods.get(j + 1));
145             }
146         } catch(IntrospectionException e) {
147             e.printStackTrace();
148         }
149         return properties;
150     }
