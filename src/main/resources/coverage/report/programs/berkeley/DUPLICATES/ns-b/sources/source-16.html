21 /**
22  * NodeSequence encapsulates the generation and maintenance of a sequence for
23  * generating node IDs, transient LSNs and other misc sequences.
24  */
25 public class NodeSequence {
26 
27     public static final int FIRST_LOCAL_NODE_ID = 1;
28     public static final int FIRST_REPLICATED_NODE_ID = -10;
29 
30     /*
31      * Node IDs: We need to ensure that local and replicated nodes use
32      * different number spaces for their ids, so there can't be any possible
33      * conflicts.  Local, non replicated nodes use positive values starting
34      * with 1, replicated nodes use negative values starting with -10.
35      *
36      * Node ID values from 0 to -9 are reserved.  0 is not used and should be
37      * avoided.  -1 is used to mean null or none, and should be used via the
38      * Node.NULL_NODE_ID constant.  -2 through -9 are reserved for future use.
39      *
40      * The local and replicated node ID sequences are initialized by the first
41      * pass of recovery, after the log has been scanned for the latest used
42      * node ID.
43      */
44     private AtomicLong lastAllocatedLocalNodeId = null;
45     private AtomicLong lastAllocatedReplicatedNodeId = null;
46 
47     /*
48      * Transient LSNs are used for not-yet-logged DeferredWrite records and
49      * for the EOF record used for Serializable isolation. Transient LSNs are
50      * used to provide unique locks, and are only used during the life of an
51      * environment, for non-persistent objects.
52      */
53     private final AtomicLong lastAllocatedTransientLsnOffset =
54         new AtomicLong(0L);
55 
56     public final EnvironmentImpl envImpl;
57 
58     /* Transient sequences. */
59     private final AtomicLong nextBackupId = new AtomicLong(0L);
60     private final AtomicLong nextDatabaseCountId = new AtomicLong(0L);
61     private final AtomicLong nextDiskOrderedCursorId = new AtomicLong(0L);
62     private final AtomicLong nextNetworkRestoreId = new AtomicLong(0L);
63 
64     public NodeSequence(EnvironmentImpl envImpl) {
65         this.envImpl = envImpl;
66     }
67 
68     /**
69      * Initialize the counters in these methods rather than a constructor
70      * so we can control the initialization more precisely.
71      */
72     void initRealNodeId() {
73         lastAllocatedLocalNodeId = new AtomicLong(FIRST_LOCAL_NODE_ID - 1);
74         lastAllocatedReplicatedNodeId =
75             new AtomicLong(FIRST_REPLICATED_NODE_ID + 1);
76     }
77 
78     /**
79      * The last allocated local and replicated node IDs are used for ckpts.
80      */
81     public long getLastLocalNodeId() {
82         return lastAllocatedLocalNodeId.get();
83     }
84 
85     public long getLastReplicatedNodeId() {
86         return lastAllocatedReplicatedNodeId.get();
87     }
88 
89     /**
90      * We get a new node ID of the appropriate kind when creating a new node.
91      */
92     public long getNextLocalNodeId() {
93         return lastAllocatedLocalNodeId.incrementAndGet();
94     }
95

99     }
100     */
101 
102     /**
103      * Initialize the node IDs, from recovery.
104      */
105     public void setLastNodeId(long lastReplicatedNodeId,
106                               long lastLocalNodeId) {
107         lastAllocatedReplicatedNodeId.set(lastReplicatedNodeId);
108         lastAllocatedLocalNodeId.set(lastLocalNodeId);
109     }
110 
111     /*
112      * Tracks the lowest replicated node ID used during a replay of the
113      * replication stream, so that it's available as the starting point if this
114      * replica transitions to being the master.
115      */
116     public void updateFromReplay(long replayNodeId) {
117         assert !envImpl.isMaster();
118         if (replayNodeId > 0 && !envImpl.isRepConverted()) {
119            throw EnvironmentFailureException.unexpectedState
120                ("replay node id is unexpectedly positive " + replayNodeId);
121         }
122 
123         if (replayNodeId < lastAllocatedReplicatedNodeId.get()) {
124             lastAllocatedReplicatedNodeId.set(replayNodeId);
125         }
126     }
127 
128     /**
129      * Assign the next available transient LSN.
130      */
131     public long getNextTransientLsn() {
132         return DbLsn.makeTransientLsn
133             (lastAllocatedTransientLsnOffset.getAndIncrement());
134     }
135
