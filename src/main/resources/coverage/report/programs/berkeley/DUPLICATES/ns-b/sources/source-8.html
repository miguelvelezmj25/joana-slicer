21 class DbEnvState {

30     public static final DbEnvState INVALID = new DbEnvState("invalid");

37     */
38 
39     DbEnvState(String name) {
40         this.name = name;
41     }
42 
43     @Override

46     }
47 
48     /* Check for valid state transitions. */
49     void checkState(DbEnvState[] validPrevStates, DbEnvState newState) {
50         if (DEBUG) {
51             System.out.println("newState = " + newState +
52                                " currentState = " + name);
53         }
54         boolean transitionOk = false;
55         for (int i = 0; i < validPrevStates.length; i++) {
56             if (this == validPrevStates[i]) {
57                 transitionOk = true;
58                 break;
59             }
60         }
61         if (!transitionOk) {
62             throw new IllegalStateException
63                 ("Can't go from environment state " + toString() +
64                  " to " + newState.toString());
65         }
66     }
