28 /**
29  * A ScavengerFileReader reads the log backwards.  If it encounters a checksum
30  * error, it goes to the start of that log file and reads forward until it
31  * encounters a checksum error.  It then continues the reading backwards in the
32  * log.
33  *
34  * The caller may set "dumpCorruptedBounds" to true if information about the
35  * start and finish of the corrupted portion should be displayed on stderr.
36  *
37  * The caller is expected to implement processEntryCallback. This method is
38  * called once for each entry that the ScavengerFileReader finds in the log.
39  */

157     protected boolean resyncReader(long nextGoodRecordPostCorruption,
158                                    boolean showCorruptedBounds)
159         throws DatabaseException {
160 
161         LastFileReader reader = null;
162         long tryReadBufferFileNum =
163             DbLsn.getFileNumber(nextGoodRecordPostCorruption);
164 
165         while (tryReadBufferFileNum >= 0) {
166             try {
167                 reader =
168                     new LastFileReader(envImpl, readBufferSize,
169                                        Long.valueOf(tryReadBufferFileNum));
170                 break;
171             } catch (ChecksumException e) {
172 
173                 /*
174                  * We encountered a problem opening this file so skip to an
175                  * earlier file.
176                  */
177                 tryReadBufferFileNum--;
178                 continue;
179             }
180         }
181 
182         boolean switchedFiles = tryReadBufferFileNum !=
183             DbLsn.getFileNumber(nextGoodRecordPostCorruption);
184 
185         if (!switchedFiles) {
186 
187             /*
188              * Read forward until a checksum fails.  This reader will not throw
189              * an exception if a checksum error is hit -- it will just return
190              * false.
191              */
192             while (reader.readNextEntry()) {
193             }
194         }
195 
196         long lastUsedLsn = reader.getLastValidLsn();
197         long nextAvailableLsn = reader.getEndOfLog();
198         if (showCorruptedBounds) {
199             System.err.println("A checksum error was found in the log.");
200             System.err.println
201                 ("Corruption begins at LSN:\n   " +
202                  DbLsn.toString(nextAvailableLsn));
203             System.err.println
204                 ("Last known good record before corruption is at LSN:\n   " +
205                  DbLsn.toString(lastUsedLsn));
206             System.err.println
207                 ("Next known good record after corruption is at LSN:\n   " +
208                  DbLsn.toString(nextGoodRecordPostCorruption));
209         }
210 
211         startLsn = lastUsedLsn;
212         initStartingPosition(nextAvailableLsn, null);
213         if (switchedFiles) {
214             currentEntryPrevOffset = 0;
215         }
216         /* Indicate resync is permitted so don't throw exception. */
217         return true;
218     }
