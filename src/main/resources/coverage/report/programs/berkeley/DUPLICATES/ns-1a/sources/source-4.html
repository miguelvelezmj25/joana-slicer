89     @Override
90     protected void cancel() {
91         Collection<DataChannel> channels;
92         synchronized (this) {
93             channels = new ArrayList<DataChannel>(activeChannels);
94             activeChannels.clear();
95         }
96         if (!channels.isEmpty()) {
97             LoggerUtils.warning
98                 (logger, repNode.getRepImpl(),
99                  "In-flight GroupService request(s) canceled: node shutdown");
100         }
101         for (DataChannel channel : channels) {
102             try {
103                 PrintWriter out =
104                     new PrintWriter(Channels.newOutputStream(channel), true);
105                 ResponseMessage rm =
106                     protocol.new Fail(FailReason.DEFAULT, "shutting down");
107                 out.println(rm.wireFormat());
108             } finally {
109                 if (channel.isOpen()) {
110                     try {
111                         channel.close();
112                     }
113                     catch (IOException e) {
114                         LoggerUtils.warning
115                             (logger, repNode.getRepImpl(),
116                              "IO error on channel close: " + e.getMessage());
117                     }
118                 }
119             }
120         }
121     }

140      * Ensures that the Monitor node, as described in the request, is a member
141      * of the group.
142      *
143      * @param ensureNode the request message describing the monitor node
144      *
145      * @return EnsureOK message if the monitor node is already part of the rep
146      * group, or was just made a part of the replication group. It returns a
147      * Fail message if it could not be made part of the group. The message
148      * associated with the response provides further details.
149      */
150     public ResponseMessage process(EnsureNode ensureNode) {
151         RepNodeImpl node = ensureNode.getNode();
152         try {
153             ensureMaster();
154             repNode.getRepGroupDB().ensureMember(node);
155             RepNodeImpl enode =
156                 repNode.getGroup().getMember(node.getName());
157             return protocol.new EnsureOK(ensureNode, enode.getNameIdPair());
158         } catch (ReplicaStateException e) {
159             return protocol.new Fail(ensureNode, FailReason.IS_REPLICA,
160                                      e.getMessage());
161         } catch (DatabaseException e) {
162             return protocol.new Fail(ensureNode, FailReason.DEFAULT,
163                                      e.getMessage());
164         }
165     }

258      * Transfer the master role from the current master to one of the specified
259      * replicas.
260      *
261      * @param transferMaster the request identifying nodes to be considered for
262      * the role of new master
263      * @return null
264      */
265     public ResponseMessage process(TransferMaster transferMaster) {
266         try {
267             ensureMaster();
268             final String nodeList = transferMaster.getNodeNameList();
269             final Set<String> replicas = parseNodeList(nodeList);
270             final long timeout = transferMaster.getTimeout();
271             final boolean force = transferMaster.getForceFlag();
272             String winner = repNode.transferMaster(replicas, timeout, force);
273             return protocol.new TransferOK(transferMaster, winner);
274         } catch (ReplicaStateException e) {
275             return protocol.new Fail(transferMaster, FailReason.IS_REPLICA,
276                                      e.getMessage());
277         } catch (MasterTransferFailureException e) {
278             return protocol.new Fail(transferMaster, FailReason.TRANSFER_FAIL,
279                                      e.getMessage());
280         } catch (DatabaseException e) {
281             return protocol.new Fail(transferMaster, FailReason.DEFAULT,
282                                      e.getMessage());
283         } catch (IllegalArgumentException e) {
284             return protocol.new Fail(transferMaster, FailReason.DEFAULT,
285                                      e.toString());
286         } catch (IllegalStateException e) {
287             return protocol.new Fail(transferMaster, FailReason.DEFAULT,
288                                      e.toString());
289         }
290     }

292     private Set<String> parseNodeList(String list) {
293         Set<String> set = new HashSet<String>();
294         StringTokenizer st = new StringTokenizer(list, ",");
295         while (st.hasMoreTokens()) {
296             set.add(st.nextToken());
297         }
298         return set;
299     }
