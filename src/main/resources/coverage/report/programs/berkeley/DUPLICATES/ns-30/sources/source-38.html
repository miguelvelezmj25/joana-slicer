20 /**
21  * Base class for all JE statistics. A single Stat embodies a value and
22  * definition. See StatGroup for a description of how to create and display
23  * statistics.
24  *
25  * Note that Stat intentionally does not contain the statistics value itself.
26  * Instead, the concrete subclass will implement the value as the appropriate
27  * primitive type. That's done to avoid wrapper classes like Integer and Long,
28  * and to  keep the overhead of statistics low.
29  */
30 public abstract class Stat<T> extends BaseStat<T> implements Cloneable {
31     private static final long serialVersionUID = 1L;
32 
33     public static final DecimalFormat FORMAT =
34         new DecimalFormat("###,###,###,###,###,###,###");
35 
36     protected final StatDefinition definition;
37 
38     /**
39      * A stat registers itself with an owning group.
40      */
41     Stat(StatGroup group, StatDefinition definition) {
42         this.definition = definition;
43         group.register(this);
44     }
45 
46     /**

48      * creating copies without using clone. For constructing an unregistered
49      * instance.

52         this.definition = definition;
53     }
