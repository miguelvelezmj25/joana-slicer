18 /**
19  * The quorum policy determine the number of nodes that must participate to
20  * pick the winner of an election, and therefore the master of the group.
21  * The default quorum policy during the lifetime of the group is
22  * QuorumPolicy.SIMPLE_MAJORITY. The only time that the application needs to
23  * specify a specific quorum policy is at node startup time, by passing one
24  * to the {@link ReplicatedEnvironment} constructor.
25  *
26  * <p>Note that {@link NodeType#SECONDARY} nodes are not counted as part of
27  * master election quorums.
28  */
29 public enum QuorumPolicy {
30 
31     /**
32      * All participants are required to vote.
33      */
34     ALL,
35 
36      /**
37       *  A simple majority of participants is required to vote.
38       */
39     SIMPLE_MAJORITY;
40 
41     /**
42      * Returns the minimum number of nodes to needed meet the quorum policy.
43      *
44      * @param groupSize the number of election participants in the replication
45      *        group
46      *
47      * @return the number of nodes that are needed for a quorum for a group
48      *         with {@code groupSize} number of election participants
49      */
50     public int quorumSize(int groupSize) {
51         switch (this) {
52             case ALL:
53                 return groupSize;
54 
55             case SIMPLE_MAJORITY:
56                 return (groupSize / 2 + 1);
57 
58             default:
59                 throw EnvironmentFailureException.unexpectedState
60                     ("Unknown quorum:" + this);
61         }
62     }
63 }
