24 /**
25  * A ReplicaThreadLocker is used with a user initiated non-transactional
26  * operation on a Replica, for a replicated DB.
27  *
28  * Like ReadonlyTxn, it enforces read-only semantics and implements the
29  * ReplicaConsistencyPolicy.  Unlike ReadonlyTxn, the environment default
30  * ReplicaConsistencyPolicy is enforced rather than the policy specified via
31  * TransactionConfig, and this policy is enforced via the openCursorHook rather
32  * than the txnBeginHook.
33  */
34 public class ReplicaThreadLocker extends ThreadLocker {
35     
36     private final RepImpl repImpl;
37 
38     public ReplicaThreadLocker(final RepImpl repImpl) {
39         super(repImpl);
40         this.repImpl = repImpl;
41     }
42 
43     @Override
44     public ThreadLocker newEmptyThreadLockerClone() {
45         return new ReplicaThreadLocker(repImpl);
46     }
47 
48     /**
49      * This overridden method prevents writes on a replica.  This check is
50      * redundant because Cursor won't allow writes to a transactional database
51      * when no Transaction is specified.  But we check here also for safety and
52      * for consistency with ReadonlyTxn.
53      */
54     @Override
55     public LockResult lockInternal(final long lsn,
56                                    final LockType lockType,
57                                    final boolean noWait,
58                                    final boolean jumpAheadOfWaiters,
59                                    final DatabaseImpl database) {
60         if (lockType.isWriteLock() && !database.allowReplicaWrite()) {
61             disallowReplicaWrite();
62         }
63         return super.lockInternal(lsn, lockType, noWait, jumpAheadOfWaiters,
64                                   database);
65     }
66 
67     /**
68      * If logging occurs before locking, we must screen out write locks here.
69      *
70      * If we allow the operation (e.g., for a non-replicated DB), then be sure
71      * to call the base class method to prepare to undo in the (very unlikely)
72      * event that logging succeeds but locking fails. [#22875]
73      */
74     @Override
75     public void preLogWithoutLock(DatabaseImpl database) {
76         if (!database.allowReplicaWrite()) {
77             disallowReplicaWrite();
78         }
79         super.preLogWithoutLock(database);
80     }
81 
82     /**
83      * Unconditionally throws ReplicaWriteException because this locker was
84      * created on a replica.
85      */
86     @Override
87     public void disallowReplicaWrite() {
88         throw new ReplicaWriteException(this, repImpl.getStateChangeEvent());
89     }
90 
91     /**
92      * Verifies that consistency requirements are met before allowing the
93      * cursor to be opened.
94      */
95     @Override
96     public void openCursorHook(DatabaseImpl dbImpl)
97         throws ReplicaConsistencyException {
98 
99         if (!dbImpl.isReplicated()) {
100             return;
101         }
102         ReadonlyTxn.checkConsistency(repImpl,
103                                      repImpl.getDefaultConsistencyPolicy());
104     }
