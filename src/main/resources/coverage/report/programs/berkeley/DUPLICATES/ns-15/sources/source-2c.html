20 /**
21  * Provides a synchronous mechanism for informing an application about a change
22  * in the state of the replication node. StateChangeException is an abstract
23  * class, with subtypes for each type of Transition.
24  * <p>
25  * A single state change can result in multiple state change exceptions (one
26  * per thread operating against the environment). Each exception is associated
27  * with the event that provoked the exception. The application can use this
28  * association to ensure that each such event is processed just once.
29  */
30 public abstract class StateChangeException extends OperationFailureException {
31     private static final long serialVersionUID = 1;
32 
33     /* Null if the event is not available. */
34     private final StateChangeEvent stateChangeEvent;
35 
36     /**
37      * For internal use only.
38      * @hidden
39      */
40     protected StateChangeException(Locker locker,
41                                    StateChangeEvent stateChangeEvent) {
42         super(locker, (locker != null),
43               makeMessage(locker, stateChangeEvent), null);
44         this.stateChangeEvent = stateChangeEvent;
45     }
46 
47     /**
48      * Used when no state change event is available
49      */
50     protected StateChangeException(String message, Exception reason) {
51         super(null, false, message, reason);
52         this.stateChangeEvent = null;
53     }
54 
55     /**
56      * Returns the event that resulted in this exception.
57      *
58      * @return the state change event
59      */
60     public StateChangeEvent getEvent() {
61         return stateChangeEvent;
62     }
63 
64     private static String makeMessage(Locker locker, StateChangeEvent event) {
65         long lockerId = (locker == null) ? 0 : locker.getId();
66         return (event != null) ?
67               ("Problem closing transaction " + lockerId +
68                ". The current state is:" + event.getState() + "." +
69                 " The node transitioned to this state at:" +
70                  new Date(event.getEventTime())) :
71                "Node state inconsistent with operation";
72     }
73 
74     /**
75      * For internal use only.
76      * @hidden
77      * Only for use by wrapSelf methods.
78      */
79     protected StateChangeException(String message,
80                                    StateChangeException cause) {
81         super(message, cause);
82         stateChangeEvent =
83             (cause != null) ? cause.stateChangeEvent : null;
84     }
85 }
