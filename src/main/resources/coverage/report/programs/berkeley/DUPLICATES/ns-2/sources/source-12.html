610     @Override
611     public void visitFrame(final int type, final int nLocal,
612             final Object[] local, final int nStack, final Object[] stack) {
613         if (!ClassReader.FRAMES || compute == FRAMES) {
614             return;
615         }
616 
617         if (type == Opcodes.F_NEW) {
618             if (previousFrame == null) {
619                 visitImplicitFirstFrame();
620             }
621             currentLocals = nLocal;
622             int frameIndex = startFrame(code.length, nLocal, nStack);
623             for (int i = 0; i < nLocal; ++i) {
624                 if (local[i] instanceof String) {
625                     frame[frameIndex++] = Frame.OBJECT
626                             | cw.addType((String) local[i]);
627                 } else if (local[i] instanceof Integer) {
628                     frame[frameIndex++] = ((Integer) local[i]).intValue();
629                 } else {
630                     frame[frameIndex++] = Frame.UNINITIALIZED
631                             | cw.addUninitializedType("",
632                                     ((Label) local[i]).position);
633                 }
634             }
635             for (int i = 0; i < nStack; ++i) {
636                 if (stack[i] instanceof String) {
637                     frame[frameIndex++] = Frame.OBJECT
638                             | cw.addType((String) stack[i]);
639                 } else if (stack[i] instanceof Integer) {
640                     frame[frameIndex++] = ((Integer) stack[i]).intValue();
641                 } else {
642                     frame[frameIndex++] = Frame.UNINITIALIZED
643                             | cw.addUninitializedType("",
644                                     ((Label) stack[i]).position);
645                 }
646             }
647             endFrame();
648         } else {
649             int delta;
650             if (stackMap == null) {
651                 stackMap = new ByteVector();
652                 delta = code.length;
653             } else {
654                 delta = code.length - previousFrameOffset - 1;
655                 if (delta < 0) {
656                     if (type == Opcodes.F_SAME) {
657                         return;
658                     } else {
659                         throw new IllegalStateException();
660                     }
661                 }
662             }
663 
664             switch (type) {
665             case Opcodes.F_FULL:
666                 currentLocals = nLocal;
667                 stackMap.putByte(FULL_FRAME).putShort(delta).putShort(nLocal);
668                 for (int i = 0; i < nLocal; ++i) {
669                     writeFrameType(local[i]);
670                 }
671                 stackMap.putShort(nStack);
672                 for (int i = 0; i < nStack; ++i) {
673                     writeFrameType(stack[i]);
674                 }
675                 break;
676             case Opcodes.F_APPEND:
677                 currentLocals += nLocal;
678                 stackMap.putByte(SAME_FRAME_EXTENDED + nLocal).putShort(delta);
679                 for (int i = 0; i < nLocal; ++i) {
680                     writeFrameType(local[i]);
681                 }
682                 break;
683             case Opcodes.F_CHOP:
684                 currentLocals -= nLocal;
685                 stackMap.putByte(SAME_FRAME_EXTENDED - nLocal).putShort(delta);
686                 break;
687             case Opcodes.F_SAME:
688                 if (delta < 64) {
689                     stackMap.putByte(delta);
690                 } else {
691                     stackMap.putByte(SAME_FRAME_EXTENDED).putShort(delta);
692                 }
693                 break;
694             case Opcodes.F_SAME1:
695                 if (delta < 64) {
696                     stackMap.putByte(SAME_LOCALS_1_STACK_ITEM_FRAME + delta);
697                 } else {
698                     stackMap.putByte(SAME_LOCALS_1_STACK_ITEM_FRAME_EXTENDED)
699                             .putShort(delta);
700                 }
701                 writeFrameType(stack[0]);
702                 break;
703             }
704 
705             previousFrameOffset = code.length;
706             ++frameCount;
707         }
708 
709         maxStack = Math.max(maxStack, nStack);
710         maxLocals = Math.max(maxLocals, currentLocals);
711     }

978     public void visitJumpInsn(final int opcode, final Label label) {
979         lastCodeOffset = code.length;
980         Label nextInsn = null;
981         // Label currentBlock = this.currentBlock;
982         if (currentBlock != null) {
983             if (compute == FRAMES) {
984                 currentBlock.frame.execute(opcode, 0, null, null);
985                 // 'label' is the target of a jump instruction
986                 label.getFirst().status |= Label.TARGET;
987                 // adds 'label' as a successor of this basic block
988                 addSuccessor(Edge.NORMAL, label);
989                 if (opcode != Opcodes.GOTO) {
990                     // creates a Label for the next basic block
991                     nextInsn = new Label();
992                 }
993             } else {
994                 if (opcode == Opcodes.JSR) {
995                     if ((label.status & Label.SUBROUTINE) == 0) {
996                         label.status |= Label.SUBROUTINE;
997                         ++subroutines;
998                     }
999                     currentBlock.status |= Label.JSR;
1000                     addSuccessor(stackSize + 1, label);
1001                     // creates a Label for the next basic block
1002                     nextInsn = new Label();
1003                     /*
1004                      * note that, by construction in this method, a JSR block
1005                      * has at least two successors in the control flow graph:
1006                      * the first one leads the next instruction after the JSR,
1007                      * while the second one leads to the JSR target.
1008                      */
1009                 } else {
1010                     // updates current stack size (max stack size unchanged
1011                     // because stack size variation always negative in this
1012                     // case)
1013                     stackSize += Frame.SIZE[opcode];
1014                     addSuccessor(stackSize, label);
1015                 }
1016             }
1017         }
1018         // adds the instruction to the bytecode of the method
1019         if ((label.status & Label.RESOLVED) != 0
1020                 && label.position - code.length < Short.MIN_VALUE) {
1021             /*
1022              * case of a backward jump with an offset < -32768. In this case we
1023              * automatically replace GOTO with GOTO_W, JSR with JSR_W and IFxxx
1024              * <l> with IFNOTxxx <l'> GOTO_W <l>, where IFNOTxxx is the
1025              * "opposite" opcode of IFxxx (i.e., IFNE for IFEQ) and where <l'>
1026              * designates the instruction just after the GOTO_W.
1027              */
1028             if (opcode == Opcodes.GOTO) {
1029                 code.putByte(200); // GOTO_W
1030             } else if (opcode == Opcodes.JSR) {
1031                 code.putByte(201); // JSR_W
1032             } else {
1033                 // if the IF instruction is transformed into IFNOT GOTO_W the
1034                 // next instruction becomes the target of the IFNOT instruction
1035                 if (nextInsn != null) {
1036                     nextInsn.status |= Label.TARGET;
1037                 }
1038                 code.putByte(opcode <= 166 ? ((opcode + 1) ^ 1) - 1
1039                         : opcode ^ 1);
1040                 code.putShort(8); // jump offset
1041                 code.putByte(200); // GOTO_W
1042             }
1043             label.put(this, code, code.length - 1, true);
1044         } else {
1045             /*
1046              * case of a backward jump with an offset >= -32768, or of a forward
1047              * jump with, of course, an unknown offset. In these cases we store
1048              * the offset in 2 bytes (which will be increased in
1049              * resizeInstructions, if needed).
1050              */
1051             code.putByte(opcode);
1052             label.put(this, code, code.length - 1, false);
1053         }
1054         if (currentBlock != null) {
1055             if (nextInsn != null) {
1056                 // if the jump instruction is not a GOTO, the next instruction
1057                 // is also a successor of this instruction. Calling visitLabel
1058                 // adds the label of this next instruction as a successor of the
1059                 // current block, and starts a new basic block
1060                 visitLabel(nextInsn);
1061             }
1062             if (opcode == Opcodes.GOTO) {
1063                 noSuccessor();
1064             }
1065         }
1066     }

1403     public void visitMaxs(final int maxStack, final int maxLocals) {
1404         if (resize) {
1405             // replaces the temporary jump opcodes introduced by Label.resolve.
1406             if (ClassReader.RESIZE) {
1407                 resizeInstructions();
1408             } else {
1409                 throw new RuntimeException("Method code too large!");
1410             }
1411         }
1412         if (ClassReader.FRAMES && compute == FRAMES) {
1413             // completes the control flow graph with exception handler blocks
1414             Handler handler = firstHandler;
1415             while (handler != null) {
1416                 Label l = handler.start.getFirst();
1417                 Label h = handler.handler.getFirst();
1418                 Label e = handler.end.getFirst();
1419                 // computes the kind of the edges to 'h'
1420                 String t = handler.desc == null ? "java/lang/Throwable"
1421                         : handler.desc;
1422                 int kind = Frame.OBJECT | cw.addType(t);
1423                 // h is an exception handler
1424                 h.status |= Label.TARGET;
1425                 // adds 'h' as a successor of labels between 'start' and 'end'
1426                 while (l != e) {
1427                     // creates an edge to 'h'
1428                     Edge b = new Edge();
1429                     b.info = kind;
1430                     b.successor = h;
1431                     // adds it to the successors of 'l'
1432                     b.next = l.successors;
1433                     l.successors = b;
1434                     // goes to the next label
1435                     l = l.successor;
1436                 }
1437                 handler = handler.next;
1438             }
1439 
1440             // creates and visits the first (implicit) frame
1441             Frame f = labels.frame;
1442             Type[] args = Type.getArgumentTypes(descriptor);
1443             f.initInputFrame(cw, access, args, this.maxLocals);
1444             visitFrame(f);
1445 
1446             /*
1447              * fix point algorithm: mark the first basic block as 'changed'
1448              * (i.e. put it in the 'changed' list) and, while there are changed
1449              * basic blocks, choose one, mark it as unchanged, and update its
1450              * successors (which can be changed in the process).
1451              */
1452             int max = 0;
1453             Label changed = labels;
1454             while (changed != null) {
1455                 // removes a basic block from the list of changed basic blocks
1456                 Label l = changed;
1457                 changed = changed.next;
1458                 l.next = null;
1459                 f = l.frame;
1460                 // a reachable jump target must be stored in the stack map
1461                 if ((l.status & Label.TARGET) != 0) {
1462                     l.status |= Label.STORE;
1463                 }
1464                 // all visited labels are reachable, by definition
1465                 l.status |= Label.REACHABLE;
1466                 // updates the (absolute) maximum stack size
1467                 int blockMax = f.inputStack.length + l.outputStackMax;
1468                 if (blockMax > max) {
1469                     max = blockMax;
1470                 }
1471                 // updates the successors of the current basic block
1472                 Edge e = l.successors;
1473                 while (e != null) {
1474                     Label n = e.successor.getFirst();
1475                     boolean change = f.merge(cw, n.frame, e.info);
1476                     if (change && n.next == null) {
1477                         // if n has changed and is not already in the 'changed'
1478                         // list, adds it to this list
1479                         n.next = changed;
1480                         changed = n;
1481                     }
1482                     e = e.next;
1483                 }
1484             }
1485 
1486             // visits all the frames that must be stored in the stack map
1487             Label l = labels;
1488             while (l != null) {
1489                 f = l.frame;
1490                 if ((l.status & Label.STORE) != 0) {
1491                     visitFrame(f);
1492                 }
1493                 if ((l.status & Label.REACHABLE) == 0) {
1494                     // finds start and end of dead basic block
1495                     Label k = l.successor;
1496                     int start = l.position;
1497                     int end = (k == null ? code.length : k.position) - 1;
1498                     // if non empty basic block
1499                     if (end >= start) {

1501                         // replaces instructions with NOP ... NOP ATHROW
1502                         for (int i = start; i < end; ++i) {
1503                             code.data[i] = Opcodes.NOP;

1505                         code.data[end] = (byte) Opcodes.ATHROW;
1506                         // emits a frame for this unreachable block
1507                         int frameIndex = startFrame(start, 0, 1);
1508                         frame[frameIndex] = Frame.OBJECT
1509                                 | cw.addType("java/lang/Throwable");
1510                         endFrame();
1511                         // removes the start-end range from the exception
1512                         // handlers
1513                         firstHandler = Handler.remove(firstHandler, l, k);
1514                     }
1515                 }
1516                 l = l.successor;
1517             }
1518 
1519             handler = firstHandler;
1520             handlerCount = 0;
1521             while (handler != null) {
1522                 handlerCount += 1;
1523                 handler = handler.next;
1524             }
1525 
1526             this.maxStack = max;
1527         } else if (compute == MAXS) {
1528             // completes the control flow graph with exception handler blocks
1529             Handler handler = firstHandler;
1530             while (handler != null) {
1531                 Label l = handler.start;
1532                 Label h = handler.handler;
1533                 Label e = handler.end;
1534                 // adds 'h' as a successor of labels between 'start' and 'end'
1535                 while (l != e) {
1536                     // creates an edge to 'h'
1537                     Edge b = new Edge();
1538                     b.info = Edge.EXCEPTION;
1539                     b.successor = h;
1540                     // adds it to the successors of 'l'
1541                     if ((l.status & Label.JSR) == 0) {
1542                         b.next = l.successors;
1543                         l.successors = b;
1544                     } else {
1545                         // if l is a JSR block, adds b after the first two edges
1546                         // to preserve the hypothesis about JSR block successors
1547                         // order (see {@link #visitJumpInsn})
1548                         b.next = l.successors.next.next;
1549                         l.successors.next.next = b;
1550                     }
1551                     // goes to the next label
1552                     l = l.successor;
1553                 }
1554                 handler = handler.next;
1555             }
1556 
1557             if (subroutines > 0) {
1558                 // completes the control flow graph with the RET successors
1559                 /*
1560                  * first step: finds the subroutines. This step determines, for
1561                  * each basic block, to which subroutine(s) it belongs.
1562                  */
1563                 // finds the basic blocks that belong to the "main" subroutine
1564                 int id = 0;
1565                 labels.visitSubroutine(null, 1, subroutines);
1566                 // finds the basic blocks that belong to the real subroutines
1567                 Label l = labels;
1568                 while (l != null) {
1569                     if ((l.status & Label.JSR) != 0) {
1570                         // the subroutine is defined by l's TARGET, not by l
1571                         Label subroutine = l.successors.next.successor;
1572                         // if this subroutine has not been visited yet...
1573                         if ((subroutine.status & Label.VISITED) == 0) {
1574                             // ...assigns it a new id and finds its basic blocks
1575                             id += 1;
1576                             subroutine.visitSubroutine(null, (id / 32L) << 32
1577                                     | (1L << (id % 32)), subroutines);
1578                         }
1579                     }
1580                     l = l.successor;
1581                 }
1582                 // second step: finds the successors of RET blocks
1583                 l = labels;
1584                 while (l != null) {
1585                     if ((l.status & Label.JSR) != 0) {
1586                         Label L = labels;
1587                         while (L != null) {
1588                             L.status &= ~Label.VISITED2;
1589                             L = L.successor;
1590                         }
1591                         // the subroutine is defined by l's TARGET, not by l
1592                         Label subroutine = l.successors.next.successor;
1593                         subroutine.visitSubroutine(l, 0, subroutines);
1594                     }
1595                     l = l.successor;
1596                 }
1597             }
1598 
1599             /*
1600              * control flow analysis algorithm: while the block stack is not
1601              * empty, pop a block from this stack, update the max stack size,
1602              * compute the true (non relative) begin stack size of the
1603              * successors of this block, and push these successors onto the
1604              * stack (unless they have already been pushed onto the stack).
1605              * Note: by hypothesis, the {@link Label#inputStackTop} of the
1606              * blocks in the block stack are the true (non relative) beginning
1607              * stack sizes of these blocks.
1608              */
1609             int max = 0;
1610             Label stack = labels;
1611             while (stack != null) {
1612                 // pops a block from the stack
1613                 Label l = stack;
1614                 stack = stack.next;
1615                 // computes the true (non relative) max stack size of this block
1616                 int start = l.inputStackTop;
1617                 int blockMax = start + l.outputStackMax;
1618                 // updates the global max stack size
1619                 if (blockMax > max) {
1620                     max = blockMax;
1621                 }
1622                 // analyzes the successors of the block
1623                 Edge b = l.successors;
1624                 if ((l.status & Label.JSR) != 0) {
1625                     // ignores the first edge of JSR blocks (virtual successor)
1626                     b = b.next;
1627                 }
1628                 while (b != null) {
1629                     l = b.successor;
1630                     // if this successor has not already been pushed...
1631                     if ((l.status & Label.PUSHED) == 0) {
1632                         // computes its true beginning stack size...
1633                         l.inputStackTop = b.info == Edge.EXCEPTION ? 1 : start
1634                                 + b.info;
1635                         // ...and pushes it onto the stack
1636                         l.status |= Label.PUSHED;
1637                         l.next = stack;
1638                         stack = l;
1639                     }
1640                     b = b.next;
1641                 }
1642             }
1643             this.maxStack = Math.max(maxStack, max);
1644         } else {
1645             this.maxStack = maxStack;
1646             this.maxLocals = maxLocals;
1647         }
1648     }

1858         int cstackSize = frame[2];

1860             stackMap.putShort(frame[0]).putShort(clocalsSize);
1861             writeFrameTypes(3, 3 + clocalsSize);
1862             stackMap.putShort(cstackSize);
1863             writeFrameTypes(3 + clocalsSize, 3 + clocalsSize + cstackSize);
1864             return;
1865         }
1866         int localsSize = previousFrame[1];
1867         int type = FULL_FRAME;
1868         int k = 0;
1869         int delta;
1870         if (frameCount == 0) {
1871             delta = frame[0];
1872         } else {
1873             delta = frame[0] - previousFrame[0] - 1;
1874         }
1875         if (cstackSize == 0) {
1876             k = clocalsSize - localsSize;
1877             switch (k) {
1878             case -3:
1879             case -2:
1880             case -1:
1881                 type = CHOP_FRAME;
1882                 localsSize = clocalsSize;
1883                 break;
1884             case 0:
1885                 type = delta < 64 ? SAME_FRAME : SAME_FRAME_EXTENDED;
1886                 break;
1887             case 1:
1888             case 2:
1889             case 3:
1890                 type = APPEND_FRAME;
1891                 break;
1892             }
1893         } else if (clocalsSize == localsSize && cstackSize == 1) {
1894             type = delta < 63 ? SAME_LOCALS_1_STACK_ITEM_FRAME
1895                     : SAME_LOCALS_1_STACK_ITEM_FRAME_EXTENDED;
1896         }
1897         if (type != FULL_FRAME) {
1898             // verify if locals are the same
1899             int l = 3;
1900             for (int j = 0; j < localsSize; j++) {
1901                 if (frame[l] != previousFrame[l]) {
1902                     type = FULL_FRAME;
1903                     break;
1904                 }
1905                 l++;
1906             }
1907         }
1908         switch (type) {
1909         case SAME_FRAME:
1910             stackMap.putByte(delta);
1911             break;
1912         case SAME_LOCALS_1_STACK_ITEM_FRAME:
1913             stackMap.putByte(SAME_LOCALS_1_STACK_ITEM_FRAME + delta);
1914             writeFrameTypes(3 + clocalsSize, 4 + clocalsSize);
1915             break;

2355     /**
2356      * Resizes and replaces the temporary instructions inserted by
2357      * {@link Label#resolve} for wide forward jumps, while keeping jump offsets
2358      * and instruction addresses consistent. This may require to resize other
2359      * existing instructions, or even to introduce new instructions: for
2360      * example, increasing the size of an instruction by 2 at the middle of a
2361      * method can increases the offset of an IFEQ instruction from 32766 to
2362      * 32768, in which case IFEQ 32766 must be replaced with IFNEQ 8 GOTO_W
2363      * 32765. This, in turn, may require to increase the size of another jump
2364      * instruction, and so on... All these operations are handled automatically
2365      * by this method.
2366      * <p>
2367      * <i>This method must be called after all the method that is being built
2368      * has been visited</i>. In particular, the {@link Label Label} objects used
2369      * to construct the method are no longer valid after this method has been
2370      * called.
2371      */
2372     private void resizeInstructions() {
2373         byte[] b = code.data; // bytecode of the method
2374         int u, v, label; // indexes in b
2375         int i, j; // loop indexes
2376         /*
2377          * 1st step: As explained above, resizing an instruction may require to
2378          * resize another one, which may require to resize yet another one, and
2379          * so on. The first step of the algorithm consists in finding all the
2380          * instructions that need to be resized, without modifying the code.
2381          * This is done by the following "fix point" algorithm:
2382          * 
2383          * Parse the code to find the jump instructions whose offset will need
2384          * more than 2 bytes to be stored (the future offset is computed from
2385          * the current offset and from the number of bytes that will be inserted
2386          * or removed between the source and target instructions). For each such
2387          * instruction, adds an entry in (a copy of) the indexes and sizes
2388          * arrays (if this has not already been done in a previous iteration!).
2389          * 
2390          * If at least one entry has been added during the previous step, go
2391          * back to the beginning, otherwise stop.
2392          * 
2393          * In fact the real algorithm is complicated by the fact that the size
2394          * of TABLESWITCH and LOOKUPSWITCH instructions depends on their
2395          * position in the bytecode (because of padding). In order to ensure the
2396          * convergence of the algorithm, the number of bytes to be added or
2397          * removed from these instructions is over estimated during the previous
2398          * loop, and computed exactly only after the loop is finished (this
2399          * requires another pass to parse the bytecode of the method).
2400          */
2401         int[] allIndexes = new int[0]; // copy of indexes
2402         int[] allSizes = new int[0]; // copy of sizes
2403         boolean[] resize; // instructions to be resized
2404         int newOffset; // future offset of a jump instruction
2405 
2406         resize = new boolean[code.length];
2407 
2408         // 3 = loop again, 2 = loop ended, 1 = last pass, 0 = done
2409         int state = 3;
2410         do {
2411             if (state == 3) {
2412                 state = 2;
2413             }
2414             u = 0;
2415             while (u < b.length) {
2416                 int opcode = b[u] & 0xFF; // opcode of current instruction
2417                 int insert = 0; // bytes to be added after this instruction
2418 
2419                 switch (ClassWriter.TYPE[opcode]) {
2420                 case ClassWriter.NOARG_INSN:
2421                 case ClassWriter.IMPLVAR_INSN:
2422                     u += 1;
2423                     break;
2424                 case ClassWriter.LABEL_INSN:
2425                     if (opcode > 201) {

2427                         // 219 to IFEQ ... JSR (inclusive), IFNULL and
2428                         // IFNONNULL
2429                         opcode = opcode < 218 ? opcode - 49 : opcode - 20;
2430                         label = u + readUnsignedShort(b, u + 1);
2431                     } else {
2432                         label = u + readShort(b, u + 1);
2433                     }
2434                     newOffset = getNewOffset(allIndexes, allSizes, u, label);
2435                     if (newOffset < Short.MIN_VALUE
2436                             || newOffset > Short.MAX_VALUE) {
2437                         if (!resize[u]) {
2438                             if (opcode == Opcodes.GOTO || opcode == Opcodes.JSR) {

2440                                 // replace this GOTO or JSR instruction with
2441                                 // a GOTO_W or a JSR_W
2442                                 insert = 2;
2443                             } else {
2444                                 // five additional bytes will be required to
2445                                 // replace this IFxxx <l> instruction with
2446                                 // IFNOTxxx <l'> GOTO_W <l>, where IFNOTxxx
2447                                 // is the "opposite" opcode of IFxxx (i.e.,
2448                                 // IFNE for IFEQ) and where <l'> designates
2449                                 // the instruction just after the GOTO_W.
2450                                 insert = 5;
2451                             }
2452                             resize[u] = true;

2456                     break;
2457                 case ClassWriter.LABELW_INSN:

2459                     break;
2460                 case ClassWriter.TABL_INSN:
2461                     if (state == 1) {
2462                         // true number of bytes to be added (or removed)
2463                         // from this instruction = (future number of padding
2464                         // bytes - current number of padding byte) -
