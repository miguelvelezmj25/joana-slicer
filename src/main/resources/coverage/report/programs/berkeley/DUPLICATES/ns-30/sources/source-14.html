21 /**
22  * Redirects logging messages to the the owning environment's file handler, so
23  * that messages can be prefixed with an environment name and sent to the
24  * correct logging file.
25  *
26  * This class must not extend FileHandler itself, since FileHandlers open their
27  * target log files at construction time, and this FileHandler is meant to be
28  * stateless.
29  */
30 public class FileRedirectHandler extends Handler {
31 
32     public FileRedirectHandler() {
33         /* No need to call super, this handler is not truly publishing. */
34     }
35 
36     @Override
37     public void publish(LogRecord record) {
38         EnvironmentImpl envImpl =
39             LoggerUtils.envMap.get(Thread.currentThread());
40 
41         /*
42          * Prefer to lose logging output, rather than risk a
43          * NullPointerException if the caller forgets to set and release the
44          * environmentImpl.
45          */
46         if (envImpl == null) {
47             return;
48         }
49 
50         /*
51          * The FileHandler is not always created for an environment, because
52          * creating a FileHandler automatically creates a logging file, and
53          * we avoid doing that for read only environments. Because of that,
54          * there may legitimately be no environment file handler.
55          */
56         if (envImpl.getFileHandler() == null) {
57             return;
58         }
59 
60         envImpl.getFileHandler().publish(record);
61     }
62 
63     @Override
64     public void close()
65         throws SecurityException {
66 
67         /*
68          * Nothing to do. The redirect target file handler is closed by
69          * the environment.
70          */
71     }
72 
73     @Override
74     public void flush() {
75 
76         /*
77          * Nothing to do. If we want to flush this logger explicitly, flush
78          * the underlying envImpl's handler.
79          */
80     }
81 }
