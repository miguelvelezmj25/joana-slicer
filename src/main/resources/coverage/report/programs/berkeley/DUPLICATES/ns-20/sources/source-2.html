18 /**
19  * The event generated when a node joins the group. A new instance of this 
20  * event is generated each time a node joins the group.
21  *
22  * The event is generated on a "best effort" basis. It may not be generated,
23  * for example, if the joining node was unable to communicate with the monitor

25  * such missing events.
26  */

29     /**
30      * The time when this node joins the group. 
31      */

33

36         this.joinTime = joinTime;
37     }
38 
39     /**

41      */
42     public Date getJoinTime() {

44     }
45
