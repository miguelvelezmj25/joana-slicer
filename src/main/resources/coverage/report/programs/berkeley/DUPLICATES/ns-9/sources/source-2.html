92     final void countNew(long lsn,
93                         Object databaseKey,
94                         LogEntryType type,
95                         int size) {
96         assert type != null;
97         /* Count in per-file and per-file-per-db summaries. */
98         long fileNum = DbLsn.getFileNumber(lsn);
99         FileSummary fileSummary = getFileSummary(fileNum);
100         fileSummary.totalCount += 1;
101         fileSummary.totalSize += size;
102         boolean isLN = isLNType(type);
103         if (isLN && fileSummary.maxLNSize < size) {
104             fileSummary.maxLNSize = size;
105         }
106         if (trackObsoleteInfo(type)) {
107             assert databaseKey != null :
108                 "No DB for lsn=" + DbLsn.getNoFormatString(lsn) +
109                 " type: " + type;
110             DbFileSummary dbFileSummary =
111                 getDbFileSummary(databaseKey, fileNum);
112             if (isLN) {
113                 fileSummary.totalLNCount += 1;
114                 fileSummary.totalLNSize += size;
115                 if (dbFileSummary != null) {
116                     dbFileSummary.totalLNCount += 1;
117                     dbFileSummary.totalLNSize += size;
118                 }
119             } else {
120                 fileSummary.totalINCount += 1;
121                 fileSummary.totalINSize += size;
122                 if (dbFileSummary != null) {
123                     dbFileSummary.totalINCount += 1;
124                     dbFileSummary.totalINSize += size;
125                 }
126             }

149         long fileNum = DbLsn.getFileNumber(lsn);
150 
151         if (countPerFile) {
152             TrackedFileSummary fileSummary = getFileSummary(fileNum);
153             if (isLN) {
154                 fileSummary.obsoleteLNCount += 1;
155                 /* The size is optional when tracking obsolete LNs. */
156                 if (size > 0) {

158                     fileSummary.obsoleteLNSizeCounted += 1;
159                 }

164             }

199     /**
200      * Counts all active LSNs in a database as obsolete in the per-file
201      * utilization summaries.  This method is called during database
202      * remove/truncate or when replaying those operations during recovery.
203      *
204      * <p>For the global tracker, must be called under the log write latch.</p>
205      *
206      * @param dbFileSummaries the map of Long file number to DbFileSummary for
207      * a database that is being deleted.
208      *
209      * @param mapLnLsn is the LSN of the MapLN when recovery is replaying the
210      * truncate/remove, or NULL_LSN when called outside of recovery; obsolete
211      * totals should only be counted when this LSN is prior to the LSN of the
212      * FileSummaryLN for the file being counted.
213      */
214     public void countObsoleteDb(
215         DbFileSummaryMap dbFileSummaries,
216         long mapLnLsn) {
217 
218         for (Map.Entry<Long, DbFileSummary> entry :
219              dbFileSummaries.entrySet()) {
220 
221             Long fileNum = entry.getKey();
222 
223             if (isFileUncounted(fileNum, mapLnLsn)) {
224                 DbFileSummary dbFileSummary = entry.getValue();
225                 TrackedFileSummary fileSummary =
226                     getFileSummary(fileNum.longValue());
227 
228                 /*
229                  * Count as obsolete the currently active amounts in the
230                  * database, which are the total amounts minus the previously
231                  * counted obsolete amounts.
232                  */
233                 int lnObsoleteCount = dbFileSummary.totalLNCount -
234                                       dbFileSummary.obsoleteLNCount;
235                 int lnObsoleteSize  = dbFileSummary.totalLNSize -
236                                       dbFileSummary.obsoleteLNSize;
237                 int inObsoleteCount = dbFileSummary.totalINCount -
238                                       dbFileSummary.obsoleteINCount;
239                 fileSummary.obsoleteLNCount += lnObsoleteCount;
240                 fileSummary.obsoleteLNSize += lnObsoleteSize;
241                 fileSummary.obsoleteINCount += inObsoleteCount;
242 
243                 /*
244                  * When a DB becomes obsolete, the size of all obsolete LNs can
245                  * now be counted accurately because all LN bytes in the DB are
246                  * now obsolete.  The lnObsoleteSize value calculated above
247                  * includes LNs that become obsolete now, plus those that
248                  * became obsolete previously but whose size was not counted.
249                  * The obsoleteLNSizeCounted field is updated accordingly
250                  * below.  In other words, DB obsolescence is self-healing with
251                  * respect to obsolete LN sizes.  [#19144]
252                  */
253                 int lnObsoleteSizeCounted = lnObsoleteCount +
254                                      (dbFileSummary.obsoleteLNCount -
255                                       dbFileSummary.obsoleteLNSizeCounted);
256                 fileSummary.obsoleteLNSizeCounted += lnObsoleteSizeCounted;
257 
258                 /*
259                  * Do not update the DbFileSummary.  It will be flushed when
260                  * the MapLN is deleted.  If later replayed during recovery, we
261                  * will call this method to update the per-file utilization.
262                  */
263             }
264         }
265     }

292     /**
293      * Returns a tracked file for the given file number, adding an empty one
294      * if the file is not already being tracked.
295      *
296      * <p>For the global tracker, must be called under the log write latch.</p>
297      */
298     TrackedFileSummary getFileSummary(long fileNum) {
299 
300         if (activeFile < fileNum) {
301             activeFile = fileNum;
302         }
303         Long fileNumLong = Long.valueOf(fileNum);
304         TrackedFileSummary file = fileSummaries.get(fileNumLong);
305         if (file == null) {
306             /* Assign fileSummaries field after modifying the new map. */
307             file = new TrackedFileSummary(this, fileNum, cleaner.trackDetail);
308             Map<Long, TrackedFileSummary> newFiles = 
309                 new HashMap<Long,TrackedFileSummary>(fileSummaries);
310             newFiles.put(fileNumLong, file);
311             fileSummaries = newFiles;
312         }
313         return file;
314     }
