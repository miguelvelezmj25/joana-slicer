228     /**
229      * Send files in response to request messages. The request sequence looks
230      * like the following:
231      *
232      *  [FileReq | FileInfoReq]+ Done
233      *
234      * The response sequence to a FileReq looks like:
235      *
236      *  FileStart <file byte stream> FileEnd
237      *
238      *  and that for a FileInfoReq, is simply a FileInfoResp
239      */
240     private void sendRequestedFiles(Protocol protocol)
241         throws IOException, ProtocolException, DatabaseException {
242 
243         String prevFileName = null;
244 
245         try {
246             while (true) {
247                 FileReq fileReq = protocol.read(namedChannel.getChannel(),
248                                                 FileReq.class);
249                 final String fileName = fileReq.getFileName();
250 
251                 /*
252                  * Calculate the full path for a specified log file name,
253                  * especially when this Feeder is configured to run with sub
254                  * directories.
255                  */
256                 FileManager fMgr = feederManager.getEnvImpl().getFileManager();
257                 File file = new File(fMgr.getFullFileName(fileName));
258 
259                 if (!file.exists()) {
260                     throw EnvironmentFailureException.unexpectedState
261                         ("Log file not found: " + fileName);
262                 }
263                 /* Freeze the length and last modified date. */
264                 final long length = file.length();
265                 final long lastModified = file.lastModified();
266                 byte digest[] = null;
267                 FileInfoResp resp = null;
268                 FileInfoResp cachedResp =
269                     feederManager.statResponses.get(fileName);
270                 byte cachedDigest[] =
271                     ((cachedResp != null) &&
272                      (cachedResp.getFileLength() == length) &&
273                      (cachedResp.getLastModifiedTime() == lastModified)) ?
274                     cachedResp.getDigestSHA1() : null;
275 
276                 if (fileReq instanceof FileInfoReq) {
277                     if  (cachedDigest != null) {
278                         digest = cachedDigest;
279                     } else if (((FileInfoReq) fileReq).getNeedSHA1()) {
280                         digest = getSHA1Digest(file, length).digest();
281                     } else {
282                         // Digest not requested
283                         digest = new byte[0];
284                     }
285                     resp = protocol.new FileInfoResp
286                         (fileName, length, lastModified, digest);
287                 } else {
288                     /* Allow deletion of previous file. */
289                     if (prevFileName != null &&
290                         !fileName.equals(prevFileName)) {
291                         dbBackup.removeFileProtection(prevFileName);
292                     }
293                     prevFileName = fileName;
294 
295                     protocol.write(protocol.new FileStart
296                                    (fileName, length, lastModified),
297                                    namedChannel);
298                     digest = sendFileContents(file, length);
299                     if ((cachedDigest != null) &&
300                          !Arrays.equals(cachedDigest, digest)) {
301                         throw EnvironmentFailureException.unexpectedState
302                             ("Inconsistent cached and computed digests");
303                     }
304                     resp = protocol.new FileEnd
305                         (fileName, length, lastModified, digest);
306                 }
307                 /* Cache for subsequent requests, if it was computed. */
308                 if (digest.length > 0) {
309                     feederManager.statResponses.put(fileName, resp);
310                 }
311                 protocol.write(resp, namedChannel);
312             }
313         } catch (ProtocolException pe) {
314             if (pe.getUnexpectedMessage() instanceof Protocol.Done) {
315                 return;
316             }
317             throw pe;
318         }
319     }

321     /**
322      * Returns the SHA1 has associated with the file.
323      *
324      * @param file
325      * @param length
326      * @return
327      * @throws IOException
328      * @throws DatabaseException
329      */
330     static MessageDigest getSHA1Digest(File file, long length)
331         throws IOException, DatabaseException {
332 
333         MessageDigest messageDigest = null;
334 
335         try {
336             messageDigest = MessageDigest.getInstance("SHA1");
337         } catch (NoSuchAlgorithmException e) {
338             throw EnvironmentFailureException.unexpectedException(e);
339         }
340         final FileInputStream fileStream = new FileInputStream(file);
341         try {
342             ByteBuffer buffer = ByteBuffer.allocate(TRANSFER_BYTES);
343             for (long bytes = length; bytes > 0; ) {
344                 int readSize = (int)Math.min(TRANSFER_BYTES, bytes);
345                 int readBytes =
346                     fileStream.read(buffer.array(), 0, readSize);
347                 if (readBytes == -1) {
348                     throw new IOException("Premature EOF. Was expecting: " +
349                                           readSize);
350                 }
351                 messageDigest.update(buffer.array(), 0, readBytes);
352                 bytes -= readBytes;
353             }
354         } finally {
355             fileStream.close();
356         }
357         return messageDigest;
358     }

360     /**
361      * Sends over the contents of the file and computes the SHA-1 hash. Note
362      * that the method does not rely on EOF detection, but rather on the
363      * promised file size, since the final log file might be growing while the
364      * transfer is in progress. The client uses the length sent in the FileResp
365      * message to maintain its position in the network stream. It expects to
366      * see a FileInfoResp once it has read the agreed upon number of bytes.
367      *
368      * Since JE log files are append only, there is no danger that we will send
369      * over any uninitialized file blocks.
370      *
371      * @param file the log file to be sent.
372      * @param length the number of bytes to send
373      * @return the digest associated with the file that was sent
374      *
375      * @throws IOException
376      */
377     private byte[] sendFileContents(File file, long length)
378         throws IOException {
379 
380         final LogVerifier verifier =
381             new LogVerifier(feederManager.getEnvImpl(), file.getName(), -1L);
382         final FileInputStream fileStream = new FileInputStream(file);
383 
384         try {
385             final FileChannel fileChannel = fileStream.getChannel();
386             messageDigest.reset();
387             final ByteBuffer buffer =
388                     ByteBuffer.allocateDirect(TRANSFER_BYTES);
389             final byte[] array =
390                 (buffer.hasArray()) ? buffer.array() : new byte[TRANSFER_BYTES];
391             int transmitBytes = 0;
392 
393             while (true) {
394                 buffer.clear();
395                 if (fileChannel.read(buffer) < 0) {
396                     verifier.verifyAtEof();
397                     break;
398                 }
399 
400                 buffer.flip();
401                 final int lim = buffer.limit();
402                 final int off;
403                 if (buffer.hasArray()) {
404                     off = buffer.arrayOffset();
405                 } else {
406                     off = 0;
407                     buffer.get(array, 0, lim);
408                     buffer.rewind();
409                 }
410                 verifier.verify(array, off, lim);
411                 messageDigest.update(array, off, lim);
412                 transmitBytes += namedChannel.getChannel().write(buffer);
413             }
414 
415             if (transmitBytes != length) {
416                 String msg = "File length:" + length + " does not match the " +
417                              "number of bytes that were transmitted:" +
418                              transmitBytes;
419 
420                 throw new IllegalStateException(msg);
421             }
422 
423             LoggerUtils.info
424                 (logger, feederManager.getEnvImpl(),
425                  "Sent file: " + file + " Length: " + length + " bytes");
426         } finally {
427             fileStream.close();
428         }
429         return messageDigest.digest();
430     }

433      * Processes the request for the list of files that constitute a valid
434      * backup. If a leased DbBackup instance is available, it uses it,
435      * otherwise it creates a new instance and uses it instead.
436      */
437     private void sendFileList(Protocol protocol)
438         throws IOException, ProtocolException, DatabaseException {
439 
440         /* Wait for the request message. */
441         protocol.read(namedChannel.getChannel(), Protocol.FileListReq.class);
442 
443         if (dbBackup == null) {
444             dbBackup = new DbBackup(feederManager.getEnvImpl());
445             dbBackup.setNetworkRestore();
446             dbBackup.startBackup();
447         } else {
448             feederManager.leaseRenewalCount++;
449         }
450 
451         /*
452          * Remove the subdirectory header of the log files, because the nodes
453          * that need to copy those log files may not configure the spreading
454          * log files into sub directories feature.
455          */
456         String[] files = dbBackup.getLogFilesInBackupSet();
457         for (int i = 0; i < files.length; i++) {
458             if (files[i].contains(File.separator)) {
459                 files[i] = files[i].substring
460                     (files[i].indexOf(File.separator) + 1, files[i].length());
461             }
462         }
463 
464         protocol.write(protocol.new FileListResp(files), namedChannel);
465     }

467     /**
468      * Verify that the protocols are compatible, switch to a different protocol
469      * version, if we need to.
470      */
471     private Protocol checkProtocol(Protocol protocol)
472         throws IOException, ProtocolException {
473 
474         ClientVersion clientVersion =
475             protocol.read(namedChannel.getChannel(),
476                           Protocol.ClientVersion.class);
477         clientId = clientVersion.getNodeId();
478         FeederManager.Lease lease = feederManager.leases.get(clientId);
479         if (lease != null) {
480             dbBackup = lease.terminate();
481         }
482         feederManager.feeders.put(clientId, this);
483         if (clientVersion.getVersion() != protocol.getVersion()) {
484             String message = "Client requested protocol version: " +
485                 clientVersion.getVersion() + " but the server version is " +
486                 protocol.getVersion();
487 
488             /*
489              * Simply log the difference on the server, it's up to the client
490              * to reject the protocol version, if it can't accommodate it.
491              */
492             LoggerUtils.warning(logger, feederManager.getEnvImpl(), message);
493         }
494         protocol.write(protocol.new ServerVersion(),  namedChannel);
495 
496         /* In future we may switch protocol versions to accommodate the client.
497          * For now, simply return the one and only version.
498          */
499         return protocol;
500     }

507
