48     public static byte[] concatByteArray(byte[] a, byte[] b) {
49         if ((a == null) || (b == null)) {
50             return (a == null) ? b : a;
51         }

53         int len = a.length + b.length;
54         byte[] dest = new byte[len];
55         /* Copy the content of a to dest. */
56         System.arraycopy(a, 0, dest, 0, a.length);
57         /* Copy the content from b to the remaining part of dest. */
58         System.arraycopy(b, 0, dest, a.length, b.length);

75     public static Block readBlock(int blockId, Cursor cursor, int numKeys)
76         throws DatabaseException {
77 
78         /* DatabaseEntry represents the key and data of each record. */
79         DatabaseEntry key = new DatabaseEntry();
80         DatabaseEntry data = new DatabaseEntry();
81         /* Adler32 to compute the rolling checksum of key/data pair. */
82         Adler32 adler32 = new Adler32();
83         int i = 0;
84         int a = 0, b = 0;
85         md.reset();
86         Block block = new Block(blockId);
87 
88         /* Please pay attention to the check order in while loop. */
89         while ((i < numKeys) &&
90                (cursor.getNext(key, data, LockMode.DEFAULT) ==
91                 OperationStatus.SUCCESS)) {
92             /* Indicates having a new block. */
93             if (i == 0) {
94                 block.setBeginKey(key.getData());
95                 block.setBeginData(data.getData());
96             }
97 
98             /* Calculate rollingChksum on "key|data" bytes. */
99             adler32.reset();
100             adler32.update(key.getData(), 0, key.getData().length);
101             adler32.update(data.getData(), 0, data.getData().length);
102             final int xi = (int) adler32.getValue();
103             a += xi;
104             b += a;
105             /* Update MessageDigest with "key|data" bytes. */
106             md.update(key.getData());
107             md.update(data.getData());
108             i++;
109         }
110 
111         /* Allocate a block and return. */
112         long cksum = (a & LDiffUtil.MASK_32BIT) | ((long) b << 32);
113         block.setRollingChksum(cksum);
114         block.setMd5Hash(md.digest());
115         block.setNumRecords(i);
116         return block;
117     }
