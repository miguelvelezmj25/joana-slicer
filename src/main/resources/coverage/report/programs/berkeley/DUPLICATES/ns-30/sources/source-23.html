31 /**
32  * Verifies the checksums in the contents of a log file in a JE {@code
33  * Environment}.
34  *
35  * <p>The caller supplies the contents of the log file by passing arrays of
36  * bytes in a series of calls to the {@link #verify} method, which verifies the
37  * checksums for log records, and by calling the {@link #verifyAtEof} when the
38  * entire contents are complete, to detect incomplete entries at the end of the
39  * file.  The primary intended use of this class is to verify the contents of
40  * log files that are being copied as part of a programmatic backup.  It is
41  * critical that invalid files are not added to a backup set, since then both
42  * the live environment and the backup will be invalid.
43  *
44  * @see com.sleepycat.je.util.LogVerificationInputStream
45  */

103      * <p>Creates a log verifier for use with an internal environment.  If
104      * {@code fileNum} is less than zero, it is derived from {@code fileName}.
105      *
106      * @param envImpl the {@code EnvironmentImpl} associated with the log
107      *
108      * @param fileName the file name of the log, for reporting in the {@code
109      * LogVerificationException}.  This should be a simple file name of the
110      * form {@code NNNNNNNN.jdb}, where NNNNNNNN is the file number in
111      * hexadecimal format.
112      *
113      * @param fileNum the file number
114      */
115     public LogVerifier(final EnvironmentImpl envImpl,
116                        final String fileName,
117                        final long fileNum) {
118         this.envImpl = envImpl;
119         this.fileName = fileName;
120         this.fileNum = (fileNum >= 0) ?
121             fileNum : envImpl.getFileManager().getNumFromName(fileName);
122         state = State.INIT;
123         entryStart = 0L;
124         prevEntryStart = 0L;
125         validator = new ChecksumValidator();
126 
127         /*
128          * The headerBuf is used to hold the fixed entry header, variable entry
129          * header portion, and file header entry.
130          */
131         headerBuf = ByteBuffer.allocate
132             (Math.max(LogEntryHeader.MAX_HEADER_SIZE, FileHeader.entrySize()));
133 
134         /* Initial log version for reading the file header. */
135         logVersion = LogEntryType.UNKNOWN_FILE_HEADER_VERSION;
136     }

138     /**
139      * Verifies the next portion of the log file.
140      *
141      * @param buf the buffer containing the log file bytes
142      *
143      * @param off the start offset of the log file bytes in the buffer
144      *
145      * @param len the number of log file bytes in the buffer
146      *
147      * @throws LogVerificationException if a checksum cannot be verified or a
148      * log entry is determined to be invalid by examining its contents
149      *
150      * @throws EnvironmentFailureException if an unexpected, internal or
151      * environment-wide failure occurs
152      */
153     public void verify(final byte[] buf, final int off, final int len)
154         throws LogVerificationException {
155 
156         final int endOffset = off + len;
157         int curOffset = off;
158         while (curOffset < endOffset) {
159             final int remaining = endOffset - curOffset;
160             switch (state) {
161             case INIT:
162                 processInit();
163                 break;
164             case FIXED_HEADER:
165                 curOffset = processFixedHeader(buf, curOffset, remaining);
166                 break;
167             case VARIABLE_HEADER:
168                 curOffset = processVariableHeader(buf, curOffset, remaining);
169                 break;
170             case FILE_HEADER_ITEM:
171                 curOffset = processFileHeaderItem(buf, curOffset, remaining);
172                 break;
173             case ITEM:
174                 curOffset = processItem(buf, curOffset, remaining);
175                 break;
176             case INVALID:
177                 throw newVerifyException
178                     ("May not read after LogVerificationException is thrown");
179             default:
180                 assert false;
181             }
182         }
183     }

185     /**
186      * Checks that the log file ends with a complete log entry, after having
187      * completed verifying the log file contents through calls to {@link
188      * #verify}.
189      *
190      * @throws LogVerificationException if the stream does not end with a
191      * complete log entry
192      *
193      * @throws EnvironmentFailureException if an unexpected, internal or
194      * environment-wide failure occurs
195      */
196     public void verifyAtEof()
197         throws LogVerificationException {
198 
199         /* State should be INIT at EOF. */
200         if (state == State.INIT) {
201             return;
202         }
203 
204         /* Ignore partial entry at end of last log file. */
205         if (fileNum == envImpl.getFileManager().getLastFileNum()) {
206             return;
207         }
208 
209         /* Report partial entry at end of any other file. */
210         throw newVerifyException("Entry is incomplete");
211     }

225     /**
226      * Processes the fixed initial portion of a log entry.  After all bytes for
227      * the fixed portion are read, moves the state to VARIABLE_HEADER if the
228      * header contains a variable portion, or to ITEM if it does not.
229      */
230     private int processFixedHeader(final byte[] buf,
231                                    final int curOffset,
232                                    final int remaining)
233         throws LogVerificationException {
234 
235         assert header == null;
236 
237         final int maxSize = LogEntryHeader.MIN_HEADER_SIZE;
238         final int processSize =
239             Math.min(remaining, maxSize - headerBuf.position());
240 
241         headerBuf.put(buf, curOffset, processSize);
242         assert headerBuf.position() <= maxSize;
243 
244         if (headerBuf.position() == maxSize) {
245             headerBuf.flip();
246             try {
247                 header = new LogEntryHeader(
248                     headerBuf, logVersion, DbLsn.makeLsn(fileNum, entryStart));
249             } catch (ChecksumException e) {
250                 throw newVerifyException(
251                     "Invalid header bytes=" +
252                         Arrays.toString(headerBuf.array()),
253                     e);
254             }
255 
256             if (header.getPrevOffset() != prevEntryStart) {
257                 throw newVerifyException(
258                     "Header prevOffset=0x" +
259                     Long.toHexString(header.getPrevOffset()) +
260                     " but prevEntryStart=0x" +
261                     Long.toHexString(prevEntryStart));
262             }
263 
264             /* If the header is invisible, turn off the invisible bit. */
265             if (header.isInvisible()) {
266                 LogEntryHeader.turnOffInvisible(headerBuf, 0);
267             }
268 
269             /* Do not validate the bytes of the checksum itself. */
270             validator.update(headerBuf.array(),
271                              LogEntryHeader.CHECKSUM_BYTES,
272                              maxSize - LogEntryHeader.CHECKSUM_BYTES);
273 
274             if (header.isVariableLength()) {
275                 headerBuf.clear();
276                 state = State.VARIABLE_HEADER;
277             } else if (header.getType() == FILE_HEADER_TYPE_NUM) {
278                 headerBuf.clear();
279                 state = State.FILE_HEADER_ITEM;
280             } else {
281                 state = State.ITEM;
282             }
283         }
284 
285         return curOffset + processSize;
286     }

288     /**
289      * Processes the variable portion of a log entry.  After all bytes for the
290      * variable portion are read, moves the state to ITEM.
291      */
292     private int processVariableHeader(final byte[] buf,
293                                       final int curOffset,
294                                       final int remaining) {
295         assert header != null;
296         assert header.isVariableLength();
297 
298         final int maxSize = header.getVariablePortionSize();
299         final int processSize =
300             Math.min(remaining, maxSize - headerBuf.position());
301 
302         headerBuf.put(buf, curOffset, processSize);
303         assert headerBuf.position() <= maxSize;
304 
305         if (headerBuf.position() == maxSize) {
306             headerBuf.flip();
307             header.readVariablePortion(headerBuf);
308             validator.update(headerBuf.array(), 0, maxSize);
309 
310             if (header.getType() == FILE_HEADER_TYPE_NUM) {
311                 headerBuf.clear();
312                 state = State.FILE_HEADER_ITEM;
313             } else {
314                 state = State.ITEM;
315             }
316         }
317 
318         return curOffset + processSize;
319     }

321     private int processFileHeaderItem(final byte[] buf,
322                                       final int curOffset,
323                                       final int remaining)
324         throws LogVerificationException {
325 
326         assert header != null;
327         assert logVersion == LogEntryType.UNKNOWN_FILE_HEADER_VERSION;
328 
329         final int maxSize = FileHeader.entrySize();
330         final int processSize =
331             Math.min(remaining, maxSize - headerBuf.position());
332 
333         headerBuf.put(buf, curOffset, processSize);
334         assert headerBuf.position() <= maxSize;
335 
336         if (headerBuf.position() == maxSize) {
337             validator.update(headerBuf.array(), 0, maxSize);
338             try {
339                 validator.validate(header.getChecksum(), fileNum, entryStart);
340             } catch (ChecksumException e) {
341                 throw newVerifyException(e);
342             }
343 
344             headerBuf.flip();
345             LogEntry fileHeaderEntry =
346                 LogEntryType.LOG_FILE_HEADER.getNewLogEntry();
347             fileHeaderEntry.readEntry(envImpl, header, headerBuf);
348             FileHeader fileHeaderItem =
349                 (FileHeader) fileHeaderEntry.getMainItem();
350 
351             /* Log version in the file header applies to all other entries. */
352             logVersion = fileHeaderItem.getLogVersion();
353 
354             prevEntryStart = entryStart;
355             entryStart += header.getSize() + maxSize;
356             state = State.INIT;
357         }
358 
359         return curOffset + processSize;
360     }

362     /**
363      * Processes the item portion of a log entry.  After all bytes for the item
364      * are read, moves the state back to INIT and bumps the entryStart.
365      */
366     private int processItem(final byte[] buf,
367                             final int curOffset,
368                             final int remaining)
369         throws LogVerificationException {
370 
371         assert header != null;
372 
373         final int maxSize = header.getItemSize();
374         final int processSize = Math.min(remaining, maxSize - itemPosition);
375 
376         validator.update(buf, curOffset, processSize);
377         itemPosition += processSize;
378         assert itemPosition <= maxSize;
379 
380         if (itemPosition == maxSize) {
381             try {
382                 validator.validate(header.getChecksum(), fileNum, entryStart);
383             } catch (ChecksumException e) {

392                 throw newVerifyException(e);
393             }

396             entryStart += header.getSize() + maxSize;
