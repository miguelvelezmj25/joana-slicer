81      * Creates a view for a given database and bindings.  The initial key range
82      * of the view will be open.
83      */
84     DataView(Database database, EntryBinding keyBinding,
85              EntryBinding valueBinding, EntityBinding entityBinding,
86              boolean writeAllowed, PrimaryKeyAssigner keyAssigner)
87         throws IllegalArgumentException {
88 
89         if (database == null) {
90             throw new IllegalArgumentException("database is null");
91         }
92         db = database;
93         try {
94             currentTxn =
95                 CurrentTransaction.getInstanceInternal(db.getEnvironment());
96             DatabaseConfig dbConfig;
97             if (db instanceof SecondaryDatabase) {
98                 secDb = (SecondaryDatabase) database;
99                 SecondaryConfig secConfig = secDb.getSecondaryConfig();
100                 secKeyCreator = secConfig.getKeyCreator();
101                 dbConfig = secConfig;
102             } else {
103                 dbConfig = db.getConfig();
104             }
105             ordered = !DbCompat.isTypeHash(dbConfig);
106             keyRangesAllowed = DbCompat.isTypeBtree(dbConfig);
107             recNumAllowed = DbCompat.isTypeQueue(dbConfig) ||
108                             DbCompat.isTypeRecno(dbConfig) ||
109                             DbCompat.getBtreeRecordNumbers(dbConfig);
110             recNumRenumber = DbCompat.getRenumbering(dbConfig);
111             dupsAllowed = DbCompat.getSortedDuplicates(dbConfig) ||
112                           DbCompat.getUnsortedDuplicates(dbConfig);
113             dupsOrdered = DbCompat.getSortedDuplicates(dbConfig);
114             transactional = currentTxn.isTxnMode() &&
115                             dbConfig.getTransactional();
116             readUncommittedAllowed = DbCompat.getReadUncommitted(dbConfig);
117             btreeRecNumDb = recNumAllowed && DbCompat.isTypeBtree(dbConfig);
118             range = new KeyRange(dbConfig.getBtreeComparator());
119         } catch (DatabaseException e) {
120             throw RuntimeExceptionWrapper.wrapIfNeeded(e);
121         }
122         this.writeAllowed = writeAllowed;
123         this.keyBinding = keyBinding;
124         this.valueBinding = valueBinding;
125         this.entityBinding = entityBinding;
126         this.keyAssigner = keyAssigner;
127         cursorConfig = CursorConfig.DEFAULT;
128 
129         if (valueBinding != null && entityBinding != null)
130             throw new IllegalArgumentException
131                 ("both valueBinding and entityBinding are non-null");
132 
133         if (keyBinding instanceof com.sleepycat.bind.RecordNumberBinding) {
134             if (!recNumAllowed) {
135                 throw new IllegalArgumentException
136                     ("RecordNumberBinding requires DB_BTREE/DB_RECNUM, " +
137                      "DB_RECNO, or DB_QUEUE");
138             }
139             recNumAccess = true;
140             if (btreeRecNumDb) {
141                 btreeRecNumAccess = true;
142             }
143         }
144         keysRenumbered = recNumRenumber || btreeRecNumAccess;
145     }

338     /**
339      * Appends a value and returns the new key.  If a key assigner is used
340      * it assigns the key, otherwise a QUEUE or RECNO database is required.
341      *
342      * Auto-commit must be performed by the caller.
343      */
344     OperationStatus append(Object value,
345                            Object[] retPrimaryKey,
346                            Object[] retValue)
347         throws DatabaseException {
348 
349         /*
350          * Flags will be NOOVERWRITE if used with assigner, or APPEND
351          * otherwise.
352          * Requires: if value param, value or entity binding
353          * Requires: if retPrimaryKey, primary key binding (no index).
354          * Requires: if retValue, value or entity binding
355          */
356         DatabaseEntry keyThang = new DatabaseEntry();
357         DatabaseEntry valueThang = new DatabaseEntry();
358         useValue(value, valueThang, null);
359         OperationStatus status;
360         if (keyAssigner != null) {
361             keyAssigner.assignKey(keyThang);
362             if (!range.check(keyThang)) {
363                 throw new IllegalArgumentException
364                     ("assigned key out of range");
365             }
366             DataCursor cursor = new DataCursor(this, true);
367             try {
368                 status = cursor.getCursor().putNoOverwrite(keyThang,
369                                                            valueThang);
370             } finally {
371                 cursor.close();
372             }
373         } else {
374             /* Assume QUEUE/RECNO access method. */
375             if (currentTxn.isCDBCursorOpen(db)) {
376                 throw new IllegalStateException
377                     ("cannot open CDB write cursor when read cursor is open");
378             }
379             status = DbCompat.append(db, useTransaction(),
380                                      keyThang, valueThang);
381             if (status == OperationStatus.SUCCESS && !range.check(keyThang)) {
382                 db.delete(useTransaction(), keyThang);
383                 throw new IllegalArgumentException
384                     ("appended record number out of range");
385             }
386         }
387         if (status == OperationStatus.SUCCESS) {
388             returnPrimaryKeyAndValue(keyThang, valueThang,
389                                      retPrimaryKey, retValue);
390         }
391         return status;
392     }

413             while (status == OperationStatus.SUCCESS) {
414                 if (keysRenumbered) {

434         throws DatabaseException {

445         } finally {

452                             /* FindBugs, this is ok. */

496     /**
497      * Populates the key entry and returns whether the key is within range.
498      */
499     boolean useKey(Object key, Object value, DatabaseEntry keyThang,
500                    KeyRange checkRange)
501         throws DatabaseException {
502 
503         if (key != null) {
504             if (keyBinding == null) {
505                 throw new IllegalArgumentException
506                     ("non-null key with null key binding");
507             }
508             keyBinding.objectToEntry(key, keyThang);
509         } else {
510             if (value == null) {
511                 throw new IllegalArgumentException("null key and null value");
512             }
513             if (entityBinding == null) {
514                 throw new IllegalStateException
515                     ("EntityBinding required to derive key from value");
516             }
517             if (!dupsView && isSecondary()) {
518                 DatabaseEntry primaryKeyThang = new DatabaseEntry();
519                 entityBinding.objectToKey(value, primaryKeyThang);
520                 DatabaseEntry valueThang = new DatabaseEntry();
521                 entityBinding.objectToData(value, valueThang);
522                 secKeyCreator.createSecondaryKey(secDb, primaryKeyThang,
523                                                  valueThang, keyThang);
524             } else {
525                 entityBinding.objectToKey(value, keyThang);
526             }
527         }
528         if (recNumAccess && DbCompat.getRecordNumber(keyThang) <= 0) {
529             return false;
530         }
531         if (checkRange != null && !checkRange.check(keyThang)) {
532             return false;
533         }
534         return true;
535     }

548      * Populates the value entry and throws an exception if the primary key
549      * would be changed via an entity binding.
550      */
551     void useValue(Object value, DatabaseEntry valueThang,
552                   DatabaseEntry checkKeyThang) {
553         if (valueBinding != null) {
554             /* Allow binding to handle null value. */
555             valueBinding.objectToEntry(value, valueThang);
556         } else if (entityBinding != null) {
557             if (value == null) {
558                 throw new IllegalArgumentException
559                     ("null value with entity binding");
560             }
561             entityBinding.objectToData(value, valueThang);
562             if (checkKeyThang != null) {
563                 DatabaseEntry thang = new DatabaseEntry();
564                 entityBinding.objectToKey(value, thang);
565                 if (!KeyRange.equalBytes(thang, checkKeyThang)) {
566                     throw new IllegalArgumentException
567                         ("cannot change primary key");
568                 }
569             }
570         } else {
571             if (value != null) {
572                 throw new IllegalArgumentException
573                     ("non-null value with null value/entity binding");
574             }
575             valueThang.setData(KeyRange.ZERO_LENGTH_BYTE_ARRAY);
576             valueThang.setOffset(0);
577             valueThang.setSize(0);
578         }
579     }

628     KeyRange subRange(KeyRange useRange,
629                       Object beginKey, boolean beginInclusive,
630                       Object endKey, boolean endInclusive)
631         throws DatabaseException, KeyRangeException {
632 
633         if (beginKey == endKey && beginInclusive && endInclusive) {
634             return subRange(useRange, beginKey);
635         }
636         if (!ordered) {
637             throw new UnsupportedOperationException
638                 ("Cannot use key ranges on an unsorted database");
639         }
640         DatabaseEntry beginThang =
641             (beginKey != null) ? makeRangeKey(beginKey) : null;
642         DatabaseEntry endThang =
643             (endKey != null) ? makeRangeKey(endKey) : null;
644 
645         return useRange.subRange(beginThang, beginInclusive,
646                                  endThang, endInclusive);
647     }

651      * dupsView, or the dupsRange if this is a dupsView, creating dupsRange if
652      * necessary.

654     KeyRange useSubRange()
655         throws DatabaseException {
656 
657         if (dupsView) {
658             synchronized (this) {
659                 if (dupsRange == null) {
660                     DatabaseConfig config =
661                         secDb.getPrimaryDatabase().getConfig();
662                     dupsRange = new KeyRange(config.getBtreeComparator());
663                 }

665             return dupsRange;
