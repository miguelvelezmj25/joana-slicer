23 /**
24  * Utility methods for combining, splitting and comparing two-part key values
25  * for duplicates databases.
26  *
27  * At the Btree storage level, for the key/data pairs in a duplicates database,
28  * the data is always zero length and the key is a two-part key. For embedded
29  * records, the key and data parts are visible at the BTree level as well. In
30  * both cases, the 'key' parameter in the API is the first part of the key. 
31  * The the 'data' parameter in the API is the second part of the key.
32  *
33  * The length of the first part is stored at the end of the combined key as a
34  * packed integer, so that the two parts can be split, combined, and compared
35  * separately.  The length is stored at the end, rather than the start, to
36  * enable key prefixing for the first part, e.g., for Strings with different
37  * lengths but common prefixes.
38  */
39 public class DupKeyData {

50      * a lower level.  It would be nice if there were a way to give ownership

65     public static byte[] combine(final byte[] key,
66                                  final int keyOff,
67                                  final int keySize,
68                                  final byte[] data,
69                                  final int dataOff,
70                                  final int dataSize) {
71         final int keySizeLen = PackedInteger.getWriteIntLength(keySize);
72         final byte[] buf = new byte[keySizeLen + keySize + dataSize];
73         System.arraycopy(key, keyOff, buf, 0, keySize);
74         System.arraycopy(data, dataOff, buf, keySize, dataSize);
75         final int nextOff =
76             PackedInteger.writeReverseInt(buf, keySize + dataSize, keySize);
77         assert nextOff == buf.length;
78         return buf;
79     }

105     public static void split(final byte[] twoPartKey,
106                              final int twoPartKeySize,
107                              final DatabaseEntry paramKey,
108                              final DatabaseEntry paramData) {
109         final int keySize =
110             PackedInteger.readReverseInt(twoPartKey, twoPartKeySize - 1);
111         assert keySize != PREFIX_ONLY;
112 
113         if (paramKey != null) {
114             final byte[] keyBuf = new byte[keySize];
115             System.arraycopy(twoPartKey, 0, keyBuf, 0, keySize);
116 
117             if (keySize == 0 || paramKey.getPartial()) {
118                 LN.setEntry(paramKey, keyBuf);
119             } else {
120                 paramKey.setData(keyBuf, 0, keySize);
121             }
122         }
123 
124         if (paramData != null) {
125             final int keySizeLen =
126                 PackedInteger.getReadIntLength(twoPartKey, twoPartKeySize - 1);
127 
128             final int dataSize = twoPartKeySize - keySize - keySizeLen;
129             final byte[] dataBuf = new byte[dataSize];
130             System.arraycopy(twoPartKey, keySize, dataBuf, 0, dataSize);
131 
132             if (dataSize == 0 || paramData.getPartial()) {
133                 LN.setEntry(paramData, dataBuf);
134             } else {
135                 paramData.setData(dataBuf, 0, dataSize);
136             }
137         }
138     }

230      * Comparator that compares the combined key/data two-part key, calling the
231      * user-defined btree and duplicate comparator as needed.
232      */
233     public static class TwoPartKeyComparator implements Comparator<byte[]> {
234 
235         private final Comparator<byte[]> btreeComparator;
236         private final Comparator<byte[]> duplicateComparator;
237 
238         public TwoPartKeyComparator(final Comparator<byte[]> btreeComparator,
239                                     final Comparator<byte[]> dupComparator) {
240             this.btreeComparator = btreeComparator;
241             this.duplicateComparator = dupComparator;
242         }
243 
244         public int compare(final byte[] twoPartKey1,
245                            final byte[] twoPartKey2) {
246 
247             /* Compare key portion. */
248             final int origKeySize1 = PackedInteger.readReverseInt
249                 (twoPartKey1, twoPartKey1.length - 1);
250 
251             final int keySize1 = (origKeySize1 == PREFIX_ONLY) ?
252                 (twoPartKey1.length - 1) :
253                 origKeySize1;
254 
255             final int origKeySize2 = PackedInteger.readReverseInt
256                 (twoPartKey2, twoPartKey2.length - 1);
257 
258             final int keySize2 = (origKeySize2 == PREFIX_ONLY) ?
259                 (twoPartKey2.length - 1) :
260                 origKeySize2;
261 
262             final int keyCmp;
263 
264             if (btreeComparator == null) {
265                 keyCmp = Key.compareUnsignedBytes(
266                     twoPartKey1, 0, keySize1, twoPartKey2, 0, keySize2);
267             } else {
268                 final byte[] key1 = new byte[keySize1];
269                 final byte[] key2 = new byte[keySize2];
270                 System.arraycopy(twoPartKey1, 0, key1, 0, keySize1);
271                 System.arraycopy(twoPartKey2, 0, key2, 0, keySize2);
272                 keyCmp = btreeComparator.compare(key1, key2);
273             }
274 
275             if (keyCmp != 0) {
276                 return keyCmp;
277             }
278 
279             if (origKeySize1 == PREFIX_ONLY || origKeySize2 == PREFIX_ONLY) {
280                 if (origKeySize1 == origKeySize2) {
281                     return 0;
282                 }
283                 return (origKeySize1 == PREFIX_ONLY) ? -1 : 1;
284             }
285 
286             /* Compare data portion. */
287             final int keySizeLen1 = PackedInteger.getReadIntLength
288                 (twoPartKey1, twoPartKey1.length - 1);
289             final int keySizeLen2 = PackedInteger.getReadIntLength
290                 (twoPartKey2, twoPartKey2.length - 1);
291 
292             final int dataSize1 = twoPartKey1.length - keySize1 - keySizeLen1;
293             final int dataSize2 = twoPartKey2.length - keySize2 - keySizeLen2;
294 
295             final int dataCmp;
296             if (duplicateComparator == null) {
297                 dataCmp = Key.compareUnsignedBytes
298                     (twoPartKey1, keySize1, dataSize1,
299                      twoPartKey2, keySize2, dataSize2);
300             } else {
301                 final byte[] data1 = new byte[dataSize1];
302                 final byte[] data2 = new byte[dataSize2];
303                 System.arraycopy(twoPartKey1, keySize1, data1, 0, dataSize1);
304                 System.arraycopy(twoPartKey2, keySize2, data2, 0, dataSize2);
305                 dataCmp = duplicateComparator.compare(data1, data2);
306             }
307             return dataCmp;
308         }
309     }
