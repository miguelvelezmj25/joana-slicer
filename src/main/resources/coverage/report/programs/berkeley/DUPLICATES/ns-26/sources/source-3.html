34 /*
35  * Class used for figuring out the difference on local and remote databases,
36  * also getting the VLSN number for different records.
37  *
38  * This class only needs to traverse the local and remote databases once and
39  * do the check, this is because the different areas are sequential on the
40  * database.
41  *
42  * It uses a hash set to save the records on the local different area, and
43  * another hash set to save the records on the remote different area. Then
44  * traverse the records in local set to see whether it exists in the remote
45  * set, and do the same thing on records in remote set.
46  */
47 public class DiffRecordAnalyzer {
48     public static final long DATABASE_END = -1;

50     /* The analysis method used by network LDiff. */
51     public static void doAnalysis(Database localDb,
52                                   Protocol protocol,
53                                   DataChannel channel,
54                                   DiffTracker tracker,
55                                   boolean doPrint)
56         throws Exception {
57 
58         List<MismatchedRegion> regions = tracker.getDiffRegions();
59         Cursor localCursor = null;
60         try {
61             localCursor = localDb.openCursor(null, null);
62 
63             for (MismatchedRegion region : regions) {
64                 if (region.isLocalAdditional()) {
65                     printLocalAdditional(localCursor, region, doPrint);
66                     continue;
67                 }
68 
69                 if (region.isRemoteAdditional()) {
70                      HashSet<Record> records =
71                          getDiffArea(protocol, channel, region);
72                      if (doPrint) {
73                          printAdditional(records, true);
74                      }
75                      records.clear();
76                      continue;
77                 }
78 
79                 HashSet<Record> localRecords =
80                     getDiffArea(localCursor, region.getLocalBeginKey(),
81                                 region.getLocalBeginData(),
82                                 region.getLocalDiffSize());
83                 HashSet<Record> remoteRecords =
84                     getDiffArea(protocol, channel, region);
85                 if (doPrint) {
86                 printDiffs(localRecords, remoteRecords);
87             }
88                 localRecords.clear();
89                 remoteRecords.clear();
90             }
91         } finally {
92             if (localCursor != null) {
93                 localCursor.close();
94             }
95         }
96     }

98     /* The analysis method used by two local databases. */
99     public static void doAnalysis(Database localDb,
100                                   Database remoteDb,
101                                   DiffTracker tracker,
102                                   boolean doPrint)
103         throws Exception {
104 
105         List<MismatchedRegion> regions = tracker.getDiffRegions();
106         Cursor localCursor = null;
107         Cursor remoteCursor = null;
108         try {
109             localCursor = localDb.openCursor(null, null);
110             remoteCursor = remoteDb.openCursor(null, null);
111 
112             for (MismatchedRegion region : regions) {
113                 if (region.isLocalAdditional()) {
114                     printLocalAdditional(localCursor, region, doPrint);
115                     continue;
116                 }
117 
118                 if (region.isRemoteAdditional()) {
119                     HashSet<Record> records =
120                         getDiffArea(remoteCursor, region.getRemoteBeginKey(),
121                                     region.getRemoteBeginData(),
122                                     region.getRemoteDiffSize());
123                     if (doPrint) {
124                         printAdditional(records, true);
125                     }
126                     records.clear();
127                     continue;
128                 }
129 
130                 HashSet<Record> localRecords =
131                     getDiffArea(localCursor, region.getLocalBeginKey(),
132                                 region.getLocalBeginData(),
133                                 region.getLocalDiffSize());
134                 HashSet<Record> remoteRecords =
135                     getDiffArea(remoteCursor, region.getRemoteBeginKey(),
136                                 region.getRemoteBeginData(),
137                                 region.getRemoteDiffSize());
138                 if (doPrint) {
139                 printDiffs(localRecords, remoteRecords);
140             }
141                 localRecords.clear();
142                 remoteRecords.clear();
143             }
144         } finally {
145             if (localCursor != null) {
146                 localCursor.close();
147             }
148             if (remoteCursor != null) {
149                 remoteCursor.close();
150             }
151         }
152     }

208     /* Return a different area on local database in a hash set. */
209     private static HashSet<Record> getDiffArea(Cursor cursor,
210                                                byte[] beginKey,
211                                                byte[] beginData,
212                                                long diffSize)
213         throws Exception {
214 
215         HashSet<Record> records = new HashSet<Record>();
216         LogManager logManager = DbInternal.getNonNullEnvImpl
217             (cursor.getDatabase().getEnvironment()).getLogManager();
218 
219         DatabaseEntry key = new DatabaseEntry(beginKey);
220         DatabaseEntry data = new DatabaseEntry(beginData);
221         boolean scanToEnd = (diffSize == DATABASE_END ? true : false);
222         long count = 1;
223 
224         for (OperationStatus status =
225              cursor.getSearchBoth(key, data, LockMode.DEFAULT);
226              status == OperationStatus.SUCCESS;
227              status = cursor.getNext(key, data, LockMode.DEFAULT)) {
228 
229             if (!scanToEnd && count > diffSize) {
230                 break;
231             }
232             records.add(new Record(key.getData(), data.getData(),
233                                    getVLSN(cursor, logManager)));
234             count++;
235         }
236 
237         return records;
238     }

249     /* Get the records of a different area on remote database in a HashSet. */
250     private static HashSet<Record> getDiffArea(Protocol protocol,
251                                                DataChannel channel,
252                                                MismatchedRegion region)
253         throws Exception {
254 
255         protocol.write(protocol.new RemoteDiffRequest(region), channel);
256 
257         /* Check whether getting records on remote database is successful. */
258         Message message = protocol.read(channel);
259         if (message.getOp() == Protocol.ERROR) {
260             throw new LDiffRecordRequestException
261                 (((Protocol.Error) message).getErrorMessage());
262         }
263 
264         if (message.getOp() != Protocol.DIFF_AREA_START) {
265             throw new ProtocolException
266                 (message, Protocol.DiffAreaStart.class);
267         }
268 
269         /* Add those different records until protocol sees an end signal. */
270         HashSet<Record> records = new HashSet<Record>();
271         while (true) {
272             try {
273                 Protocol.RemoteRecord record =
274                     protocol.read(channel, Protocol.RemoteRecord.class);
275                 records.add(new Record(record.getKey(),
276                                        record.getData(),
277                                        record.getVLSN()));
278             } catch (ProtocolException pe) {
279                 if (pe.getUnexpectedMessage().getOp() !=
280                     Protocol.DIFF_AREA_END) {
281                     throw pe;
282                 }
283                 break;
284             }
285         }
286 
287         return records;
288     }

290     /* Compare the differences between two sets. */
291     private static void printDiffs(HashSet<Record> localDiffs,
292                                      HashSet<Record> remoteDiffs) {
293         System.err.println("************************************************");
294         System.err.println("Different records between local and remote database " +
295                            "in a specific different area.");
296         for (Record record : localDiffs) {
297             if (!remoteDiffs.contains(record)) {
298                 printRecord(record, "remote", true);
299             }
300         }
301 
302         for (Record record : remoteDiffs) {
303             if (!localDiffs.contains(record)) {
304                 printRecord(record, "local", true);
305             }
306         }
307         System.err.println("************************************************");
308     }

310     /*
311      * Get the LSN that the cursor currently points to, there are two cases
312      * that a record doesn't have a VLSN:
313      * 1. compare between two local databases.
314      * 2. compare between two converted environments.
315      *
316      * In the above two cases, we actually return NULL_VLSN instead of null, so
317      * that the message doesn't complain about a null value.
318      */
319     private static VLSN getVLSN(Cursor cursor, LogManager logManager)
320         throws Exception {
321 
322         CursorImpl cursorImpl = DbInternal.getCursorImpl(cursor);
323         cursorImpl.latchBIN();
324         final long lsn = cursorImpl.getCurrentLsn();
325         cursorImpl.releaseBIN();
326 
327         WholeEntry entry = logManager.getLogEntryAllowInvisible(lsn);
328 
329         VLSN vlsn = entry.getHeader().getVLSN();
330         if (vlsn == null) {
331             vlsn = VLSN.NULL_VLSN;
332         }
333 
334         return vlsn;
335     }
