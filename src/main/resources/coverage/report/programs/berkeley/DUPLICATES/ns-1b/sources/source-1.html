162     /**
163      * The dispatcher method that starts up new log file feeders.
164      */
165     @Override
166     public void run() {
167         try {
168             while (true) {
169                 final DataChannel channel =
170                     channelQueue.poll(POLL_TIMEOUT, TimeUnit.MILLISECONDS);
171                 if (channel == RepUtils.CHANNEL_EOF_MARKER) {
172                     LoggerUtils.info(logger, envImpl,
173                                      "Log file Feeder manager soft shutdown.");
174                     return;
175                 }
176                 new LogFileFeeder(this, channel).start();
177             }
178         } catch (InterruptedException ie) {
179             LoggerUtils.info
180                 (logger, envImpl, "Log file feeder manager interrupted");
181         } catch (Exception e) {
182             LoggerUtils.severe(logger, envImpl,
183                                "unanticipated exception: " + e.getMessage());
184             throw new EnvironmentFailureException
185                 (envImpl, EnvironmentFailureReason.UNCAUGHT_EXCEPTION, e);
186         } finally {
187             shutdown();
188         }
189     }
190 
191     public void shutdown() {
192         LoggerUtils.fine
193             (logger, envImpl, "Shutting down log file feeder manager");
194 
195         if (!shutdown.compareAndSet(false, true)) {
196             return;
197         }
198         shutdownThread(logger);
199 
200         /* shutdown active feeder threads */
201         for (LogFileFeeder feeder :
202              new ArrayList<LogFileFeeder>(feeders.values())) {
203             feeder.shutdown();
204         }
205         leaseTimer.cancel();
206         /*
207          * Terminate any outstanding leases, so we don't hold back file
208          * deletion by the cleaner.
209          */
210         for (Lease l : new ArrayList<Lease>(leases.values())) {
211             l.terminate();
212         }
213         serviceDispatcher.cancel(FEEDER_SERVICE);
214         cleanup();
215         LoggerUtils.fine(logger, envImpl,
216                          "Shut down log file feeder manager completed");
217     }

228     /**
229      * Provides the lease mechanism used to maintain a handle to the DbBackup
230      * object across Server client disconnects.
231      */
232     class Lease extends TimerTask {
233         private final int id;
234         private DbBackup dbBackup;
235 
236         public Lease(int id, long duration, DbBackup dbbackup) {
237             super();
238             this.dbBackup = dbbackup;
239             this.id = id;
240             Lease oldLease = leases.put(id, this);
241             if (oldLease != null) {
242                 throw EnvironmentFailureException.unexpectedState
243                     ("Found an old lease for node: " + id);
244             }
245             leaseTimer.schedule(this, duration);
246         }
247 
248         @Override
249         /* The timer went off, expire the lease if it hasn't been terminated */
250         public synchronized void run() {
251             if (dbBackup == null) {
252                 return;
253             }
254             dbBackup.endBackup();
255             terminate();
256         }
257 
258         /**
259          * Fetches the leased DbBackup instance and terminates the lease.
260          *
261          * @return the dbBackup instance, if the lease hasn't already been
262          * terminated
263          */
264         public synchronized DbBackup terminate() {
265             if (dbBackup == null) {
266                 return null;
267             }
268             cancel();
269             Lease l = leases.remove(id);
270             assert(l == this);
271             DbBackup saveDbBackup = dbBackup;
272             dbBackup = null;
273             return saveDbBackup;
274         }
275 
276         public synchronized DbBackup getOpenDbBackup() {
277             return (dbBackup != null) && dbBackup.backupIsOpen() ?
278                     dbBackup :
279                     null;
280         }
281     }
