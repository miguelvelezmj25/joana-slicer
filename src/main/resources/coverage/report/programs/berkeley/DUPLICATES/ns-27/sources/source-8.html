201         private long lastLSN = DbLsn.NULL_LSN;
202         private final PrintStream out1;
203 
204         VerifyReader(EnvironmentImpl envImpl, PrintStream out, long startLsn,
205                      long endLsn) {
206             super(envImpl,
207                   10000,
208                   true,              // forward
209                   startLsn,
210                   null,              // singleFileNumber,
211                   DbLsn.NULL_LSN,
212                   endLsn);
213             this.out1 = out;
214         }
215 
216         @Override
217         protected boolean isTargetEntry() {
218             return (currentEntryHeader.getReplicated());
219         }
220 
221         @Override
222         protected boolean processEntry(ByteBuffer entryBuffer) {
223             VLSN currentVLSN = currentEntryHeader.getVLSN();
224             long currentLSN = getLastLsn();
225 
226             if (currentVLSN == null) {
227                 throw new RuntimeException
228                      (DbLsn.getNoFormatString(currentLSN) +
229                       "Should be a replicated entry");
230             }
231 
232             if (currentVLSN.isNull()) {
233                 out1.println("unexpected LSN " +
234                             DbLsn.getNoFormatString(getLastLsn()) +
235                             " has vlsn " + currentVLSN);
236             }
237 
238             if ((lastVLSN != null) && lastVLSN.isNull()) {
239 
240                 /* First entry seen */
241                 out1.println("first VLSN = " + currentVLSN + " at lsn " +
242                             DbLsn.getNoFormatString(getLastLsn()));
243             } else if (!currentEntryHeader.isInvisible() &&
244                        !currentVLSN.follows(lastVLSN)) {
245 
246                 /* Note the first entry, check for a gap. */
247                 out1.println("gap of " +
248                             (currentVLSN.getSequence() -
249                             lastVLSN.getSequence()) +
250                             " Last=" + lastVLSN + " at lsn " +
251                             DbLsn.getNoFormatString(lastLSN) +
252                             " next=" + currentVLSN + " at lsn " +
253                             DbLsn.getNoFormatString(currentLSN));
254 
255             }
256 
257             /* Note the invisible log entries. */
258             if (showInvisible && currentEntryHeader.isInvisible()) {
259                 out1.println("VLSN " + currentVLSN + " at lsn " +
260                             DbLsn.getNoFormatString(currentLSN) +
261                             " is invisible.");
262             }
263 
264             if (!currentEntryHeader.isInvisible()) {
265                 lastVLSN = currentVLSN;
266                 lastLSN = currentLSN;
267             }
268             entryBuffer.position(entryBuffer.position() +
269                                  currentEntryHeader.getItemSize());
270             return true;
271         }
272 
273         void displayLast() {
274             out1.println("LastVLSN = " + lastVLSN + " at " +
275                          DbLsn.getNoFormatString(lastLSN));
276         }

280      * A Decoder reader can dump the LNs from one of the JE internal databases
281      * and display the LNs in a deserialized format, for easy debugging. The
282      * target internal databases are the VLSNIndex and the RepGroup. We could
283      * add FileSummaryLNs later.
284      */
285     private abstract class DecoderReader extends FileReader {
286         protected final PrintStream outStream;
287         private final Map<Byte, LNLogEntry<?>> targetMap;
288         protected LNLogEntry<?> targetEntry;
289         private final DatabaseId targetDbId;
290 
291         DecoderReader(Environment env,
292                       PrintStream out,
293                       long startLsn,
294                       long endLsn,
295                       String dbName) {
296             super(DbInternal.getNonNullEnvImpl(env),
297                   10000,
298                   true,              // forward
299                   startLsn,
300                   null,              // singleFileNumber,
301                   DbLsn.NULL_LSN,
302                   endLsn);
303             this.outStream = out;
304 
305             targetMap = new HashMap<Byte, LNLogEntry<?>>();
306 
307             for (LogEntryType entryType : LogEntryType.getAllTypes()) {
308                 if (entryType.isUserLNType()) {
309                     targetMap.put(entryType.getTypeNum(),
310                                   (LNLogEntry<?>) entryType.getNewLogEntry());
311                 }
312             }
313 
314             DatabaseConfig dbConfig = new DatabaseConfig();
315             dbConfig.setReadOnly(true);
316             DbInternal.setUseExistingConfig(dbConfig, true);
317             Database db = env.openDatabase(null, dbName, dbConfig);
318             targetDbId = DbInternal.getDbImpl(db).getId();
319             db.close();
320         }
321 
322         @Override
323         protected boolean isTargetEntry() {
324             /* Is it a target entry? */
325             targetEntry = targetMap.get(currentEntryHeader.getType());
326             return targetEntry != null;
327         }
328 
329         @Override
330         protected boolean processEntry(ByteBuffer entryBuffer) {
331             targetEntry.readEntry(envImpl, currentEntryHeader, entryBuffer);
332             targetEntry.postFetchInit(false /*isDupDb*/);
333 
334             if (!targetEntry.getDbId().equals(targetDbId)) {
335                 return false;
336             }
337 
338             outStream.print("LSN=" + DbLsn.getNoFormatString(getLastLsn()));
339 
340             display();
341             return true;
342         }
343 
344         protected abstract void display();
345     }

370 
371                 if (keyVal == VLSNRange.RANGE_KEY) {
372                     outStream.print(" range: ");
373                     VLSNRange range = VLSNRange.readFromDatabase(data);
374                     outStream.println(range);
375                 } else {
376                     outStream.print(" key=" + keyVal);
377                     VLSNBucket bucket = VLSNBucket.readFromDatabase(data);
378                     outStream.println(" " + bucket);
379                     if (verbose) {
380                         outStream.println("-------------------------------");
381                         bucket.dump(outStream);
382                         outStream.println("-------------------------------\n");
383                     }
