21 /**
22  * Widens a value returned by another input when any readXxx method is called.
23  * Used to cause an Accessor to read a widened value.
24  *
25  * For non-key fields we support all Java primitive widening:
26  * - byte to short, int, long, float, double, BigInteger and BigDecimal
27  * - short to int, long, float, double, BigInteger and BigDecimal
28  * - char to int, long, float, double, BigInteger and BigDecimal
29  * - int to long, float, double, BigInteger and BigDecimal
30  * - long to float, double, BigInteger and BigDecimal
31  * - float to double
32  *
33  * For non-key fields we also support:
34  * - Java reference widening
35  * - primitive to primitive wrapper
36  * - Java primitive widening to corresponding primitive wrappers
37  * - Java widening of primitive wrapper to primitive wrapper
38  *
39  * For secondary keys fields we ONLY support:
40  * - primitive to primitive wrapper
41  *
42  * But for primary keys and composite key fields we ONLY support:
43  * - primitive to primitive wrapper
44  * - primitive wrapper to primitive
45  * These conversions don't require any converter, since the stored format is
46  * not changed.  A WidenerInput is not used for these changes.
47  *
48  * @author Mark Hayes
49  */

283         throws RefreshException {
284 
285         switch (fromFormatId) {
286         case Format.ID_BOOL:
287             checkToFormat(Format.ID_BOOL_W);
288             return input.readBoolean();

291         case Format.ID_BYTE_W:

293             return (b != null) ? byteToObject(b) : null;
294         case Format.ID_SHORT:

299         case Format.ID_INT:

304         case Format.ID_LONG:
