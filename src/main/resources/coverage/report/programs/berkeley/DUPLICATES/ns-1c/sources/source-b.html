34 /**
35  * The class for firing MonitorChangeEvents.
36  *
37  * Each time when there happens a MonitorChangeEvents, it refreshes the group
38  * information so that it can send messages to current monitors.
39  */
40 public class MonitorEventManager {
41 
42     /* The time when this node joins the group, 0 if it hasn't joined yet. */
43     private long joinTime = 0L;
44 
45     private final RepNode repNode;
46 
47     public MonitorEventManager(RepNode repNode) {
48         this.repNode = repNode;
49     }
50 
51     /* Return the time when JoinGroupEvent for this RepNode fires. */
52     public long getJoinTime() {
53         return joinTime;
54     }
55 
56     /* Disable the LeaveGroupEvent because the node is abnormally closed. */
57     public void disableLeaveGroupEvent() {
58         joinTime = 0L;
59     }
60 
61     /**
62      * Fire a GroupChangeEvent.
63      */
64     public void notifyGroupChange(String nodeName, GroupChangeType opType)
65         throws DatabaseException {
66 
67         RepGroupImpl repGroup = repNode.getGroup();
68         GroupChange changeEvent =
69             getProtocol(repGroup).new GroupChange(repGroup, nodeName, opType);
70         refreshMonitors(repGroup, changeEvent);
71     }
72 
73     /**
74      * Fire a JoinGroupEvent.
75      */
76     public void notifyJoinGroup()
77         throws DatabaseException {
78 
79         if (joinTime > 0) {
80             /* Already notified. */
81             return;
82         }
83 
84         joinTime = System.currentTimeMillis();
85         RepGroupImpl repGroup = repNode.getGroup();
86         JoinGroup joinEvent =
87             getProtocol(repGroup).new JoinGroup(repNode.getNodeName(),
88                                                 repNode.getMasterName(),
89                                                 joinTime);
90         refreshMonitors(repGroup, joinEvent);
91     }
92 
93     /**
94      * Fire a LeaveGroupEvent and wait for responses.
95      */
96     public void notifyLeaveGroup(LeaveReason reason)
97         throws DatabaseException, InterruptedException {
98 
99         if (joinTime == 0) {
100             /* No join event, therefore no matching leave event. */
101             return;
102         }
103 
104         RepGroupImpl repGroup = repNode.getGroup();
105         LeaveGroup leaveEvent =
106             getProtocol(repGroup).new LeaveGroup(repNode.getNodeName(),
107                                                  repNode.getMasterName(),
108                                                  reason,
109                                                  joinTime,
110                                                  System.currentTimeMillis());
111         final FutureTrackingCompService<MessageExchange> compService =
112             refreshMonitors(repGroup, leaveEvent);
113 
114         /* Wait for the futures to be evaluated. */
115        Utils.checkFutures
116         (compService, 10, TimeUnit.SECONDS, repNode.getLogger(),
117          repNode.getRepImpl(), null);
118     }
119 
120     /* Create a monitor protocol. */
121     private com.sleepycat.je.rep.monitor.Protocol
122         getProtocol(RepGroupImpl repGroup) {
123 
124         return new com.sleepycat.je.rep.monitor.Protocol
125             (repGroup.getName(), NameIdPair.NOCHECK, null,
126              repNode.getRepImpl().getChannelFactory());
127     }
128 
129     /* Refresh all the monitors with specified message. */
130     private FutureTrackingCompService<MessageExchange>
131         refreshMonitors(RepGroupImpl repGroup,
132                         RequestMessage requestMessage) {
133         Set<InetSocketAddress> monitors = repGroup.getAllMonitorSockets();
134         if (monitors.size() > 0) {
135             LoggerUtils.info(repNode.getLogger(), repNode.getRepImpl(),
136                              "Refreshed " + monitors.size() + " monitors.");
137         }
138         /* Broadcast and forget. */
139         return Utils.broadcastMessage(monitors,
140                                       MonitorService.SERVICE_NAME,
141                                       requestMessage,
142                                       repNode.getElections().getThreadPool());
143     }
144 }
