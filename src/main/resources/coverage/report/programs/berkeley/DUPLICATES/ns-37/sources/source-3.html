28 /**
29  * A cursor-like interface that enforces a key range.  The method signatures
30  * are actually those of SecondaryCursor, but the pKey parameter may be null.
31  * It was done this way to avoid doubling the number of methods.
32  *
33  * <p>This is not a fully general implementation of a range cursor and should
34  * not be used directly by applications; however, it may evolve into a
35  * generally useful range cursor some day.</p>

37  * @author Mark Hayes
38  */

177      * Dups the cursor, sets the cursor and secCursor fields to the duped
178      * cursor, and returns the old cursor.  Always call endOperation in a
179      * finally clause after calling beginOperation.
180      *
181      * <p>If the returned cursor == the cursor field, the cursor is
182      * uninitialized and was not duped; this case is handled correctly by
183      * endOperation.</p>
184      */
185     private Cursor beginOperation()
186         throws DatabaseException {
187 
188         Cursor oldCursor = cursor;
189         if (initialized) {
190             cursor = dupCursor(cursor, true);
191             if (secCursor != null) {
192                 secCursor = (SecondaryCursor) cursor;
193             }
194         } else {
195             return cursor;
196         }
197         return oldCursor;
198     }

200     /**
201      * If the operation succeeded, leaves the duped cursor in place and closes
202      * the oldCursor.  If the operation failed, moves the oldCursor back in
203      * place and closes the duped cursor.  oldCursor may be null if
204      * beginOperation was not called, in cases where we don't need to dup
205      * the cursor.  Always call endOperation when a successful operation ends,
206      * in order to set the initialized field.
207      */
208     private void endOperation(Cursor oldCursor,
209                               OpResult result,
210                               DatabaseEntry key,
211                               DatabaseEntry pKey,
212                               DatabaseEntry data)
213         throws DatabaseException {
214 
215         if (result.isSuccess()) {
216             if (oldCursor != null && oldCursor != cursor) {
217                 closeCursor(oldCursor);
218             }
219             if (key != null) {
220                 swapData(key, privKey);
221             }
222             if (pKey != null && secCursor != null) {
223                 swapData(pKey, privPKey);
224             }
225             if (data != null) {
226                 swapData(data, privData);
227             }
228             initialized = true;
229         } else {
230             if (oldCursor != null && oldCursor != cursor) {
231                 closeCursor(cursor);
232                 cursor = oldCursor;
233                 if (secCursor != null) {
234                     secCursor = (SecondaryCursor) cursor;
235                 }
236             }
237         }
238     }

341      * This method will restart the operation when a key range is used and an
342      * insertion at the end of the key range is performed in another thread.
343      * The restarts are needed because a sequence of cursor movements is
344      * performed, and serializable isolation cannot be relied on to prevent
345      * insertions in other threads. Without the restarts, getLast could return
346      * NOTFOUND when keys in the range exist. This may only be an issue for JE
347      * since it uses record locking, while DB core uses page locking.
348      */
349     public OpResult getLast(DatabaseEntry key,
350                             DatabaseEntry pKey,
351                             DatabaseEntry data,
352                             OpReadOptions options)
353         throws DatabaseException {
354 
355         OpResult result = OpResult.FAILURE;
356 
357         if (!range.hasBound()) {
358             setParams(key, pKey, data);
359             result = doGetLast(options);
360             endOperation(null, result, null, null, null);
361             return result;
362         }
363 
364         Cursor oldCursor = beginOperation();
365         try {
366             if (pkRange != null) {
367                 result = getLastInPKeyRange(options);
368 
369                 /* Final check on candidate key and pKey value. */
370                 if (result.isSuccess() &&

372                     result = OpResult.FAILURE;

374             } else {
375                 result = getLastInKeyRange(options);
376 
377                 /* Final check on candidate key value. */
378                 if (result.isSuccess() &&

390     /**
391      * Performs getLast operation when a main key range is specified but
392      * pkRange is null. Does everything but the final checks for key in range,
393      * i.e., when SUCCESS is returned the caller should do the final check.
394      */
395     private OpResult getLastInKeyRange(OpReadOptions options)
396         throws DatabaseException {
397 
398         /* Without an endKey, getLast returns the candidate record. */
399         if (range.endKey == null) {
400             return doGetLast(options);
401         }
402 
403         /*
404          * K stands for the main key at the cursor position in the comments
405          * below.
406          */
407         while (true) {
408             KeyRange.copy(range.endKey, privKey);
409             OpResult result = doGetSearchKeyRange(options);
410 
411             if (result.isSuccess()) {
412 
413                 /* Found K >= endKey. */
414                 if (range.endInclusive &&
415                     range.compare(range.endKey, privKey) == 0) {
416 
417                     /* K == endKey and endKey is inclusive. */
418 
419                     if (!sortedDups) {
420                         /* If dups are not configured, we're done. */
421                         return result;
422                     }
423 
424                     /*
425                      * If there are dups, we're positioned at endKey's first
426                      * dup and we want to move to its last dup. Move to the
427                      * first dup for the next main key (getNextNoDup) and then
428                      * the prev record. In the absence of insertions by other
429                      * threads, the prev record is the last dup for endKey.
430                      */
431                     result = doGetNextNoDup(options);
432                     if (result.isSuccess()) {
433 
434                         /*
435                          * K > endKey. Move backward to the last dup for
436                          * endKey.
437                          */
438                         result = doGetPrev(options);
439                     } else {
440 
441                         /*
442                          * endKey is the last main key in the DB. Its last dup
443                          * is the last key in the DB.
444                          */
445                         result = doGetLast(options);
446                     }
447                 } else {
448 
449                     /*
450                      * K > endKey or endKey is exclusive (and K >= endKey). In
451                      * both cases, moving to the prev key finds the last key in
452                      * the range, whether or not there are dups.
453                      */
454                     result = doGetPrev(options);
455                 }
456             } else {
457 
458                 /*
459                  * There are no keys >= endKey in the DB. The last key in the
460                  * range is the last key in the DB.
461                  */
462                 result = doGetLast(options);
463             }
464 
465             if (!result.isSuccess()) {
466                 return result;
467             }
468 
469             if (!range.checkEnd(privKey, true)) {
470 
471                 /*
472                  * The last call above (getPrev or getLast) returned a key
473                  * outside the endKey range. Another thread must have inserted
474                  * this key. Start over.
475                  */
476                 continue;
477             }
478 
479             return result;
480         }
481     }

483     /**
484      * Performs getLast operation when both a main key range (which must be a
485      * single key range) and a pkRange are specified. Does everything but the
486      * final checks for key and pKey in range, i.e., when SUCCESS is returned
487      * the caller should do the final two checks.
488      */
489     private OpResult getLastInPKeyRange(OpReadOptions options)
490         throws DatabaseException {
491 
492         /* We can do an exact search when range and pkRange are single keys. */
493         if (pkRange.isSingleKey()) {
494             KeyRange.copy(range.beginKey, privKey);
495             KeyRange.copy(pkRange.beginKey, privPKey);
496             return doGetSearchBoth(options);
497         }
498 
499         /*
500          * When dups are not configured, getSearchKey for the main key returns
501          * the only possible candidate record.
502          */
503         if (!sortedDups) {
504             KeyRange.copy(range.beginKey, privKey);
505             return doGetSearchKey(options);
506         }
507 
508         /*
509          * K stands for the main key and D for the duplicate (data item) at the
510          * cursor position in the comments below
511          */
512         while (true) {
513 
514             if (pkRange.endKey != null) {
515 
516                 KeyRange.copy(range.beginKey, privKey);
517                 KeyRange.copy(pkRange.endKey, privPKey);
518                 OpResult result = doGetSearchBothRange(options);
519 
520                 if (result.isSuccess()) {
521 
522                     /* Found D >= endKey. */
523                     if (!pkRange.endInclusive ||
524                         pkRange.compare(pkRange.endKey, privPKey) != 0) {
525 
526                         /*
527                          * D > endKey or endKey is exclusive (and D >= endKey).
528                          * In both cases, moving to the prev dup finds the last
529                          * key in the range.
530                          */
531                         result = doGetPrevDup(options);
532 
533                         if (!result.isSuccess()) {
534                             return result;
535                         }
536 
537                         if (!pkRange.checkEnd(privPKey, true)) {
538 
539                             /*
540                              * getPrevDup returned a key outside the endKey
541                              * range. Another thread must have inserted this
542                              * key. Start over.
543                              */
544                             continue;
545                         }
546                     }
547                     /* Else D == endKey and endKey is inclusive. */
548 
549                     return result;
550                 }
551                 /* Else there are no dups >= endKey.  Fall through. */
552             }
553 
554             /*
555              * We're here for one of two reasons:
556              *  1. pkRange.endKey == null.
557              *  2. There are no dups >= endKey for the main key (status
558              *     returned by getSearchBothRange above was not SUCCESS).
559              * In both cases, the last dup in the range is the last dup for the
560              * main key.
561              */
562             KeyRange.copy(range.beginKey, privKey);
563             OpResult result = doGetSearchKey(options);
564 
565             if (!result.isSuccess()) {
566                 return result;
567             }
568 
569             /*
570              * K == the main key and D is its first dup. We want to move to its
571              * last dup. Move to the first dup for the next main key;
572              * (getNextNoDup) and then the prev record. In the absence of
573              * insertions by other threads, the prev record is the last dup for
574              * the main key.
575              */
576             result = doGetNextNoDup(options);
577 
578             if (result.isSuccess()) {
579 
580                 /*
581                  * K > main key and D is its first dup. Move to the prev record
582                  * which should be the last dup for the main key.
583                  */
584                 result = doGetPrev(options);
585             } else {
586 
587                 /*
588                  * The main key specified is the last main key in the DB. Its
589                  * last dup is the last record in the DB.
590                  */
591                 result = doGetLast(options);
592             }
593 
594             if (!result.isSuccess()) {
595                 return result;
596             }
597 
598             if (!range.checkEnd(privKey, true)) {
599 
600                 /*
601                  * The last call above (getPrev or getLast) returned a key
602                  * outside the endKey range. Another thread must have inserted
603                  * this key. Start over.
604                  */
605                 continue;
606             }
607 
608             return result;
609         }
610     }
