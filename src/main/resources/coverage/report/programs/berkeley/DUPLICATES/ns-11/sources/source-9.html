18 /**
19  * A subclass of Latch that may be used for debugging performance issues.  This
20  * latch can be used in place of an exclusive latch or object mutex in order to
21  * see who is waiting for a latch acquisition, how long they're waiting, and
22  * who the previous holder was.  It crudely writes to System.out, but this can
23  * easily be changed to a java.util.Log or EventTrace as desired.  You can
24  * specify a threshold for the wait and previous holder time (nanos).
25  *
26  * Note that this class has not recently been used because it is not
27  * implemented for shared (Btree) latches.  The next time it is used, it should
28  * be integrated with the LatchFactory.
29  */

130     public void acquireExclusive() {
131         if (!debug) {
132             super.acquireExclusive();
133             return;
134         }
135 
136         try {
137             EventTrace.addEvent(new AcquireRequestEvent());
138             if (acquireExclusiveNoWait()) {
139                 EventTrace.addEvent
140                     (new AcquireCompleteEvent(System.nanoTime(), 0));
141                 return;
142             }
143 
144             long startWait = System.nanoTime();
145             super.acquireExclusive();
146             long endWait = System.nanoTime();
147             long ourWaitTime = endWait - startWait;
148             EventTrace.addEvent
149                 (new AcquireCompleteEvent(System.nanoTime(), ourWaitTime));
150             long previousHoldTime = releaseTime - acquireTime;
151             if (previousHoldTime > holdThreshold ||
152                 ourWaitTime > waitThreshold) {
153                 System.out.println
154                     (String.format("%1tT %s waited %,d nanosec for %s\n" +
155                                    " Previous held by %s for %,d nanosec.",
156                                    System.currentTimeMillis(),
157                                    Thread.currentThread(), ourWaitTime,
158                                    getName(), lastThread, previousHoldTime));
159                 EventTrace.dumpEvents(System.out);
160                 EventTrace.disableEvents = false;
161             }
162         } finally {
163             acquireTime = System.nanoTime();
164             lastThread = Thread.currentThread();
165         }
166     }
