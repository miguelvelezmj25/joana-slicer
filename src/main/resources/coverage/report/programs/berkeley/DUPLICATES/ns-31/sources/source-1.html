369     V returnValue(OpResult opResult) {
370         V value;
371         if (opResult.isSuccess()) {
372             value = adapter.entryToValue(key, pkey, data);
373         } else {
374             value = null;
375         }
376         /* Clear entries to save memory. */
377         adapter.clearEntries(key, pkey, data);
378         return value;
379     }
