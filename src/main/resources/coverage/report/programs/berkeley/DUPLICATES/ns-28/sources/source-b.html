33 /**
34  * Common base class for mirror comparisons.  Supports both authenticator and
35  * host verifier implementations.
36  */
37 
38 class SSLMirrorMatcher {
39 
40     /*
41      * The Principal that represents us when in the expected peer's ssl mode.
42      */
43     final private Principal ourPrincipal;
44     final private InstanceLogger logger;
45 
46     /**
47      * Construct an SSLMirrorMatcher
48      *
49      * @param params The instantiation parameters.
50      * @param clientMode set to true if the matcher will be evaluated
51      * as a client that has a server as a peer, or false if it will be
52      * evaluated as a server that has received a connection from a client.
53      * @throws IllegalArgumentException if the instance cannot be created due
54      * to a problem related to the input parameters
55      */
56     public SSLMirrorMatcher(InstanceParams params, boolean clientMode)
57         throws IllegalArgumentException {
58 
59         ourPrincipal = determinePrincipal(params.getContext(), clientMode);
60         if (ourPrincipal == null) {
61             throw new IllegalArgumentException(
62                 "Unable to determine a local principal for comparison " +
63                 "with peer principals");
64         }
65         logger = params.getContext().getLoggerFactory().getLogger(getClass());
66     }
67 
68     /**
69      * Checks whether the SSL session peer's certificate DN matches our own.
70      *
71      * @param sslSession the SSL session that has been established with a peer
72      * @return true if the peer's certificate DN matches ours
73      */
74     public boolean peerMatches(SSLSession sslSession) {
75 
76         if (ourPrincipal == null) {
77             return false;
78         }
79 
80         /*
81          * Get the peer principal, which should also be an X500Principal.
82          * We validate that here.
83          */
84         Principal peerPrincipal = null;
85         try {
86             peerPrincipal = sslSession.getPeerPrincipal();
87         } catch (SSLPeerUnverifiedException pue) {
88             return false;
89         }
90 
91         if (peerPrincipal == null ||
92             ! (peerPrincipal instanceof X500Principal)) {
93             logger.log(
94                 INFO,
95                 "Unable to attempt peer validation - peer Principal is: " +
96                 peerPrincipal);
97             return false;
98         }
99 
100         return ourPrincipal.equals(peerPrincipal);
101     }
102 
103     /**
104      * Attempt to determine the Principal that we take on when connecting
105      * in client or server context based on the ReplicationNetworkConfig.
106      * If we are unable to determine that principal, return null.
107      */
108     private Principal determinePrincipal(
109         InstanceContext context, boolean clientMode)
110         throws IllegalArgumentException {
111 
112         final ReplicationSSLConfig config =
113             (ReplicationSSLConfig) context.getRepNetConfig();
114 
115         /*
116          * Determine what alias would be used.  It is allowable for this to be
117          * null.
118          */
119         String aliasProp = clientMode ?
120             config.getSSLClientKeyAlias() :
121             config.getSSLServerKeyAlias();
122 
123         final KeyStore keyStore = SSLChannelFactory.readKeyStore(context);
124 
125         if (aliasProp == null || aliasProp.isEmpty()) {
126             /* Since we weren't told which one to use, there better be
127              * only one option, or this might behave unexpectedly. */
128             try {
129                 if (keyStore.size() < 1) {
130                     logger.log(INFO, "KeyStore is empty");
131                     return null;
132                 } else if (keyStore.size() > 1) {
133                     logger.log(INFO, "KeyStore has multiple entries but no " +
134                                "alias was specified.  Using the first one " +
135                                "available.");
136                 }
137                 final Enumeration<String> e = keyStore.aliases();
138                 aliasProp = e.nextElement();
139             } catch (KeyStoreException kse) {
140                 throw new IllegalArgumentException(
141                     "Error accessing aliases from the keystore", kse);
142             }
143         }
144 
145         Certificate cert = null;
146         try {
147             cert = keyStore.getCertificate(aliasProp);
148         } catch (KeyStoreException kse) {
149             /* Shouldn't be possible */
150             throw new IllegalArgumentException(
151                 "Error accessing certificate with alias " + aliasProp +
152                 " from the keystore", kse);
153         }
154 
155         if (cert == null) {
156             logger.log(INFO, "No certificate for alias " + aliasProp +
157                        " found in KeyStore");
158             throw new IllegalArgumentException(
159                 "Unable to find a certificate in the keystore");
160         }
161 
162         if (!(cert instanceof X509Certificate)) {
163             logger.log(INFO, "The certificate for alias " + aliasProp +
164                        " is not an X509Certificate.");
165             throw new IllegalArgumentException(
166                 "Unable to find a valid certificate in the keystore");
167         }
168 
169         final X509Certificate x509Cert = (X509Certificate) cert;
170         return x509Cert.getSubjectX500Principal();
171     }
