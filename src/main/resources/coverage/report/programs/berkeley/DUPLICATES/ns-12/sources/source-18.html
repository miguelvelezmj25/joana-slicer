30     /**
31      * Persistent fields. Layout on disk is
32      * (invariant) checksum - 4 bytes
33      * (invariant) entry type - 1 byte
34      * (invariant) entry flags - 1 byte
35      * (invariant) offset of previous log entry - 4 bytes
36      * (invariant) item size (not counting header size) - 4 bytes
37      * (optional) vlsn - 8 bytes
38      *
39      * Flags:
40      * The provisional bit can be set for any log type in the log. It's an
41      * indication to recovery that the entry shouldn't be processed when
42      * rebuilding the tree. See com.sleepycat.je.log.Provisional.java for
43      * the reasons why it's set.
44      *
45      * The replicated bit is set when this particular log entry is
46      * part of the replication stream and contains a VLSN in the header.
47      *
48      * The invisible bit is set when this log entry has been rolled back as
49      * part of replication syncup. The ensuing log entry has not been
50      * checksum-corrected, and to read it, the invisible bit must be cloaked.
51      *
52      * The VLSN_PRESENT bit is set when a VLSN is present for log version 8+,
53      * and is set when the replicated bit is *not* set in the case of a cleaner
54      * migrated LN.  Prior to version 8, the replicated bit alone indicates
55      * that a VLSN is present.  For all versions, if the replicated bit is set
56      * then a VLSN is always present. [#19476]
57      *
58      *                    first version of        migrated LN
59      *                    a replicated LN
60      *                    ---------------         -----------
61      * log version 7-     replicated = true       replicated = false
62      * (JE 4.1 and        vlsn present = false    vlsn present = false
63      *  earlier)          vlsn exists in header   no vlsn in header
64      *
65      * log version 8+     replicated = true       replicated = false
66      * preserve record    vlsn present = true     vlsn present = false
67      * version = false    vlsn exists in header   no vlsn in header
68      *
69      * log version 8+     replicated = true       replicated = false
70      * preserve record    vlsn present = true     vlsn present = true
71      * version = true     vlsn exists in header   vlsn exists in header
72      */

117     /*
118      * Prior to log version 6, a type-specific version was stored in each
119      * entry, and was packed together with the flags in a single byte.
120      *
121      * For version 6, we changed to use a global version (not type specific),
122      * but it was stored in each entry, packed with the flags as in earlier
123      * versions, as well as being stored redundantly in the FileHeader.  The
124      * entry header and file header versions are always the same for all
125      * entries in a file.  We flip the log file to guarantee this, when running
126      * for the first time with an upgraded JE with a new log version.
127      *
128      * For version 7 and above, the version is stored only in the FileHeader,
129      * freeing the space formerly taken by the version in each entry for use
130      * by flag bits.  The version is not stored in each entry; however, the
131      * version is still maintained in this in-memory object for two reasons:
132      *
133      * 1. When reading log files prior to version 6, each entry potentially has
134      *    a different version.
135      * 2. Convenience of access to the version when processing log entries.
136      *
137      * [#16939]
138      */

148      * For reading a log entry.
149      *
150      * @param entryBuffer the buffer containing at least the first
151      * MIN_HEADER_SIZE bytes of the entry header.
152      *
153      * @param logVersion is the log version of the file that contains the given
154      * buffer, and is obtained from the file header.  Note that for the file
155      * header entry itself, UNKNOWN_FILE_HEADER_VERSION may be passed.
156      *
157      * @param lsn is the LSN of the entry, for exception reporting.
158      */
159     public LogEntryHeader(ByteBuffer entryBuffer, int logVersion, long lsn)
160         throws ChecksumException {
161 
162         assert logVersion == LogEntryType.UNKNOWN_FILE_HEADER_VERSION ||
163             (logVersion >= LogEntryType.FIRST_LOG_VERSION &&
164              logVersion <= LogEntryType.LOG_VERSION) : logVersion;
165 
166         checksumVal = LogUtils.readUnsignedInt(entryBuffer);
167         entryType = entryBuffer.get();
168         if (!LogEntryType.isValidType(entryType)) {
169             throw new ChecksumException(
170                 "Invalid log entry type: " + entryType +
171                 " lsn=" + DbLsn.getNoFormatString(lsn) +
172                 " bufPosition=" + entryBuffer.position() +
173                 " bufRemaining=" + entryBuffer.remaining());
174         }
175 
176         if (entryType == FILE_HEADER_TYPE_NUM) {
177             /* Actual version will be set by setFileHeaderVersion. */
178             entryVersion = LogEntryType.UNKNOWN_FILE_HEADER_VERSION;
179             /* Discard flags byte: none are allowed for the file header. */
180             entryBuffer.get();
181             initFlags(0);
182         } else {
183             if (logVersion == LogEntryType.UNKNOWN_FILE_HEADER_VERSION ) {
184                 /*
185                  * If we are reading a log header the type should be
186                  * FILE_HEADER_TYPE_NUM.
187                  */
188                 throw new ChecksumException(
189                     "Wrong entry type for header: " + entryType +
190                     " lsn=" + DbLsn.getNoFormatString(lsn) +
191                     " bufPosition=" + entryBuffer.position() +
192                     " bufRemaining=" + entryBuffer.remaining());
193             } else if (logVersion <= 6) {
194                 /* Before version 7, flags and version were packed together. */
195                 entryVersion = entryBuffer.get();
196                 initFlags(entryVersion & VERSION_6_FLAGS);
197                 entryVersion &= IGNORE_VERSION_6_FLAGS;
198                 /* For log version 6, the entry version should always match. */
199                 assert (logVersion == 6) ? (entryVersion == 6) : true;
200             } else {
201                 /* For log version 7+, only flags are stored in the entry. */
202                 entryVersion = logVersion;
203                 initFlags(entryBuffer.get());
204             }
205         }
206         prevOffset = LogUtils.readUnsignedInt(entryBuffer);
207         itemSize = LogUtils.readInt(entryBuffer);
208         if (itemSize < 0) {
209             throw new ChecksumException(
210                 "Invalid log entry size: " + itemSize +
211                 " lsn=" + DbLsn.getNoFormatString(lsn) +
212                 " bufPosition=" + entryBuffer.position() +
213                 " bufRemaining=" + entryBuffer.remaining());
214         }
215     }

272     private void initFlags(int entryFlags) {
273         if ((entryFlags & PROVISIONAL_ALWAYS_MASK) != 0) {
274             provisional = Provisional.YES;
275         } else if ((entryFlags & PROVISIONAL_BEFORE_CKPT_END_MASK) != 0) {
276             provisional = Provisional.BEFORE_CKPT_END;
277         } else {
278             provisional = Provisional.NO;
279         }
280         replicated = ((entryFlags & REPLICATED_MASK) != 0);
281         invisible = ((entryFlags & INVISIBLE) != 0);
282         vlsnPresent = ((entryFlags & VLSN_PRESENT) != 0) || replicated;

374     /**
375      * Serialize this object into the buffer and leave the buffer positioned in
376      * the right place to write the following item.  The checksum, prevEntry,
377      * and vlsn values will filled in later on.
378      *
379      * public for unit tests.
380      */
381     public void writeToLog(ByteBuffer entryBuffer) {
382 
383         /* Skip over the checksumVal, proceed to the entry type. */
384         entryBuffer.position(ENTRYTYPE_OFFSET);
385         entryBuffer.put(entryType);
386 
387         /* Flags */
388         byte flags = 0;
389         if (provisional == Provisional.YES) {
390             flags |= PROVISIONAL_ALWAYS_MASK;
391         } else if (provisional == Provisional.BEFORE_CKPT_END) {
392             flags |= PROVISIONAL_BEFORE_CKPT_END_MASK;
393         }
394         if (replicated) {
395             flags |= REPLICATED_MASK;
396         }
397         if (vlsnPresent) {
398             flags |= VLSN_PRESENT;
399         }
400         entryBuffer.put(flags);
401 
402         /*
403          * Leave room for the prev offset, which must be added under
404          * the log write latch. Proceed to write the item size.
405          */
406         entryBuffer.position(ITEMSIZE_OFFSET);
407         LogUtils.writeInt(entryBuffer, itemSize);
408 
409         /*
410          * Leave room for a VLSN if needed, must also be generated
411          * under the log write latch.
412          */
413         if (vlsnPresent) {
414             entryBuffer.position(entryBuffer.position() + VLSN.LOG_SIZE);
415         }
416     }

418     /**
419      * Add those parts of the header that must be calculated later to the
420      * entryBuffer, and also assign the fields in this class.
421      * That's
422      * - the prev offset, which must be done within the log write latch to
423      *   be sure what that lsn is
424      * - the VLSN, for the same reason
425      * - the checksumVal, which must be added last, after all other
426      *   fields are marshalled.
427      * (public for unit tests)
428      */
429     public ByteBuffer addPostMarshallingInfo(ByteBuffer entryBuffer,
430                                              long lastOffset,
431                                              VLSN vlsn) {
432 
433         /* Add the prev pointer */
434         prevOffset = lastOffset;
435         entryBuffer.position(PREV_OFFSET);
436         LogUtils.writeUnsignedInt(entryBuffer, prevOffset);
437 
438         if (vlsn != null) {
439             this.vlsn = vlsn;
440             entryBuffer.position(VLSN_OFFSET);
441 
442             vlsn.writeToLog(entryBuffer);
443         }
444 
445         /*
446          * Now calculate the checksumVal and write it into the buffer.  Be sure
447          * to set the field in this instance, for use later when printing or
448          * debugging the header.
449          */
450         Checksum checksum = Adler32.makeChecksum();
451         checksum.update(entryBuffer.array(),
452                         entryBuffer.arrayOffset() + CHECKSUM_BYTES,
453                         entryBuffer.limit() - CHECKSUM_BYTES);
454         entryBuffer.position(0);
455         checksumVal = checksum.getValue();
456         LogUtils.writeUnsignedInt(entryBuffer, checksumVal);
457 
458         /* Leave this buffer ready for copying into another buffer. */
459         entryBuffer.position(0);
460 
461         return entryBuffer;
462     }

513     /**
514      * For use in special case where commits are transformed to aborts because
515      * of i/o errors during a logBuffer flush. See [11271].
516      * Assumes that the entryBuffer is positioned at the start of the item.
517      * Return with the entryBuffer positioned to the end of the log entry.
518      */
519     void convertCommitToAbort(ByteBuffer entryBuffer) {
520         assert (entryType == LogEntryType.LOG_TXN_COMMIT.getTypeNum());
521 
522         /* Remember the start of the entry item. */
523         int itemStart = entryBuffer.position();
524 
525         /* Back up to where the type is stored and change the type. */
526         int entryTypePosition =
527             itemStart - (getSize() - ENTRYTYPE_OFFSET);
528         entryBuffer.position(entryTypePosition);
529         entryBuffer.put(LogEntryType.LOG_TXN_ABORT.getTypeNum());
530 
531         /*
532          * Recalculate the checksum. This byte buffer could be large,
533          * so don't just turn the whole buffer into an array to pass
534          * into the checksum object.
535          */
536         Checksum checksum = Adler32.makeChecksum();
537         int checksumSize = itemSize + (getSize() - CHECKSUM_BYTES);
538         checksum.update(entryBuffer.array(),
539                         entryTypePosition + entryBuffer.arrayOffset(),
540                         checksumSize);
541         entryBuffer.position(itemStart - getSize());
542         checksumVal = checksum.getValue();
543         LogUtils.writeUnsignedInt(entryBuffer, checksumVal);
544     }
