15 import java.util.concurrent.CountDownLatch;
16 import java.util.concurrent.TimeUnit;

18 import com.sleepycat.je.rep.elections.Proposer.Proposal;

20 /**
21  * Ensures that a VLSN is not advanced at this node while an election is in
22  * progress. Note that this is difficult, if not impossible to achieve
23  * efficiently in a distributed environment across the entire group, when
24  * communications may not always be reliable. So, the implementation really
25  * represents a good faith effort to freeze the VLSN. JE HA itself should be
26  * able to make forward progress in the event of such a failure.
27  *
28  * The class coordinates three threads: the acceptor, the learner, and the
29  * replay thread. There is exactly one instance of each thread per replication
30  * node, so it coordinates the activity of these three threads.
31  *
32  * The typical serialized sequence of calls is therefore:
33  *
34  * latch.freeze() -- invoked in response to a Promise by an Acceptor
35  * latch.vlsnEvent() -- one or more of them in response to ongoing election
36  * latch.awaitThaw() -- by the replica thread waiting for the freeze to lift
37  *
38  * Both vlsnEvent() and awaitThaw() are NOPs in the absence of a freeze.
39  *
40  * @see <a href="https://sleepycat.oracle.com/trac/wiki/ElectionsImplementation#FreezingVLSNs">Freezing VLSNs</a>
41  */
42 public class CommitFreezeLatch {

45     private Proposal proposal = null;

55     private long freezeEnd = 0;

80     /**
81      * Initiates or extends a freeze on a VLSN in response to a new election
82      * that is in progress. It's invoked by the Acceptor thread.
83      *
84      * @param freezeProposal identifies the election that is provoking the freeze
85      */
86     public synchronized void freeze(Proposal freezeProposal) {
87         if ((proposal != null) && (freezeProposal.compareTo(proposal) <= 0)) {
88             // Older proposal ignore it.
89             return;
90         }
91         if (latch != null) {
92             /* Enable waiters who will reacquire the new latch below. */
93             latch.countDown();
94         }
95         latch = new CountDownLatch(1);
96         proposal = freezeProposal;
97         freezeEnd = System.currentTimeMillis() + timeOut;
98         return;
99     }
100 
101     /**
102      * Invoked by the Learner thread whenever it receives an election result.
103      * The freeze on the VLSN is only lifted if the proposal associated with
104      * the event is current, that is, it represents a proposal that is newer
105      * than the one used to establish the freeze.
106      *
107      * @param listenerProposal identifies the election that just concluded
108      */
109     public synchronized void vlsnEvent(Proposal listenerProposal) {
110         if (proposal == null) {
111             // No VLSN to unfreeze
112             return;
113         }
114         if (listenerProposal.compareTo(this.proposal) >= 0) {
115             latch.countDown();
116         }
117     }
118 
119     /**
120      * Clears the latch freeing any waiters.
121      */
122     public synchronized void clearLatch() {
123         if (latch != null) {
124             latch.countDown();
125         }
126         latch = null;
127         proposal = null;
128         freezeEnd = 0;
129     }

131     /**
132      * Used to wait for an event that unfreezes the VLSN. In our case this
133      * event is a message to the Learner agent announcing the result of an
134      * election. Note that the latch must be re-initialized after a return from
135      * this await method.
136      *
137      * This method is invoked by the Replay thread. Completion of an awaitThaw
138      * always results in the freeze being lifted.
139      *
140      * @return true if the await was satisfied due to completion of an
141      * election, false if no freeze was in effect, or the latch was timed out.
142      *
143      * @throws InterruptedException
144      */
145     public boolean awaitThaw()
146         throws InterruptedException {
147 
148         CountDownLatch awaitLatch;
149         long awaitTimeout;
150 
151         synchronized (this) {
152             /* Copy out the values of interest  */
153             awaitLatch = latch;
154             if (awaitLatch == null) {
155                 return false;
156             }
157             awaitTimeout = this.freezeEnd - System.currentTimeMillis();
158         }
159         freezeCount++;
160 
161         boolean done = awaitLatch.await(awaitTimeout, TimeUnit.MILLISECONDS);
162 
163         synchronized (this) {
164             if (done) {
165                 awaitElectionCount++;
166                 clearLatch();
167                 return true;
168             }
169             if (this.freezeEnd - System.currentTimeMillis() <= 0) {
170                 awaitTimeoutCount++;
171                 /* freeze end was not extended, election completed. */
172                 clearLatch();
173                 return false;
174             }
175         }
176         /* Re-acquire the new latch and wait for the extended timeout. */
177         return awaitThaw();
178     }
