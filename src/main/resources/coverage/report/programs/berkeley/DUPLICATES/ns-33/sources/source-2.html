44     public final void initializeProxy(Collection<E> collection) {
45         elements = (E[]) new Object[collection.size()];
46         int i = 0;
47         for (E element : collection) {
48             elements[i] = element;
49             i += 1;
50         }
51     }

53     public final Collection<E> convertProxy() {
54         Collection<E> collection = newInstance(elements.length);
55         for (E element : elements) {

103     static Object[] getElements(RawObject collection) {
104         Object value = null;
105         while (value == null && collection != null) {
106             Map<String, Object> values = collection.getValues();
107             if (values != null) {
108                 value = values.get("elements");
109                 if (value == null) {
110                     collection = collection.getSuper();
111                 }
112             }
113         }
114         if (value == null || !(value instanceof RawObject)) {
115             throw new IllegalStateException
116                 ("Collection proxy for a secondary key field must " +
117                  "contain a field named 'elements'");
118         }
119         RawObject rawObj = (RawObject) value;
120         Format format = (Format) rawObj.getType();
121         if (!format.isArray() ||
122             format.getComponentType().getId() != Format.ID_OBJECT) {
123             throw new IllegalStateException
124                 ("Collection proxy 'elements' field must be an Object array");
125         }
126         return rawObj.getElements();
127     }

129     static void setElements(RawObject collection, Object[] elements) {
130         RawObject value = null;
131         while (value == null && collection != null) {
132             Map<String, Object> values = collection.getValues();
133             if (values != null) {
134                 value = (RawObject) values.get("elements");
135                 if (value != null) {
136                     values.put("elements",
137                                new RawObject(value.getType(), elements));
138                 } else {
139                     collection = collection.getSuper();
140                 }
141             }
142         }
143         if (value == null) {
144             throw DbCompat.unexpectedState();
145         }
146     }
