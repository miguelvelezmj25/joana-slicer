23  * An iterator to iterate over the free ports on an interface.
24  */
25 public class FreePortLocator {

40     /**
41      * Constructor identifying the interface and the port range within which
42      * to look for free ports. The port range specified by the arguments
43      * must be < 32768, that is, it should be outside the dynamic port range
44      * that is typically configured on most machines.
45      *
46      * @see <a href="https://sleepycat.oracle.com/trac/wiki/JEKV/UnitTest#Avoidingproblemswithanonymousports.html">Anonymous ports</a>
47      * for details regarding port configuration for tests.
48      */
49     public FreePortLocator(String hostname, int portStart, int portEnd) {
50         super();
51         assert portStart < portEnd;
52 
53         if ((portStart > 0x7fff) || (portEnd > 0x7fff)) {
54             throw new IllegalArgumentException
55                 ("Invalid port range:" + portStart + " - " + portEnd + ". " +
56                  "The port range must not extend past:" + 0x7fff +
57                  " since the allocated ports could then overlap with " +
58                  "dynamically assigned ports used by other services. ");
59         }
60 
61         this.hostname = hostname;
62         this.portStart = portStart;
63         this.portEnd = portEnd;
64         currPort = portStart;
65     }

75     /**
76      * Returns the next free port. Note that it's possible that on a busy
77      * machine another process may grab the "free" port before it's actually
78      * used.
79      *
80      * There is somewhat AIsh aspect to the code below. In general it tries to
81      * be very conservative, using different techniques so that it works
82      * reasonably well on Linux, Mac OS and Windows.
83      *
84      * Note: The use of setReuseAddress after a bind operation may look
85      * dubious, since it runs counter to the API doc, but it helps based on
86      * actual tests. It's also the idiom used by Apache Camel to find a
87      * free port. It, at least, can't hurt.
88      */
89     public int next() {
90         while (++currPort < portEnd) {
91 
92             /* Try connecting to the port to see if somebody is listening. */
93             Socket s = null;
94             try {
95                 s = new Socket(hostname, currPort);
96                 /* Somebody is listening on the port. */
97                 if (debug) {
98                     System.err.println(
99                         "FreePortLocator: " + currPort + " busy - socket");
100                     Thread.dumpStack();
101                 }
102                 continue;
103             } catch (IOException e) {
104                 /* Nobody is listening, continue with other tests. */
105             } finally {
106                 if (s != null){
107                     try {
108                         s.close();
109                     } catch (IOException e) {
110                         /* Unexpected, something's wrong, ignore the port. */
111                         if (debug) {
112                             System.err.println(
113                                 "FreePortLocator: " + currPort +
114                                 " busy - socket close: " + e);
115                             e.printStackTrace();
116                         }
117                        continue;
118                     }
119                 }
120             }
121 
122             /* Try without a hostname */
123             ServerSocket ss = null;
124             DatagramSocket ds = null;
125             try {
126                 ss = new ServerSocket(currPort);
127                 ss.setReuseAddress(true);
128                 ds = new DatagramSocket(currPort);
129                 ds.setReuseAddress(true);
130             } catch (IOException e) {
131                 if (debug) {
132                     System.err.println(
133                         "FreePortLocator: " + currPort +
134                         " busy - server, datagram: " + e);
135                     e.printStackTrace();
136                 }
137                 continue;
138             } finally {
139                 if (ds != null) {
140                     ds.close();
141                 }
142 
143                 if (ss != null) {
144                     try {
145                         ss.close();
146                     } catch (IOException e) {
147                         if (debug) {
148                             System.err.println(
149                                 "FreePortLocator: " + currPort +
150                                 " busy - server close: " + e);
151                             e.printStackTrace();
152                         }
153                         continue;
154                     }
155                 }
156             }
157 
158             ss = null;
159             ds = null;
160 
161             /* try with a hostname */
162            final InetSocketAddress sa =
163                new InetSocketAddress(hostname, currPort);
164             try {
165                 ss = new ServerSocket();
166                 ss.setReuseAddress(true);
167                 ss.bind(sa);
168 
169                 ds = new DatagramSocket(sa);
170                 ds.setReuseAddress(true);
171             } catch (IOException e) {
172                 if (debug) {
173                     System.err.println(
174                         "FreePortLocator: " + currPort +
175                         " busy - server, datagram hostname: " + e);
176                     e.printStackTrace();
177                 }
178                 continue;
179             } finally {
180                 if (ds != null) {
181                     ds.close();
182                 }
183 
184                 if (ss != null) {
185                     try {
186                         ss.close();
187                     } catch (IOException e) {
188                         if (debug) {
189                             System.err.println(
190                                 "FreePortLocator: " + currPort +
191                                 " busy - server hostname close: " + e);
192                             e.printStackTrace();
193                         }
194                         continue;
195                     }
196                 }
197             }
198 
199             /* Survived port test gauntlet, return it. */
200             if (debug) {
201                 System.err.println(
202                     "FreePortLocator: " + currPort + " free");
203             }
204             return currPort;
205         }
206 
207         throw new IllegalStateException
208             ("No more ports available in the range: " +
209              portStart + " - " + portEnd);
210     }
