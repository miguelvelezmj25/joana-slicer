114     /**
115      * Parse the command line parameters.
116      *
117      * @param argv Input command line parameters.
118      */
119     private void parseArgs(String argv[]) {
120         int argc = 0;
121         int nArgs = argv.length;
122         String netPropsName = null;
123 
124         if (nArgs == 0) {
125             printUsage(null);
126         }
127 
128         while (argc < nArgs) {
129             String thisArg = argv[argc++];
130             if (thisArg.equals("-nodeName")) {
131                 if (argc < nArgs) {
132                     nodeName = argv[argc++];
133                 } else {
134                     printUsage("-nodeName requires an argument");
135                 }
136             } else if (thisArg.equals("-groupName")) {
137                 if (argc < nArgs) {
138                     groupName = argv[argc++];
139                 } else {
140                     printUsage("-groupName requires an argument");
141                 }
142             } else if (thisArg.equals("-nodeHost")) {
143                 if (argc < nArgs) {
144                     StringTokenizer st =
145                         new StringTokenizer(argv[argc++], ":");
146                     if (st.countTokens() != 2) {
147                         printUsage("Argument for -nodeHost is not valid.");
148                     }
149                     try {
150                         socketAddress = new InetSocketAddress
151                             (st.nextToken(), Integer.parseInt(st.nextToken()));
152                     } catch (NumberFormatException e) {
153                         printUsage("the port of -nodeHost is not valid");
154                     }
155                 } else {
156                     printUsage("-nodeHost requires an argument");
157                 }
158             } else if (thisArg.equals("-socketTimeout")) {
159                 if (argc < nArgs) {
160                     try {
161                         socketTimeout = Integer.parseInt(argv[argc++]);
162                     } catch (NumberFormatException e) {
163                         printUsage("Argument for -socketTimeout is not valid");
164                     }
165                 } else {
166                     printUsage("-socketTimeout requires an argument");
167                 }
168             } else if (thisArg.equals("-netProps")) {
169                 if (argc < nArgs) {
170                     netPropsName = argv[argc++];
171                 } else {
172                     printUsage("-netProps requires an argument");
173                 }
174             } else {
175                 printUsage(thisArg + " is not a valid argument");
176             }
177         }
178 
179         if (socketTimeout <= 0) {
180             printUsage("-socketTimeout requires a positive integer number");
181         }
182 
183         ReplicationNetworkConfig repNetConfig =
184             ReplicationNetworkConfig.createDefault();
185         if (netPropsName != null) {
186             try {
187                 repNetConfig =
188                     ReplicationNetworkConfig.create(new File(netPropsName));
189             } catch (FileNotFoundException fnfe) {
190                 printUsage("The net properties file " + netPropsName +
191                            " does not exist: " + fnfe.getMessage());
192             } catch (IllegalArgumentException iae) {
193                 printUsage("The net properties file " + netPropsName +
194                            " is not valid: " + iae.getMessage());
195             }
196         }
197 
198         if (nodeName == null || groupName == null || socketAddress == null) {
199             printUsage("Node name, group name and the node host port are " +
200                        "mandatory arguments, please configure.");
201         }
202 
203         this.channelFactory = initializeFactory(repNetConfig, nodeName);
204     }

299     /* Get the state of the specified node. */
300     public NodeState getNodeState()
301         throws IOException, ServiceConnectFailedException {
302 
303         BinaryNodeStateProtocol protocol =
304             new BinaryNodeStateProtocol(NameIdPair.NOCHECK, null);
305         DataChannel channel = null;
306 
307         try {
308             /* Build the connection. */
309             channel = channelFactory.connect(
310                 socketAddress,
311                 new ConnectOptions().
312                 setTcpNoDelay(true).
313                 setOpenTimeout(socketTimeout).
314                 setReadTimeout(socketTimeout));
315             ServiceDispatcher.doServiceHandshake
316                 (channel, BinaryNodeStateService.SERVICE_NAME);
317 
318             /* Send a NodeState request to the node. */
319             protocol.write
320                 (protocol.new BinaryNodeStateRequest(nodeName, groupName),
321                  channel);
322 
323             /* Get the response and return the NodeState. */
324             BinaryNodeStateResponse response =
325                 protocol.read(channel, BinaryNodeStateResponse.class);
326 
327             return response.convertToNodeState();
328         } finally {
329             if (channel != null) {
330                 channel.close();
331             }
332         }
333     }
