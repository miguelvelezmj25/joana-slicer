191      * The keySet, entrySet and valueSet are created during Map construction
192      * rather than lazily when requested (as done with the java.util.Map
193      * implementations).  This is done to avoid synchronization every time they
194      * are requested.  Since they are requested often but a StoredMap is
195      * created infrequently, this gives the best performance.  The additional
196      * views are small objects and are cheap to construct.
197      */
198     private void initView() {
199 
200         /* entrySet */
201         if (areKeyRangesAllowed()) {
202             entrySet = new StoredSortedEntrySet<K, V>(view);
203         } else {
204             entrySet = new StoredEntrySet<K, V>(view);
205         }
206 
207         /* keySet */
208         DataView newView = view.keySetView();
209         if (areKeyRangesAllowed()) {
210             keySet = new StoredSortedKeySet<K>(newView);
211         } else {
212             keySet = new StoredKeySet<K>(newView);
213         }
214 
215         /* valueSet */
216         newView = view.valueSetView();
217         if (areKeyRangesAllowed() && newView.canDeriveKeyFromValue()) {
218             valueSet = new StoredSortedValueSet<V>(newView);
219         } else {
220             valueSet = new StoredValueSet<V>(newView);
221         }
222     }

377     public V putIfAbsent(K key, V value) {
378         DataCursor cursor = null;
379         boolean doAutoCommit = beginAutoCommit();
380         try {
381             cursor = new DataCursor(view, true);
382             V oldValue;
383             while (true) {
384                 OperationStatus status =
385                     cursor.putNoOverwrite(key, value, false /*useCurrentKey*/);
386                 if (status == OperationStatus.SUCCESS) {
387                     /* We inserted the key.  Return null.  */
388                     oldValue = null;
389                     break;
390                 } else {
391                     status = cursor.getSearchKey(key, null /*value*/,
392                                                  false /*lockForWrite*/);
393                     if (status == OperationStatus.SUCCESS) {
394                         /* The key is present. Return the current value. */
395                         oldValue = (V) cursor.getCurrentValue();
396                         break;
397                     } else {
398 
399                         /*
400                          * If Serializable isolation is not configured, another
401                          * thread can delete the record after our attempt to
402                          * insert it failed above.  Loop back and try again.
403                          */
404                         continue;
405                     }
406                 }
407             }
408             closeCursor(cursor);
409             commitAutoCommit(doAutoCommit);
410             return oldValue;
411         } catch (Exception e) {
412             closeCursor(cursor);
413             throw handleException(e, doAutoCommit);
