67     /**
68      * Returns the max amount of memory in the heap available, using an
69      * approach that depends on the JVM vendor, OS, etc.
70      *
71      * May return Long.MAX_VALUE if there is no inherent limit.
72      */
73     public static long getRuntimeMaxMemory() {
74 
75         /* Runtime.maxMemory is unreliable on MacOS Java 1.4.2. */
76         if ("Mac OS X".equals(System.getProperty("os.name"))) {
77             final String jvmVersion = System.getProperty("java.version");
78             if (jvmVersion != null && jvmVersion.startsWith("1.4.2")) {
79                 return Long.MAX_VALUE; /* Undetermined heap size. */
80             }
81         }
82 
83         /*
84          * Runtime.maxMemory is unreliable on Zing. Call
85          * MemoryMXBean.getApplicationObjectHeapUsableMemory instead.
86          */
87         if (ZING_JVM) {
88             try {
89                 final Class<?> factoryClass =
90                     Class.forName(ZING_MANAGEMENT_FACTORY_CLASS);
91 
92                 final Method getBeanMethod =
93                     factoryClass.getMethod("getMemoryMXBean");
94 
95                 final Object memoryBean = getBeanMethod.invoke(null);
96                 final Class<?> beanClass = memoryBean.getClass();
97 
98                 final Method getMaxMemoryMethod = beanClass.getMethod(
99                     "getApplicationObjectHeapUsableMemory");
100 
101                 return (Long) getMaxMemoryMethod.invoke(memoryBean);
102 
103             } catch (Exception e) {
104                 throw new IllegalStateException(ZING_ACCESS_ERROR, e);
105             }
106         }
107 
108         /* Standard approach. */
109         return Runtime.getRuntime().maxMemory();
110     }

112     /**
113      * Returns the size of the System Zing Memory pool. This is the max memory
114      * for all running Zing JVMs.
115      */
116     public static long getSystemZingMemorySize() {
117         try {
118             if (!ZING_JVM) {
119                 throw new IllegalStateException("Only allowed under Zing");
120             }
121 
122             final Class<?> factoryClass =
123                 Class.forName(ZING_MANAGEMENT_FACTORY_CLASS);
124 
125             final Method getPoolsMethod =
126                 factoryClass.getMethod("getMemoryPoolMXBeans");
127 
128             final List<?> pools =
129                 (List<?>) getPoolsMethod.invoke(null);
130 
131             final Class<?> poolClass = pools.get(0).getClass();
132             final Method getNameMethod = poolClass.getMethod("getName");
133             final Method getSizeMethod = poolClass.getMethod("getCurrentSize");
134 
135             for (Object pool : pools) {
136                 if ("System Zing Memory".equals(getNameMethod.invoke(pool))) {
137                     return (Long) getSizeMethod.invoke(pool);
138                 }
139             }
140 
141             throw new IllegalStateException(
142                 "System Zing Memory pool not found");
143 
144         } catch (Exception e) {
145             throw new IllegalStateException(ZING_ACCESS_ERROR, e);
146         }
147     }
