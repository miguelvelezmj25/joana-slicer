21 /**
22  * Redirects logging messages to the owning environment's application
23  * configured handler, if one was specified through
24  * EnvironmentConfig.setLoggingHandler(). Handlers for JE logging can be
25  * configured through EnvironmentConfig, to support handlers which:
26  * - require a constructor with arguments
27  * - is specific to this environment, and multiple environments exist in the
28  *   same process.
29  */
30 public class ConfiguredRedirectHandler extends Handler {
31 
32     public ConfiguredRedirectHandler() {
33         /* No need to call super, this handler is not truly publishing. */
34     }
35 
36     @Override
37     public void publish(LogRecord record) {
38         Handler h = getEnvSpecificConfiguredHandler();
39         if ((h != null) && (h.isLoggable(record))) {
40             h.publish(record);
41         }
42     }
43 
44     private Handler getEnvSpecificConfiguredHandler() {
45         EnvironmentImpl envImpl =
46             LoggerUtils.envMap.get(Thread.currentThread());
47 
48         /*
49          * Prefer to lose logging output, rather than risk a
50          * NullPointerException if the caller forgets to set and release the
51          * environmentImpl.
52          */
53         if (envImpl == null) {
54             return null;
55         }
56 
57         return envImpl.getConfiguredHandler();
58     }
59 
60     @Override
61     public void close()
62         throws SecurityException {
63         Handler h = getEnvSpecificConfiguredHandler();
64         if (h != null) {
65             h.close();
66         }
67     }
68 
69     @Override
70     public void flush() {
71         Handler h = getEnvSpecificConfiguredHandler();
72         if (h != null) {
73             h.flush();
74         }
75     }
76 }
