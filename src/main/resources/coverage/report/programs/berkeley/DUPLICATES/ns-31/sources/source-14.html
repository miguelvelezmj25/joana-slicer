54 /**
55  * The EntityIndex returned by SecondaryIndex.subIndex.  A SubIndex, in JE
56  * internal terms, is a duplicates btree for a single key in the main btree.
57  * From the user's viewpoint, the keys are primary keys.  This class implements
58  * that viewpoint.  In general, getSearchBoth and getSearchBothRange are used
59  * where in a normal index getSearchKey and getSearchRange would be used.  The
60  * main tree key is always implied, not passed as a parameter.
61  *
62  * @author Mark Hayes
63  */
64 class SubIndex<PK, E> implements EntityIndex<PK, E> {

238     private OpResult deleteInternal(Transaction txn,
239                                     PK key,
240                                     OpWriteOptions options)
241         throws DatabaseException {
242 
243         DatabaseEntry pkeyEntry = new DatabaseEntry();
244         DatabaseEntry dataEntry = BasicIndex.NO_RETURN_ENTRY;
245         pkeyBinding.objectToEntry(key, pkeyEntry);
246 
247         boolean autoCommit = false;
248         Environment env = db.getEnvironment();
249         if (transactional &&
250             txn == null &&
251             DbCompat.getThreadTransaction(env) == null) {
252             txn = env.beginTransaction
253                 (null, secIndex.getAutoCommitTransactionConfig());
254             autoCommit = true;
255         }
256 
257         boolean failed = true;
258         CursorConfig cursorConfig = null;
259         if (concurrentDB) {
260             cursorConfig = new CursorConfig();
261             DbCompat.setWriteCursor(cursorConfig, true);
262         } 
263         SecondaryCursor cursor = db.openSecondaryCursor(txn, cursorConfig);
264         try {
265             /* <!-- begin JE only --> */
266             if (DbCompat.IS_JE) {
267                 ReadOptions readOptions;
268                 if (options.jeOptions != null &&
269                     options.jeOptions.getCacheMode() != null) {
270                     readOptions = new ReadOptions();
271                     readOptions.setLockMode(LockMode.RMW);
272                     readOptions.setCacheMode(options.jeOptions.getCacheMode());
273                 } else {
274                     readOptions = LockMode.RMW.toReadOptions();
275                 }
276                 OperationResult result = cursor.get(
277                     keyEntry, pkeyEntry, dataEntry, Get.SEARCH_BOTH,
278                     readOptions);
279                 if (result != null) {
280                     result = cursor.delete(options.jeOptions);
281                 }
282                 failed = false;
283                 return OpResult.make(result);
284             }
285             /* <!-- end JE only --> */
286             OperationStatus status = cursor.getSearchBoth
287                 (keyEntry, pkeyEntry, dataEntry,
288                  locking ? LockMode.RMW : null);
289             if (status == OperationStatus.SUCCESS) {
290                 status = cursor.delete();
291             }
292             failed = false;
293             return OpResult.make(status);
294         } finally {
295             cursor.close();
296             if (autoCommit) {
297                 if (failed) {
298                     txn.abort();
299                 } else {
300                     txn.commit();
301                 }
302             }
303         }
304     }

375     private <V> EntityCursor<V> cursor(Transaction txn,
376                                        PK fromKey,
377                                        boolean fromInclusive,
378                                        PK toKey,
379                                        boolean toInclusive,
380                                        ValueAdapter<V> adapter,
381                                        CursorConfig config)
382         throws DatabaseException {
383 
384         DatabaseEntry fromEntry = null;

386             fromEntry = new DatabaseEntry();

388         }
389         DatabaseEntry toEntry = null;

392             pkeyBinding.objectToEntry(toKey, toEntry);
393         }
394         KeyRange pkeyRange = emptyPKeyRange.subRange
395             (fromEntry, fromInclusive, toEntry, toInclusive);
396         return cursor(txn, pkeyRange, adapter, config);
