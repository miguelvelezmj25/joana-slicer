18 /**
19  * TestHook is used to induce testing behavior that can't be provoked
20  * externally.  For example, unit tests may use hooks to throw IOExceptions, or
21  * to cause waiting behavior.
22  *
23  * To use this, a unit test should implement TestHook with a class that
24  * overrides the desired method. The desired code will have a method that
25  * allows the unit test to specify a hook, and will execute the hook if it is
26  * non-null.  This should be done within an assert like so:
27  *
28  *    assert TestHookExecute(myTestHook);
29  *
30  * See Tree.java for examples.
31  */
32 public interface TestHook<T> {
33 
34     public void hookSetup();
35 
36     public void doIOHook()
37         throws IOException;
38 
39     public void doHook();
40 
41     public void doHook(T obj);
42 
43     public T getHookValue();
44 }
