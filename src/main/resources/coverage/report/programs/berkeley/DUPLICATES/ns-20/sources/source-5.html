64 /**
65  * Provides a lightweight mechanism to track the current master node and the
66  * members of the replication group. The information provided by the monitor
67  * can be used to route update requests to the node that is currently the
68  * master and distribute read requests across the other members of the group.
69  * <p>
70  * The Monitor is typically run on a machine that participates in load
71  * balancing, request routing or is simply serving as a basis for application
72  * level monitoring and does not have a replicated environment. To avoid
73  * creating a single point of failure, an application may need to create
74  * multiple monitor instances, with each monitor running on a distinct machine.
75  * <p>
76  * Applications with direct access to a {@link
77  * com.sleepycat.je.rep.ReplicatedEnvironment ReplicatedEnvironment} can use
78  * its {@link <a href="{@docRoot}/../ReplicationGuide/replicawrites.html">
79  * synchronous and asynchronous mechanisms</a>} for determining the master node
80  * and group composition changes. The Monitor class is not needed by such
81  * applications.
82  * <p>
83  * The Monitor generally learns about changes to group status through events
84  * issued by replication group members. In addition, the Monitor maintains a
85  * daemon thread which periodically pings members of the group so that the
86  * Monitor can proactively discover group status changes that occur when it is
87  * down or has lost network connectivity.
88  * <p>
89  * The following code excerpt illustrates the typical code sequence used to
90  * initiate a Monitor. Exception handling has been omitted to simplify the
91  * example.
92  *
93  * <pre class="code">
94  * MonitorConfig monConfig = new MonitorConfig();
95  * monConfig.setGroupName("PlanetaryRepGroup");
96  * monConfig.setNodeName("mon1");
97  * monConfig.setNodeHostPort("monhost1.acme.com:7000");
98  * monConfig.setHelperHosts("mars.acme.com:5000,jupiter.acme.com:5000");
99  *
100  * Monitor monitor = new Monitor(monConfig);
101  *
102  * // If the monitor has not been registered as a member of the group,
103  * // register it now. register() returns the current node that is the
104  * // master.
105  *
106  * ReplicationNode currentMaster = monitor.register();
107  *
108  * // Start up the listener, so that it can be used to track changes
109  * // in the master node, or group composition. It can also be used to help
110  * // determine the electable nodes that are currently active and participating
111  * // in the replication group.
112  * monitor.startListener(new MyChangeListener());
113  * </pre>
114  *
115  * @see MonitorChangeListener
116  * @see <a href="{@docRoot}../ReplicationGuide/monitors.html">Writing Monitor
117  * Nodes</a>
118  * @see <a
119  * href="{@docRoot}../examples/je/rep/quote/package-summary.html">je.rep.quote
120  * Examples</a>
121  */

322      * Starts the listener so it's actively listening for election results and
323      * broadcasts of replication group changes.
324      * <p>
325      * {@link Monitor#register} should be called before starting the listener.
326      * If the monitor has not been registered, it will not be updated, and its
327      * listener will not be invoked.
328      * <p>
329      * Once the registration has been completed, the Monitor can start
330      * listening even if none of the other nodes in the group are available.
331      * It will be contacted automatically by the other nodes as they come up.
332      * <p>
333      * If the group has a Master, invoking <code>startListener</code> results
334      * in a synchronous callback to the application via the {@link
335      * MonitorChangeListener#notify(NewMasterEvent)} method.  If there is no
336      * Master at this time, the callback takes place asynchronously, after the
337      * method returns, when a Master is eventually elected.
338      * <p>
339      * Starting the listener will start the underlying ping thread, which
340      * proactively checks group status for changes that might have been
341      * missed when this Monitor instance has lost network connectivity or
342      * is down.
343      *
344      * @param newListener the listener used to monitor events of interest.
345      *
346      * @throws EnvironmentFailureException if an unexpected, internal or
347      * environment-wide failure occurs.
348      *
349      * @throws IOException if the monitor socket could not be set up
350      *
351      * @throws IllegalArgumentException if an invalid parameter is specified.
352      *
353      * @throws IllegalStateException if the monitor has been shutdown, or a
354      * listener has already been established.
355      */
356     public void startListener(MonitorChangeListener newListener)
357         throws DatabaseException, IOException {
358 
359         if (shutdown.get()) {
360             throw new IllegalStateException("The monitor has been shutdown");
361         }
362         if (newListener == null) {
363             throw new IllegalArgumentException
364                 ("A MonitorChangeListener must be associated with "  +
365                  " this Monitor when invoking this method");
366         }
367         if (this.monitorChangeListener != null) {
368             throw new IllegalStateException
369                 ("A Listener has already been established");
370         }
371 
372         this.monitorChangeListener = newListener;
373 
374         serviceDispatcher =
375             new ServiceDispatcher(monitorConfig.getNodeSocketAddress(),
376                                   channelFactory);
377         serviceDispatcher.start();
378         Protocol electionProtocol =
379             new Protocol(TimebasedProposalGenerator.getParser(),
380                          MasterValue.getParser(),
381                          monitorConfig.getGroupName(),
382                          nameIdPair,
383                          null,
384                          channelFactory);
385         learner = new Learner(electionProtocol, serviceDispatcher);
386         serviceDispatcher.register(new MonitorService(this,
387                                                       serviceDispatcher));
388         masterChangeListener = new MasterChangeListener();

392             /* Notify the listener about the current master. */
393             final ReplicationGroup repGroup = repGroupAdmin.getGroup();
394             final RepGroupImpl group = RepInternal.getRepGroupImpl(repGroup);
395 
396             /*
397              * In the absence of a network failure, the query should result in
398              * a call to the notify method of MonitorChangeListener.
399              */
400             learner.queryForMaster(group.getAllLearnerSockets());
401 
402             /* Notify JoinGroupEvents for those current active nodes. */
403             notifyJoinGroupEventsForActiveNodes(repGroup);
404 
405             /* Start an underlying ping thread. */
406             pingThread = new PingThread(repGroup);
407             pingThread.start();
408         } catch (UnknownMasterException ume) {
409             /* The Listener will be informed when a Master is elected. */
410             LoggerUtils.logMsg
411                 (logger, formatter, Level.INFO, "No current master.");
412         }

425      * Notify JoinGroupEvents for currently active nodes in replication group,
426      * and update the list of group members.
427      */
428     private void notifyJoinGroupEventsForActiveNodes(ReplicationGroup group) {
429         NodeStateProtocol stateProtocol =
430             new NodeStateProtocol(group.getName(),
431                                   NameIdPair.NOCHECK,
432                                   null,
433                                   channelFactory);
434         for (ReplicationNode repNode : group.getNodes()) {
435 
436             /*
437              * Note any existing nodes with persistent id as group
438              * members. Nodes w/ transient id never send add or remove
439              * events, so they are not included in groupMembers.
440              */
441             if (!repNode.getType().hasTransientId()) {
442                 groupMembers.add(repNode.getName());
443             }
444 
445             /*
446              * Send out a NodeState request message for this node, but only for
447              * electable and secondary nodes, since monitor nodes don't support
448              * the NodeStateService.
449              */
450             if (repNode.getType().isMonitor()) {
451                 continue;
452             }
453             MessageExchange me = stateProtocol.new MessageExchange
454                 (repNode.getSocketAddress(),
455                  NodeStateService.SERVICE_NAME,
456                  stateProtocol.new NodeStateRequest(repNode.getName()));
457             me.run();
458             ResponseMessage resp = me.getResponseMessage();
459             if (resp instanceof NodeStateResponse) {
460                 NodeStateResponse response = (NodeStateResponse) resp;
461                 notifyJoin(new JoinGroupEvent(response.getNodeName(),
462                                               response.getMasterName(),
463                                               response.getJoinTime()));
464             }
465         }
466     }

557     /**
558      * Notify the MonitorChangeListener that a GroupChangeEvent occurred.
559      */
560     synchronized void notifyGroupChange(GroupChangeEvent event) {
561         if (disableNotify) {
562             return;
563         }
564 
565         final String name = event.getNodeName();
566         switch (event.getChangeType()) {
567             case REMOVE:
568                 /* If the REMOVE event has been fired before, do nothing. */
569                 if (!groupMembers.contains(name)) {
570                     return;
571                 }
572                 /* Remove the deleted node from the group members set. */
573                 groupMembers.remove(name);
574                 break;
575             case ADD:
576                 /* If the ADD event has been fired before, do nothing. */
577                 if (groupMembers.contains(name)) {
578                     return;
579                 }
580 
581                 /* Add the node to the group members set. */
582                 groupMembers.add(name);
583                 break;
584             default:
585                 throw new IllegalArgumentException
586                     ("Unrecognized GroupChangeType: " + event.getChangeType());
587         }
588 
589         monitorChangeListener.notify(event);
590     }

592     /**
593      * Notify the MonitorChangeListener that a JoinGroupEvent happens.
594      */
595     synchronized void notifyJoin(JoinGroupEvent event) {
596         if (disableNotify) {
597             return;
598         }
599 
600         final String name = event.getNodeName();
601         /* If this JoinGroupEvent has been fired before, do nothing. */
602         if (joinEvents.containsKey(name)) {
603             return;
604         }
605 
606         /*
607          * Save the JoinGroupEvent for this node so that it can be used while
608          * notifying an abnormal LeaveGroupEvent for this node.
609          */
610         joinEvents.put(name, event);
611         monitorChangeListener.notify(event);
612     }

625     /**
626      * The Listener used to learn about new Masters
627      */
628     private class MasterChangeListener implements Learner.Listener {
629         /* The current learned value. */
630         private MasterValue currentValue = null;
631 
632         /**
633          * Implements the Listener protocol.
634          */
635         @Override
636         public void notify(Proposal proposal, Value value) {
637             /* We have a winning new proposal, is it truly different? */
638             if (value.equals(currentValue)) {
639                 return;
640             }
641             currentValue = (MasterValue) value;
642             try {
643                 String currentMasterName = currentValue.getNodeName();
644                 LoggerUtils.logMsg(logger, formatter, Level.INFO,
645                                    "Monitor notified of new Master: " +
646                                    currentMasterName);
647                 if (monitorChangeListener == null) {
648                     /* No interest */
649                     return;
650                 }
651                 monitorChangeListener.notify
652                     (new NewMasterEvent(currentValue));
653             } catch (Exception e) {
654                 LoggerUtils.logMsg
655                     (logger, formatter, Level.SEVERE,
656                      "Monitor change event processing exception: " +
657                      e.getMessage());
658             }
659         }

662     /*
663      * PingThread periodically queries the replication group state in order
664      * to proactively find group changes that this monitor may have missed
665      * if the monitor was down or had network connectivity problems. Any
666      * missed changes are propagated as the appropriate type of events.
667      *
668      * PingThread takes these steps:
669      * 1. Get the current group information. If not available, use the last
670      *    valid group information. Get information for both removed and
671      *    non-removed nodes.
672      * 2. Walk through the removed nodes set. If there is a node which is in
673      *    the removed nodes set but still contained in groupMembers, send a
674      *    REMOVE GroupChangeEvent.
675      * 2a. Also walk through the group members remembered by the monitor, and
676      *     send REMOVE GroupChangeEvent for each one that is not recorded in
677      *     the current group info for removed or unremoved nodes, to account
678      *     for deleted nodes.
679      * 3. Walk through all unremoved nodes. There are three cases:
680      *    a. If a node is reachable (it acks state request), but it's not
681      *       in groupMembers, emit an ADD GroupChangeEvent.  Only do this for
682      *       electable and monitor nodes -- secondary nodes don't send these
683      *       events.
684      *    b. If a node is reachable, but joinEvents.containsKey(node name)
685      *       returns false, send a JoinGroupEvent.
686      *    c. If a node is unreachable, but joinEvents.containsKey(node name)
687      *       returns true, we want to send a LeaveGroupEvent, but guard against
688      *       this being a transient situation. We do some retries and if we get
689      *       the same result for all retries, send a missed LeaveGroupEvent.
690      * 3a. Also walk through all nodes removed nodes, and do cases 3.b. and
691      *     3.c. for those nodes
692      * 4. Walk through all nodes in joinEvents, and perform 3.c. for ones which
693      *    are not present as removed or unremoved nodes in the group.  These
694      *    nodes are secondary nodes, which are removed from the group when the
695      *    node becomes unavailable.
696      */
697     private class PingThread extends Thread {
698         private volatile boolean running = true;
699         private ReplicationGroup group;
700         private final int retries;
701         private final long retryInterval;
702         private final int socketConnectTimeout;
703 
704         /*
705          * Track the missed LeaveEvents, mapping from the node name to the
706          * frequency it is thought to be missed.
707          */
708         private final Map<String, Integer> missedLeaveEvents =
709             new HashMap<String, Integer>();
710 
711         /* Construct an underlying PingThread. */
712         public PingThread(ReplicationGroup group) {
713             this.group = group;
714             this.retries = monitorConfig.getNumRetries();
715             this.retryInterval = monitorConfig.getRetryInterval();
716             this.socketConnectTimeout =
717                 monitorConfig.getSocketConnectTimeout();
718             setDaemon(true);
719         }
720 
721         @Override
722         public void run() {
723             try {
724                 while (running) {
725                     for (int i = 0; i < retries && running; i++) {
726 
727                         queryNodes();
728 
729                         /*
730                          * Ensure that the monitor hasn't missed any election
731                          * results due to network interruptions by explicitly
732                          * querying for a new master on a periodic basis.
733                          */
734                         RepGroupImpl groupImpl =
735                             RepInternal.getRepGroupImpl(group);
736                         /*
737                          * The query for a master will result in
738                          * NewMasterEvents, if there are any changes in the
739                          * master.
740                          */
741                         learner.queryForMaster(
742                             groupImpl.getAllLearnerSockets());
743 
744                         /* Sleep a while after querying all the nodes. */
745                         sleep(retryInterval);
746                     }
747                     missedLeaveEvents.clear();
748                 }
749             } catch (InterruptedException e) {
750                  LoggerUtils.logMsg(logger, formatter, Level.INFO,
751                                     "The daemon PingThread is interrupted: " +
752                                     e.getMessage());
753             }
754         }
755 
756         /**
757          * Ping all nodes to find out about possible missed events.
758          * Manufacture notifications for any missed events.
759          */
760         private void queryNodes() {
761             /* Get the current valid group information. */
762             final ReplicationGroup repGroup = getValidGroup();
763 
764             final Set<? extends ReplicationNode> removedNodes =
765                 repGroup.getRepGroupImpl().getRemovedNodes();
766 
767             /* Send missed REMOVE GroupChangeEvents for removed nodes. */
768             for (final ReplicationNode repNode : removedNodes) {
769                 notifyGroupChange
770                     (new GroupChangeEvent(repGroup,
771                                           repNode.getName(),
772                                           GroupChangeType.REMOVE));
773             }
774 
775             /*
776              * Also send REMOVE GroupChangeEvents for nodes in groupMembers
777              * that are no longer in the replication group, to account for
778              * deleted nodes, which will not be remembered as removed nodes
779              */
780             for (final String nodeName : groupMembers) {
781                 if (repGroup.getRepGroupImpl().getNode(nodeName) == null) {
782                     notifyGroupChange(
783                         new GroupChangeEvent(
784                             repGroup, nodeName, GroupChangeType.REMOVE));
785                 }
786             }
787 
788             /*
789              * Ping nodes to send missed ADD GroupChangeEvents, JoinGroupEvent
790              * and LeaveGroupEvent.
791              */
792             for (ReplicationNode repNode : repGroup.getNodes()) {
793                 pingNode(repNode, repGroup, false);
794             }
795 
796             /* Also ping removed nodes for missed leave events */
797             for (final ReplicationNode repNode : removedNodes) {
798                 pingNode(repNode, repGroup, true);
799             }
800 
801             /*
802              * Send leave events for secondary nodes that no longer appear in
803              * the group, since those nodes are completely removed as soon as
804              * they disconnect from the feeder, which, because they are not
805              * removed from the feeder otherwise, should only happen due to a
806              * shutdown or network failure.
807              *
808              * TODO: Provide for distinguishing secondary and deleted electable
809              * nodes.
810              */
811             for (final String nodeName : joinEvents.keySet()) {
812                 final ReplicationNode repNode =
813                     repGroup.getRepGroupImpl().getNode(nodeName);
814                 if (repNode == null) {
815                     notifyMissedLeaveEvents(nodeName);
816                 }
817             }
818         }
819 
820         /** Ping node to issue missed events. */
821         private void pingNode(ReplicationNode repNode,
822                               ReplicationGroup repGroup,
823                               boolean removed) {
824 
825             /* Monitor nodes don't respond to pings */
826             if (repNode.getType().isMonitor()) {
827                 return;
828             }
829 
830             final String name = repNode.getName();
831             try {
832                 DbPing ping = new DbPing(repNode,
833                                          getGroupName(),
834                                          socketConnectTimeout,
835                                          channelFactory);
836                 NodeState state = ping.getNodeState();
837 
838                 /* Send ADD GroupChangeEvent if needed */
839                 if (!groupMembers.contains(name) &&
840                     !repNode.getType().isSecondary() &&
841                     !repNode.getType().isExternal() &&
842                     !removed) {
843                     notifyGroupChange(
844                         new GroupChangeEvent(repGroup,
845                                              name,
846                                              GroupChangeType.ADD));
847                 }
848 
849                 /* Send JoinGroupEvent if needed */
850                 if (!joinEvents.containsKey(name)) {
851                     notifyJoin(new JoinGroupEvent(name,
852                                                   state.getMasterName(),
853                                                   state.getJoinTime()));
854                 }
855             } catch (IOException e) {
856                 /* Increase the counter of this down node. */
857                 notifyMissedLeaveEvents(name);
858             } catch (ServiceConnectFailedException e) {
859                 /* Increase the counter of this down node. */
860                 notifyMissedLeaveEvents(name);
861             } catch (ProtocolException e) {
862 
863                 /*
864                  * Thrown if a node is restarted with a new name on a
865                  * hostname/port combination used for an earlier node, meaning
866                  * the old node is offline.  Increase the counter of this down
867                  * node.
868                  */
869                 notifyMissedLeaveEvents(name);
870             }
871         }
872 
873         /*
874          * If the master is currently unknown, use the last valid group
875          * information so that the ping thread can continue working.
876          */
877         private ReplicationGroup getValidGroup() {
878             ReplicationGroup repGroup = null;
879             try {
880                 repGroup = getGroup();
881                 group = repGroup;
882             } catch (Exception e) {
883                 repGroup = group;
884             }
885 
886             return repGroup;
887         }
888 
889         /* Notify a missed LeaveGroupEvent. */
890         private void notifyMissedLeaveEvents(String name) {
891             final JoinGroupEvent event = joinEvents.get(name);
892             if (event == null) {
893                 return;
894             }
895 
896             int counter = (missedLeaveEvents.get(name) == null) ?
897                 1 : missedLeaveEvents.get(name) + 1;
898             missedLeaveEvents.put(name, counter);
899 
900             if (missedLeaveEvents.get(name) == retries) {
901                 notifyLeave(new LeaveGroupEvent(
902                                 name, event.getMasterName(),
903                                 LeaveReason.ABNORMAL_TERMINATION,
904                                 event.getJoinTime().getTime(),
905                                 System.currentTimeMillis()));
906             }
907 
908         }
