24 /**
25  * The locus for management of this node's active arbitration state, and of the
26  * mechanisms available to this node for doing arbitration.
27  * <p>
28  * A node is in active arbitration state if 
29  * <ul>
30  * <li> is the master
31  * <li> is lacking the required durability quorum
32  * <li> is maintaining its authoritative master status and its ability to
33  * commit writes through the good graces of an ArbiterProvider.
34  * </ul>
35  * <p>
36  * The Arbiter detects which arbitration options are available in the JE HA
37  * group.
38  */
39 public class Arbiter {
40 
41     /**
42      * True if this node is in active arbitration.
43      */
44     private volatile boolean active;
45 
46     private final ArbiterProvider provider;
47 
48     private final RepImpl repImpl;
49     private final Logger logger;
50 
51     /**
52      * Examine environment configuration and rep group membership to figure out
53      * which arbitration options are in operation for this HA group.
54      */
55     public Arbiter(RepImpl repImpl) {
56         this.repImpl = repImpl;
57         provider = new DesignatedPrimaryProvider(repImpl);
58         logger = LoggerUtils.getLogger(this.getClass());
59     }
60 
61     /**
62      * The replication node knows that it has lost its durability quorum, and
63      * wants to try to enter active arbitration mode. 
64      * @return true if the node successfully transitions to active arbitration,
65      * or was already in active arbitration.
66      */
67     public synchronized boolean activateArbitration() {
68         if (provider.attemptActivation()) {
69             active = true;
70         } else {
71             active = false;
72         }
73         return active;
74     }
75 
76     /**
77      * The node has determined that it need not be in active arbitration. 
78      * End the active arbitration state. If the node was not in active
79      * arbitration, do nothing.
80      */
81     public void endArbitration() {
82         synchronized(this) {
83             if (!active) {
84                 return;
85             }
86         
87             provider.endArbitration(); 
88             active = false;
89         }
90 
91         LoggerUtils.info(logger, repImpl, "Arbitration is inactivated");
92     }
93 
94     /**
95      * Return true if it's possible that this node can switch into active
96      * arbitration. The criteria for activation depend on the type of
97      * arbitration enabled for this node. 
98      * <p>
99      * For example, if designated primary arbitration is used, then it's only
100      * possible to move into active arbitration if the Designated Primary
101      * configuration parameter is set for this node. If LWT Node arbitration is
102      * used, then this node must have a valid connection to the arbiter node.
103      */
104     public boolean activationPossible() {
105         return provider.activationPossible();
106     }
107 
108     /**
109      * Return true if this node is in active arbitration, and if arbitration
110      * should take precedence over the election quorum policy.
111      */
112     public boolean isApplicable(QuorumPolicy quorumPolicy) {
113         return active && (quorumPolicy.equals(QuorumPolicy.SIMPLE_MAJORITY));
114     }
115 
116     /**
117      * Return true if this node is in active arbitration, and if arbitration
118      * should take precedence over the durability quorum policy.
119      */
120     public boolean isApplicable(ReplicaAckPolicy ackPolicy) {
121         return active && (ackPolicy.equals(ReplicaAckPolicy.SIMPLE_MAJORITY));
122     }
123 
124     /**
125      * Return the arbitration-influenced election quorum size. Arbitration 
126      * may reduce the value that would usually be indicated by the quorum
127      * policy.
128      */
129     public int getElectionQuorumSize(QuorumPolicy quorumPolicy) {
130         return provider.getElectionQuorumSize(quorumPolicy);
131     }
132 
133     /**
134      * Return the arbitration-influenced durability quorum size. Arbitration
135      * may reduce the value that would usually be indicated by the ack policy.
136      */
137     public int getAckCount(ReplicaAckPolicy ackPolicy) {
138         return provider.getAckCount(ackPolicy);
139     }
140 
141     /**
142      * The replication configuration was changed. Check the new configuration
143      * to see it impacts active arbitration state or makes more arbitration
144      * mechanisms available. For example, if we are in active arbitration using
145      * designated primary arbitration, a change to the node's designated
146      * primary configuration parameter may affect whether this node can stay in
147      * active arbitration.
148      */
149     public synchronized void 
150         processConfigChange (ReplicationMutableConfig newConfig) {
151 
152         if (!active) {
153             return;
154         }
155 
156         if (provider.shouldEndArbitration(newConfig)) {
157             endArbitration();
158         }
159     }
160 
161     /**
162      * Return true if this node is in active arbitration.
163      */
164     public synchronized boolean isActive() {
165         return active;
166     }
167 }
