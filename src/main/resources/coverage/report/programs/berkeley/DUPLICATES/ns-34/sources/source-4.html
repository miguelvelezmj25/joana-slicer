26 /**
27  * An {@code ant} task for running the {@link ClassEnhancer}.
28  *
29  * <p>{@code ClassEnhancerTask} objects are thread-safe.  Multiple threads may
30  * safely call the methods of a shared {@code ClassEnhancerTask} object.</p>
31  *
32  * <p>Note that in the BDB Java Edition product, the {@code ClassEnhancerTask}
33  * class is included in {@code je-<version>.jar}.  However, in the BDB
34  * (C-based) product, it is not included in {@code db.jar} because the build is
35  * not dependent on the Ant libraries.  Therefore, in the BDB product, the
36  * application must compile the {@code
37  * java/src/com/sleepycat/persist/model/ClassEnhancerTask.java} source file and
38  * ensure that the compiled class is available to the Ant task.  For example
39  * the following Ant task definitions could be used.</p>
40  *
41  * <p>For BDB Java Edition product:</p>
42  * <pre class="code">
43  * {@literal <taskdef name="enhance-persistent-classes"}
44  *          {@literal classname="com.sleepycat.persist.model.ClassEnhancerTask"}
45  *          {@literal classpath="${je.home}/lib/je-<version>.jar"/>}</pre>
46  * 
47  * <p>For BDB (C-based Edition) product:</p>
48  * <pre class="code">
49  * {@literal <taskdef name="enhance-persistent-classes"}
50  *          {@literal classname="com.sleepycat.persist.model.ClassEnhancerTask"}
51  *          {@literal classpath="/path-to-jar/db.jar:/path-to-ClassEnhancerTask-class"/>}</pre>
52  *
53  * <p>The class enhancer task element has no attributes.  It may contain one or
54  * more nested {@code fileset} elements specifying the classes to be enhanced.
55  * The class files are replaced when they are enhanced, without changing the
56  * file modification date.  For example:</p>
57  *
58  * <pre class="code">
59  * {@literal <target name="main">}
60  *     {@literal <enhance-persistent-classes verbose="no">}
61  *         {@literal <fileset dir="classes"/>}
62  *     {@literal </enhance-persistent-classes>}
63  * {@literal </target>}</pre>
64  *
65  * <p>The verbose attribute may be specified as "true", "yes" or "on" (like
66  * other Ant boolean attributes) to print the name of each class file that is
67  * enhanced.  The total number of class files enhanced will always be
68  * printed.</p>
69  *
70  * @author Mark Hayes
71  */
72 public class ClassEnhancerTask extends Task {
73 
74     private List<FileSet> fileSets = new ArrayList<FileSet>();
75     private boolean verbose;
76 
77     public void execute() throws BuildException {
78         if (fileSets.size() == 0) {
79             throw new BuildException("At least one fileset must be specified");
80         }
81         try {
82             int nFiles = 0;
83             ClassEnhancer enhancer = new ClassEnhancer();
84             enhancer.setVerbose(verbose);
85             for (FileSet fileSet : fileSets) {
86                 DirectoryScanner scanner =
87                     fileSet.getDirectoryScanner(getProject());
88                 String[] fileNames = scanner.getIncludedFiles();
89                 for (String fileName : fileNames) {
90                     File file = new File(scanner.getBasedir(), fileName);
91                     try {
92                         nFiles += enhancer.enhanceFile(file);
93                     } catch (IOException e) {
94                         throw new BuildException(e);
95                     }
96                 }
97             }
98             if (nFiles > 0) {
99                 System.out.println("Enhanced: " + nFiles + " files");
100             }
101         } catch (RuntimeException e) {
102             e.printStackTrace();
103             throw e;
104         }
105     }
106 
107     public void addConfiguredFileset(FileSet files) {
108         fileSets.add(files);
109     }
110 
111     public void setVerbose(boolean verbose) {
112         this.verbose = verbose;
113     }
114 }
