44  * @deprecated As of JE 4, JEMBeanHelper is deprecated in favor of the concrete

46  * registered and enabled by the environment by setting the following JVM

48  *     JEMonitor:

374      * Get MBean attribute metadata for this environment.
375      * @param targetEnv The target JE environment. May be null if the
376      * environment is not open.
377      * @return list of MBeanAttributeInfo objects describing the available
378      * attributes.
379      */
380     public List<MBeanAttributeInfo> getAttributeList(Environment targetEnv) {
381 
382         /* Turn off reset because the mbean metadata is being refreshed. */
383         setNeedReset(false);
384 
385         ArrayList<MBeanAttributeInfo> attrList = 
386             new ArrayList<MBeanAttributeInfo>();
387 
388         /* Add attributes for all JE environments. */
389         for (int i = 0; i < COMMON_ATTR.length; i++) {
390             attrList.add(COMMON_ATTR[i]);
391         }
392 
393         if (targetEnv == null) {
394             if (canConfigure) {
395                 /* Add attributes for configuring an environment. */
396                 for (int i = 0; i < CREATE_ATTR.length; i++) {
397                     attrList.add(CREATE_ATTR[i]);
398                 }
399             }
400         } else {
401             /* Add attributes for an open environment. */
402             for (int i = 0; i < OPEN_ATTR.length; i++) {
403                 attrList.add(OPEN_ATTR[i]);
404             }
405 
406             /* Add attributes for an open, transactional environment. */
407             try {
408                 EnvironmentConfig config = targetEnv.getConfig();
409                 if (config.getTransactional()) {
410                     for (int i = 0; i < TRANSACTIONAL_ATTR.length; i++) {
411                         attrList.add(TRANSACTIONAL_ATTR[i]);
412                     }
413                 }
414             } catch (DatabaseException ignore) {
415                     /* ignore */
416             }
417         }
418 
419         return attrList;
420     }

423      * Get an attribute value for the given environment. Check
424      * JEMBeanHelper.getNeedReset() after this call because the helper may
425      * detect that the environment has changed and that the MBean metadata
426      * should be reset.
427      *
428      * @param targetEnv The target JE environment. May be null if the
429      * environment is not open.
430      * @param attributeName attribute name.
431      * @return attribute value.
432      */
433     public Object getAttribute(Environment targetEnv, String attributeName)
434         throws AttributeNotFoundException,
435                MBeanException {
436 
437         /* Sanity check. */
438         if (attributeName == null) {
439             throw new AttributeNotFoundException
440                 ("Attribute name cannot be null");
441         }
442 
443         /* These attributes are available regardless of environment state. */
444         try {
445             if (attributeName.equals(ATT_ENV_HOME)) {
446                 return environmentHome.getCanonicalPath();
447             } else if (attributeName.equals(ATT_OPEN)) {
448                 boolean envIsOpen = (targetEnv != null);
449                 resetIfOpenStateChanged(envIsOpen);
450                 return new Boolean(envIsOpen);
451             } else if (attributeName.equals(ATT_SET_READ_ONLY)) {
452                 return new Boolean(openConfig.getReadOnly());
453             } else if (attributeName.equals(ATT_SET_TRANSACTIONAL)) {
454                 return new Boolean(openConfig.getTransactional());
455             } else if (attributeName.equals(ATT_SET_SERIALIZABLE)) {
456                 return new Boolean(openConfig.getTxnSerializableIsolation());
457             } else {
458                 /* The rest are JE environment attributes. */
459                 if (targetEnv != null) {
460 
461                     EnvironmentConfig config = targetEnv.getConfig();
462 
463                     if (attributeName.equals(ATT_IS_READ_ONLY)) {
464                         return new Boolean(config.getReadOnly());
465                     } else if (attributeName.equals(ATT_IS_TRANSACTIONAL)) {
466                         return new Boolean(config.getTransactional());
467                     } else if (attributeName.equals(ATT_CACHE_SIZE)) {
468                         return new Long(config.getCacheSize());
469                     } else if (attributeName.equals(ATT_CACHE_PERCENT)) {
470                         return new Integer(config.getCachePercent());
471                     } else if (attributeName.equals(ATT_LOCK_TIMEOUT)) {
472                         return new Long(config.getLockTimeout());
473                     } else if (attributeName.equals(ATT_IS_SERIALIZABLE)) {
474                         return new
475                             Boolean(config.getTxnSerializableIsolation());
476                     } else if (attributeName.equals(ATT_TXN_TIMEOUT)) {
477                         return new Long(config.getTxnTimeout());
478                     } else {
479                         throw new AttributeNotFoundException
480                             ("attribute " + attributeName + " is not valid.");
481                     }
482                 }
483                 return null;
484             }
485         } catch (Exception e) {
486 
487             /*
488              * Add both the message and the exception for easiest deciphering
489              * of the problem. Sometimes the original exception stacktrace gets
490              * hidden in server logs.
491              */
492             throw new MBeanException(e, e.getMessage());
493         }
494     }

497      * Set an attribute value for the given environment.
498      *
499      * @param targetEnv The target JE environment. May be null if the
500      * environment is not open.
501      * @param attribute name/value pair
502      */
503     public void setAttribute(Environment targetEnv, Attribute attribute)
504         throws AttributeNotFoundException,
505                InvalidAttributeValueException {
506 
507         if (attribute == null) {
508             throw new AttributeNotFoundException("Attribute cannot be null");
509         }
510 
511         /* Sanity check parameters. */
512         String name = attribute.getName();
513         Object value = attribute.getValue();
514 
515         if (name == null) {
516             throw new AttributeNotFoundException
517                 ("Attribute name cannot be null");
518         }
519 
520         if (value == null) {
521             throw new InvalidAttributeValueException
522                 ("Attribute value for attribute " + name + " cannot be null");
523         }
524 
525         try {
526             if (name.equals(ATT_SET_READ_ONLY)) {
527                 openConfig.setReadOnly(((Boolean) value).booleanValue());
528             } else if (name.equals(ATT_SET_TRANSACTIONAL)) {
529                 openConfig.setTransactional(((Boolean) value).booleanValue());
530             } else if (name.equals(ATT_SET_SERIALIZABLE)) {
531                 openConfig.setTxnSerializableIsolation
532                     (((Boolean) value).booleanValue());
533             } else {
534                 /* Set the specified attribute if the environment is open. */
535                 if (targetEnv != null) {
536 
537                     EnvironmentMutableConfig config =
538                         targetEnv.getMutableConfig();
539 
540                     if (name.equals(ATT_CACHE_SIZE)) {
541                         config.setCacheSize(((Long) value).longValue());
542                         targetEnv.setMutableConfig(config);
543                     } else if (name.equals(ATT_CACHE_PERCENT)) {
544                         config.setCachePercent(((Integer) value).intValue());
545                         targetEnv.setMutableConfig(config);
546                     } else {
547                         throw new AttributeNotFoundException
548                             ("attribute " + name + " is not valid.");
549                     }
550                 } else {
551                     throw new AttributeNotFoundException
552                         ("attribute " + name + " is not valid.");
553                 }
554             }
555         } catch (NumberFormatException e) {
556             throw new InvalidAttributeValueException("attribute name=" + name);
557         } catch (DatabaseException e) {
558             throw new InvalidAttributeValueException
559                 ("attribute name=" + name + e.getMessage());
560         }
561     }

581

590 
591             /* Add checkpoint only for transactional environments. */
592             boolean isTransactional = false;
593             try {
594                 EnvironmentConfig config = targetEnv.getConfig();
595                 isTransactional = config.getTransactional();
596             } catch (DatabaseException e) {
597                 /* Don't make any operations available. */
598                 return new ArrayList<MBeanOperationInfo>();
599             }
600 
601             if (isTransactional) {
602                 operationList.add(OP_CHECKPOINT_INFO);
603                 operationList.add(OP_TXN_STAT_INFO);
604             } else {
605                 operationList.add(OP_SYNC_INFO);
606             }
607         }
608 
609         return operationList;
610     }

612     /**
613      * Invoke an operation for the given environment.
614      *
615      * @param targetEnv The target JE environment. May be null if the
616      * environment is not open.
617      * @param actionName operation name.
618      * @param params operation parameters. May be null.
619      * @param signature operation signature. May be null.
620      * @return the operation result
621      */
622     public Object invoke(Environment targetEnv,
623                          String actionName,
624                          Object[] params,
625                          String[] signature)
626         throws MBeanException {
627 
628         /* Sanity checking. */
629         if (actionName == null) {
630             throw new IllegalArgumentException("actionName cannot be null");
631         }
632 
633         try {
634             if (targetEnv != null) {
635                 if (actionName.equals(OP_CLEAN)) {
636                     int numFiles = targetEnv.cleanLog();
637                     return new Integer(numFiles);
638                 } else if (actionName.equals(OP_EVICT)) {
639                     targetEnv.evictMemory();
640                     return null;
641                 } else if (actionName.equals(OP_CHECKPOINT)) {
642                     CheckpointConfig config = new CheckpointConfig();
643                     if ((params != null) && (params.length > 0)) {
644                         Boolean force = (Boolean) params[0];
645                         config.setForce(force.booleanValue());
646                     }
647                     targetEnv.checkpoint(config);
648                     return null;
649                 } else if (actionName.equals(OP_SYNC)) {
650                     targetEnv.sync();
651                     return null;
652                 } else if (actionName.equals(OP_ENV_STAT)) {
653                     return targetEnv.getStats
654                         (getStatsConfig(params)).toString();
655                 } else if (actionName.equals(OP_TXN_STAT)) {
656                     return targetEnv.getTransactionStats
657                         (getStatsConfig(params)).toString();
658                 } else if (actionName.equals(OP_DB_NAMES)) {
659                     return targetEnv.getDatabaseNames();
660                 } else if (actionName.equals(OP_DB_STAT)) {
661                     DatabaseStats stats = getDatabaseStats(targetEnv, params);
662                     return stats != null ? stats.toString() : null;
663                 }
664             }
665 
666             return new IllegalArgumentException
667                 ("actionName: " + actionName + " is not valid");
668         } catch (Exception e) {
669 
670             /*
671              * Add both the message and the exception for easiest deciphering
672              * of the problem. Sometimes the original exception stacktrace gets
673              * hidden in server logs.
674              */
675             throw new MBeanException(e, e.getMessage());
676         }

697      * Helper to get statistics for a given database.
698      * @param params operation parameters
699      * @return DatabaseStats object
700      */
701     private DatabaseStats getDatabaseStats(Environment targetEnv,
702                                            Object[] params)
703         throws IllegalArgumentException,
704                DatabaseNotFoundException,
705                DatabaseException {
706 
707         if ((params == null) || (params.length < 3)) {
708             return null;
709         }
710         String dbName = (String)params[2];
711 
712         Database db = null;
713         try {
714             DatabaseConfig dbConfig = new DatabaseConfig();
715             dbConfig.setReadOnly(true);
716             DbInternal.setUseExistingConfig(dbConfig, true);
717             try {
718                 db = targetEnv.openDatabase(null, dbName, dbConfig);
719             } catch (DatabaseExistsException e) {
720                 /* Should never happen, ExlcusiveCreate is false. */
721                 throw EnvironmentFailureException.unexpectedException(e);
722             }
723             return db.getStats(getStatsConfig(params));
724         } finally {
725             if (db != null) {
726                 db.close();
727             }
728         }
729     }
