53 /**
54  * Used to retrieve as much data as possible from a corrupted environment. 
55  * This utility is meant to be used programmatically, and is the equivalent
56  * to the -R or -r options for {@link DbDump}.
57  * <p>
58  * To scavenge a database:
59  *<pre>
60  *  DbScavenger scavenger =
61  *      new DbScavenger(env, outputDirectory, <boolean>, <boolean>, <boolean>);
62  *  scavenger.dump();
63  *</pre> 
64  *
65  *<p>
66  * The recovered databases will put placed in the outputDirectory with ".dump"
67  * file suffixes.  The format of the .dump files will be suitable for use with
68  * DbLoad.
69  */

143      * Start the scavenger run.
144      */
145     @Override
146     public void dump()
147         throws EnvironmentNotFoundException,
148                EnvironmentLockedException,
149                IOException {
150 
151         openEnv(false);
152 
153         envImpl = DbInternal.getNonNullEnvImpl(env);
154         DbConfigManager cm = envImpl.getConfigManager();
155         readBufferSize = cm.getInt(EnvironmentParams.LOG_ITERATOR_READ_SIZE);
156 
157         /*
158          * Find the end of the log.
159          */
160         LastFileReader reader = new LastFileReader(envImpl, readBufferSize);
161         while (reader.readNextEntry()) {
162         }
163 
164         /* Tell the fileManager where the end of the log is. */
165         long lastUsedLsn = reader.getLastValidLsn();
166         long nextAvailableLsn = reader.getEndOfLog();
167         envImpl.getFileManager().setLastPosition(nextAvailableLsn,
168                                                  lastUsedLsn,
169                                                  reader.getPrevOffset());
170 
171         try {
172             /* Pass 1: Scavenge the dbtree. */
173             if (verbose) {
174                 System.out.println("Pass 1: " + new Date());
175             }
176             scavengeDbTree(lastUsedLsn, nextAvailableLsn);
177 
178             /* Pass 2: Scavenge the databases. */
179             if (verbose) {
180                 System.out.println("Pass 2: " + new Date());
181             }
182             scavenge(lastUsedLsn, nextAvailableLsn);
183 
184             if (verbose) {
185                 System.out.println("End: " + new Date());
186             }
187         } finally {
188             closeOutputStreams();
189         }
190     }

193      * Scan the log looking for records that are relevant for scavenging the db
194      * tree.
195      */
196     private void scavengeDbTree(long lastUsedLsn, long nextAvailableLsn)
197         throws DatabaseException {
198 
199         positiveCommittedTxnIdsSeen = new BitMap();
200         negativeCommittedTxnIdsSeen = new BitMap();
201         lnNodesSeen = new TreeSet<CompareSlot>();
202 
203         final ScavengerFileReader scavengerReader =
204             new ScavengerFileReader(envImpl, readBufferSize, lastUsedLsn,
205                                     DbLsn.NULL_LSN, nextAvailableLsn) {
206                 protected void processEntryCallback(LogEntry entry,
207                                                     LogEntryType entryType)
208                     throws DatabaseException {
209 
210                     processDbTreeEntry(entry, entryType);
211                 }
212             };
213 
214         scavengerReader.setTargetType(LogEntryType.LOG_MAPLN);
215         scavengerReader.setTargetType(LogEntryType.LOG_NAMELN_TRANSACTIONAL);
216         scavengerReader.setTargetType(LogEntryType.LOG_NAMELN);
217         scavengerReader.setTargetType(LogEntryType.LOG_TXN_COMMIT);
218         scavengerReader.setTargetType(LogEntryType.LOG_TXN_ABORT);
219         lastTime = System.currentTimeMillis();
220         long fileNum = -1;
221         while (scavengerReader.readNextEntry()) {
222             fileNum = reportProgress(fileNum,
223                                      scavengerReader.getLastLsn());
224         }

227     private long reportProgress(long fileNum, long lastLsn) {
228 
229         long currentFile = DbLsn.getFileNumber(lastLsn);
230         if (verbose) {

244     /*
245      * Look at an entry and determine if it should be processed for scavenging.
246      */
247     private boolean checkProcessEntry(LogEntry entry,
248                                       LogEntryType entryType,
249                                       boolean pass2) {
250         boolean isTransactional = entryType.isTransactional();
251 
252         /*
253          * If entry is txnal...
254          *  if a commit record, add to committed txn id set
255          *  if an abort record, ignore it and don't process.
256          *  if an LN, check if it's in the committed txn id set.
257          *     If it is, continue processing, otherwise ignore it.
258          */
259         if (isTransactional) {
260             final long txnId = entry.getTransactionId();
261             if (entryType.equals(LogEntryType.LOG_TXN_COMMIT)) {
262                 setCommittedTxn(txnId);
263                 /* No need to process this entry further. */
264                 return false;
265             }
266 
267             if (entryType.equals(LogEntryType.LOG_TXN_ABORT)) {
268                 /* No need to process this entry further. */
269                 return false;
270             }
271 
272             if (!isCommittedTxn(txnId)) {
273                 return false;
274             }
275         }
276 
277         /*
278          * Check the nodeId to see if we've already seen it or not.
279          */
280         if (entry instanceof LNLogEntry) {
281 
282             final LNLogEntry<?> lnEntry = (LNLogEntry<?>) entry;
283             final long dbId = lnEntry.getDbId().getId();
284             final DatabaseImpl db = dbIdToImpl.get(dbId);
285             /* Must call postFetchInit if true is returned. */
286             if (db != null) {
287                 lnEntry.postFetchInit(db);
288             } else {
289                 lnEntry.postFetchInit(false /*isDupDb*/);
290             }
291 
292             /*
293              * If aggressive or if processing DbTree entries, don't worry about
294              * whether this node has been processed already.
295              */
296             if (doAggressiveScavengerRun || !pass2) {
297                 return true;
298             }
299 
300             if (db == null) {
301                 throw EnvironmentFailureException.unexpectedState
302                     ("Database info not available for DB ID: " + dbId);
303             }
304             return lnNodesSeen.add(new CompareSlot(db, lnEntry));
305         }
306 
307         return false;
308     }

310     /*
311      * Called once for each log entry during the pass 1 (dbtree).
312      */
313     private void processDbTreeEntry(LogEntry entry, LogEntryType entryType)
314         throws DatabaseException {
315 
316         boolean processThisEntry =
317             checkProcessEntry(entry, entryType, false);
318 
319         if (processThisEntry &&
320             (entry instanceof LNLogEntry)) {
321             LNLogEntry<?> lnEntry = (LNLogEntry<?>) entry;
322             LN ln = lnEntry.getLN();
323             if (ln instanceof NameLN) {
324                 String name = StringUtils.fromUTF8(lnEntry.getKey());
325                 Long dbId = Long.valueOf(((NameLN) ln).getId().getId());
326                 if (dbIdToName.containsKey(dbId) &&
327                     !dbIdToName.get(dbId).equals(name)) {
328                     throw EnvironmentFailureException.unexpectedState
329                         ("Already name mapped for dbId: " + dbId +
330                          " changed from " + dbIdToName.get(dbId) +
331                          " to " + name);
332                 } else {
333                     dbIdToName.put(dbId, name);
334                 }
335             }
336 
337             if (ln instanceof MapLN) {
338                 DatabaseImpl db = ((MapLN) ln).getDatabase();
339                 Long dbId = db.getId().getId();
340                 /* Use latest version to get most recent comparators. */
341                 if (!dbIdToImpl.containsKey(dbId)) {
342                     dbIdToImpl.put(dbId, db);
343                 }
344             }
345         }
346     }

348     /*
349      * Pass 2: scavenge the regular (non-dbtree) environment.
350      */
351     private void scavenge(long lastUsedLsn, long nextAvailableLsn)
352         throws DatabaseException {
353 
354         final ScavengerFileReader scavengerReader =
355             new ScavengerFileReader(envImpl, readBufferSize, lastUsedLsn,
356                                     DbLsn.NULL_LSN, nextAvailableLsn) {
357                 protected void processEntryCallback(LogEntry entry,
358                                                     LogEntryType entryType)
359                     throws DatabaseException {
360 
361                     processRegularEntry(entry, entryType);
362                 }
363             };
364 
365         /*
366          * Note: committed transaction id map has been created already, no
367          * need to read TXN_COMMITS on this pass.
368          */
369         for (LogEntryType entryType : LogEntryType.getAllTypes()) {
370             if (entryType.isUserLNType()) {
371                 scavengerReader.setTargetType(entryType);
372             }
373         }
374         scavengerReader.setDumpCorruptedBounds(dumpCorruptedBounds);
375 
376         long progressFileNum = -1;
377         while (scavengerReader.readNextEntry()) {
378             progressFileNum = reportProgress(progressFileNum,
379                                              scavengerReader.getLastLsn());
380         }
381     }

383     /*
384      * Process an entry during pass 2.
385      */
386     private void processRegularEntry(LogEntry entry, LogEntryType entryType)
387         throws DatabaseException {
388 
389         boolean processThisEntry =
390             checkProcessEntry(entry, entryType, true);
391 
392         if (processThisEntry) {
393             LNLogEntry<?> lnEntry = (LNLogEntry<?>) entry;
394             Long dbId = Long.valueOf(lnEntry.getDbId().getId());
395             LN ln = lnEntry.getLN();
396 
397             /* Create output file even if we don't process a deleted entry. */
398             PrintStream out = getOutputStream(dbId);
399 
400             if (!ln.isDeleted()) {
401                 DatabaseEntry key = new DatabaseEntry();
402                 DatabaseEntry data = new DatabaseEntry();
403                 lnEntry.getUserKeyData(key, data);
404                 dumpOne(out, key.getData(), formatUsingPrintable);
405                 dumpOne(out, data.getData(), formatUsingPrintable);
406                 if ((++flushCounter % FLUSH_INTERVAL) == 0) {
407                     out.flush();
408                     flushCounter = 0;
409                 }
410             }
411         }
412     }

415      * Return the output stream for the .dump file for database with id dbId.
416      * If an output stream has not already been created, then create one.
417      */
418     private PrintStream getOutputStream(Long dbId)
419         throws DatabaseException {
420 
421         PrintStream ret = dbIdToOutputStream.get(dbId);
422         if (ret != null) {
423             return ret;
424         }
425         String name = dbIdToName.get(dbId);
426         if (name == null) {
427             name = "db" + dbId;
428         }
429         File file = new File(outputDirectory, name + ".dump");
430         try {
431             ret = new PrintStream(new FileOutputStream(file), false);
432         } catch (FileNotFoundException e) {
433             throw EnvironmentFailureException.unexpectedException(e);
434         }
435         dbIdToOutputStream.put(dbId, ret);
436         DatabaseImpl db = dbIdToImpl.get(dbId);
437         boolean dupSort = (db != null) ? db.getSortedDuplicates() : false;
438         printHeader(ret, dupSort, formatUsingPrintable);
439         return ret;
440     }
