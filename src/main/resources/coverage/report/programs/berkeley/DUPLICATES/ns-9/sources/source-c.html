66     /**
67      * Returns the approximate byte size of all obsolete LN entries, using the
68      * average LN size for LN sizes that were not counted.
69      */
70     public int getObsoleteLNSize() {
71 
72         if (totalLNCount == 0) {
73             return 0;
74         }
75 
76         /* Normalize obsolete amounts to account for double-counting. */
77         final int obsLNCount = Math.min(obsoleteLNCount, totalLNCount);
78         final int obsLNSize = Math.min(obsoleteLNSize, totalLNSize);
79         final int obsLNSizeCounted = Math.min(obsoleteLNSizeCounted,
80                                               obsLNCount);
81 
82         /*
83          * Use the tracked obsolete size for all entries for which the size was
84          * counted, plus the average size for all obsolete entries whose size
85          * was not counted.
86          */
87         long obsSize = obsLNSize;
88         final int obsCountNotCounted = obsLNCount - obsLNSizeCounted;
89         if (obsCountNotCounted > 0) {
90 
91             /*
92              * When there are any obsolete LNs with sizes uncounted, we add an
93              * obsolete amount that is the product of the number of LNs
94              * uncounted and the average LN size.
95              */
96             final float avgLNSizeNotCounted = getAvgObsoleteLNSizeNotCounted();
97             if (!Float.isNaN(avgLNSizeNotCounted)) {
98                 obsSize += (int) (obsCountNotCounted * avgLNSizeNotCounted);
99             }
100         }
101 
102         /* Don't return an impossibly large estimate. */
103         return (obsSize > totalLNSize) ? totalLNSize : (int) obsSize;
104     }

106     /**
107      * Returns the average size for LNs with sizes not counted, or NaN if
108      * there are no such LNs.
109      *
110      * In FileSummaryLN version 3 and greater the obsolete size is normally
111      * counted, but not in exceptional circumstances such as recovery.  If it
112      * is not counted, obsoleteLNSizeCounted will be less than obsoleteLNCount.
113      *
114      * In log version 8 and greater, we don't count the size when the LN is not
115      * resident in cache during update/delete, and CLEANER_FETCH_OBSOLETE_SIZE
116      * is false (the default setting).
117      *
118      * We added maxLNSize in version 8 for use in estimating obsolete LN sizes.
119      *
120      * To compute the average LN size, we only consider the LNs (both obsolete
121      * and non-obsolete) for which the size has not been counted.  This
122      * increases accuracy when counted and uncounted LN sizes are not uniform.
123      * An example is when large LNs are inserted and deleted.  The size of the
124      * deleted LN log entry (which is small) is always counted, but the
125      * previous version (which has a large size) may not be counted.
126      */
127     public float getAvgObsoleteLNSizeNotCounted() {
128 
129         /* Normalize obsolete amounts to account for double-counting. */
130         final int obsLNCount = Math.min(obsoleteLNCount, totalLNCount);
131         final int obsLNSize = Math.min(obsoleteLNSize, totalLNSize);
132         final int obsLNSizeCounted = Math.min(obsoleteLNSizeCounted,
133                                               obsLNCount);
134 
135         final int obsCountNotCounted = obsLNCount - obsLNSizeCounted;
136         if (obsCountNotCounted <= 0) {
137             return Float.NaN;
138         }
139 
140         final int totalSizeNotCounted = totalLNSize - obsLNSize;
141         final int totalCountNotCounted = totalLNCount - obsLNSizeCounted;
142 
143         if (totalSizeNotCounted <= 0 || totalCountNotCounted <= 0) {
144             return Float.NaN;
145         }
146 
147         return totalSizeNotCounted / ((float) totalCountNotCounted);
148     }

150     /**
151      * Returns the maximum possible obsolete LN size, using the maximum LN size
152      * for LN sizes that were not counted.
153      */
154     public int getMaxObsoleteLNSize() {
155 
156         /*
157          * In log version 7 and earlier the maxLNSize is not available.  It is
158          * safe to use getObsoleteLNSize in that case, because LSN locking was
159          * not used and the obsolete size was counted for updates and deletes.
160          */
161         if (maxLNSize == 0) {
162             return getObsoleteLNSize();
163         }
164 
165         if (totalLNCount == 0) {
166             return 0;
167         }
168 
169         /* Normalize obsolete amounts to account for double-counting. */
170         final int obsLNCount = Math.min(obsoleteLNCount, totalLNCount);
171         final int obsLNSize = Math.min(obsoleteLNSize, totalLNSize);
172         final int obsLNSizeCounted = Math.min(obsoleteLNSizeCounted,
173                                               obsLNCount);
174 
175         /*
176          * Use the tracked obsolete size for all entries for which the size was
177          * counted, plus the maximum possible size for all obsolete entries
178          * whose size was not counted.
179          */
180         long obsSize = obsLNSize;
181         final long obsCountNotCounted = obsLNCount - obsLNSizeCounted;
182         if (obsCountNotCounted > 0) {
183 
184             /*
185              * When there are any obsolete LNs with sizes uncounted, we add an
186              * obsolete amount that is the minimum of two values.  Either value
187              * may be much higher than the true obsolete amount, but by taking
188              * their minimum we use a much more realistic obsolete amount.
189              *
190              * maxLNSizeNotCounted is the maximum obsolete not counted, based
191              * on the multiplying maxLNSize and the number of obsolete LNs not
192              * counted.
193              *
194              * maxObsSizeNotCounted is also an upper bound on the obsolete size
195              * not not counted.  The (totalLNSize - obsLNSize) gives the amount
196              * non-obsolete plus the obsolete amount not counted.  From this we
197              * subtract the minimum non-obsolete size, based on the minimum
198              * size of any LN.  This leaves the maximum obsolete size not
199              * counted.
200              *
201              * Note that the mutiplication immediately below would overflow if
202              * type 'int' instead of 'long' were used for the operands.  This
203              * was fixed in [#21106].
204              */
205             final long maxLNSizeNotCounted = obsCountNotCounted * maxLNSize;
206 
207             final long maxObsSizeNotCounted = totalLNSize - obsLNSize -
208                 ((totalLNCount - obsLNCount) * LNLogEntry.MIN_LOG_SIZE);
209 
210             obsSize += Math.min(maxLNSizeNotCounted, maxObsSizeNotCounted);
211         }
212 
213         /* Don't return an impossibly large estimate. */
214         return (obsSize > totalLNSize) ? totalLNSize : (int) obsSize;
215     }

251     private int calculateObsoleteSize(int lnObsoleteSize) {
252         if (totalSize <= 0) {
253             return 0;
254         }
255             /* Leftover (non-IN non-LN) space is considered obsolete. */
256         final int leftoverSize = totalSize - (totalINSize + totalLNSize);
257 
258         int obsoleteSize = lnObsoleteSize +
259             getObsoleteINSize() +
260             leftoverSize;
261 
262         /*
263          * Don't report more obsolete bytes than the total.  We may
264          * calculate more than the total because of (intentional)
265          * double-counting during recovery.
266          */
267         if (obsoleteSize > totalSize) {
268             obsoleteSize = totalSize;
269         }
270         return obsoleteSize;
271     }
