54 /**
55  * JE compression consists of removing BIN slots for deleted and expired
56  * records, and pruning empty IN/BINs from the tree which is also called a
57  * reverse split.
58  *
59  * One of the reasons compression is treated specially is that slot compression
60  * cannot be performed inline as part of a delete operation.  When we delete an
61  * LN, a cursor is always present on the LN.  The API dictates that the cursor
62  * will remain positioned on the deleted record.  In addition, if the deleting
63  * transaction aborts we must restore the slot and the possibility of a split
64  * during an abort is something we wish to avoid; for this reason, compression
65  * will not occur if the slot's LSN is locked.  In principle, slot compression
66  * could be performed during transaction commit, but that would be expensive
67  * because a Btree lookup would be required, and this would negatively impact
68  * operation latency.  For all these reasons, slot compression is performed
69  * after the delete operation is complete and committed, and not in the thread
70  * performing the operation or transaction commit.
71  *
72  * Compression is of two types:
73  *
74  * + "Queued compression" is carried out by the INCompressor daemon thread.
75  *    Both slot compression and pruning are performed.
76  *
77  * + "Lazy compression" is carried out opportunistically at various times when
78  *    compression is beneficial.
79  *
80  * The use of BIN-deltas has a big impact on slot compression because dirty
81  * slots cannot be compressed until we know that a full BIN will be logged
82  * next. If a dirty slot were compressed prior to logging a BIN-delta, the
83  * record of the compression would be lost and the slot would "reappear" when
84  * the BIN is reconstituted. Normally we do not compress dirty slots when a
85  * delta would next be logged. However, there are times when we do compress
86  * dirty slots, and in that case the "prohibit next logged delta" flag is set
87  * on the BIN.
88  *
89  * Queued compression prior to logging a BIN-delta is also wasteful because the
90  * dequeued entry cannot be processed.  Therefore, lazy compression is relied
91  * on when a BIN-delta will next be logged.  Because, BIN-deltas are logged
92  * more often than BINs, lazy compression is used for slot compression more
93  * often than queued compression.
94  *
95  * Lazy compression is used for compressing expired slots, in addition to
96  * deleted slots. This is done opportunistically as described above. Expired
97  * slots are normally not dirty, so they can often be compressed even when a
98  * BIN will be logged next as a BIN-delta. The same is true of deleted slots
99  * that are not dirty, although these occur infrequently.
100  *
101  * Since we don't lazy-compress BIN-deltas, how can we prevent their expired
102  * slots from using space for long time periods? Currently the expired space
103  * will be reclaimed only when the delta is mutated to a full BIN and then
104  * compressed, including when the full BIN is cleaned. This is the same as
105  * reclaiming space for deleted slots, so this is acceptable for now at least.
106  *
107  * You may wonder, since lazy compression is necessary, why use queued
108  * compression for slot compression at all? Queued compression is useful for
109  * the following reasons:
110  *
111  * + If a BIN-delta will not be logged next, queued compression will cause the
112  *   compression to occur sooner than with lazy compression.
113  *
114  * + When a cursor is on a BIN or a deleted entry is locked during lazy
115  *   compression, we cannot compress the slot. Queuing allows it to be
116  *   compressed sooner than if we waited for the next lazy compression.
117  *
118  * + The code to process a queue entry must do slot compression anyway, even if
119  *   we only want to prune the BIN.  We have to account for the case where all
120  *   slots are deleted but not yet compressed.  So the code to process the
121  *   queue entry could not be simplified even if we were to decide not to queue
122  *   entries for slot compression. 
123  */

346      * The real work to doing a compress. This may be called by the compressor
347      * thread or programatically.
348      */
349     public synchronized void doCompress()
350         throws DatabaseException {
351 
352         /*
353          * Make a snapshot of the current work queue so the compressor thread
354          * can safely iterate over the queue. Note that this impacts lazy
355          * compression, because it lazy compressors will not see BINReferences
356          * that have been moved to the snapshot.
357          */
358         final Map<Long, BINReference> queueSnapshot;
359         final int binQueueSize;
360         synchronized (binRefQueueSync) {
361             binQueueSize = binRefQueue.size();
362             if (binQueueSize <= 0) {
363                 return;
364             }
365             queueSnapshot = binRefQueue;
366             binRefQueue = new HashMap<>();
367         }
368 
369         /* There is work to be done. */
370         resetPerRunCounters();
371         LoggerUtils.fine(logger, envImpl,
372                          "InCompress.doCompress called, queue size: " +
373                          binQueueSize);
374         if (LatchSupport.TRACK_LATCHES) {
375             LatchSupport.expectBtreeLatchesHeld(0);
376         }
377 
378         /*
379          * Compressed entries must be counted as obsoleted.  A separate
380          * tracker is used to accumulate tracked obsolete info so it can be
381          * added in a single call under the log write latch.  We log the
382          * info for deleted subtrees immediately because we don't process
383          * deleted IN entries during recovery; this reduces the chance of
384          * lost info.
385          */
386         final LocalUtilizationTracker localTracker =
387             new LocalUtilizationTracker(envImpl);
388 
389         /* Use local caching to reduce DbTree.getDb overhead. */
390         final Map<DatabaseId, DatabaseImpl> dbCache = new HashMap<>();
391 
392         final DbTree dbTree = envImpl.getDbTree();
393         final BINSearch binSearch = new BINSearch();
394 
395         try {
396             for (final BINReference binRef : queueSnapshot.values()) {
397 
398                 if (envImpl.isClosed()) {
399                     return;
400                 }
401 
402                 if (!findDBAndBIN(binSearch, binRef, dbTree, dbCache)) {
403 
404                     /*
405                      * Either the db is closed, or the BIN doesn't exist.
406                      * Don't process this BINReference.
407                      */
408                     continue;
409                 }
410 
411                 /* Compress deleted slots and prune if possible. */
412                 compressBin(binSearch.db, binSearch.bin, binRef, localTracker);
413             }
414 
415             /* SR [#11144]*/
416             assert TestHookExecute.doHookIfSet(beforeFlushTrackerHook);
417 
418             /*
419              * Count obsolete nodes and write out modified file summaries
420              * for recovery.  All latches must have been released.
421              */
422             envImpl.getUtilizationProfile().flushLocalTracker(localTracker);
423 
424         } finally {
425             dbTree.releaseDbs(dbCache);
426             if (LatchSupport.TRACK_LATCHES) {
427                 LatchSupport.expectBtreeLatchesHeld(0);
428             }
429             accumulatePerRunCounters();
430         }
431     }

433     /**
434      * Compresses a single BIN and then deletes the BIN if it is empty.
435      *
436      * @param bin is latched when this method is called, and unlatched when it
437      * returns.
438      */
439     private void compressBin(
440         DatabaseImpl db,
441         BIN bin,
442         BINReference binRef,
443         LocalUtilizationTracker localTracker) {
444 
445         /* Safe to get identifier keys; bin is latched. */
446         final byte[] idKey = bin.getIdentifierKey();
447         boolean empty = (bin.getNEntries() == 0);
448 
449         try {
450             if (!empty) {
451 
452                 /*
453                  * Deltas in cache cannot be compressed.
454                  *
455                  * We strive not to add a slot to the queue when we will log a
456                  * delta.  However, it is possible that an entry is added, or
457                  * that an entry is not cleared by lazy compression prior to
458                  * logging a full BIN.  Clean-up for such queue entries is
459                  * here.
460                 */
461                 if (bin.isBINDelta()) {
462                     return;
463                 }
464 
465                 /* If there are cursors on the BIN, requeue and try later. */
466                 if (bin.nCursors() > 0) {
467                     addBinRefToQueue(binRef);
468                     cursorsBinsThisRun++;
469                     return;
470                 }
471 
472                 /*
473                  * If a delta should be logged, do not compress dirty slots,
474                  * since this would prevent logging a delta.
475                  */
476                 if (!bin.compress(
477                     !bin.shouldLogDelta() /*compressDirtySlots*/,
478                     localTracker)) {
479 
480                     /* If compression is incomplete, requeue and try later. */
481                     addBinRefToQueue(binRef);
482                     return;
483                 }
484 
485                 /* After compression the BIN may be empty. */
486                 empty = (bin.getNEntries() == 0);
487             }
488         } finally {
489             bin.releaseLatch();
490         }
491 
492         /* After releasing the latch, prune the BIN if it is empty. */
493         if (empty) {
494             pruneBIN(db, binRef, idKey);
495         }
496     }

499      * If the target BIN is empty, attempt to remove the empty branch of the
500      * tree.

502     private void pruneBIN(DatabaseImpl dbImpl,
503                           BINReference binRef,
504                           byte[] idKey) {
505 
506         try {
507             final Tree tree = dbImpl.getTree();
508             tree.delete(idKey);
509             processedBinsThisRun++;
510         } catch (NodeNotEmptyException NNEE) {
511 
512             /*
513              * Something was added to the node since the point when the
514              * deletion occurred; we can't prune, and we can throw away this
515              * BINReference.
516              */
517              nonEmptyBinsThisRun++;
518         } catch (CursorsExistException e) {
519             /* If there are cursors in the way of the delete, retry later. */
520             addBinRefToQueue(binRef);
521             cursorsBinsThisRun++;
522         }
523     }

556     /**
557      * Lazily/opportunistically compress a full BIN.
558      *
559      * The target IN should be latched when we enter, and it will be remain
560      * latched.
561      *
562      * If compression succeeds, does not prune empty BINs, but does queue them
563      * for pruning later. If compression fails because a record lock cannot be
564      * obtained, queues the BIN to retry later.
565      *
566      * Note that we do not bother to delete queue entries for the BIN if
567      * compression succeeds.  Queue entries are normally removed quickly by the
568      * compressor.  In the case where queue entries happen to exist when we do
569      * the final compression below, we rely on the compressor to clean them up
570      * later on when they are processed.
571      */
572     public void lazyCompress(final IN in, final boolean compressDirtySlots) {
573 
574         assert in.isLatchOwner();
575 
576         /* Only full BINs can be compressed. */
577         if (!in.isBIN() || in.isBINDelta()) {
578             return;
579         }
580 
581         final BIN bin = (BIN) in;
582 
583         /*
584          * Cursors prohibit compression. We queue for later when there is
585          * anything that can be compressed.
586          */
587         if (bin.nCursors() > 0) {
588             for (int i = 0; i < bin.getNEntries(); i += 1) {
589                 if (bin.isDefunct(i)) {
590                     addBinToQueue(bin);
591                     break;
592                 }
593             }
594             return;
595         }
596 
597         if (bin.compress(compressDirtySlots, null /*localTracker*/)) {
598             if (bin.getNEntries() == 0) {
599                 /* The BIN is empty. Prune it later. */
600                 addBinToQueue(bin);
601             }
602         } else {
603             /* A record lock prevented slot removal. Try again later. */
604             addBinToQueue(bin);
605         }
606 
607         lazyProcessed++;
608     }

611      * Find the db and bin for a BINReference.
612      * @return true if the db is open and the target bin is found.
613      */
614     private boolean findDBAndBIN(
615         BINSearch binSearch,
616         BINReference binRef,
617         DbTree dbTree,
618         Map<DatabaseId, DatabaseImpl> dbCache)
619         throws DatabaseException {
620 
621         /*
622          * Find the database.  Do not call releaseDb after this getDb, since
623          * the entire dbCache will be released later.
624          */
625         binSearch.db = dbTree.getDb(
626             binRef.getDatabaseId(), lockTimeout, dbCache);
627 
628         if (binSearch.db == null || binSearch.db.isDeleted()) {
629             /* The db was deleted. Ignore this BIN Ref. */
630             dbClosedBinsThisRun++;
631             return false;
632         }
633 
634         /* Perform eviction before each operation. */
635         envImpl.daemonEviction(true /*backgroundIO*/);
636 
637         /* Find the BIN. */
638         binSearch.bin = searchForBIN(binSearch.db, binRef);
639 
640         if (binSearch.bin == null ||
641             binSearch.bin.getNodeId() != binRef.getNodeId()) {
642             /* The BIN may have been split. */
643             if (binSearch.bin != null) {
644                 binSearch.bin.releaseLatch();
645             }
646             splitBinsThisRun++;
647             return false;
648         }
649 
650         return true;
651     }
