533      * after the recovery checkpoint, before the Replicator constructor returns
534      * and before any user level HA operations can start. That's not sufficient
535      * because the recovery checkpoint is shortening the recovery interval for
536      * future recoveries, and any information that has been garnered must be
537      * persisted. Here's an example of what might happen after a series of

542      *    recovery did not find any VLSNs in log, because log is brand new

551      *  recovery crawls log from ckpt 1 start onward, finds VLSNs 1-3
552      *  recovery logs ckpt 2 start
553      *  recovery logs ckpt 2 end
554      *  VLSN index instantiated, VLSNs 1-3 added in but not written too disk

556      *  crash ... Environment recovers
557      *  recovery crawls log from ckpt start 2 start onward, finds no VLSNs.

856     /**
857      * Generate the next VLSN and update the DTVLSN value in the item. Note
858      * that this method is only invoked when the node is in the Master state,
859      * since the master assigns new VLSNs and DTVLSNs, and the replicas simply
860      * preserve them.
861      *
862      * The DTVLSN value must be calculated under the same latch as the updating
863      * of the VLSN to ensure that the following invariants are maintained:
864      *
865      * lsn1 > lsn2 ==> VLSN(lsn1) > VLSN(lsn2)
866      * vlsn2 > vlsn1 ==> DTVLSN(vlsn2) >= DTVLSN(vlsn1)
867      *
868      * where vlsn2 and vlsn1 are transaction commit or abort records.
869      *
870      * Replicas, when replaying their stream, verify that this invariant is
871      * maintained.
872      *
873      * Commit/Abort records for Replication groups that have a single electable
874      * and durable node have their dtvlsn written as the associated VLSN, that
875      * is, DTVLSN(vlsn) == vlsn. For all other RG configurations, DTVLSN(vlsn)
876      * < vlsn.
877      *
878      * Commit/Abort Log records that are created by replaying an HA stream from
879      * a pre DTVLSN feeder, will have their dtvlsns set to
880      * VLSN.UNINITIALIZED_VLSN_SEQUENCE during replica replay. They do not
881      * follow this code path.
882      *
883      * @param entry the log entry with which the VLSN will be associated. If
884      * the log entry represents a commit or abort entry, its DTVLSN is modified
885      * so that it's correct when it's serialized out.
886      */
887     @Override
888     public VLSN assignVLSNs(LogEntry entry) {
889         final VLSN vlsn = vlsnIndex.bump();
890 
891         final byte itemType = entry.getLogType().getTypeNum();
892         if (itemType != LogEntryType.LOG_TXN_COMMIT.getTypeNum() &&
893             itemType !=  LogEntryType.LOG_TXN_ABORT.getTypeNum()) {
894             return  vlsn;
895         }
896 
897         /*
898          * A commit or abort record. Compute the DTVLSN value to associate with
899          * the commit record and store it into the buffer at the appropriate
900          * position.
901          */
902         final long dtvlsn;
903 
904         if (simulatePreDTVLSNMaster) {
905             dtvlsn = VLSN.UNINITIALIZED_VLSN_SEQUENCE;
906         } else if (repNode.isNeedsAcks()) {
907             /*
908              * Use the dtvlsn value being tracked via acknowledgments from
909              * replicas when replication is being used for durability.
910              */
911             dtvlsn = getRepNode().getDTVLSN();
912         } else {
913             /*
914              * Replicated environment, but replication is not being used for
915              * durability. That is, the commit is self-acknowledged, set dtvlsn
916              * == vlsn.
917              */
918             dtvlsn = getRepNode().updateDTVLSN(vlsn.getSequence());
919         }
920 
921         final VersionedWriteTxnEnd txnEnd =
922             (VersionedWriteTxnEnd)entry.getMainItem();
923 
924         /*
925          * As a cheap sanity check, commits/aborts on the master are created
926          * with VLSN.NULL values, so they can be verified here.
927          */
928         final long checkDTVLSN = txnEnd.getDTVLSN();
929         if (checkDTVLSN != VLSN.NULL_VLSN_SEQUENCE) {
930             throw new IllegalStateException("NULL DTVLSN expected at VLSN:" +
931                                             vlsn + " not " + checkDTVLSN);
932         }
933 
934         txnEnd.setDTVLSN(dtvlsn);
935 
936         /* Verify invariant */
937         if (prevLoggedVLSN > vlsn.getSequence()) {
938             if (dtvlsn < prevLoggedDTVLSN) {
939                 String msg =
940                     "DTVLSNs must be in ascending order in the stream. " +
941                     " prev DTVLSN:" + prevLoggedDTVLSN +
942                     " next DTVLSN:" + dtvlsn + " at VLSN: " + vlsn;
943                 throw EnvironmentFailureException.unexpectedState(this, msg);
944             }
945         }
946         prevLoggedVLSN = vlsn.getSequence();
947         prevLoggedDTVLSN = dtvlsn;
948 
949         return vlsn;
950     }

1241      * used to delay the commit operation until a sufficient number of commits
1242      * have been received.
1243      *
1244      * In addition, when mastership transfers are done, and this node is the
1245      * original master, commits and aborts are blocked so as to avoid hard
1246      * recovery after electing a new master, see [#18081].
1247      *
1248      * @param txn the master transaction being committed
1249      *
1250      * @throws InsufficientReplicasException if the feeder is not in contact
1251      * with enough replicas.
1252      * @throws RestartRequiredException if the environment is invalid.
1253      * @throws UnknownMasterException if the current master is unknown.
1254      * @throws ReplicaWriteException if the node transitioned to a Replica
1255      * after the transaction was initiated.
1256      */
1257     public void preLogCommitHook(MasterTxn txn)
1258         throws InsufficientReplicasException,
1259                RestartRequiredException,
1260                UnknownMasterException,
1261                ReplicaWriteException,
1262                EnvironmentFailureException {
1263 
1264         checkIfInvalid();
1265         checkIfMaster(txn);
1266         checkBlock(txn);
1267 
1268         /* Still a master, check for a sufficient number of connections */
1269         int activeReplicaCount =
1270             repNode.feederManager().activeAckReplicaCount();
1271         ReplicaAckPolicy ackPolicy =
1272             txn.getCommitDurability().getReplicaAck();
1273         int requiredAckCount = txn.getRequiredAckCount();
1274 
1275         if (envLogger.isLoggable(Level.FINE)) {
1276             LoggerUtils.fine(envLogger, this,
1277                              "Txn " + txn.getId() + " requires: " +
1278                              requiredAckCount + " active: " +
1279                              activeReplicaCount +
1280                              " replica acks. Commit Policy: " + ackPolicy);
1281         }
1282 
1283         if (requiredAckCount > activeReplicaCount) {
1284             /* Check for possible activation of Primary */
1285             if (ackPolicy.equals(ReplicaAckPolicy.SIMPLE_MAJORITY) &&
1286                 repNode.getArbiter().activateArbitration()) {
1287                 txn.resetRequiredAckCount();
1288             } else if (useArbiter(txn)) {
1289                 /*
1290                  * Note we could change the check to allow a degraded
1291                  * write from any group size. Limit is place at rep group
1292                  * size of two due to the priority of requirements
1293                  * and lower cost of testing.
1294                  */
1295                 txn.setArbiterAck(true);
1296             } else {
1297                 /*
1298                  * Capture the set to ensure it's consistent with the exception
1299                  * message.
1300                  */
1301                 final boolean includeArbiters =
1302                     !ackPolicy.equals(ReplicaAckPolicy.ALL);
1303                 final Set<String> activeAckRepSet =
1304                     repNode.feederManager().activeAckReplicas(includeArbiters);
1305 
1306                 if (requiredAckCount > activeAckRepSet.size()) {
1307                     /* No change in window, throw exception */
1308                     InsufficientReplicasException ire =
1309                     new InsufficientReplicasException
1310                         (txn, ackPolicy, requiredAckCount, activeAckRepSet);
1311                     LoggerUtils.info(envLogger, this, ire.getMessage());
1312                     throw ire;
1313                 }
1314 
1315                 /**
1316                  * A new replica became active in the window between the
1317                  * first fast check and the second slower check, just continue
1318                  */
1319             }
1320         }
1321         feederTxns.setupForAcks(txn);
1322     }

1324     /*
1325      * Block transaction commits/aborts if this node is the original master
1326      * and we're doing Master Transfer.
1327      */
1328     private void checkBlock(MasterTxn txn) {
1329         try {
1330 
1331             /*
1332              * Lock out the setting of the block latch by Master Transfer in
1333              * the interval between waiting on the latch and setting the VLSN
1334              * for the commit: Master Transfer needs to get a coherent idea of
1335              * the final VLSN when it sets the latch.  This lock will be
1336              * released by the {@code postLogXxxHook()} functions, one of which
1337              * is guaranteed to be called, unless an Environment-invalidating
1338              * exception occurs.
1339              */
1340             if (txn.lockOnce()) {
1341                 blockLatchLock.readLock().lockInterruptibly();
1342             }
1343 
1344             if (blockTxnLatch.getCount() > 0) {
1345                 LoggerUtils.info(envLogger, this,
1346                                  "Block transaction: " + txn.getId() +
1347                                  " pending master transfer. Write locks = " +
1348                                  txn.getWriteLockIds());
1349             }
1350 
1351             final long txnTimeout = txn.getTxnTimeout();
1352             if (txnTimeout <= 0) {
1353                 blockTxnLatch.await();
1354             } else if (! blockTxnLatch.await(txnTimeout,
1355                                              TimeUnit.MILLISECONDS)) {
1356 
1357                 final String message =
1358                     "Timed out waiting for master transfer. " +
1359                     "Configured transaction timeout:" + txnTimeout + "ms";
1360 
1361                 throw new TransactionTimeoutException(txn, message);
1362             }
1363 
1364             checkIfInvalid();
1365 
1366             /*
1367              * Check again, after the block! The block may be a result of a
1368              * master->replica transfer, and if this node transitions from
1369              * master to replica, this node will be disqualified from being
1370              * able to commit transactions.
1371              */
1372             checkIfMaster(txn);
1373 
1374         } catch (InterruptedException e) {
1375             throw new ThreadInterruptedException(this, e);
1376         }
1377     }

1407                                            int ackTimeout)
1408         throws InsufficientAcksException,
1409                InterruptedException,
1410                EnvironmentFailureException {
1411         if (txn.unlockOnce()) {
1412             blockLatchLock.readLock().unlock();
1413         }

1429         }

1438             }
1439         }
1440

1443             feederTxns.awaitReplicaAcks(txn, ackTimeout);
1444         } catch (InsufficientAcksException e) {
1445             if (txn.getArbiterAck() == false && useArbiter(txn)) {
