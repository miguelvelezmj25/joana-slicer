32 /**
33  * A visitor to visit a Java field. The methods of this class must be called in
34  * the following order: ( <tt>visitAnnotation</tt> |
35  * <tt>visitTypeAnnotation</tt> | <tt>visitAttribute</tt> )* <tt>visitEnd</tt>.
36  * 
37  * @author Eric Bruneton
38  */
39 public abstract class FieldVisitor {
40 
41     /**
42      * The ASM API version implemented by this visitor. The value of this field
43      * must be one of {@link Opcodes#ASM4} or {@link Opcodes#ASM5}.
44      */
45     protected final int api;
46 
47     /**
48      * The field visitor to which this visitor must delegate method calls. May
49      * be null.
50      */
51     protected FieldVisitor fv;
52 
53     /**
54      * Constructs a new {@link FieldVisitor}.
55      * 
56      * @param api
57      *            the ASM API version implemented by this visitor. Must be one
58      *            of {@link Opcodes#ASM4} or {@link Opcodes#ASM5}.
59      */
60     public FieldVisitor(final int api) {
61         this(api, null);
62     }
63 
64     /**
65      * Constructs a new {@link FieldVisitor}.
66      * 
67      * @param api
68      *            the ASM API version implemented by this visitor. Must be one
69      *            of {@link Opcodes#ASM4} or {@link Opcodes#ASM5}.
70      * @param fv
71      *            the field visitor to which this visitor must delegate method
72      *            calls. May be null.
73      */
74     public FieldVisitor(final int api, final FieldVisitor fv) {
75         if (api != Opcodes.ASM4 && api != Opcodes.ASM5) {
76             throw new IllegalArgumentException();
77         }
78         this.api = api;
79         this.fv = fv;
80     }
81 
82     /**
83      * Visits an annotation of the field.
84      * 
85      * @param desc
86      *            the class descriptor of the annotation class.
87      * @param visible
88      *            <tt>true</tt> if the annotation is visible at runtime.
89      * @return a visitor to visit the annotation values, or <tt>null</tt> if
90      *         this visitor is not interested in visiting this annotation.
91      */
92     public AnnotationVisitor visitAnnotation(String desc, boolean visible) {
93         if (fv != null) {
94             return fv.visitAnnotation(desc, visible);
95         }
96         return null;
97     }
98 
99     /**
100      * Visits an annotation on the type of the field.
101      * 
102      * @param typeRef
103      *            a reference to the annotated type. The sort of this type
104      *            reference must be {@link TypeReference#FIELD FIELD}. See
105      *            {@link TypeReference}.
106      * @param typePath
107      *            the path to the annotated type argument, wildcard bound, array
108      *            element type, or static inner type within 'typeRef'. May be
109      *            <tt>null</tt> if the annotation targets 'typeRef' as a whole.
110      * @param desc
111      *            the class descriptor of the annotation class.
112      * @param visible
113      *            <tt>true</tt> if the annotation is visible at runtime.
114      * @return a visitor to visit the annotation values, or <tt>null</tt> if
115      *         this visitor is not interested in visiting this annotation.
116      */
117     public AnnotationVisitor visitTypeAnnotation(int typeRef,
118             TypePath typePath, String desc, boolean visible) {
119         if (api < Opcodes.ASM5) {
120             throw new RuntimeException();
121         }
122         if (fv != null) {
123             return fv.visitTypeAnnotation(typeRef, typePath, desc, visible);
124         }
125         return null;
126     }
127 
128     /**
129      * Visits a non standard attribute of the field.
130      * 
131      * @param attr
132      *            an attribute.
133      */
134     public void visitAttribute(Attribute attr) {
135         if (fv != null) {
136             fv.visitAttribute(attr);
137         }
138     }
139 
140     /**
141      * Visits the end of the field. This method, which is the last one to be
142      * called, is used to inform the visitor that all the annotations and
143      * attributes of the field have been visited.
144      */
145     public void visitEnd() {
146         if (fv != null) {
147             fv.visitEnd();
148         }
149     }
