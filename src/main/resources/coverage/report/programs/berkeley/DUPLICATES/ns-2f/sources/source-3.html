35     /**
36      * Create the restore marker file from Exception origException and return
37      * an EFE that can be thrown by the caller. The EFE will invalidate the
38      * environment.
39      *
40      * @param failureType the failure type that should be recorded in the
41      * RestoreRequired log entry.
42      * @param origException the exception contains the properties that are
43      * stored to the marker file.
44      */
45     public static EnvironmentFailureException createMarkerFileFromException(
46         RestoreRequired.FailureType failureType,
47         Throwable origException,
48         EnvironmentImpl envImpl,
49         EnvironmentFailureReason reason) {
50 
51         String markerFileError = "";
52 
53         /*
54          * If env is read-only (for example when using the DbVerify command
55          * line) we cannot create the marker file, but we should still create
56          * and return an invalidating EFE indicating persistent corruption.
57          */
58         if (!envImpl.isReadOnly()) {
59             final Properties props = new Properties();
60 
61             props.setProperty(
62                 EXCEPTION_KEY, origException.toString() + "\n" +
63                     LoggerUtils.getStackTrace(origException));
64 
65             final RestoreMarker restoreMarker = new RestoreMarker(
66                 envImpl.getFileManager(), envImpl.getLogManager());
67 
68             try {
69                 restoreMarker.createMarkerFile(failureType, props);
70             } catch (RestoreMarker.FileCreationException e) {
71                 markerFileError = " " + e.getMessage();
72             }
73         }
74 
75         return new EnvironmentFailureException(
76             envImpl,
77             reason,
78             "Persistent corruption detected: " + origException.toString() +
79                 markerFileError,
80             origException);
81     }
82 
83     /*
84      * Get a message referencing the original data corruption exception.
85      */
86     public static String getRestoreRequiredMessage(
87         RestoreRequired restoreRequired) {
88 
89         Properties p = restoreRequired.getProperties();
90         return RESTORE_REQUIRED_MSG + p.get(EXCEPTION_KEY);
91     }
