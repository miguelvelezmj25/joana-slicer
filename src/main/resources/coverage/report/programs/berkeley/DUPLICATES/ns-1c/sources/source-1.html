22 import com.sleepycat.je.rep.utilint.NamedChannelWithTimeout;

24 /**
25  * The ChannelTimeoutTask ensures that all channels registered with it are
26  * periodically checked to ensure that they are active. The period roughly
27  * corresponds to a second, although intervening GC activity may expand this
28  * period considerably. Note that elapsedMs used for timeouts is always ticked
29  * up in 1 second increments. Thus multiple seconds of real time may correspond
30  * to a single second of "timer time" if the system is paricularly busy, or the
31  * gc has been particularly active.
32  *
33  * This property allows the underlying timeout implementation to compensate for
34  * GC pauses in which activity on the channel at the java level would have been
35  * suspended and thus reduces the number of false timeouts.
36  */
37 public class ChannelTimeoutTask extends TimerTask {
38 
39     private final long ONE_SECOND_MS = 1000l;
40 
41     /* Elapsed time as measured by the timer task. It's always incremented
42      * in one second intervals.
43      */
44     private long elapsedMs = 0;
45 
46     private final List<NamedChannelWithTimeout> channels =
47         Collections.synchronizedList(new LinkedList<NamedChannelWithTimeout>());
48 
49     /**
50      * Creates and schedules the timer task.
51      * @param timer the timer associated with this task
52      */
53     public ChannelTimeoutTask(Timer timer) {
54         timer.schedule(this, ONE_SECOND_MS, ONE_SECOND_MS);
55     }
56 
57     /**
58      * Runs once a second checking to see if a channel is still active. Each
59      * channel establishes its own timeout period using elapsedMs to check for
60      * timeouts. Inactive channels are removed from the list of registered
61      * channels.
62      */
63     @Override
64     public void run() {
65         elapsedMs += ONE_SECOND_MS;
66         synchronized (channels) {
67             for (Iterator<NamedChannelWithTimeout> i = channels.iterator();
68                  i.hasNext();) {
69                 if (!i.next().isActive(elapsedMs)) {
70                    i.remove();
71                 }
72             }
73         }
74     }
75 
76     /**
77      * Registers a channel so that the timer can make periodic calls to
78      * isActive(). Note that closing a channel renders it inactive and causes
79      * it to be removed from the list by the run() method. Consequently, there
80      * is no corresponding unregister operation.
81      *
82      * @param channel the channel being registered.
83      */
84     public void register(NamedChannelWithTimeout channel) {
85         channels.add(channel);
86     }
87 }
