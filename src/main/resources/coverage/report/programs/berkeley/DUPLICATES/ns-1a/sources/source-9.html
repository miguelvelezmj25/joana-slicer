47     /**
48      * Verifies that the environment config is suitable for a replicated
49      * environment.
50      *
51      * @param envConfig the environment config being checked.
52      *
53      * @throws IllegalArgumentException via ReplicatedEnvironment ctor.
54      */
55     private static void checkEnvConfig(EnvironmentConfig envConfig)
56         throws IllegalArgumentException {
57 
58         if (!envConfig.getTransactional()) {
59             throw new IllegalArgumentException
60                 ("A replicated environment must be transactional");
61         }
62         String logMemOnly = envConfig.getConfigParam
63                             (EnvironmentParams.LOG_MEMORY_ONLY.getName());
64         if (Boolean.parseBoolean(logMemOnly)) {
65             throw new IllegalArgumentException
66                 ("A replicated environment must not log to memory");
67         }
68     }

70     /**
71      * Create a new ReplicationConfig for use in creating Replicator handles.
72      * Be sure to only pick out replication related properties.
73      */
74     public ReplicationConfig makeReplicationConfig() {
75 
76         /*
77          * TODO: the code would be nicer if we replaced the properties bag with
78          * a class that knows how to iterate over the param values, associating
79          * the ConfigParam object with it. As a future task, this is similar to
80          * code in DbConfigManager, and ought to be refactored out. We can also
81          * strip out the multivalued param support.
82          */
83         Properties repProperties = new Properties();
84 
85         /* Check that the properties have valid names and values. */
86         Enumeration<?> propNames = props.propertyNames();
87         while (propNames.hasMoreElements()) {
88             String name = (String) propNames.nextElement();
89             /* Is this a valid property name? */
90             ConfigParam param =
91                 EnvironmentParams.SUPPORTED_PARAMS.get(name);
92 
93             if (param == null) {
94                 /* See if the parameter is an multi-value parameter. */
95                 String mvParamName = ConfigParam.multiValueParamName(name);
96                 param = EnvironmentParams.SUPPORTED_PARAMS.get(mvParamName);
97                 if (param == null) {
98                     throw EnvironmentFailureException.unexpectedState
99                         (name +
100                          " is not a valid BDBJE environment configuration");
101                 }
102             }
103 
104             if (param.isForReplication()) {
105                 repProperties.setProperty(name, props.getProperty(name));
106             }
107         }
108 
109         return RepInternal.makeReplicationConfig
110             (repProperties, validateParams);
111     }
