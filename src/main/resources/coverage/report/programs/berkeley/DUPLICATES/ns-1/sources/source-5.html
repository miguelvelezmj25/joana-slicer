33 /**
34  * Measures performance of a full scan with DiskOrderedCursor compared to a regular cursor walk.
35  *
36  * <p>The DB population and scan are run as two separate invocations, so that the file system cache
37  * can be cleared prior to the scan.
38  *
39  * <p>To populate the DB we insert and update (with the same value) each record to produce a log
40  * that is around 50% utilized, to add realism. With insert-only, a log that is 100% utilized would
41  * result in less real/random IO.
42  *
43  * <p>Variants on scan: - key-only or key-plus-data - with or without duplicates - preload or not
44  *
45  * <p>Other scan parameters: - internalMemoryLimit - lsnBatchSize
46  *
47  * <p>Data set parameters, normally held constant across multiple runs: - writes are in key order or
48  * random order - number of records - key and data size - dups per key (for dup db) - JE cache size
49  *
50  * <p>MeasureDiskOrderedScan.sh (in this directory) is a script for running this program that can be
51  * copied and modified to run a subset of tests.
52  */

254     close();
255   }

295   private void populate() {
296     final DatabaseEntry key = new DatabaseEntry();
297     final DatabaseEntry data = new DatabaseEntry();
298     for (long i = 0; i < nRecords; i += 1) {
299       if (sequentialWrites) {
300         makeLongKey(key, i);
301       } else {
302         makeRandomKey(key);
303       }
304       makeData(data);
305       OperationStatus status;
306       /* Insert */
307       if (dupDb) {
308         status = db.putNoDupData(null, key, data);
309       } else {
310         status = db.putNoOverwrite(null, key, data);
311       }
312       if (status != OperationStatus.SUCCESS) {
313         fail(status);
314       }
315       /* Update to create waste */
316       status = db.put(null, key, data);
317       if (status != OperationStatus.SUCCESS) {
318         fail(status);
319       }
320     }
321   }

323   private void dirtyReadScan() {
324     final DatabaseEntry key = new DatabaseEntry();
325     final DatabaseEntry data = new DatabaseEntry();
326     if (keysOnly) {
327       data.setPartial(0, 0, true);
328     }
329     final Cursor cursor = db.openCursor(null, null);
330     int nScanned = 0;
331     while (cursor.getNext(key, data, LockMode.READ_UNCOMMITTED) == OperationStatus.SUCCESS) {
332       if (sequentialWrites) {
333         checkLongKey(key, nScanned);
334       } else {
335         checkAnyKey(key);
336       }
337       if (!keysOnly) {
338         checkData(data);
339       }
340       nScanned += 1;
341     }
342     cursor.close();
343     checkEquals(nRecords, nScanned);
344   }
