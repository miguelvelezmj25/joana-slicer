100 /**
101  * Statistics for a replicated environment.
102  * <p>
103  * The statistics are logically grouped into four categories. Viewing the
104  * statistics through {@link ReplicatedEnvironmentStats#toString()} displays
105  * the values in these categories, as does viewing the stats through the {@link
106  * <a href="{@docRoot}/../jconsole/JConsole-plugin.html">RepJEMonitor
107  * mbean</a>}.  Viewing the stats with {@link
108  * ReplicatedEnvironmentStats#toStringVerbose()} will provide more detailed
109  * descriptions of the stats and stat categories.
110  * <p>
111  * The current categories are:
112  * <ul>
113  *  <li><b>FeederManager</b>: A feed is the {@link <a
114  * href="{@docRoot}/../ReplicationGuide/introduction.html#replicationstreams">replication
115  * stream</a>} between a master and replica. The current number of feeders
116  * gives a sense of the connectivity of the replication group.
117  *  </li>
118  *  <li><b>BinaryProtocol</b>: These statistics center on the network traffic
119  *   engendered by the replication stream, and provide a sense of the network
120  *   bandwidth seen by the replication group.
121  *  </li>
122  *  <li><b>Replay</b>: The act of receiving and applying the replication stream
123  *  at the Replica node is called Replay. These stats give a sense of how much
124  *  load the replica node is experiencing when processing the traffic from the
125  *  replication group.
126  *  </li>
127  *  <li><b>ConsistencyTracker</b>: The tracker is invoked when consistency
128  *  policies are used at a replica node. This provides a measure of delays
129  *  experienced by read requests at a replica, in order to conform with the
130  *  consistency specified by the application.
131  *  </li>
132  * </ul>
133  *
134  * @see <a href="{@docRoot}/../jconsole/JConsole-plugin.html">Viewing
135  * Statistics with JConsole</a>
136  */

157     ReplicatedEnvironmentStats(RepImpl repImpl, StatsConfig config) {
158         final RepNode repNode = repImpl.getRepNode();
159         final FeederManager feederManager = repNode.feederManager();
160 
161         feederManagerStats = feederManager.getFeederManagerStats(config);
162         feederTxnStats = repNode.getFeederTxns().getStats(config);
163 
164         final Replica replica = repNode.getReplica();
165         replayStats = replica.getReplayStats(config);
166         trackerStats = replica.getTrackerStats(config);
167         protocolStats = feederManager.getProtocolStats(config);
168         vlsnIndexStats = repImpl.getVLSNIndex().getStats(config);
169 
170         protocolStats.addAll(replica.getProtocolStats(config));
171         addMessageRateStats();
172         addBytesRateStats();
173     }
