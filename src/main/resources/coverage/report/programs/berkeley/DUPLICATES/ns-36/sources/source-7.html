378      */
379     public static int readSortedInt(byte[] buf, int off) {
380     
381         int byteLen;
382         boolean negative;
383         
384         /* The first byte of the buf stores the length of the value part. */
385         int b1 = buf[off++] & 0xff;
386         /* Adjust the byteLen to the real length of the value part. */
387         if (b1 < 0x08) {
388             byteLen = 0x08 - b1;
389             negative = true;
390         } else if (b1 > 0xf7) {
391             byteLen = b1 - 0xf7;
392             negative = false;
393         } else {
394             return b1 - 127;
395         }
396         
397         /* 
398          * The following bytes on the buf store the value as a big endian 
399          * integer. We extract the significant bytes from the buf and put them
400          * into the value in big endian order.
401          */
402         int value;
403         if (negative) {
404             value = 0xFFFFFFFF;
405         } else {
406             value = 0;
407         }
408         if (byteLen > 3) {
409             value = (value << 8) | (buf[off++] & 0xFF);
410         } 
411         if (byteLen > 2) {
412             value = (value << 8) | (buf[off++] & 0xFF);
413         }
414         if (byteLen > 1) {
415             value = (value << 8) | (buf[off++] & 0xFF);
416         }
417         value = (value << 8) | (buf[off++] & 0xFF);
418         
419         /* 
420          * After get the adjusted value, we have to adjust it back to the
421          * original value.
422          */
423         if (negative) {
424             value -= 119;
425         } else {
426             value += 121;
427         }
428         return value;

553      * Writes a packed sorted integer starting at the given buffer offset and
554      * returns the next offset to be written.
555      *
556      * @param buf the buffer to write to.
557      *
558      * @param offset the offset in the buffer at which to start writing.
559      *
560      * @param value the integer to be written.
561      *
562      * @return the offset past the bytes written.
563      */
564     public static int writeSortedInt(byte[] buf, int offset, int value) {
565         
566         /* 
567          * Values in the inclusive range [-119,120] are stored in a single 
568          * byte. For values outside that range, the first byte stores the 
569          * number of additional bytes. The additional bytes store 
570          * (value + 119 for negative and value - 121 for positive) as an 
571          * unsigned big endian integer.
572          */
573         int byte1Off = offset;
574         offset++;
575         if (value < -119) {
576         
577             /* 
578              * If the value < -119, then first adjust the value by adding 119.
579              * Then the adjusted value is stored as an unsigned big endian
580              * integer.
581              */
582             value += 119;
583             
584             /* 
585              * Store the adjusted value as an unsigned big endian integer. 
586              * For an negative integer, from left to right, the first 
587              * significant byte is the byte which is not equal to 0xFF. Also 
588              * please note that, because the adjusted value is stored in big
589              * endian integer, we extract the significant byte from left to 
590              * right.
591              *
592              * In the left to right order, if the first byte of the adjusted 
593              * value is a significant byte, it will be stored in the 2nd byte 
594              * of the buf. Then we will look at the 2nd byte of the adjusted 
595              * value to see if this byte is the significant byte, if yes, this 
596              * byte will be stored in the 3rd byte of the buf, and the like.
597              */
598             if ((value | 0x00FFFFFF) != 0xFFFFFFFF) {
599                 buf[offset++] = (byte) (value >> 24);
600             }
601             if ((value | 0x0000FFFF) != 0xFFFFFFFF) {
602                 buf[offset++] = (byte) (value >> 16);
603             }
604             if ((value | 0x000000FF) != 0xFFFFFFFF) {
605                 buf[offset++] = (byte) (value >> 8);
606             }
607             buf[offset++] = (byte) value;
608             
609             /* 
610              * valueLen is the length of the value part stored in buf. Because
611              * the first byte of buf is used to stored the length, we need 
612              * to subtract one.
613              */
614             int valueLen = offset - byte1Off - 1;
615             
616             /* 
617              * The first byte stores the number of additional bytes. Here we
618              * store the result of 0x08 - valueLen, rather than directly store
619              * valueLen. The reason is to implement natural sort order for 
620              * byte-by-byte comparison.
621              */
622             buf[byte1Off] = (byte) (0x08 - valueLen);
623         } else if (value > 120) {
624         
625             /* 
626              * If the value > 120, then first adjust the value by subtracting 
627              * 121. Then the adjusted value is stored as an unsigned big endian
628              * integer.
629              */
630             value -= 121;
631             
632             /* 
633              * Store the adjusted value as an unsigned big endian integer. 
634              * For a positive integer, from left to right, the first 
635              * significant byte is the byte which is not equal to 0x00. 
636              *
637              * In the left to right order, if the first byte of the adjusted 
638              * value is a significant byte, it will be stored in the 2nd byte 
639              * of the buf. Then we will look at the 2nd byte of the adjusted 
640              * value to see if this byte is the significant byte, if yes, this 
641              * byte will be stored in the 3rd byte of the buf, and the like.
642              */
643             if ((value & 0xFF000000) != 0) {
644                 buf[offset++] = (byte) (value >> 24);
645             }
646             if ((value & 0xFFFF0000) != 0) {
647                 buf[offset++] = (byte) (value >> 16);
648             }
649             if ((value & 0xFFFFFF00) != 0) {
650                 buf[offset++] = (byte) (value >> 8);
651             }
652             buf[offset++] = (byte) value;
653             
654             /* 
655              * valueLen is the length of the value part stored in buf. Because
656              * the first byte of buf is used to stored the length, we need to
657              * subtract one.
658              */
659             int valueLen = offset - byte1Off - 1;   
660 
661             /* 
662              * The first byte stores the number of additional bytes. Here we
663              * store the result of 0xF7 + valueLen, rather than directly store
664              * valueLen. The reason is to implement natural sort order for 
665              * byte-by-byte comparison.
666              */
667             buf[byte1Off] = (byte) (0xF7 + valueLen);
668         } else {
669 
670             /* 
671              * If -119 <= value <= 120, only one byte is needed to store the
672              * value. The stored value is the original value plus 127.
673              */
674             buf[byte1Off] = (byte) (value + 127);
675         }
676         return offset;
677     }
