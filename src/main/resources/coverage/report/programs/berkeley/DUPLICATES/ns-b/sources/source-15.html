751     /**
752      * @throws IllegalArgumentException via Environment ctor and
753      * setMutableConfig.
754      */
755     private long calcMaxMemory(DbConfigManager configManager) {
756 
757         /*
758          * Calculate the total memory allotted to JE.
759          * 1. If je.maxMemory is specified, use that. Check that it's not more
760          * than the JVM memory.
761          * 2. Otherwise, take je.maxMemoryPercent * JVM max memory.
762          */
763         long newMaxMemory =
764             configManager.getLong(EnvironmentParams.MAX_MEMORY);
765         long jvmMemory = JVMSystemUtils.getRuntimeMaxMemory();
766 
767         if (newMaxMemory != 0) {
768             /* Application specified a cache size number, validate it. */
769             if (jvmMemory < newMaxMemory) {
770                 throw new IllegalArgumentException
771                     (EnvironmentParams.MAX_MEMORY.getName() +
772                      " has a value of " + newMaxMemory +
773                      " but the JVM is only configured for " +
774                      jvmMemory +
775                      ". Consider using je.maxMemoryPercent.");
776             }
777             if (newMaxMemory < MIN_MAX_MEMORY_SIZE) {
778                 throw new IllegalArgumentException
779                     (EnvironmentParams.MAX_MEMORY.getName() +
780                      " is " + newMaxMemory +
781                      " which is less than the minimum: " +
782                      MIN_MAX_MEMORY_SIZE);
783             }
784         } else {
785 
786             /*
787              * When no explicit cache size is specified and the JVM memory size
788              * is unknown, assume a default sized (64 MB) heap.  This produces
789              * a reasonable cache size when no heap size is known.
790              */
791             if (jvmMemory == Long.MAX_VALUE) {
792                 jvmMemory = N_64MB;
793             }
794 
795             /* Use the configured percentage of the JVM memory size. */
796             int maxMemoryPercent =
797                 configManager.getInt(EnvironmentParams.MAX_MEMORY_PERCENT);
798             newMaxMemory = (maxMemoryPercent * jvmMemory) / 100;
799         }
800 
801         return newMaxMemory;
802     }

804     /**
805      * Initialize at construction time and when the cache is resized.
806      *
807      * @param newMaxMemory is the new total cache budget or is less than 0 if
808      * the total should remain unchanged.
809      *
810      * @param newEnv is true if this is the first time we are resetting the
811      * budget for a new environment.  Note that a new environment has not yet
812      * been added to the set of shared cache environments.
813      */
814     void reset(long newMaxMemory,
815                boolean newEnv,
816                DbConfigManager configManager)
817         throws DatabaseException {
818 
819         long oldLogBufferBudget = logBufferBudget;
820 
821         /*
822          * Update the new total cache budget.
823          */
824         if (newMaxMemory < 0) {
825             newMaxMemory = getMaxMemory();
826         } else {
827             totals.setMaxMemory(newMaxMemory);
828         }
829 
830         /*
831          * This environment's portion is adjusted for a shared cache.  Further
832          * below we make buffer and tracker sizes a fixed percentage (7% and
833          * 2%, by default) of the total shared cache size.  The math for this
834          * starts by dividing the total size by number of environments to get
835          * myCachePortion.  Then we take 7% or 2% of myCachePortion to get each
836          * environment's portion.  In other words, if there are 10 environments
837          * then each gets 7%/10 and 2%/10 of the total cache size, by default.
838          *
839          * Note that when we resize the shared cache, we resize the buffer
840          * pools and tracker budgets for all environments.  Resizing the
841          * tracker budget has no overhead, but resizing the buffer pools causes
842          * new buffers to be allocated.  If reallocation of the log buffers is
843          * not desirable, the user can configure a byte amount rather than a
844          * percentage.
845          */
846         long myCachePortion;
847         if (envImpl.getSharedCache()) {
848             int nEnvs = DbEnvPool.getInstance().getNSharedCacheEnvironments();
849             if (newEnv) {
850                 nEnvs += 1;
851             }
852             myCachePortion = newMaxMemory / nEnvs;
853         } else {
854             myCachePortion = newMaxMemory;
855         }
856 
857         /*
858          * Calculate the memory budget for log buffering.  If the LOG_MEM_SIZE
859          * parameter is not set, start by using 7% (1/16th) of the cache
860          * size. If it is set, use that explicit setting.
861          *
862          * No point in having more log buffers than the maximum size. If
863          * this starting point results in overly large log buffers,
864          * reduce the log buffer budget again.
865          */
866         long newLogBufferBudget =
867             configManager.getLong(EnvironmentParams.LOG_MEM_SIZE);
868         if (newLogBufferBudget == 0) {
869             newLogBufferBudget = myCachePortion >> 4;
870         } else if (newLogBufferBudget > myCachePortion / 2) {
871             newLogBufferBudget = myCachePortion / 2;
872         }
873 
874         /*
875          * We have a first pass at the log buffer budget. See what
876          * size log buffers result. Don't let them be too big, it would
877          * be a waste.
878          */
879         int numBuffers =
880             configManager.getInt(EnvironmentParams.NUM_LOG_BUFFERS);
881         long startingBufferSize = newLogBufferBudget / numBuffers;
882         int logBufferSize =
883             configManager.getInt(EnvironmentParams.LOG_BUFFER_MAX_SIZE);
884         if (startingBufferSize > logBufferSize) {
885             startingBufferSize = logBufferSize;
886             newLogBufferBudget = numBuffers * startingBufferSize;
887         } else if (startingBufferSize <
888                    EnvironmentParams.MIN_LOG_BUFFER_SIZE) {
889             startingBufferSize = EnvironmentParams.MIN_LOG_BUFFER_SIZE;
890             newLogBufferBudget = numBuffers * startingBufferSize;
891         }
892 
893         long newCriticalThreshold =
894             (newMaxMemory *
895              envImpl.getConfigManager().getInt
896                 (EnvironmentParams.EVICTOR_CRITICAL_PERCENTAGE))/100;
897 
898         long newTrackerBudget =
899             (myCachePortion *
900              envImpl.getConfigManager().getInt
901                 (EnvironmentParams.CLEANER_DETAIL_MAX_MEMORY_PERCENTAGE))/100;
902 
903         long newMinTreeMemoryUsage = Math.min
904             (configManager.getLong(EnvironmentParams.MIN_TREE_MEMORY),
905              myCachePortion - newLogBufferBudget);
906 
907         /*
908          * If all has gone well, update the budget fields.  Once the log buffer
909          * budget is determined, the remainder of the memory is left for tree
910          * nodes.
911          */
912         logBufferBudget = newLogBufferBudget;
913         totals.setCriticalThreshold(newCriticalThreshold);
914         trackerBudget = newTrackerBudget;
915         minTreeMemoryUsage = newMinTreeMemoryUsage;
916 
917         /* The log buffer budget is counted in the cache usage. */
918         totals.updateCacheUsage(logBufferBudget - oldLogBufferBudget);
919 
920         /*
921          * Only reset the log buffer pool if the log buffer has already been
922          * initialized (we're updating an existing budget) and the log buffer
923          * budget has changed (resetting it is expensive and may cause I/O).
924          */
925         if (!newEnv && oldLogBufferBudget != logBufferBudget) {
926             envImpl.getLogManager().resetPool(configManager);
927         }
928     }
