84 /**
85  * The Replica class is the locus of the replay operations and replica
86  * transaction consistency tracking and management operations at a replica
87  * node.
88  *
89  * A single instance of this class is created when the replication node is
90  * created and exists for the lifetime of the replication node, although it is
91  * only really used when the node is operating as a Replica.
92  *
93  * Note that the Replica (like the FeederManager) does not have its own
94  * independent thread of control; it runs in the RepNode's thread. To make the
95  * network I/O as aync as possible, and avoid stalls during network I/O the
96  * input and output are done in separate threads. The overall thread
97  * and queue organization is as sketched below:
98  *
99  *  read from network -> RepNodeThread (does read)       -> replayQueue
100  *  replayQueue       -> ReplayThread                     -> outputQueue
101  *  outputQueue       -> ReplicaOutputThread (does write) -> writes to network
102  *
103  * This three thread organization has the following benefits over a single
104  * thread replay model:
105  *
106  * 1) It makes the heartbeat mechanism used to determine whether the HA sockets
107  * are in use more reliable. This is because a heartbeat response cannot
108  * be blocked by lock contention in the replay thread, since a heartbeat
109  * can be sent spontaneously (without an explicit heartbeat request from the
110  * feeder) by the ReplicaOutputThread, if a heartbeat had not been sent during
111  * a heartbeat interval period.
112  *
113  * 2) The cpu load in the replay thread is reduced by offloading the
114  * network-specific aspects of the processing to different threads. It's
115  * important to keep the CPU load in this thread at a minimum so we can use
116  * a simple single thread replay scheme.
117  *
118  * 3) Prevents replay thread stalls by input and output buffering in the two
119  * threads on either side of it.
120  *
121  * With jdk 1.7 we could eliminate the use of these threads and switch over to
122  * the new aysnc I/O APIs, but that involves a lot more code restructuring.
123  */

367      * The core control loop when the node is serving as a Replica. Note that
368      * if a Replica is also serving the role of a feeder, it will run
369      * additional feeder loops in separate threads. The loop exits when it
370      * encounters one of the following possible conditions:
371      *
372      * 1) The connection to the master can no longer be maintained, due to
373      * connectivity issues, or because the master has explicitly shutdown its
374      * connections due to an election.
375      *
376      * 2) The node becomes aware of a change in master, that is, assertSync()
377      * fails.
378      *
379      * 3) The loop is interrupted, which is interpreted as a request to
380      * shutdown the replication node as a whole.
381      *
382      * 4) It fails to establish its node information in the master as it
383      * attempts to join the replication group for the first time.
384      *
385      * Normal exit from this run loop results in the rep node retrying an
386      * election and continuing in its new role as determined by the outcome of
387      * the election. A thrown exception, on the other hand, results in the rep
388      * node as a whole terminating its operation and no longer participating in
389      * the replication group, that is, it enters the DETACHED state.
390      *
391      * Note that the in/out streams are handled synchronously on the replica,
392      * while they are handled asynchronously by the Feeder.
393      *
394      * @throws InterruptedException
395      * @throws DatabaseException if the environment cannot be closed/for a
396      * re-init
397      * @throws GroupShutdownException
398      */

400         throws InterruptedException,
401                DatabaseException,
402                GroupShutdownException {
403

405         int retryCount = 0;
406         try {
407

409                 try {

432                         retryCount = 0;

436                                      retryCount + "/" + e.retries +
437                                      " Will retry replica loop after " +

445                      * Exit replica loop, wait for disk space to become
446                      * available in main rep node loop.
447                      */
448                     break;
449                 }
450             }
451         } finally {
452

542     protected void doRunReplicaLoopInternalWork()
543         throws Exception {
544 
545         final int timeoutMs = repNode.getConfigManager().
546                 getDuration(RepParams.REPLICA_TIMEOUT);
547         replicaFeederChannel.setTimeoutMs(timeoutMs);
548 
549         replayQueue.clear();
550         repImpl.getReplay().reset();
551 
552         replicaOutputThread = new ReplicaOutputThread(repImpl);
553         replicaOutputThread.start();
554 
555         final ReplayThread replayThread = new ReplayThread();
556         replayThread.start();
557         long maxPending = 0;
558 
559         try {
560             while (true) {
561                 final Message message = protocol.read(replicaFeederChannel);
562 
563                 if (repNode.isShutdownOrInvalid() || (message == null)) {
564                     return;
565                 }
566 
567                 /* Throw DiskLimitException if there is a violation. */
568                 repNode.getRepImpl().checkDiskLimitViolation();
569 
570                 while (!replayQueue.
571                         offer(message,
572                               ReplayThread.QUEUE_POLL_INTERVAL_NS,
573                               TimeUnit.NANOSECONDS)) {
574                     /* Offer timed out. */
575                     if (!replayThread.isAlive()) {
576                         return;
577                     }
578                     /* Retry the offer */
579                     nMessageQueueOverflows.increment();
580                 }
581 
582                 final int pending = replayQueue.size();
583                 if (pending > maxPending) {
584                     maxPending = pending;
585                 }
586             }
587         } catch (IOException ioe) {
588 
589             /*
590              * Make sure messages in the queue are processed. Ensure, in
591              * particular, that shutdown requests are processed and not ignored
592              * due to the IOEException resulting from a closed connection.
593              */
594             replayThread.exitRequest = ReplayExitType.SOFT;
595         } finally {
596 
597             if (replayThread.exitRequest == ReplayExitType.SOFT) {
598 
599                 /*
600                  * Drain all queued messages, exceptions may be generated
601                  * in the process. They logically precede IO exceptions.
602                  */
603                 replayThread.join();
604             }
605 
606             try {

638          * Acknowledge the shutdown message right away, since the checkpoint
639          * operation can take a long time to complete. Long enough to exceed
640          * the feeder timeout on the master. The master only needs to know that
641          * the replica has received the message.

648          * longer used past this point and will be reclaimed as part of normal

653         /*
654          * TODO: Share the following code with the standalone Environment
655          * shutdown, or better yet, call EnvironmentImpl.doClose here.
656          */
657 
658         /*
659          * Begin shutdown of the deamons before checkpointing.  Cleaning during
660          * the checkpoint is wasted and slows down the checkpoint, plus it may
661          * cause additional checkpoints.
662          */
663         repNode.getRepImpl().requestShutdownDaemons();

671         config.setMinimizeRecoveryTime(true);

841     /**
842      * Invoked when this node seamlessly changes roles from master to replica
843      * without a recovery. The ability to do this transition without a recovery
844      * is desirable because it's a faster transition, and avoids the GC
845      * overhead of releasing the JE cache, and the I/O overhead of recreating
846      * the in-memory btree.
847      * <p>
848      * The two key cases where this happens are:
849      * A) a network partition occurs, and the group elects a new master. The
850      * orphaned master did not crash and its environment is still valid, and
851      * when it regains contact with the group, it discovers that it has been
852      * deposed. It transitions into a replica status.
853      * <p>
854      * B) a master transfer request moves mastership from this node to another
855      * member of the group. This node's environment is still valid, and it
856      * transitions to replica state.
857      * <p>
858      * The transition from master to replica requires resetting state so all
859      * is as expected for a Replica. There are two categories of work:
860      * - network connections: shutting down feeder connections and
861      * reinitializing feeder infrastructure so that a future replica->master
862      * transition will work.
863      * - resetting transaction state. All MasterTxns must be transformed
864      * into ReplayTxns, bearing the same transaction id and holding the same
865      * locks.
866      * <p>
867      * Note: since non-masters can't commit txns, the inflight MasterTxns are
868      * destined to be aborted in the future. An alternative to resetting
869      * transaction state would be to mark them in some way so that the later HA
870      * syncup/ replay ignores operations pertaining to these ill fated txns. We
871      * didn't chose that approach because the simplicity of the replay is a
872      * plus; it is almost entirely ignorant of the semantics of the replication
873      * stream. Also, replays have potential for complexity, either because
874      * syncups could restart if masters change or become unavailable, or
875      * because there may be future performance optimizations in that area.
876      * <p>
877      * Resetting transaction state is tricky because the MasterTxn is
878      * accessible to the application code. While the Replay thread is
879      * attempting to transform the MasterTxn, application threads may be
880      * attempting to commit or abort the transactions. Note that application
881      * threads will not be trying to add locks, because the node will be in
882      * UNKNOWN state, and writes will be prohibited by the MasterTxn.
883      * <p>
884      * MasterTransfers do impose a blocking period on transaction commits and
885      * aborts, but even there, windows exist in the post-block period where
886      * the application may try to abort the transaction. Network partitions
887      * do no form of blocking, and have a wider window when the application
888      * and RepNode thread must be coordinated. Here's a diagram of the time
889      * periods of concern
890      * <p>
891      * t1 - master transfer request issued (only when master transfer)
892      * t2 - user txns which attempt to abort or commit are blocked on
893      *      RepImpl.blockTxnLatch (only when mt)
894      * t3 - node detects that it has transitioned to UNKNOWN and lost
895      *      master status. MasterTxns are now stopped from acquiring
896      *      locks or committing and will throw UnknownMasterException.
897      * t4 - feeder connections shutdown
898      * t5 - node begins conversion to replica state
899      * t6 - blockTxnLatch released (only when master transfer)
900      * t7 - existing MasterTxns converted into ReplayTxns, locks moved into
901      *      new ReplayTxns. Blocked txns must be released before this
902      *      conversion, because the application thread is holding the
903      *      txn mutex, and conversion needs to take that mutex.
904      * <p>
905      * At any time during this process, the application threads may attempt to
906      * abort or commit outstanding txns, or acquire read or write locks. After
907      * t3, any attempts to lock, abort or commit will throw an
908      * UnknownMasterException or ReplicaWriteException, and in the normal
909      * course of events, the txn would internally abort. But once t5 is
910      * reached, we want to prevent any changes to the number of write locks in
911      * the txn so as to prevent interference with the conversion of the master
912      * txns and any danger of converting only part of a txn. We set the
913      * volatile, transient MasterTxn.freeze field at t5 to indicate that there
914      * should be no change to the contents of the transaction. When freeze is
915      * true, any attempts to abort or commit the transaction will throw
916      * Unknown/ReplicaWriteException, and the txn will be put into MUST_ABORT
917      * state, but the existing locks will be unchanged.
918      * <p>
919      * In a network partition, it's possible that the txn will be aborted or
920      * committed locally before t5. In that case, there may be a hard rollback
921      * when the node syncs up with the new master, and finds the anomalous
922      * abort record. In masterTransfer, the window is smaller, and the blocking
923      * latch ensures that no commits can happen bettween t1-t5. After t5, the
924      * node will not be a master, so there can be no commits. Aborts may happen
925      * and can cause hard rollbacks, but no data will be lost.
926      * <p>
927      * The freeze field is similar to the blockTxnLatch, and we considered
928      * using the blockTxnLatch to stabilize the txns, but ruled it out because:
929      * - the locking hierarchy where the application thread holds the txn
930      *   mutex while awaiting the block txn latch prevents txn conversion.
931      * - the blockTxnLatch is scoped to the MasterTransfer instance, which may
932      *   not be in play for network partitioning.
933      */
934     void replicaTransitionCleanup() {
935 
936         /*
937          * Logically an assert, use an exception rather than Java assert
938          * because we want this check to be enabled at all times. If
939          * unexpectedly in master state, invalidate the environment, so we do a
940          * recovery and are sure to cleanup.
941          */
942         if (repImpl.getState() == State.MASTER) {
943             throw EnvironmentFailureException.unexpectedState(repImpl,
944                 "Should not be in MASTER state when converting from master " +
945                 "to replica state");
946         }
947 
948         /*
949          * Find all MasterTxns, and convert them to ReplayTxns.  The set of
950          * existing MasterTxns cannot increase at this point, because the node
951          * is not in MASTER state. Freeze all txns and prevent change.
952          */
953         Set<MasterTxn> existingMasterTxns = repImpl.getExistingMasterTxns();
954         LoggerUtils.info(logger, repImpl,
955                          "Transitioning node to replica state, " +
956                          existingMasterTxns.size() + " txns to clean up");
957 
958         /* Prevent aborts on all MasterTxns; hold their contents steady */
959         for (MasterTxn masterTxn: existingMasterTxns) {
960             masterTxn.freeze();
961         }
962 
963         /*
964          * Unblock any transactions that are stuck in their commit processing,
965          * awaiting the release of the master transfer block. Such
966          * transactions hold a mutex on the transaction, and the mutex would
967          * block any of the lock stealing that will occur below. Note that if
968          * we are doing this transition because of a network partition, there
969          * will be no blocked transactions.
970          */
971         repImpl.unblockTxnCompletion();
972 
973         for (MasterTxn masterTxn: existingMasterTxns) {
974 
975             /*
976              * Convert this masterTxn to a ReplayTxn and move any existing
977              * write locks to it. Unfreeze and then abort the masterTxn.
978              */
979             ReplayTxn replayTxn =
980                 masterTxn.convertToReplayTxnAndClose(logger,
981                                                      repImpl.getReplay());
982 
983             if (replayTxn == null) {
984                 LoggerUtils.info(logger, repImpl, "Master Txn " +
985                                  masterTxn.getId() +
986                                  " has no locks, nothing to transfer");
987             } else {
988                 repImpl.getTxnManager().registerTxn(replayTxn);
989                 LoggerUtils.info(logger, repImpl,
990                                  "state for replay transaction " +
991                                  replayTxn.getId() + " = " +
992                                  replayTxn.getState());
993             }
994         }
995 
996         /*
997          * We're done with the transition, clear any active master transfers,
998          * if they exist.
999          */
1000         repNode.clearActiveTransfer();
1001     }

1115     /**
1116      * Defines the possible types of exits that can be requested from the
1117      * ReplayThread.
1118      */
1119     private enum ReplayExitType {
1120         IMMEDIATE, /* An immediate exit; ignore queued requests. */
1121         SOFT       /* Process pending requests in queue, then exit */
1122     }
1123 
1124     /**
1125      * The thread responsible for the replay of messages delivered over the
1126      * replication stream. Reading and replay are done in separate threads for
1127      * two reasons:
1128      *
1129      * 1) It allows the two activities to make independent progress. The
1130      * network can be read and messages assembled even if the replay activity
1131      * has stalled. 2) The two threads permit use of two cores to perform the
1132      * replay thus making it less likely that cpu is the replay bottleneck.
1133      *
1134      * The inputs and outputs of this thread are schematically described as:
1135      *
1136      * replayQueue -> ReplayThread -> outputQueue
1137      *
1138      * It's the second component of the three thread structure outlined in the
1139      * Replica's class level comment.
1140      */
1141     class ReplayThread extends StoppableThread {
1142 
1143         /**
1144          * Thread exit exception. It's null if the thread exited due to an
1145          * exception. It's the responsibility of the main replica thread to
1146          * propagate the exception across the thread boundary in this case.
1147          */
1148         volatile private Exception exception;
1149 
1150         /**
1151          * Set asynchronously when a shutdown is being requested.
1152          */
1153         volatile ReplayExitType exitRequest = null;
1154 
1155         /* The queue poll interval, 1 second */
1156         private final static long QUEUE_POLL_INTERVAL_NS = 1000000000l;
1157 
1158         protected ReplayThread() {
1159             super(repImpl, "ReplayThread");
1160         }
1161 
1162         @Override
1163         protected int initiateSoftShutdown() {
1164            /* Use immediate, since the stream will continue to be read. */
1165            exitRequest = ReplayExitType.IMMEDIATE;
1166            return 0;
1167         }
1168 
1169         @Override
1170         public void run() {
1171 
1172             LoggerUtils.info(logger, repImpl,
1173                              "Replay thread started. Message queue size:" +
1174                               replayQueue.remainingCapacity());
1175 
1176             final int dbTreeCacheClearingOpCount =
1177                 repNode.getDbTreeCacheClearingOpCount();
1178 
1179             long opCount = 0;
1180 
1181             try {
1182                 while (true) {
1183 
1184                     final long pollIntervalNs =
1185                         replay.getPollIntervalNs(QUEUE_POLL_INTERVAL_NS);
1186 
1187                     final Message message =
1188                         replayQueue.poll(pollIntervalNs,
1189                                           TimeUnit.NANOSECONDS);
1190 
1191                     if ((exitRequest == ReplayExitType.IMMEDIATE) ||
1192                         ((exitRequest == ReplayExitType.SOFT) &&
1193                          (message == null)) ||
1194                          repNode.isShutdownOrInvalid()) {
1195 
1196                         if (exitRequest == ReplayExitType.SOFT) {
1197                             replay.flushPendingAcks(Long.MAX_VALUE);
1198                         }
1199                         return;
1200                     }
1201 
1202                     final long startNs = System.nanoTime();
1203                     replay.flushPendingAcks(startNs);
1204 
1205                     repNode.getMasterStatus().assertSync();
1206 
1207                     if (message == null) {
1208                         /* Timeout on poll. */
1209                         continue;
1210                     }
1211                     assert TestHookExecute.doHookIfSet(replayHook, message);
1212 
1213                     final MessageOp messageOp = message.getOp();
1214 
1215                     if (messageOp == Protocol.SHUTDOWN_REQUEST) {
1216                         throw processShutdown((ShutdownRequest) message);
1217                     }
1218
