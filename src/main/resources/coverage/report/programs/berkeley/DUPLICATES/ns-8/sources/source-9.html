665                 state == DbState.PREEMPTED || state == null) {
666                 return;

669             /*
670              * databaseImpl and handleLocker are set to null only while
671              * synchronized, at which time the state is also changed to
672              * non-open. So they should not be null here.
673              */
674             dbImpl = databaseImpl;
675             assert dbImpl != null;
676             assert handleLocker != null;

678             /*
679              * Check env only after checking for closed db, to mimic close()
680              * behavior for Cursors, etc, and avoid unnecessary exception
681              * handling.  [#21264]
682              */
683             final EnvironmentImpl envImpl = checkEnv();
684 
685             /*
686              * The state should be changed ASAP during close, so that
687              * addCursor and removeCursor will see the updated state ASAP.
688              */
689             state = newState;
690             preemptedCause = preemptedException;
691 
692             /*
693              * Throw an IllegalStateException if there are open cursors or
694              * associated secondaries.
695              */
696             if (newState == DbState.CLOSED) {

698                     handleRefErrors.append(" ").
699                            append(openCursors.get()).

701                 }

704                     handleRefErrors.append(" ").
705                            append(simpleAssocSecondaries.size()).
706                            append(" associated SecondaryDatabases.");
707                 }

709                     foreignKeySecondaries.size() > 0) {
710                     handleRefErrors.append(" ").
711                            append(foreignKeySecondaries.size()).
712                            append(
713                            " associated foreign key SecondaryDatabases.");
714                 }
715             }
716

718 
719             removeReferringAssociations();

721             dbImpl.removeReferringHandle(this);
722             envImpl.getDbTree().releaseDb(dbImpl);
723 
724             /*
725              * If the handle was preempted, we mark the locker as
726              * only-abortable with the DatabasePreemptedException.  If
727              * the handle locker is a user txn, this causes the
728              * DatabasePreemptedException to be thrown if the user
729              * attempts to commit, or continue to use, the txn, rather
730              * than throwing a LockConflictException.  [#17015]
731              */
732             if (newState == DbState.PREEMPTED) {
733                 handleLocker.setOnlyAbortable(preemptedException);
734             }
735 
736             /*
737              * Tell our protecting txn that we're closing. If this type
738              * of transaction doesn't live beyond the life of the
739              * handle, it will release the db handle lock.
740              */

742                 if (isWritable() && (dbImpl.noteWriteHandleClose() == 0)) {
743                     try {
744                         TriggerManager.runCloseTriggers(handleLocker, dbImpl);
745                     } catch (RuntimeException e) {
746                         triggerException = e;
747                     }
748                 }
749                 handleLocker.operationEnd(true);
750             } else {
751                 handleLocker.operationEnd(false);
752             }
753         }
754 
755         /*
756          * Notify the database when a handle is closed.  This should not be
757          * done while synchronized since it may perform database removal or
758          * sync.  Statements above are synchronized but this section must not
759          * be.
760          *
761          * Note that handleClosed may throw an exception, so any following code
762          * may not be executed.
763          */
764         dbImpl.handleClosed(doSyncDw, deleteTempDb);
765 
766         /* Throw exceptions for previously encountered problems. */
767         if (handleRefErrors.length() > 0) {
768             throw new IllegalStateException(
769                 "Database closed while still referenced by other handles." +
770                 handleRefErrors.toString());
771         }
772         if (triggerException != null) {
773             throw triggerException;
774         }
