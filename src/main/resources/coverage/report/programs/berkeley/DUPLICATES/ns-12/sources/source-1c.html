293      * Log an item, first logging any items on the lazyLogQueue, and finally
294      * flushing and sync'ing (if requested).
295      */
296     public LogItem log(LogParams params)
297         throws DatabaseException {
298         COUNT_LOG.incrementAndGet();
299 
300         final LogItem item = new LogItem();
301 
302         /*
303          * In a read-only env we return NULL_LSN (the default value for
304          * LogItem.lsn) for all entries.  We allow this to proceed, rather
305          * than throwing an exception, to support logging INs for splits that
306          * occur during recovery, for one reason.  Logging LNs in a read-only
307          * env is not allowed, and this is checked in the LN class.
308          */
309         if (readOnly) {
310             return item;
311         }
312 
313         try {
314             /* Flush any pending lazy entries. */
315             for (LazyQueueEntry lqe = lazyLogQueue.poll();
316                  lqe != null;
317                  lqe = lazyLogQueue.poll()) {
318 
319                 LogParams lqeParams = new LogParams();
320                 lqeParams.entry = lqe.entry;
321                 lqeParams.provisional = Provisional.NO;
322                 lqeParams.repContext = lqe.repContext;
323 
324                 logItem(new LogItem(), lqeParams);
325             }
326 
327             final LogEntry logEntry = params.entry;
328 
329             /*
330              * If possible, marshall this entry outside the log write latch to
331              * allow greater concurrency by shortening the write critical
332              * section.  Note that the header may only be created during
333              * marshalling because it calls entry.getSize().
334              */
335             if (logEntry.getLogType().marshallOutsideLatch()) {
336 
337                 item.header = new LogEntryHeader(
338                     logEntry, params.provisional, params.repContext);
339 
340                 item.buffer = marshallIntoBuffer(item.header, logEntry);
341             }
342 
343             logItem(item, params);
344 
345             if (params.fsyncRequired || params.flushRequired) {
346 
347                 /* Flush log buffers and write queue, and optionally fsync. */
348                 grpManager.flushAndSync(params.fsyncRequired);
349 
350             } else if (params.switchedLogBuffer) {
351                 /*
352                  * The operation does not require writing to the log file, but
353                  * since we switched log buffers, this thread will write the
354                  * previously dirty log buffers (not this thread's log entry
355                  * though). This is done for NOSYNC durability so those types
356                  * of transactions won't fill all the log buffers thus forcing
357                  * to have to write the buffers under the log write latch.
358                  */
359                 logBufferPool.writeDirty(false /*flushWriteQueue*/);
360             }
361 
362             TestHookExecute.doHookIfSet(flushHook);
363 
364             /*
365              * We've logged this log entry from the replication stream. Let the
366              * Replicator know, so this node can create a VLSN->LSN mapping. Do
367              * this before the ckpt so we have a better chance of writing this
368              * mapping to disk.
369              */
370             if (params.repContext.inReplicationStream()) {
371 
372                 assert (item.header.getVLSN() != null) :
373                     "Unexpected null vlsn: " + item.header + " " +
374                     params.repContext;
375 
376                 /* Block the VLSN registration, used by unit tests. */
377                 TestHookExecute.doHookIfSet(delayVLSNRegisterHook);
378 
379                 envImpl.registerVLSN(item);
380             }
381 
382         } catch (EnvironmentFailureException e) {
383 
384             /*
385              * Final checks are below for unexpected exceptions during the
386              * critical write path.  Most should be caught by
387              * serialLogInternal, but the catches here account for other
388              * exceptions above.  Note that Errors must be caught here as well
389              * as Exceptions.  [#21929]
390              *
391              * If we've already invalidated the environment, rethrow so as not
392              * to excessively wrap the exception.
393              */
394             if (!envImpl.isValid()) {
395                 throw e;
396             }
397             throw EnvironmentFailureException.unexpectedException(envImpl, e);
398 
399         } catch (Exception e) {
400             throw EnvironmentFailureException.unexpectedException(envImpl, e);
401 
402         } catch (Error e) {
403             envImpl.invalidate(e);
404             throw e;
405         }
406 
407         /*
408          * Periodically, as a function of how much data is written, ask the
409          * checkpointer or the cleaner to wake up.
410          */
411         envImpl.getCheckpointer().wakeupAfterWrite();
412         envImpl.getCleaner().wakeupAfterWrite(item.size);
413 
414         /* Update background writes. */
415         if (params.backgroundIO) {
416             envImpl.updateBackgroundWrites(
417                 item.size, logBufferPool.getLogBufferSize());
418         }
419 
420         return item;
421     }

423     private void logItem(final LogItem item, final LogParams params)
424         throws IOException, DatabaseException {
425 
426         final UtilizationTracker tracker = envImpl.getUtilizationTracker();
427 
428         final LogWriteInfo lwi = serialLog(item, params, tracker);
429 
430         if (lwi != null) {
431 
432             /*
433              * Add checksum, prev offset, and VLSN to the entry.
434              * Copy data into the log buffer.
435              */
436             item.buffer = item.header.addPostMarshallingInfo(
437                 item.buffer, lwi.fileOffset, lwi.vlsn);
438 
439             lwi.lbs.put(item.buffer);
440         }
441 
442         /* Update obsolete info under the LWL */
443         updateObsolete(params, tracker);
444 
445         /* Expiration tracking is protected by the Btree latch, not the LWL. */
446         if (params.expirationTrackerToUse != null) {
447             params.expirationTrackerToUse.track(params.entry, item.size);
448         }
449 
450         /* Queue flushing of expiration tracker after a file flip. */
451         if (params.expirationTrackerCompleted != null) {
452             envImpl.getExpirationProfile().addCompletedTracker(
453                 params.expirationTrackerCompleted);
454         }
455     }

457     /**
458      * This method handles exceptions to be certain that the Environment is
459      * invalidated when any exception occurs in the critical write path, and it
460      * checks for an invalid environment to be sure that no subsequent write is
461      * allowed.  [#21929]
462      *
463      * Invalidation is necessary because a logging operation does not ensure
464      * that the internal state -- correspondence of LSN pointer, log buffer
465      * position and file position, and the integrity of the VLSN index [#20919]
466      * -- is maintained correctly when an exception occurs.  Allowing a
467      * subsequent write can cause log corruption.
468      */
469     private LogWriteInfo serialLog(
470         final LogItem item,
471         final LogParams params,
472         final UtilizationTracker tracker)
473         throws IOException {
474 
475         synchronized (logWriteMutex) {
476 
477             /* Do not attempt to write with an invalid environment. */
478             envImpl.checkIfInvalid();
479 
480             try {
481                 return serialLogWork(item, params, tracker);
482 
483             } catch (EnvironmentFailureException e) {
484                 /*
485                  * If we've already invalidated the environment, rethrow so
486                  * as not to excessively wrap the exception.
487                  */
488                 if (!envImpl.isValid()) {
489                     throw e;
490                 }
491 
492                 /* Otherwise, invalidate the environment. */
493                 throw EnvironmentFailureException.unexpectedException(
494                     envImpl, e);
495 
496             } catch (Exception e) {
497                 throw EnvironmentFailureException.unexpectedException(
498                     envImpl, e);
499 
500             } catch (Error e) {
501                 /* Errors must be caught here as well as Exceptions.[#21929] */
502                 envImpl.invalidate(e);
503                 throw e;
504             }
505         }
506     }

508     /**
509      * This method is used as part of writing data to the log. Called
510      * under the LogWriteLatch.
511      * Data is either written to the LogBuffer or allocates space in the
512      * LogBuffer. The LogWriteInfo object is used to save information about
513      * the space allocate in the LogBuffer. The caller uses the object to
514      * copy data into the underlying LogBuffer. A null value returned
515      * indicates that the item was written to the log. This occurs when the
516      * data item is too big to fit into an empty LogBuffer.
517      *
518      * @param params log params.
519      * @param tracker utilization.
520      * @return a LogWriteInfo object used to access allocated LogBuffer space.
521      *           If null, the data was written to the log.
522      * @throws IOException
523      */
524     private LogWriteInfo serialLogWork(
525         final LogItem item,
526         final LogParams params,
527         final UtilizationTracker tracker)
528         throws IOException {
529 
530         /*
531          * Do obsolete tracking before marshalling a FileSummaryLN into the
532          * log buffer so that a FileSummaryLN counts itself.
533          * countObsoleteNode must be called before computing the entry
534          * size, since it can change the size of a FileSummaryLN entry that
535          * we're logging
536          */
537         final LogEntryType entryType = params.entry.getLogType();
538 
539         if (!DbLsn.isTransientOrNull(params.oldLsn)) {
540             if (params.obsoleteDupsAllowed) {
541                 tracker.countObsoleteNodeDupsAllowed(
542                     params.oldLsn, entryType, params.oldSize, params.nodeDb);
543             } else {
544                 tracker.countObsoleteNode(
545                     params.oldLsn, entryType, params.oldSize, params.nodeDb);
546             }
547         }
548 
549         /* Count auxOldLsn for same database; no specified size. */
550         if (!DbLsn.isTransientOrNull(params.auxOldLsn)) {
551             if (params.obsoleteDupsAllowed) {
552                 tracker.countObsoleteNodeDupsAllowed(
553                     params.auxOldLsn, entryType, 0, params.nodeDb);
554             } else {
555                 tracker.countObsoleteNode(
556                     params.auxOldLsn, entryType, 0, params.nodeDb);
557             }
558         }
559 
560         /*
561          * Compute the VLSNs and modify the DTVLSN in commit/abort entries
562          * before the entry is marshalled or its size is required. At that
563          * at this point we are committed to writing a log entry with  the
564          * computed VLSN.
565          */
566         final VLSN vlsn;
567 
568         if (params.repContext.getClientVLSN() != null ||
569             params.repContext.mustGenerateVLSN()) {
570 
571             if (params.repContext.mustGenerateVLSN()) {
572                 vlsn = envImpl.assignVLSNs(params.entry);
573             } else {
574                 vlsn = params.repContext.getClientVLSN();
575             }
576         } else {
577             vlsn = null;
578         }
579 
580         /*
581          * If an entry must be protected within the log write latch for
582          * marshalling, take care to also calculate its size in the
583          * protected section. Note that we have to get the size *before*
584          * marshalling so that the currentLsn and size are correct for
585          * utilization tracking.
586          */
587         final boolean marshallOutsideLatch = (item.buffer != null);
588         final int entrySize;
589 
590         if (marshallOutsideLatch) {
591             entrySize = item.buffer.limit();
592             assert item.header != null;
593         } else {
594             assert item.header == null;
595             item.header = new LogEntryHeader(
596                 params.entry, params.provisional, params.repContext);
597             entrySize = item.header.getEntrySize();
598         }
599 
600         /*
601          * Get the next free slot in the log, under the log write latch.
602          */
603         if (params.forceNewLogFile) {
604             fileManager.forceNewLogFile();
605         }
606 
607         final boolean flippedFile = fileManager.shouldFlipFile(entrySize);
608         final long currentLsn = fileManager.calculateNextLsn(flippedFile);
609 
610         /*
611          * TODO: Count file header, since it is not logged via LogManager.
612          * Some tests (e.g., INUtilizationTest) will need to be adjusted.
613          *
614         final int fileHeaderSize = FileManager.firstLogEntryOffset();
615         if (DbLsn.getFileOffset(currentLsn) == fileHeaderSize) {
616             final long fileNum = DbLsn.getFileNumber(currentLsn);
617 
618             tracker.countNewLogEntry(
619                 DbLsn.makeLsn(fileNum, 0), LogEntryType.LOG_FILE_HEADER,
620                 fileHeaderSize, null);
621         }
622         */
623 
624         /*
625          * countNewLogEntry and countObsoleteNodeInexact cannot change
626          * a FileSummaryLN size, so they are safe to call after
627          * getSizeForWrite.
628          */
629         tracker.countNewLogEntry(
630             currentLsn, entryType, entrySize, params.nodeDb);
631 
632         /*
633          * LN deletions and dup DB LNs are obsolete immediately.  Inexact
634          * counting is used to save resources because the cleaner knows
635          * that all such LNs are obsolete.
636          */
637         if (params.entry.isImmediatelyObsolete(params.nodeDb)) {
638             tracker.countObsoleteNodeInexact(
639                 currentLsn, entryType, entrySize, params.nodeDb);
640         }
641 
642         /*
643          * This entry must be marshalled within the log write latch.
644          */
645         if (!marshallOutsideLatch) {
646             assert item.buffer == null;
647             item.buffer = marshallIntoBuffer(item.header, params.entry);
648         }
649 
650         /* Sanity check */
651         if (entrySize != item.buffer.limit()) {
652             throw EnvironmentFailureException.unexpectedState(
653                 "Logged entry entrySize= " + entrySize +
654                 " but marshalledSize=" + item.buffer.limit() +
655                 " type=" + entryType + " currentLsn=" +
656                 DbLsn.getNoFormatString(currentLsn));
657         }
658 
659         /*
660          * Ask for a log buffer suitable for holding this new entry. If
661          * entrySize is larger than the LogBuffer capacity, this will flush
662          * all dirty buffers and return the next empty (but too small) buffer.
663          * The returned buffer is not latched.
664          */
665         final LogBuffer lastLogBuffer =
666             logBufferPool.getWriteBuffer(entrySize, flippedFile);
667 
668         /*
669          * Bump the LSN values, which gives us a valid previous pointer,
670          * which is part of the log entry header. This must be done:
671          *  - before logging the currentLsn.
672          *  - after calling getWriteBuffer, to flush the prior file when
673          *    flippedFile is true.
674          */
675         final long prevOffset = fileManager.advanceLsn(
676             currentLsn, entrySize, flippedFile);
677 
678         if (lastLogBuffer != prevLogBuffer) {
679             params.switchedLogBuffer = true;
680         }
681         prevLogBuffer = lastLogBuffer;
682 
683         final LogBufferSegment useBuffer;
684 
685         lastLogBuffer.latchForWrite();
686         try {
687             useBuffer = lastLogBuffer.allocate(entrySize);
688 
689             if (useBuffer != null) {
690                 /* Register the lsn while holding the buffer latch. */
691                 lastLogBuffer.registerLsn(currentLsn);
692             } else {
693                 /*
694                  * The item buffer is larger than the LogBuffer capacity, so
695                  * write the item buffer to the file directly. Note that
696                  * getWriteBuffer has flushed all dirty buffers.
697                  *
698                  * First add checksum, prev offset, and VLSN to the entry.
699                  */
700                 item.buffer = item.header.addPostMarshallingInfo(
701                     item.buffer, prevOffset, vlsn);
702 
703                 final boolean flushWriteQueue =
704                     params.flushRequired && !params.fsyncRequired;
705 
706                 fileManager.writeLogBuffer(
707                     new LogBuffer(item.buffer, currentLsn),
708                     flushWriteQueue);
709 
710                 assert lastLogBuffer.getDataBuffer().position() == 0;
711 
712                 /* Leave a clue that the buffer size needs to be increased. */
713                 nTempBufferWrites.increment();
714             }
715         } finally {
716             lastLogBuffer.release();
717         }
718 
719         /*
720          * If the txn is not null, the first entry is an LN. Update the txn
721          * with info about the latest LSN. Note that this has to happen
722          * within the log write latch.
723          */
724         params.entry.postLogWork(item.header, currentLsn, vlsn);
725 
726         item.lsn = currentLsn;
727         item.size = entrySize;
728 
729         /* If the expirationTracker field is null, no tracking should occur. */
730         if (expirationTracker != null) {
731             /*
732              * When logging to a new file, also flip the expirationTracker
733              * under the LWL and return expirationTrackerCompleted so it will
734              * be queued for flushing.
735              */
736             final long newFile = DbLsn.getFileNumber(item.lsn);
737             if (flippedFile && newFile != expirationTracker.getFileNum()) {
738                 params.expirationTrackerCompleted = expirationTracker;
739                 expirationTracker = new ExpirationTracker(newFile);
740             }
741             /*
742              * Increment the pending calls under the LWL, so we can determine
743              * when we're finished.
744              */
745             expirationTracker.incrementPendingTrackCalls();
746             params.expirationTrackerToUse = expirationTracker;
747         }
748 
749         return (useBuffer == null ?
750                 null : new LogWriteInfo(useBuffer, vlsn, prevOffset));
751     }

754      * Serialize a loggable object into this buffer.
755      */
756     private ByteBuffer marshallIntoBuffer(LogEntryHeader header,
757                                           LogEntry entry) {
758         int entrySize = header.getSize() + header.getItemSize();
759 
760         ByteBuffer destBuffer = ByteBuffer.allocate(entrySize);
761         header.writeToLog(destBuffer);
762

764         entry.writeEntry(destBuffer);
765 
766         /* Set the limit so it can be used as the size of the entry. */
767         destBuffer.flip();
768 
769         return destBuffer;

836     /**
837      * Instantiate all the objects in the log entry at this LSN.
838      * @param lsn location of entry in log.
839      * @param invisibleReadAllowed true if it's expected that the target log
840      * entry might be invisible. Correct the known-to-be-bad checksum before
841      * proceeding.
842      * @return log entry that embodies all the objects in the log entry.
843      */
844     private WholeEntry getLogEntry(
845         long lsn,
846         int lastLoggedSize,
847         boolean invisibleReadAllowed)
848         throws FileNotFoundException {
849 
850         /* Fail loudly if the environment is invalid. */
851         envImpl.checkIfInvalid();
852 
853         LogSource logSource = null;
854         try {
855 
856             /*
857              * Get a log source for the log entry which provides an abstraction
858              * that hides whether the entry is in a buffer or on disk. Will
859              * register as a reader for the buffer or the file, which will take
860              * a latch if necessary. Latch is released in finally block.
861              */
862             logSource = getLogSource(lsn);
863 
864             try {
865                 return getLogEntryFromLogSource(
866                     lsn, lastLoggedSize, logSource, invisibleReadAllowed);
867 
868             } catch (ChecksumException ce) {
869 
870                 /*
871                  * When using a FileSource, a checksum error indicates a
872                  * persistent corruption. An EFE with LOG_CHECKSUM is created
873                  * in the catch below and EFE.isCorrupted will return true.
874                  */
875                 if (!(logSource instanceof LogBuffer)) {
876                     assert logSource instanceof FileSource;
877                     throw ce;
878                 }
879 
880                 /*
881                  * When using a LogBuffer source, we must try to read the entry
882                  * from disk to see if the corruption is persistent.
883                  */
884                 final LogBuffer logBuffer = (LogBuffer) logSource;
885                 FileHandle fileHandle = null;
886                 long fileLength = -1;
887                 try {
888                     fileHandle =
889                         fileManager.getFileHandle(DbLsn.getFileNumber(lsn));
890                     fileLength = fileHandle.getFile().length();
891                 } catch (IOException ioe) {
892                     /* FileNotFound or another IOException was thrown. */
893                 }
894 
895                 /*
896                  * If the file does not exist (FileNotFoundException is thrown
897                  * above) or the firstLsn in the buffer does not appear in the
898                  * file (the buffer was not flushed), then the corruption is
899                  * not persistent and we throw a EFE for which isCorrupted
900                  * will return false.
901                  */
902                 if (fileHandle == null ||
903                     fileLength <=
904                         DbLsn.getFileOffset(logBuffer.getFirstLsn())) {
905 
906                     throw EnvironmentFailureException.unexpectedException(
907                         envImpl,
908                         "Corruption detected in log buffer, " +
909                             "but was not written to disk.",
910                         ce);
911                 }
912 
913                 /*
914                  * The log entry should have been written to the file. Try
915                  * getting the log entry from the FileSource. If a
916                  * ChecksumException is thrown, the corruption is persistent
917                  * and an EFE with LOG_CHECKSUM is thrown below.
918                  */
919                 final FileSource fileSource = new FileHandleSource(
920                     fileHandle, readBufferSize, fileManager);
921                 try {
922                     return getLogEntryFromLogSource(
923                         lsn, lastLoggedSize, fileSource,
924                         invisibleReadAllowed);
925                 } finally {
926                     fileSource.release();
927                 }
928             }
929 
930         } catch (ChecksumException e) {
931             /*
932              * WARNING: EFE with LOG_CHECKSUM indicates a persistent corruption
933              * and therefore LogSource.release must not be called until after
934              * invalidating the environment (in the finally below). The buffer
935              * latch prevents the corrupt buffer from being logged by another
936              * thread.
937              */
938             throw VerifierUtils.createMarkerFileFromException(
939                 RestoreRequired.FailureType.LOG_CHECKSUM,
940                 e,
941                 envImpl,
942                 EnvironmentFailureReason.LOG_CHECKSUM);
943 
944         } catch (Error e) {
945             envImpl.invalidate(e);
946             throw e;
947 
948         } finally {
949             if (logSource != null) {
950                 logSource.release();
951             }
952         }
953     }

1017     /**
1018      * Gets log entry from the given source; the caller is responsible for
1019      * calling logSource.release and handling ChecksumException.
1020      *
1021      * Is non-private for unit testing.
1022      *
1023      * @param lsn location of entry in log
1024      * @param lastLoggedSize is the entry size if known, or zero if unknown.
1025      * @param invisibleReadAllowed if true, we will permit the read of invisible
1026      * log entries, and we will adjust the invisible bit so that the checksum
1027      * will validate
1028      * @return log entry that embodies all the objects in the log entry
1029      */
1030     WholeEntry getLogEntryFromLogSource(long lsn,
1031                                         int lastLoggedSize,
1032                                         LogSource logSource,
1033                                         boolean invisibleReadAllowed)
1034         throws ChecksumException, DatabaseException {
1035 
1036         /*
1037          * Read the log entry header into a byte buffer. If the
1038          * lastLoggedSize is available (non-zero), we can use it to avoid a
1039          * repeat-read further below. Otherwise we use the configured
1040          * LOG_FAULT_READ_SIZE, and a repeat-read may occur if the log
1041          * entry is larger than the buffer.
1042          *
1043          * Even when lastLoggedSize is non-zero, we do not assume that it
1044          * is always accurate, because this is not currently guaranteed
1045          * in corner cases such as transaction aborts. We do the initial
1046          * read with lastLoggedSize. If lastLoggedSize is larger than the
1047          * actual size, we will simply read more bytes than needed. If
1048          * lastLoggedSize is smaller than the actual size, we will do a
1049          * repeat-read further below.
1050          */
1051         long fileOffset = DbLsn.getFileOffset(lsn);
1052 
1053         ByteBuffer entryBuffer = (lastLoggedSize > 0) ?
1054             logSource.getBytes(fileOffset, lastLoggedSize) :
1055             logSource.getBytes(fileOffset);
1056 
1057         if (entryBuffer.remaining() < LogEntryHeader.MIN_HEADER_SIZE) {
1058             throw new ChecksumException(
1059                 "Incomplete log entry header in " + logSource +
1060                 " needed=" + LogEntryHeader.MIN_HEADER_SIZE +
1061                 " remaining=" + entryBuffer.remaining() +
1062                 " lsn=" + DbLsn.getNoFormatString(lsn));
1063         }
1064 
1065         /* Read the fixed length portion of the header. */
1066         LogEntryHeader header = new LogEntryHeader(
1067             entryBuffer, logSource.getLogVersion(), lsn);
1068 
1069         /* Read the variable length portion of the header. */
1070         if (header.isVariableLength()) {
1071             if (entryBuffer.remaining() <
1072                 header.getVariablePortionSize()) {
1073                 throw new ChecksumException(
1074                     "Incomplete log entry header in " + logSource +
1075                     " needed=" + header.getVariablePortionSize() +
1076                     " remaining=" + entryBuffer.remaining() +
1077                     " lsn=" + DbLsn.getNoFormatString(lsn));
1078             }
1079             header.readVariablePortion(entryBuffer);
1080         }
1081 
1082         ChecksumValidator validator = null;
1083         if (doChecksumOnRead) {
1084             int itemStart = entryBuffer.position();
1085 
1086             /*
1087              * We're about to read an invisible log entry, which has knowingly
1088              * been left on disk with a bad checksum. Flip the invisible bit in
1089              * the backing byte buffer now, so the checksum will be valid. The
1090              * LogEntryHeader object itself still has the invisible bit set,
1091              * which is useful for debugging.
1092              */
1093             if (header.isInvisible()) {
1094                 LogEntryHeader.turnOffInvisible
1095                     (entryBuffer, itemStart - header.getSize());
1096             }
1097 
1098             /* Add header to checksum bytes */
1099             validator = new ChecksumValidator();
1100             int headerSizeMinusChecksum = header.getSizeMinusChecksum();
1101             entryBuffer.position(itemStart -
1102                                  headerSizeMinusChecksum);
1103             validator.update(entryBuffer, headerSizeMinusChecksum);
1104             entryBuffer.position(itemStart);
1105         }
1106 
1107         /*
1108          * Now that we know the size, read the rest of the entry if the first
1109          * read didn't get enough.
1110          */
1111         int itemSize = header.getItemSize();
1112         if (entryBuffer.remaining() < itemSize) {
1113             entryBuffer = logSource.getBytes(
1114                 fileOffset + header.getSize(), itemSize);
1115             if (entryBuffer.remaining() < itemSize) {
1116                 throw new ChecksumException(
1117                     "Incomplete log entry item in " + logSource +
1118                     " needed=" + itemSize +
1119                     " remaining=" + entryBuffer.remaining() +
1120                     " lsn=" + DbLsn.getNoFormatString(lsn));
1121             }
1122             nRepeatFaultReads.increment();
1123         }
1124 
1125         /*
1126          * Do entry validation. Run checksum before checking the entry type, it
1127          * will be the more encompassing error.
1128          */
1129         if (doChecksumOnRead) {
1130             /* Check the checksum first. */
1131             validator.update(entryBuffer, itemSize);
1132             validator.validate(header.getChecksum(), lsn);
1133         }
1134 
1135         /*
1136          * If invisibleReadAllowed == false, we should not be fetching an
1137          * invisible log entry.
1138          */
1139         if (header.isInvisible() && !invisibleReadAllowed) {
1140             throw new EnvironmentFailureException
1141                 (envImpl, EnvironmentFailureReason.LOG_INTEGRITY,
1142                  "Read invisible log entry at " +
1143                  DbLsn.getNoFormatString(lsn) + " " + header);
1144         }
1145 
1146         assert LogEntryType.isValidType(header.getType()):
1147             "Read non-valid log entry type: " + header.getType();
1148 
1149         /* Read the entry. */
1150         LogEntry logEntry =
1151             LogEntryType.findType(header.getType()).getNewLogEntry();
1152         logEntry.readEntry(envImpl, header, entryBuffer);
1153 
1154         /* For testing only; generate a read io exception. */
1155         if (readHook != null) {
1156             try {
1157                 readHook.doIOHook();
1158             } catch (IOException e) {
1159                 /* Simulate what the FileManager would do. */
1160                 throw new EnvironmentFailureException
1161                     (envImpl, EnvironmentFailureReason.LOG_READ, e);
1162             }
1163         }
1164 
1165         return new WholeEntry(header, logEntry);
1166     }
