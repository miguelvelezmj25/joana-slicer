651         if (triggerBytes == null && transientTriggers == null) {
652             return null;
653         }
654 
655         /* Just return them, if already constructed. */
656         List<Trigger> myTriggers = triggers.get();
657         if (myTriggers != null) {
658             return myTriggers;
659         }
660 
661         /*
662          * Unmarshall triggers, add transient triggers, and update the
663          * reference atomically. If another thread unmarshalls and updates it
664          * first, use the value set by the other thread.  This ensures that a
665          * single instance is always used.
666          */
667         myTriggers = TriggerUtils.unmarshallTriggers(getName(), triggerBytes,
668                                                      envImpl.getClassLoader());
669         if (myTriggers == null) {
670             myTriggers = new LinkedList<Trigger>();
671         }
672         if (transientTriggers != null) {
673             myTriggers.addAll(transientTriggers);
674         }
675         if (triggers.compareAndSet(null, myTriggers)) {
676             return myTriggers;
677         }
678         myTriggers = triggers.get();
679         assert myTriggers != null;
680         return myTriggers;
681     }

886      * @param dbName pass it in since it may not be available during database
887      * creation
888      * @param newTriggers the triggers to associate with the database
889      * @param overridePersistentTriggers whether to overwrite persistent
890      * triggers
891      *
892      * @return true if a {@link PersistentTrigger} was changed, and therefore
893      * may need to be stored.
894      */
895     public boolean setTriggers(Locker locker,
896                                String dbName,
897                                List<Trigger> newTriggers,
898                                boolean overridePersistentTriggers) {
899 
900         if ((newTriggers != null) && (newTriggers.size() == 0)) {
901             newTriggers = null;
902         }
903 
904         /* Construct new persistent triggers. */
905         final byte newTriggerBytes[][];
906         final boolean persistentChange;
907 
908         if (overridePersistentTriggers) {
909             if (newTriggers == null) {
910                 newTriggerBytes = null;
911                 persistentChange = (this.triggerBytes != null);
912             } else {
913                 /* Create the new trigger bytes. */
914                 int nTriggers = 0;
915                 for (Trigger trigger : newTriggers) {
916                     if (trigger instanceof PersistentTrigger) {
917                         nTriggers += 1;
918                     }
919                 }
920                 if (nTriggers == 0) {
921                     newTriggerBytes = null;
922                     persistentChange = (this.triggerBytes != null);
923                 } else {
924                     newTriggerBytes = new byte[nTriggers][];
925                     int i=0;
926                     for (Trigger trigger : newTriggers) {
927                         if (trigger instanceof PersistentTrigger) {
928                             newTriggerBytes[i++] = objectToBytes
929                                 (trigger, "trigger " + trigger.getName());
930                             trigger.setDatabaseName(dbName);
931                         }
932                     }
933                     persistentChange =
934                         !Arrays.equals(triggerBytes, newTriggerBytes);
935                 }
936             }
937         } else {
938             newTriggerBytes = triggerBytes;
939             persistentChange = false;
940         }
941 
942         /* Add transient triggers. */
943         final List<Trigger> newTransientTriggers;
944         final boolean transientChange;
945 
946         if (newTriggers == null) {
947             newTransientTriggers = null;
948             transientChange = (transientTriggers != null);
949         } else {
950             newTransientTriggers = new LinkedList<Trigger>();
951             final Map<Trigger, Object> diffs =
952                 new IdentityHashMap<Trigger, Object>();
953             for (Trigger trigger : newTriggers) {
954                 if (!(trigger instanceof PersistentTrigger)) {
955                     diffs.put(trigger, null);
956                     newTransientTriggers.add(trigger);
957                     trigger.setDatabaseName(dbName);
958                 }
959             }
960             if (transientTriggers == null) {
961                 transientChange = (newTransientTriggers.size() > 0);
962             } else if (transientTriggers.size() !=
963                        newTransientTriggers.size()) {
964                 transientChange = true;
965             } else {
966                 for (Trigger trigger : transientTriggers) {
967                     diffs.remove(trigger);
968                 }
969                 transientChange = (diffs.size() > 0);
970             }
971         }
972 
973         if (persistentChange || transientChange) {
974             TriggerManager.invokeAddRemoveTriggers(locker,
975                                                    getTriggers(),
976                                                    newTriggers);
977             /* Don't change fields until after getTriggers() call above. */
978             triggerBytes = newTriggerBytes;
979             transientTriggers =
980                 ((newTransientTriggers != null) &&
981                  (newTransientTriggers.size() > 0)) ?
982                 newTransientTriggers :
983                 null;
984             this.triggers.set(newTriggers);
985         }
986 
987         return persistentChange;
988     }

1201     /**
1202      * Called after a handle onto this DB is closed.
1203      */
1204     public void handleClosed(boolean doSyncDw, boolean deleteTempDb)
1205         throws DatabaseException {
1206 
1207         if (referringHandles.isEmpty()) {
1208 
1209             /*
1210              * Transient triggers are discarded when the last handle is
1211              * closed.
1212              */
1213             clearTransientTriggers();
1214 
1215             /*
1216              * Remove a temporary database with no handles open.
1217              *
1218              * We are not synchronized here in any way that would prevent
1219              * another thread from opening a handle during this process, before
1220              * the NameLN is locked.  So we use noWait locking.  If a lock is
1221              * not granted, then another handle was opened and we cannot remove
1222              * the database until later.
1223              *
1224              * We pass the database ID to dbRemove in order to remove the
1225              * database only if the name matches the ID.  This accounts for the
1226              * remote possibility that the database is renamed or another
1227              * database is created with the same name during this process,
1228              * before the NameLN is locked.
1229              *
1230              * We can use a BasicLocker because temporary databases are always
1231              * non-transactional.
1232              */
1233             if (deleteTempDb && isTemporary()) {
1234                 Locker locker =
1235                     BasicLocker.createBasicLocker(envImpl, true /* noWait */);
1236                 boolean operationOk = false;
1237                 try {
1238                     envImpl.getDbTree().dbRemove(locker, getName(), getId());
1239                     operationOk = true;
1240                 } catch (DbTree.NeedRepLockerException e) {
1241                     /* Should never happen; a temp db is never replicated. */
1242                     throw EnvironmentFailureException.unexpectedException(
1243                         envImpl, e);
1244                 } catch (DatabaseNotFoundException e) {
1245                     /* Do nothing if DB was removed or renamed. */
1246                 } catch (LockConflictException e) {
1247                     /*
1248                      * We will have to remove this database later.  Note that
1249                      * we catch LockConflictException for simplicity but we
1250                      * expect either LockNotAvailableException or
1251                      * LockNotGrantedException.
1252                      */
1253                 } catch (Error E) {
1254                     envImpl.invalidate(E);
1255                     throw E;
1256                 } finally {
1257                     locker.operationEnd(operationOk);
1258                 }
1259             }
1260 
1261             /*
1262              * Sync a durable deferred write database with no handles open.  If
1263              * a handle is opened during this process, then the sync may be
1264              * unnecessary but it will not cause a problem.
1265              */
1266             if (doSyncDw && isDurableDeferredWrite()) {
1267                 sync(true);
1268             }
1269         }
1270     }
