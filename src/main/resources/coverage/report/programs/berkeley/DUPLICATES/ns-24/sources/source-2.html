304     /**
305      * Determines whether we should lock the block-latch lock.
306      * <p>
307      * We acquire the lock during pre-log hook, and release it during post-log
308      * hook.  Specifically, there are the following cases:
309      * <ol>
310      * <li>
311      * For a normal commit, we acquire it in {@code preLogCommitHook()} and
312      * release it in {@code postLogCommitHook()}
313      * <li>
314      * For a normal abort (invoked by the application on the {@code
315      * Txn.abort()} API), we acquire the lock in {@code preLogAbortHook()} and
316      * release it in {@code postLogAbortHook()}.
317      * <li>
318      * When a commit fails in such a way as to call {@code
319      * Txn.throwPreCommitException()}, we go through the abort path as well.
320      * In this case:
321      * <ul>
322      * <li>we will of course already have called {@code preLogCommitHook()};
323      * <li>the abort path calls {@code preLogAbortHook()} and {@code
324      * postLogAbortHook()} as always;
325      * <li>finally we call {@code postLogCommitAbortHook()}
326      * </ul>
327      * Fortunately we can avoid the complexity of dealing with a second
328      * (recursive) lock acquisition here, because by the time either post-hook
329      * is called we've done any writing of VLSNs.  Thus, when we want to
330      * take the lock, we take it if it hasn't already been taken, and do
331      * nothing if it has; when releasing, we release it if we have it, and do
332      * nothing if we don't.
333      * </ol>
334      * <p>
335      * See additional javadoc at {@code RepImpl.blockLatchLock}
336      */
337     public boolean lockOnce() {
338         if (locked) {
339             return false;
340         }
341         locked = true;
342         return true;
343     }

454     /**
455      * Convert a MasterTxn that has any write locks into a ReplayTxn, and close
456      * the MasterTxn after it is disemboweled. A MasterTxn that only has read
457      * locks is unchanged and is still usable by the application. To be clear,
458      * the application can never use a MasterTxn to obtain a lock if the node
459      * is in Replica mode, but may indeed be able to use a read-lock-only
460      * MasterTxn if the node cycles back into Master status.
461      *
462      * For converted MasterTxns, all write locks are transferred to a replay
463      * transaction, read locks are released, and the txn is closed. Used when a
464      * node is transitioning from master to replica mode without recovery,
465      * which may happen for an explicit master transfer request, or merely for
466      * a network partition/election of new
467      * master.
468      *
469      * The newly created replay transaction will need to be in the appropriate
470      * state, holding all write locks, so that the node in replica form can
471      * execute the proper syncups.  Note that the resulting replay txn will
472      * only be aborted, and will never be committed, because the txn originated
473      * on this node, which is transitioning from master -> replica.
474      *
475      * We only transfer write locks. We need not transfer read locks, because
476      * replays only operate on writes, and are never required to obtain read
477      * locks. Read locks are released though, because (a) this txn is now only
478      * abortable, and (b) although the Replay can preempt any read locks held
479      * by the MasterTxn, such preemption will add delay.
480      *
481      * @return a ReplayTxn, if there were locks in this transaction, and
482      * there's a need to create a ReplayTxn.
483      */
484     public ReplayTxn convertToReplayTxnAndClose(Logger logger,
485                                                 Replay replay) {
486 
487         /* Assertion */
488         if (!freeze) {
489             throw EnvironmentFailureException.unexpectedState
490                 (envImpl,
491                  "Txn " + getId() +
492                  " should be frozen when converting to replay txn");
493         }
494 
495         /*
496          * This is an important and relatively rare operation, and worth
497          * logging.
498          */
499         LoggerUtils.info(logger, envImpl,
500                          "Transforming txn " + getId() +
501                          " from MasterTxn to ReplayTxn");
502 
503         int hookCount = 0;
504         ReplayTxn replayTxn = null;
505         boolean needToClose = true;
506         try {
507             synchronized (this) {
508 
509                 if (isClosed()) {
510                     LoggerUtils.info(logger, envImpl,
511                                      "Txn " + getId() +
512                                      " is closed, no tranform needed");
513                     needToClose = false;
514                     return null;
515                 }
516 
517                 /*
518                  * Get the list of write locks, and process them in lsn order,
519                  * so we properly maintain the lastLoggedLsn and firstLoggedLSN
520                  * fields of the newly created ReplayTxn.
521                  */
522                 final Set<Long> lockedLSNs = getWriteLockIds();
523 
524                 /*
525                  * This transaction may not actually have any write locks. In
526                  * that case, we permit it to live on.
527                  */
528                 if (lockedLSNs.size() == 0) {
529                     LoggerUtils.info(logger, envImpl, "Txn " + getId() +
530                                      " had no write locks, didn't create" +
531                                      " ReplayTxn");
532                     needToClose = false;
533                     return null;
534                 }
535 
536                 /*
537                  * We have write locks. Make sure that this txn can now
538                  * only be aborted. Otherwise, there could be this window
539                  * in this method:
540                  *  t1: locks stolen, no locks left in this txn
541                  *  t2: txn unfrozen, commits and aborts possible
542                  *    -- at this point, another thread could sneak in and
543                  *    -- try to commit. The txn would commmit successfully,
544                  *    -- because a commit w/no write locks is a no-op.
545                  *    -- but that would convey the false impression that the
546                  *    -- txn's write operations had commmitted.
547                  *  t3: txn is closed
548                  */
549                 setOnlyAbortable(new UnknownMasterException
550                                  (envImpl.getName() +
551                                   " is no longer a master"));
552                 replayTxn = replay.getReplayTxn(getId(), false);
553 
554                 /*
555                  * Order the lsns, so that the locks are taken in the proper
556                  * order, and the txn's firstLoggedLsn and lastLoggedLsn fields
557                  * are properly set.
558                  */
559                 List<Long> sortedLsns = new ArrayList<>(lockedLSNs);
560                 Collections.sort(sortedLsns);
561                 LoggerUtils.info(logger, envImpl,
562                                  "Txn " + getId()  + " has " +
563                                  lockedLSNs.size() + " locks to transform");
564 
565                 /*
566                  * Transfer each lock. Note that ultimately, since mastership
567                  * is changing, and replicated commits will only be executed
568                  * when a txn has originated on that node, the target ReplayTxn
569                  * can never be committed, and will only be aborted.
570                  */
571                 for (Long lsn: sortedLsns) {
572 
573                     LoggerUtils.info(logger, envImpl,
574                                      "Txn " + getId() +
575                                      " is transferring lock " + lsn);
576 
577                     /*
578                      * Use a special method to steal the lock. Another approach
579                      * might have been to have the replayTxn merely attempt a
580                      * lock(); as an importunate txn, the replayTxn would
581                      * preempt the MasterTxn's lock. However, that path doesn't
582                      * work because lock() requires having a databaseImpl in
583                      * hand, and that's not available here.
584                      */
585                     replayTxn.stealLockFromMasterTxn(lsn);
586 
587                     /*
588                      * Copy all the lock's info into the Replay and remove it
589                      * from the master. Normally, undo clears write locks, but
590                      * this MasterTxn will not be executing undo.
591                      */
592                     WriteLockInfo replayWLI = replayTxn.getWriteLockInfo(lsn);
593                     WriteLockInfo masterWLI = getWriteLockInfo(lsn);
594                     replayWLI.copyAllInfo(masterWLI);
595                     removeLock(lsn);
596                 }
597 
598                 /*
599                  * Txns have collections of undoDatabases and deletedDatabases.
600                  * Undo databases are normally incrementally added to the txn
601                  * as locks are obtained Unlike normal locking or recovery
602                  * locking, in this case we don't have a reference to the
603                  * databaseImpl that goes with this lock, so we copy the undo
604                  * collection in one fell swoop.
605                  */
606                 replayTxn.copyDatabasesForConversion(this);
607 
608                 /*
609                  * This txn is no longer responsible for databaseImpl
610                  * cleanup, as that issue now lies with the ReplayTxn, so
611                  * remove the collection.
612                  */
613                 deletedDatabases = null;
614 
615                 /*
616                  * All locks have been removed from this transaction. Clear
617                  * the firstLoggedLsn and lastLoggedLsn so there's no danger
618                  * of attempting to undo anything; this txn is no longer
619                  * responsible for any records.
620                  */
621                 lastLoggedLsn = DbLsn.NULL_LSN;
622                 firstLoggedLsn = DbLsn.NULL_LSN;
623 
624                 /* If this txn also had read locks, clear them */
625                 clearReadLocks();
626             }
627         } finally {
628 
629             assert TestHookExecute.doHookIfSet(convertHook, hookCount++);
630 
631             unfreeze();
632 
633             assert TestHookExecute.doHookIfSet(convertHook, hookCount++);
634 
635             /*
636              * We need to abort the txn, but we can't call abort() because that
637              * method checks whether we are the master! Instead, call the
638              * internal method, close(), in order to end this transaction and
639              * unregister it from the transactionManager. Must be called
640              * outside the synchronization block.
641              */
642             if (needToClose) {
643                 LoggerUtils.info(logger, envImpl, "About to close txn " +
644                                  getId() + " state=" + getState());
645                 close(false /*isCommit */);
646                 LoggerUtils.info(logger, envImpl, "Closed txn " +  getId() +
647                                  " state=" + getState());

663     /**
664      * Used to hold the transaction stable while it is being cloned as a
665      * ReplayTxn, during master->replica transitions. Essentially, there
666      * are two parties that now have a reference to this transaction -- the
667      * originating application thread, and the RepNode thread that is
668      * trying to set up internal state so it can begin to act as a replica.
669      *
670      * The transaction will throw UnknownMasterException or
671      * ReplicaWriteException if the transaction is frozen, so that the
672      * application knows that the transaction is no longer viable, but it
673      * doesn't attempt to do most of the follow-on cleanup and release of locks
674      * that failed aborts and commits normally attempt. One aspect of
675      * transaction cleanup can't be skipped though. It is necessary to do the
676      * post log hooks to free up the block txn latch lock so that the
677      * transaction can be closed by the RepNode thread. For example:
678      * - application thread starts transaction
679      * - application takes the block txn latch lock and attempts commit or
680      * abort, but is stopped because the txn is frozen by master transfer.
681      * - the application must release the block txn latch lock.
682      * @see Replica#replicaTransitionCleanup
683      */
684     @Override
685     protected void checkIfFrozen(boolean isCommit) {
686         if (freeze) {
687             try {
688                 ((RepImpl) envImpl).checkIfMaster(this);
689             } catch (DatabaseException e) {
690                 if (isCommit) {
691                     postLogCommitAbortHook();
692                 } else {
693                     postLogAbortHook();
694                 }
695                 throw e;
696             }
697         }
698     }
