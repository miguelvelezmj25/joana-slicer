21 /**
22  * Create a thread factory that returns threads that are legitimate
23  * StoppableThreads. Like StoppableThreads, if an environment is provided, the
24  * threads will invalidate if an exception is not handled, and are registered
25  * with the exception listener.If a logger is provided, StoppableThreads log
26  * exception information.
27  *
28  * This factory is used in conjunction with the ExecutorService and
29  * ThreadExecutorPool models.
30  */
31 public class StoppableThreadFactory implements ThreadFactory {
32 
33     private final String threadName;
34     private final Logger logger;
35     private final EnvironmentImpl envImpl;
36 
37     /**
38      * This kind of StoppableThreadFactory will cause invalidation if an
39      * unhandled exception occurs.
40      */
41     public StoppableThreadFactory(EnvironmentImpl envImpl,
42                                   String threadName,
43                                   Logger logger) {
44         this.threadName = threadName;
45         this.logger = logger;
46         this.envImpl = envImpl;
47     }
48 
49     /**
50      * This kind of StoppableThreadFactory will NOT cause invalidation if an
51      * unhandled exception occurs, because there is no environment provided.
52      */
53     public StoppableThreadFactory(String threadName, Logger logger) {
54         this(null, threadName, logger);
55     }
56 
57     public Thread newThread(Runnable runnable) {
58         return new StoppablePoolThread(envImpl, runnable, threadName, logger);
59     }
60 
61     /*
62      * A fairly plain implementation of the abstract StoppableThread class,
63      * for use by the factory.
64      */
65     private static class StoppablePoolThread extends StoppableThread {

72             super(envImpl, null, runnable, threadName);
73             this.logger = logger;

75
