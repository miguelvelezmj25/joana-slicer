27  */

39     }

41     /**
42      * Implements the Listener protocol. The method should not have any
43      * operations that might wait, since notifications are single threaded.
44      */
45     @Override
46     public void notify(Proposal proposal, Value value) {
47 
48         try {
49             repNode.getVLSNFreezeLatch().vlsnEvent(proposal);
50             /* We have a new proposal, is it truly different? */
51             if (value.equals(currentValue)) {
52                 LoggerUtils.fine(logger, repNode.getRepImpl(),
53                                  "Master change listener -- no value change." +
54                                  "Proposal: " + proposal + " Value: " + value);
55                 return;
56             }
57 
58             MasterValue masterValue = ((MasterValue) value);
59 
60             LoggerUtils.fine(logger, repNode.getRepImpl(),
61                     "Master change listener notified. Proposal:" +
62                     proposal + " Value: " + value);
63             LoggerUtils.info(logger, repNode.getRepImpl(),
64                     "Master changed to " +
65                      masterValue.getNameId().getName());
66 
67             repNode.getMasterStatus().setGroupMaster
68                 (masterValue.getHostName(),
69                  masterValue.getPort(),
70                  masterValue.getNameId());
71 
72             /* Propagate the information to any monitors. */
73             repNode.getElections().asyncInformMonitors(proposal, value);
74         } finally {
75             currentValue = value;
76         }
77     }
