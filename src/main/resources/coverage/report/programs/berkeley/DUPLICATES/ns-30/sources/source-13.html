55         if (disableEvents) {
56             return true;
57         }
58         int nextEventIdx = currentEvent.getAndIncrement() % MAX_EVENTS;
59         events[nextEventIdx] = event;
60         threadIdHashes[nextEventIdx] =
61             System.identityHashCode(Thread.currentThread());
62         return true;
63     }

80     public static void dumpEvents(PrintStream out) {
81 
82         if (disableEvents) {
83             return;
84         }
85         out.println("----- Event Dump -----");
86         EventTrace[] oldEvents = events;
87         int[] oldThreadIdHashes = threadIdHashes;
88         disableEvents = true;
89 
90         int j = 0;
91         for (int i = currentEvent.get(); j < MAX_EVENTS; i++) {
92             EventTrace ev = oldEvents[i % MAX_EVENTS];
93             if (ev != null) {
94                 int thisEventIdx = i % MAX_EVENTS;
95                 out.print(oldThreadIdHashes[thisEventIdx] + " ");
96                 out.println(j + "(" + thisEventIdx + "): " + ev);
97             }
98             j++;
99         }
