23 /**
24  * JE instances of java.util.logging.Logger are configured to use this
25  * implementation of java.util.logging.ConsoleHandler. By default, the
26  * handler's level is {@link Level#OFF}. To enable the console output, use the
27  * standard java.util.logging.LogManager configuration to set the desired
28  * level:
29  * <pre>
30  * com.sleepycat.je.util.ConsoleHandler.level=ALL
31  * </pre>
32  * JE augments the java.util.logging API with a JE environment parameter for
33  * setting handler levels. This is described in greater detail in 
34  * {@link <a href="{@docRoot}/../GettingStartedGuide/managelogging.html">
35  * Chapter 12.Administering Berkeley DB Java Edition Applications</a>}
36  *
37  * @see <a href="{@docRoot}/../GettingStartedGuide/managelogging.html">
38  *     Chapter 12. Logging</a> 
39  * @see <a href="{@docRoot}/../traceLogging.html">Using JE Trace Logging</a>
40  */
41 public class ConsoleHandler extends java.util.logging.ConsoleHandler {
42     
43     /* 
44      * Using a JE specific handler lets us enable and disable output for the
45      * entire library, and specify an environment specific format.
46      */
47     public ConsoleHandler(Formatter formatter, EnvironmentImpl envImpl) {
48         super();
49 
50         /* Messages may be formatted with an environment specific tag. */
51         setFormatter(formatter);
52 
53         Level level = null;
54         String propertyName = getClass().getName() + ".level";
55 
56         if (envImpl != null) {
57             level = 
58                 LoggerUtils.getHandlerLevel(envImpl.getConfigManager(), 
59                                             EnvironmentParams.JE_CONSOLE_LEVEL,
60                                             propertyName);
61         } else {
62             /* If envImpl instance is null, level is decided by properties. */
63             String levelProperty = LoggerUtils.getLoggerProperty(propertyName);
64             if (levelProperty == null) {
65                 level = Level.OFF;
66             } else {
67                 level = Level.parse(levelProperty);
68             }
69         }
70 
71         setLevel(level);
72     }
73 }
