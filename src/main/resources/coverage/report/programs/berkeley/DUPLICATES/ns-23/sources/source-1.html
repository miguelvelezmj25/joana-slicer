60      * Client side authentication

62     static class ClientTokenOp extends ServiceHandshake.ClientInitOp {
63 
64         private final SubscriptionAuthHandler auth;
65         ClientTokenOp(ServiceHandshake.ClientHandshake initState,
66                       SubscriptionAuthHandler auth) {
67             super(initState);
68             this.auth = auth;
69         }
70 
71         @Override
72         public ServiceHandshake.InitResult processOp(
73             ServiceHandshake.IOAdapter ioAdapter) throws IOException {
74 
75             final byte[] token = auth.getToken();
76             if (token == null || token.length == 0) {
77                 throw new IOException("Token cannot be null or empty");
78             }
79 
80             /* write size of token */
81             final ByteBuffer szBuf = ByteBuffer.allocate(4);
82             LogUtils.writeInt(szBuf, token.length);
83             ioAdapter.write(szBuf.array());
84             /* write token */
85             final ByteBuffer tokenBuf = ByteBuffer.allocate(token.length);
86             LogUtils.writeBytesNoLength(tokenBuf, token);
87             ioAdapter.write(tokenBuf.array());
88 
89             final byte[] responseByte = new byte[1];
90             final int result = ioAdapter.read(responseByte);
91             if (result < 0) {
92                 throw new IOException(
93                     "No service authenticate response byte: " + result);
94             }
95             final ServiceDispatcher.Response
96                 response = ServiceDispatcher.Response.get(responseByte[0]);
97             if (response == null) {
98                 throw new IOException("Unexpected read response byte: " +
99                                       responseByte[0]);
100             }
101             setResponse(response);
102             return ServiceHandshake.InitResult.DONE;
103         }
104     }
