50 /**
51  * TextProtocol provides the support for implementing simple low performance
52  * protocols involving replication nodes. The protocol is primarily text based,
53  * and checks group membership and version matches with every message favoring
54  * flexibility over performance.
55  *
56  * The base class is primarily responsible for the message formatting and
57  * message envelope validation. The subclasses define the specific messages
58  * that constitute the protocol and the request/response semantics.
59  *
60  * Every message has the format:
61  * <version>|<name>|<id>|<op>|<op-specific payload>
62  *
63  * <version> is the version of the protocol in use.
64  * <name>    identifies a group participating in an election. It avoids
65  *           accidental cross-talk across groups holding concurrent elections.
66  * <id>      identifies the originator of the message within the group.
67  * <op>      the operation identified by the specific message.
68  * <op-specific payload> the payload associated with the particular operation.
69  */

326      * Parses a line into a Request/Response message.
327      *
328      * @param line containing the message
329      * @return a message instance
330      * @throws InvalidMessageException
331      */
332     public Message parse(String line)
333         throws InvalidMessageException {
334 
335         String[] tokens = line.split(SEPARATOR_REGEXP);
336 
337         final int index = TOKENS.OP_TOKEN.ordinal();
338         if (index >= tokens.length) {
339             throw new InvalidMessageException(
340                 MessageError.BAD_FORMAT,
341                 "Missing message op in message: " + line);
342         }
343         final MessageOp op = ops.get(tokens[index]);
344         if (op == null) {
345             throw new InvalidMessageException(MessageError.BAD_FORMAT,
346                                               "Text Protocol" +
347                                               " unknown op:" + tokens[index] +
348                                               " in message: " + line);
349         }
350 
351         try {
352             Class<? extends Message> c = op.getMessageClass();
353             Constructor<? extends Message> cons =
354                 c.getConstructor(c.getEnclosingClass(),
355                                  line.getClass(),
356                                  tokens.getClass());
357             Message message = cons.newInstance(this, line, tokens);
358             return message;
359         } catch (InstantiationException e) {
360             throw EnvironmentFailureException.unexpectedException(e);
361         } catch (IllegalAccessException e) {
362             throw EnvironmentFailureException.unexpectedException(e);
363         } catch (SecurityException e) {
364             throw EnvironmentFailureException.unexpectedException(e);
365         } catch (NoSuchMethodException e) {
366             throw EnvironmentFailureException.unexpectedException(e);
367         } catch (InvocationTargetException e) {
368             /* Unwrap the exception. */
369             Throwable target = e.getTargetException();
370             if (target instanceof RuntimeException) {
371                 final String message = "message: " + line +
372                   " exception:" + target.getClass().getName() +
373                   " exception message:" + target.getMessage();
374                 throw new InvalidMessageException(MessageError.BAD_FORMAT,
375                                                   message);
376 
377             } else if (target instanceof InvalidMessageException) {
378                 throw (InvalidMessageException) target;
379             }
380             throw EnvironmentFailureException.unexpectedException(e);
381         }
382     }

387     public abstract class Message implements WireFormatable {
388         /* The sender of the message. */
389         private int senderId = 0;
390 
391         /*
392          * The version of this message, as it's deserialized and sent across
393          * the network. The default is that messages are sent in the VERSION of
394          * the current protocol, but in cases of mixed-version upgrades, the
395          * message may be sent in an earlier version format.
396          *
397          * When this message is a RequestMessage, the sender will always
398          * initially send it out its own native version, but may resend it in
399          * an earlier version, if the recipient can't understand the native
400          * version. When the message is a ResponseMessage, the sender can reply
401          * either in its native version, or in an earlier version if the
402          * requester is an older version of JE.
403          */
404         protected String sendVersion;
405 
406         /* The line representing the message. */
407         private final String line;
408 
409         /* The tokenized form of the above line. */
410         private final String[] tokens;
411 
412         /* The current variable arg token */
413         private int currToken = TOKENS.FIRST_PAYLOAD_TOKEN.ordinal();
414 
415         protected String messagePrefixNocheck;
416

419          * message, which does not use the line or tokens.
420          */
421         protected Message() {
422             line = null;
423             tokens = null;
424             setSendVersion(VERSION);
425         }
426 
427         /**
428          * Every message must define a constructor of this form so that it can
429          * be de-serialized. The constructor is invoked using reflection by the
430          * parse() method.
431          *
432          * @param line the line constituting the message
433          * @param tokens the line in token form
434          * @throws InvalidMessageException
435          * @throws EnvironmentFailureException on format errors
436          */
437         protected Message(String line, String[] tokens)
438             throws InvalidMessageException {
439 
440             this.line = line;
441             this.tokens = tokens;
442 
443             /* Validate the leading fixed fields. */
444             final String version = getTokenString(TOKENS.VERSION_TOKEN);
445             if (new Double(VERSION) < new Double(version)) {
446                 throw new InvalidMessageException
447                     (MessageError.VERSION_MISMATCH,
448                      "Version argument mismatch." +
449                      " Expected: " + VERSION + ", found: " + version +
450                      ", in message: " + line);
451             }
452 
453             /*
454              * Set the sender version of a request message. This version
455              * information will be used by the receiver to determine what
456              * version should be used for the response message.
457              */
458             setSendVersion(version);
459 
460             final String messageGroupName = getTokenString(TOKENS.NAME_TOKEN);
461             if (!groupName.equals(messageGroupName)) {
462                 throw new InvalidMessageException
463                 (MessageError.GROUP_MISMATCH,
464                  "Group name mismatch; this group name: " + groupName +
465                  ", message group name: " + messageGroupName +
466                  ", in message: " + line);
467             }
468 
469             senderId =
470                 new Integer(getTokenString(TOKENS.ID_TOKEN)).intValue();
471             if ((memberIds != null) &&
472                 (memberIds.size() > 0) &&
473                 (nameIdPair.getId() != NameIdPair.NOCHECK_NODE_ID) &&
474                 (senderId != NameIdPair.NOCHECK_NODE_ID) &&
475                 (senderId != nameIdPair.getId()) &&
476                 !memberIds.contains(senderId)) {
477                 throw new InvalidMessageException
478                    (MessageError.NOT_A_MEMBER,
479                     "Sender's member id: " + senderId +
480                     ", message op: " + getTokenString(TOKENS.OP_TOKEN) +
481                     ", was not a member of the group: " + memberIds +
482                     ", in message: " + line);
483             }
484         }
