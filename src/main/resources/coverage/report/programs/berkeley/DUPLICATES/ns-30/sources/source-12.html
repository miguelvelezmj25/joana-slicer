23  * a specified time period of values supplied with associated times, to support

98      * @param time the current time in milliseconds
99      */
100     public synchronized void add(double value, long time) {
101         assert time > 0;
102         if (time <= prevTime) {
103             return;
104         }
105         if (prevTime == 0) {
106             avg = value;
107         } else {
108 
109             /*
110              * Compute the exponential moving average, as described in:
111              * http://en.wikipedia.org/wiki/
112              *   Moving_average#Application_to_measuring_computer_performance
113              */
114             double m = Math.exp(-((time - prevTime)/((double) periodMillis)));
115             avg = ((1-m) * value) + (m * avg);
116         }
117         prevTime = time;
118     }

136     }
