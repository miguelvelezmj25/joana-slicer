38     /**
39      * Each file for which cleaning is in progress has one of the following
40      * status values.  Files numbers migrate from one status to another, in
41      * the order declared below.
42      */
43     enum FileStatus {
44 
45         /**
46          * A file's status is initially TO_BE_CLEANED when it is selected as
47          * part of a batch of files that, when deleted, will bring total
48          * utilization down to the minimum configured value.  All files with
49          * this status will be cleaned in lowest-cost-to-clean order.  For two
50          * files of equal cost to clean, the lower numbered (oldest) files is
51          * selected; this is why the fileInfoMap is sorted by key (file
52          * number).
53          */
54         TO_BE_CLEANED,
55 
56         /**
57          * When a TO_BE_CLEANED file is selected for processing by
58          * FileProcessor, it is moved to the BEING_CLEANED status.  This
59          * distinction is used to prevent a file from being processed by more
60          * than one thread.
61          */
62         BEING_CLEANED,
63 
64         /**
65          * A file is moved to the CLEANED status when all its log entries have
66          * been read and processed.  However, entries needing migration will be
67          * marked with the BIN entry MIGRATE flag, entries that could not be
68          * locked will be in the pending LN set, and the DBs that were pending
69          * deletion will be in the pending DB set.
70          */
71         CLEANED,
72 
73         /**
74          * A file is moved to the CHECKPOINTED status at the end of a
75          * checkpoint if it was CLEANED at the beginning of the checkpoint.
76          * Because all dirty BINs are flushed during the checkpoints, no files
77          * in this set will have entries with the MIGRATE flag set.  However,
78          * some entries may be in the pending LN set and some DBs may be in the
79          * pending DB set.
80          */
81         CHECKPOINTED,
82 
83         /**
84          * A file is moved from the CHECKPOINTED status to the FULLY_PROCESSED
85          * status when the pending LN/DB sets become empty.  Since a pending LN
86          * was not locked successfully, we don't know its original file.  But
87          * we do know that when no pending LNs are present for any file, all
88          * log entries in CHECKPOINTED files are either obsolete or have been
89          * migrated.  Note, however, that the parent BINs of the migrated
90          * entries may not have been logged yet.
91          *
92          * No special handling is required to coordinate syncing of deferred
93          * write databases for pending, deferred write LNs, because
94          * non-temporary deferred write DBs are always synced during
95          * checkpoints, and temporary deferred write DBs are not recovered.
96          * Note that although DW databases are non-txnal, their LNs may be
97          * pended because of lock collisions.
98          */
99         FULLY_PROCESSED,
100     }

161      * Returns the best file that qualifies for cleaning, or null if no file
162      * qualifies.
163      *
164      * @param forceCleaning is true to always select a file, even if its
165      * utilization is above the minimum utilization threshold.
166      *
167      * @return {file number, required utilization for 2-pass cleaning},
168      * or null if no file qualifies for cleaning.
169      */
170     synchronized Pair<Long, Integer> selectFileForCleaning(
171         UtilizationCalculator calculator,
172         SortedMap<Long, FileSummary> fileSummaryMap,
173         boolean forceCleaning) {
174 
175         final Set<Long> toBeCleaned = getToBeCleanedFiles();
176 
177         if (!toBeCleaned.isEmpty()) {
178             final Long fileNum = toBeCleaned.iterator().next();
179             final FileInfo info = setStatus(fileNum, FileStatus.BEING_CLEANED);
180             return new Pair<>(fileNum, info.requiredUtil);
181         }
182 
183         final Pair<Long, Integer> result = calculator.getBestFile(
184             fileSummaryMap, forceCleaning);
185 
186         if (result == null) {
187             return null;
188         }
189 
190         final Long fileNum = result.first();
191         final int requiredUtil = result.second();
192 
193         assert !fileInfoMap.containsKey(fileNum);
194 
195         final FileInfo info = setStatus(fileNum, FileStatus.BEING_CLEANED);
196         info.requiredUtil = requiredUtil;
197 
198         return result;
199     }

389     /**
390      * When a checkpoint is complete, move the previously CLEANED and
391      * FULLY_PROCESSED files to the CHECKPOINTED and reserved status.
392      * Reserved files are removed from the FileSelector and their reserved
393      * status is maintained in FileProtector.
394      *
395      * @return map of {fileNum, FileInfo} for the files whose status was
396      * changed to reserved.
397      */
398     synchronized Map<Long, FileInfo> updateFilesAtCheckpointEnd(
399         final EnvironmentImpl env,
400         final CheckpointStartCleanerState info) {
401 
402         if (info.isEmpty()) {
403             return Collections.emptyMap();
404         }
405 
406         final Map<Long, FileInfo> reservedFiles = new HashMap<>();
407         final Set<Long> previouslyCleanedFiles = info.getCleanedFiles();
408         final Set<Long> previouslyProcessedFiles =
409             info.getFullyProcessedFiles();
410 
411         if (previouslyCleanedFiles != null) {
412 
413             assert checkStatus(previouslyCleanedFiles, FileStatus.CLEANED);
414 
415             if (anyPendingDuringCheckpoint) {
416                 setStatus(previouslyCleanedFiles, FileStatus.CHECKPOINTED);
417             } else {
418                 makeReservedFiles(env, previouslyCleanedFiles, reservedFiles);
419             }
420         }
421 
422         if (previouslyProcessedFiles != null) {
423 
424             assert checkStatus(previouslyProcessedFiles,
425                                FileStatus.FULLY_PROCESSED);
426 
427             makeReservedFiles(env, previouslyProcessedFiles, reservedFiles);
428         }
429 
430         updateProcessedFiles();
431 
432         return reservedFiles;
433     }

435     private void makeReservedFiles(
436         final EnvironmentImpl env,
437         final Set<Long> safeToDeleteFiles,
438         final Map<Long, FileInfo> reservedFiles) {
439 
440         final FileProtector fileProtector = env.getFileProtector();
441         final MemoryBudget memoryBudget = env.getMemoryBudget();
442 
443         for (Long file : safeToDeleteFiles) {
444             final FileInfo info = removeFile(file, memoryBudget);
445             fileProtector.reserveFile(file, info.lastVlsn);
446             reservedFiles.put(file, info);
447         }
448 
449         env.getUtilizationProfile().removeFileSummaries(safeToDeleteFiles);
450     }
