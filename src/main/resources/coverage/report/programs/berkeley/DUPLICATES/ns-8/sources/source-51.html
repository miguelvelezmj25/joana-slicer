24 /**
25  * The Transaction object is the handle for a transaction.  Methods off the
26  * transaction handle are used to configure, abort and commit the transaction.
27  * Transaction handles are provided to other Berkeley DB methods in order to
28  * transactionally protect those operations.
29  *
30  * <p>A single Transaction may be used to protect operations for any number of
31  * Databases in a given environment.  However, a single Transaction may not be
32  * used for operations in more than one distinct environment.</p>
33  *
34  * <p>Transaction handles are free-threaded; transactions handles may be used
35  * concurrently by multiple threads. Once the {@link Transaction#abort
36  * Transaction.abort} or {@link Transaction#commit Transaction.commit} method
37  * is called, the handle may not be accessed again, regardless of the success
38  * or failure of the method, with one exception:  the {@code abort} method may
39  * be called any number of times to simplify error handling.</p>
40  *
41  * <p>To obtain a transaction with default attributes:</p>
42  *
43  * <blockquote><pre>
44  *     Transaction txn = myEnvironment.beginTransaction(null, null);
45  * </pre></blockquote>
46  *
47  * <p>To customize the attributes of a transaction:</p>
48  *
49  * <blockquote><pre>
50  *     TransactionConfig config = new TransactionConfig();
51  *     config.setReadUncommitted(true);
52  *     Transaction txn = myEnvironment.beginTransaction(null, config);
53  * </pre></blockquote>
54  */
55 public class Transaction {

201

601     private void doCommit(Durability durability, boolean explicitSync) {
602         try {
603             checkEnv();
604             checkOpen();
605             env.removeReferringHandle(this);
606             if (explicitSync) {
607                 /* A sync-specific api was invoked. */
608                 if (txn.getExplicitDurabilityConfigured()) {
609                     throw new IllegalArgumentException
610                         ("Mixed use of deprecated durability API for the " +
611                          "transaction commit with the new durability API for" +
