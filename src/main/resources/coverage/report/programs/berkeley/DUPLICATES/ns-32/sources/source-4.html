20 /**
21  * A subclass of Converter that allows specifying keys to be deleted.
22  *
23  * <p>When a Converter is used with an entity class, secondary keys cannot be
24  * automatically deleted based on field deletion, because field Deleter objects
25  * are not used in conjunction with a Converter mutation.  The EntityConverter
26  * can be used instead of a plain Converter to specify the key names to be
27  * deleted.</p>
28  *
29  * <p>It is not currently possible to rename or insert secondary keys when
30  * using a Converter mutation with an entity class.</p>
31  *
32  * @see Converter
33  * @see com.sleepycat.persist.evolve Class Evolution
34  * @author Mark Hayes
35  */
36 public class EntityConverter extends Converter {
37 
38     private static final long serialVersionUID = -988428985370593743L;
39 
40     private Set<String> deletedKeys;
41 
42     /**
43      * Creates a mutation for converting all instances of the given entity
44      * class version to the current version of the class.
45      *
46      * @param entityClassName the entity class to which this mutation applies.
47      * @param classVersion the class version to which this mutation applies.
48      * @param conversion converter instance.
49      * @param deletedKeys the set of key names that are to be deleted.
50      */
51     public EntityConverter(String entityClassName,
52                            int classVersion,
53                            Conversion conversion,
54                            Set<String> deletedKeys) {
55         super(entityClassName, classVersion, null, conversion);
56 
57         /* Eclipse objects to assigning with a ternary operator. */
58         if (deletedKeys != null) {
59             this.deletedKeys = new HashSet(deletedKeys);
60         } else {
61             this.deletedKeys = Collections.emptySet();
62         }
63     }
64

67      *
68      * @return the set of key names that are to be deleted.
69      */
70     public Set<String> getDeletedKeys() {
71         return Collections.unmodifiableSet(deletedKeys);
72     }
