103     private CurrentTransaction(Environment env) {
104         envRef = new WeakReference<Environment>(env);
105         try {
106             EnvironmentConfig config = env.getConfig();
107             txnMode = config.getTransactional();
108             lockingMode = DbCompat.getInitializeLocking(config);
109             if (txnMode || lockingMode) {
110                 writeLockMode = LockMode.RMW;
111             } else {
112                 writeLockMode = LockMode.DEFAULT;
113             }
114             cdbMode = DbCompat.getInitializeCDB(config);
115             if (cdbMode) {
116                 localCdbCursors = new ThreadLocal();
117             }
118         } catch (DatabaseException e) {
119             throw RuntimeExceptionWrapper.wrapIfNeeded(e);
120         }

222     public final Transaction beginTransaction(TransactionConfig config)
223         throws DatabaseException {
224 
225         Environment env = getEnvironment();
226         Trans trans = (Trans) localTrans.get();
227         if (trans != null) {
228             if (trans.txn != null) {
229                 if (!DbCompat.NESTED_TRANSACTIONS) {
230                     throw new IllegalStateException
231                         ("Nested transactions are not supported");
232                 }
233                 Transaction parentTxn = trans.txn;
234                 trans = new Trans(trans, config);
235                 trans.txn = env.beginTransaction(parentTxn, config);
236                 localTrans.set(trans);
237             } else {
238                 trans.txn = env.beginTransaction(null, config);
239                 trans.config = config;
240             }
241         } else {
242             trans = new Trans(null, config);
243             trans.txn = env.beginTransaction(null, config);
244             localTrans.set(trans);
245         }
246         return trans.txn;
247     }

364      * Opens a cursor for a given database, dup'ing an existing CDB cursor if
365      * one is open for the current thread.
366      */
367     Cursor openCursor(Database db,
368                       CursorConfig cursorConfig,
369                       boolean writeCursor,
370                       Transaction txn)
371         throws DatabaseException {
372 
373         if (cdbMode) {
374             CdbCursors cdbCursors = null;
375             WeakHashMap cdbCursorsMap = (WeakHashMap) localCdbCursors.get();
376             if (cdbCursorsMap == null) {
377                 cdbCursorsMap = new WeakHashMap();
378                 localCdbCursors.set(cdbCursorsMap);
379             } else {
380                 cdbCursors = (CdbCursors) cdbCursorsMap.get(db);
381             }
382             if (cdbCursors == null) {
383                 cdbCursors = new CdbCursors();
384                 cdbCursorsMap.put(db, cdbCursors);
385             }
386 
387             /*
388              * In CDB mode the cursorConfig specified by the user is ignored
389              * and only the writeCursor parameter is honored.  This is the only
390              * meaningful cursor attribute for CDB, and here we count on
391              * writeCursor flag being set correctly by the caller.
392              */
393             List cursors;
394             CursorConfig cdbConfig;
395             if (writeCursor) {
396                 if (cdbCursors.readCursors.size() > 0) {
397 
398                     /*
399                      * Although CDB allows opening a write cursor when a read
400                      * cursor is open, a self-deadlock will occur if a write is
401                      * attempted for a record that is read-locked; we should
402                      * avoid self-deadlocks at all costs
403                      */
404                     throw new IllegalStateException
405                         ("Cannot open CDB write cursor when read cursor " +
406                          "is open");
407                 }
408                 cursors = cdbCursors.writeCursors;
409                 cdbConfig = new CursorConfig();
410                 DbCompat.setWriteCursor(cdbConfig, true);
411             } else {
412                 cursors = cdbCursors.readCursors;
413                 cdbConfig = null;
414             }
415             Cursor cursor;
416             if (cursors.size() > 0) {
417                 Cursor other = ((Cursor) cursors.get(0));
418                 cursor = other.dup(false);
419             } else {
420                 cursor = db.openCursor(null, cdbConfig);
421             }
422             cursors.add(cursor);
423             return cursor;
424         } else {
425             return db.openCursor(txn, cursorConfig);
426         }
427     }

430      * Duplicates a cursor for a given database.
431      *
432      * @param writeCursor true to open a write cursor in a CDB environment, and
433      * ignored for other environments.
434      *
435      * @param samePosition is passed through to Cursor.dup().
436      *
437      * @return the open cursor.
438      *
439      * @throws DatabaseException if a database problem occurs.
440      */
441     Cursor dupCursor(Cursor cursor, boolean writeCursor, boolean samePosition)
442         throws DatabaseException {
443 
444         if (cdbMode) {
445             WeakHashMap cdbCursorsMap = (WeakHashMap) localCdbCursors.get();
446             if (cdbCursorsMap != null) {
447                 Database db = cursor.getDatabase();
448                 CdbCursors cdbCursors = (CdbCursors) cdbCursorsMap.get(db);
449                 if (cdbCursors != null) {
450                     List cursors = writeCursor ? cdbCursors.writeCursors
451                                                : cdbCursors.readCursors;
452                     if (cursors.contains(cursor)) {
453                         Cursor newCursor = cursor.dup(samePosition);
454                         cursors.add(newCursor);
455                         return newCursor;
456                     }
457                 }
458             }
459             throw new IllegalStateException("Cursor to dup not tracked");
460         } else {
461             return cursor.dup(samePosition);
462         }
463     }

466      * Closes a cursor.
467      *
468      * @param cursor the cursor to close.
469      *
470      * @throws DatabaseException if a database problem occurs.
471      */
472     void closeCursor(Cursor cursor)
473         throws DatabaseException {
474 
475         if (cursor == null) {
476             return;
477         }
478         if (cdbMode) {
479             WeakHashMap cdbCursorsMap = (WeakHashMap) localCdbCursors.get();
480             if (cdbCursorsMap != null) {
481                 Database db = cursor.getDatabase();
482                 CdbCursors cdbCursors = (CdbCursors) cdbCursorsMap.get(db);
483                 if (cdbCursors != null) {
484                     if (cdbCursors.readCursors.remove(cursor) ||
485                         cdbCursors.writeCursors.remove(cursor)) {
486                         cursor.close();
487                         return;
488                     }
489                 }
490             }
491             throw new IllegalStateException
492                 ("Closing CDB cursor that was not known to be open");
493         } else {
494             cursor.close();
495         }
496     }
