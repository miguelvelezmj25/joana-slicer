72     /**
73      * Creates a catalog based on a given database. To save resources, only a
74      * single catalog object should be used for each unique catalog database.
75      *
76      * @param database an open database to use as the class catalog.  It must
77      * be a BTREE database and must not allow duplicates.
78      *
79      * @throws DatabaseException if an error occurs accessing the database.
80      *
81      * @throws IllegalArgumentException if the database is not a BTREE database
82      * or if it configured to allow duplicates.
83      */
84     public StoredClassCatalog(Database database)
85         throws DatabaseException, IllegalArgumentException {
86 
87         db = database;
88         DatabaseConfig dbConfig = db.getConfig();
89         EnvironmentConfig envConfig = db.getEnvironment().getConfig();
90 
91         writeLockMode = (DbCompat.getInitializeLocking(envConfig) ||
92                          envConfig.getTransactional()) ? LockMode.RMW
93                                                        : LockMode.DEFAULT;
94         cdbMode = DbCompat.getInitializeCDB(envConfig);
95         txnMode = dbConfig.getTransactional();
96 
97         if (!DbCompat.isTypeBtree(dbConfig)) {
98             throw new IllegalArgumentException
99                 ("The class catalog must be a BTREE database.");
100         }
101         if (DbCompat.getSortedDuplicates(dbConfig) ||
102             DbCompat.getUnsortedDuplicates(dbConfig)) {
103             throw new IllegalArgumentException
104                 ("The class catalog database must not allow duplicates.");
105         }
106 
107         /*
108          * Create the class format and class info maps. Note that these are not
109          * synchronized, and therefore the methods that use them are
110          * synchronized.
111          */
112         classMap = new HashMap<String, ClassInfo>();
113         formatMap = new HashMap<BigInteger, ObjectStreamClass>();
114 
115         DatabaseEntry key = new DatabaseEntry(LAST_CLASS_ID_KEY);
116         DatabaseEntry data = new DatabaseEntry();
117         if (dbConfig.getReadOnly()) {
118             /* Check that the class ID record exists. */
119             OperationStatus status = db.get(null, key, data, null);
120             if (status != OperationStatus.SUCCESS) {
121                 throw DbCompat.unexpectedState
122                     ("A read-only catalog database may not be empty");
123             }
124         } else {
125             /* Add the initial class ID record if it doesn't exist.  */
126             data.setData(new byte[1]); // zero ID
127 
128             /*
129              * Query the record before writing it to the database, to avoid
130              * ReplicaWriteException while opening a StoredClassCatalog on the
131              * replicas.
132              */
133             OperationStatus status = db.get(null, key, data, null);
134             if (status == OperationStatus.NOTFOUND) {
135                 db.putNoOverwrite(null, key, data);
136             }
137         }
138     }

167     /**
168      * Internal function for getting the class format.  Allows passing the
169      * DatabaseEntry object for the data, so the bytes of the class format can
170      * be examined afterwards.
171      */
172     private ObjectStreamClass getClassFormat(byte[] classID,
173                                              DatabaseEntry data)
174         throws DatabaseException, ClassNotFoundException {
175 
176         /* First check the map and, if found, add class info to the map. */
177 
178         BigInteger classIDObj = new BigInteger(classID);
179         ObjectStreamClass classFormat = formatMap.get(classIDObj);
180         if (classFormat == null) {
181 
182             /* Make the class format key. */
183 
184             byte[] keyBytes = new byte[classID.length + 1];
185             keyBytes[0] = REC_CLASS_FORMAT;
186             System.arraycopy(classID, 0, keyBytes, 1, classID.length);
187             DatabaseEntry key = new DatabaseEntry(keyBytes);
188 
189             /* Read the class format. */
190 
191             OperationStatus status = db.get(null, key, data, LockMode.DEFAULT);
192             if (status != OperationStatus.SUCCESS) {
193                 throw new ClassNotFoundException("Catalog class ID not found");
194             }
195             try {
196                 ObjectInputStream ois =
197                     new ObjectInputStream(
198                         new ByteArrayInputStream(data.getData(),
199                                                  data.getOffset(),
200                                                  data.getSize()));
201                 classFormat = (ObjectStreamClass) ois.readObject();
202             } catch (IOException e) {
203                 throw RuntimeExceptionWrapper.wrapIfNeeded(e);
204             }
205 
206             /* Update the class format map. */
207 
208             formatMap.put(classIDObj, classFormat);
209         }
210         return classFormat;
211     }

213     /**
214      * Get the ClassInfo for a given class name, adding it and its
215      * ObjectStreamClass to the database if they are not already present, and
216      * caching both of them using the class info and class format maps.  When a
217      * class is first loaded from the database, the stored ObjectStreamClass is
218      * compared to the current ObjectStreamClass loaded by the Java class
219      * loader; if they are different, a new class ID is assigned for the
220      * current format.
221      */
222     private ClassInfo getClassInfo(ObjectStreamClass classFormat)
223         throws DatabaseException, ClassNotFoundException {
224 
225         /*
226          * First check for a cached copy of the class info, which if
227          * present always contains the class format object.
228          */
229         String className = classFormat.getName();
230         ClassInfo classInfo = classMap.get(className);
231         if (classInfo != null) {
232             return classInfo;
233         } else {
234             /* Make class info key.  */
235             char[] nameChars = className.toCharArray();
236             byte[] keyBytes = new byte[1 + UtfOps.getByteLength(nameChars)];
237             keyBytes[0] = REC_CLASS_INFO;
238             UtfOps.charsToBytes(nameChars, 0, keyBytes, 1, nameChars.length);
239             DatabaseEntry key = new DatabaseEntry(keyBytes);
240 
241             /* Read class info.  */
242             DatabaseEntry data = new DatabaseEntry();
243             OperationStatus status = db.get(null, key, data, LockMode.DEFAULT);
244             if (status != OperationStatus.SUCCESS) {
245                 
246                 /*
247                  * Not found in the database, write class info and class
248                  * format.
249                  */
250                 classInfo = putClassInfo(new ClassInfo(), className, key,
251                                          classFormat);
252             } else {
253                 /*
254                  * Read class info to get the class format key, then read class
255                  * format.
256                  */
257                 classInfo = new ClassInfo(data);
258                 DatabaseEntry formatData = new DatabaseEntry();
259                 ObjectStreamClass storedClassFormat =
260                     getClassFormat(classInfo.getClassID(), formatData);
261 
262                 /*
263                  * Compare the stored class format to the current class format,
264                  * and if they are different then generate a new class ID.
265                  */
266                 if (!areClassFormatsEqual(storedClassFormat,
267                                           getBytes(formatData),
268                                           classFormat)) {
269                     classInfo = putClassInfo(classInfo, className, key,
270                                              classFormat);
271                 }
272 
273                 /* Update the class info map.  */
274                 classInfo.setClassFormat(classFormat);
275                 classMap.put(className, classInfo);
276             }
277         }
278         return classInfo;
279     }

281     /**
282      * Assign a new class ID (increment the current ID record), write the
283      * ObjectStreamClass record for this new ID, and update the ClassInfo
284      * record with the new ID also.  The ClassInfo passed as an argument is the
285      * one to be updated.
286      */
287     private ClassInfo putClassInfo(ClassInfo classInfo,
288                                    String className,
289                                    DatabaseEntry classKey,
290                                    ObjectStreamClass classFormat)
291         throws DatabaseException {
292 
293         /* An intent-to-write cursor is needed for CDB. */
294         CursorConfig cursorConfig = null;
295         if (cdbMode) {
296             cursorConfig = new CursorConfig();
297             DbCompat.setWriteCursor(cursorConfig, true);
298         }
299         Cursor cursor = null;
300         Transaction txn = null;
301         try {
302             if (txnMode) {
303                 txn = db.getEnvironment().beginTransaction(null, null);
304             }
305             cursor = db.openCursor(txn, cursorConfig);
306 
307             /* Get the current class ID. */
308             DatabaseEntry key = new DatabaseEntry(LAST_CLASS_ID_KEY);
309             DatabaseEntry data = new DatabaseEntry();
310             OperationStatus status = cursor.getSearchKey(key, data,
311                                                          writeLockMode);
312             if (status != OperationStatus.SUCCESS) {
313                 throw DbCompat.unexpectedState("Class ID not initialized");
314             }
315             byte[] idBytes = getBytes(data);
316 
317             /* Increment the ID by one and write the updated record.  */
318             idBytes = incrementID(idBytes);
319             data.setData(idBytes);
320             cursor.put(key, data);
321 
322             /*
323              * Write the new class format record whose key is the ID just
324              * assigned.
325              */
326             byte[] keyBytes = new byte[1 + idBytes.length];
327             keyBytes[0] = REC_CLASS_FORMAT;
328             System.arraycopy(idBytes, 0, keyBytes, 1, idBytes.length);
329             key.setData(keyBytes);
330 
331             ByteArrayOutputStream baos = new ByteArrayOutputStream();
332             ObjectOutputStream oos;
333             try {
334                 oos = new ObjectOutputStream(baos);
335                 oos.writeObject(classFormat);
336             } catch (IOException e) {
337                 throw RuntimeExceptionWrapper.wrapIfNeeded(e);
338             }
339             data.setData(baos.toByteArray());
340 
341             cursor.put(key, data);
342 
343             /*
344              * Write the new class info record, using the key passed in; this
345              * is done last so that a reader who gets the class info record
346              * first will always find the corresponding class format record.
347              */
348             classInfo.setClassID(idBytes);
349             classInfo.toDbt(data);
350 
351             cursor.put(classKey, data);
352 
353             /*
354              * Update the maps before closing the cursor, so that the cursor
355              * lock prevents other writers from duplicating this entry.
356              */
357             classInfo.setClassFormat(classFormat);
358             classMap.put(className, classInfo);
359             formatMap.put(new BigInteger(idBytes), classFormat);
360             return classInfo;
361         } finally {
362             if (cursor != null) {
363                 cursor.close();
364             }
365             if (txn != null) {
366                 txn.commit();
367             }
368         }
369     }

430      * Return whether two class formats are equal.  This determines whether a
431      * new class format is needed for an object being serialized.  Formats must
432      * be identical in all respects, or a new format is needed.
433      */
434     private static boolean areClassFormatsEqual(ObjectStreamClass format1,
435                                                 byte[] format1Bytes,
436                                                 ObjectStreamClass format2) {
437         try {
438             if (format1Bytes == null) { // using cached format1 object
439                 format1Bytes = getObjectBytes(format1);
440             }
441             byte[] format2Bytes = getObjectBytes(format2);
442             return java.util.Arrays.equals(format2Bytes, format1Bytes);
443         } catch (IOException e) { return false; }
444     }

451     private static byte[] getBytes(DatabaseEntry dbt) {
452         byte[] b = dbt.getData();
453         if (b == null) {
454             return null;
455         }
456         if (dbt.getOffset() == 0 && b.length == dbt.getSize()) {
457             return b;
458         }
459         int len = dbt.getSize();
460         if (len == 0) {
461             return ZERO_LENGTH_BYTE_ARRAY;
462         } else {
463             byte[] t = new byte[len];
464             System.arraycopy(b, dbt.getOffset(), t, 0, t.length);
465             return t;
466         }
467     }
