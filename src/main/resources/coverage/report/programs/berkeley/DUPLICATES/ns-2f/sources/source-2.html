79     /**
80      * Applies the new configuration, then cancels and reschedules the verify
81      * task as needed.
82      */
83     public void configVerifyTask(DbConfigManager configMgr) {
84 
85         if (!updateConfig(configMgr)) {
86             return;
87         }
88 
89         synchronized (this) {
90             if (!shutdownRequest) {
91                 cancel();
92 
93                 if (cronSchedule != null) {
94                     verifyTask = new VerifyTask(envImpl);
95 
96                     /*
97                      * Use Timer.scheduleAtFixedRate to instead of
98                      * Timer.schedule, since this is a long running task and
99                      * the next task is NOT expected to be scheduled for
100                      * 24 hours later, it is expected to be scheduled at
101                      * a fixed time.
102                      */
103                     timer.scheduleAtFixedRate(
104                         verifyTask, verifyDelay, verifyInterval);
105                 }
106             }
107         }
108     }

111         if (verifyTask != null) {

117          * EnvironmentMutableConfig.

168     /**
169      * Applies the new configuration and returns whether it changed.
170      */
171     private boolean updateConfig(DbConfigManager configMgr) {
172 
173         /*
174          * If set to false (which is not the default).
175          */
176         if (!configMgr.getBoolean(ENV_RUN_VERIFIER)) {
177             if (cronSchedule == null) {
178                 return false;
179             }
180             cronSchedule = null;
181             verifyDelay = 0;
182             verifyInterval = 0;
183             return true;
184         } else {
185             String newCronSchedule = configMgr.get(VERIFY_SCHEDULE);
186 
187             /*
188              * If the data verifier schedule is set by system property and
189              * it is not set through the JE source code explicitly, then
190              * the system property will be used.
191              *
192              * This is used for JE unit test and JE standalone test.
193              */
194             String sysPropVerifySchedule =
195                 System.getProperty(VERIFIER_SCHEDULE);
196             if (sysPropVerifySchedule != null &&
197                 !configMgr.isSpecified(VERIFY_SCHEDULE)) {
198                 newCronSchedule = sysPropVerifySchedule;
199             }
200 
201             if (CronScheduleParser.checkSame(cronSchedule, newCronSchedule)) {
202                 return false;
203             }
204             CronScheduleParser csp = new CronScheduleParser(newCronSchedule);
205             verifyDelay = csp.getDelayTime();
206             verifyInterval = csp.getInterval();
207             cronSchedule = newCronSchedule;
208             return true;
209         }
210     }

212     class VerifyTask extends TimerTask {
213         private final EnvironmentImpl envImpl;
214         private volatile boolean isRunning;
215 
216         VerifyTask(EnvironmentImpl envImpl) {
217             this.envImpl = envImpl;
218         }
219         
220         private void updateConfig() {
221             DbConfigManager configMgr = envImpl.getConfigManager();
222 
223             verifyLog = configMgr.getBoolean(VERIFY_LOG);
224             verifyBtree = configMgr.getBoolean(VERIFY_BTREE);
225 
226             dbLogVerifier.setReadDelay(
227                 configMgr.getDuration(VERIFY_LOG_READ_DELAY),
228                 TimeUnit.MILLISECONDS);
229 
230             final VerifyConfig btreeVerifyConfig = new VerifyConfig();
231             btreeVerifyConfig.setVerifySecondaries(
232                 configMgr.getBoolean(VERIFY_SECONDARIES));
233             btreeVerifyConfig.setVerifyDataRecords(
234                 configMgr.getBoolean(VERIFY_DATA_RECORDS));
235             btreeVerifyConfig.setVerifyObsoleteRecords(
236                 configMgr.getBoolean(VERIFY_OBSOLETE_RECORDS));
237             btreeVerifyConfig.setBatchSize(
238                 configMgr.getInt(VERIFY_BTREE_BATCH_SIZE));
239             btreeVerifyConfig.setBatchDelay(
240                 configMgr.getDuration(VERIFY_BTREE_BATCH_DELAY),
241                 TimeUnit.MILLISECONDS);
242             dbTreeVerifier.setBtreeVerifyConfig(btreeVerifyConfig);
243 
244             /*
245              * 1. Why call dbTreeVerifier.setVerifyFlag here, rather than
246              * immediately after call cancel in configVerifyTask?
247              *    If we do that, then maybe before BtreeVerifier really gets
248              *  the stop status of the flag, we set the status of the flag
249              *  to be OK again. Then the previous task can not be stopped
250              *  as soon as possible.
251              *    If we do here, this means that one new task has already
252              *    started, so the previous task has already been stopped.
253              * 2. Why use synchronized?
254              *    Consider the following scenario:
255              *           shutdown Thread                 task Thread
256              *
257              *                                         !shutdownRequest
258              *      
259              *      shutdownRequest = true
260              *      dbTreeVerifier.setStopVerifyFlag
261              *
262              *                                     dbTreeVerifier.setVerifyFlag
263              */
264             synchronized (DataVerifier.this) {
265                 if (!shutdownRequest) {
266                     dbLogVerifier.setStopVerifyFlag(false);
267                     dbTreeVerifier.setStopVerifyFlag(false);
268                 }
269             }
270         }
271 
272         @Override
273         public void run() {
274             /*
275              * If the scheduled execution time of the scheduled run lags
276              * very far from the current time due to the long-time current
277              * verification, the scheduled run is just skipped.
278              */
279             if (System.currentTimeMillis() - scheduledExecutionTime() >=
280                 envImpl.getConfigManager().getDuration(VERIFY_MAX_TARDINESS)) {
281                 return;
282             }
283 
284             isRunning = true;
285             boolean success = false;
286             updateConfig();
287             try {
288                 if (verifyLog) {
289                     dbLogVerifier.verifyAll();
290                 }
291                 if (verifyBtree) {
292                     dbTreeVerifier.verifyAll();
293                 }
294                 success = true;
295             } catch (EnvironmentFailureException efe) {
296                 /* Do nothing. Just cancel this timer in finally. */
297             } catch (Throwable e) {
298                 if (envImpl.isValid()) {
299                     StoppableThread.handleUncaughtException(
300                         envImpl.getLogger(), envImpl, Thread.currentThread(),
301                         e);
302                 }
303             } finally {
304                 if (!success) {
305                     requestShutdown();
306                 }
307                 isRunning = false;
308             }
309         }
310     }
