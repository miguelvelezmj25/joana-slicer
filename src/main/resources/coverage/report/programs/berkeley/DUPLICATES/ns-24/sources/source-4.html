335     /**
336      * Rollback all write operations that are logged with an LSN > the
337      * matchpointLsn parameter. This is logically a truncation of the log
338      * entries written by this transaction. Any log entries created by this
339      * transaction are marked obsolete.
340      *
341      * Note that this is by no means a complete implementation of what would be
342      * needed to support user visible savepoints. This method only rolls back
343      * write operations and doesn't handle other types of state, like read
344      * locks and open cursors.
345      *
346      * There are several key assumptions:
347      * - the transaction does not hold read locks.
348      * - the transaction will either be resumed, and any rolled back
349      * operations will be repeated, or the transaction will be aborted
350      * in its entirety.
351      *
352      * If all operations in the transaction are rolled back, this transaction
353      * is also unregistered and closed.
354      *
355      * Rolling back a log entry through rollback is akin to truncating the
356      * transactional log. The on-disk entries should not be referred to by
357      * anything in the in-memory tree or the transaction chain. JE's append
358      * only storage and the fact that the transactional log entries are
359      * intertwined through the physical log prohibits any log truncation. To
360      * mimic log truncation, any rolled back log entry is marked as
361      * obsolete. Since only the last version of any data record is alive,
362      * any future uses of this transaction must use the obsoleteDupsAllowed
363      * option (see Txn.countObsoleteExact) to prevent asserts about duplicate
364      * obsolete offsets. For example, suppose the transaction logs this:
365      *
366      * 100 LNa (version1)
367      * 200 LNa (version2)
368      * 300 LNa (version3)
369      *
370      * At this point in time, LSN 100 and 200 are obsolete.
371      *
372      * Now, suppose we roll back to LSN 100. LSNs 200 and 300 are marked
373      * obsolete by the rollback.(although LSN 200 was already obsolete).  It is
374      * true that for an instance in time LSN 100 is incorrectly marked as
375      * obsolete, when it's really alive. But this transaction is going to
376      * either abort or resume exactly as it was before, so LSN 100 is going to
377      * be obsolete again.
378      *
379      * Suppose txn.abort() is called. The abort() logic will mark LSN 100 as
380      * obsolete, since it is the latest version of the record in the
381      * transaction. Using the obsoleteDupsAllowed option avoids an assertion on
382      * the double recording of LSN 100.
383      *
384      * Alternatively, suppose LNa (version2) is retransmitted and logged as LSN
385      * 400. Normal execution of LN.log() marks LSN 100 as obsolete, which would
386      * trigger the assertion were it not for obsoleteDupsAllowed.
387      *
388      * @return list of LSNs that were rolled back
389      */
390     public Collection<Long> rollback(long matchpointLsn)
391         throws DatabaseException {
392 
393         List<Long> rollbackLsns = new ArrayList<Long>();
394         LoggerUtils.finest(logger, envImpl, "Partial Rollback of " + this);
395         synchronized (this) {
396             checkState(true);
397 
398             /* This transaction didn't log anything, nothing to rollback. */
399             if (lastLoggedLsn == NULL_LSN) {
400                 return rollbackLsns;
401             }
402 
403             /*
404              * This transaction doesn't include any operations that are after
405              * the matchpointLsn. There is nothing to rollback.
406              */
407             if (DbLsn.compareTo(lastLoggedLsn, matchpointLsn) <= 0) {
408                 return rollbackLsns;
409             }
410 
411             setRollback();
412             undoWrites(matchpointLsn, rollbackLsns);
413         }
414 
415         /*
416          * The call to undoWrites() may have rolled everything back, and set
417          * lastLoggedLsn to NULL_LSN.
418          */
419         if (lastLoggedLsn == NULL_LSN) {
420             /* Everything was rolled back. */
421             try {
422 
423                 /*
424                  * Purge any databaseImpls not needed as a result of the abort.
425                  * Be sure to do this outside the synchronization block, to
426                  * avoid conflict w/checkpointer.
427                  */
428                 cleanupDatabaseImpls(false);
429             } finally {
430                 close(false /* isCommit */);
431             }
432         }
433 
434         /*
435          * We don't expect there to be any database handles associated with
436          * a ReplayTxn, because only DatabaseImpls are used. Because of that,
437          * there should be no cleanup needed.
438          */
439         if (openedDatabaseHandles != null) {
440             throw EnvironmentFailureException.unexpectedState
441                 ("Replay Txn " + getId() + " has a openedDatabaseHandles");
442         }
443 
444         /*
445          * There is no need to call cleanupDatabaseImpls if the txn still holds
446          * locks. The operations in this txn will either be entirely aborted,
447          * or will be repeated, so any cleanup will happen when the txn ends.
448          */
449         return rollbackLsns;
450     }

452     /**
453      * Rollback the changes to this txn's write locked nodes up to but not
454      * including the entry at the specified matchpoint. When we log a
455      * transactional entry, we record the LSN of the original,
456      * before-this-transaction version as the abort LSN. This means that if
457      * there are multiple updates to a given record in a single transaction,
458      * each update only references that original version and its true
459      * predecessor.
460      *
461      * This was done to streamline abort processing, so that an undo reverts
462      * directly to the original version rather than stepping through all the
463      * intermediates. The intermediates are skipped.  However, undo to a
464      * matchpoint may need to stop at an intermediate point, so we need to
465      * create a true chain of versions.
466      *
467      * To do so, we read the transaction backwards from the last logged LSN
468      * to reconstruct a transaction chain that links intermediate versions
469      * of records. For example, suppose our transaction looks like this and
470      * that we are undoing up to LSN 250
471      *
472      * lsn=100 node=A (version 1)
473      * lsn=200 node=B (version 1)
474      *                        <-- matchpointLsn
475      * lsn=300 node=C (version 1)
476      * lsn=400 node=A (version 2)
477      * lsn=500 node=B (version 2)
478      * lsn=600 node=A (version 3)
479      * lsn=700 node=A (version 4)
480      *
481      * To setup the old versions, We walk from LSN 700 -> 100
482      *   700 (A) rolls back to 600
483      *   600 (A) rolls back to 400
484      *   500 (B) rolls back to 200
485      *   400 (A) rolls back to 100
486      *   300 (C) rolls back to an empty slot (NULL_LSN).
487      *
488      * A partial rollback also requires resetting the lastLoggedLsn field,
489      * because these operations are no longer in the btree and their on-disk
490      * entries are no longer valid.
491      *
492      * Lastly, the appropriate write locks must be released.
493      * @param matchpointLsn the rollback should go up to but not include this
494      * LSN.
495      */
496     private void undoWrites(long matchpointLsn, List<Long> rollbackLsns)
497         throws DatabaseException {
498 
499         /*
500          * Generate a map of node->List of intermediate LSNs for this node.
501          * to re-create the transaction chain.
502          */
503         TreeLocation location = new TreeLocation();
504         Long undoLsn = lastLoggedLsn;
505 
506         TxnChain chain =
507             new TxnChain(undoLsn, id, matchpointLsn, undoDatabases, envImpl);
508 
509         try {
510             while ((undoLsn != DbLsn.NULL_LSN) &&
511                     DbLsn.compareTo(undoLsn, matchpointLsn) > 0) {
512 
513                 UndoReader undo =
514                     UndoReader.create(envImpl, undoLsn, undoDatabases);
515 
516                 RevertInfo revertTo = chain.pop();
517 
518                 logFinest(undoLsn, undo, revertTo);
519 
520                 /*
521                  * When we undo this log entry, we've logically truncated
522                  * it from the log. Remove it from the btree and mark it
523                  * obsolete.
524                  */
525                 RecoveryManager.rollbackUndo(
526                     logger, Level.FINER, location,
527                     undo.db, undo.logEntry, undoLsn, revertTo);
528 
529                 countObsoleteInexact(undoLsn, undo);
530                 rollbackLsns.add(undoLsn);
531 
532                 /*
533                  * Move on to the previous log entry for this txn and update
534                  * what is considered to be the end of the transaction chain.
535                  */
536                 undoLsn = undo.logEntry.getUserTxn().getLastLsn();
537                 lastLoggedLsn = undoLsn;
538             }
539 
540             /*
541              * Correct the fields which hold LSN and VLSN state that may
542              * now be changed.
543              */
544             lastApplied = chain.getLastValidVLSN();
545             if (!updateLoggedForTxn()) {
546                 firstLoggedLsn = NULL_LSN;
547             }
548 
549         } catch (DatabaseException e) {
550             LoggerUtils.traceAndLogException(envImpl, "Txn", "undo",
551                                      "For LSN=" +
552                                      DbLsn.getNoFormatString(undoLsn), e);
553             throw e;
554         } catch (RuntimeException e) {
555             throw EnvironmentFailureException.unexpectedException
556                 ("Txn undo for LSN=" + DbLsn.getNoFormatString(undoLsn), e);
557         }
558 
559         if (lastLoggedLsn == DbLsn.NULL_LSN) {
560 
561             /*
562              * The whole txn is rolled back, and it may not appear again. This
563              * is the equivalent of an abort. Do any delete processing for an
564              * abort which is needed.
565              *
566              * Set database state for deletes before releasing any write
567              * locks.
568              */
569             setDeletedDatabaseState(false);
570         }
571 
572         /* Clear any write locks that are no longer needed. */
573         clearWriteLocks(chain.getRemainingLockedNodes());
574     }

576     /**
577      * Count an LN obsolete that is being made invisble by rollback.
578      *
579      * Use inexact counting.  Since invisible entries are not processed by the
580      * cleaner, recording the obsolete offset would be a waste of resources.
581      * Since we don't count offsets, we don't need to worry about duplicate
582      * offsets.
583      *
584      * Some entries may be double counted if they were previously counted
585      * obsolete, for example, when multiple versions of an LN were logged.
586      * This is tolerated for an exceptional situation like rollback.
587      */
588     private void countObsoleteInexact(long undoLsn, UndoReader undo) {
589 
590         /*
591          * "Immediately obsolete" LNs are counted as obsolete when they are
592          * logged, so no need to repeat here.
593          */
594         if (undo.logEntry.isImmediatelyObsolete(undo.db)) {
595             return;
596         }
597 
598         envImpl.getLogManager().countObsoleteNode(undoLsn,
599                                                   null, /*type*/
600                                                   undo.logEntrySize,
601                                                   undo.db,
602                                                   false /*countExact*/);
603     }
