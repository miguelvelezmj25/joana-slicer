82     public LocalTransaction getLocalTransaction() {
83 
84         /*
85          * If there is no JEConnection associated with this ManagedConnection
86          * yet, then the ManagedConnection holds on to the JELocalTransaction.
87          * Once a JEConnection is associated (it may not ever happen), we hand
88          * off the JELocalTransaction to the JEConnection and forget about it
89          * in the ManagedConnection.
90          */
91         if (conn == null) {
92             savedLT = new JELocalTransaction(env, savedTransConfig, this);
93             return savedLT;
94         }
95 
96         JELocalTransaction lt = conn.getLocalTransaction();
97         if (lt == null) {
98             if (savedLT == null) {
99                 lt = new JELocalTransaction(env, savedTransConfig, this);
100             } else {
101                 lt = savedLT;
102             }
103             conn.setLocalTransaction(lt);
104             savedLT = null;
105         }
106         return lt;
107     }

147     protected void sendConnectionEvent(ConnectionEvent connEvent) {
148         for (int i = listeners.size() - 1; i >= 0; i--) {
149             ConnectionEventListener listener =
150                 listeners.get(i);
151             if (connEvent.getId() == ConnectionEvent.CONNECTION_CLOSED) {
152                 listener.connectionClosed(connEvent);
153             } else if (connEvent.getId() ==
154                        ConnectionEvent.CONNECTION_ERROR_OCCURRED) {
155                 listener.connectionErrorOccurred(connEvent);
156             } else if (connEvent.getId() ==
157                        ConnectionEvent.LOCAL_TRANSACTION_STARTED) {
158                 listener.localTransactionStarted(connEvent);
159             } else if (connEvent.getId() ==
160                        ConnectionEvent.LOCAL_TRANSACTION_COMMITTED) {
161                 listener.localTransactionCommitted(connEvent);
162             } else if (connEvent.getId() ==
163                        ConnectionEvent.LOCAL_TRANSACTION_ROLLEDBACK) {
164                 listener.localTransactionRolledback(connEvent);
165             }
166         }
167     }

242     private Database

244                              String dbName,
245                              DatabaseConfig config)
246         throws DatabaseException {
247 
248         Database db;
249         if (config.getExclusiveCreate()) {
250             db = env.openDatabase(null, dbName, config);
251             databaseHandleCache.put(dbName, db);
252         } else {
253             db = databaseHandleCache.get(dbName);
254             if (db == null) {
255                 db = env.openDatabase(null, dbName, config);
256                 databaseHandleCache.put(dbName, db);

259             }
