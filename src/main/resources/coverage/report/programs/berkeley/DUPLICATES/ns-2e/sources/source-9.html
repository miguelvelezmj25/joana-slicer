41 /**
42  * @hidden
43  * For internal use only.
44  * DbRunAction is a debugging aid that can invoke a JE operation or background
45  * activity from the command line.
46  *
47  *   batchClean calls Environment.cleanLog() in a loop
48  *   checkpoint calls Environment.checkpoint() with force=true
49  *   compress calls Environment.compress
50  *   evict calls Environment.preload, then evictMemory
51  *   removeDb calls Environment.removeDatabase, but doesn't do any cleaning
52  *   removeDbAndClean calls removeDatabase, then cleanLog in a loop
53  *   activateCleaner wakes up the cleaner, and then the main thread waits
54  *     until you type "y" to the console before calling Environment.close().
55  *     The control provided by the prompt is necessary for daemon activities
56  *     because often threads check and bail out if the environment is closed.
57  *   verifyUtilization calls CleanerTestUtils.verifyUtilization() to compare
58  *     utilization as calculated by UtilizationProfile to utilization as
59  *     calculated by UtilizationFileReader.
60  */
61 public class DbRunAction {

308     private static void preload(Environment env, String dbName)
309         throws Exception {
310 
311         System.out.println("Preload starting");
312         Database db = env.openDatabase(null, dbName, null);
313         Cursor cursor = db.openCursor(null, null);
314         try {
315             DatabaseEntry key = new DatabaseEntry();
316             DatabaseEntry data = new DatabaseEntry();
317             int count = 0;
318             while (cursor.getNext(key, data, LockMode.DEFAULT) ==
319                    OperationStatus.SUCCESS) {
320                 count++;
321                 if ((count % 50000) == 0) {
322                     System.out.println(count + "...");
323                 }
324             }
325             System.out.println("Preloaded " + count + " records");
326         } finally {
327             cursor.close();
328             db.close();
329         }

336         /* Push the cache size down by half to force eviction. */

339         EnvironmentMutableConfig c = new EnvironmentMutableConfig();

342 
343         long start = System.currentTimeMillis();

345         long end = System.currentTimeMillis();
346
