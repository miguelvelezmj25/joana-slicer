24 /**
25  * Read-only catalog operations used when initializing new formats.  This
26  * catalog is used temprarily when the main catalog has not been updated yet,
27  * but the new formats need to do catalog lookups.
28  *
29  * @see PersistCatalog#addNewFormat
30  *
31  * @author Mark Hayes
32  */
33 class ReadOnlyCatalog implements Catalog {
34 
35     private final ClassLoader classLoader;
36     private List<Format> formatList;
37     private Map<String, Format> formatMap;
38 
39     ReadOnlyCatalog(ClassLoader classLoader,
40                     List<Format> formatList,
41                     Map<String, Format> formatMap) {
42         this.classLoader = classLoader;
43         this.formatList = formatList;
44         this.formatMap = formatMap;
45     }
46 
47     public int getInitVersion(Format format, boolean forReader) {
48         return Catalog.CURRENT_VERSION;
49     }
50 
51     public Format getFormat(int formatId, boolean expectStored) {
52         try {
53             Format format = formatList.get(formatId);
54             if (format == null) {
55                 throw DbCompat.unexpectedState
56                     ("Format does not exist: " + formatId);
57             }
58             return format;
59         } catch (NoSuchElementException e) {
60             throw DbCompat.unexpectedState
61                 ("Format does not exist: " + formatId);
62         }
63     }
64 
65     public Format getFormat(Class cls, boolean checkEntitySubclassIndexes) {
66         Format format = formatMap.get(cls.getName());
67         if (format == null) {
68             throw new IllegalArgumentException
69                 ("Class is not persistent: " + cls.getName());
70         }
71         return format;
72     }
73 
74     public Format getFormat(String className) {

76     }
