38 /**
39  * Implementation of the LDiff service that process requests from the LDiff
40  * client. It's the network level interface to the remote processing done as

42  *

98         public void runLDiff(DbBlocks request, Protocol protocol)
99             throws IOException {
100 
101             Database db = null;
102             Cursor cursor = null;
103             try{
104                 db = env.openDatabase
105                     (null, request.getDbName(), dbConfig);
106                 protocol.write(protocol.new BlockListStart(), channel);
107                 LDiffConfig cfg = new LDiffConfig();
108                 cfg.setBlockSize(request.getBlockSize());
109                 LDiff ldf = new LDiff(cfg);
110                 /* Use the Iterator to stream the blocks across the wire. */
111                 Iterator<Block> blocks = ldf.iterator(db);
112                 while (blocks.hasNext()) {
113                     protocol.write(protocol.new BlockInfo(blocks.next()),
114                             channel);
115                 }
116                 protocol.write(protocol.new BlockListEnd(), channel);
117 
118                 /* Start to do the record difference analysis. */
119                 Message msg = protocol.read(channel);
120                 if (msg.getOp() == Protocol.REMOTE_DIFF_REQUEST) {
121                     cursor = db.openCursor(null, null);
122                     sendDiffArea(cursor, (RemoteDiffRequest) msg, protocol);
123                     runDiffAnalysis(cursor, protocol);
124                 } else if (msg.getOp() != Protocol.DONE) {
125                     protocol.write(protocol.new ProtocolError
126                             ("Invalid message: " + msg), channel);
127                 }
128             } catch (DatabaseNotFoundException e) {
129                 protocol.write(protocol.new DbMismatch(e.getMessage()),
130                                channel);
131             } finally {
132                 if (cursor != null) {
133                     cursor.close();
134                 }
135                 if (db != null) {
136                     db.close();
137                 }
138             }
139         }

141         /* Get records for all different areas and send out. */
142         private void runDiffAnalysis(Cursor cursor,
143                                      Protocol protocol)
144             throws IOException {
145 
146             while (true) {
147                 Message msg = protocol.read(channel);
148                 if (msg.getOp() == Protocol.REMOTE_DIFF_REQUEST) {
149                     sendDiffArea(cursor, (RemoteDiffRequest) msg, protocol);
150                 } else {
151                     if (msg.getOp() != Protocol.DONE) {
152                         protocol.write(protocol.new ProtocolError
153                                 ("Invalid message: " + msg), channel);
154                     }
155                     break;
156                 }
157             }
158         }
159 
160         /* Send the different records of an area to the requested machine. */
161         private void sendDiffArea(Cursor cursor,
162                                   RemoteDiffRequest request,
163                                   Protocol protocol)
164             throws IOException {
165 
166             /* Get the records in the different area. */
167             HashSet<Record> records = null;
168             try {
169                 records = DiffRecordAnalyzer.getDiffArea(cursor, request);
170             } catch (Exception e) {
171                 protocol.write(protocol.new Error(e.getMessage()), channel);
172                 throw new LDiffRecordRequestException(e.getMessage());
173             }
174 
175             /* Write them out to the requested machine. */
176             protocol.write(protocol.new DiffAreaStart(), channel);
177             for (Record record: records) {
178                 protocol.write(protocol.new RemoteRecord(record), channel);
179             }
180             protocol.write(protocol.new DiffAreaEnd(), channel);
181         }

187                            (env.getDatabaseNames().size()), channel);
188         }

190         @Override
191         public void run() {
192             final Protocol protocol;
193 
194             try {
195                 env = repImpl.makeEnvironment();
196                 protocol = new Protocol(new NameIdPair("Ldiff", -1), repImpl);
197                 try {
198                     channel.getSocketChannel().configureBlocking(true);
199                     Message msg = protocol.read(channel);
200                     if (msg.getOp() == Protocol.DB_BLOCKS) {
201                         runLDiff((DbBlocks)msg, protocol);
202                     } else if (msg.getOp() == Protocol.ENV_DIFF) {
203                         runEnvDiff((EnvDiff)msg, protocol);
204                     }
205                 } catch (ProtocolException e) {
206                     /* Unexpected message. */
207                     protocol.write
208                         (protocol.new ProtocolError(e.getMessage()),
209                          channel);
210                 } finally {
211                     if (channel.isOpen()) {
212                         channel.close();
213                     }
214                 }
215             } catch (IOException e) {
216 
217                 /*
218                  * Channel has already been closed, or the close itself
219                  * failed.
220                  */
221             } finally {
222                 if (env != null) {
223                     env.close();
224                 }
225                 if (!busy.compareAndSet(true, false)) {
226                     throw EnvironmentFailureException.unexpectedState
227                         ("Service is not busy");
228                 }
229             }
230         }
