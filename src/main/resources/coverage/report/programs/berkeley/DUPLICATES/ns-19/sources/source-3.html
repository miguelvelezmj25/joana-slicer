61 /**
62  * Represents the environment in which elections are run on behalf of a node.
63  * There is exactly one instance of an Elections for each node. Elections are
64  * initiated via this class.
65  *
66  * One of the primary goals of this interface is to keep Elections as free
67  * standing as possible, so that we can change how elections are held, or
68  * aspects of the election infrastructure with minimal impact on replication.
69  * For example, elections currently used tcp for communication of election
70  * messages but may want to switch over to udp. Such a change should be
71  * confined to just the Elections module. Other changes might include changes
72  * to the strategy used to suggest Values and the weight associated with a
73  * suggested Value.
74  *
75  * The following are the principal points of interaction between Elections and
76  * Replication:
77  *
78  * 1) The initiation of elections via the initiateElections() method.
79  *
80  * 2) The suggestion of nodes as masters and the ranking of the
81  * suggestion. This is done via the Acceptor.SuggestionGenerator interface. An
82  * instance of this interface is supplied when the Elections class is
83  * instantiated. Note that the implementation must also initiate a freeze of
84  * VLSNs to ensure that the ranking does not change as the election
85  * progresses. The VLSN can make progress when the node is informed via its
86  * Listener that an election with a higher Proposal number (than the one in the
87  * Propose request) has finished.
88  *
89  * 3) Obtaining the result of an election initiated in step 1. This is done via
90  * the Learner.Listener interface. An instance of this class is supplied when
91  * the Election class is first instantiated.
92  *
93  */

325      * Initiates an election. Note that this may just be one of many possible
326      * elections that are in progress in a replication group. The method does
327      * not wait for this election to complete, but instead returns as soon as
328      * any election result (including one initiated by some other Proposer)
329      * becomes available via the Learner.
330      *
331      * A proposal submitted as part of this election may lose out to other
332      * concurrent elections, or there may not be a sufficient number of
333      * Acceptor agents active or reachable to reach a quorum. In such cases,
334      * the election will not produce a result. That is, there will be no
335      * notification to the Learners. Note that only one election can be
336      * initiated at a time at a node If a new election is initiated while one
337      * is already in progress, then the method will wait until it completes
338      * before starting a new one.
339      *
340      * The results of this and any other elections that may have been initiated
341      * concurrently by other nodes are made known to the Learner agents. Note
342      * that this method does not return a result, since the concurrent arrival
343      * of results could invalidate the result even before its returned.
344      *
345      * @param newGroup the definition of the group to be used for this election
346      * @param quorumPolicy the policy to be used to reach a quorum.
347      * @param maxRetries the max number of times a proposal may be retried
348      * @throws InterruptedException
349      */
350     public synchronized void initiateElection(RepGroupImpl newGroup,
351                                               QuorumPolicy quorumPolicy,
352                                               int maxRetries)
353         throws InterruptedException {
354 
355         updateRepGroup(newGroup);
356         long startTime = System.currentTimeMillis();
357         nElections++;
358         LoggerUtils.logMsg(logger, envImpl, formatter, Level.INFO,
359                            "Election initiated; election #" + nElections);
360         if (electionThread != null) {
361             /*
362              * The factor of four used below to arrive at a timeout value is a
363              * heuristic: A factor of two to cover any pending message exchange
364              * and another factor of two as a grace period. We really don't
365              * expect to hit this timeout in the absence of networking issues,
366              * hence the thread dump to understand the reason in case there's
367              * some bug.
368              */
369             final int waitMs = protocol.getReadTimeout() * 4;
370             // A past election request, wait until the election has quiesced
371             LoggerUtils.logMsg(logger, envImpl, formatter, Level.INFO,
372                                "Election in progress. Waiting ... for " +
373                                waitMs + "ms");
374 
375             electionThread.join(waitMs);
376             if (electionThread.isAlive()) {
377                 /* Dump out threads for future analysis if it did not quit. */
378                 LoggerUtils.logMsg(logger, envImpl, formatter, Level.INFO,
379                                    "Election did not finish as expected." +
380                                    " resorting to shutdown");
381                 LoggerUtils.fullThreadDump(logger, envImpl, Level.INFO);
382                 electionThread.shutdown();
383             }
384 
385             final Exception exception =
386                 electionThread.getSavedShutdownException();
387             if (exception != null) {
388                 throw new EnvironmentFailureException
389                 (envImpl,
390                  EnvironmentFailureReason.UNEXPECTED_EXCEPTION,
391                  exception);
392             }
393         }
394 
395         CountDownLatch countDownLatch = null;
396         synchronized (electionListener) {
397             // Watch for any election results from this point forward
398             countDownLatch = electionListener.setLatch();
399         }
400 
401         RetryPredicate retryPredicate =
402             new RetryPredicate(repNode, maxRetries, countDownLatch);
403         electionThread = new ElectionThread(quorumPolicy, retryPredicate,
404                                             envImpl,
405                                             (envImpl == null) ? null :
406                                             envImpl.getName());
407 
408         electionThread.start();
409         try {
410             /* Wait until we hear of some "new" election result */
411             countDownLatch.await();
412             if (retryPredicate.pendingRetries <= 0) {
413                 /* Ran out of retries -- a test situation */
414                 LoggerUtils.logMsg(logger, envImpl, formatter, Level.INFO,
415                                    "Retry count exhausted: " +
416                                    retryPredicate.maxRetries);
417             }
418 
419             /*
420              * Note that the election thread continues to run past this point
421              * and may be active upon re-entry
422              */
423         } catch (InterruptedException e) {
424             LoggerUtils.logMsg(logger, envImpl, formatter, Level.WARNING,
425                                "Election initiation interrupted");
426             shutdown();
427             throw e;
428         }
429         LoggerUtils.logMsg(logger, envImpl, formatter, Level.INFO,
430                            "Election finished. Elapsed time: " +
431                            (System.currentTimeMillis() - startTime) + "ms");
432     }

654          * the primary.
655          */
656         private final int primaryRetries;
657 
658         private static final int BACKOFF_SLEEP_MIN = 1;
659         private static final int BACKOFF_SLEEP_MAX = 32;
660 
661         private int backoffSleepInterval = BACKOFF_SLEEP_MIN;
662 
663         RetryPredicate(RepNode repNode,
664                        int maxRetries,
665                        CountDownLatch electionLatch) {
666             this.repNode = repNode;
667             this.maxRetries = maxRetries;
668             pendingRetries = maxRetries;
669             this.electionLatch = electionLatch;
670             final RepImpl repImpl = repNode.getRepImpl();
671             final IntConfigParam retriesParam =
672                 RepParams.ELECTIONS_PRIMARY_RETRIES;
673             primaryRetries = (repImpl != null) ?
674                  repImpl.getConfigManager().getInt(retriesParam) :
675                  Integer.parseInt(retriesParam.getDefault());
676         }

678         /**
679          * Returns the time to backoff before a retry. The backoff is
680          * non-linear.
681          *
682          * @return the time to backoff in ms
683          */
684         private int backoffWaitTime() {
685             backoffSleepInterval =
686                 Math.min(BACKOFF_SLEEP_MAX, backoffSleepInterval * 2);
687             return backoffSleepInterval * 1000;
688         }

690         /**
691          * Implements the protocol
692          */
693         @Override
694         public boolean retry() throws InterruptedException {
695             if ((maxRetries - pendingRetries) >= primaryRetries) {
696                 if ((repNode != null) &&
697                      repNode.getArbiter().activateArbitration()) {
698                     pendingRetries = maxRetries;
699                     return true;
700                 }
701             }
702             if (pendingRetries-- <= 0) {
703                 /* Free up the main election thread */
704                 electionLatch.countDown();
705                 return false;
706             }
707 
708             electionLatch.await(backoffWaitTime(), TimeUnit.MILLISECONDS);
709             if (electionLatch.getCount() == 0) {
710                 /* An election completed, we can quit issuing proposals */
711                 return false;
712             }
713             /* Timed out and did not hear any election results. */
714             return true;
715         }

717         /**
718          * The number of times a retry was attempted
719          */
720         @Override
721         public int retries() {
722             return (maxRetries-pendingRetries);
723         }

767                  * nodes of election results, but continuing to only wait for
768                  * the completion of notifications to ELECTABLE nodes.  That
769                  * change would increase the chance that SECONDARY nodes have
770                  * up-to-date information about the master, but would avoid

907     /**
908      * Task to re-inform learners of election results by re-broadcasting the
909      * results of an election from the master. This re-broadcast is intended to
910      * help in network partition situations. See [#20220] for details.
911      */
912     private class RebroadcastTask extends TimerTask {
913 
914         /* Lock to ensure that async executions don't overlap. */
915         private final ReentrantLock lock = new ReentrantLock();
916         private int acquireFailCount = 0;
917         private final int periodMs;
918 
919         public RebroadcastTask(int periodMs) {
920             this.periodMs = periodMs;
921         }
922 
923         public int getPeriod() {
924            return periodMs;
925         }
926 
927         /**
928          * If the node is a master, it broadcasts election results to nodes
929          * that are not currently connected to it via feeders.
930          *
931          * It's worth noting that since this is a timer task method it must be
932          * be lightweight. So the actual broadcast is done in an asynchronous
933          * method using a thread from the election thread pool.
934          */
935         @Override
936         public void run() {
937             try {
938                 if (!lock.tryLock()) {
939                     if ((++acquireFailCount % 100) == 0) {
940                         LoggerUtils.logMsg(logger, envImpl, formatter,
941                                            Level.WARNING,
942                                            "Failed to acquire lock after " +
943                                            acquireFailCount + " retries");
944 
945                     }
946                     return;
947                 }
948                 acquireFailCount = 0;
949                 if (!repNode.getMasterStatus().isGroupMaster()) {
950                     return;
951                 }
952 
953                 /*
954                  * Re-informing when the node is a master is just an
955                  * optimization, it does not impact correctness. Further
956                  * minimize network traffic by trying just the nodes that are
957                  * currently disconnected.
958                  */
959                 final FeederManager feederManager = repNode.feederManager();
960                 final Set<String> active = feederManager.activeReplicas();
961                 active.add(repNode.getNodeName());
962 
963                 final Set<InetSocketAddress> learners =
964                     new HashSet<>();
965                 for (final RepNodeImpl rn : repGroup.getAllLearnerMembers()) {
966                     if (!active.contains(rn.getName())) {
967                         learners.add(rn.getSocketAddress());
968                     }
969                 }
970 
971                 if (learners.size() == 0) {
972                     return;
973                 }
974 
975                 LoggerUtils.logMsg(logger, envImpl, formatter, Level.FINE,
976                                    "informing learners:" +
977                                    Arrays.toString(learners.toArray()) +
978                                    " active: " +
979                                    Arrays.toString(active.toArray()));
980 
981                 pool.execute(new Runnable() {
982                         @Override
983                         public void run() {
984                             learner.reinformLearners(learners, pool);
985                         }
986                     });
987             } catch (Exception e) {
988                 LoggerUtils.logMsg(logger, envImpl, formatter, Level.SEVERE,
989                                    "Unexpected exception:" + e.getMessage());
990             } finally {
991                 if (lock.isHeldByCurrentThread()) {
992                     lock.unlock();
993                 }
994             }
995         }
996     }
