40 /**
41  * Summarizes the utilized and unutilized portion of each log file by examining
42  * each log entry.  Does not use the Cleaner UtilizationProfile information in
43  * order to provide a second measure against which to evaluation the
44  * UtilizationProfile accuracy.
45  *
46  * Limitations
47  * ===========
48  * BIN-deltas are all considered obsolete, as an implementation short cut and
49  * for efficiency. 90% (by default) of deltas are obsolete anyway, and it
50  * would be expensive to fetch the parent BIN to find the lookup key.
51  *
52  * Assumes that any currently open transactions will be committed.  For
53  * example, if a deletion or update has been performed but not yet committed,
54  * the old record will be considered obsolete.  Perhaps this behavior could be
55  * changed in the future by attempting to lock a record (non-blocking) and
56  * considering a locked record to be non-obsolete; this might make it match
57  * live utilization counting more closely.
58  *
59  * Accesses the Btree, using JE cache memory if necessary and contending with
60  * other accessors, to check whether an entry is active.
61  *
62  * Historical note: This implementation, which uses the Btree to determine
63  * whether a node is active, replaced an earlier implementation that attempted
64  * to duplicate the Btree in memory and read the entire log.  This older
65  * implementation had inaccuracies and was less efficient.  With the new
66  * implementation it is also possible to calculation utilization for a range of
67  * LSNs, reading only that portion of the log.  [#22208]
68  */

98     protected boolean isTargetEntry() {
99 
100         /* 
101          * UtilizationTracker is supposed to mimic the UtilizationProfile. 
102          * Accordingly it does not count the file header or invisible log 
103          * entries because those entries are not covered by the U.P.
104          */
105         return ((currentEntryHeader.getType() !=

107                 !currentEntryHeader.isInvisible());

110     protected boolean processEntry(ByteBuffer entryBuffer)
111         throws DatabaseException {
112 
113         final LogEntryType lastEntryType =
114             LogEntryType.findType(currentEntryHeader.getType());
115         final LogEntry entry = lastEntryType.getNewLogEntry();
116         entry.readEntry(envImpl, currentEntryHeader, entryBuffer);
117 
118         ExtendedFileSummary summary = 
119                 (ExtendedFileSummary) summaries.get(window.currentFileNum());
120         if (summary == null) {
121             summary = new ExtendedFileSummary();
122             summaries.put(window.currentFileNum(), summary);
123         }
124 
125         final int size = getLastEntrySize();
126 
127         summary.totalCount += 1;
128         summary.totalSize += size;
129 
130         if (entry instanceof LNLogEntry) {
131             final LNLogEntry<?> lnEntry = (LNLogEntry<?>) entry;
132             final DatabaseImpl dbImpl = getActiveDb(lnEntry.getDbId());
133             final boolean isActive = (dbImpl != null) &&
134                                      !lnEntry.isImmediatelyObsolete(dbImpl) &&
135                                      isLNActive(lnEntry, dbImpl);
136             applyLN(summary, size, isActive);
137         } else if (entry instanceof BINDeltaLogEntry ||
138                    entry instanceof OldBINDeltaLogEntry) {
139             /* Count Delta as IN. */
140             summary.totalINCount += 1;
141             summary.totalINSize += size;
142             /* Most deltas are obsolete, so count them all obsolete. */
143             summary.obsoleteINCount += 1;
144             summary.recalcObsoleteINSize += size;
145         } else if (entry instanceof INLogEntry) {
146             final INLogEntry<?> inEntry = (INLogEntry<?>) entry;
147             final DatabaseImpl dbImpl = getActiveDb(inEntry.getDbId());
148             final boolean isActive = dbImpl != null &&
149                                      isINActive(inEntry, dbImpl);
150             applyIN(summary, size, isActive);
151         }
152 
153         return true;
154     }

169      * Mimics lookup in com.sleepycat.je.cleaner.FileProcessor.processLN.
170      */
171     private boolean isLNActive(LNLogEntry<?> lnEntry, DatabaseImpl dbImpl) {
172         lnEntry.postFetchInit(dbImpl);
173         final byte[] key = lnEntry.getKey();
174         final Tree tree = dbImpl.getTree();
175         final TreeLocation location = new TreeLocation();
176 
177         final boolean parentFound = tree.getParentBINForChildLN(
178             location, key, false /*splitsAllowed*/,
179             false /*blindDeltaOps*/, CacheMode.DEFAULT);
180 
181         final BIN bin = location.bin;
182 
183         try {
184             if (!parentFound || bin.isEntryKnownDeleted(location.index)) {
185                 return false;
186             }
187             final int index = location.index;
188             final long treeLsn = bin.getLsn(index);
189             if (treeLsn == DbLsn.NULL_LSN) {
190                 return false;
191             }
192             final long logLsn = getLastLsn();
193             return treeLsn == logLsn;
194 
195         } finally {
196             if (bin != null) {
197                 bin.releaseLatch();
198             }
199         }
200     }

202     /**
203      * Mimics lookup in com.sleepycat.je.cleaner.FileProcessor.processIN.
204      */
205     private boolean isINActive(INLogEntry<?> inEntry, DatabaseImpl dbImpl) {
206 
207         final long logLsn = getLastLsn();
208         final IN logIn = inEntry.getIN(dbImpl);
209         logIn.setDatabase(dbImpl);
210         final Tree tree = dbImpl.getTree();
211         if (logIn.isRoot()) {
212             return logLsn == tree.getRootLsn();
213         }
214 
215         logIn.latch(CacheMode.DEFAULT);
216 
217         final SearchResult result = tree.getParentINForChildIN(
218             logIn, true, /*useTargetLevel*/
219             true, /*doFetch*/ CacheMode.DEFAULT);
220 
221         if (!result.exactParentFound) {
222             return false;
223         }
224         try {
225             long treeLsn = result.parent.getLsn(result.index);
226 
227             if (treeLsn == DbLsn.NULL_LSN) {
228                 return false;
229             }
230 
231             if (treeLsn == logLsn) {
232                 return true;
233             }
234 
235             if (!logIn.isBIN()) {
236                 return false;
237             }
238 
239             /* The treeLsn may refer to a BIN-delta. */
240             final IN treeIn =
241                 result.parent.fetchIN(result.index, CacheMode.DEFAULT);
242 
243             treeLsn = treeIn.getLastFullLsn();
244 
245             return treeLsn == logLsn;
246         } finally {
247             result.parent.releaseLatch();
248         }
249     }

295             (envImpl, readBufferSize, startLsn, finishLsn);
