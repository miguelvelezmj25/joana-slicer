26 /**
27  * ElectionQuorum centralizes decision making about what constitutes a
28  * successful election quorum and the definition of an authoritative master.
29  */
30 public class ElectionQuorum {
31 
32     private final RepImpl repImpl;
33     private final Logger logger;
34 
35     /*
36      * If non-zero use this value to override the normal group size
37      * calculations.
38      */
39     private volatile int electableGroupSizeOverride;
40 
41     public ElectionQuorum(RepImpl repImpl) {
42 
43         this.repImpl = repImpl;
44         logger = LoggerUtils.getLogger(getClass());
45 
46         electableGroupSizeOverride = repImpl.getConfigManager().
47             getInt(RepParams.ELECTABLE_GROUP_SIZE_OVERRIDE);
48         if (electableGroupSizeOverride > 0) {
49             LoggerUtils.warning(logger, repImpl,
50                                 "Electable group size override set to:" +
51                                 electableGroupSizeOverride);
52         }
53     }
54 
55     /** For unit testing */
56     public ElectionQuorum() {
57         repImpl = null;
58         logger = null;
59     }

61     /*
62      * Sets the override value for the Electable Group size.
63      */
64     public void setElectableGroupSizeOverride(int override) {
65         if (electableGroupSizeOverride != override) {
66             LoggerUtils.warning(logger, repImpl,
67                                 "Electable group size override changed to:" +
68                                 override);
69         }
70         this.electableGroupSizeOverride = override;
71     }
72 
73     public int getElectableGroupSizeOverride() {
74         return electableGroupSizeOverride;
75     }
76 
77     /**
78      * Predicate to determine whether we have a quorum based upon the quorum
79      * policy.
80      */
81     public boolean haveQuorum(QuorumPolicy quorumPolicy, int votes) {
82         return votes >= getElectionQuorumSize(quorumPolicy);
83     }
84 
85     /**
86      * Returns a definitive answer to whether this node is currently the master
87      * by checking both its status as a master and that a sufficient number
88      * of nodes agree that it's the master based on the number of feeder
89      * connections to it. Currently, the sufficient number is just a simple
90      * majority. Such an authoritative answer is needed in a network partition
91      * situation to detect a master that may be isolated on the minority side
92      * of a network partition.
93      *
94      * @return true if the node is definitely the master. False if it's not or
95      * we cannot be sure.
96      */
97     boolean isAuthoritativeMaster(MasterStatus masterStatus,
98                                   FeederManager feederManager) {
99         if (!masterStatus.isGroupMaster()) {
100             return false;
101         }
102 
103         return (feederManager.activeReplicaCount() + 1) >=
104             getElectionQuorumSize(QuorumPolicy.SIMPLE_MAJORITY);
105     }
106 
107     /**
108      * Return the number of nodes that are required to achieve consensus on the
109      * election. Over time, this may evolve to be a more detailed description
110      * than simply the size of the quorum. Instead, it may return the set of
111      * possible voters.
112      *
113      * Special situations, like an active designated primary or an election
114      * group override will change the default quorum size.
115      *
116      * @param quorumPolicy
117      * @return the number of nodes required for a quorum
118      */
119     private int getElectionQuorumSize(QuorumPolicy quorumPolicy) {
120         if (electableGroupSizeOverride > 0) {
121             return quorumPolicy.quorumSize(electableGroupSizeOverride);
122         }
123 
124         /*
125          * If arbitration is active, check whether arbitration determines the
126          * election group size.
127          */
128         RepNode repNode = repImpl.getRepNode();
129         Arbiter arbiter = repNode.getArbiter();
130         if (arbiter.isApplicable(quorumPolicy)) {
131             return arbiter.getElectionQuorumSize(quorumPolicy);
132         }
133 
134         return quorumPolicy.quorumSize
135             (repNode.getGroup().getElectableGroupSize());
136     }
137 
138     /**
139      * Return whether nodes of the specified type should participate in
140      * elections.
141      *
142      * @param nodeType the node type
143      * @return whether nodes of that type should participate in elections
144      */
145     public boolean nodeTypeParticipates(final NodeType nodeType) {
146 
147         /* Only electable nodes participate in elections */
148         return nodeType.isElectable();
149     }
