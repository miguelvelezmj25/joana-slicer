66 /**
67  * ServiceDispatcher listens on a specific socket for service requests
68  * and dispatches control to the service that is being requested. A service
69  * request message has the format:
70  *
71  * Service:<one byte ServiceName.length><ServiceName>
72  *
73  * The format of the message is binary, with all text being encoded in ascii.
74  *
75  * Upon receipt of service request message, the new SocketChannel is queued for
76  * processing by the service in the Queue associated with the service. The
77  * SocketChannel is the responsibility of the service after this point. It can
78  * configure the channel to best suit the requirements of the specific service.
79  *
80  * The dispatcher returns a single byte to indicate success or failure. The
81  * byte value encodes a ServiceDispatcher.Response enumerator.
82  *
83  */

200 
201         serverChannel = ServerSocketChannel.open();
202         serverChannel.configureBlocking(false);
203         scKey = serverChannel.register(selector, SelectionKey.OP_ACCEPT);
204         ServerSocket acceptSocket = serverChannel.socket();
205         /* No timeout */
206         acceptSocket.setSoTimeout(0);
207 
208         InetSocketAddress bindAddress = socketAddress;
209         if (repImpl != null) {
210             if (repImpl.getConfigManager().getBoolean(SO_REUSEADDR)) {
211 
212                 /* Only turn it on if requested. Otherwise let it default. */
213 
214                  serverChannel.setOption(StandardSocketOptions.SO_REUSEADDR,
215                                          true);
216 
217                 acceptSocket.setReuseAddress(true);
218             }
219 
220             if (repImpl.getConfigManager().getBoolean(BIND_INADDR_ANY)) {
221                 bindAddress = new InetSocketAddress((InetAddress)null,
222                                                     socketAddress.getPort());

227             repImpl.getConfigManager().getInt(SO_BIND_WAIT_MS) : 0;
228

231         final int retryWaitMs = 1000;
232         int totalWaitMs;

273             LoggerUtils.logMsg(logger, repImpl, formatter, Level.INFO,
274                                "ServiceDispatcher HostPort=" +
275                                socketAddress.getHostName() + ":" +

416      * Returns the next socketChannel created in response to a request for the
417      * service. The socketChannel and the associated socket is configured as
418      * requested in the arguments.
419      *
420      * @param serviceName the service for which the channel must be created.
421      * @param blocking true if the channel must be configured to block
422      * @param soTimeout the timeout for the underlying socket
423      * @return the configured channel or null if there are no more channels,
424      * because the service has been shut down.
425      * @throws InterruptedException
426      */
427     public DataChannel takeChannel(String serviceName,
428                                    boolean blocking,
429                                    int soTimeout)
430         throws InterruptedException {
431 
432         while (true) {
433             Service service = serviceMap.get(serviceName);
434             if (service == null) {
435                 throw EnvironmentFailureException.unexpectedState
436                 ("Service: " + serviceName + " was not registered");
437             }
438             if (! (service instanceof QueuingService)) {
439                 throw EnvironmentFailureException.unexpectedState
440                 ("Service: " + serviceName + " is not a queuing service");
441             }
442             Socket socket = null;
443             DataChannel channel = null;
444             try {
445                 channel = ((QueuingService)service).take();
446                 assert channel != null;
447 
448                 if (channel == RepUtils.CHANNEL_EOF_MARKER) {
449                     /* A pseudo channel to indicate EOF, return null */
450                     return null;
451                 }
452 
453                 channel.getSocketChannel().configureBlocking(blocking);
454 
455                 socket = channel.getSocketChannel().socket();
456                 socket.setSoTimeout(soTimeout);
457                 if (blocking) {
458 
459                     /*
460                      * Ensure that writes have been flushed.  All message
461                      * exchanges should be complete here, and we don't expect
462                      * there to be any pending writes, but do this here in
463                      * blocking mode to ensure that the writes complete without
464                      * the need for a loop.
465                      */
466                     channel.flush();
467                 }
468 
469                 return channel;
470             } catch (IOException e) {
471                 LoggerUtils.logMsg(logger, repImpl, formatter, Level.WARNING,
472                                    "Unable to configure channel " +
473                                    "for '" + serviceName + "' service: " +
474                                    LoggerUtils.exceptionTypeAndMsg(e));
475                 try {
476                     channel.close();
477                 } catch (IOException e1) {
478                     LoggerUtils.logMsg(logger, repImpl, formatter, Level.FINEST,
479                                        "Cleanup failed for service: " +
480                                        serviceName + "\n" +
481                                        LoggerUtils.exceptionTypeAndMsg(e1));
482                 }
483                 /* Wait for the next request. */
484                 continue;
485             }
486         }

636     /**
637      * Processes an accept event on the server socket. As a result of the
638      * processing a new socketChannel is created, and the selector is
639      * registered with the new channel so that it can process subsequent read
640      * events.
641      */
642     private void processAccept() {
643 
644         SocketChannel socketChannel = null;
645         try {
646             socketChannel = serverChannel.accept();
647 
648             if (!processAcceptRequests) {
649                 closeChannel(socketChannel);
650                 return;
651             }
652             socketChannel.configureBlocking(false);
653             final DataChannel dataChannel =
654                 getChannelFactory().acceptChannel(socketChannel);
655 
656             /*
657              * If authenticationMethod is provided, use it. Otherwise if no
658              * authenticationMethods are provided but the channel is capable
659              * of determining trust, pass an empty Authentication
660              * array to ServerHandshake in order to trigger the trust check.
661              */
662             final AuthenticationMethod[] authInfo =
663                 (dataChannel.isTrustCapable() && authOptions == null) ?
664                 new AuthenticationMethod[0] :
665                 authOptions;
666 
667             final ServerHandshake initState =
668                 new ServerHandshake(dataChannel, this, authInfo);
669 
670             /* Register the selector with the base SocketChannel */

681      * Processes read events on newly established socket channels. Input on the
682      * channel is verified to ensure that it is a service request. The read is
683      * accomplished in two parts, a read for the fixed size prefix and the name
684      * length byte, followed by a read of the variable length name itself.
685      *
686      * Errors result in the channel being closed(with the key being canceled
687      * as a result) and a null value being returned.
688      *
689      * If the service request is sane, we may require the connecting
690      * entity to authenticate itself.
691      *
692      * @param initState the InitState object associated with the new channel
693      *
694      * @return the ServiceName or null if there was insufficient input, or an
695      * error was encountered.
696      */
697     private String processRead(ServerHandshake initState) {
698         try {
699             final InitResult result = initState.process();
700             if (result == InitResult.FAIL) {
701                 /* Probably already closed, but make sure */
702                 initState.getChannel().close();
703                 return null;
704             }
705             if (result == InitResult.REJECT) {
706                 initState.getChannel().write(Response.INVALID.byteBuffer());
707                 initState.getChannel().close();
708                 return null;
709             }
710 
711             if (result == InitResult.DONE) {
712                 return initState.getServiceName();
713             }
714             /* Initial sequence not complete as yet, keep reading */
715             return null;
716         } catch (IOException e) {
717             LoggerUtils.logMsg(logger, repImpl, formatter, Level.WARNING,
718                                "Exception during read: " +
719                                 LoggerUtils.exceptionTypeAndMsg(e));
720             closeChannel(initState.getChannel());
721             return null;
722         }
723     }

742     /**
743      * The central run method. It dispatches to the "accept" and "read" event
744      * processing methods. Upon a completed read, it verifies the validity of
745      * the service name and queues the channel for subsequent consumption
746      * by the service.
747      *
748      */
749     @Override
750     public void run() {
751         LoggerUtils.logMsg(logger, repImpl, formatter, Level.INFO,
752                            "Started ServiceDispatcher. HostPort=" +
753                            socketAddress.getHostName() + ":" +
754                            socketAddress.getPort());
755         LoggerUtils.logMsg(logger, repImpl, formatter, Level.INFO,
756                            "DataChannel factory: " +
757                            getChannelFactory().getClass().getName());
758         try {
759             while (true) {
760                 try {
761                     /**
762                      * To make the dispatcher resilient to IP address change,
763                      * we periodically check for such change and rebind the
764                      * socket if that occurs.
765                      *
766                      * Speculation and rational:
767                      * New communications fail sliently which is possibly
768                      * caused by that each TCP session use both IP and port
769                      * number as the identifier. Thus it will drop packages
770                      * after IP address change. Yet no exception is raised in
771                      * that situation. Therefore, we cannot rely on exception
772                      * for detecting such a change, but instead use an active
773                      * approach.
774                      */
775                     boolean changed = false;
776                     try {
777                         changed = ipChanged();
778                     } catch (Exception e) {
779                         LoggerUtils.logMsg
780                             (logger, repImpl, formatter, Level.INFO,
781                              "Exception while check IP: " +
782                              LoggerUtils.exceptionTypeAndMsg(e));
783                     }
784                     if (changed) {
785                         rebindSocket();
786                     }
787                     final int result = selector.select(1000);
788                     if (isShutdown()) {
789                         return;
790                     }
791                     if (result == 0) {
792                         continue;
793                     }
794                 } catch (Exception e) {
795                     LoggerUtils.logMsg
796                         (logger, repImpl, formatter, Level.SEVERE,
797                          "Server socket exception: " +
798                         LoggerUtils.getStackTrace(e));
799                     throw EnvironmentFailureException.unexpectedException(e);
800                 }
801                 Set<SelectionKey> skeys = selector.selectedKeys();
802                 for (SelectionKey key : skeys) {
803                     switch (key.readyOps()) {
804 
805                         case SelectionKey.OP_ACCEPT:
806                             processAccept();
807                             break;
808 
809                         case SelectionKey.OP_READ:
810                             final ServerHandshake initState =
811                                 (ServerHandshake) key.attachment();
812 
813                             final String serviceName = processRead(initState);
814                             if (serviceName == null) {
815                                 break;
816                             }
817                             key.cancel();
818                             processService(initState.getChannel(), serviceName);
819                             break;
820 
821                         default:
822                             throw EnvironmentFailureException.unexpectedState
823                                 ("Unexpected ops bit set: " + key.readyOps());
824                     }
825                 }
826                 /* All keys have been processed clear them. */
827                 skeys.clear();
828             }
829         } finally {
830             /*
831              * Clean up any in-process connections that are still in the
832              * handshake phase.
833              */
834             Iterator<SelectionKey> skIter = selector.keys().iterator();
835             while (skIter.hasNext()) {
836                 SelectionKey key = skIter.next();
837                 final ServerHandshake initState =
838                     (ServerHandshake) key.attachment();
839                 if (initState != null) {
840                     LoggerUtils.logMsg(logger, repImpl, formatter, Level.INFO,
841                                        "Server closing in-process handshake");
842                     closeChannel(initState.getChannel());
843                     key.cancel();
844                 }
845             }
846             closeChannel(serverChannel);
847             cleanup();
848         }
849     }
