40 /**
41  * @hidden
42  *
43  * DbNullNode is an internal debugging aid that simply starts up a node as part
44  * of a group. It's primarily useful for diagnosing node start up bugs, since
45  * it enables you to start up the nodes in a group if you have the associated
46  * log files.
47  * <p>
48  * If the environment does not exist, and the -createEnv is specified it will
49  * create the environment (with the aid of helpers), using network restore to
50  * initialize it. This behavior can be used to advantage at a Replica node if
51  * the log files have been damaged for example and the Replica cannot proceed.
52  * It effectively restores the environment for the Replica, which can then
53  * proceed with the restored files using the actual application.
54  * <p>
55  * Similarly, if a node does not exist in the environment and -createNode is
56  * specified, it will create a new node in either the existing environment, or
57  * if -createEnv is specified in a new environment.
58  * <p>
59  * Note that the hostnames mentioned in the member database must be resolvable
60  * before the group can start up. The hostnames can be obtained by running
61  * DbDumpGroup. The hostnames can be "faked" on Linux, by adding appropriate
62  * entries to /etc/hosts or cnames to the DNS.
63  * <p>
64  * If the hosts all use the same port, you will need to make provisions for
65  * virtual ip addresses that are then associated with the host. On Linux this
66  * can be done by adding an entry of the following type for each virtual ip in
67  * /etc/network/interfaces:
68  *<p>
69  * iface eth0:1 inet static
70  * address 192.168.1.201
71  * netmask 255.255.255.0
72  * gateway 192.168.2.1
73  * hwaddress ether XX:YY:ZZ:AA:BB:CC
74  * <p>
75  * Use eth0:2, eth0:3, etc for each new virtual ip that is needed. Also,
76  * substitute the real NIC address for XX:YY:ZZ:AA:BB:CC in the iface stanza.
77  *
78  * Be careful when modifying your machine's configuration information since it
79  * can have unintended side-effects. So back up the above config files before
80  * making any changes.
81  */

312      * Opens a replicated environment and idles in a loop printing out its
313      * state periodically.
314      *
315      * @throws InsufficientLogException so that the log files can be restored
316      * if necessary

318     private void openAndIdle()
319         throws InsufficientLogException {
320 
321         ReplicationConfig repConfig = new ReplicationConfig();
322         repConfig.setNodeName(nodeName);
323         repConfig.setGroupName(groupName);
324         repConfig.setNodeHostPort(hostPort);
325         repConfig.setDesignatedPrimary(designatedPrimary);
326         repConfig.setElectableGroupSizeOverride(electableGroupSize);
327         repConfig.setNodeType(nodeType);
328 
329         if (helpers != null) {

332 
333         EnvironmentConfig envConfig = new EnvironmentConfig();
334         envConfig.setTransactional(true);
335         envConfig.setAllowCreate(createEnv);
336         envConfig.setCacheMode(cacheMode);
337         ReplicatedEnvironment repEnv =
338             new ReplicatedEnvironment(envHome, repConfig, envConfig);
339         RepNode repNode = RepInternal.getNonNullRepImpl(repEnv).getRepNode();
340 
341         System.err.println("Handle created:" + repEnv +
342                            "  Node idling indefinitely...");
343         try {
344             while (true) {
345                 System.out.println(new Date() +
346                                    " State:" + repEnv.getState() + " " +
347                                    " VLSN range:" +
348                                    repNode.getVLSNIndex().getRange() +
349                                    " DTVLSN:" + repNode.getDTVLSN() +
350                                    repNode.dumpState());
351                 writeUpdates(repEnv);
352                 Thread.sleep(pollIntervalMs);
353             }
354         } catch (InterruptedException e) {
355             System.err.println("Exiting");
356         }
357     }

359     /**
360      * Write updates if master.
361      */
362     private void writeUpdates(ReplicatedEnvironment repEnv) {
363         if ((updates == 0) || ! repEnv.getState().isMaster()) {
364             return;
365         }
366 
367         Database db = null;
368         Transaction txn = null;
369         try {
370             txn = repEnv.beginTransaction(null, TransactionConfig.DEFAULT);
371             db = repEnv.openDatabase(txn, "DbNullDb", dbConfig);
372             txn.commit();
373             txn = null;
374             DatabaseEntry key = new DatabaseEntry(new byte[]{1});
375             IntegerBinding.intToEntry(0, key);
376             txn = repEnv.beginTransaction(null, txnConfig);
377             for (int i = 0; i < updates; i++) {
378                 db.put(txn, key, key);
379             }
380             txn.commit();
381             System.err.println(new Date() + " Wrote " + updates + " updates");
382             txn = null;
383         } catch (Exception e) {
384             System.err.println("Update failed:" + e.getMessage());
385         } finally {
386             if (txn != null) {
387                 txn.abort();
388             }
389             if (db != null) {
390                 db.close();
391             }
392         }
393     }
