159     /**
160      * Determines log compatibility. Returns null if they are compatible or the
161      * server would like to defer the rejection to the replica. Return a
162      * JEVersionsReject message if the server does not wish to communicate with
163      * the replica.
164      *
165      * This check requires the log version of the replicas to be greater than
166      * or equal to {@value LogEntryType#LOG_VERSION_REPLICATE_OLDER} . Allowing
167      * the replica version to be behind the feeder version supports replication
168      * during upgrades where the feeder is upgraded before the replica.
169      * [#22336]
170      *
171      * This check also requires that the JE version of the replica is at least
172      * the minJEVersion specified by the RepGroupImpl, if any.  This check
173      * makes sure that nodes running an older software version cannot join the
174      * group after a new and incompatible feature has been used.
175      */
176     private JEVersionsReject checkJECompatibility(final Protocol protocol,
177                                                   final JEVersions jeVersions) {
178 
179         final int replicaLogVersion = jeVersions.getLogVersion();
180         if (replicaLogVersion < LogEntryType.LOG_VERSION_REPLICATE_OLDER - 1) {
181             return protocol.new JEVersionsReject(
182                 "Incompatible log versions. " +
183                 "Feeder log version: " + getCurrentLogVersion() +
184                 ", Feeder JE version: " + getCurrentJEVersion() +
185                 ", Replica log version: " + replicaLogVersion +
186                 ", Replica JE version: " + jeVersions.getVersion());
187         }
188 
189         final JEVersion minJEVersion = repNode.getGroup().getMinJEVersion();
190         if (minJEVersion.compareTo(jeVersions.getVersion()) > 0) {
191             return protocol.new JEVersionsReject(
192                 "Unsupported JE version. " +
193                 "Feeder JE version: " + getCurrentJEVersion() +
194                 ", Feeder min JE version: " + minJEVersion +
195                 ", Replica JE version: " + jeVersions.getVersion());
196         }
197 
198         return null;
199     }

230     /**
231      * Executes the feeder side of the handshake.
232      * @throws ProtocolException
233      * @throws ExitException
234      */
235     public Protocol execute()
236         throws DatabaseException,
237                IOException,
238                ProtocolException,
239                ExitException {
240 
241         LoggerUtils.info(logger, repNode.getRepImpl(),
242                          "Feeder-replica handshake start");
243 
244         /* First negotiate a compatible protocol */
245         Protocol protocol = negotiateProtocol();
246 
247         /* Now exchange JE version information using the negotiated protocol */
248         replicaJEVersions = (ReplicaJEVersions) protocol.read(namedChannel);
249         final String versionMsg =
250             " Replica " + replicaNameIdPair.getName() +
251             " Versions" +
252             " JE: " + replicaJEVersions.getVersion().getVersionString() +
253             " Log: " + replicaJEVersions.getLogVersion() +
254             " Protocol: " + protocol.getVersion();
255         LoggerUtils.fine(logger, repNode.getRepImpl(), versionMsg);
256         JEVersionsReject reject =
257             checkJECompatibility(protocol, replicaJEVersions);
258 
259         if (reject != null) {
260             final String msg = "Version incompatibility: " +
261                 reject.getErrorMessage() +
262                 " with replica " + replicaNameIdPair.getName();
263             LoggerUtils.warning(logger, repNode.getRepImpl(), msg);
264             writeMessage(protocol, reject);
265             throw new ExitException(msg);
266         }
267 
268         /*
269          * Use the minimum common log version. This ensures that the feeder
270          * can generate it and that the replica can understand it.
271          */
272         streamLogVersion =
273             Math.min(getCurrentLogVersion(), replicaJEVersions.getLogVersion());
274 
275         writeMessage(protocol,
276                      protocol.new FeederJEVersions(
277                          getCurrentJEVersion(),
278                          streamLogVersion,
279                          repNode.getMinJEVersion()));
280 
281         /* Ensure that the feeder sends the agreed upon version. */
282         protocol.setStreamLogVersion(streamLogVersion);
283 
284         /* Verify replica membership info */
285         verifyMembershipInfo(protocol);
286 
287         checkClockSkew(protocol);
288         LoggerUtils.info
289             (logger, repNode.getRepImpl(),
290              "Feeder-replica " + replicaNameIdPair.getName() +
291              " handshake completed." +
292              versionMsg +
293              " Stream Log: " + protocol.getStreamLogVersion());
294 
295         return protocol;
296     }

323      * Verifies that the group as configured here at the Feeder matches the
324      * configuration of the replica.
325      *
326      * @param protocol the protocol to use for this verification
327      *
328      * @throws IOException
329      * @throws DatabaseException
330      */
331     private void verifyMembershipInfo(Protocol protocol)
332         throws IOException,
333                DatabaseException,
334                ExitException {
335 
336         NodeGroupInfo nodeGroup =
337             (NodeGroupInfo)(protocol.read(namedChannel));
338         final RepGroupImpl group = repNode.getGroup();
339         RepNodeImpl node = group.getNode(nodeGroup.getNodeName());
340 
341         try {
342 
343             if (nodeGroup.getNodeId() != replicaNameIdPair.getId()) {
344                 throw new ExitException
345                     ("The replica node ID sent during protocol negotiation: " +
346                      replicaNameIdPair +
347                      " differs from the one sent in the MembershipInfo " +
348                      "request: " + nodeGroup.getNodeId());
349             }
350 
351             if (nodeGroup.getNodeType().hasTransientId()) {
352 
353                 /*
354                  * Note the secondary or external node if this is a new node.
355                  * Otherwise, fall through, and the subsequent code will
356                  * notice the incompatible node type.
357                  */
358                 if (node == null) {
359 
360                     /* A new node with transient id */
361                     node = new RepNodeImpl(nodeGroup);
362                     try {
363                         repNode.addTransientIdNode(node);
364                     } catch (IllegalStateException | NodeConflictException e) {
365                         throw new ExitException(e, true);
366                     }
367                 }
368             } else if (node == null || !node.isQuorumAck()) {
369                 /* Not currently a confirmed member. */
370                 if (nodeGroup.getNodeType().isArbiter()) {
371                     Set<RepNodeImpl> arbMembers = group.getArbiterMembers();
372                     if (arbMembers.size() > 0) {
373                         throw new ExitException(
374                             "An Arbiter node already exists in the "+
375                             "replication group.");
376                     }
377                 }
378                 try {
379 
380                     /*
381                      * It is possible that the thread that added the
382                      * node to the group database has not added the
383                      * member to the in memory representation. We retry
384                      * several times to give the other thread a chance to
385                      * update the in memory structure.
386                      */
387                     repNode.getRepGroupDB().ensureMember(nodeGroup);
388                     for (int i=0; i < GROUP_RETRY; i++) {
389                         node =
390                             repNode.getGroup().getMember(
391                                 nodeGroup.getNodeName());
392                         if (node != null) {
393                             break;
394                         }
395                         try {
396                             Thread.sleep(GROUP_RETRY_SLEEP_MS);
397                         } catch (Exception e) {
398 
399                         }
400                     }
401                     if (node == null) {
402                         throw EnvironmentFailureException.unexpectedState
403                             ("Node: " + nodeGroup.getNameIdPair() +
404                              " not found");
405                     }
406                 } catch (InsufficientReplicasException |
407                          InsufficientAcksException |
408                          LockTimeoutException e) {
409                     throw new ExitException(e, false);
410                 } catch (NodeConflictException e) {
411                     throw new ExitException(e, true);
412                 }
413             } else if (node.isRemoved()) {
414                 throw new ExitException
415                     ("Node: " + nodeGroup.getNameIdPair() +
416                      " is no longer a member of the group." +
417                      " It was explicitly removed.");
418             }
419 
420             doGroupChecks(nodeGroup, group);
421             doNodeChecks(nodeGroup, node);
422             maybeUpdateJEVersion(nodeGroup, group, node);
423         } catch (ExitException exception) {
424             LoggerUtils.info
425                 (logger, repNode.getRepImpl(), exception.getMessage());
426             if (exception.failReplica()) {
427                 /*
428                  * Explicit message to force replica to invalidate the
429                  * environment.
430                  */
431                 writeMessage(protocol,
432                              protocol.new NodeGroupInfoReject(
433                              exception.getMessage()));
434             }
435             throw exception;
436         }
437 
438         /* Id is now established for sure, update the pair. */
439         replicaNameIdPair.update(node.getNameIdPair());
440         namedChannel.setNameIdPair(replicaNameIdPair);
441         LoggerUtils.fine(logger, repNode.getRepImpl(),
442                          "Channel Mapping: " + replicaNameIdPair + " is at " +
443                          namedChannel.getChannel());
444         writeMessage(protocol,
445                      protocol.new NodeGroupInfoOK(
446                          group.getUUID(), replicaNameIdPair));
447     }

533     /**
534      * Update the node's JE version from the provided group info if storing the
535      * JE version is supported and the current version differs from the stored
536      * one.  It's OK if the attempt does not fully succeed due to a lack of
537      * replicas or acknowledgments: we can try again the next time.
538      *
539      * @throws ExitException if the attempt fails because the updated node's
540      * socket address conflicts with another node
541      */
542     private void maybeUpdateJEVersion(final NodeGroupInfo nodeGroup,
543                                       final RepGroupImpl group,
544                                       final RepNodeImpl node)
545         throws ExitException {
546 
547         if ((group.getFormatVersion() >= RepGroupImpl.FORMAT_VERSION_3) &&
548             (nodeGroup.getJEVersion() != null) &&
549             !nodeGroup.getJEVersion().equals(node.getJEVersion())) {
550 
551             /*
552              * Try updating the JE version information, given that the group
553              * format supports this.  Don't require a quorum of acknowledgments
554              * since the fact that the handshake for this replica is underway
555              * may mean that a quorum is not available.  Saving the JE version
556              * is only an optimization, so it is OK if this attempt fails to be
557              * persistent.
558              */
559             try {
560                 repNode.getRepGroupDB().updateMember(
561                     new RepNodeImpl(nodeGroup), false);
562             } catch (InsufficientReplicasException |
563                      InsufficientAcksException |
564                      LockTimeoutException e) {
565                 /* Ignored */
566             } catch (NodeConflictException e) {
567                 throw new ExitException(e, true);
568             }
569         }
570     }

572     /**
573      * Negotiates and returns the protocol that will be used in all subsequent
574      * interactions with the Replica, if the replica accepts to it.
575      *
576      * @return the protocol instance to be used for subsequent interactions
577      *
578      * @throws IOException
579      * @throws ExitException
580      */
581     private Protocol negotiateProtocol()
582         throws IOException, ExitException {
583 
584         Protocol defaultProtocol =
585             Protocol.getProtocol(repNode, getCurrentProtocolVersion());
586 
587         /*
588          * Wait to receive the replica's version, decide which protocol version
589          * to use ourselves, and then reply with our version.
590          */
591         ReplicaProtocolVersion message =
592             (ReplicaProtocolVersion) defaultProtocol.read(namedChannel);
593 
594         replicaNameIdPair = message.getNameIdPair();
595 
596         Feeder dup =
597             repNode.feederManager().getFeeder(replicaNameIdPair.getName());
598         if ((dup != null) ||
599             (message.getNameIdPair().getName().
600                     equals(feederNameIdPair.getName()))) {
601             /* Reject the connection. */
602             writeMessage(defaultProtocol,
603                          defaultProtocol.new DuplicateNodeReject(
604                              "This node: " + replicaNameIdPair +
605                              " is already in active use at the feeder "));
606             SocketAddress dupAddress =
607                 namedChannel.getChannel().getSocketChannel().socket().
608                 getRemoteSocketAddress();
609 
610             throw new ExitException
611                 ("A replica with the id: " + replicaNameIdPair +
612                  " is already active with this feeder. " +
613                  " The duplicate replica resides at: " +
614                  dupAddress);
615         }
616 
617         /*
618          * If the Replica's version is acceptable, use it, otherwise return the
619          * default protocol at this node, in case the Replica can support it.
620          */
621         final int replicaVersion = message.getVersion();
622         Protocol protocol =
623             Protocol.get(repNode, replicaVersion, getCurrentProtocolVersion());
624         if (protocol == null) {
625             protocol = defaultProtocol;
626         }
627         defaultProtocol.write
628              (defaultProtocol.new FeederProtocolVersion(protocol.getVersion()),
629               namedChannel);
630         return protocol;
631     }
