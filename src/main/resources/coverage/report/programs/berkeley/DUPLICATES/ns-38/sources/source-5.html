14 package com.sleepycat.utilint;

16 import java.util.HashMap;
17 import java.util.Map;
18 import java.util.logging.Logger;
19 
20 /**
21  * Maintain interval and cumulative stats for a given set of operations, as
22  * well as a activityCounter that generates thread dumps if operations take too
23  * long. The markStart and markFinish methods can be used to bracket each
24  * tracked operation.
25  */
26 public class StatsTracker<T> {
27 
28     /* Latency stats. */
29     private final Map<T, LatencyStat> intervalLatencies;
30     private final Map<T, LatencyStat> cumulativeLatencies;
31 
32     /*
33      * ActivityCounter tracks throughput and dumps thread stacktraces when
34      * throughput drops.
35      */
36     private final ActivityCounter activityCounter;
37 
38     /**
39      * The logger is used for activity stack traces.
40      */
41     public StatsTracker(T[] opTypes,
42                         Logger stackTraceLogger,
43                         int activeThreadThreshold,
44                         long threadDumpIntervalMillis,
45                         int threadDumpMax,
46                         int maxTrackedLatencyMillis) {
47 
48         this.intervalLatencies = new HashMap<T, LatencyStat>();
49         this.cumulativeLatencies = new HashMap<T, LatencyStat>();
50 
51         for (T opType : opTypes) {
52             intervalLatencies.put
53                 (opType, new LatencyStat(maxTrackedLatencyMillis));
54             cumulativeLatencies.put
55                 (opType, new LatencyStat(maxTrackedLatencyMillis));
56         }
57 
58         activityCounter = new ActivityCounter(activeThreadThreshold,
59                                               threadDumpIntervalMillis,
60                                               threadDumpMax, 
61                                               stackTraceLogger);
62     }
63 
64     /** 
65      * Track the start of a operation.
66      * @return the value of System.nanoTime, for passing to markFinish.
67      */
68     public long markStart() {
69         activityCounter.start();
70         return System.nanoTime();
71     }
72 
73     /**
74      * Track the end of an operation.
75      * @param startTime should be the value returned by the corresponding call
76      * to markStart
77      */
78     public void markFinish(T opType, long startTime) {
79         markFinish(opType, startTime, 1);
80     }
81     /**
82      * Track the end of an operation.
83      * @param startTime should be the value returned by the corresponding call
84      * to markStart
85      */
86     public void markFinish(T opType, long startTime, int numOperations) {
87         try {
88             if (numOperations == 0) {
89                 return;
90             }
91 
92             if (opType != null) {
93                 long elapsed = System.nanoTime() - startTime;
94                 intervalLatencies.get(opType).set(numOperations, elapsed);
95                 cumulativeLatencies.get(opType).set(numOperations, elapsed);
96             }
97         } finally {
98             /* Must be invoked to clear the ActivityCounter stats. */
99             activityCounter.finish();
100         }
101     }
102 
103     /**
104      * Should be called after each interval latency stat collection, to reset
105      * for the next period's collection.
106      */
107     public void clearLatency() {
108         for (Map.Entry<T, LatencyStat> e : intervalLatencies.entrySet()) {
109             e.getValue().clear();
110         }
111     }
112 
113     public Map<T, LatencyStat> getIntervalLatency() {
114         return intervalLatencies;
115     }
116 
117     public Map<T, LatencyStat> getCumulativeLatency() {
118         return cumulativeLatencies;
119     }
120 
121     /**
122      * For unit test support.
123      */
124     public int getNumCompletedDumps() {
125         return activityCounter.getNumCompletedDumps();
126     }
127 }
