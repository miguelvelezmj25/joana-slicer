23 /**
24  * Implements Iterator for an arbitrary EntityCursor.
25  *
26  * @author Mark Hayes
27  */
28 class BasicIterator<V> implements Iterator<V> {
29 
30     private EntityCursor<V> entityCursor;
31     private ForwardCursor<V> forwardCursor;
32     private LockMode lockMode;
33     private V nextValue;
34 
35     /**
36      * An EntityCursor is given and the remove() method is supported.
37      */
38     BasicIterator(EntityCursor<V> entityCursor, LockMode lockMode) {
39         this.entityCursor = entityCursor;
40         this.forwardCursor = entityCursor;
41         this.lockMode = lockMode;
42     }
43 
44     /**
45      * A ForwardCursor is given and the remove() method is not supported.
46      */
47     BasicIterator(ForwardCursor<V> forwardCursor, LockMode lockMode) {
48         this.forwardCursor = forwardCursor;
49         this.lockMode = lockMode;
50     }
51 
52     public boolean hasNext() {
53         if (nextValue == null) {
54             try {
55                 nextValue = forwardCursor.next(lockMode);
56             } catch (DatabaseException e) {
57                 throw RuntimeExceptionWrapper.wrapIfNeeded(e);
58             }
59             return nextValue != null;
60         } else {
61             return true;
62         }
63     }
64 
65     public V next() {
66         if (hasNext()) {
67             V v = nextValue;
68             nextValue = null;
69             return v;
70         } else {
71             throw new NoSuchElementException();
72         }
73     }
74 
75     public void remove() {
76         if (entityCursor == null) {
77             throw new UnsupportedOperationException();
78         }
79         try {
80             if (!entityCursor.delete()) {
81                 throw new IllegalStateException
82                     ("Record at cursor position is already deleted");
83             }
84         } catch (DatabaseException e) {
85             throw RuntimeExceptionWrapper.wrapIfNeeded(e);
86         }
87     }
88 }
