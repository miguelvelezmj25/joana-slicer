165     /*
166      * Fields for the control flow and data flow graph analysis algorithms (used
167      * to compute the maximum stack size or the stack map frames). A control
168      * flow graph contains one node per "basic block", and one edge per "jump"
169      * from one basic block to another. Each node (i.e., each basic block) is
170      * represented by the Label object that corresponds to the first instruction
171      * of this basic block. Each node also stores the list of its successors in
172      * the graph, as a linked list of Edge objects.
173      * 
174      * The control flow analysis algorithms used to compute the maximum stack
175      * size or the stack map frames are similar and use two steps. The first
176      * step, during the visit of each instruction, builds information about the
177      * state of the local variables and the operand stack at the end of each
178      * basic block, called the "output frame", <i>relatively</i> to the frame
179      * state at the beginning of the basic block, which is called the "input
180      * frame", and which is <i>unknown</i> during this step. The second step, in
181      * {@link MethodWriter#visitMaxs}, is a fix point algorithm that computes
182      * information about the input frame of each basic block, from the input
183      * state of the first basic block (known from the method signature), and by
184      * the using the previously computed relative output frames.
185      * 
186      * The algorithm used to compute the maximum stack size only computes the
187      * relative output and absolute input stack heights, while the algorithm
188      * used to compute stack map frames computes relative output frames and
189      * absolute input frames.
190      */
191 
192     /**
193      * Start of the output stack relatively to the input stack. The exact
194      * semantics of this field depends on the algorithm that is used.
195      * 
196      * When only the maximum stack size is computed, this field is the number of
197      * elements in the input stack.
198      * 
199      * When the stack map frames are completely computed, this field is the
200      * offset of the first output stack element relatively to the top of the
201      * input stack. This offset is always negative or null. A null offset means
202      * that the output stack must be appended to the input stack. A -n offset
203      * means that the first n output stack elements must replace the top n input

319      * Adds a forward reference to this label. This method must be called only
320      * for a true forward reference, i.e. only if this label is not resolved
321      * yet. For backward references, the offset of the reference can be, and
322      * must be, computed and stored directly.
323      * 
324      * @param sourcePosition
325      *            the position of the referencing instruction. This position
326      *            will be used to compute the offset of this forward reference.

329      *            be stored.

331     private void addReference(final int sourcePosition,
332             final int referencePosition) {
333         if (srcAndRefPositions == null) {
334             srcAndRefPositions = new int[6];
335         }
336         if (referenceCount >= srcAndRefPositions.length) {
337             int[] a = new int[srcAndRefPositions.length + 6];
338             System.arraycopy(srcAndRefPositions, 0, a, 0,
339                     srcAndRefPositions.length);
340             srcAndRefPositions = a;
341         }
342         srcAndRefPositions[referenceCount++] = sourcePosition;
343         srcAndRefPositions[referenceCount++] = referencePosition;
344     }

347      * Resolves all forward references to this label. This method must be called
348      * when this label is added to the bytecode of the method, i.e. when its
349      * position becomes known. This method fills in the blanks that where left
350      * in the bytecode by each forward reference previously added to this label.
351      * 
352      * @param owner
353      *            the code writer that calls this method.
354      * @param position
355      *            the position of this label in the bytecode.
356      * @param data
357      *            the bytecode of the method.
358      * @return <tt>true</tt> if a blank that was left for this label was to
359      *         small to store the offset. In such a case the corresponding jump
360      *         instruction is replaced with a pseudo instruction (using unused
361      *         opcodes) using an unsigned two bytes offset. These pseudo
362      *         instructions will need to be replaced with true instructions with
363      *         wider offsets (4 bytes instead of 2). This is done in
364      *         {@link MethodWriter#resizeInstructions}.
365      * @throws IllegalArgumentException
366      *             if this label has already been resolved, or if it has not
367      *             been created by the given code writer.
368      */
369     boolean resolve(final MethodWriter owner, final int position,
370             final byte[] data) {
371         boolean needUpdate = false;
372         this.status |= RESOLVED;
373         this.position = position;
374         int i = 0;
375         while (i < referenceCount) {
376             int source = srcAndRefPositions[i++];
377             int reference = srcAndRefPositions[i++];
378             int offset;
379             if (source >= 0) {
380                 offset = position - source;
381                 if (offset < Short.MIN_VALUE || offset > Short.MAX_VALUE) {
382                     /*
383                      * changes the opcode of the jump instruction, in order to
384                      * be able to find it later (see resizeInstructions in
385                      * MethodWriter). These temporary opcodes are similar to
386                      * jump instruction opcodes, except that the 2 bytes offset
387                      * is unsigned (and can therefore represent values from 0 to
388                      * 65535, which is sufficient since the size of a method is
389                      * limited to 65535 bytes).
390                      */
391                     int opcode = data[reference - 1] & 0xFF;
392                     if (opcode <= Opcodes.JSR) {
393                         // changes IFEQ ... JSR to opcodes 202 to 217
394                         data[reference - 1] = (byte) (opcode + 49);
395                     } else {
396                         // changes IFNULL and IFNONNULL to opcodes 218 and 219
397                         data[reference - 1] = (byte) (opcode + 20);
398                     }
399                     needUpdate = true;
400                 }
401                 data[reference++] = (byte) (offset >>> 8);
402                 data[reference] = (byte) offset;
403             } else {
404                 offset = position + source + 1;
405                 data[reference++] = (byte) (offset >>> 24);
406                 data[reference++] = (byte) (offset >>> 16);
407                 data[reference++] = (byte) (offset >>> 8);
408                 data[reference] = (byte) offset;
409             }
410         }
411         return needUpdate;
412     }

481     /**
482      * Finds the basic blocks that belong to a given subroutine, and marks these
483      * blocks as belonging to this subroutine. This method follows the control
484      * flow graph to find all the blocks that are reachable from the current
485      * block WITHOUT following any JSR target.
486      * 
487      * @param JSR
488      *            a JSR block that jumps to this subroutine. If this JSR is not
489      *            null it is added to the successor of the RET blocks found in
490      *            the subroutine.
491      * @param id
492      *            the id of this subroutine.
493      * @param nbSubroutines
494      *            the total number of subroutines in the method.
495      */
496     void visitSubroutine(final Label JSR, final long id, final int nbSubroutines) {
497         // user managed stack of labels, to avoid using a recursive method
498         // (recursivity can lead to stack overflow with very large methods)
499         Label stack = this;
500         while (stack != null) {
501             // removes a label l from the stack
502             Label l = stack;
503             stack = l.next;
504             l.next = null;
505 
506             if (JSR != null) {
507                 if ((l.status & VISITED2) != 0) {
508                     continue;
509                 }
510                 l.status |= VISITED2;
511                 // adds JSR to the successors of l, if it is a RET block
512                 if ((l.status & RET) != 0) {
513                     if (!l.inSameSubroutine(JSR)) {
514                         Edge e = new Edge();
515                         e.info = l.inputStackTop;
516                         e.successor = JSR.successors.successor;
517                         e.next = l.successors;
518                         l.successors = e;
519                     }
520                 }
521             } else {
522                 // if the l block already belongs to subroutine 'id', continue
523                 if (l.inSubroutine(id)) {
524                     continue;
525                 }
526                 // marks the l block as belonging to subroutine 'id'
527                 l.addToSubroutine(id, nbSubroutines);
528             }
529             // pushes each successor of l on the stack, except JSR targets
530             Edge e = l.successors;
531             while (e != null) {
532                 // if the l block is a JSR block, then 'l.successors.next' leads
533                 // to the JSR target (see {@link #visitJumpInsn}) and must
534                 // therefore not be followed
535                 if ((l.status & Label.JSR) == 0 || e != l.successors.next) {
536                     // pushes e.successor on the stack if it not already added
537                     if (e.successor.next == null) {
538                         e.successor.next = stack;
539                         stack = e.successor;
540                     }
541                 }
542                 e = e.next;
543             }
544         }
545     }
