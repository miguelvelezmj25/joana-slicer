30 /**
31  * The methods in this class are disabled if the GlobalCBVLSN is defunct -- see
32  * {@link GlobalCBVLSN}.
33  *
34  * <p>When the GlobalCBVLSN is not defunct, this class supports updating the
35  * group database with each node's local CBVLSN when it is in the Master
36  * state. There is one instance per feeder connection, plus one for the
37  * Master. There is, logically, a LocalCBVLSNTracker instance associated
38  * with each instance of the updater. The instance is local for an update
39  * associated with a node in the Master state and is remote for each
40  * Replica.</p>
41  *
42  * <p>The nodeCBVLSN can only increase during the lifetime of the
43  * LocalCBVLSNUpdater instance. Note however that the value of the node's
44  * CBVLSN as stored in the database, which represents the values from multiple
45  * updaters associated with a node over its lifetime, may both decrease and
46  * increase over its lifetime. The decreases are due primarily to rollbacks,
47  * and should be relatively rare.</p>
48  *
49  * <p>The updaters used to maintain the Replica's local CBVLSNs are stored in
50  * the Feeder.InputThread. The lifetime of such a LocalCBVLSNUpdater is
51  * therefore determined by the lifetime of the connection between the Master
52  * and the Replica. The node CBVLSN is updated each time a heart beat response
53  * is processed by the FeederInput thread. It's also updated when an old Master
54  * detects that a Replica needs a network restore. In this case, it updates
55  * cbvlsn to the value expected from the node after a network restore so that
56  * the global CBVLSN can continue to make forward progress and not hold up the
57  * cleaner.</p>
58  *
59  * <p>The Master maintains an updater for its own CBVLSN in the FeederManager.
60  * This updater exists as long as the node retains its Master state.</p>
61  *
62  * <p>Local CBVLSNs are used only to contribute to the calculation of the
63  * global CBVLSN. The global CBVLSN acts as the cleaner throttle on old nodes.
64  * Any invariants, such as the rule that the cleaner throttle cannot regress,
65  * are applied when doing the global calculation.</p>
66  *
67  * <p>Note that CBVLSNs are not stored in the database for secondary nodes, but
68  * transient information about them is still maintained.</p>
69  */

113     public LocalCBVLSNUpdater(final NameIdPair nameIdPair,
114                               final NodeType trackedNodeType,
115                               final RepNode repNode) {
116         defunct = repNode.isGlobalCBVLSNDefunct();
117         if (!defunct) {
118             this.nameIdPair = nameIdPair;
119             this.trackedNodeType = trackedNodeType;
120             this.repNode = repNode;
121 
122             logger = LoggerUtils.getLogger(getClass());
123         } else {
124             this.nameIdPair = null;
125             this.trackedNodeType = null;
126             this.repNode = null;
127             logger = null;
128         }
129         nodeCBVLSN = NULL_VLSN;
130         updatePending = false;
131     }

133     /**
134      * Sets the current CBVLSN for this node, and trips the updatePending
135      * flag so that we know there is something to store to the RepGroupDB.
136      *
137      * @param syncableVLSN the new local CBVLSN
138      */
139     private void set(VLSN syncableVLSN, String source) {
140         assert !defunct;
141 
142         assert repNode.isMaster() :
143                "LocalCBVLSNUpdater.set() can only be called by the master";
144 
145         if (!nodeCBVLSN.equals(syncableVLSN)) {
146             LoggerUtils.fine(logger, repNode.getRepImpl(),
147                              "update local CBVLSN for " + nameIdPair +
148                              " from nodeCBVLSN " + nodeCBVLSN + " to " +
149                              syncableVLSN + " from " + source);
150             if (nodeCBVLSN.compareTo(syncableVLSN) >= 0) {
151 
152                 /*
153                  * LCBVLSN must not decrease, since it can result in a GCBVLSN
154                  * value that's outside a truncated VLSNIndex range. See SR
155                  * [#17343]
156                  */
157                 throw EnvironmentFailureException.unexpectedState
158                     (repNode.getRepImpl(),
159                      "nodeCBVLSN" + nodeCBVLSN + " >= " + syncableVLSN +
160                      " attempted update local CBVLSN for " + nameIdPair +
161                      " from " + source);
162             }
163             nodeCBVLSN = syncableVLSN;
164             updatePending = true;
165         }
166     }

196     }

198     private void doUpdate(VLSN vlsn, String source) {
199         assert !defunct;
200         set(vlsn, source);
201         repNode.getRepImpl().updateCBVLSN(this);
202     }
203 
204     /**
205      * If the GlobalCBVLSN is not defunct, update the database, with the local
206      * CBVLSN associated with the node ID if required. Note that updates can
207      * only be invoked on the master. If the GlobalCBVLSN is defunct, do
208      * nothing.
209      */
210     public void update() {
211 
212         if (defunct) {
213             return;
214         }
215 
216         if (!updatePending) {
217             return;
218         }
219 
220         if (suppressGroupDBUpdates) {
221             /* Short circuit the database update. For testing only. */
222             updatePending = false;
223             return;
224         }
225 
226         if (repNode.isShutdownOrInvalid()) {
227 
228             /*
229              * Don't advance VLSNs after a shutdown request, to minimize the
230              * need for a hard recovery.
231              */
232             return;
233         }
234 
235         try {
236             VLSN candidate = nodeCBVLSN;
237 
238             if (candidate.isNull()) {
239                 return;
240             }
241 
242             if (candidate.compareTo(repNode.getGlobalCBVLSN()) < 0) {
243                 /* Don't let the group CBVLSN regress.*/
244                 return;
245             }
246 
247             final boolean updated = repNode.getRepGroupDB().updateLocalCBVLSN(
248                 nameIdPair, candidate, trackedNodeType);
249             /* If not updated, we'll try again later. */
250             if (updated) {
251                 updatePending = false;
252             }
253         } catch (EnvironmentFailureException e) {
254 
255             /*
256              * Propagate environment failure exception so that the master
257              * can shut down.
258              */
259             throw e;
260         } catch (LockNotAvailableException lnae) {
261             /*
262              * Expected exception, due to use of nowait transaction
263              */
264             LoggerUtils.info(repNode.getLogger(), repNode.getRepImpl(),
265                              " lock not available without waiting. " +
266                              "local cbvlsn update skipped for node: " +
267                               nameIdPair + " Error: " + lnae.getMessage());
268         } catch (DatabaseException e) {
269             LoggerUtils.warning(repNode.getLogger(), repNode.getRepImpl(),
270                                 "local cbvlsn update failed for node: " +
271                                 nameIdPair + " Error: " + e.getMessage() +
272                                 "\n" + LoggerUtils.getStackTrace(e));
273         }
274     }
