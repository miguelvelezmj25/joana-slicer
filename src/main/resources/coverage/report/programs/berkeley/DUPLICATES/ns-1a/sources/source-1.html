24 /**
25  * Defines the protocol used in support of node state querying.
26  *
27  * Because this protocol has to transfer byte array between two nodes, so
28  * instead of using the former NodeStateProtocol, we introduce this new
29  * protocol which inherits from BinaryProtocol.
30  *
31  * Note: once we support active version update, we can use one protocol only.
32  *
33  * The message request sequence:
34  *    NODE_STATE_REQ -> NODE_STATE_RESP
35  */
36 public class BinaryNodeStateProtocol extends BinaryProtocol {

43     public final static MessageOp BIN_NODE_STATE_RESP =
44         new MessageOp((short) 2, BinaryNodeStateResponse.class);
