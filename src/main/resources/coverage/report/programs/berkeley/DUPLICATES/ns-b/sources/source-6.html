38 /**
39  * DbConfigManager holds the configuration parameters for an environment.
40  *
41  * In general, all configuration parameters are represented by a ConfigParam
42  * defined in com.sleepycat.je.config.EnvironmentParams and can be represented
43  * by a property described by the EnvironmentConfig String constants.
44  * Environment parameters have some interesting twists because there are some
45  * attributes that are scoped by handle, such as the commit durability
46  * (txnSync, txnNoSync, etc) parameters.
47  *
48  * DbConfigManager is instantiated first by the EnvironmentImpl, and is loaded
49  * with the base configuration parameters. If replication is enabled,
50  * additional properties are added when the ReplicatedEnvironment is
51  * instantiated. In order to keep replication code out of the base code,
52  * replication parameters are loaded by way of the addConfigurations method.
53  */

262      * Validate a collection of configurations, checking that
263      *   - the name and value are valid
264      *   - a replication param is not being set through an EnvironmentConfig
265      * class, and a non-rep param is not set through a ReplicationConfig
266      * instance.
267      *
268      * This may happen at Environment start time, or when configurations have
269      * been mutated. The configurations have been collected from a file, or
270      * from a Properties object, and haven't gone through the usual validation
271      * path that occurs when XXXConfig.setConfigParam is called.
272      *
273      * SuppressWarnings is used here because Enumeration doesn't work well with
274      * Properties in Java 1.5
275      *
276      * @throws IllegalArgumentException via XxxConfig(Properties) ctor.
277      */
278     @SuppressWarnings("unchecked")
279     public static void validateProperties(Properties props,
280                                           boolean isRepConfigInstance,
281                                           String configClassName)
282         throws IllegalArgumentException {
283 
284         /* Check that the properties have valid names and values. */
285         Enumeration propNames = props.propertyNames();
286         while (propNames.hasMoreElements()) {
287             String name = (String) propNames.nextElement();
288             /* Is this a valid property name? */
289             ConfigParam param =
290                 EnvironmentParams.SUPPORTED_PARAMS.get(name);
291 
292             if (param == null) {
293                 /* See if the parameter is an multi-value parameter. */
294                 String mvParamName = ConfigParam.multiValueParamName(name);
295                 param = EnvironmentParams.SUPPORTED_PARAMS.get(mvParamName);
296 
297                 if (param == null) {
298 
299                     /*
300                      * Remove the property only if:
301                      * 1. The parameter name indicates it's a replication
302                      *    parameter
303                      * 2. The Environment is being opened in standalone mode
304                      * 3. The parameter is being initialized in the properties
305                      *    file
306                      * See SR [#19080].
307                      */
308                     if (configClassName == null && !isRepConfigInstance &&
309                         name.contains(EnvironmentParams.REP_PARAM_PREFIX)) {
310                         props.remove(name);
311                         continue;
312                     }
313 
314                     throw new IllegalArgumentException
315                         (name +
316                          " is not a valid BDBJE environment configuration");
317                 }
318             }
319 
320             /*
321              * Only verify that the parameter is "for replication" if this is
322              * being validated on behalf of a FooConfig class, not a
323              * je.properties file.
324              */
325             if (configClassName != null) {
326                 /* We're validating a config instance, not a file. */
327                 if (isRepConfigInstance) {
328                     if (!param.isForReplication()) {
329                         throw new IllegalArgumentException
330                             (name +
331                              " is not a replication parameter and cannot " +
332                              " be set through " + configClassName);
333                     }
334                 } else {
335                     if (param.isForReplication()) {
336                         throw new IllegalArgumentException
337                             (name +
338                              " is a replication parameter and cannot be set " +
339                              " through " + configClassName);
340                     }
341                 }
342             }
343 
344             /* Is this a valid property value? */
345             param.validateValue(props.getProperty(name));
346         }
347     }

350      * Apply the configurations specified in the je.properties file to override
351      * the programmatically set configuration values held in the property bag.

355     @SuppressWarnings("unchecked")
356     public static void applyFileConfig(File envHome,
357                                        Properties props,
358                                        boolean forReplication)
359         throws IllegalArgumentException {
360 
361         File paramFile = null;
362         try {
363             Properties fileProps = new Properties();
364             if (envHome != null) {
365                 if (envHome.isFile()) {
366                     paramFile = envHome;
367                 } else {
368                     paramFile = new File(envHome, PROPFILE_NAME);
369                 }
370                 FileInputStream fis = new FileInputStream(paramFile);
371                 fileProps.load(fis);
372                 fis.close();
373             }
374 
375             /*
376              * Validate the existing file. No config instance name is used
377              * because we're validating a je.properties file.
378              */
379             validateProperties(fileProps,
380                                false,
381                                null); /* config instance name, don't use. */
382 
383             /* Add them to the configuration object. */
384             Iterator iter = fileProps.entrySet().iterator();
385             while (iter.hasNext()) {
386                 Map.Entry propPair = (Map.Entry) iter.next();
387                 String name = (String) propPair.getKey();
388                 String value = (String) propPair.getValue();
389                 setConfigParam(props,
390                                name,
391                                value,
392                                false, /* don't need mutability, we're
393                                          initializing */
394                                false, /* value already validated when set in
395                                          config object */
396                                forReplication,
397                                false); /* verifyForReplication */
398             }
399         } catch (FileNotFoundException e) {
400 
401             /*
402              * Klockwork - ok
403              * Eat the exception, okay if the file doesn't exist.
404              */
405         } catch (IOException e) {
406             IllegalArgumentException e2 = new IllegalArgumentException
407                 ("An error occurred when reading " + paramFile);
408             e2.initCause(e);
409             throw e2;
410         }
411     }
