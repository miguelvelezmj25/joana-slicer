324     public static Type[] getArgumentTypes(final String methodDescriptor) {
325         char[] buf = methodDescriptor.toCharArray();
326         int off = 1;
327         int size = 0;
328         while (true) {
329             char car = buf[off++];
330             if (car == ')') {
331                 break;
332             } else if (car == 'L') {
333                 while (buf[off++] != ';') {
334                 }
335                 ++size;
336             } else if (car != '[') {
337                 ++size;
338             }
339         }
340         Type[] args = new Type[size];
341         off = 1;
342         size = 0;
343         while (buf[off] != ')') {
344             args[size] = getType(buf, off);
345             off += args[size].len + (args[size].sort == OBJECT ? 2 : 0);
346             size += 1;
347         }
348         return args;

397      * Computes the size of the arguments and of the return value of a method.
398      * 
399      * @param desc
400      *            the descriptor of a method.
401      * @return the size of the arguments of the method (plus one for the
402      *         implicit this argument), argSize, and the size of its return
403      *         value, retSize, packed into a single int i =
404      *         <tt>(argSize &lt;&lt; 2) | retSize</tt> (argSize is therefore equal to
405      *         <tt>i &gt;&gt; 2</tt>, and retSize to <tt>i &amp; 0x03</tt>).
406      */
407     public static int getArgumentsAndReturnSizes(final String desc) {
408         int n = 1;
409         int c = 1;
410         while (true) {
411             char car = desc.charAt(c++);
412             if (car == ')') {
413                 car = desc.charAt(c);
414                 return n << 2
415                         | (car == 'V' ? 0 : (car == 'D' || car == 'J' ? 2 : 1));
416             } else if (car == 'L') {
417                 while (desc.charAt(c++) != ';') {
418                 }
419                 n += 1;
420             } else if (car == '[') {
421                 while ((car = desc.charAt(c)) == '[') {
422                     ++c;
423                 }
424                 if (car == 'D' || car == 'J') {
425                     n -= 1;
426                 }
427             } else if (car == 'D' || car == 'J') {
428                 n += 2;
429             } else {
430                 n += 1;
431             }
432         }
433     }

436      * Returns the Java type corresponding to the given type descriptor. For
437      * method descriptors, buf is supposed to contain nothing more than the
438      * descriptor itself.
439      * 
440      * @param buf
441      *            a buffer containing a type descriptor.
442      * @param off
443      *            the offset of this descriptor in the previous buffer.
444      * @return the Java type corresponding to the given type descriptor.
445      */
446     private static Type getType(final char[] buf, final int off) {
447         int len;
448         switch (buf[off]) {
449         case 'V':
450             return VOID_TYPE;
451         case 'Z':
452             return BOOLEAN_TYPE;
453         case 'C':
454             return CHAR_TYPE;
455         case 'B':
456             return BYTE_TYPE;
457         case 'S':
458             return SHORT_TYPE;
459         case 'I':
460             return INT_TYPE;
461         case 'F':
462             return FLOAT_TYPE;
463         case 'J':
464             return LONG_TYPE;
465         case 'D':
466             return DOUBLE_TYPE;
467         case '[':
468             len = 1;
469             while (buf[off + len] == '[') {
470                 ++len;
471             }
472             if (buf[off + len] == 'L') {
473                 ++len;
474                 while (buf[off + len] != ';') {
475                     ++len;
476                 }
477             }
478             return new Type(ARRAY, buf, off, len + 1);
479         case 'L':
480             len = 1;
481             while (buf[off + len] != ';') {
482                 ++len;
483             }
484             return new Type(OBJECT, buf, off + 1, len - 1);
485             // case '(':
486         default:
487             return new Type(METHOD, buf, off, buf.length - off);
488         }
489     }

658      * string buffer.

662      */
663     private void getDescriptor(final StringBuffer buf) {
664         if (this.buf == null) {
665             // descriptor is in byte 3 of 'off' for primitive types (buf ==
666             // null)
667             buf.append((char) ((off & 0xFF000000) >>> 24));
668         } else if (sort == OBJECT) {
669             buf.append('L');
670             buf.append(this.buf, off, len);
671             buf.append(';');
672         } else { // sort == ARRAY || sort == METHOD
673             buf.append(this.buf, off, len);
674         }

761                 } else if (d == Boolean.TYPE) {

763                 } else if (d == Byte.TYPE) {
764                     car = 'B';
765                 } else if (d == Character.TYPE) {
766                     car = 'C';
767                 } else if (d == Short.TYPE) {
768                     car = 'S';
769                 } else if (d == Double.TYPE) {
770                     car = 'D';
771                 } else if (d == Float.TYPE) {
772                     car = 'F';
773                 } else /* if (d == Long.TYPE) */{
774                     car = 'J';
775                 }

778             } else if (d.isArray()) {
