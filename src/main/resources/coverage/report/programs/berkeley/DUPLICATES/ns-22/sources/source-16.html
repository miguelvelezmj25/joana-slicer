27 /**
28  * The VLSNReader returns picks out replicated log entries from the log. It
29  * works in tandem with the VLSNIndex, using vlsn->lsn mappings if those are
30  * available, and otherwise scanning the log for replicated entries.
31  *
32  * A VLSNReader is not thread safe, and can only be used serially.
33  */
34 abstract class VLSNReader extends FileReader {
35 
36     final VLSNIndex vlsnIndex;
37 
38     /*
39      * currentVLSN is the target VLSN that the reader is looking for. It is
40      * set if the reader is scanning vlsn sequentially, either forwards or
41      * backwards.
42      */
43     VLSN currentVLSN;
44 
45     /*
46      * The current log entry that the reader is positioned at, in wire record
47      * format.
48      */
49     OutputWireRecord currentFeedRecord;
50 
51     /*
52      * True if the reader has been positioned at a point in the file. Forward
53      * scanning readers may read from either the log buffer or the log files.
54      * It's preferable to read first from the log buffers, in case the log
55      * entry is cached (or the log entry hasn't been written to disk).
56      */
57 
58     /* stats */
59     long nScanned;    // Num log entries seen by the reader
60     long nReposition; // Number of times the reader has used a vlsn->lsn
61                       // mapping and have reset the read window
62 
63     VLSNReader(EnvironmentImpl envImpl,
64                VLSNIndex vlsnIndex,
65                boolean forward,
66                long startLsn,
67                int readBufferSize,
68                long finishLsn)
69         throws DatabaseException {
70 
71         /*
72          * If we go backwards, endOfFileLsn and startLsn must not be null.
73          * Make them the same, so we always start at the same very end.
74          */
75         super(envImpl,

77               forward,

79               null,            // singleFileNumber
80               startLsn,        // endOfFileLsn
81               finishLsn);

85     }

87     void setPosition(long startLsn)
88         throws ChecksumException, FileNotFoundException, DatabaseException {
89 
90         if (startLsn == DbLsn.NULL_LSN) {
91             return;
92         }
93 
94         /*
95          * An assertion: a reposition should never make the reader lose ground.
96          */
97         if (forward) {
98             if (DbLsn.compareTo(getLastLsn(), startLsn) > 0) {
99                 throw EnvironmentFailureException.unexpectedState
100                     ("Feeder forward scanning should not be repositioned to " +
101                      " a position earlier than the current position. Current" +
102                      " lsn = " + DbLsn.getNoFormatString(getLastLsn()) + 
103                      " reposition = " + DbLsn.getNoFormatString(startLsn));
104             }
105         } else {
106             if (DbLsn.compareTo(getLastLsn(), startLsn) < 0) {
107                 throw EnvironmentFailureException.unexpectedState
108                     ("Feeder backward scanning should not be repositioned to " +
109                      " a position later than the current position. Current" +
110                      " lsn = " + DbLsn.getNoFormatString(getLastLsn()) + 
111                      " reposition = " + DbLsn.getNoFormatString(startLsn));
112             }
113         }
114 
115         long fileNum = DbLsn.getFileNumber(startLsn);
116         long offset = DbLsn.getFileOffset(startLsn);
117 
118         if (window.containsLsn(fileNum, offset)) {
119             window.positionBuffer(offset);
120         } else {
121             window.slideAndFill(fileNum, offset, offset, forward);
122         }
123 
124         if (forward) {
125             nextEntryOffset = offset;
126         } else {
127             currentEntryPrevOffset = offset;
128         }
129         nReposition++;
130     }

137         ByteBuffer buffer = entryBuffer.slice();
138         buffer.limit(currentEntryHeader.getItemSize());
139         currentFeedRecord =

141 
142         entryBuffer.position(entryBuffer.position() +
143                              currentEntryHeader.getItemSize());
144         return true;
145     }
