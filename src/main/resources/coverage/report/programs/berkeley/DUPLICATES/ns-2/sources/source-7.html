621     /**
622      * Constructs a new {@link ClassWriter} object and enables optimizations for
623      * "mostly add" bytecode transformations. These optimizations are the
624      * following:
625      * 
626      * <ul>
627      * <li>The constant pool from the original class is copied as is in the new
628      * class, which saves time. New constant pool entries will be added at the
629      * end if necessary, but unused constant pool entries <i>won't be
630      * removed</i>.</li>
631      * <li>Methods that are not transformed are copied as is in the new class,
632      * directly from the original class bytecode (i.e. without emitting visit
633      * events for all the method instructions), which saves a <i>lot</i> of
634      * time. Untransformed methods are detected by the fact that the
635      * {@link ClassReader} receives {@link MethodVisitor} objects that come from
636      * a {@link ClassWriter} (and not from any other {@link ClassVisitor}
637      * instance).</li>
638      * </ul>
639      * 
640      * @param classReader
641      *            the {@link ClassReader} used to read the original class. It
642      *            will be used to copy the entire constant pool from the
643      *            original class and also to copy other fragments of original
644      *            bytecode where applicable.
645      * @param flags
646      *            option flags that can be used to modify the default behavior
647      *            of this class. <i>These option flags do not affect methods
648      *            that are copied as is in the new class. This means that the
649      *            maximum stack size nor the stack frames will be computed for
650      *            these methods</i>. See {@link #COMPUTE_MAXS},
651      *            {@link #COMPUTE_FRAMES}.
652      */
653     public ClassWriter(final ClassReader classReader, final int flags) {
654         this(flags);
655         classReader.copyPool(this);
656         this.cr = classReader;
657     }

663     @Override
664     public final void visit(final int version, final int access,
665             final String name, final String signature, final String superName,
666             final String[] interfaces) {
667         this.version = version;
668         this.access = access;
669         this.name = newClass(name);
670         thisName = name;
671         if (ClassReader.SIGNATURES && signature != null) {
672             this.signature = newUTF8(signature);
673         }
674         this.superName = superName == null ? 0 : newClass(superName);
675         if (interfaces != null && interfaces.length > 0) {
676             interfaceCount = interfaces.length;
677             this.interfaces = new int[interfaceCount];
678             for (int i = 0; i < interfaceCount; ++i) {
679                 this.interfaces[i] = newClass(interfaces[i]);
680             }
681         }
682     }

753     @Override
754     public final void visitInnerClass(final String name,
755             final String outerName, final String innerName, final int access) {
756         if (innerClasses == null) {
757             innerClasses = new ByteVector();
758         }
759         // Sec. 4.7.6 of the JVMS states "Every CONSTANT_Class_info entry in the
760         // constant_pool table which represents a class or interface C that is
761         // not a package member must have exactly one corresponding entry in the
762         // classes array". To avoid duplicates we keep track in the intVal field
763         // of the Item of each CONSTANT_Class_info entry C whether an inner
764         // class entry has already been added for C (this field is unused for
765         // class entries, and changing its value does not change the hashcode
766         // and equality tests). If so we store the index of this inner class
767         // entry (plus one) in intVal. This hack allows duplicate detection in
768         // O(1) time.
769         Item nameItem = newClassItem(name);
770         if (nameItem.intVal == 0) {
771             ++innerClassesCount;
772             innerClasses.putShort(nameItem.index);
773             innerClasses.putShort(outerName == null ? 0 : newClass(outerName));
774             innerClasses.putShort(innerName == null ? 0 : newUTF8(innerName));
775             innerClasses.putShort(access);
776             nameItem.intVal = innerClassesCount;
777         } else {
778             // Compare the inner classes entry nameItem.intVal - 1 with the
779             // arguments of this method and throw an exception if there is a
780             // difference?
781         }
782     }

805     /**
806      * Returns the bytecode of the class that was build with this class writer.
807      * 
808      * @return the bytecode of the class that was build with this class writer.
809      */
810     public byte[] toByteArray() {
811         if (index > 0xFFFF) {
812             throw new RuntimeException("Class file too large!");
813         }
814         // computes the real size of the bytecode of this class
815         int size = 24 + 2 * interfaceCount;
816         int nbFields = 0;
817         FieldWriter fb = firstField;
818         while (fb != null) {
819             ++nbFields;
820             size += fb.getSize();
821             fb = (FieldWriter) fb.fv;
822         }
823         int nbMethods = 0;
824         MethodWriter mb = firstMethod;
825         while (mb != null) {
826             ++nbMethods;
827             size += mb.getSize();
828             mb = (MethodWriter) mb.mv;
829         }
830         int attributeCount = 0;
831         if (bootstrapMethods != null) {
832             // we put it as first attribute in order to improve a bit
833             // ClassReader.copyBootstrapMethods
834             ++attributeCount;
835             size += 8 + bootstrapMethods.length;
836             newUTF8("BootstrapMethods");
837         }
838         if (ClassReader.SIGNATURES && signature != 0) {
839             ++attributeCount;
840             size += 8;
841             newUTF8("Signature");
842         }
843         if (sourceFile != 0) {
844             ++attributeCount;
845             size += 8;
846             newUTF8("SourceFile");
847         }
848         if (sourceDebug != null) {
849             ++attributeCount;
850             size += sourceDebug.length + 6;
851             newUTF8("SourceDebugExtension");
852         }
853         if (enclosingMethodOwner != 0) {
854             ++attributeCount;
855             size += 10;
856             newUTF8("EnclosingMethod");
857         }
858         if ((access & Opcodes.ACC_DEPRECATED) != 0) {
859             ++attributeCount;
860             size += 6;
861             newUTF8("Deprecated");
862         }
863         if ((access & Opcodes.ACC_SYNTHETIC) != 0) {
864             if ((version & 0xFFFF) < Opcodes.V1_5
865                     || (access & ACC_SYNTHETIC_ATTRIBUTE) != 0) {
866                 ++attributeCount;
867                 size += 6;
868                 newUTF8("Synthetic");
869             }
870         }
871         if (innerClasses != null) {
872             ++attributeCount;
873             size += 8 + innerClasses.length;
874             newUTF8("InnerClasses");
875         }
876         if (ClassReader.ANNOTATIONS && anns != null) {
877             ++attributeCount;
878             size += 8 + anns.getSize();
879             newUTF8("RuntimeVisibleAnnotations");
880         }
881         if (ClassReader.ANNOTATIONS && ianns != null) {
882             ++attributeCount;
883             size += 8 + ianns.getSize();
884             newUTF8("RuntimeInvisibleAnnotations");
885         }
886         if (ClassReader.ANNOTATIONS && tanns != null) {
887             ++attributeCount;
888             size += 8 + tanns.getSize();
889             newUTF8("RuntimeVisibleTypeAnnotations");
890         }
891         if (ClassReader.ANNOTATIONS && itanns != null) {
892             ++attributeCount;
893             size += 8 + itanns.getSize();
894             newUTF8("RuntimeInvisibleTypeAnnotations");
895         }
896         if (attrs != null) {
897             attributeCount += attrs.getCount();
898             size += attrs.getSize(this, null, 0, -1, -1);
899         }
900         size += pool.length;
901         // allocates a byte vector of this size, in order to avoid unnecessary
902         // arraycopy operations in the ByteVector.enlarge() method
903         ByteVector out = new ByteVector(size);
904         out.putInt(0xCAFEBABE).putInt(version);
905         out.putShort(index).putByteArray(pool.data, 0, pool.length);
906         int mask = Opcodes.ACC_DEPRECATED | ACC_SYNTHETIC_ATTRIBUTE
907                 | ((access & ACC_SYNTHETIC_ATTRIBUTE) / TO_ACC_SYNTHETIC);
908         out.putShort(access & ~mask).putShort(name).putShort(superName);
909         out.putShort(interfaceCount);
910         for (int i = 0; i < interfaceCount; ++i) {
911             out.putShort(interfaces[i]);
912         }
913         out.putShort(nbFields);
914         fb = firstField;
915         while (fb != null) {
916             fb.put(out);
917             fb = (FieldWriter) fb.fv;
918         }
919         out.putShort(nbMethods);
920         mb = firstMethod;
921         while (mb != null) {
922             mb.put(out);
923             mb = (MethodWriter) mb.mv;
924         }
925         out.putShort(attributeCount);
926         if (bootstrapMethods != null) {
927             out.putShort(newUTF8("BootstrapMethods"));
928             out.putInt(bootstrapMethods.length + 2).putShort(
929                     bootstrapMethodsCount);
930             out.putByteArray(bootstrapMethods.data, 0, bootstrapMethods.length);
931         }
932         if (ClassReader.SIGNATURES && signature != 0) {
933             out.putShort(newUTF8("Signature")).putInt(2).putShort(signature);
934         }
935         if (sourceFile != 0) {
936             out.putShort(newUTF8("SourceFile")).putInt(2).putShort(sourceFile);
937         }
938         if (sourceDebug != null) {
939             int len = sourceDebug.length;
940             out.putShort(newUTF8("SourceDebugExtension")).putInt(len);
941             out.putByteArray(sourceDebug.data, 0, len);
942         }
943         if (enclosingMethodOwner != 0) {
944             out.putShort(newUTF8("EnclosingMethod")).putInt(4);
945             out.putShort(enclosingMethodOwner).putShort(enclosingMethod);
946         }
947         if ((access & Opcodes.ACC_DEPRECATED) != 0) {
948             out.putShort(newUTF8("Deprecated")).putInt(0);
949         }
950         if ((access & Opcodes.ACC_SYNTHETIC) != 0) {
951             if ((version & 0xFFFF) < Opcodes.V1_5
952                     || (access & ACC_SYNTHETIC_ATTRIBUTE) != 0) {
953                 out.putShort(newUTF8("Synthetic")).putInt(0);
954             }
955         }
956         if (innerClasses != null) {
957             out.putShort(newUTF8("InnerClasses"));
958             out.putInt(innerClasses.length + 2).putShort(innerClassesCount);
959             out.putByteArray(innerClasses.data, 0, innerClasses.length);
960         }
961         if (ClassReader.ANNOTATIONS && anns != null) {
962             out.putShort(newUTF8("RuntimeVisibleAnnotations"));
963             anns.put(out);
964         }
965         if (ClassReader.ANNOTATIONS && ianns != null) {
966             out.putShort(newUTF8("RuntimeInvisibleAnnotations"));
967             ianns.put(out);
968         }
969         if (ClassReader.ANNOTATIONS && tanns != null) {
970             out.putShort(newUTF8("RuntimeVisibleTypeAnnotations"));
971             tanns.put(out);
972         }
973         if (ClassReader.ANNOTATIONS && itanns != null) {
974             out.putShort(newUTF8("RuntimeInvisibleTypeAnnotations"));
975             itanns.put(out);
976         }
977         if (attrs != null) {
978             attrs.put(this, null, 0, -1, -1, out);
979         }
980         if (invalidFrames) {
981             anns = null;
982             ianns = null;
983             attrs = null;
984             innerClassesCount = 0;
985             innerClasses = null;
986             bootstrapMethodsCount = 0;
987             bootstrapMethods = null;
988             firstField = null;
989             lastField = null;
990             firstMethod = null;
991             lastMethod = null;
992             computeMaxs = false;
993             computeFrames = true;
994             invalidFrames = false;
995             new ClassReader(out.data).accept(this, ClassReader.SKIP_FRAMES);
996             return toByteArray();
997         }
998         return out.data;
999     }

1005     /**
1006      * Adds a number or string constant to the constant pool of the class being
1007      * build. Does nothing if the constant pool already contains a similar item.
1008      * 
1009      * @param cst
1010      *            the value of the constant to be added to the constant pool.
1011      *            This parameter must be an {@link Integer}, a {@link Float}, a
1012      *            {@link Long}, a {@link Double}, a {@link String} or a
1013      *            {@link Type}.
1014      * @return a new or already existing constant item with the given value.
1015      */
1016     Item newConstItem(final Object cst) {
1017         if (cst instanceof Integer) {
1018             int val = ((Integer) cst).intValue();
1019             return newInteger(val);
1020         } else if (cst instanceof Byte) {
1021             int val = ((Byte) cst).intValue();
1022             return newInteger(val);
1023         } else if (cst instanceof Character) {
1024             int val = ((Character) cst).charValue();
1025             return newInteger(val);
1026         } else if (cst instanceof Short) {
1027             int val = ((Short) cst).intValue();
1028             return newInteger(val);
1029         } else if (cst instanceof Boolean) {
1030             int val = ((Boolean) cst).booleanValue() ? 1 : 0;
1031             return newInteger(val);
1032         } else if (cst instanceof Float) {
1033             float val = ((Float) cst).floatValue();
1034             return newFloat(val);
1035         } else if (cst instanceof Long) {
1036             long val = ((Long) cst).longValue();
1037             return newLong(val);
1038         } else if (cst instanceof Double) {
1039             double val = ((Double) cst).doubleValue();
1040             return newDouble(val);
1041         } else if (cst instanceof String) {
1042             return newString((String) cst);
1043         } else if (cst instanceof Type) {
1044             Type t = (Type) cst;
1045             int s = t.getSort();
1046             if (s == Type.OBJECT) {
1047                 return newClassItem(t.getInternalName());
1048             } else if (s == Type.METHOD) {
1049                 return newMethodTypeItem(t.getDescriptor());
1050             } else { // s == primitive type or array
1051                 return newClassItem(t.getDescriptor());
1052             }
1053         } else if (cst instanceof Handle) {
1054             Handle h = (Handle) cst;
1055             return newHandleItem(h.tag, h.owner, h.name, h.desc);
1056         } else {
1057             throw new IllegalArgumentException("value " + cst);
1058         }
1059     }

1240      * Adds an invokedynamic reference to the constant pool of the class being
1241      * build. Does nothing if the constant pool already contains a similar item.
1242      * <i>This method is intended for {@link Attribute} sub classes, and is
1243      * normally not needed by class generators or adapters.</i>
1244      * 
1245      * @param name
1246      *            name of the invoked method.
1247      * @param desc
1248      *            descriptor of the invoke method.
1249      * @param bsm
1250      *            the bootstrap method.
1251      * @param bsmArgs
1252      *            the bootstrap method constant arguments.
1253      * 
1254      * @return a new or an already existing invokedynamic type reference item.
1255      */
1256     Item newInvokeDynamicItem(final String name, final String desc,
1257             final Handle bsm, final Object... bsmArgs) {
1258         // cache for performance
1259         ByteVector bootstrapMethods = this.bootstrapMethods;
1260         if (bootstrapMethods == null) {
1261             bootstrapMethods = this.bootstrapMethods = new ByteVector();
1262         }
1263 
1264         int position = bootstrapMethods.length; // record current position
1265 
1266         int hashCode = bsm.hashCode();
1267         bootstrapMethods.putShort(newHandle(bsm.tag, bsm.owner, bsm.name,
1268                 bsm.desc));
1269 
1270         int argsLength = bsmArgs.length;
1271         bootstrapMethods.putShort(argsLength);
1272 
1273         for (int i = 0; i < argsLength; i++) {
1274             Object bsmArg = bsmArgs[i];
1275             hashCode ^= bsmArg.hashCode();
1276             bootstrapMethods.putShort(newConst(bsmArg));
1277         }
1278 
1279         byte[] data = bootstrapMethods.data;
