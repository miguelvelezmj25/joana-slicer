71      * Returns the environment handle that will be passed in as an argument to
72      * a database open trigger.
73      *
74      * To ensure that an environment handle is always available, an internal
75      * handle is created and stored in the EnvironmentImpl. The lifetime of the
76      * internal handle (Environment or ReplicatedEnvironment) roughly aligns
77      * with the lifetime of the underlying EnvironmentImpl, or it's subtype
78      * RepImpl.
79      *
80      * For standalone environments, using explicit transactions, the
81      * environment handle that's passed as the argument is the one used to
82      * initiate the transaction. When using AutoTransactions to open a
83      * database, the environment argument to the trigger is the internal
84      * environment handle.
85      *
86      * For replicated environments, the argument to the trigger is the internal
87      * environment handle in all cases. This is done to make the behavior of
88      * the parameter deterministic and independent of the interaction of the
89      * application level database open operations with those initiated from the
90      * "replay" stream.
91      *
92      * @param transaction the transaction associated with the trigger
93      *
94      * @return the environment or null (if the environment is
95      * non-transactional)
96      */
97     private static Environment
98         getOpenTriggerEnvironment(Transaction transaction) {
99 
100         if (transaction == null) {
101             return null;
102         }
103 
104         final EnvironmentImpl envImpl =
105             DbInternal.getTxn(transaction).getEnvironmentImpl();
106 
107         /*
108          * Always return the same internal environment handle for replicated
109          * environments.
110          */
111         if (envImpl.isReplicated()) {
112             return envImpl.getInternalEnvHandle();
113         }
114 
115         /*
116          * Returns the environment handle associated with the transaction. It's
117          * the internal handle for auto transactions, and the application
118          * supplied handle used during transaction creation in all other cases.
119          */
120         return DbInternal.getEnvironment(transaction);
121     }

318      * Generic method for invoking any trigger operation. It iterates over all
319      * the triggers associated with the database and if the trigger fails
320      * invalidates the environment.
321      *
322      * @param dbImpl the database associated with potential triggers
323      *
324      * @param locker provides the transaction associated with the operation
325      *
326      * @param invoker encapsulates the trigger invoker
327      */
328     private static void runTriggers(final DatabaseImpl dbImpl,
329                                     final Locker       locker,
330                                     TriggerInvoker     invoker) {
331 
332         final List<Trigger> triggers = dbImpl.getTriggers();
333 
334         if (triggers == null) {
335             return;
336         }
337 
338         Transaction triggerTransaction =
339             (locker instanceof Txn) ? ((Txn)locker).getTransaction() : null;
340 
341         try {
342             for (Trigger trigger : triggers) {
343                 Trigger dbt = trigger;
344                 invoker.run(triggerTransaction, dbt);
345             }
346         } catch (Exception e) {
347             final EnvironmentImpl env = dbImpl.getEnv();
348             throw EnvironmentFailureException.unexpectedException(env, e);
349         }
350 
351         /*
352          * Note the use of a trigger for the database so that the appropriate
353          * commit/abort triggers can be run.
354          */
355         if (invoker.invokeTransactionTrigger()) {
356             DbInternal.getTxn(triggerTransaction).noteTriggerDb(dbImpl);
357         }
358     }

388      * Invoke the triggers associated with the addition or removal of the
389      * trigger itself. They are typically invoked upon database open, or
390      * database removal.
391      *
392      * @param locker the locker associated with the trigger update operation
393      * @param oldTriggers the current list of triggers
394      * @param newTriggers the new list of triggers
395      */
396     public static void invokeAddRemoveTriggers(Locker locker,
397                                                List<Trigger> oldTriggers,
398                                                List<Trigger> newTriggers) {
399 
400         Set<String> oldNames = new MapOver<String, Trigger>(oldTriggers) {
401             @Override
402             protected String fun(Trigger e) {
403                 return e.getName();
404             }
405         }.run(new HashSet<String>());
406 
407         Set<String> newNames = new MapOver<String, Trigger>(newTriggers) {
408             @Override
409             protected String fun(Trigger e) {
410                 return e.getName();
411             }
412         }.run(new HashSet<String>());
413 
414         Transaction txn = (locker instanceof Txn) ?
415                 ((Txn)locker).getTransaction() : null;
416 
417         /* First invoke removeTrigger */
418         if (oldTriggers != null) {
419             for (Trigger trigger : oldTriggers) {
420                 if (!newNames.contains(trigger.getName())) {
421                     trigger.removeTrigger(txn);
422                 }
423             }
424         }
425 
426         /* Now invoke addTrigger */
427         if (newTriggers != null) {
428             for (Trigger trigger : newTriggers) {
429                 if (!oldNames.contains(trigger.getName())) {
430                     trigger.addTrigger(txn);
431                 }
432             }
433         }
434     }

437      * Lisp inspired Map function.
438      *
439      * @param <R> The result element type for the list being returned.
440      * @param <E> The type of the element being mapped over.
441      */
442     public static abstract class MapOver<R,E> {
443         final Collection<E> c;
444 
445         public MapOver(Collection<E> c) {
446             this.c = c;
447         }
448 
449         @SuppressWarnings("unchecked")
450         public <S extends Collection<R>> S run() {
451             Collection<R> l = new LinkedList<R>();
452             return (S) run(l);
453         }
454 
455         public <S extends Collection<R>> S run(S l) {
456             if (c == null) {
457                 return l;
458             }
459             for (E e : c) {
460                 l.add(fun(e));
461             }
462             return l;
463         }
464 
465         /* The function invoked for each element in the collection. */
466         protected abstract R fun(E e);
467     }
