14 package com.sleepycat.je.rep.util.ldiff;

16 import java.util.ArrayList;
17 import java.util.List;
18 
19 /*
20  * This class is used for tracking the different regions between local and 
21  * remote database, it saves the begin key/data pair and size of different area
22  * on both local and remote database.
23  */
24 public class DiffTracker {
25     /* Start block for the block different area. */
26     private Block startBlock;
27     /* Start position for the different block in the database. */
28     private long remoteStart;
29     /* Size of a block different area. */
30     private long remoteDiffSize;
31     /* Size of each block. */
32     private final long blockSize;
33     /* Collection of different areas. */
34     private final List<MismatchedRegion> regions;
35 
36     public DiffTracker(long blockSize) {
37         this.blockSize = blockSize;
38         regions = new ArrayList<MismatchedRegion>();
39     }
40 
41     /* Set begin key/data pair and start position of a different block area. */
42     public void setBlockDiffBegin(Block startBlock, int startIndex) {
43         this.startBlock = startBlock;
44         remoteStart = startIndex * blockSize;
45     }
46 
47     /* Calculate the size of a block different area. */
48     public void calBlockDiffSize(int currentIndex) {
49         remoteDiffSize = (currentIndex - 1) * blockSize - remoteStart;
50     }
51 
52     /* Add the different local and remote different areas to the collection. */
53     public void addDiffRegion(Window window) {
54         MismatchedRegion region = new MismatchedRegion();
55         setBlockDiff(region, startBlock, remoteDiffSize, true);
56         setWindowDiff(region, window, window.getDiffSize(), true);
57 
58         if (!region.isNull()) {
59             regions.add(region);
60         }
61 
62         remoteDiffSize = 0;
63     }
64 
65     private void setWindowDiff(MismatchedRegion region, 
66                                Window window,
67                                long diffSize, 
68                                boolean doCheck) {
69         if (doCheck && window.getDiffSize() == 0) {
70             return;
71         }
72 
73         region.setLocalBeginKey(window.getBeginKey());
74         region.setLocalBeginData(window.getBeginData());
75         region.setLocalDiffSize(diffSize);
76     }
77 
78     private void setBlockDiff(MismatchedRegion region,
79                               Block block,
80                               long diffSize,
81                               boolean doCheck) {
82         if (doCheck && remoteDiffSize == 0) {
83             return;
84         }
85 
86         region.setRemoteBeginKey(block.getBeginKey());
87         region.setRemoteBeginData(block.getBeginData());
88         region.setRemoteDiffSize(diffSize);
89     }
90 
91     /* Add the window additional area to the difference collection. */
92     public void addWindowAdditionalDiffs(Window window) {
93         MismatchedRegion region = new MismatchedRegion();
94         setWindowDiff(region, window, DiffRecordAnalyzer.DATABASE_END, false);
95         regions.add(region);
96     }
97 
98     /* Add the block additional area to the different collection. */
99     public void addBlockBagAdditionalDiffs(Window window, BlockBag blkBag) {
100         MismatchedRegion region = new MismatchedRegion();
101         setBlockDiff(region, blkBag.getBlock(), 
102                      DiffRecordAnalyzer.DATABASE_END, false);
103         setWindowDiff(region, window, DiffRecordAnalyzer.DATABASE_END, true);
104         regions.add(region);
105     }
106 
107     /* Return the different regions. */
108     public List<MismatchedRegion> getDiffRegions() {
109         return regions;
110     }
111 }
