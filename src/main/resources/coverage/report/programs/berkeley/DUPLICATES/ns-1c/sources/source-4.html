45     /**
46      * See if there are a sufficient number of replicas alive to support
47      * the commit for this transaction. Used as an optimizing step before any
48      * writes are executed, to reduce the number of outstanding writes that
49      * suffer from insufficient ack problems.
50      *
51      * If this node is not the master, just return. A different kind of check
52      * will catch the fact that this node cannot support writes.
53      *
54      * TODO: Read only transactions on the master should not have to wait.
55      * In the future, introduce either a read-only attribute as part of
56      * TransactionConfig or a read only transaction class to optimize this.
57      *
58      * @param insufficientReplicasTimeout timeout in ms
59      * @throws InsufficientReplicasException if there are not enough replicas
60      * connected to this feeder to be able to commit the transaction.
61      */
62     public void ensureReplicasForCommit(MasterTxn txn,
63                                         int insufficientReplicasTimeout)
64         throws DatabaseException, InterruptedException,
65                InsufficientReplicasException {
66 
67         RepNode repNode = repImpl.getRepNode();
68         if (!repNode.isMaster()) {
69             return;
70         }
71 
72         ReplicaAckPolicy ackPolicy =
73             txn.getDefaultDurability().getReplicaAck();
74         int requiredReplicaAckCount = getCurrentRequiredAckCount(ackPolicy);
75         if (logger.isLoggable(FINE)) {
76             LoggerUtils.fine(logger, repImpl,
77                              "Txn " + txn + ": checking that " +
78                              requiredReplicaAckCount +
79                              " feeders exist before starting commit");
80         }
81 
82         /* No need to wait for anyone else, only this node is needed. */
83         if (requiredReplicaAckCount == 0) {
84             return;
85         }
86 
87         if (repNode.feederManager().awaitFeederReplicaConnections
88             (requiredReplicaAckCount, insufficientReplicasTimeout)) {
89             /* Wait was successful */
90             return;
91         }
92 
93         /*
94          * Timed out, not enough replicas connected, or feeder was shutdown
95          * normally, that is, without any exceptions while waiting.
96          */
97         if (!repNode.isMaster()) {
98 
99             /*
100              * Continue if we are no longer the master after the wait. The
101              * transaction will fail if it tries to acquire write locks, or
102              * at commit.
103              */
104             return;
105         }
106 
107         if (ackPolicy.equals(ReplicaAckPolicy.SIMPLE_MAJORITY) &&
108             repNode.getArbiter().activateArbitration()) {
109             return;
110         }
111 
112         final boolean includeArbiters =
113             !ackPolicy.equals(ReplicaAckPolicy.ALL);
114         throw new InsufficientReplicasException(
115             txn, ackPolicy, requiredReplicaAckCount,
116             repNode.feederManager().activeAckReplicas(includeArbiters));
117     }

132     /**
133      * Determine if this transaction has been adequately acknowledged.
134      *
135      * @throws InsufficientAcksException if the transaction's durability
136      * requirements have not been met.
137      */
138     public void ensureSufficientAcks(FeederTxns.TxnInfo txnInfo,
139                                      int timeoutMs)
140         throws InsufficientAcksException {
141 
142         int pendingAcks = txnInfo.getPendingAcks();
143         if (pendingAcks == 0) {
144             return;
145         }
146 
147         MasterTxn txn = txnInfo.getTxn();
148         final int requiredAcks = getCurrentRequiredAckCount(
149             txn.getCommitDurability().getReplicaAck());
150         int requiredAckDelta = txn.getRequiredAckCount() - requiredAcks;
151         if (requiredAckDelta >= pendingAcks) {
152 
153             /*
154              * The group size was reduced while waiting for acks and the
155              * acks received are sufficient given the new reduced group
156              * size.
157              */
158             return;
159         }
160 
161         /* Snapshot the state to be used in the error message */
162         final String dumpState = repImpl.dumpAckFeederState();
163 
164         /*
165          * Repeat the check to ensure that acks have not been received in
166          * the time between the completion of the await() call above and
167          * the creation of the exception message. This tends to happen when
168          * there are lots of threads in the process thus potentially
169          * delaying the resumption of this thread following the timeout
170          * resulting from the await.
171          *
172          * It should be noted that some transactions may be setup to not
173          * decrement the wait latch count for Arbiter acks. Checking an
174          * Arbiters feeder VLSN here will account for the Arbiter ack.
175          */
176         final FeederManager feederManager =
177             repImpl.getRepNode().feederManager();
178         int currentFeederCount =
179             feederManager.getNumCurrentAckFeeders(txn.getCommitVLSN());
180         if (currentFeederCount >= requiredAcks) {
181             String msg = "txn " + txn.getId() +
182                 " commit vlsn:" + txnInfo.getCommitVLSN() +
183                 " acknowledged after explicit feeder check" +
184                 " latch count:" + txnInfo.getPendingAcks() +
185                 " state:" + dumpState +
186                 " required acks:" + requiredAcks;
187 
188             LoggerUtils.info(logger, repImpl, msg);
189             return;
190         }
191 
192         /*
193          * We can avoid the exception if it's possible for this node to enter
194          * activate arbitration. It's useful to check for this again here in
195          * case we happen to lose connections to replicas in the (brief)
196          * period since the pre-log hook.  Note that in this case we merely
197          * want to check; we don't want to switch into active arbitration
198          * unless/until we actually lose the connection to the replica at
199          * commit time. TODO: this doesn't seem right! Shouldn't we require
200          * activation at this point!!!
201          */
202         if (repImpl.getRepNode().getArbiter().activationPossible()) {
203             return;
204         }
205         throw new InsufficientAcksException(txn, pendingAcks, timeoutMs,
206                                             dumpState);
207 
208     }

210     /**
211      * Returns the minimum number of acknowledgments required to satisfy the
212      * ReplicaAckPolicy for a given group size. Does not include the master.
213      * The method factors in considerations like the current arbitration status
214      * of the environment and the composition of the replication group.
215      *
216      * TODO: it seems sufficient to return a number, as opposed to a set of
217      * qualified ack nodes, as long as {@link #replicaAcksQualify} will only
218      * count qualified acks against the required count. That does mean that
219      * getCurrentRequiredAckCount and noteReplicaAcks for a transaction must be
220      * kept consistent.
221      *
222      * @return the number of nodes that are needed, not including the master.
223      */
224     public int getCurrentRequiredAckCount(ReplicaAckPolicy ackPolicy) {
225 
226         /*
227          * If the electableGroupSizeOverride is determining the size of the
228          * election quorum, let it also influence the durability quorum.
229          */
230         RepNode repNode = repImpl.getRepNode();
231         int electableGroupSizeOverride =
232             repNode.getElectionQuorum().getElectableGroupSizeOverride();
233         if (electableGroupSizeOverride > 0) {
234 
235             /*
236              * Use the override-defined group size to determine the
237              * number of acks.
238              */
239             return ackPolicy.minAckNodes(electableGroupSizeOverride) - 1;
240         }
241 
242         Arbiter arbiter = repNode.getArbiter();
243         if (arbiter.isApplicable(ackPolicy)) {
244             return arbiter.getAckCount(ackPolicy);
245         }
246 
247         return ackPolicy.minAckNodes
248             (repNode.getGroup().getAckGroupSize()) - 1;
249     }
