30 /**
31  * The Iterator returned by all stored collections.
32  *
33  * <p>While in general this class conforms to the {@link Iterator} interface,
34  * it is important to note that all iterators for stored collections must be
35  * explicitly closed with {@link #close()}.  The static method {@link
36  * #close(Iterator)} allows calling close for all iterators without
37  * harm to iterators that are not from stored collections, and also avoids
38  * casting.  If a stored iterator is not closed, unpredictable behavior
39  * including process death may result.</p>
40  *
41  * <p>This class implements the {@link Iterator} interface for all stored
42  * iterators.  It also implements {@link ListIterator} because some list
43  * iterator methods apply to all stored iterators, for example, {@link
44  * #previous} and {@link #hasPrevious}.  Other list iterator methods are always
45  * supported for lists, but for other types of collections are only supported
46  * under certain conditions.  See {@link #nextIndex}, {@link #previousIndex},
47  * {@link #add} and {@link #set} for details.</p>
48  *
49  * <p>In addition, this class provides the following methods for stored
50  * collection iterators only.  Note that the use of these methods is not
51  * compatible with the standard Java collections interface.</p>
52  * <ul>
53  * <li>{@link #close()}</li>
54  * <li>{@link #close(Iterator)}</li>
55  * <li>{@link #count()}</li>
56  * <li>{@link #getCollection}</li>
57  * <li>{@link #setReadModifyWrite}</li>
58  * <li>{@link #isReadModifyWrite}</li>
59  * </ul>
60  *
61  * @author Mark Hayes
62  */

487      * Inserts the specified element into the list or inserts a duplicate into
488      * other types of collections (optional operation).
489      * This method conforms to the {@link ListIterator#add} interface when
490      * the collection is a list and the RECNO-RENUMBER access method is used.
491      * Otherwise, this method may only be called when duplicates are allowed.
492      * If duplicates are unsorted, the new value will be inserted in the same
493      * manner as list elements.
494      * If duplicates are sorted, the new value will be inserted in sort order.
495      *
496      * <p>Note that for the JE product, RECNO-RENUMBER databases are not
497      * supported, and therefore this method may only be used to add
498      * duplicates.</p>
499      *
500      * @param value the new value.
501      *
502      * <!-- begin JE only -->
503      * @throws OperationFailureException if one of the <a
504      * href="../je/OperationFailureException.html#writeFailures">Write
505      * Operation Failures</a> occurs.
506      *
507      * @throws EnvironmentFailureException if an unexpected, internal or
508      * environment-wide failure occurs.
509      * <!-- end JE only -->
510      *
511      * @throws UnsupportedOperationException if the collection is a sublist, or
512      * if the collection is indexed, or if the collection is read-only, or if
513      * the collection is a list and the RECNO-RENUMBER access method was not
514      * used, or if the collection is not a list and duplicates are not allowed.
515      *
516      * @throws IllegalStateException if the collection is empty and is not a
517      * list with RECNO-RENUMBER access.
518      *
519      * @throws IllegalArgumentException if a duplicate value is being added
520      * that already exists and duplicates are sorted.
521      *
522      * @throws RuntimeExceptionWrapper if a checked exception is thrown,
523      * including a {@code DatabaseException} on BDB (C Edition).
524      */
525     public void add(E value) {
526 
527         coll.checkIterAddAllowed();
528         try {
529             OperationStatus status = OperationStatus.SUCCESS;
530             if (toNext != 0 && toPrevious != 0) { // database is empty
531                 if (coll.view.keysRenumbered) { // recno-renumber database
532                     /*
533                      * Close cursor during append and then reopen to support
534                      * CDB restriction that append may not be called with a
535                      * cursor open; note the append will still fail if the
536                      * application has another cursor open.
537                      */
538                     close();
539                     status = coll.view.append(value, null, null);
540                     cursor = new DataCursor(coll.view, writeAllowed);
541                     reset();
542                     next(); // move past new record
543                 } else { // hash/btree with duplicates
544                     throw new IllegalStateException
545                         ("Collection is empty, cannot add() duplicate");
546                 }
547             } else { // database is not empty
548                 boolean putBefore = false;
549                 if (coll.view.keysRenumbered) { // recno-renumber database
550                     moveToCurrent();
551                     if (hasNext()) {
552                         status = cursor.putBefore(value);
553                         putBefore = true;
554                     } else {
555                         status = cursor.putAfter(value);
556                     }
557                 } else { // hash/btree with duplicates
558                     if (coll.areDuplicatesOrdered()) {
559                         status = cursor.putNoDupData(null, value, null, true);
560                     } else if (toNext == 0) {
561                         status = cursor.putBefore(value);
562                         putBefore = true;
563                     } else {
564                         status = cursor.putAfter(value);
565                     }
566                 }
567                 if (putBefore) {
568                     toPrevious = 0;
569                     toNext = MOVE_NEXT;
570                 }
571             }
572             if (status == OperationStatus.KEYEXIST) {
573                 throw new IllegalArgumentException("Duplicate value");
574             } else if (status != OperationStatus.SUCCESS) {
575                 throw DbCompat.unexpectedState("Could not insert: " + status);
576             }
577             setAndRemoveAllowed = false;
578         } catch (Exception e) {
579             throw StoredContainer.convertException(e);
580         }
581     }
