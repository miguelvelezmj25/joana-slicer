26 /**
27  * This is an implementation of SSLAuthenticator which authenticates based
28  * on the Distinguished Name (DN) in the SSL peer's certificate.  Matching
29  * is done using Java regular expressions against the RFC1779-formatted DN.
30  * This is typically used to match against the CN portion of the name.
31  */
32 
33 class SSLDNMatcher {
34 
35     private final Pattern pattern;
36 
37     /**
38      * Construct an SSLDNMatcher
39      *
40      * @param params The instantiation params.  The classParams must be
41      * a pattern to be matched to a Distinguished Name in an SSL certificate.
42      * The match pattern must be a valid Java regular expression.
43      * @throws IllegalArgumentException if the pattern is not a valid
44      * regular expression
45      */
46     SSLDNMatcher(InstanceParams params)
47         throws IllegalArgumentException {
48 
49         this.pattern = compileRegex(params.getClassParams());
50     }
51 
52     /*
53      * Check whether the peer certificate matches the configured expression.
54      */
55     public boolean peerMatches(SSLSession sslSession) {
56         Principal principal = null;
57         try {
58             principal = sslSession.getPeerPrincipal();
59         } catch (SSLPeerUnverifiedException pue) {
60             return false;
61         }
62 
63         if (principal != null) {
64             if (principal instanceof X500Principal) {
65                 final X500Principal x500Principal = (X500Principal) principal;
66                 final String name =
67                     x500Principal.getName(X500Principal.RFC1779);
68                 final Matcher m = pattern.matcher(name);
69                 if (m.matches()) {
70                     return true;
71                 }
72             }
73         }
74         return false;
75     }
76 
77     private static Pattern compileRegex(String regex)
78         throws IllegalArgumentException {
79         try {
80             return Pattern.compile(regex);
81         } catch(PatternSyntaxException pse) {
82             throw new IllegalArgumentException(
83                 "pattern is invalid", pse);
84         }
85     }
86 
87     static void validateRegex(String regex)
88         throws IllegalArgumentException {
89 
90         /* ignore the result */
91         compileRegex(regex);
92     }
93 }
