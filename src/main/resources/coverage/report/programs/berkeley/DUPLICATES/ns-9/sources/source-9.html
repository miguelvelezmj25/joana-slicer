644      * benefit, which is to avoid a tree search if the entry is obsolete.  Many
645      * more lookups for non-obsolete entries than obsolete entries will
646      * typically be done.  Because of the high cost of the linear search,
647      * especially when processing large log files, we do not check the TFS.
648      * [#19626]
649      *
650      * @param fileNum the file being cleaned.
651      *
652      * @param fileSummary used to return the true utilization.
653      *
654      * @param inSummary used to return IN utilization info for debugging.
655      *

780                 if (!countOnly && envImpl.isClosing()) {
781                     return false;

785                 if (!countOnly) {

808                     !isIN &&

810                     !isOldBINDelta &&
811                     !isDbTree) {

813                     isObsolete = true;

819                  * considered obsolete by the cleaner.  Processing an old delta

821                  * dirtying and flushing a BIN), and for duplicates databases
822                  * could cause an exception due to the key format change.
823                  * [#21405]

826                     isOldBINDelta &&
827                     fileLogVersion < 8) {
828                     isObsolete = true;
829                 }

832                 final DatabaseId dbId = reader.getDatabaseId();
833                 if (dbId != null) {
834                     databases.add(dbId);

839                  * contention in DbTree.getDb.

847                      * prevent starving other threads that need exclusive

862                      * deleted after the delete cleanup is finished.

873                  * be obsolete, just as DINs and DBINs must be obsolete (and

890                      * deleted LNs are obsolete. Either the delete committed
891                      * and the BIN parent is marked with a pending deleted bit,
892                      * or the delete rolled back, in which case there is no
893                      * reference to this entry. JE 1.7.1 and earlier require a

899                     }

902                     if (!isObsolete &&

910                      */

912

914                             lnEntry.getExpiration(),
915                             lnEntry.isExpirationInHours());

919                             0 - envImpl.getTtlLnPurgeDelay())) {
920

922                                 fileSelector.addPendingLN(
923                                     logLsn,
924                                     new LNInfo(null /*LN*/, dbId,

928                             }

930                             isObsolete = true;

935                              * adding obsolete offset.

945                 if (isObsolete) {

952                         } else if (isIN) {
953                             nINsObsoleteThisRun++;

955                     }

957                     if (checkPendingDbSet != null && dbId != null) {

959                     }

961                     reader.countObsolete();

963                 }
964

966                 reader.countExpired();
967 
968                 /* Don't process further if we are only calculating. */
969                 if (countOnly) {
970                     continue;

972 
973                 /* Evict before processing each entry. */
974                 if (Cleaner.DO_CRITICAL_EVICTION) {
975                     envImpl.daemonEviction(true /*backgroundIO*/);
976                 }
977

980                 assert checkPendingDbSet != null;

1026

1030                 }
1031             }

1034             if (countOnly) {
1035                 return true;
1036             }

1044                           checkPendingDbSet);
1045                 /* Sleep if background read/write limit was exceeded. */
1046                 envImpl.sleepAfterBackgroundIO();

1050             for (final DatabaseId pendingDbId : checkPendingDbSet) {
1051                 final DatabaseImpl db = dbMapTree.getDb
1052                     (pendingDbId, cleaner.lockTimeout, dbCache);
1053                 cleaner.addPendingDB(db);
1054             }
1055

1061 
1062         } catch (ChecksumException e) {
1063             throw new EnvironmentFailureException
1064                 (envImpl, EnvironmentFailureReason.LOG_CHECKSUM, e);
1065         } finally {
1066             /* Subtract the overhead of this method from the budget. */
1067             budget.updateAdminMemoryUsage(0 - adjustMem);
1068

1070             dbMapTree.releaseDbs(dbCache);
1071         }

1074         fileSelector.addCleanedFile(
1075             fileNum, databases, reader.getFirstVLSN(), reader.getLastVLSN(),

1081     /**
1082      * Processes the first LN in the look ahead cache and removes it from the
1083      * cache. While the BIN is latched, look through the BIN for other LNs in
1084      * the cache; if any match, process them to avoid a tree search later.
1085      */
1086     private void processLN(
1087         final Long fileNum,
1088         final TreeLocation location,
1089         final LookAheadCache lookAheadCache,
1090         final Map<DatabaseId, DatabaseImpl> dbCache,
1091         final Set<DatabaseId> checkPendingDbSet) {
1092 
1093         /* Get the first LN from the queue. */
1094         final Long offset = lookAheadCache.nextOffset();
1095         final LNInfo info = lookAheadCache.remove(offset);
1096 
1097         final LN lnFromLog = info.getLN();
1098         final byte[] keyFromLog = info.getKey();
1099         final long logLsn = DbLsn.makeLsn(fileNum, offset);
1100 
1101         /*
1102          * Refresh the DB before processing an LN, in case the DB has been
1103          * deleted since it was added to the lookAheadCache.  If the DB has
1104          * been deleted, perform the housekeeping tasks for an obsolete LN.
1105          *
1106          * Normally the DB will already be present (and non-deleted) in the
1107          * dbCache. But because of the lookAheadCache and the periodic clearing
1108          * of the dbCache in processFile, it's possible for a DB to be deleted
1109          * in between placing it in the lookAheadCache and the call to
1110          * processLN. [#22202]
1111          */
1112         final DatabaseId dbId = info.getDbId();
1113 
1114         final DatabaseImpl db = envImpl.getDbTree().getDb(
1115             dbId, cleaner.lockTimeout, dbCache);
1116 
1117         if (db == null || db.isDeleted()) {
1118             nLNsObsoleteThisRun++;
1119             if (checkPendingDbSet != null) {
1120                 checkPendingDbSet.add(dbId);
1121             }
1122             return;
1123         }
1124 
1125         nLNsCleanedThisRun++;
1126 
1127         /* Status variables are used to generate debug tracing info. */
1128         boolean processedHere = true; // The LN was cleaned here.
1129         boolean obsolete = false;     // The LN is no longer in use.
1130         boolean completed = false;    // This method completed.
1131 
1132         BIN bin = null;
1133         Map<Long, LNInfo> pendingLNs = null;
1134 
1135         try {
1136             final Tree tree = db.getTree();
1137             assert tree != null;
1138 
1139             /* Find parent of this LN. */
1140             final boolean parentFound = tree.getParentBINForChildLN(
1141                 location, keyFromLog, false /*splitsAllowed*/,
1142                 false /*blindDeltaOps*/, Cleaner.UPDATE_GENERATION);
1143 
1144             bin = location.bin;
1145             final int index = location.index;
1146 
1147             if (!parentFound) {
1148 
1149                 nLNsDeadThisRun++;
1150                 obsolete = true;
1151                 completed = true;
1152                 return;
1153             }
1154 
1155             /*
1156              * Now we're at the BIN parent for this LN.  If knownDeleted, LN is
1157              * deleted and can be purged.
1158              */
1159             if (bin.isEntryKnownDeleted(index)) {
1160                 nLNsDeadThisRun++;
1161                 obsolete = true;
1162                 completed = true;
1163                 return;
1164             }
1165 
1166             /* Process this LN that was found in the tree. */
1167             processedHere = false;
1168 
1169             LNInfo pendingLN =
1170                 processFoundLN(info, logLsn, bin.getLsn(index), bin, index);
1171 
1172             if (pendingLN != null) {
1173                 pendingLNs = new HashMap<>();
1174                 pendingLNs.put(logLsn, pendingLN);
1175             }
1176 
1177             completed = true;
1178 
1179             /*
1180              * For all other non-deleted LNs in this BIN, lookup their LSN
1181              * in the LN queue and process any matches.
1182              */
1183             for (int i = 0; i < bin.getNEntries(); i += 1) {
1184 
1185                 final long binLsn = bin.getLsn(i);
1186 
1187                 if (i != index &&
1188                     !bin.isEntryKnownDeleted(i) &&
1189                     !bin.isEntryPendingDeleted(i) &&
1190                     DbLsn.getFileNumber(binLsn) == fileNum) {
1191 
1192                     final Long myOffset = DbLsn.getFileOffset(binLsn);
1193                     final LNInfo myInfo = lookAheadCache.remove(myOffset);
1194 
1195                     /* If the offset is in the cache, it's a match. */
1196                     if (myInfo != null) {
1197                         nLNQueueHitsThisRun++;
1198                         nLNsCleanedThisRun++;
1199 
1200                         pendingLN =
1201                             processFoundLN(myInfo, binLsn, binLsn, bin, i);
1202 
1203                         if (pendingLN != null) {
1204                             if (pendingLNs == null) {
1205                                 pendingLNs = new HashMap<>();
1206                             }
1207                             pendingLNs.put(binLsn, pendingLN);
1208                         }
1209                     }
1210                 }
1211             }
1212         } finally {
1213             if (bin != null) {
1214                 bin.releaseLatch();
1215             }
1216 
1217             /* BIN must not be latched when synchronizing on FileSelector. */
1218             if (pendingLNs != null) {
1219                 for (Map.Entry<Long, LNInfo> entry : pendingLNs.entrySet()) {
1220                     fileSelector.addPendingLN(
1221                         entry.getKey(), entry.getValue());
1222                 }
1223             }
1224 
1225             if (processedHere) {
1226                 cleaner.logFine(Cleaner.CLEAN_LN, lnFromLog, logLsn,
1227                     completed, obsolete, false /*migrated*/);
1228             }
1229         }
1230     }

1232     /**
1233      * Processes an LN that was found in the tree.  Lock the LN's LSN and
1234      * then migrates the LN, if the LSN of the LN log entry is the active LSN
1235      * in the tree.
1236      *
1237      * @param info identifies the LN log entry.
1238      *
1239      * @param logLsn is the LSN of the log entry.
1240      *
1241      * @param treeLsn is the LSN found in the tree.
1242      *
1243      * @param bin is the BIN found in the tree; is latched on method entry and
1244      * exit.
1245      *
1246      * @param index is the BIN index found in the tree.
1247      *
1248      * @return a non-null LNInfo if it should be added to the pending LN list,
1249      * after releasing the BIN latch.
1250      */
1251     private LNInfo processFoundLN(
1252         final LNInfo info,
1253         final long logLsn,
1254         final long treeLsn,
1255         final BIN bin,
1256         final int index) {
1257 
1258         final LN lnFromLog = info.getLN();
1259         final byte[] key = info.getKey();
1260 
1261         final DatabaseImpl db = bin.getDatabase();
1262         final boolean isTemporary = db.isTemporary();
1263 
1264         /* Status variables are used to generate debug tracing info. */
1265         boolean obsolete = false;  // The LN is no longer in use.
1266         boolean migrated = false;  // The LN was in use and is migrated.
1267         boolean completed = false; // This method completed.
1268 
1269         BasicLocker locker = null;
1270         try {
1271             final Tree tree = db.getTree();
1272             assert tree != null;
1273 
1274             /*
1275              * Before migrating an LN, we must lock it and then check to see
1276              * whether it is obsolete or active.
1277              *
1278              * 1. If the LSN in the tree and in the log are the same, we will
1279              * attempt to migrate it.
1280              *
1281              * 2. If the LSN in the tree is < the LSN in the log, the log entry
1282              * is obsolete, because this LN has been rolled back to a previous
1283              * version by a txn that aborted.
1284              *
1285              * 3. If the LSN in the tree is > the LSN in the log, the log entry
1286              * is obsolete, because the LN was advanced forward by some
1287              * now-committed txn.
1288              *
