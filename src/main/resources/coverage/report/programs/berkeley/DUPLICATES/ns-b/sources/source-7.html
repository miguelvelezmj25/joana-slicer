37 /**
38  * Singleton collection of environments.  Responsible for environment open and
39  * close, supporting this from multiple threads by synchronizing on the pool.
40  *
41  * To avoid multiple environment openings from being blocked by recovery in
42  * getEnvironment(), the EnvironmentImpl constructor is broken into two parts,
43  * with the second part (EnvironmentImpl.finishInit) doing the recovery.
44  *
45  * When synchronizing on two or more of the following objects the
46  * synchronization order must be as follows.  Synchronization is not performed
47  * in constructors, of course, because no other thread can access the object.
48  *
49  * Synchronization order:  Environment, DbEnvPool, EnvironmentImpl, Evictor
50  */
51 public class DbEnvPool {

95     /**
96      * Find a single environment, used by Environment handles and by command
97      * line utilities.
98      *
99      * @return a non-null EnvironmentImpl.
100      */
101     public EnvironmentImpl getEnvironment(File envHome,
102                                           EnvironmentConfig config,
103                                           boolean checkImmutableParams,
104                                           RepConfigProxy repConfigProxy)
105         throws EnvironmentNotFoundException, EnvironmentLockedException {
106 
107         String environmentKey = null;
108         EnvironmentImpl envImpl = null;
109         synchronized (this) {
110             environmentKey = getEnvironmentMapKey(envHome);
111             envImpl = envs.get(environmentKey);
112 
113             if (envImpl != null) {
114 
115                 /*
116                  * If the envImpl intance returned is standalone, but users are
117                  * actually creating a replicated environment, throw out an
118                  * UnsupportedOperationException. We needn't worry about the
119                  * read only property, since a replicated environment can't be
120                  * read only.
121                  */
122                 if (!envImpl.isReplicated() && (repConfigProxy != null)) {
123                     throw new UnsupportedOperationException
124                        ("This environment was previously opened as a " +
125                         "standalone environment. It cannot be re-opened for " +
126                         "replication.");
127                 }
128 
129                 /*
130                  * If the envImpl instance returned is replicated, but users
131                  * are actually creating a standalone environment and it is not
132                  * read only, then throw out an UnsupportedOperationException.
133                  */
134                 if (envImpl.isReplicated() && (repConfigProxy == null) &&
135                     !config.getReadOnly()) {
136                     throw new UnsupportedOperationException
137                         ("This environment was previously opened for " +
138                          "replication. It cannot be re-opened in read/write " +
139                          "mode for standalone operation.");
140                 }
141 
142                 /*
143                  * If envImpl is used for an Arbiter.
144                  */
145                 if (envImpl.isArbiter()) {
146                     throw new UnsupportedOperationException(
147                         "An Arbiter is currently using " +
148                         "this directory. " + envHome.getAbsolutePath());
149                 }
150 
151                 envImpl.checkIfInvalid();
152 
153                 if (checkImmutableParams) {
154 
155                     /*
156                      * If a non-null configuration parameter was passed to the
157                      * Environment ctor and the underlying EnvironmentImpl
158                      * already exists, check that the configuration parameters
159                      * specified match those of the currently open environment.
160                      * An exception is thrown if the check fails.
161                      *
162                      * Don't do this check if we create the environment here
163                      * because the creation might modify the parameters, which
164                      * would create a Catch-22 in terms of validation.  For
165                      * example, je.maxMemory will be overridden if the JVM's
166                      * -mx flag is less than that setting, so the new resolved
167                      * config parameters won't be the same as the passed
168                      * in config.
169                      */
170                     envImpl.checkImmutablePropsForEquality
171                         (DbInternal.getProps(config));
172                 }
173                 /* Successful, increment reference count */
174                 envImpl.incOpenCount();
175             } else {
176 
177                 /*
178                  * If a shared cache is used, get another (any other,
179                  * doesn't matter which) environment that is sharing the
180                  * global cache.
181                  */
182                 EnvironmentImpl sharedCacheEnv = config.getSharedCache() ?
183                     getAnySharedCacheEnv() : null;
184 
185                 /*
186                  * Environment must be instantiated. If it can be created,
187                  * the configuration must have allowCreate set.  Note that
188                  * the environment is added to the SharedEvictor before the
189                  * EnvironmentImpl ctor returns, by
190                  * RecoveryManager.buildTree.
191                  */
192                 envImpl =
193                     (repConfigProxy == null) ?
194                      new EnvironmentImpl(envHome, config, sharedCacheEnv) :
195                      loadRepImpl(envHome, config, sharedCacheEnv,
196                                  repConfigProxy);
197                 assert config.getSharedCache() == envImpl.getSharedCache();
198 
199                 envImpl.incOpenCount();
200                 envs.put(environmentKey, envImpl);
201                 addToSharedCacheEnvs(envImpl);
202             }
203         }
204 
205         /*
206          * An new EnvironmentImpl was created. Call finishInit outside the
207          * synchronized block to support concurrent recovery for more than one
208          * environment.
209          *
210          * Note that finishInit must be called even if an existing envImpl was
211          * found, because initialization (recovery) for that envImpl may not be
212          * complete.  finishInit is synchronized and this ensures that recovery
213          * will be complete when it returns.
214          *
215          * If this environment finishInit() fails in any way, make sure it
216          * is removed from the envs map. If it isn't, it will block all
217          * future attempts to create the environment.
218          */
219         TestHookExecute.doHookIfSet(beforeFinishInitHook, envImpl);
220         boolean success = false;
221         try {
222             if (envImpl.finishInit(config)) {
223                 /* Initialization (recovery) was performed. */
224                 synchronized(this) {
225                     finishAdditionOfSharedCacheEnv(envImpl);
226                 }
227             }
228             success = true;
229         } finally {
230             if (!success) {
231                 synchronized(this) {
232                     envs.remove(environmentKey);
233                     sharedCacheEnvs.remove(envImpl);
234                 }
235             }
236         }
237 
238         return envImpl;
239     }

241     /**
242      * Use reflection to create a RepImpl, to avoid introducing HA compilation
243      * dependencies to non-replication code.
244      */
245      private EnvironmentImpl loadRepImpl(File envHome,
246                                          EnvironmentConfig config,
247                                          EnvironmentImpl sharedCacheEnv,
248                                          RepConfigProxy repConfigProxy)
249         throws DatabaseException {
250 
251         final String repClassName = "com.sleepycat.je.rep.impl.RepImpl";
252         final String envImplName = "com.sleepycat.je.dbi.EnvironmentImpl";
253         final String repProxy = "com.sleepycat.je.dbi.RepConfigProxy";
254         try {
255             final Class<?> repClass = Class.forName(repClassName);
256             return (EnvironmentImpl)
257                 repClass.getConstructor(envHome.getClass(),
258                                         config.getClass(),
259                                         Class.forName(envImplName),
260                                         Class.forName(repProxy)).
261                 newInstance(envHome, config, sharedCacheEnv, repConfigProxy);
262         } catch (InvocationTargetException e) {
263             if (e.getCause() instanceof RuntimeException) {
264                 /* Propate runtime exceptions thrown by the ctor. */
265                 throw (RuntimeException)e.getCause();
266             }
267             throw EnvironmentFailureException.unexpectedException(e);
268         } catch (Exception e) {
269 
270             /*
271              * This intentially violates our guideline for not catching
272              * Exception in order to avoid many catches for all the checked
273              * exceptions thrown by Class.forName and Class.getConstructor.  No
274              * other methods in the try block throw checked exceptions.
275              */
276             throw EnvironmentFailureException.unexpectedException(e);
277         }
278     }

281     private void addToSharedCacheEnvs(EnvironmentImpl envImpl)
282         throws DatabaseException {
283

292     /* Post-processing of SharedCacheEnv addition, after recovery is done. */
293     private void finishAdditionOfSharedCacheEnv(EnvironmentImpl envImpl)
294         throws DatabaseException {
295 
296         if (envImpl.getSharedCache()) {
297             if (!sharedCacheEnvs.contains(envImpl)) {
298                 throw EnvironmentFailureException.unexpectedState();
299             }
300             assert envImpl.getEvictor().checkEnv(envImpl);
301             resetSharedCache(-1, envImpl);

325     /**
326      * Called by EnvironmentImpl.close to perform the close operation while
327      * synchronized on the DbEnvPool.
328      *
329      * Synchronization on this DbEnvPool during the close is used to protect
330      * its data structures.  Unfortunately, this means that a long checkpoint
331      * during a close will block other closes and opens.  We may want to
332      * improve this in the future.  However, at least there is a user
333      * workaround: perform a full checkpoint before closing the environment.
334      */
335     synchronized void closeEnvironment(EnvironmentImpl envImpl,
336                                        boolean doCheckpoint,
337                                        boolean isAbnormalClose) {
338         synchronized (envImpl) {
339             /* Hold the reference count stable. */
340             if (envImpl.decOpenCount()) {
341                 try {
342                     envImpl.doClose(doCheckpoint, isAbnormalClose);
343                 } finally {
344                     removeEnvironment(envImpl);
345                 }
346             }
347         }
348     }

364     /**
365      * Removes an EnvironmentImpl from the pool after it has been closed.  This
366      * method is called while synchronized.  Note that the environment was
367      * removed from the SharedEvictor by EnvironmentImpl.shutdownEvictor.
368      */
369     private void removeEnvironment(EnvironmentImpl envImpl) {
370 
371         final String environmentKey =
372             getEnvironmentMapKey(envImpl.getEnvironmentHome());
373 
374         final boolean found = envs.remove(environmentKey) != null;
375 
376         if (sharedCacheEnvs.remove(envImpl)) {
377 
378             assert found && envImpl.getSharedCache();
379             assert !envImpl.getEvictor().checkEnv(envImpl);
380 
381             if (sharedCacheEnvs.isEmpty()) {
382                 envImpl.getEvictor().shutdown();
383                 envImpl.getOffHeapCache().shutdown();
384             } else {
385                 envImpl.getMemoryBudget().subtractCacheUsage();
386                 resetSharedCache(-1, null);
387             }
388         } else {
389             assert !found || !envImpl.getSharedCache();
390         }
391 
392         /*
393          * Latch notes may only be cleared when there is no possibility that
394          * any environment is open.
395          */
396         if (envs.isEmpty()) {
397             LatchSupport.clear();
398         }
399     }

430      *
431      * @param newMaxMemory is the new total cache budget or is less than 0 if

433      * when it has changed via setMutableConfig.
434      *
435      * @param skipEnv is an environment that should not be reset, or null.

438      */
439     private void resetSharedCache(long newMaxMemory, EnvironmentImpl skipEnv)
440         throws DatabaseException {
441 
442         for (EnvironmentImpl envImpl : sharedCacheEnvs) {
443 
444             /*
445              * To avoid spurious exceptions, don't reset invalid envs that have
446              * not yet been removed.  They aren't usable, and we expect them
447              * to be closed and removed very soon.
448              */

450                 envImpl.getMemoryBudget().reset(newMaxMemory,
451                                                 false /*newEnv*/,
452                                                 envImpl.getConfigManager());
453             }
