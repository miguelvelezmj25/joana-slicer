55  * The Learner agent.  It runs in its own dedicated thread, listening for
56  * messages announcing the results of elections and, in turn, invoking

136      * Processes a result message
137      *
138      * @param proposal the winning proposal
139      * @param value the winning value
140      */
141     synchronized public void processResult(Proposal proposal, Value value) {
142         if ((currentProposal != null) &&
143             (proposal.compareTo(currentProposal) < 0)) {
144             LoggerUtils.logMsg(logger, envImpl, formatter, Level.FINE,
145                                "Ignoring obsolete winner: " + proposal);
146             return;
147         }
148         currentProposal = proposal;
149         currentValue = value;
150 
151         /* We have a new winning proposal and value, inform the listeners */
152         synchronized (listeners) {
153             for (Listener listener : listeners) {
154                 try {
155                     listener.notify(currentProposal, currentValue);
156                 } catch (Exception e) {
157                     e.printStackTrace();
158                     /* Report the exception and keep going. */
159                     LoggerUtils.logMsg
160                         (logger, envImpl, formatter, Level.SEVERE,
161                          "Exception in Learner Listener: " + e.getMessage());
162                     continue;
163                 }
164             }
165         }
166     }

168     /**
169      * The main Learner loop. It accepts requests and propagates them to its
170      * Listeners, if the proposal isn't out of date.
171      */
172     @Override
173     public void run() {
174         serviceDispatcher.register(SERVICE_NAME, channelQueue);
175         LoggerUtils.logMsg
176             (logger, envImpl, formatter, Level.FINE, "Learner started");
177         DataChannel channel = null;
178         try {
179             while (true) {
180                 channel = serviceDispatcher.takeChannel
181                     (SERVICE_NAME,
182                      true /* blocking socket */,
183                      protocol.getReadTimeout());
184 
185                 if (channel == null) {
186                     /* A soft shutdown. */
187                    return;
188                 }
189 
190                 BufferedReader in = null;
191                 PrintWriter out = null;
192                 try {
193                     in = new BufferedReader
194                         (new InputStreamReader(
195                             Channels.newInputStream(channel)));
196                     final String requestLine = in.readLine();
197                     if (requestLine == null) {
198                         continue;
199                     }
200                     final RequestMessage requestMessage;
201                     try {
202                         requestMessage = protocol.parseRequest(requestLine);
203                     } catch (InvalidMessageException ime) {
204                         protocol.processIME(channel, ime);
205                         continue;
206                     }
207 
208                     final MessageOp op = requestMessage.getOp();
209                     LoggerUtils.logMsg(logger, envImpl, formatter,
210                                        Level.FINEST,
211                                        "learner request: " + op +
212                                        " sender: " +
213                                        requestMessage.getSenderId());
214                     if (op == protocol.RESULT) {
215                         Result result = (Result) requestMessage;
216                         processResult(result.getProposal(), result.getValue());
217                     } else if (op == protocol.MASTER_QUERY) {
218                         processMasterQuery(channel, requestMessage);
219                     } else if (op == protocol.SHUTDOWN) {
220                         LoggerUtils.logMsg
221                             (logger, envImpl, formatter, Level.FINE,
222                              "Learner thread exiting");
223                         break;
224                     } else {
225                         final String message =
226                             "Malformed request: '" + requestLine + "'" +
227                             " Unexpected op:" + op;
228                         final InvalidMessageException ime =
229                             new InvalidMessageException(MessageError.BAD_FORMAT,
230                                                         message);
231                         protocol.processIME(channel, ime);
232                         continue;
233                     }
234                 } catch (IOException e) {
235                     LoggerUtils.logMsg
236                         (logger, envImpl, formatter, Level.INFO,
237                          "IO exception: " + e.getMessage());
238                 } catch (Exception e) {
239                     throw EnvironmentFailureException.unexpectedException(e);
240                 } finally {
241                     Utils.cleanup(logger, envImpl, formatter, channel, in, out);
242                 }
243             }
244         } catch (InterruptedException e) {
245             if (isShutdown()) {
246                 /* Treat it like a shutdown, exit the thread. */
247                 return;
248             }
249             LoggerUtils.logMsg(logger, envImpl, formatter, Level.WARNING,
250                                "Learner unexpected interrupted");
251             throw EnvironmentFailureException.unexpectedException(e);
252        } finally {
253             serviceDispatcher.cancel(SERVICE_NAME);
254             cleanup();
255        }
256     }

259      * Responds to a query for the current master. A response is only
260      * generated if the node is currently in the Master or Replica state to
261      * ensure that the information is reasonably current.
262      */
263     synchronized private void processMasterQuery(DataChannel channel,
264                                                  RequestMessage requestMessage)
265     {
266         if ((currentProposal == null) || (currentValue == null)) {
267             /* Don't have any election results to share. */
268             return;
269         }
270 
271         if ((envImpl == null) || !((RepImpl) envImpl).getState().isActive()) {
272             /* Knowledge of master is potentially obsolete */
273             return;
274         }
275 
276         PrintWriter out = null;
277         try {
278             out = new PrintWriter(Channels.newOutputStream(channel), true);
279             final MasterQueryResponse responseMessage = protocol.new
280                 MasterQueryResponse(currentProposal, currentValue);
281 
282             /*
283              * The request message may be of an earlier version. If so, this
284              * node transparently read the older version. JE only throws out
285              * InvalidMessageException when the version of the request message
286              * is newer than the current protocol. To avoid sending a response
287              * that the requester cannot understand, we send a response in the
288              * same version as that of the original request message.
289              */
290             responseMessage.setSendVersion(requestMessage.getSendVersion());
291             out.println(responseMessage.wireFormat());
292         } finally {
293             if (out != null) {
294                 out.close();
295             }
296         }
297     }

300      * Queries other learners, in parallel, to determine whether they know of
301      * an existing master in the group. If one is found, the result is
302      * processed via {@link #processResult} as though it were an election
303      * result that was sent to the Learner, resulting in the node transitioning
304      * to the master or replica state as appropriate.
305      * <p>
306      * Note that this node itself is not allowed to become a master as a result
307      * of such a query. It must only do so via an election.
308      *
309      * @param learnerSockets the sockets associated with learners at other
310      * nodes. The nodes are queried on these sockets.
311      */
312     public void queryForMaster(Set<InetSocketAddress> learnerSockets) {
313         if (learnerSockets.size() <= 0) {
314                 return;
315         }
316         int threadPoolSize = Math.min(learnerSockets.size(), 10);
317         final ExecutorService pool =
318             Executors.newFixedThreadPool
319                (threadPoolSize, new StoppableThreadFactory("JE Learner",
320                                                            logger));
321         try {
322             RequestMessage masterQuery = protocol.new MasterQuery();
323             FutureTrackingCompService<MessageExchange> compService =
324                 Utils.broadcastMessage(learnerSockets,
325                                        Learner.SERVICE_NAME,
326                                        masterQuery,
327                                        pool);
328             /*
329              * 2 * read timeout below to roughly cover the max time for a
330              * message exchange.
331              */
332             new Utils.WithFutureExceptionHandler<MessageExchange>
333             (compService, 2 * protocol.getReadTimeout(), TimeUnit.MILLISECONDS,
334                 logger, (RepImpl)envImpl, formatter) {
335 
336                 @Override
337                 protected void processResponse(MessageExchange me) {
338 
339                     if (me.getResponseMessage().getOp() ==
340                         protocol.MASTER_QUERY_RESPONSE){
341                         MasterQueryResponse accept =
342                             (MasterQueryResponse) me.getResponseMessage();
343                         MasterValue masterValue =
344                             (MasterValue) accept.getValue();
345                         if ((masterValue != null) &&
346                             masterValue.getNameId().
347                             equals(protocol.getNameIdPair())) {
348 
349                             /*
350                              * Should not transition to master as a result
351                              * of a query it risks imposing a hard recovery
352                              * on the replicas.
353                              */
354                             return;
355                         }
356                         processResult(accept.getProposal(), masterValue);
357                     }
358                 }
359 
360                 @Override
361                 protected boolean isShutdown() {
362                     return Learner.this.isShutdown();
363                 }
364             }.execute();
365         } finally {
366             pool.shutdownNow();
367         }
368     }

371      * Returns the socket address for the current master, or null if one
372      * could not be determined from the available set of learners. This API
373      * is suitable for tools which need to contact the master for a specific
374      * service, e.g. to delete a replication node, or to add a monitor. This
375      * method could be used in principle to establish other types of nodes as
376      * well via a tool, but that is currently done by the handshake process.
377      *
378      * @param protocol the protocol to be used when determining the master
379      *
380      * @param learnerSockets the learner to be queried for the master
381      * @param logger for log messages
382      * @return the MasterValue identifying the master
383      * @throws UnknownMasterException if no master could be established
384      */
385     static public MasterValue findMaster
386         (final Protocol protocol,
387          Set<InetSocketAddress> learnerSockets,
388          final Logger logger,
389          final RepImpl repImpl,
390          final Formatter formatter )
391         throws UnknownMasterException {
392 
393         if (learnerSockets.size() <= 0) {
394                 return null;
395         }
396         int threadPoolSize = Math.min(learnerSockets.size(), 10);
397         final ExecutorService pool =
398             Executors.newFixedThreadPool(threadPoolSize);
399         try {
400             FutureTrackingCompService<MessageExchange> compService =
401                 Utils.broadcastMessage(learnerSockets,
402                                        Learner.SERVICE_NAME,
403                                        protocol.new MasterQuery(),
404                                        pool);
405 
406             final List<MasterQueryResponse> results = new LinkedList<>();
407             new Utils.WithFutureExceptionHandler<MessageExchange>
408             (compService, 2 * protocol.getReadTimeout(), TimeUnit.MILLISECONDS,

410

412                 protected void processResponse(MessageExchange me) {
413

415

417                         protocol.MASTER_QUERY_RESPONSE){
418                         results.add((MasterQueryResponse)response);
419                     } else {
420                         LoggerUtils.logMsg(logger, repImpl, formatter,
421                                            Level.WARNING,
422                                            "Unexpected MasterQuery response:" +
423                                            response.wireFormat());
424                     }
425                 }
426 
427                 @Override
428                 protected boolean isShutdown() {
429                     return (repImpl != null) && !repImpl.isValid();
430                 }
431 
432             }.execute();

435             for (MasterQueryResponse result : results) {
436                 if ((bestResponse == null) ||

454      * A method to re-broadcast this Learner's notion of the master. This
455      * re-broadcast is done primarily to inform an obsolete master that it's no
456      * longer the current master. Obsolete master situations arise in network

458      * in an election, nor is it informed about the results. The re-broadcast

460      * master receives the new results after the network partition has been
461      * fixed, it will revert to being a replica.

464      * @param threadPool the pool used to dispatch broadcast requests in

468                                  ExecutorService threadPool) {

471         synchronized (this) {
472             if ((currentProposal == null) || (currentValue == null)) {
473                 return;
474             }
475             winningProposal =
476                 new WinningProposal(currentProposal, currentValue, null);
477         }
478 
479         final RepImpl repImpl = (RepImpl)envImpl;

481             return;
482         }
