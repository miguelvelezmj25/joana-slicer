30 /**
31  * Object to sync-up the Feeder and Subscriber to establish the VLSN from
32  * which subscriber should should start stream log entries from feeder.
33  */

48                                   Logger logger) {

53         this.type = type;

57     /**
58      * Execute sync-up to the Feeder.  Request Feeder to start a replication
59      * stream from a start VLSN, if it is available. Otherwise return NULL
60      * VLSN to subscriber.
61      *
62      * @param reqVLSN  VLSN requested by subscriber to stream log entries
63      *
64      * @return start VLSN from subscribe can stream log entries
65      * @throws InternalException if fail to execute syncup
66      */
67     public VLSN execute(VLSN reqVLSN) throws InternalException {
68 
69         final long startTime = System.currentTimeMillis();
70 
71         LoggerUtils.info(logger, repImpl,
72                          "Subscriber-Feeder " + namedChannel.getNameIdPair() +
73                          " syncup started.");
74 
75         try {
76             /* first query the start VLSN from feeder */
77             final VLSN startVLSN = getStartVLSNFromFeeder(reqVLSN);
78             if (!startVLSN.equals(VLSN.NULL_VLSN)) {
79                 LoggerUtils.info(logger, repImpl,
80                                  "Response from feeder  " +
81                                  namedChannel.getNameIdPair() +
82                                  ": the start VLSN " + startVLSN +
83                                  ", the requested VLSN " + reqVLSN +
84                                  ", send startStream request with filter.");
85 
86                 /* start streaming from feeder if valid start VLSN */
87                 protocol.write(protocol.new StartStream(startVLSN, filter),
88                                namedChannel);
89             } else {
90                 LoggerUtils.info(logger, repImpl,
91                                  "Unable to stream from Feeder " +
92                                  namedChannel.getNameIdPair() +
93                                  " from requested VLSN " + reqVLSN);
94             }
95             return startVLSN;
96         } catch (IllegalStateException | IOException e) {
97             throw new InternalException(e.getMessage());
98         } finally {
99             LoggerUtils.info(logger, repImpl,
100                              String.format("Subscriber to feeder " +
101                                            namedChannel.getNameIdPair() +
102                                            " sync-up done, elapsed time: %,dms",
103                                            System.currentTimeMillis() -
104                                            startTime));
105         }
106     }

108     /**
109      * Request a start VLSN from feeder. The feeder will return a valid
110      * start VLSN, which can be equal to or earlier than the request VLSN,
111      * or null if feeder is unable to service the requested VLSN.
112      *
113      * @param requestVLSN start VLSN requested by subscriber
114      *
115      * @return VLSN a valid start VLSN from feeder, or null if it unavailable
116      * at the feeder
117      * @throws IOException if unable to read message from channel
118      * @throws IllegalStateException if the feeder sends an unexpected message
119      */
120     private VLSN getStartVLSNFromFeeder(VLSN requestVLSN)
121             throws IOException, IllegalStateException {
122 
123         LoggerUtils.fine(logger, repImpl,
124                          "Subscriber send requested VLSN " + requestVLSN + 
125                          " to feeder " + namedChannel.getNameIdPair());
126 
127         /* ask the feeder for the requested VLSN. */
128         protocol.write(protocol.new EntryRequest(requestVLSN, type),
129                        namedChannel);
130 
131         /*
132          * Expect the feeder to return one of following if type is
133          * EntryRequestType.DEFAULT:
134          *  a) not_found if the requested VLSN is too low
135          *  b) the requested VLSN if the requested VLSN is found
136          *  c) the alt match VLSN if the requested VLSN is too high
137          *
138          * If type is EntryRequestType.AVAILABLE:
139          *  a) the lowest available VLSN if the requested VLSN is too low
140          *  b) the requested VLSN if the requested VLSN is found
141          *  c) the highest available VLSN if the request VLSN is too high
142 
143          * If type is EntryRequestType.NOW:
144          *  a) always returns highest available VLSN
145          */
146         final Message message = protocol.read(namedChannel);
147         final VLSN vlsn;
148         if (message instanceof Entry) {
149             vlsn = ((Entry) message).getWireRecord().getVLSN();
150 
151             /* must be exact match for the default type */
152             if (type.equals(EntryRequestType.DEFAULT)) {
153                 assert (vlsn.equals(requestVLSN));
154             }
155 
156             /* dump traces */
157             if (vlsn.equals(requestVLSN)) {
158                 LoggerUtils.finest(logger, repImpl,
159                                    "Subscriber successfully requested VLSN " +
160                                    requestVLSN + " from feeder " +
161                                    namedChannel.getNameIdPair() +
162                                    ", request type: " + type);
163             }
164 
165             if (vlsn.compareTo(requestVLSN) < 0) {
166                 LoggerUtils.finest(logger, repImpl,
167                                    "Requested VLSN " + requestVLSN +
168                                    " is not available from feeder " +
169                                    namedChannel.getNameIdPair() +
170                                    " instead, start stream from a lowest " +
171                                    "available VLSN " + vlsn +
172                                    ", request type: " + type);
173             }
174 
175             if (vlsn.compareTo(requestVLSN) > 0) {
176                 if (type.equals(EntryRequestType.NOW)) {
177                     LoggerUtils.finest(logger, repImpl,
178                                        "Stream from highest available vlsn " +
179                                        "from feeder " +
180                                        namedChannel.getNameIdPair() + ":" +
181                                        vlsn + ", request type: " + type);
182                 } else {
183                     LoggerUtils.finest(logger, repImpl,
184                                        "Requested VLSN " + requestVLSN +
185                                        " is not available from feeder " +
186                                        namedChannel.getNameIdPair() +
187                                        " instead, start stream from a highest" +
188                                        " available VLSN " + vlsn +
189                                        ", request type: " + type);
190                 }
191             }
192 
193         } else if (message instanceof AlternateMatchpoint) {
194             /* now and available type should not see alter match point */
195             if (type.equals(EntryRequestType.NOW) ||
196                 type.equals(EntryRequestType.AVAILABLE)) {
197                 String msg = "Receive unexpected response " + message +
198                              "from feeder " + namedChannel.getNameIdPair() +
199                              ", request type: " + type;
200                 LoggerUtils.warning(logger, repImpl, msg);
201                 throw new IllegalStateException(msg);
202             }
203 
204             vlsn = ((AlternateMatchpoint) message).getAlternateWireRecord()
205                                                    .getVLSN();
206             /* must be an earlier VLSN */
207             assert (vlsn.compareTo(requestVLSN) < 0);
208             LoggerUtils.finest(logger, repImpl,
209                                "Feeder " + namedChannel.getNameIdPair() +
210                                " returns a valid start VLSN" + vlsn +
211                                " but earlier than requested one " +
212                                requestVLSN + ", request type: " + type);
213 
214         } else  if (message instanceof EntryNotFound) {
215             /* now and available type should not see not found */
216             if (type.equals(EntryRequestType.NOW) ||
217                 type.equals(EntryRequestType.AVAILABLE)) {
218                 /*
219                  * even for a brand new environment, the VLSN range at feeder
220                  * is not empty so we should not see entry not found
221                  */
222                 String msg = "Receive unexpected response " + message +
223                              "from feeder " + namedChannel.getNameIdPair() +
224                              ", request type: " + type;
225                 LoggerUtils.warning(logger, repImpl, msg);
226                 throw new IllegalStateException(msg);
227             }
228 
229             vlsn = VLSN.NULL_VLSN;
230             LoggerUtils.finest(logger, repImpl,
231                                "Feeder " + namedChannel.getNameIdPair() +
232                                " is unable to service the request vlsn " +
233                                requestVLSN + ", request type: " + type);
234         } else {
235             /* unexpected response from feeder */
236             String msg = "Receive unexpected response " + message +
237                          "from feeder " + namedChannel.getNameIdPair() +
238                          ", request type: " + type;
239             LoggerUtils.warning(logger, repImpl, msg);
240             throw new IllegalStateException(msg);
241         }
242 
243         return vlsn;
244     }
