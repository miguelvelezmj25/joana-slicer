408         private final FeederFilter feederFilter;
409 
410         StartStream(VLSN startVLSN) {
411             super(startVLSN);
412             feederFilter = null;

420         public StartStream(ByteBuffer buffer) {
421             super(buffer);
422 
423             /* Feeder filtering not supported before protocol version 6 */
424             if (getVersion() < VERSION_6) {
425                 feederFilter = null;
426                 return;
427             }
428 
429             final int length = LogUtils.readInt(buffer);
430             if (length == 0) {
431                 /* no filter is provided by client */
432                 feederFilter = null;
433                 return;
434             }
435 
436             /* reconstruct filter from buffer */
437             final byte filterBytes[] =
438                 LogUtils.readBytesNoLength(buffer, length);
439             final ByteArrayInputStream bais =
440                 new ByteArrayInputStream(filterBytes);
441             ObjectInputStream ois = null;
442             try {
443                 ois = new ObjectInputStream(bais);
444                 feederFilter = (FeederFilter) ois.readObject();
445             } catch (ClassNotFoundException | IOException e) {
446                 logger.warning(e.getLocalizedMessage());
447                 throw new IllegalStateException(e);
448             } finally {
449                 if (ois != null) {
450                     try {
451                         ois.close();
452                     } catch (IOException e) {
453                         logger.finest("exception raised when closing the " +
454                                       "object input stream object " +
455                                       e.getLocalizedMessage());
456                     }
457                 }
458             }
459         }
460 
461         public FeederFilter getFeederFilter() {
462             return feederFilter;
463         }
464 
465         @Override
466         public ByteBuffer wireFormat() {
467             /* Feeder filtering not supported before protocol version 6 */
468             if (getVersion() < VERSION_6) {
469                 return super.wireFormat();
470             }
471 
472             final int feederBufferSize;
473             final byte[] filterBytes;
474 
475             if (feederFilter != null) {
476                 final ByteArrayOutputStream baos = new ByteArrayOutputStream();
477                 ObjectOutputStream oos = null;
478                 try {
479                     oos = new ObjectOutputStream(baos);
480                     oos.writeObject(feederFilter);
481                     oos.flush();
482                 } catch (IOException e) {
483                     logger.warning(e.getLocalizedMessage());
484                     throw new IllegalStateException(e);
485                 } finally {
486                     if (oos != null) {
487                         try {
488                             oos.close();
489                         } catch (IOException e) {
490                             logger.finest("exception raised when closing the " +
491                                           "object output stream object " +
492                                           e.getLocalizedMessage());
493                         }
494                     }
495                 }
496                 filterBytes = baos.toByteArray();
497                 feederBufferSize = filterBytes.length;
498             } else {
499                 filterBytes = null;
500                 feederBufferSize = 0;
501             }
502 
503             /* build message buffer */
504             final int bodySize = wireFormatSize() + 4 + feederBufferSize;
505             final ByteBuffer messageBuffer =
506                 allocateInitializedBuffer(bodySize);
507             /* write 8 bytes of VLSN */
508             LogUtils.writeLong(messageBuffer, vlsn.getSequence());
509             /* write 4 bytes of feeder buf size */
510             LogUtils.writeInt(messageBuffer, feederBufferSize);
511             /* write feeder buffer */
512             if (feederBufferSize > 0) {
513                 LogUtils.writeBytesNoLength(messageBuffer, filterBytes);
514             }
515             messageBuffer.flip();
516             return messageBuffer;
517         }
