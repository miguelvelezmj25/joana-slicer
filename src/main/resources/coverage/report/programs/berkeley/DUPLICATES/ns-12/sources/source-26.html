90     protected boolean processEntry(ByteBuffer entryBuffer) {
91         byte currentType = currentEntryHeader.getType();
92         LogEntryType type = LogEntryType.findType(currentType);
93         LogEntry entry = null;
94 
95         if (needMatchEntry()) {
96             entry = type.getSharedLogEntry();
97             entry.readEntry(envImpl, currentEntryHeader, entryBuffer);
98 
99             if (!matchEntry(entry)) {
100                 return true;
101             }
102         }
103 
104         int itemSize = currentEntryHeader.getItemSize();
105         int headerSize = currentEntryHeader.getSize();
106 
107         /*
108          * Record various stats based on the entry header.
109          *
110          * Get the info object for it, if this is the first time it's seen,
111          * create an info object and insert it.
112          */
113         EntryInfo info = entryInfoMap.get(type);
114         if (info == null) {
115             info = new EntryInfo();
116             entryInfoMap.put(type, info);
117         }
118 
119         /* Update counts. */
120         info.count++;
121         totalCount++;
122         if (currentEntryHeader.getProvisional() == Provisional.YES) {
123             info.provisionalCount++;
124         }
125         int size = itemSize + headerSize;
126         info.totalBytes += size;
127         info.headerBytes += headerSize;
128         totalLogBytes += size;
129 
130         if ((info.minBytes == 0) || (info.minBytes > size)) {
131             info.minBytes = size;
132         }
133         if (info.maxBytes < size) {
134             info.maxBytes = size;
135         }
136 
137         if (verbose) {
138             if (firstLsnRead == DbLsn.NULL_LSN) {
139                 firstLsnRead = getLastLsn();
140             }
141 
142             if (currentType == LogEntryType.LOG_CKPT_END.getTypeNum()) {
143                 /* Start counting a new interval. */
144                 ckptCounter.endCkptLsn = getLastLsn();
145                 ckptCounter = new CheckpointCounter();
146                 ckptList.add(ckptCounter);
147             } else {
148                 ckptCounter.increment(this, currentType);
149             }
150         }
151 
152         if (type.isUserLNType()) {
153             /* Read the entry into the ByteBuffer. */
154             if (entry == null) {
155                 entry = type.getSharedLogEntry();
156                 entry.readEntry(envImpl, currentEntryHeader, entryBuffer);
157             }
158             LNLogEntry<?> lnEntry = (LNLogEntry<?>) entry;
159 
160             /*
161              * The getUnconvertedXxx methods are used because we don't have a
162              * DatabaseImpl for calling LNLogEntry.postFetchInit, and we can
163              * tolerate statistics that use the old duplicates format.
164              */
165             int keyLen = lnEntry.getUnconvertedKeyLength();
166 
167             realTotalKeyBytes += keyLen;
168             realTotalKeyCount += 1;
169 
170             if ((realMinKeyBytes == 0) || (realMinKeyBytes > keyLen)) {
171                 realMinKeyBytes = keyLen;
172             }
173             if (realMaxKeyBytes < keyLen) {
174                 realMaxKeyBytes = keyLen;
175             }
176 
177             if (!entry.isDeleted()) {
178                 int dataLen = lnEntry.getUnconvertedDataLength();
179 
180                 realTotalDataBytes += dataLen;
181                 realTotalDataCount += 1;
182 
183                 if ((realMinDataBytes == 0) || (realMinDataBytes > dataLen)) {
184                     realMinDataBytes = dataLen;
185                 }
186                 if (realMaxDataBytes < dataLen) {
187                     realMaxDataBytes = dataLen;
188                 }
189             }
190         }
191 
192         /*
193          * If we have not read the entry, skip over it.
194          */
195         if (entry == null) {
196             int nextEntryPosition = entryBuffer.position() + itemSize;
197             entryBuffer.position(nextEntryPosition);
198         }
199         return true;
200     }

418         System.out.println("\nPer checkpoint interval info:");
419 
420         /*
421          * Print out checkpoint interval info.
422          * If the log looks like this:
423          *
424          * start of log
425          * ckpt1 start
426          * ckpt1 end
427          * ckpt2 start
428          * ckpt2 end
429          * end of log
430          *
431          * There are 3 ckpt intervals
432          * start of log->ckpt1 end
433          * ckpt1 end -> ckpt2 end
434          * ckpt2 end -> end of log
435          */
436         System.out.println
437             (f.format("lnTxn") +
438              f.format("ln") +
439              f.format("mapLNTxn") +
440              f.format("mapLN") +
441              f.format("end to end") +  // ckpt n-1 end -> ckpt n end
442              f.format("end to start") +// ckpt n-1 end -> ckpt n start
443              f.format("start to end") +// ckpt n start -> ckpt n end
444              f.format("maxLNReplay") +
445              f.format("ckptEnd"));
446 
447         long logFileMax =
448             envImpl.getConfigManager().getLong(EnvironmentParams.LOG_FILE_MAX);
449 
450         Iterator<CheckpointCounter> iter = ckptList.iterator();
451         CheckpointCounter prevCounter = null;
452         while (iter.hasNext()) {
453             CheckpointCounter c = iter.next();
454             StringBuilder sb = new StringBuilder();
455 
456             /* Entry type counts. */
457             int maxTxnLNs = c.preStartLNTxnCount + c.postStartLNTxnCount;
458             sb.append(f.format(maxTxnLNs));
459             int maxLNs = c.preStartLNCount + c.postStartLNCount;
460             sb.append(f.format(maxLNs));
461             sb.append(f.format(c.preStartMapLNTxnCount +

463             sb.append(f.format(c.preStartMapLNCount +
464                                c.postStartMapLNCount));
465 
466             /* Checkpoint interval distance. */
467             long end = (c.endCkptLsn == DbLsn.NULL_LSN) ?
468                 getLastLsn() :
469                 c.endCkptLsn;
470             long endToEndDistance = 0;
471 
472             FileManager fileMgr = envImpl.getFileManager();
473             if (prevCounter == null) {
474                 endToEndDistance = DbLsn.getWithCleaningDistance(
475                     end, firstLsnRead, logFileMax, fileMgr);
476             } else {
477                 endToEndDistance = DbLsn.getWithCleaningDistance(
478                     end, prevCounter.endCkptLsn, logFileMax, fileMgr);
479             }
480             sb.append(f.format(endToEndDistance));
481 
482             /*
483              * Interval between last checkpoint end and this checkpoint start.
484              */

486                 c.startCkptLsn;
487             long endToStartDistance = 0;
488 
489             if (prevCounter == null) {
490                 endToStartDistance = DbLsn.getWithCleaningDistance(

523             } else {
524                 sb.append("   ").append(DbLsn.getNoFormatString(c.endCkptLsn));
525             }
526

528             prevCounter = c;
529         }

569     static class CheckpointCounter {
570         public long startCkptLsn = DbLsn.NULL_LSN;
571         public long endCkptLsn = DbLsn.NULL_LSN;
572         public int preStartLNTxnCount;
573         public int preStartLNCount;
574         public int preStartMapLNTxnCount;
575         public int preStartMapLNCount;
576         public int postStartLNTxnCount;
577         public int postStartLNCount;
578         public int postStartMapLNTxnCount;
579         public int postStartMapLNCount;
580 
581         public void increment(FileReader reader,  byte currentEntryTypeNum) {
582             LogEntryType entryType =
583                 LogEntryType.findType(currentEntryTypeNum);
584 
585             if (entryType == LogEntryType.LOG_CKPT_START) {
586                 startCkptLsn = reader.getLastLsn();
587             } else if (entryType.isUserLNType()) {
588                 if (entryType.isTransactional()) {
589                     if (startCkptLsn == DbLsn.NULL_LSN) {
590                         preStartLNTxnCount++;
591                     } else {
592                         postStartLNTxnCount++;
593                     }
594                 } else {
595                     if (startCkptLsn == DbLsn.NULL_LSN) {
596                         preStartLNCount++;
597                     } else {
598                         postStartLNCount++;
599                     }
600                 }
601             } else if (entryType == LogEntryType.LOG_MAPLN) {
602                 if (startCkptLsn == DbLsn.NULL_LSN) {
603                     preStartMapLNCount++;
604                 } else {
605                     postStartMapLNCount++;
606                 }
607             }
608         }
