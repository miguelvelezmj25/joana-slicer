22 /**
23  * Encapsulates the handling of timeouts: ENV_SETUP_TIMEOUT and
24  * ENV_UNKNOWN_STATE_TIMEOUT, used when a replicated environment handle is
25  * opened and a node joins the group.
26  * <p>
27  * There are three timeouts that are relevant at the time a Replica joins a
28  * group. They are listed below in the order in which each is applied.
29  *
30  * 1) The ENV_UNKNOWN_STATE_TIMEOUT which is basically an election timeout. If
31  * set and an election is not concluded in this time period, the environment
32  * handle is opened in the unknown state.
33  *
34  * 2) The ENV_SETUP_TIMEOUT. This timeout determines the maximum amount of time
35  * allowed to hold an election and sync up with a master if the joins as a
36  * replica.
37  *
38  * 3) The consistency timeout as determined by the consistency policy in the
39  * event that the node joins as a replica.
40  *
41  * The first two timeouts are managed by this class. RepNode.joinGroup uses the
42  * timeouts supplied by the getTimeout() method to wait for each timeout if
43  * both are specified.
44  *
45  * joinGroup first waits up to the unknown state timeout for an election to be
46  * concluded. If the node is not in the unknown state at the timeout,
47  * it advances to the env setup timeout by invoking setSetupTimeout() and
48  * proceeds to wait up to this timeout for the syncup activity to complete.
49  *
50  */
51 class JoinGroupTimeouts {
52 
53     /*
54      * The timeout associated with opening a handle in the unknown state. It's
55      * max int if the handle should not be opened in that state.
56      */
57     private final int unknownStateTimeout;
58 
59     /* The timeout associated with the total setup of the handle. */
60     private final int setupTimeout;
61 
62     /*
63      * The timeout that's currently active, it can be either of the two values
64      * values above.
65      */
66     private int timeout;
67 
68     /* used as the basis for determining time limits from timeouts. */
69     private final long start = System.currentTimeMillis();
70 
71     JoinGroupTimeouts(DbConfigManager configManager) {
72        setupTimeout = configManager.getDuration(ENV_SETUP_TIMEOUT);
73 
74         if (configManager.getDuration(ENV_UNKNOWN_STATE_TIMEOUT) == 0) {
75             /* Support deprecated usage. */
76             final boolean allowUnknownStateEnv =
77                     configManager.getBoolean(ALLOW_UNKNOWN_STATE_ENV_OPEN);
78             unknownStateTimeout =
79                  (allowUnknownStateEnv ? setupTimeout : Integer.MAX_VALUE);
80         } else {
81             unknownStateTimeout = configManager.
82                     getDuration(ENV_UNKNOWN_STATE_TIMEOUT);
83             if (unknownStateTimeout > setupTimeout) {
84                 String message = String.format(
85                   " The timeout ENV_UNKNOWN_STATE_TIMEOUT(%,d ms)" +
86                   " exceeds the timeout ENV_SETUP_TIMEOUT(%,d ms)",
87                   unknownStateTimeout,
88                   setupTimeout);
89 
90                 throw new IllegalArgumentException(message);
91             }
92         }
93         /* Set the first timeout. */
94         timeout = Math.min(unknownStateTimeout, setupTimeout);
95     }
96 
97     /**
98      * Returns the currently active timeout, adjusted for the time that has
99      * already elapsed.
100      */
101     int getTimeout() {
102         return Math.max(timeout - (int)(System.currentTimeMillis() - start),
103                         0);
104     }
105 
106     /**
107      * Returns the setup timeout
108      */
109     public int getSetupTimeout() {
110        return setupTimeout;
111     }
112 
113     /**
114      * Returns true if the currently active timeout is the one for the
115      * transition out of the unknown state.
116      */
117     boolean timeoutIsForUnknownState() {
118         return timeout == unknownStateTimeout;
119     }
120 
121     /**
122      * Set the currently active timeout to be the env setup timeout.
123      */
124     void setSetupTimeout() {
125         timeout = setupTimeout;
126     }
127 }
