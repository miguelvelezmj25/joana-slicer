540     /* Get results of the invoked MBean operation. */
541     private synchronized Map<String, String> generateStats(ObjectName name) {
542         Map<String, String> map = new LinkedHashMap<String, String>();
543         try {
544             if (connection != null &&
545                 connection.queryNames(name, null) != null) {
546                 String status = (String)
547                     connection.invoke(name, opName, envStatParams, signature);
548                 StringTokenizer st1 = new StringTokenizer(status, "\n");
549                 String title = null;
550                 while (st1.hasMoreTokens()) {
551                     String expression = st1.nextToken();
552                     if (expression != null) {
553                         if (expression.indexOf("=") < 0) {
554                             StringTokenizer st2 =
555                                 new StringTokenizer(expression, ":");
556                             title = st2.nextToken();
557                             map.put(title, "");
558                         } else {
559                             StringTokenizer st2 =
560                                 new StringTokenizer(expression, "=");
561                             String stats = "    " + st2.nextToken();
562                             String value = st2.nextToken().trim();
563                             map.put(stats + ":" + title, value);
564                         }
565                     }
566                 }
567             }
568         } catch (javax.management.InstanceNotFoundException e) {
569 
570             /*
571              * If the connection is broken while it is trying to invoke, close
572              * and release all resources.
573              */
574             forceClose();
575         } catch (Exception e) {
576             e.printStackTrace();
577         }
578 
579         return map;
580     }

591     private synchronized ArrayList<ObjectName> getBeansNames() {
592         if (connection == null) {
593             return null;
594         }
595 
596         ArrayList<ObjectName> names = null;
597         try {
598             ObjectName name = new ObjectName(mBeanNamePrefix);
599 
600             if (connection.queryNames(name, null).size() != 0) {
601                 names = new ArrayList<ObjectName>
602                     (connection.queryNames(name, null));
603             }
604         } catch (java.rmi.RemoteException e) {
605 
606             /*
607              * Because the interval of plugin and the logging thread is not the
608              * same, sometimes logging thread would try to invoke a non-active
609              * connection and throws out RemoteException.  We need to stop the
610              * thread and release the resources.
611              */
612             forceClose();
613         } catch (Exception e) {
614             e.printStackTrace();
615         }
616 
617         return names;

681     /* Make the stats output in CSV format. */
682     private void getCSVOutput(StringBuilder buffer,
683                               Map<String, String> map,
684                               boolean init,
685                               ObjectName name) {
686         LogObject item = logMap.get(name);
687 
688         if (map == null || map.size() == 0) {
689             return;
690         }
691 
692         /* Before write to log, remove those unlog stats. */
693         if (item != null) {
694             Object[] keys = map.keySet().toArray();
695             for (String title : item.getTurnOff()) {
696                 for (Object key : keys) {
697                     if (key.toString().contains(title)) {
698                         map.remove(key.toString());
699                     }
700                 }
701             }
702             emptyArray(keys);
703             keys = null;
704         }
705         item = null;
706 
707         if (!init) {
708             buffer.append(System.currentTimeMillis());
709         } else {
710             buffer.append("TIME");
711         }
712         for (int i = 0; i < statsTitles.length; i++) {
713             map.remove(statsTitles[i]);
714         }
715         if (map.size() > 0) {
716             buffer.append(",");
717             int count = 1;
718             for (Map.Entry<String, String> entry : map.entrySet()) {
719                 String title = entry.getKey().substring
720                     (0, entry.getKey().indexOf(":")).trim();
721                 if (!init) {
722                     title = "\"" + entry.getValue() + "\"";
723                 }
724                 if (count < map.size())
725                     title = title + ",";
726                 buffer.append(title);
727                 count++;
728             }
729         }
730     }
