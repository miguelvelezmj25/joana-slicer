133             String fieldName = clsMeta.getPrimaryKey().getName();
134             FieldInfo field = FieldInfo.getField(nonKeyFields, fieldName);

149             for (SecondaryKeyMetadata secKeyMeta :

151                 String fieldName = secKeyMeta.getName();
152                 FieldInfo field = FieldInfo.getField(nonKeyFields, fieldName);
153                 if (field == null) {

157                 }

167                     }

170                         throw new IllegalArgumentException
171                             ("ONE_TO_ONE and MANY_TO_ONE keys must not" +
172                              " have an array or Collection type: " +
173                              getClassName() + '.' + fieldName);
174                     }
175                 }

177                     secKeyMeta.getDeleteAction() == DeleteAction.NULLIFY) {
178                     throw new IllegalArgumentException
179                         ("NULLIFY may not be used with primitive fields: " +

181                 }

183                 secKeyFields.add(field);

634                  format = format.getComplexSuper()) {
635                 nFields += format.getNFields();
636                 depth += 1;
637             }
638             ComplexFormat[] hierarchy = new ComplexFormat[depth];
639             int level = depth;
640             for (ComplexFormat format = this;
641                  format != null;
642                  format = format.getComplexSuper()) {
643                 level -= 1;
644                 hierarchy[level] = format;
645             }
646             assert level == 0;
647 
648             /* Populate levels and fields in parallel. */
649             levels = new int[nFields];
650             fields = new FieldInfo[nFields];
651             int index = 0;
652 
653             /*
654              * The primary key is the first field read/written.  We use the
655              * first primary key field encountered going from this class upward
656              * in the class hierarchy.
657              */
658             if (getEntityFormat() != null) {
659                 for (level = depth - 1; level >= 0; level -= 1) {
660                     ComplexFormat format = hierarchy[level];
661                     if (format.priKeyField != null) {
662                         levels[index] = level;
663                         fields[index] = format.priKeyField;
664                         index += 1;
665                         break;
666                     }
667                 }
668                 assert index == 1;
669             }
670 
671             /*
672              * Secondary key fields are read/written next, from the highest
673              * base class downward.
674              */
675             for (level = 0; level < depth; level += 1) {
676                 ComplexFormat format = hierarchy[level];
677                 for (FieldInfo field : format.secKeyFields) {
678                     levels[index] = level;
679                     fields[index] = field;
680                     index += 1;
681                 }
682             }
683 
684             /*
685              * Other fields are read/written last, from the highest base class
686              * downward.
687              */
688             for (level = 0; level < depth; level += 1) {
689                 ComplexFormat format = hierarchy[level];
690                 for (FieldInfo field : format.nonKeyFields) {
691                     levels[index] = level;
692                     fields[index] = field;
693                     index += 1;
694                 }
695             }
696 
697             /* We're finished -- update the volatile fields for next time. */
698             assert index == fields.length;
699             rawInputFields = fields;
700             rawInputLevels = levels;
701             rawInputDepth = depth;
702         }
703 
704         /*
705          * Create an objects array that is parallel to the fields and levels
706          * arrays, but contains the RawObject for each slot from which the
707          * field value can be retrieved.  The predetermined level for each
708          * field determines which RawObject in the instance hierarchy to use.
709          */
710         RawObject[] objectsByLevel = new RawObject[depth];
711         int level = depth;
712         for (RawObject raw = rawObject; raw != null; raw = raw.getSuper()) {
713             if (level == 0) {
714                 throw new IllegalArgumentException
715                     ("RawObject has too many superclasses: " +
716                      rawObject.getType().getClassName());
717             }
718             level -= 1;
719             objectsByLevel[level] = raw;
720         }
721         if (level > 0) {
722             throw new IllegalArgumentException
723                 ("RawObject has too few superclasses: " +
724                  rawObject.getType().getClassName());
725         }
726         assert level == 0;
727         RawObject[] objects = new RawObject[fields.length];
728         for (int i = 0; i < objects.length; i += 1) {
729             objects[i] = objectsByLevel[levels[i]];
730         }
731 
732         /* Create the persistent object and convert all RawObject fields. */
733         EntityInput in = new RawComplexInput
734             (catalog, rawAccess, converted, fields, objects);
735         Object o = newInstance(in, rawAccess);
736         converted.put(rawObject, o);
737         if (getEntityFormat() != null) {
738             readPriKey(o, in, rawAccess);
739         }
740         return readObject(o, in, rawAccess);
741     }

919     @Override
920     boolean evolve(Format newFormatParam, Evolver evolver) {
921 
922         /* Disallow evolution to a non-complex format. */
923         if (!(newFormatParam instanceof ComplexFormat)) {
924             evolver.addMissingMutation
925                 (this, newFormatParam,
926                  "Converter is required when a complex type is changed " +
927                  "to a simple type or enum type");
928             return false;
929         }
930         ComplexFormat newFormat = (ComplexFormat) newFormatParam;
931         Mutations mutations = evolver.getMutations();
932         boolean thisChanged = false;
933         boolean hierarchyChanged = false;
934         Map<String, String> allKeyNameMap = new HashMap<String, String>();
935 
936         /* The Evolver has already ensured that entities evolve to entities. */
937         assert isEntity() == newFormat.isEntity();
938         assert isEntity() == (entityMeta != null);
939         assert newFormat.isEntity() == (newFormat.entityMeta != null);
940 
941         /*
942          * Keep track of the old and new entity class names for use in deleting
943          * and renaming secondary keys below.  If the oldEntityClass is
944          * non-null this also signifies an entity class or subclass.  Note that
945          * getEntityFormat cannot be called on a newly created format during
946          * evolution because its super format property is not yet initialized.
947          * [#16253]
948          */
949         String oldEntityClass;
950         String newEntityClass;
951         if (isEntity()) {
952             oldEntityClass = getClassName();
953             newEntityClass = newFormat.getClassName();
954         } else {
955             oldEntityClass = null;
956             newEntityClass = null;
957         }
958 
959         /*
960          * Evolve all superclass formats, even when a deleted class appears in
961          * the hierarchy.  This ensures that the super format's
962          * getLatestVersion/getComplexLatest method can be used accurately
963          * below.
964          */
965         for (ComplexFormat oldSuper = getComplexSuper();
966              oldSuper != null;
967              oldSuper = oldSuper.getComplexSuper()) {
968             Converter converter = mutations.getConverter
969                 (oldSuper.getClassName(), oldSuper.getVersion(), null);
970             if (converter != null) {
971                 evolver.addMissingMutation
972                     (this, newFormatParam,
973                      "Converter is required for this subclass when a " +
974                      "Converter appears on its superclass: " + converter);
975                 return false;
976             }
977             if (!evolver.evolveFormat(oldSuper)) {
978                 return false;
979             }
980             if (!oldSuper.isCurrentVersion()) {
981                 if (oldSuper.isDeleted()) {
982                     if (!oldSuper.evolveDeletedClass(evolver)) {
983                         return false;
984                     }
985                 }
986                 if (oldSuper.oldToNewKeyMap != null) {
987                     allKeyNameMap.putAll(oldSuper.oldToNewKeyMap);
988                 }
989                 hierarchyChanged = true;
990             }
991         }
992 
993         /*
994          * Compare the old and new class hierarhies and decide whether each
995          * change is allowed or not:
996          * + Old deleted and removed superclass -- allowed
997          * + Old empty and removed superclass -- allowed
998          * + Old non-empty and removed superclass -- not allowed
999          * + Old superclass repositioned in the hierarchy -- not allowed
1000          * + New inserted superclass -- allowed
1001          */
1002         Class newFormatCls = newFormat.getExistingType();
1003         Class newSuper = newFormatCls;
1004         List<Integer> newLevels = new ArrayList<Integer>();
1005         int newLevel = 0;
1006         newLevels.add(newLevel);
1007 
1008         /*
1009          * When this format has a new superclass, we treat it as a change to
1010          * this format as well as to the superclass hierarchy.
1011          */
1012         if (getSuperFormat() == null) {
1013             if (newFormatCls.getSuperclass() != Object.class) {
1014                 thisChanged = true;
1015                 hierarchyChanged = true;
1016             }
1017         } else {
1018             if (!getSuperFormat().getLatestVersion().getClassName().equals
1019                     (newFormatCls.getSuperclass().getName())) {
1020                 thisChanged = true;
1021                 hierarchyChanged = true;
1022             }
1023         }
1024 
1025         for (ComplexFormat oldSuper = getComplexSuper();
1026              oldSuper != null;
1027              oldSuper = oldSuper.getComplexSuper()) {
1028 
1029             /* Find the matching superclass in the new hierarchy. */
1030             String oldSuperName = oldSuper.getLatestVersion().getClassName();
1031             Class foundNewSuper = null;
1032             int tryNewLevel = newLevel;
1033             for (Class newSuper2 = newSuper.getSuperclass();
1034                  newSuper2 != Object.class;
1035                  newSuper2 = newSuper2.getSuperclass()) {
1036                 tryNewLevel += 1;
1037                 if (oldSuperName.equals(newSuper2.getName())) {
1038                     foundNewSuper = newSuper2;
1039                     newLevel = tryNewLevel;
1040                     if (oldSuper.isEntity()) {
1041                         assert oldEntityClass == null;
1042                         assert newEntityClass == null;
1043                         oldEntityClass = oldSuper.getClassName();
1044                         newEntityClass = foundNewSuper.getName();
1045                     }
1046                     break;
1047                 }
1048             }
1049 
1050             if (foundNewSuper != null) {
1051 
1052                 /*
1053                  * We found the old superclass in the new hierarchy.  Traverse
1054                  * through the superclass formats that were skipped over above
1055                  * when finding it.
1056                  */
1057                 for (Class newSuper2 = newSuper.getSuperclass();
1058                      newSuper2 != foundNewSuper;
1059                      newSuper2 = newSuper2.getSuperclass()) {
1060 
1061                     /*
1062                      * The class hierarchy changed -- a new class was inserted.
1063                      */
1064                     hierarchyChanged = true;
1065 
1066                     /*
1067                      * Check that the new formats skipped over above are not at
1068                      * a different position in the old hierarchy.
1069                      */
1070                     for (ComplexFormat oldSuper2 = oldSuper.getComplexSuper();
1071                          oldSuper2 != null;
1072                          oldSuper2 = oldSuper2.getComplexSuper()) {
1073                         String oldSuper2Name =
1074                             oldSuper2.getLatestVersion().getClassName();
1075                         if (oldSuper2Name.equals(newSuper2.getName())) {
1076                             evolver.addMissingMutation
1077                                 (this, newFormatParam,
1078                                  "Class Converter is required when a " +
1079                                  "superclass is moved in the class " +
1080                                  "hierarchy: " + newSuper2.getName());
1081                             return false;
1082                         }
1083                     }
1084                 }
1085                 newSuper = foundNewSuper;
1086                 newLevels.add(newLevel);
1087             } else {
1088 
1089                 /*
1090                  * We did not find the old superclass in the new hierarchy.
1091                  * The class hierarchy changed, since an old class no longer
1092                  * appears.
1093                  */
1094                 hierarchyChanged = true;
1095 
1096                 /* Check that the old class can be safely removed. */
1097                 if (!oldSuper.isDeleted()) {
1098                     ComplexFormat oldSuperLatest =
1099                         oldSuper.getComplexLatest();
1100                     if (oldSuperLatest.getNFields() != 0) {
1101                         evolver.addMissingMutation
1102                             (this, newFormatParam,
1103                              "When a superclass is removed from the class " +
1104                              "hierarchy, the superclass or all of its " +
1105                              "persistent fields must be deleted with a " +
1106                              "Deleter: " +
1107                              oldSuperLatest.getClassName());
1108                         return false;
1109                     }
1110                 }
1111 
1112                 if (oldEntityClass != null && isCurrentVersion()) {
1113                     Map<String, SecondaryKeyMetadata> secKeys =
1114                         oldSuper.clsMeta.getSecondaryKeys();
1115                     for (FieldInfo field : oldSuper.secKeyFields) {
1116                         SecondaryKeyMetadata meta =
1117                             getSecondaryKeyMetadataByFieldName
1118                                 (secKeys, field.getName());
1119                         assert meta != null;
1120                         allKeyNameMap.put(meta.getKeyName(), null);
1121                     }
1122                 }
1123 
1124                 /*
1125                  * Add the DO_NOT_READ_ACCESSOR level to prevent an empty class
1126                  * (no persistent fields) from being read via the Accessor.
1127                  */
1128                 newLevels.add(EvolveReader.DO_NOT_READ_ACCESSOR);
1129             }
1130         }
1131 
1132         /* Make FieldReaders for this format if needed. */
1133         int result = evolveAllFields(newFormat, evolver);
1134         if (result == Evolver.EVOLVE_FAILURE) {
1135             return false;
1136         }
1137         if (result == Evolver.EVOLVE_NEEDED) {
1138             thisChanged = true;
1139         }
1140         if (oldToNewKeyMap != null) {
1141             allKeyNameMap.putAll(oldToNewKeyMap);
1142         }
1143 
1144         /* Require new version number if this class was changed. */
1145         if (thisChanged &&
1146             !evolver.checkUpdatedVersion
1147                 ("Changes to the fields or superclass were detected", this,
1148                  newFormat)) {
1149             return false;
1150         }
1151 
1152         /* Rename and delete the secondary databases. */
1153         if (allKeyNameMap.size() > 0 &&
1154             oldEntityClass != null &&
1155             newEntityClass != null &&
1156             isCurrentVersion()) {
1157             for (Map.Entry<String, String> entry : allKeyNameMap.entrySet()) {
1158                 String oldKeyName = entry.getKey();
1159                 String newKeyName = entry.getValue();
1160                 if (newKeyName != null) {
1161                     evolver.renameSecondaryDatabase
1162                         (oldEntityClass, newEntityClass,
1163                          oldKeyName, newKeyName);
1164                 } else {
1165                     evolver.deleteSecondaryDatabase
1166                         (oldEntityClass, oldKeyName);
1167                 }
1168             }
1169         }
1170 
1171         /*
1172          * Use an EvolveReader if needed.
1173          *
1174          * We force evolution to occur if the old format did not use the new
1175          * String format.  We do not require the user to bump the version
1176          * number, since the format change is internal.  Note that we could
1177          * optimize by only forcing evolution if this format may contain
1178          * Strings. [#19247]
1179          */
1180         if (hierarchyChanged || thisChanged || !newStringFormat) {
1181             Reader reader = new EvolveReader(newLevels);
1182             evolver.useEvolvedFormat(this, reader, newFormat);
1183         } else {
1184             evolver.useOldFormat(this, newFormat);
1185         }
1186         return true;
1187     }

1416     /**

1421      * is needed, or FieldReader.EVOLVE_FAILURE if an evolution error occurs.
1422      */

1424                                         List<FieldInfo> newFields,
1425                                         boolean isOldSecKeyField,
1426                                         List<FieldInfo> otherNewFields,
1427                                         ComplexFormat newFormat,
1428                                         Evolver evolver) {

1431         boolean localEvolveNeeded = false;
1432         boolean readerNeeded = false;
1433         List<FieldReader> fieldReaders = new ArrayList<FieldReader>();
1434         FieldReader currentReader = null;
1435         int newFieldsMatched = 0;
1436 
1437         /*
1438          * Add FieldReaders to the list in old field storage order, since that
1439          * is the order in which field values must be read.
1440          */

1442         for (int oldFieldIndex = 0;
1443              oldFieldIndex < oldFields.size();
1444              oldFieldIndex += 1) {
1445 
1446             FieldInfo oldField = oldFields.get(oldFieldIndex);
1447             String oldName = oldField.getName();
1448             SecondaryKeyMetadata oldMeta = null;
1449             if (isOldSecKeyField) {
1450                 oldMeta = getSecondaryKeyMetadataByFieldName
1451                     (clsMeta.getSecondaryKeys(), oldName);
1452                 assert oldMeta != null;
1453             }
1454

1456             Renamer renamer = mutations.getRenamer
1457                 (getClassName(), getVersion(), oldName);

1459                 (getClassName(), getVersion(), oldName);

1461                 (getClassName(), getVersion(), oldName);

1463                 evolver.addInvalidMutation
1464                     (this, newFormat, deleter,
1465                      "Field Deleter is not allowed along with a Renamer or " +
1466                      "Converter for the same field: " + oldName);
1467                 evolveFailure = true;
1468                 continue fieldLoop;
1469             }
1470 
1471             /*
1472              * Match old and new field by name, taking into account the Renamer
1473              * mutation.  If the @SecondaryKey annotation was added or removed,
1474              * the field will have moved from one of the two field lists to the
1475              * other.
1476              */
1477             String newName = (renamer != null) ?
1478                 renamer.getNewName() : oldName;
1479             boolean nameChanged = false;
1480             if (!oldName.equals(newName)) {
1481                 if (newToOldFieldMap == null) {
1482                     newToOldFieldMap = new HashMap<String, String>();
1483                 }
1484                 newToOldFieldMap.put(newName, oldName);
1485                 nameChanged = true; 
1486             }
1487             int newFieldIndex = FieldInfo.getFieldIndex(newFields, newName);
1488             FieldInfo newField = null;
1489             boolean isNewSecKeyField = isOldSecKeyField;
1490             if (newFieldIndex >= 0) {
1491                 newField = newFields.get(newFieldIndex);
1492                 
1493                 /* 
1494                  * Change the key name in incorrectlyOrderedSecKeys of the new
1495                  * format.
1496                  */
1497                 if (nameChanged &&
1498                     newFormat.incorrectlyOrderedSecKeys != null && 
1499                     newFormat.incorrectlyOrderedSecKeys.remove(oldName)) {
1500                     newFormat.incorrectlyOrderedSecKeys.add(newName);
1501                 }
1502                 
1503                 /* 
1504                  * [#18961] If the order of the field has been changed, we will
1505                  * create a PlainFieldReader for it.
1506                  */
1507                 if (newFieldIndex != oldFieldIndex) {
1508                     localEvolveNeeded = true;
1509                     readerNeeded = true;
1510                 }
1511             } else {
1512                 newFieldIndex = FieldInfo.getFieldIndex
1513                     (otherNewFields, newName);
1514                 if (newFieldIndex >= 0) {
1515                     newField = otherNewFields.get(newFieldIndex);
1516                     isNewSecKeyField = !isOldSecKeyField;
1517                 }

1519                 readerNeeded = true;
1520                 
1521                 /* 
1522                  * Remove the key in incorrectlyOrderedSecKeys of the new
1523                  * format.
1524                  */

1526                     newFormat.incorrectlyOrderedSecKeys.remove(oldName);

1528             }
1529

1531             if (deleter != null) {
1532                 if (newField != null) {
1533                     evolver.addInvalidMutation
1534                         (this, newFormat, deleter,
1535                          "Field Deleter is not allowed when the persistent " +
1536                          "field is still present: " + oldName);
1537                     evolveFailure = true;
1538                 }

1540                 if (currentReader instanceof SkipFieldReader &&
1541                     currentReader.acceptField
1542                         (oldFieldIndex, newFieldIndex, isNewSecKeyField)) {

1546                         (oldFieldIndex, oldField);
1547                     fieldReaders.add(currentReader);
1548                     readerNeeded = true;

1550                 }

1553                         oldToNewKeyMap = new HashMap<String, String>();
1554                     }
1555                     oldToNewKeyMap.put(oldMeta.getKeyName(), null);
1556                 }

1558             } else {
1559                 if (newField == null) {
1560                     evolver.addMissingMutation
1561                         (this, newFormat,
1562                          "Field is not present or not persistent: " +
1563                          oldName);
1564                     evolveFailure = true;
1565                     continue fieldLoop;
1566                 }
1567             }
1568 
1569             /*
1570              * The old field corresponds to a known new field, and no Deleter
1571              * mutation applies.
1572              */

1574

1576             SecondaryKeyMetadata newMeta = null;
1577             if (isOldSecKeyField && isNewSecKeyField) {

1580                 assert newMeta != null;
1581

1585                     evolveFailure = true;
1586                     continue fieldLoop;
1587                 }

1591                  * use in renaming the secondary database and for key
1592                  * extraction.
1593                  */
1594                 String oldKeyName = oldMeta.getKeyName();
1595                 String newKeyName = newMeta.getKeyName();
1596                 if (!oldKeyName.equals(newKeyName)) {
1597                     if (oldToNewKeyMap == null) {
1598                         oldToNewKeyMap = new HashMap<String, String>();
1599                     }
1600                     oldToNewKeyMap.put(oldName, newName);
1601                     localEvolveNeeded = true;
1602                 }
1603             } else if (isOldSecKeyField && !isNewSecKeyField) {
1604                 if (oldToNewKeyMap == null) {
1605                     oldToNewKeyMap = new HashMap<String, String>();
1606                 }
1607                 oldToNewKeyMap.put(oldMeta.getKeyName(), null);
1608             }
1609

1611             if (converter != null) {
1612                 if (isOldSecKeyField) {
1613                     evolver.addInvalidMutation
1614                         (this, newFormat, converter,
1615                          "Field Converter is not allowed for secondary key " +
1616                          "fields: " + oldName);
1617                     evolveFailure = true;
1618                 } else {
1619                     currentReader = new ConvertFieldReader
1620                         (converter, oldFieldIndex, newFieldIndex,
1621                          isNewSecKeyField);
1622                     fieldReaders.add(currentReader);
1623                     readerNeeded = true;
1624                     localEvolveNeeded = true;
1625                 }
1626                 continue fieldLoop;
1627             }
1628

1642                     continue fieldLoop;

1645                     !evolver.isClassConverted(formatVersion)) {
1646                     allClassesConverted = false;
1647                 }

1649                     evolver.getSubclassFormats(formatVersion);

1654                             continue fieldLoop;

1657                             !evolver.isClassConverted(format2)) {
1658                             allClassesConverted = false;
1659                         }

1669              * needed.  If no widener is needed, fall through and apply a

1676                 !oldLatestFormat.isDeleted()) {

1686                      isNewSecKeyField);

1694                     !oldLatestFormat.isPrimitive() &&
1695                     !oldLatestFormat.isDeleted() &&
1696                     !evolver.isClassConverted(oldLatestFormat)) {

1700                         refWidened = true;
1701                     }
1702                 }
1703                 if (refWidened) {
1704                     /* A reference type has been widened.  Fall through. */
1705                     localEvolveNeeded = true;
1706                 } else {
1707                     /* Types are not compatible. */
1708                     evolver.addMissingMutation
1709                         (this, newFormat,
1710                          "Old field type: " + oldLatestFormat.getClassName() +
1711                          " is not compatible with the new type: " +
1712                          newFieldFormat.getClassName() +
1713                          " for field: " + oldName);

1715                     continue fieldLoop;
1716                 }
1717             }

1723              */
1724             if (currentReader instanceof PlainFieldReader &&
1725                 currentReader.acceptField
1726                     (oldFieldIndex, newFieldIndex, isNewSecKeyField)) {

1730                     (oldFieldIndex, newFieldIndex, isNewSecKeyField);
