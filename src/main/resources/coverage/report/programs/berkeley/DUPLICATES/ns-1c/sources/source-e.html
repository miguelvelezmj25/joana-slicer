1016     /**
1017      * Transfer the master role to one of the specified replicas.
1018      * <p>
1019      * We delegate most of the real work to an instance of the {@link
1020      * MasterTransfer} class.  Here, after some simple initial validity
1021      * checking, we're concerned with coordinating the potential for multiple
1022      * overlapping Master Transfer operation attempts.  The possible outcomes
1023      * are:
1024      * <ol>
1025      * <li>complete success ({@code done == true})
1026      * <ul>
1027      * <li>
1028      * don't unblock txns here; that'll happen automatically as part of the
1029      * usual handling when the environment transitions from master->replica
1030      * state.
1031      * <li>
1032      * don't clear xfrInProgress, because we don't want to allow another
1033      * attempt to supersede
1034      * </ul>
1035      * <li>timeout before establishing a winner (no superseder)
1036      * <ul>
1037      * <li>unblock txns
1038      * <li>clear xfrInProgress
1039      * </ul>
1040      * <li>superseded (see {@link #setUpTransfer})
1041      * <ul>
1042      * <li>abort existing op (if permitted), unblock txns before unleashing the
1043      * new one
1044      * <li>replace xfrInProgress
1045      * </ul>
1046      * <li>env is closed (or invalidated because of an error) during the
1047      * operation
1048      * <ul>
1049      * <li>release the block
1050      * <li>leave xfrInProgress as is.
1051      * </ul>
1052      * </ol>
1053      *
1054      * @param replicas candidate targets for new master role
1055      * @param timeout time limit, in msec
1056      * @param force whether to replace any existing, in-progress
1057      * transfer operation
1058      */
1059     public String transferMaster(Set<String> replicas,
1060                                  long timeout,
1061                                  boolean force) {
1062         if (replicas == null || replicas.isEmpty()) {
1063             throw new IllegalArgumentException
1064                 ("Parameter 'replicas' cannot be null or empty");
1065         }
1066         if (!nodeState.getRepEnvState().isMaster()) {
1067             throw new IllegalStateException("Not currently master");
1068         }
1069         if (replicas.contains(getNodeName())) {
1070 
1071             /*
1072              * The local node is on the list of candidate new masters, and
1073              * we're already master: the operation is trivially satisfied.
1074              */
1075             return getNodeName();
1076         }
1077         for (String rep : replicas) {
1078             RepNodeImpl node = group.getNode(rep);
1079             if (node == null || node.isRemoved()) {
1080                 throw new IllegalArgumentException
1081                     ("Node '" + rep +
1082                      "' is not currently an active member of the group");
1083             } else if (!node.getType().isElectable()) {
1084                 throw new IllegalArgumentException
1085                     ("Node '" + rep +
1086                      "' must have node type ELECTABLE, but had type " +
1087                      node.getType());
1088             }
1089         }
1090 
1091         MasterTransfer xfr = setUpTransfer(replicas, timeout, force);
1092         boolean done = false;
1093         try {
1094             String winner = xfr.transfer();
1095             done = true;
1096             return winner;
1097         } finally {
1098             synchronized (this) {
1099                 if (xfrInProgress == xfr && !done) {
1100                     xfrInProgress = null;
1101                 }
1102             }
1103         }
1104     }
