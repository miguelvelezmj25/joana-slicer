26 /**
27  * Implementation of a master node acting as a FeederSource. The
28  * MasterFeederSource is stateful, because it keeps its own FeederReader which
29  * acts as a cursor or scanner across the log files, so it can only be used by
30  * a single Feeder.
31  */
32 public class MasterFeederSource implements FeederSource {
33 
34     private final FeederReader feederReader;
35 
36     /* Protects files being read from being deleted. See FileProtector. */
37     private final FileProtector.ProtectedFileRange protectedFileRange;
38 
39     public MasterFeederSource(EnvironmentImpl envImpl,
40                               VLSNIndex vlsnIndex,
41                               NameIdPair replicaNameIdPair,
42                               VLSN startVLSN)
43         throws DatabaseException, IOException {
44 
45         int readBufferSize =
46             envImpl.getConfigManager().getInt
47             (EnvironmentParams.LOG_ITERATOR_READ_SIZE);
48 
49         feederReader = new FeederReader(envImpl,
50                                         vlsnIndex,
51                                         DbLsn.NULL_LSN, // startLsn
52                                         readBufferSize);
53 
54         long startLsn = feederReader.initScan(startVLSN);
55 
56         /*
57          * Syncup currently protects all files in the entire VLSNIndex range.
58          * This allows us to safely protect files from the matchpoint onward.
59          */
60         protectedFileRange = envImpl.getFileProtector().protectFileRange(
61             FileProtector.FEEDER_NAME + "-" + replicaNameIdPair,
62             DbLsn.getFileNumber(startLsn),
63             true /*protectVlsnIndex*/);
64     }
65 
66     /**
67      * Must be called to allow deletion of files protected by this feeder.
68      */
69     @Override
70     public void shutdown(EnvironmentImpl envImpl) {
71         envImpl.getFileProtector().removeFileProtection(protectedFileRange);
72     }
73 
74     /*
75      * @see com.sleepycat.je.rep.stream.FeederSource#getLogRecord
76      * (com.sleepycat.je.utilint.VLSN, int)
77      */
78     @Override
79     public OutputWireRecord getWireRecord(VLSN vlsn, int waitTime)
80         throws DatabaseException, InterruptedException, IOException {
81 
82         try {
83             OutputWireRecord record =
84                 feederReader.scanForwards(vlsn, waitTime);
85 
86             if (record == null) {
87                 return null;
88             }
89 
90             /*
91              * Advance the protected file range when we advance to a new file,
92              * to allow deletion of older files. Use getRangeStart (which is
93              * not synchronized) to check whether the file has advanced, before
94              * calling advanceRange (which is synchronized). This check must be
95              * inexpensive and non-blocking.
96              */
97             long lastFile = feederReader.getLastFile(record);
98             if (lastFile > protectedFileRange.getRangeStart()) {
99                 protectedFileRange.advanceRange(lastFile);
100             }
101 
102             return record;
103         } catch (DatabaseException e) {
104             /* Add more information */
105             e.addErrorMessage
106                 ("MasterFeederSource fetching vlsn=" + vlsn +
107                  " waitTime=" + waitTime);
108             throw e;
109         }
110     }
111 
112     @Override
113     public String dumpState() {
114         return feederReader.dumpState();
115     }
116 }
