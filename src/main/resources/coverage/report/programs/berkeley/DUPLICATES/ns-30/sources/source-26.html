94     /**
95      * Adds a new value to the average, ignoring values that are less than
96      * {@link #MIN_PERIOD} milliseconds older than the last entry.
97      *
98      * @param value the new value
99      * @param time the current time in milliseconds
100      */
101     public synchronized void add(long value, long time) {
102         assert time > 0;
103         if (prevTime != 0) {
104             final long deltaTime = time - prevTime;
105             if (deltaTime < MIN_PERIOD) {
106                 return;
107             }
108             avg.add(((double) (value - prevValue)) / ((double) deltaTime),
109                     time);
110         }
111         prevValue = value;
112         prevTime = time;
113     }

129     /**

131      * and the argument safely.
132      */
133     private void addInternal(LongAvgRate other) {
134         assert(Thread.holdsLock(this));
135         assert(Thread.holdsLock(other));
136 
137         /*

141         if ((other.prevTime - prevTime) > MIN_PERIOD) {
142             avg.add(other.avg);
143             prevValue = other.prevValue;

145         }

190     /** Returns the current average rate as a primitive value. */
191     private long getPrimitive() {
192         final double inMillis = avg.getPrimitive();
193         if (reportTimeUnit == MILLISECONDS) {
194             return Math.round(inMillis);
195         } else if (reportTimeUnit.compareTo(MILLISECONDS) < 0) {
196             return Math.round(
197                 inMillis / reportTimeUnit.convert(1, MILLISECONDS));
198         } else {

200         }
