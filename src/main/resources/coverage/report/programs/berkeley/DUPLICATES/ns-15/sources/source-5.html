32 /**
33  * This exception indicates that the log files constituting the Environment are
34  * insufficient and cannot be used as the basis for continuing with the
35  * replication stream provided by the current master.
36  * <p>
37  * This exception is typically thrown by the ReplicatedEnvironment constructor
38  * when a node has been down for a long period of time and is being started up
39  * again. It may also be thrown when a brand new node attempts to become a
40  * member of the group and it does not have a sufficiently current set of log
41  * files. If the group experiences sustained network connectivity problems,
42  * this exception may also be thrown by an active Replica that has been unable
43  * to stay in touch with the members of its group for an extended period of
44  * time.
45  * <p>
46  * In the typical case, application handles the exception by invoking
47  * {@link NetworkRestore#execute} to obtain the log files it needs from one of
48  * the members of the replication group. After the log files are obtained, the
49  * node recreates its environment handle and resumes participation as an active
50  * member of the group.
51  *
52  * @see NetworkRestore
53  */
54 public class InsufficientLogException extends RestartRequiredException {

142     private Properties initProperties(String groupName) {
143         Properties p = new Properties();
144 
145         /*
146          * These apply to the destination node which is experiencing the
147          * insufficient logs.
148          */
149         p.setProperty(GROUP_NAME, groupName);
150         p.setProperty(NODE_NAME, repImpl.getNameIdPair().getName());
151         p.setProperty(HOSTNAME, repImpl.getHostName());
152         p.setProperty(PORT, Integer.toString(repImpl.getPort()));
153         p.setProperty(ENV_DIR, repImpl.getEnvironmentHome().getPath());
154         p.setProperty(REFRESH_VLSN, Long.toString(refreshVLSN.getSequence()));
155 
156         /*
157          * There is a set of nodes which might act as the source for the
158          * network restore. Since we can't store arrays, append an index
159          * to the property name for each log provider.
160          */
161         p.setProperty(P_NUMPROVIDERS, Integer.toString(logProviders.size()));
162         int i = 0;
163         for (ReplicationNode rn: logProviders) {
164             p.setProperty(P_NODENAME + i, rn.getName());
165             p.setProperty(P_HOSTNAME + i, rn.getHostName());
166             p.setProperty(P_PORT + i, Integer.toString(rn.getPort()));
167             p.setProperty(P_NODETYPE + i, rn.getType().name());
168             i++;
169         }
170         return p;
171     }

192     /**
193      * @hidden
194      *
195      * Creates an instance of the exception when a LOG_RESTORE_REQUIRED was
196      * found at recovery, and network restore must be initiated before the
197      * recovery can succeed. The flow is:
198      * 0. A network restore is underway. It writes a marker file, essentially
199      *    serializing this ILE as a property list, and writing it into the log.
200      *    Something then interrupts this network restore. The process dies,
201      *    so all knowledge of the interrupted network restore is lost in
202      *    memory but the the marker file acts as a persistent breadcrumb.
203      * 1. Since knowledge of the network restore was lost, the application tries
204      *    to open and recover the target node. The LOG_RESTORE_REQUIRED entry
205      *    is seen, which means that recovery can't continue.
206      * 2. Within recovery, a new ILE is created using information from the
207      *    LOG_RESTORE_REQUIRED. It's thrown, which ends recovery.
208      * 3. The caller realizes that a network restore has to be carried out
209      *    before this environment can be recovered. It uses the ILE instance
210      *    created in step 2 to start a new invocation of network restore.
211      * 4. Network restore starts. Since network restore needs a RepImpl, one
212      *    is instantiated using info from the ILE
213      * 5. When Network restore succeeds, it removes the marker file.
214      *
215      * @param helperHosts extra helper hosts are derived from those used
216      * to open the environment.
217      */
218     public InsufficientLogException(Properties properties,
219                                     String helperHosts) {
220         super(null, EnvironmentFailureReason.INSUFFICIENT_LOG);
221 
222         /*
223          * Don't initialize the repImpl until it's needed. If we try to do
224          * so in step2, we'll be in a loop, trying to recover an environment
225          * from  within recovery.
226          */
227         String vlsnVal =  properties.getProperty(REFRESH_VLSN);
228         this.refreshVLSN = new VLSN(Long.parseLong(vlsnVal));
229         this.logProviders = new HashSet<>();
230 
231         if (helperHosts != null) {
232             for (String hostPortPair : helperHosts.split(",")) {
233                 final String hpp = hostPortPair.trim();
234                 if (hpp.length() > 0) {
235                     LogFileSource source = 
236                         new LogFileSource("NoName", // not important
237                                           NodeType.ELECTABLE.name(),
238                                           HostPortPair.getHostname(hpp),
239                                           HostPortPair.getPort(hpp));
240                     logProviders.add(source);
241                 }
242             }
243         }
244         this.props = properties;
245     }

247     private void initRepImpl() {
248         /*
249          * Setup log providers. Since can't use something that supports array
250          * types like JSON, and must use a property list, the provider flags
251          * are named NODE_NAME0, HOSTNAME0, etc.
252          */
253         int numLogProviders =
254             Integer.parseInt(props.getProperty(P_NUMPROVIDERS));
255         for (int i = 0; i < numLogProviders; i++) {
256             String name = props.getProperty(P_NODENAME + i);
257             String nodeType = props.getProperty(P_NODETYPE + i);
258             String hostname = props.getProperty(P_HOSTNAME + i);
259             int port = Integer.parseInt(props.getProperty(P_PORT + i));
260             logProviders.add(new LogFileSource(name, nodeType, hostname, port));
261         }
262 
263         /*
264          * Create new, read only, internal handle type environment for use
265          * by the network backup. 
266          */
267         EnvironmentConfig envConfig = new EnvironmentConfig();
268         envConfig.setReadOnly(true);
269         envConfig.setConfigParam(EnvironmentParams.ENV_RECOVERY.getName(),
270                                  "false");
271         envConfig.setTransactional(true);
272 
273         String hostname = props.getProperty(HOSTNAME);
274         int portVal = Integer.parseInt(props.getProperty(PORT));
275         ReplicationConfig repConfig =
276             new ReplicationConfig(props.getProperty(GROUP_NAME),
277                                   props.getProperty(NODE_NAME),
278                                   HostPortPair.getString(hostname, portVal));
279 
280         repConfig.setConfigParam(RepParams.NETWORKBACKUP_USE.getName(), "true");
281         ReplicationNetworkConfig defaultNetConfig =
282             ReplicationNetworkConfig.createDefault();
283         repConfig.setRepNetConfig(defaultNetConfig);
284 
285         File envDir = new File(props.getProperty(ENV_DIR));
286         ReplicatedEnvironment restoreEnv =
287             RepInternal.createInternalEnvHandle(envDir,
288                                                 repConfig,
289                                                 envConfig);
290         this.repImpl = RepInternal.getRepImpl(restoreEnv);
291         openedByILE = true;
292     }
