58     public MonitorService(Monitor monitor,
59                           ServiceDispatcher dispatcher) {
60         super(SERVICE_NAME, dispatcher);
61         this.monitor = monitor;

64                                 null,

72     /**
73      * Notify the monitor about the group change (add/remove a node) event.
74      */
75     public ResponseMessage process(GroupChange groupChange) {
76         GroupChangeEvent event =
77             new GroupChangeEvent(new ReplicationGroup(groupChange.getGroup()),
78                                  groupChange.getNodeName(),
79                                  groupChange.getOpType());
80         try {
81             assert doHookIfSet(processGroupChangeHook, event);
82         } catch (IllegalStateException e) {
83             return null;
84         }
85         monitor.notifyGroupChange(event);
86         return null;
87     }
88 
89     /**
90      * Notify the monitor about a node has joined the group.
91      */
92     public ResponseMessage process(JoinGroup joinGroup) {
93         JoinGroupEvent event = new JoinGroupEvent(joinGroup.getNodeName(),
94                                                   joinGroup.getMasterName(),
95                                                   joinGroup.getJoinTime());
96         try {
97             assert doHookIfSet(processJoinGroupHook, event);
98         } catch (IllegalStateException e) {
99             return null;
100         }
101         monitor.notifyJoin(event);
102         return null;
103     }
104 
105     /**
106      * Notify the monitor about a node has left the group.
107      */
108     public ResponseMessage process(LeaveGroup leaveGroup) {
109         LeaveGroupEvent event =
110             new LeaveGroupEvent(leaveGroup.getNodeName(),
111                                 leaveGroup.getMasterName(),
112                                 leaveGroup.getLeaveReason(),
113                                 leaveGroup.getJoinTime(),
114                                 leaveGroup.getLeaveTime());
115         try {
116             assert doHookIfSet(processLeaveGroupHook, event);
117         } catch (IllegalStateException e) {
118             return null;
119         }
120         monitor.notifyLeave(event);
121         return null;
122     }
