18 /**
19  * A base class for serial bindings creators that provides control over the
20  * allocation of the output buffer.
21  *
22  * <p>Serial bindings append data to a {@link FastOutputStream} instance.  This
23  * object has a byte array buffer that is resized when it is full.  The
24  * reallocation of this buffer can be a performance factor for some
25  * applications using large objects.  To manage this issue, the {@link
26  * #setSerialBufferSize} method may be used to control the initial size of the
27  * buffer, and the {@link #getSerialOutput} method may be overridden by
28  * subclasses to take over creation of the FastOutputStream object.</p>
29  *
30  * @see <a href="SerialBinding.html#evolution">Class Evolution</a>
31  *
32  * @author Mark Hayes
33  */
34 public class SerialBase {
35 
36     private int outputBufferSize;
37 
38     /**
39      * Initializes the initial output buffer size to zero.
40      *
41      * <p>Unless {@link #setSerialBufferSize} is called, the default {@link
42      * FastOutputStream#DEFAULT_INIT_SIZE} size will be used.</p>
43      */
44     public SerialBase() {
45         outputBufferSize = 0;
46     }
47 
48     /**
49      * Sets the initial byte size of the output buffer that is allocated by the
50      * default implementation of {@link #getSerialOutput}.
51      *
52      * <p>If this property is zero (the default), the default {@link
53      * FastOutputStream#DEFAULT_INIT_SIZE} size is used.</p>
54      *
55      * @param byteSize the initial byte size of the output buffer, or zero to
56      * use the default size.
57      */

60     }
61

64      *
65      * @return the initial byte size of the output buffer.
66      *
67      * @see #setSerialBufferSize
68      */
69     public int getSerialBufferSize() {

71     }
72 
73     /**
74      * Returns an empty SerialOutput instance that will be used by the serial
75      * binding or key creator.
76      *
77      * <p>The default implementation of this method creates a new SerialOutput
78      * with an initial buffer size that can be changed using the {@link
79      * #setSerialBufferSize} method.</p>
80      *
81      * <p>This method may be overridden to return a FastOutputStream instance.

85      *
86      * @param object is the object to be written to the serial output, and may

97         } else {

99         }
