40 /**
41  * An LNLogEntry is the in-memory image of an LN logrec describing a write op
42  * (insertion, update, or deletion) performed by a locker T on a record R.
43  * T always locks R in exclusive (WRITE or WRITE_RANGE) mode before performing
44  * any write ops on it, and it retains its exclusive lock on R until it
45  * terminates (commits or aborts). (Non-transactional lockers can be viewed as
46  * "simple" transactions that perform at most one write op, and then
47  * immediately commit).
48  *
49  * On disk, an LN logrec contains :
50  *
51  * 1 <= version <= 5
52  *
53  *   LN data
54  *   databaseid
55  *   key
56  *   abortLsn             -- if transactional
57  *   abortKnownDeleted    -- if transactional
58  *   txn id               -- if transactional
59  *   prev LSN of same txn -- if transactional
60  *
61  * 6 <= versions <= 10 :
62  *
63  *   databaseid
64  *   abortLsn             -- if transactional
65  *   abortKnownDeleted    -- if transactional
66  *   txn id               -- if transactional
67  *   prev LSN of same txn -- if transactional
68  *   data
69  *   key
70  *
71  * 11 == version :
72  *
73  *   databaseid
74  *   abortLsn               -- if transactional
75  *   1-byte flags
76  *     abortKnownDeleted
77  *     embeddedLN
78  *     haveAbortKey
79  *     haveAbortData
80  *     haveAbortVLSN
81  *   txn id                 -- if transactional
82  *   prev LSN of same txn   -- if transactional
83  *   abort key              -- if haveAbortKey
84  *   abort data             -- if haveAbortData
85  *   abort vlsn             -- if haveAbortVLSN
86  *   data
87  *   key
88  *
89  *   In forReplication mode, these flags and fields are omitted:
90  *     embeddedLN, haveAbortKey, haveAbortData, haveAbortVLSN,
91  *     abort key, abort data, abort vlsn
92  *
93  * 12 <= version :
94  *
95  *   1-byte flags
96  *     abortKnownDeleted
97  *     embeddedLN
98  *     haveAbortKey
99  *     haveAbortData
100  *     haveAbortVLSN
101  *     haveAbortLSN
102  *     haveAbortExpiration
103  *     haveExpiration
104  *   databaseid
105  *   abortLsn                -- if transactional and haveAbortLSN
106  *   txn id                  -- if transactional
107  *   prev LSN of same txn    -- if transactional
108  *   abort key               -- if haveAbortKey
109  *   abort data              -- if haveAbortData
110  *   abort vlsn              -- if haveAbortVLSN
111  *   abort expiration        -- if haveAbortExpiration
112  *   expiration              -- if haveExpiration
113  *   data
114  *   key
115  *
116  *   In forReplication mode, these flags and fields are omitted:
117  *     abortKnownDeleted, embeddedLN, haveAbortKey, haveAbortData,
118  *     haveAbortVLSN, abort key, abort data, abort vlsn
119  *
120  * NOTE: LNLogEntry is sub-classed by NameLNLogEntry, which adds some extra
121  * fields after the record key.
122  */

399      * Method shared by LNLogEntry subclasses.
400      *
401      * @param keyIsLastSerializedField specifies whether the key length can be
402      * omitted because the key is the last field.  This should be false when
403      * an LNLogEntry subclass adds fields to the serialized format.
404      */
405     final void readBaseLNEntry(
406         EnvironmentImpl envImpl,
407         LogEntryHeader header,
408         ByteBuffer entryBuffer,
409         boolean keyIsLastSerializedField) {
410 
411         reset();
412 
413         int logVersion = header.getVersion();
414         boolean unpacked = (logVersion < 6);
415         int recStartPosition = entryBuffer.position();
416 
417         if (logVersion >= 12) {
418             setFlags(entryBuffer.get());
419         }
420 
421         /*
422          * For log version 6 and above we store the key last so that we can
423          * avoid storing the key size. Instead, we derive it from the LN size
424          * and the total entry size. The DatabaseId is also packed.
425          */
426         if (logVersion < 6) {
427             /* LN is first for log versions prior to 6. */
428             ln = newLNInstance(envImpl);
429             ln.readFromLog(entryBuffer, logVersion);
430         }
431 
432         /* DatabaseImpl Id. */
433         dbId = new DatabaseId();
434         dbId.readFromLog(entryBuffer, logVersion);
435 
436         /* Key. */
437         if (logVersion < 6) {
438             key = LogUtils.readByteArray(entryBuffer, true/*unpacked*/);
439         }
440 
441         if (entryType.isTransactional()) {
442 
443             /*
444              * AbortLsn. If it was a marker LSN that was used to fill in a
445              * create, mark it null.
446              */
447             if (haveAbortLSN || logVersion < 12) {
448                 abortLsn = LogUtils.readLong(entryBuffer, unpacked);
449                 if (DbLsn.getFileNumber(abortLsn) ==
450                     DbLsn.getFileNumber(DbLsn.NULL_LSN)) {
451                     abortLsn = DbLsn.NULL_LSN;
452                 }
453             }
454 
455             if (logVersion < 12) {
456                 setFlags(entryBuffer.get());
457                 haveAbortLSN = (abortLsn != DbLsn.NULL_LSN);
458             }
459 
460             /* txn id and prev LSN by same txn. */
461             txn = new Txn();
462             txn.readFromLog(entryBuffer, logVersion);
463 
464         } else if (logVersion == 11) {
465             setFlags(entryBuffer.get());
466         }
467 
468         if (logVersion >= 11) {
469             if (haveAbortKey) {
470                 abortKey = LogUtils.readByteArray(entryBuffer, false);
471             }
472             if (haveAbortData) {
473                 abortData = LogUtils.readByteArray(entryBuffer, false);
474             }
475             if (haveAbortVLSN) {
476                 abortVLSN = LogUtils.readPackedLong(entryBuffer);
477             }
478         }
479 
480         if (logVersion >= 12) {
481             if (haveAbortExpiration) {
482                 abortExpiration = LogUtils.readPackedInt(entryBuffer);
483                 if (abortExpiration < 0) {
484                     abortExpiration = (- abortExpiration);
485                     abortExpirationInHours = true;
486                 }
487             }
488             if (haveExpiration) {
489                 expiration = LogUtils.readPackedInt(entryBuffer);
490                 if (expiration < 0) {
491                     expiration = (- expiration);
492                     expirationInHours = true;
493                 }
494             }
495         }
496 
497         if (logVersion >= 6) {
498 
499             ln = newLNInstance(envImpl);
500             ln.readFromLog(entryBuffer, logVersion);
501 
502             int keySize;
503             if (keyIsLastSerializedField) {
504                 int bytesWritten = entryBuffer.position() - recStartPosition;
505                 keySize = header.getItemSize() - bytesWritten;
506             } else {
507                 keySize = LogUtils.readPackedInt(entryBuffer);
508             }
509             key = LogUtils.readBytesNoLength(entryBuffer, keySize);
510         }
511 
512         /* Save transient fields after read. */
513 
514         if (header.getVLSN() != null) {
515             ln.setVLSNSequence(header.getVLSN().getSequence());
516         }
517 
518         /* Dup conversion will be done by postFetchInit. */
519         dupStatus =
520             (logVersion < 8) ? DupStatus.NEED_CONVERSION : DupStatus.UNKNOWN;
521     }

539     @Override
540     public boolean isReplicationFormatWorthwhile(final ByteBuffer logBuffer,
541                                                  final int srcVersion,
542                                                  final int destVersion) {
543 
544         /* The replication format is optimized only in versions >= 11. */
545         if (destVersion < 11) {
546             return false;
547         }
548 
549         /*
550          * It is too much trouble to parse versions older than 12, because the
551          * flags are not at the front in older versions.
552          */
553         if (srcVersion < 12) {
554             return false;
555         }
556 
557         final byte flags = logBuffer.get(0);
558 
559         /*
560          * If we have an abort key or data, assume that the savings is
561          * substantial enough to be worthwhile.
562          *
563          * The abort key is unusual and implies that data is hidden in the key
564          * using a partial comparator, so we assume it is probably large,
565          * relative to the total size.
566          *
567          * If there is abort data, it may be small, however, because the
568          * presence of abort data implies that this is an update or deletion,
569          * there will also be an abort LSN and an abort VLSN (with HA). Plus,
570          * abort data is likely to be around the same size as the non-abort
571          * data, and keys are normally smallish, meaning that the abort data is
572          * largish relative to the total record size. So we assume the savings
573          * are substantial enough.
574          */
575         return (flags &
576             (HAVE_ABORT_KEY_MASK | HAVE_ABORT_DATA_MASK)) != 0;
577     }
