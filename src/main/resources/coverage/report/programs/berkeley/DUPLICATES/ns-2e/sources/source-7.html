41  * Loads a database from a dump file generated by {@link DbDump}. 
42  * This utility may be used programmatically or from the command line.
43  *
44  * <p>When using this utility as a command line program, and the
45  * application uses custom key comparators, be sure to add the jars or
46  * classes to the classpath that contain the application's comparator
47  * classes.</p>
48  *
49  *<pre>
50  * java { com.sleepycat.je.util.DbLoad |
51  *        -jar je-&lt;version&gt;.jar DbLoad }
52  *     -h &lt;dir&gt;            # environment home directory
53  *    [-f &lt;fileName&gt;]      # input file
54  *    [-n]                 # no overwrite mode
55  *    [-T]                 # input file is in text mode
56  *    [-I]                 # ignore unknown parameters
57  *    [-c name=value]      # config values
58  *    [-s &lt;databaseName&gt; ] # database to load
59  *    [-v]                 # show progress
60  *    [-V]                 # print JE version number
61  *</pre>
62  * See {@link DbLoad#main} for a full description of the 
63  * command line arguments.
64  * <p>
65  * To load a database to a stream from code:
66  * <pre>
67  *    DbLoad loader = new DbLoad();
68  *    loader.setEnv(env);
69  *    loader.setDbName(dbName);
70  *    loader.setInputStream(stream);
71  *    loader.setNoOverwrite(noOvrwr);

77  * each database, multiple databases can be dumped and loaded using a single
78  * stream. The {@link DbDump#dump} method leaves the stream positioned after
79  * the last line written and the {@link DbLoad#load} method leaves the stream

110      * The main used by the DbLoad utility.
111      *
112      * @param argv The arguments accepted by the DbLoad utility.
113      *
114      * <pre>
115      * usage: java { com.sleepycat.je.util.DbLoad | -jar
116      * je-&lt;version&gt;.jar DbLoad }
117      *             [-f input-file] [-n] [-V] [-v] [-T] [-I]
118      *             [-c name=value]
119      *             [-s database] -h dbEnvHome
120      * </pre>
121      *
122      * <p>-f - the file to load from (in DbDump format)<br>
123      * -n - no overwrite mode.  Do not overwrite existing data.<br>
124      * -V - display the version of the JE library.<br>
125      * -T - input file is in Text mode.<br>
126      * -I - ignore unknown parameters in the config file.<br></p>
127      *
128      * <p>If -f is not specified, the dump is read from System.in.</p>
129      *
130      * <p>The -T option allows JE applications to easily load text files into
131      * databases.</p>
132      *
133      * <p>The -I option allows loading databases that were dumped with the
134      * Berkeley DB C product, when the dump file contains parameters not known
135      * to JE.</p>
136      *
137      * <p>The input must be paired lines of text, where the first line of the
138      * pair is the key item, and the second line of the pair is its
139      * corresponding data item.</p>
140      *
141      * <p>A simple escape mechanism, where newline and backslash (\) characters
142      * are special, is applied to the text input. Newline characters are
143      * interpreted as record separators. Backslash characters in the text will
144      * be interpreted in one of two ways: If the backslash character precedes
145      * another backslash character, the pair will be interpreted as a literal
146      * backslash. If the backslash character precedes any other character, the
147      * two characters following the backslash will be interpreted as a
148      * hexadecimal specification of a single character; for example, \0a is a
149      * newline character in the ASCII character set.</p>
150      *
151      * <p>For this reason, any backslash or newline characters that naturally
152      * occur in the text input must be escaped to avoid misinterpretation by
153      * db_load.</p>
154      *
155      * <p>-c name=value - Specify configuration options ignoring any value they
156      * may have based on the input. The command-line format is name=value. See
157      * the Supported Keywords section below for a list of keywords supported by
158      * the -c option.</p>
159      *
160      * <p>-s database - the database to load.<br>
161      * -h dbEnvHome - the directory containing the database environment.<br>
162      * -v - report progress <br></p>
163      *
164      * <p>Supported Keywords<br>
165      * version=N - specify the version of the input file.  Currently only
166      * version 3 is supported.<br>
167      * format - specify the format of the file.  Allowable values are "print"
168      * and "bytevalue".<br>
169      * dupsort - specify whether the database allows duplicates or not.
170      * Allowable values are "true" and "false".<br>
171      * type - specifies the type of database.  Only "btree" is allowed.<br>
172      * database - specifies the name of the database to be loaded.<br></p>
173      *
174      * @throws EnvironmentFailureException if an unexpected, internal or
175      * environment-wide failure occurs.
176      */
177     public static void main(String argv[])
178         throws Exception {
179 
180         DbLoad loader = parseArgs(argv);
181         try {
182             loader.load();
183         } catch (Throwable e) {
184             e.printStackTrace();
185         }
186 
187         loader.env.close();
188     }

399 
400         if (progressInterval > 0) {
401             System.out.println("Load start: " + new Date());
402         }
403 
404         if (textFileMode) {
405             formatUsingPrintable = true;
406         } else {
407             loadHeader();
408         }
409 
410         if (dbName == null) {
411             throw new IllegalArgumentException
412                 ("Must supply a database name if -l not supplied.");
413         }
414

416         dbConfig.setSortedDuplicates(dupSort);
417         dbConfig.setAllowCreate(true);
418         Database db;
419         try {
420             db = env.openDatabase(null, dbName, dbConfig);
421         } catch (DatabaseNotFoundException e) {
422             /* Should never happen, AllowCreate is true. */
423             throw EnvironmentFailureException.unexpectedException(e);

443     private void loadConfigLine(String line) {
444         int equalsIdx = line.indexOf('=');
445         if (equalsIdx < 0) {
446             throw new IllegalArgumentException
447                 ("Invalid header parameter: " + line);
448         }
449 
450         String keyword = line.substring(0, equalsIdx).trim().toLowerCase();
451         String value = line.substring(equalsIdx + 1).trim();
452 
453         if (keyword.equals("version")) {
454             if (DEBUG) {
455                 System.out.println("Found version: " + line);
456             }
457             if (!value.equals("3")) {
458                 throw new IllegalArgumentException
459                     ("Version " + value + " is not supported.");
460             }
461         } else if (keyword.equals("format")) {
462             value = value.toLowerCase();
463             if (value.equals("print")) {
464                 formatUsingPrintable = true;
465             } else if (value.equals("bytevalue")) {
466                 formatUsingPrintable = false;
467             } else {
468                 throw new IllegalArgumentException

473             }

476             if (value.equals("true") ||
477                 value.equals("1")) {

480                        value.equals("0")) {
481                 dupSort = false;
482             } else {
483                 throw new IllegalArgumentException

485             }

488             }

525     private void loadData(Database db)
526         throws DatabaseException, IOException {
527 
528         String keyLine = reader.readLine();
529         String dataLine = null;
530         int count = 0;
531         long totalBytesRead = 0;
532         long lastTime = System.currentTimeMillis();
533         long bytesReadThisInterval = 0;
534 
535         while (keyLine != null &&
536                !keyLine.equals("DATA=END")) {
537             dataLine = reader.readLine();
538             if (dataLine == null) {
539                 throw new IllegalArgumentException("No data to match key " +
540                                                    keyLine);
541             }
542             /* Add one for \n or \r. */
543             bytesReadThisInterval += dataLine.length() + 1;
544             byte[] keyBytes = loadLine(keyLine.trim());
545             byte[] dataBytes = loadLine(dataLine.trim());
546 
547             DatabaseEntry key = new DatabaseEntry(keyBytes);
548             DatabaseEntry data = new DatabaseEntry(dataBytes);
549 
550             if (noOverwrite) {
551                 if (db.putNoOverwrite(null, key, data) ==
552                     OperationStatus.KEYEXIST) {
553                     /* Calling println is OK only from command line. */
554                     if (commandLine) {
555                         System.err.println("Key exists: " + key);
556                     }
557                 }
558             } else {
559                 db.put(null, key, data);
560             }
561 
562             count++;
563             if ((progressInterval > 0) &&
564                 (bytesReadThisInterval > progressInterval)) {
565                 totalBytesRead += bytesReadThisInterval;
566                 bytesReadThisInterval -= progressInterval;
567                 long now = System.currentTimeMillis();
568                 System.out.println("loaded " + count + " records  " +
569                                    (now - lastTime) + " ms - % completed: " +
570                                    ((100 * totalBytesRead) / totalLoadBytes));
571                 lastTime = now;
572             }
573 
574             keyLine = reader.readLine();
575             if (keyLine == null) {
576                 throw new IllegalArgumentException("No \"DATA=END\"");
577             }
578             bytesReadThisInterval += keyLine.length() + 1;
579         }
580     }

582     private byte[] loadLine(String line)
583         throws DatabaseException {
584 
585         if (formatUsingPrintable) {
586             return readPrintableLine(line);
587         }
588         int nBytes = line.length() / 2;
589         byte[] ret = new byte[nBytes];
590         int charIdx = 0;
591         for (int i = 0; i < nBytes; i++, charIdx += 2) {
592             int b2 = Character.digit(line.charAt(charIdx), 16);
593             b2 <<= 4;
594             b2 += Character.digit(line.charAt(charIdx + 1), 16);
595             ret[i] = (byte) b2;
596         }
597         return ret;
598     }

603     private byte[] readPrintableLine(String line)
604         throws DatabaseException {
605 
606         /* nBytes is the max number of bytes that this line could turn into. */
607         int maxNBytes = line.length();
608         byte[] ba = new byte[maxNBytes];
609         int actualNBytes = 0;
610 
611         for (int charIdx = 0; charIdx < maxNBytes; charIdx++) {
612             char c = line.charAt(charIdx);
613             if (c == '\\') {
614                 if (++charIdx < maxNBytes) {
615                     char c1 = line.charAt(charIdx);
616                     if (c1 == '\\') {
617                         ba[actualNBytes++] = backSlashValue;
618                     } else {
619                         if (++charIdx < maxNBytes) {
620                             char c2 = line.charAt(charIdx);
621                             int b = Character.digit(c1, 16);
622                             b <<= 4;
623                             b += Character.digit(c2, 16);
624                             ba[actualNBytes++] = (byte) b;
625                         } else {
626                             throw 
627                                 new IllegalArgumentException("Corrupted file");
628                         }
629                     }
630                 } else {
631                     throw new IllegalArgumentException("Corrupted file");
632                 }
633             } else {
634                 ba[actualNBytes++] = (byte) (c & 0xff);
635             }
636         }
637 
638         if (maxNBytes == actualNBytes) {
639             return ba;
640         } else {
641             byte[] ret = new byte[actualNBytes];
642             System.arraycopy(ba, 0, ret, 0, actualNBytes);
643             return ret;
644         }
645     }
