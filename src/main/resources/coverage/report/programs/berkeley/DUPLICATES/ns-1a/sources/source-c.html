18 /**
19  * Defines the protocol used in support of group membership.
20  *
21  * API to Master
22  *   ENSURE_NODE -> ENSURE_OK | FAIL
23  *   REMOVE_MEMBER -> OK | FAIL
24  *   TRANSFER_MASTER -> TRANSFER_OK | FAIL
25  *   DELETE_MEMBER -> OK | FAIL
26  *
27  * Monitor to Master
28  *   GROUP_REQ -> GROUP | FAIL

449         @Override
450         public String wireFormat() {
451 
452             /*
453              * Use the requested group version, unless it is newer than the
454              * current group version.
455              */
456             int groupFormatVersion = getGroupFormatVersion(sendVersion);
457             if (group.getFormatVersion() < groupFormatVersion) {
458                 groupFormatVersion = group.getFormatVersion();
459             }
460             return wireFormatPrefix() + SEPARATOR +
461                 group.serializeHex(groupFormatVersion);
462         }
