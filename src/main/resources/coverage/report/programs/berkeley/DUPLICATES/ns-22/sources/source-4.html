37 /**
38  * The FeederReader is a flavor of VLSNReader which supports replication
39  * stream feeding. It assumes that reading will always go forward in the log.
40  * Special features are:
41  *
42  * - The reader can read either from a log buffer or from the file. Sometimes
43  *   log entries are logged but are not yet available on disk. In general, it's
44  *   better to read from the log buffers rather then the file.
45  *
46  * - The reader can block for a given time period, waiting for the next vlsn to
47  *    appear
48  */
49 public class FeederReader extends VLSNReader {

112      * Set up the FeederReader to start scanning from this VLSN. If we find a
113      * mapping for this VLSN, we'll start precisely at its LSN, else we'll have
114      * to start from an earlier location. This initialization can't be done in
115      * the constructor, because the Feeder is set up to require the
116      * construction of the source before we know where to start.
117      *
118      * @throws IOException
119      *
120      * @return the startLsn

122     public long initScan(VLSN startVLSN)
123         throws IOException {
124 
125         if (startVLSN.equals(VLSN.NULL_VLSN)) {
126             throw EnvironmentFailureException.unexpectedState

128         }
129

131         VLSN startPoint = startVLSN;
132         if (currentRange.getLast().compareTo(startVLSN) < 0) {
133             /*
134              * When feeding, we may be starting at the VLSN following the last
135              * VLSN in the node.

137             startPoint = currentRange.getLast();
138         }

141         assert startLsn != DbLsn.NULL_LSN;
142

144         nextEntryOffset = window.getEndOffset();

167     /**
168      * Forward scanning for feeding the replica: get the log record for this
169      * VLSN. If the log record hasn't been created yet, wait for a period
170      * specified by "waitTime".
171      *
172      * Where possible, the FeederReader fetches the log record from the cache
173      * within the VLSNIndex. (See the VLSNIndex for a description of this two
174      * level cache). If the requested VLSN is not available from the cache, the
175      * reader fetches the item from the JE log -- either from the log buffers
176      * or from disk.
177      *
178      * The FeederReader is like a cursor on the log, and retains a position
179      * in the log.  When there are log item cache hits, the FeederReader's
180      * position can fall behind, because it is being bypassed. It is possible
181      * for log cleaning to take place between the point of the FeederReader's
182      * stale position and the end of the log. If so, the FeederReader must
183      * not attempt to scan from its current position, because it might
184      * run afoul of gap created by the cleaned and delete log files. When
185      * there have been log item cache hits, the FeederReader must jump its
186      * position forward using the vlsnIndex mappings to safely skip over
187      * any cleaned gaps in the log.
188      */
189     public OutputWireRecord scanForwards(VLSN vlsn, int waitTime)
190         throws InterruptedException {
191 
192         assert initDone;
193 
194         LogItem logItem = null;
195 
196         try {
197             logItem = vlsnIndex.waitForVLSN(vlsn, waitTime);
198         } catch (WaitTimeOutException e) {
199             /* This vlsn not yet available */
200             return null;
201         }
202 
203         currentVLSN = vlsn;
204 
205         if ((logItem != null) && (!bypassCache)) {
206 
207             /* We've found the requested log item in the cache. */
208             assert logItem.header.getVLSN().equals(vlsn);
209             prevCacheHits++;
210             return new OutputWireRecord(envImpl, logItem);
211         }
212 
213         /*
214          * We must go to the log for this requested VLSN. Use the VLSNIndex for
215          * the closest position in the log file to find the next replicated log
216          * entry.
217          *
218          * If there are no cache hits and the reader has been supplying log
219          * entries sequentially, we know that it is already positioned at the
220          * immediately preceding log entry and that we can scan from there to
221          * the current requested record. In that case, we are only hoping that
222          * the VLSNIndex can supply the exact location of the current requested
223          * record in order to reduce the scanning.
224          *
225          * If there have been cache hits, the reader's current position is some
226          * unknown distance back. In that case, scanning from the current
227          * position could run into a cleaned gap in the log files, and could
228          * fail. Because of that, we must reposition to a VLSN that is <= to
229          * the current requested VLSN. We know that such a VLSN must exist and
230          * have a valid lsn mapping, because the begin and end point in the
231          * vlsn range always exists.
232          */
233         long repositionLsn;
234         if (prevCacheHits > 0) {
235             repositionLsn = scanner.getApproximateLsn(vlsn);
236 
237             /*
238              * Guard against sliding the window backwards. This could happen if
239              * by dint of previous scans, the reader is fortuitously positioned
240              * at a point in the log that is before the current target VLSN,
241              * but after any available mappings. For example, suppose the
242              * VLSNIndex has VLSNs 10, 50, 100. Suppose the reader is
243              * positioned at VLSN 20, and we have supplied VLSNs 21->40 from
244              * the cache.  VLSN 41 has not hit in the cache, and we must fetch
245              * the log record from disk. We do not want to slide the
246              * FeederReader from its current position at 21 back to VLSN 10.
247              */
248             if (DbLsn.compareTo(getLastLsn(), repositionLsn) >= 0) {
249                 repositionLsn = DbLsn.NULL_LSN;
250             }
251         } else {
252             repositionLsn = scanner.getPreciseLsn(vlsn);
253         }
254 
255         /*
256          * We're going to start scanning, so reset the prevCacheHits field, and
257          * position the reader at the optimal spot.
258          */
259         prevCacheHits = 0;
260         try {
261             /* setPosition is a noop if repositionLsn is null. */
262             setPosition(repositionLsn);
263         } catch (ChecksumException e) {
264             throw new EnvironmentFailureException
265                 (envImpl,
266                  EnvironmentFailureReason.LOG_CHECKSUM,
267                  "trying to reposition FeederReader to " +
268                  DbLsn.getNoFormatString(repositionLsn) + " prevWindow=" +
269                  window, e);
270         } catch (FileNotFoundException e) {
271             throw new EnvironmentFailureException
272                 (envImpl, EnvironmentFailureReason.LOG_FILE_NOT_FOUND,
273                  "Trying to reposition FeederReader to " +
274                  DbLsn.getNoFormatString(repositionLsn) +
275                  " for vlsn:" + vlsn + " prevWindow=" + window, e);
276         }
277 
278         if (readNextEntry()) {
279             return currentFeedRecord;
280         }
281 
282         throw EnvironmentFailureException.unexpectedState
283             (envImpl, "VLSN=" + vlsn + " repositionLsn = " +
284              DbLsn.getNoFormatString(repositionLsn) + window);
285     }

305         nScanned++;
306

308             return false;
309         }
310 
311         if (entryIsReplicated()) {
312             VLSN entryVLSN = currentEntryHeader.getVLSN();
313 
314             int compareResult = entryVLSN.compareTo(currentVLSN);
315             checkForPassingTarget(compareResult);
316

318             return (compareResult == 0);

325      * The SwitchWindow can fill itself from either the log file or the log
326      * buffers.
327      */
328     static class SwitchWindow extends ReadWindow {
329 
330         private final LogManager logManager;
331 
332         SwitchWindow(int readBufferSize, EnvironmentImpl envImpl) {
333             super(readBufferSize, envImpl);
334 
335             logManager = envImpl.getLogManager();
336         }
337 
338         /*
339          * Reposition to the specified file, and fill starting at
340          * targetOffset. For this use case, we are always going forwards, and
341          * windowStartOffset should == targetOffset. Position the window's
342          * buffer to point at the log entry indicated by targetOffset
343          */
344         @Override
345         public void slideAndFill(long windowFileNum,
346                                  long windowStartOffset,
347                                  long targetOffset,
348                                  boolean forward)
349             throws ChecksumException,
350                    FileNotFoundException,
351                    DatabaseException {
352 
353             if (!fillFromLogBuffer(windowFileNum, targetOffset)) {
354                 /* The entry was not in the LogBufferPool. */
355                 super.slideAndFill(windowFileNum,
356                                    windowStartOffset,
357                                    targetOffset,
358                                    forward);
359             }

362         /**
363          * Fill the read window's buffer from a LogBuffer.
364          * @return true if the read window was filled.
365          * @throws DatabaseException
366          */
367         private boolean fillFromLogBuffer(long windowFileNum,
368                                           long targetOffset)
369             throws DatabaseException {
370 
371             LogBuffer logBuffer = null;
372 
373             try {
374                 long fileLocation = DbLsn.makeLsn(windowFileNum, targetOffset);
375                 logBuffer = logManager.getReadBufferByLsn(fileLocation);
376                 if (logBuffer == null) {
377                     return false;
378                 }
379 
380                 /*
381                  * Copy at much as we can of the logBuffer into the window's
382                  * readBuffer. We don't call ByteBuffer.put(ByteBuffer) because
383                  * the logBuffer may be larger than the window readBuffer, and
384                  * we don't want to get an overflow. Instead, we convert to an
385                  * array and carefully size the copy. A LogBuffer is positioned
386                  * for writing, and hasn't yet been flipped.  LogManager.get()
387                  * does an absolute retrieval of bytes from the buffer, because
388                  * it knows that the log entry exists, and is only reading one
389                  * entry. We need to flip the buffer, because we don't know
390                  * apriori how much is in the buffer, and we want to scan it.
391                  */
392 
393                 /*
394                  * Put the logBuffer's contents into wholeContents, and
395                  * position wholeContents at the desired target offset. If
396                  * this logBuffer had been the currentWriteBuffer, it's
397                  * positioned for writing and must be flipped for reading.
398                  */
399                 ByteBuffer wholeContents =
400                     logBuffer.getDataBuffer().duplicate();
401                 if (wholeContents.position() != 0) {
402                     wholeContents.flip();
403                 }
404                 long firstOffset =
405                     DbLsn.getFileOffset(logBuffer.getFirstLsn());
406                 wholeContents.position((int) (targetOffset - firstOffset));
407 
408                 /* Make a buffer which starts at target. */
409                 ByteBuffer startAtTarget = wholeContents.slice();
410                 byte[] data = startAtTarget.array();
411                 int availableContentLen = startAtTarget.limit();
412                 int copyLength =
413                     (availableContentLen > readBuffer.capacity()) ?
414                     readBuffer.capacity() : availableContentLen;
415 
416                 readBuffer.clear();
417                 readBuffer.put(data, startAtTarget.arrayOffset(), copyLength);
418                 readBuffer.flip();
419 
420                 /* LogBuffers were just written and use the current version. */
421                 setFileNum(windowFileNum, LogEntryType.LOG_VERSION);
422                 startOffset = targetOffset;
423                 endOffset = startOffset + readBuffer.limit();
424                 readBuffer.position(0);
425                 return true;
426             } finally {
427                 if (logBuffer != null) {
428                     logBuffer.release();
429                 }
430             }
431         }
432 
433         /**
434          * Fill up the read buffer with more data, moving along to the
435          * following file (next largest number) if needed. Unlike other file
436          * readers, we are reading log files that are concurrently growing, so
437          * this read window must also know to look in the log buffers.
438          *
439          * The contract between the feeder reader and the VLSNIndex lets us
440          * assume that the feeder reader is only active when it is sure that
441          * there is more data available somewhere -- whether it's in the log
442          * buffers, write queue, or on disk.
443          *
444          * @return true if the fill moved us to a new file.
445          * @see ReadWindow#fillNext
446          */
447         @Override
448             protected boolean fillNext(boolean singleFile, int bytesNeeded)
449             throws ChecksumException, DatabaseException, EOFException {
450 
451             /*
452              * The SwitchReadWindow should only be used for feeding, and
453              * singleFile should never be true.
454              */
455             assert !singleFile;
456 
457             adjustReadBufferSize(bytesNeeded);
458 
459             /*
460              * Try to fill the window by asking for the next offset from
461              * the log buffers.
462              */
463             if (fillFromLogBuffer(currentFileNum(), endOffset)) {
464                 /* Didn't move to a new file. */
465                 return false;
466             }
467 
468             /*
469              * If that didn't work, there are these possible reasons why:
470              * a - it's a valid offset, but it's no longer in a log buffer, it
471              *     was written to disk.
472              * b - it's not a valid offset, because the log file flipped.
473              * In both cases, go to the FileManager and see if there's more log
474              * to be found.
475              */
476 
477             FileHandle fileHandle = null;
478             try {
479 
480                 /* Get a file handle to read in more log. */
481                 fileHandle = fileManager.getFileHandle(currentFileNum());
482 
483                 /* Attempt to read more from this file. */
484                 startOffset = endOffset;
485                 if (fillFromFile(fileHandle, startOffset)) {
486                     /*
487                      * Successfully filled the read buffer, but didn't move to
488                      * a new file.
489                      */
490                     return false;
491                 }
492 
493                 fileHandle.release();
494                 fileHandle = null;
495 
496                 /* This file is done -- can we read in the next file? */
497                 if (singleFile) {
498                     throw new EOFException();
499                 }
500 
501                 /*
502                  * Remember that the nextFile may not be fileNum + 1 if
503                  * there has been log cleaning.
504                  */
505                 Long nextFile =
506                     fileManager.getFollowingFileNum(currentFileNum(),
507                                                     true /* forward */);
508 
509                 /*
510                  * But if there's no next file, let's assume that the desired
511                  * data is still in the log buffers, and the next lsn is the
512                  * first entry in the subsequent file number.  Start the read
513                  * from the first real log entry, because the file header entry
514                  * is not in the log buffers.
515                  */
516                 if (nextFile == null) {
517                     nextFile = currentFileNum() + 1;
518                 }
519 
520                 if (fillFromLogBuffer(nextFile,
521                                       FileManager.firstLogEntryOffset())) {
522                     /*
523                      * We filled the read buffer, and jumped to a new
524                      * file.
525                      */
526                     return true;
527                 }
528 
529                 /*
530                  * Didn't find the next bytes in the log buffer, go look on
531                  * disk.
532                  */
533                 fileHandle = fileManager.getFileHandle(nextFile);
534                 setFileNum(nextFile, fileHandle.getLogVersion());
535                 startOffset = 0;
536                 boolean moreData = fillFromFile(fileHandle, 0);
537                 assert moreData :
538                    "FeederReader should find more data in next file";
539                 return true;
540             } catch (IOException e) {
541                 e.printStackTrace();
542                 throw EnvironmentFailureException.unexpectedException
543                     ("Problem in ReadWindow.fill, reading from  = " +
544                      currentFileNum(), e);
545 
546             } finally {
547                 if (fileHandle != null) {
548                     fileHandle.release();
549                 }
550             }
551         }
552     }
