341     /**
342      * Checks if the argument represents the same node, ignoring fields that
343      * might legitimately vary over time.  Like the equals method, considers
344      * all fields, except ignores the quorumAck field (which may change
345      * temporarily), the nodeId (since it may not have been resolved as yet),
346      * and the isRemoved, barrierState, changeVersion, and jeVersion fields

348      *

354         if (this == mi) {
355             return true;

359             return false;

361

363             return false;

365

367             if (mi.hostName != null) {

370         } else if (!hostName.equals(mi.hostName)) {
371             return false;

373

375         if (!nameIdPair.getName().equals(mi.nameIdPair.getName())) {
376             return false;
377         }
378 
379         if (getType() != mi.getType()) {
380             return false;
381         }
382 
383         /*
384          * Ignore quorumAck, isRemoved, barrierState, changeVersion, and
385          * jeVersion
386          */
387 
388         return true;
