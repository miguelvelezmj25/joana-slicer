161     private void parseArgs(String argv[]) {
162 
163         int argc = 0;
164         int nArgs = argv.length;
165 
166         if (nArgs == 0) {
167             printUsage(null);
168             System.exit(0);
169         }
170 
171         while (argc < nArgs) {
172             String thisArg = argv[argc++];
173             if (thisArg.equals("-q")) {
174                 quiet = true;
175             } else if (thisArg.equals("-u")) {
176                 sorted = true;
177             } else if (thisArg.equals("-d")) {
178                 details = true;
179             } else if (thisArg.equals("-r")) {
180                 doRecalcUtil = true;
181             } else if (thisArg.equals("-R")) {
182                 doRecalcExpired = true;
183             } else if (thisArg.equals("-V")) {
184                 System.out.println(JEVersion.CURRENT_VERSION);
185                 System.exit(0);
186             } else if (thisArg.equals("-h")) {
187                 if (argc < nArgs) {
188                     envHome = new File(argv[argc++]);
189                 } else {
190                     printUsage("-h requires an argument");
191                 }
192             } else if (thisArg.equals("-s")) {
193                 if (argc < nArgs) {
194                     startLsn = CmdUtil.readLsn(argv[argc++]);
195                 } else {
196                     printUsage("-s requires an argument");
197                 }
198             } else if (thisArg.equals("-e")) {
199                 if (argc < nArgs) {
200                     finishLsn = CmdUtil.readLsn(argv[argc++]);
201                 } else {
202                     printUsage("-e requires an argument");
203                 }
204             } else if (thisArg.equals("-t")) {
205                 if (argc < nArgs) {
206                     String s = argv[argc++];
207                     DateFormat format = new SimpleDateFormat(DATE_FORMAT);
208                     ParsePosition pp = new ParsePosition(0);
209                     Date date = format.parse(s, pp);
210                     if (date != null) {
211                         targetTime = date.getTime();
212                     } else {
213                         printUsage(
214                             "-t doesn't match format: " + DATE_FORMAT +
215                             " example: " + DATE_EXAMPLE);
216                     }
217                 } else {
218                     printUsage("-t requires an argument");
219                 }
220             }
221         }
222 
223         if (envHome == null) {
224             printUsage("-h is a required argument");
225         }
226 
227         if (doRecalcUtil && doRecalcExpired) {
228             printUsage("-r and -R cannot both be used");
229         }
