146      * An IN has just come into memory, add it to the list.
147      */
148     public void add(IN in) {
149         /* Ignore additions until the INList is enabled. */
150         if (!enabled) {
151             return;
152         }
153 
154         /* Be sure to check for BIN first, since it's a subclass of IN! */
155         if (in.isBIN()) {
156             nCachedBINs.incrementAndGet();
157             if (in.isBINDelta(false)) {
158                 nCachedBINDeltas.incrementAndGet();
159             }
160         } else {
161             nCachedUpperINs.incrementAndGet();
162         }
163 
164         /*
165          * Use putIfAbsent to ensure that we never overwrite an IN, since this
166          * can cause Btree corruption.  Throw a fatal EFE if the IN is already
167          * present to detect potential corruption bugs early. [#21686]
168          */
169         IN oldValue = ins.putIfAbsent(in, in);
170         if (oldValue != null) {
171             throw EnvironmentFailureException.unexpectedState
172                 (envImpl,
173                  "Failed adding new IN node=" + in.getNodeId() +
174                  " dbIdentity=" + System.identityHashCode(in.getDatabase()) +
175                  " db=" + in.getDatabase().dumpString(0) +
176                  "\nExisting IN node=" + oldValue.getNodeId() +
177                  " dbIdentity=" +
178                  System.identityHashCode(oldValue.getDatabase()) +
179                  " db=" + oldValue.getDatabase().dumpString(0));
180         }
181 
182         long size = in.getBudgetedMemorySize();
183         memRecalcAdd(in, size);
184         envImpl.getMemoryBudget().updateTreeMemoryUsage(size);
185         in.setInListResident(true);
186     }

206      *

209     private boolean removeInternal(IN in) {
210 
211         /* Be sure to check for BIN first, since it's a subclass of IN! */
212         if (in.isBIN()) {
213             nCachedBINs.decrementAndGet();
214             if (in.isBINDelta(false/*checkLatched*/)) {
215                 nCachedBINDeltas.decrementAndGet();
216             }
217         } else {
218             nCachedUpperINs.decrementAndGet();
219         }

222 
223         boolean latchAcquired = false;
224         if (!in.isLatchOwner()) {
225             in.latch(CacheMode.UNCHANGED);
226             latchAcquired = true;
227         }
228

230             evictor.remove(in);
231             in.setInListResident(false);
232             envImpl.getOffHeapCache().removeINFromMain(in);
233         } finally {

257     /**
258      * A direct Iterator on the INList may return INs that have been removed,
259      * since the underlying ConcurrentHashMap doesn't block changes to the list
260      * during the iteration.  This Iterator implementation wraps a direct
261      * Iterator and returns only those INs that are on the INList.
262      *
263      * Note that this doesn't guarantee that an IN will not be removed from the
264      * INList after being returned by this iterator.  But filtering out the INs
265      * already removed will avoid wasting effort in the evictor, checkpointer,
266      * and other places where INs are iterated and processed.
267      */
268     private class Iter implements Iterator<IN> {
269 
270         private final Iterator<IN> baseIter;
271         private IN next;
272         private IN lastReturned;
273 
274         private Iter() {
275             baseIter = ins.keySet().iterator();
276         }
277 
278         public boolean hasNext() {
279             if (next != null) {
280                 return true;
281             } else {
282                 return advance();
283             }
284         }
285 
286         public IN next() {
287             if (next == null) {
288                 if (!advance()) {
289                     throw new NoSuchElementException();
290                 }
291             }
292             lastReturned = next;
293             next = null;
294             return lastReturned;
295         }
296 
297         private boolean advance() {
298             while (baseIter.hasNext()) {
299                 IN in = baseIter.next();
300                 if (in.getInListResident()) {
301                     next = in;
302                     return true;
303                 }
304             }
305             return false;
306         }

347     /*
348      * The following set of memRecalc methods allow an iteration over the
349      * INList to recalculate the tree memory budget.  This is done during a
350      * checkpoint by the DirtyINMap class.
351      *
352      * We flip the INList toggle, recalcToggle, at the beginning of the recalc.
353      * At that point, if recalcConsistent is true, all IN toggles have the
354      * opposite value of recalcToggle.  As we process INs we flip their
355      * toggles.  We can tell whether we have already processed an IN by
356      * comparing its toggle to recalcToggle.  If they are equal, we have
357      * already processed the IN.
358      *
359      * The scenarios below describe how the recalcTotal is updated for a
360      * particular IN.
361      *
362      * Scenario #1: IN size is unchanged during the iteration
363      *  begin
364      *   iterate -- add total IN size, mark processed
365      *  end
366      *
367      * Scenario #2: IN size is updated during the iteration
368      *  begin
369      *   update  -- do not add delta because IN is not yet processed
370      *   iterate -- add total IN size, mark processed
371      *   update  -- do add delta because IN was already processed
372      *  end
373      *
374      * Scenario #3: IN is added during the iteration but not iterated
375      *  begin
376      *   add -- add IN size, mark processed
377      *  end
378      *
379      * Scenario #4: IN is added during the iteration and is iterated
380      *  begin
381      *   add     -- add IN size, mark processed
382      *   iterate -- do not add size because IN was already processed
383      *  end
384      *
385      * Scenario #5: IN is removed during the iteration but not iterated
386      *  begin
387      *   remove  -- do not add delta because IN is not yet processed
388      *  end
389      *
390      * Scenario #6: IN is removed during the iteration and is iterated
391      *  begin
392      *   iterate -- add total IN size, mark processed
393      *   remove  -- add delta because IN was already processed
394      *  end
395      *
396      * If recalcConsistent is false, the last attempted recalc was not
397      * compeleted.  In that case the next reset pass will simply set the toggle
398      * in every IN so that they are consistent.  The pass following that will
399      * then do a normal recalc.  At the end of any pass, we only update the
400      * memory budget if the last recalc was consistent (or this is the first
401      * recalc), and the current recalc is completed.
402      *
403      * We do not synchronize when changing state variables.  In memRecalcBegin
404      * and memRecalcEnd it is possible for an IN to be added or removed by
405      * another thread in the window between settting recalcInProgress and
406      * setting or getting the recalclTotal.  In memRecalcUpdate a similar thing
407      * can happen in the window between checking the IN toggle and adding to
408      * recalcTotal, if memRecaclcIterate is called by the checkpointer in that
409      * window. If this occurs, the reset total can be inaccurate by the amount
410      * that was changed in the window.  We have chosen to live with this
411      * possible inaccuracy rather than synchronize these methods.  We would
412      * have to synchronize every time we add/remove INs and update the size of
413      * an IN, which could introduce a new point of contention.
414      */
415

417      * We are starting the iteration of the INList.  Flip the INList toggle
418      * and set the total amount to zero.

423     public void memRecalcBegin() {
424         recalcTotal.set(0);
425         recalcInProgress = true;
426         recalcToggle = !recalcToggle;
427     }

429     /**
430      * An IN was encountered during the iteration through the entire INList.
431      * Add its size to the recalc total if we have not already processed it,
432      * and mark it as processed.  If it was already processed, memRecalcAdd
433      * must have been called for the IN when it was added to the INList during
434      * the iteration.
435      */
436     public void memRecalcIterate(IN in) {
437         assert recalcInProgress;
438         if (recalcConsistent &&
439             recalcToggle != in.getRecalcToggle()) {
440             long delta = in.resetAndGetMemorySize();
441             recalcTotal.addAndGet(delta);
442         }
443         in.setRecalcToggle(recalcToggle);
444     }
445 
446     /**
