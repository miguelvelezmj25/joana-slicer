130     private static class TimeConsistencyPolicyFormat
131         implements ConsistencyPolicyFormat<TimeConsistencyPolicy> {
132 
133         @Override
134         public String policyToString(final TimeConsistencyPolicy policy) {
135             return policy.getName() +
136                 "(" + policy.getPermissibleLag(TimeUnit.MILLISECONDS) +
137                 " ms," + policy.getTimeout(TimeUnit.MILLISECONDS) +
138                 " ms)";
139         }
140 
141         @Override
142         public TimeConsistencyPolicy stringToPolicy(final String string) {
143             /* Format: (<lag>, <timeout>) */
144             String args =
145                 string.substring(TimeConsistencyPolicy.NAME.length());
146             if (args.charAt(0) != '(' ||
147                 args.charAt(args.length()-1) != ')') {
148                 throw new IllegalArgumentException
149                     ("Incorrect property value syntax: " + string);
150             }
151             int arg1 = args.indexOf(',');
152             if (arg1 == -1) {
153                 throw new IllegalArgumentException
154                     ("Incorrect property value syntax: " + string);
155             }
156             int lag = PropUtil.parseDuration(args.substring(1, arg1));
157             int arg2 = args.indexOf(')');
158             if (arg2 == -1) {
159                 throw new IllegalArgumentException
160                     ("Incorrect property value syntax: " + string);
161             }
162             int timeout =
163                 PropUtil.parseDuration(args.substring(arg1 + 1, arg2));
164             return new TimeConsistencyPolicy
165                 (lag, TimeUnit.MILLISECONDS, timeout, TimeUnit.MILLISECONDS);
166         }
167     }

215      *
216      * @param propertyValue the formatted string representing the policy.

219      *

221      * and setter.

223     public static ReplicaConsistencyPolicy
224         getReplicaConsistencyPolicy(String propertyValue)
225         throws IllegalArgumentException {
226 
227         final String upperCasePropertyValue =
228             propertyValue.toUpperCase(java.util.Locale.ENGLISH);
229         for (final Map.Entry<String, ConsistencyPolicyFormat<?>> entry :
230                  consistencyPolicyFormats.entrySet()) {
231             final String name = entry.getKey();
232             if (upperCasePropertyValue.equals(name) ||
233                 (upperCasePropertyValue.startsWith(name) &&
234                  upperCasePropertyValue.length() > name.length() &&
235                  !Character.isLetter(
236                      upperCasePropertyValue.charAt(name.length())))) {
237                 ConsistencyPolicyFormat<?> format = entry.getValue();
238                 return format.stringToPolicy(propertyValue);
239             }
240         }
241         throw new IllegalArgumentException
242             ("Invalid consistency policy: " + propertyValue);

541      * A simple debugging utility used to obtain information about the
542      * execution environment that's only available through some system utility,
543      * like netstat, or jps, etc. It's up to the caller to ensure the
544      * availability of the utility and ensure that it's on the search path.
545      *
546      * @args the arguments to a ProcessBuilder with args[0] being the command
547      * and args[1-...] being its args
548      *
549      * @return a string denoting the output from the command. Or a string
550      * prefixed by the word EXCEPTION, if an exception was encountered,
551      * followed by the exception class name and exception message.
552      */
553     public static String exec(String... args) {
554 
555         final ByteArrayOutputStream bao = new ByteArrayOutputStream(1024);
556         final PrintStream output = new PrintStream(bao);
557 
558         try {
559 
560             final ProcessBuilder builder = new ProcessBuilder(args);
561             builder.redirectErrorStream(true);
562 
563             final Process process = builder.start();
564             final InputStream is = process.getInputStream();
565             final InputStreamReader isr = new InputStreamReader(is);
566             final BufferedReader br = new BufferedReader(isr);
567 
568             String line;
569             while ((line = br.readLine()) != null) {
570                 output.println(line);
571             }
572 
573         } catch (Exception e) {
574             output.println("EXCEPTION:" + " class:" + e.getClass().getName() +
575                            " message:" + e.getMessage());
576         }
577 
578         return bao.toString();
579     }
