81 /**
82  * An ASM ClassVisitor that examines a class, throws NotPersistentException if
83  * it is not persistent, or enhances it if it is persistent.  A class is
84  * persistent if it contains the @Entity or @Persistent annotations.  A
85  * resulting enhanced class implements the com.sleepycat.persist.impl.Enhanced
86  * interface.
87  *
88  * <p>NotPersistentException is thrown to abort the transformation in order to
89  * avoid making two passes over the class file (one to look for the annotations
90  * and another to enhance the bytecode) or outputing a class that isn't
91  * enhanced.  By aborting the transformation as soon as we detect that the
92  * annotations are missing, we make only one partial pass for a non-persistent
93  * class.</p>

96  */

241         /*
242         System.out.println("AllFields: " + nonKeyFields);
243         //*/
244         if (nonKeyFields.size() == 0) {
245             return;
246         }
247         isCompositeKey = true;
248         for (FieldInfo field : nonKeyFields) {
249             if (field.order == null) {
250                 isCompositeKey = false;
251             }
252         }
253         if (isCompositeKey) {
254             Collections.sort(nonKeyFields, new Comparator<FieldInfo>() {
255                 public int compare(FieldInfo f1, FieldInfo f2) {
256                     return f1.order.value - f2.order.value;
257                 }
258             });
259         } else {
260             for (int i = 0; i < nonKeyFields.size();) {
261                 FieldInfo field = nonKeyFields.get(i);
262                 if (field.isTransient) {
263                     nonKeyFields.remove(i);
264                 } else if (field.isPriKey) {
265                     if (priKeyField == null) {
266                         priKeyField = field;
267                         nonKeyFields.remove(i);
268                     }
269                 } else if (field.isSecKey) {
270                     secKeyFields.add(field);
271                     nonKeyFields.remove(i);
272                 } else {
273                     i += 1;
274                 }
275             }
276             Comparator<FieldInfo> cmp = new Comparator<FieldInfo>() {
277                 public int compare(FieldInfo f1, FieldInfo f2) {
278                     return f1.name.compareTo(f2.name);
279                 }
280             };
281             Collections.sort(secKeyFields, cmp);
282             Collections.sort(nonKeyFields, cmp);
283         }

286         System.out.println("SecKeys: " + secKeyFields);

927                 mv.visitLabel(labels[i]);
928                 genReadField(mv, field);
929                 if (i < nFields - 1) {
930                     Label nextCase = labels[i + 1];
931                     mv.visitVarInsn(ILOAD, 3);
932                     if (i == 0) {
933                         mv.visitJumpInsn(IFNE, nextCase);
934                     } else {
935                         switch (i) {
936                         case 1:
