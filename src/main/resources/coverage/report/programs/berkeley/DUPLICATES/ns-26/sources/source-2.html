21 /**
22  * A bag of Blocks used during the LDiff process. Blocks are accessed by their
23  * checksum; when checksums collide, blocks are returned in insertion order.
24  */
25 public class BlockBag implements Iterable<Block> {
26     /* Map checksums to the corresponding block's index in blocks. */
27     private final HashMap<Long, List<Integer>> chksums;
28     /* Maintain the list of blocks in insertion order. */
29     private final List<Block> blocks;
30 
31     /*
32      * The index in blocks of the first block that has not yet been removed.
33      * Items in blocks prior to blockIndex have been deleted from the bag.
34      */
35     private int blockIndex;
36 
37     /**
38      * Instantiate a new BlockBag object.
39      */
40     public BlockBag() {
41         blockIndex = 0;
42         blocks = new ArrayList<Block>();
43         chksums = new HashMap<Long, List<Integer>>();
44     }
45 
46     /**
47      * Adds a new Block to the bag.
48      *
49      * @param b The Block to be added.
50      */
51     public void add(Block b) {
52         final Long chksum = b.getRollingChksum();
53         final Integer indx = blocks.size();
54         blocks.add(b);
55         List<Integer> indices = chksums.get(chksum);
56         if (indices == null) {
57             indices = new ArrayList<Integer>();
58         }
59         indices.add(indx);
60         chksums.put(chksum, indices);
61     }
62 
63     /**
64      * Returns all Blocks in the bag with a given checksum.
65      *
66      * @param chksum The checksum to match
67      * @return A List of blocks with the given checksum, in insertion order, or
68      * null if no matching blocks were found.
69      */
70     public List<Block> get(long chksum) {
71         List<Integer> indices;
72         List<Block> ret;
73 
74         ret = new ArrayList<Block>();
75         indices = chksums.get(new Long(chksum));
76         if (indices == null) {
77             return null;
78         }
79         for (Integer indx : indices) {
80             int i = indx.intValue();
81             if (i >= blockIndex)
82                 ret.add(blocks.get(i));
83         }
84 
85         if (ret.size() == 0) {
86             return null;
87         }
88         return ret;
89     }

101     /**
102      * Removes the given Block, plus any blocks inserted previous to the given
103      * Block.
104      *
105      * @param b The Block to remove.
106      * @return A List of all unmatched blocks, or null
107      */
108     public List<Block> remove(Block b) {
109         final int startIndex = blockIndex;
110         while (blockIndex < blocks.size()) {
111             Block b2 = blocks.get(blockIndex);
112             blockIndex++;
113             if (b == b2) {
114                 break;
115             }
116         }
117 
118         return (blockIndex - startIndex <= 1) ? null : blocks.subList(
119                 startIndex, blockIndex - 1);
120     }

128     public List<Block> removeAll() {

132         while (blockIndex < blocks.size()) {
133             Block b = blocks.get(blockIndex);

136         }
137 
138         if (ret.size() == 0) {

140         }
141         return ret;
142     }
