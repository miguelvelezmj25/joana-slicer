38      * Creates a map of Long file number to DbFileSummary.  The init() method
39      * must be called after creating this object.
40      *
41      * <p>Always counts this object and its contained objects in the memory
42      * budget.  If countParentMapEntry is true, also counts a single HashMap
43      * entry that contains this object.  This option allows all memory budget
44      * adjustments for LocalUtilizationTracker to be contained in this

70     /**
71      * Returns the DbFileSummary for the given file, allocating it if
72      * necessary.
73      *
74      * <p>Must be called under the log write latch.</p>
75      *
76      * @param fileNum the file identifying the summary.
77      *
78      * @param adjustMemBudget see init().
79      *
80      * @param checkResurrected is true if this method should check fileNum and
81      * return null if the file does not exist. When checkResurrected is false,
82      * the expensive call to File.exists will not be made.
83      *
84      * @param fileManager is used to check for resurrected files and may be
85      * null if checkResurrected is false.
86      */
87     public DbFileSummary get(Long fileNum,
88                              boolean adjustMemBudget,
89                              boolean checkResurrected,
90                              FileManager fileManager) {
91 
92         assert adjustMemBudget == (budget != null);
93 
94         /*
95          * Note that the call below to isFileValid (which calls File.exists) is
96          * only made if the file number is less than the last file in the log,
97          * and the file is not already present in the map.  When the file is
98          * not the last file, we are recording obsoleteness and the file should
99          * already be in the map.  So we only incur the overhead of File.exists
100          * when resurrecting a file, which should be pretty rare.
101          *
102          * The reliability of this approach is questionable. Earlier we had an
103          * assertion that double-checked this condition after adding a new map
104          * entry and the assertion sometimes fired, indicating that the file
105          * was deleted during the execution of this method. Luckily, we plan
106          * to remove per-DB utilization metadata completely in the future.
107          */
108         DbFileSummary summary = map.get(fileNum);
109         if (summary == null) {
110             if (checkResurrected && 
111                 fileNum < fileManager.getCurrentFileNum() &&
112                 !fileManager.isFileValid(fileNum)) {
113                 /* Will return null. */
114             } else {
115                 summary = new DbFileSummary();
116                 Object oldVal = map.put(fileNum, summary);
117                 assert oldVal == null;
118                 memSize += FILE_ENTRY_OVERHEAD;
119                 if (adjustMemBudget) {
120                    budget.updateTreeAdminMemoryUsage(FILE_ENTRY_OVERHEAD);     
121                 }
122             }
123         }
124         return summary;
125     }

172     public Map<Long, DbFileSummary> cloneMap() {
173         final Map<Long, DbFileSummary> clone =
174             new HashMap<Long, DbFileSummary>(map.size());
175         final Iterator<Map.Entry<Long, DbFileSummary>> i =
176             map.entrySet().iterator();
177         while (i.hasNext()) {
178             final Map.Entry<Long, DbFileSummary> entry = i.next();
179             final Long fileNum = entry.getKey();

195     public void repair(EnvironmentImpl env) {
196         Long[] existingFiles = env.getFileManager().getAllFileNumbers();
197         Iterator<Long> iter = map.keySet().iterator();
198         while (iter.hasNext()) {
199             Long fileNum = iter.next();
200             if (Arrays.binarySearch(existingFiles, fileNum) < 0) {
201                 iter.remove();
202                 budget.updateTreeAdminMemoryUsage(0 - FILE_ENTRY_OVERHEAD);
