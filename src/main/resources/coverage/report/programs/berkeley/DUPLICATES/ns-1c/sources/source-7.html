75 /**
76  * FeedManager is responsible for the creation and management of the Feeders
77  * used to respond to connections initiated by a Replica. runfeeders() is the
78  * central loop that listens for replica connections and manages the lifecycle
79  * of individual Feeders. It's re-entered each time the node becomes a Master
80  * and is exited when its status changes.
81  *
82  * There is a single instance of FeederManager that is created for a
83  * replication node. There are many instances of Feeders per FeederManager.
84  * Each Feeder instance represents an instance of a connection between the node
85  * serving as the feeder and the replica.
86  *
87  * Note that the FeederManager and the Replica currently reuse the Replication
88  * node's thread of control. When we implement r2r we will need to revisit the
89  * thread management to provide for concurrent operation of the FeederManger
90  * and the Replica.
91  */

460     void activateFeeder(Feeder feeder) {
461         synchronized (nascentFeeders) {
462             synchronized (activeFeeders) {
463                 boolean removed = nascentFeeders.remove(feeder);
464                 if (feeder.isShutdown()) {
465                     return;
466                 }
467                 assert(removed);
468                 String replicaName = feeder.getReplicaNameIdPair().getName();
469                 assert(!feeder.getReplicaNameIdPair().equals(NameIdPair.NULL));
470                 Feeder dup = activeFeeders.get(replicaName);
471                 if ((dup != null) && !dup.isShutdown()) {
472                     throw EnvironmentFailureException.
473                         unexpectedState(repNode.getRepImpl(),
474                                         feeder.getReplicaNameIdPair() +
475                                         " is present in both nascent and " +
476                                         "active feeder sets");
477                 }
478                 activeFeeders.put(replicaName, feeder);
479                 if (feeder.getReplicaNode().getType().isArbiter()) {
480                     assert(arbiterFeeders.get() == 0);
481                     arbiterFeeders.incrementAndGet();
482                     arbiterFeederName = replicaName;
483 
484                 } else if (feeder.getReplicaNode().getType().isElectable()) {
485                     ackFeeders.incrementAndGet();
486                 }
487 
488                 MasterTransfer xfr = repNode.getActiveTransfer();
489                 if (xfr != null) {
490                     xfr.addFeeder(feeder);
491                 }
492             }
493         }
494     }

540     /**
541      * The core feeder listener loop that is run either in a Master node, or in
542      * a Replica that is serving as a Feeder to other Replica nodes. The core
543      * loop accepts connections from Replicas as they come in and establishes a
544      * Feeder on that connection.
545      *
546      * The loop can be terminated for one of the following reasons:
547      *
548      *  1) A change in Masters.
549      *
550      *  2) A forced shutdown, via a thread interrupt.
551      *
552      *  3) A server socket level exception.
553      *
554      * The timeout on the accept is used to ensure that the check is done at
555      * least once per timeout period.
556      */
557     void runFeeders()
558         throws DatabaseException {
559 
560         if (shutdown.get()) {
561             throw EnvironmentFailureException.unexpectedState
562                 ("Feeder manager was shutdown");
563         }
564         Exception feederShutdownException = null;
565         LoggerUtils.info(logger, repNode.getRepImpl(),
566                          "Feeder manager accepting requests.");
567 
568         /* Init GlobalCBVLSN using minJEVersion in the rep group DB. */
569         repNode.globalCBVLSN.init(repNode, repNode.getMinJEVersion());
570 
571         /* This updater represents the masters's local cbvlsn, which the master
572            updates directly. */
573         final LocalCBVLSNUpdater updater = new LocalCBVLSNUpdater(
574             repNode.getNameIdPair(), repNode.getNodeType(), repNode);
575         final LocalCBVLSNTracker tracker = repNode.getCBVLSNTracker();
576 
577         try {
578 
579             /*
580              * Ensure that the Global CBVLSN is initialized for the master when
581              * it first comes up; it's subsequently maintained in the loop
582              * below.
583              */
584             updater.updateForMaster(tracker);
585 
586             repNode.getServiceDispatcher().
587                 register(FEEDER_SERVICE, channelQueue);
588 
589             /*
590              * The Feeder is ready for business, indicate that the node is
591              * ready by counting down the latch and releasing any waiters.
592              */
593             repNode.getReadyLatch().countDown();
594 
595             while (true) {
596                 final DataChannel feederReplicaChannel =
597                     channelQueue.poll(pollTimeoutMs, TimeUnit.MILLISECONDS);
598 
599                 if (feederReplicaChannel == RepUtils.CHANNEL_EOF_MARKER) {
600                     LoggerUtils.info(logger, repNode.getRepImpl(),
601                                      "Feeder manager soft shutdown.");
602                     return;
603                 }
604 
605                 repNode.getMasterStatus().assertSync();
606                 if (feederReplicaChannel == null) {
607                     if (repNode.isShutdownOrInvalid()) {
608                         /* Timeout and shutdown request */
609                         LoggerUtils.info(logger, repNode.getRepImpl(),
610                                          "Feeder manager forced shutdown.");
611                         return;
612                     }
613

616                      * by delay updating the CBVLSN, to allow tests to control
617                      * the timing.
618                      */
619                     try {

621                             delayCBVLSNUpdateHook, repNode.getNameIdPair());
622                     } catch (IllegalStateException e) {

627                      * Take this opportunity to update this node's CBVLSN The
628                      * replicas are sending in their CBVLSNs through the
629                      * heartbeat responses, but a master does not send any
630                      * heartbeat responses, and needs a different path to
631                      * update its local CBVLSN.
632                      */
633                     updater.updateForMaster(tracker);
634

636                     dtvlsnFlusher.flush();

640                      * This must be done while the feeder is active.
641                      */
642                     repNode.globalCBVLSN.setDefunctJEVersion(repNode);
643 
644                     continue;
645                 }

649                     Feeder feeder = new Feeder(this, feederReplicaChannel);
650                     nascentFeeders.add(feeder);
651                     feeder.startFeederThreads();
652                 } catch (IOException e) {

655                      * Indicates a feeder socket level exception.
656                      */

659                          "Feeder I/O exception: " + e.getMessage());
660                     try {

662                     } catch (IOException e1) {
663                         LoggerUtils.fine
664                             (logger, repNode.getRepImpl(),
665                              "Exception during cleanup." + e.getMessage());
666                     }
667                     continue;
668                 }
669             }
670         } catch (MasterSyncException e) {
671             LoggerUtils.info(logger, repNode.getRepImpl(),
672                              "Master change: " + e.getMessage());
673 
674             feederShutdownException = new UnknownMasterException("Node " +
675                                 repNode.getRepImpl().getName() +
676                                 " is not a master anymore");
677         } catch (InterruptedException e) {
678             if (this.repNodeShutdownException != null) {
679 
680                 /*
681                  * The interrupt was issued to propagate an exception from one
682                  * of the Feeder threads. It's not a normal exit.
683                  */
684                 LoggerUtils.warning(logger, repNode.getRepImpl(),
685                                     "Feeder manager unexpected interrupt");
686                 throw repNodeShutdownException; /* Terminate the rep node */
687             }
688             if (repNode.isShutdown()) {
689                 LoggerUtils.info(logger, repNode.getRepImpl(),
690                                  "Feeder manager interrupted for shutdown");
691                 return;
692             }
693             feederShutdownException = e;
694             LoggerUtils.warning(logger, repNode.getRepImpl(),
695                                 "Feeder manager unexpected interrupt");
696         } finally {
697             repNode.resetReadyLatch(feederShutdownException);
698             repNode.getServiceDispatcher().cancel(FEEDER_SERVICE);
699             shutdownFeeders(feederShutdownException);
700             LoggerUtils.info(logger, repNode.getRepImpl(),
701                              "Feeder manager exited. CurrentTxnEnd VLSN: " +
702                              repNode.getCurrentTxnEndVLSN());
703         }
704     }

711     private void shutdownFeeders(Exception feederShutdownException) {
712         boolean changed = shutdown.compareAndSet(false, true);
713         if (!changed) {
714             return;
715         }
716 
717         try {
718             /* Copy sets for safe iteration in the presence of deletes.*/
719             final Set<Feeder> feederSet;
720             synchronized (nascentFeeders) {
721                 synchronized (activeFeeders) {
722                     feederSet = new HashSet<Feeder>(activeFeeders.values());
723                     feederSet.addAll(nascentFeeders);
724                 }
725             }
726

729                 feeder.shutdown(feederShutdownException);

737                  * don't throw any exception
738                  */
739                 activeFeeders.clear(null);
740             } else {
741                 activeFeeders.clear(feederShutdownException);
742             }
743             nascentFeeders.clear();
744         }

838     /**
839      * Update the Master's DTVLSN if we can conclude based upon the state of
840      * the replicas that the DTVLSN needs to be advanced.
841      *
842      * This method is invoked when a replica heartbeat reports a more recent
843      * txn VLSN. This (sometimes) redundant form of DTVLS update is useful in
844      * circumstances when the value could not be maintained via the usual ack
845      * response processing:
846      *
847      * 1) The application is using no ack transactions explicitly.
848      *
849      * 2) There were ack transaction timeouts due to network problems and the
850      * acks were never received or were received after the timeout had expired.
851      */
852     public void updateDTVLSN(long heartbeatVLSN) {
853 
854         final long currDTVLSN = repNode.getDTVLSN();
855         if (heartbeatVLSN <= currDTVLSN) {
856             /* Nothing to update, a lagging replica that's catching up */
857             return;
858         }
859 
860         final DurabilityQuorum durabilityQuorum = repNode.getDurabilityQuorum();
861         final int durableAckCount = durabilityQuorum.
862             getCurrentRequiredAckCount(ReplicaAckPolicy.SIMPLE_MAJORITY);
863 
864         long min = Long.MAX_VALUE;
865 
866         synchronized (activeFeeders) {
867 
868             int ackCount = 0;
869             for (Feeder feeder : activeFeeders.values()) {
870 
871                 if (!durabilityQuorum.
872                     replicaAcksQualify(feeder.getReplicaNode())) {
873                     continue;
874                 }
875 
876                 final long replicaTxnVLSN =
877                     feeder.getReplicaTxnEndVLSN().getSequence();
878 
879                 if (replicaTxnVLSN <= currDTVLSN) {
880                     continue;
881                 }
882 
883                 if (replicaTxnVLSN < min) {
884                     min = replicaTxnVLSN;
885                 }
886 
887                 if (++ackCount >= durableAckCount) {
888                     /*
889                      * If a majority of replicas have vlsns >= durable txn
890                      * vlsn, advance the DTVLSN.
891                      */
892                     repNode.updateDTVLSN(min);
893                     return;
894                 }
895             }
896 
897             /* DTVLSN unchanged. */
898             return;
899         }
900     }

921     /**
922      * Writes a null (no modifications) commit record when it detects that the
923      * DTVLSN is ahead of the persistent DTVLSN and needs to be updated.
924      *
925      * Note that without this mechanism, the in-memory DTVLSN would always be
926      * ahead of the persisted VLSN, since in general DTVLSN(vlsn) < vlsn. That
927      * is, the commit or abort log record containing the DTVLSN always has a
928      * more recent VLSN than the one it contains.
929      */
930     private class DTVLSNFlusher {
931 
932         /**
933          * The number of feeder ticks for which the in-memory DTVLSN must be
934          * stable before it's written to the log as a null TXN. We are using
935          * this "tick" indirection to avoid yet another call to the clock. A
936          * "tick" in this context is the FEEDER_MANAGER_POLL_TIMEOUT.
937          */
938         final int targetStableTicks;
939 
940         /**
941          * The number of ticks for which the DTVLSN has been stable.
942          */
943         private int stableTicks = 0;
944 
945         public DTVLSNFlusher() {
946             final int heartbeatMs = repNode.getConfigManager().
947                 getInt(RepParams.HEARTBEAT_INTERVAL);
948             targetStableTicks =
949                 (int) Math.max(1, (2 * heartbeatMs) / pollTimeoutMs);
950         }
951 
952         /**
953          * Used to track whether the DTVLSN has been stable enough to write
954          * out. While it's changing application commits and aborts are writing
955          * it out, so no need to write it here.
956          */
957         private long stableDTVLSN = VLSN.NULL_VLSN_SEQUENCE;
958 
959         /**
960          * Update each time we actually persist the DTVLSN via a null txn. It
961          * represents the DTVLSN that's been written out.
962          */
963         private long persistedDTVLSN = VLSN.NULL_VLSN_SEQUENCE;
964 
965         /* Identifies the Txn that was used to persist the DTVLSN. */
966         private long nullTxnVLSN = VLSN.NULL_VLSN_SEQUENCE;
967 
968         /**
969          * Persists the DTVLSN if necessary. The DTVLSN is persisted if the
970          * version in memory is more current than the version on disk and has
971          * not changed for targetStableTicks.
972          */
973         void flush() {
974             final long dtvlsn = repNode.getDTVLSN();
975 
976             if (dtvlsn == nullTxnVLSN) {
977                 /* Don't save VLSN from null transaction as DTVLSN */
978                 return;
979             }
980 
981             if (dtvlsn > stableDTVLSN) {
982                 stableTicks = 0;
983                 stableDTVLSN = dtvlsn;
984 
985                 /* The durable DTVLSN is being actively updated. */
986                 return;
987             }
988 
989             if (dtvlsn < stableDTVLSN) {
990                 /* Enforce the invariant that the DTVLSN cannot decrease. */
991                 throw new IllegalStateException("The DTVLSN sequence cannot decrease" +
992                                                 "current DTVLSN:" + dtvlsn +
993                                                 " previous DTVLSN:" + stableDTVLSN);
994             }
995 
996             /* DTVLSN == stableDTVLSN */
997             if (++stableTicks <= targetStableTicks) {
998                 /*
999                  * Increase the stable tick counter. it has not been stable
1000                  * long enough.
1001                  */
1002                 return;
1003             }
1004 
1005             stableTicks = 0;
1006 
1007             /* dtvlsn has been stable */
1008             if (stableDTVLSN > persistedDTVLSN) {
1009                 if (repNode.getActiveTransfer() != null) {
1010                     /*
1011                      * Don't attempt writing a transaction. while a transfer is
1012                      * in progress and txns will be blocked.
1013                      */
1014                     LoggerUtils.info(logger, repNode.getRepImpl(),
1015                                      "Skipped null txn updating DTVLSN: " +
1016                                      dtvlsn + " Master transfer in progress");
1017                     return;
1018                 }
1019                 final RepImpl repImpl = repNode.getRepImpl();
1020                 final MasterTxn nullTxn =
1021                     MasterTxn.createNullTxn(repImpl, NULL_TXN_CONFIG,
1022                                             repImpl.getNameIdPair());
1023                 /*
1024                  * We don't want to wait for any reason, if the txn fails,
1025                  * we can try later.
1026                  */
1027                 nullTxn.setTxnTimeout(1);
1028                 try {
1029                     nullTxn.commit();
1030                     LoggerUtils.fine(logger, repNode.getRepImpl(),
1031                                      "Persist DTVLSN: " + dtvlsn +
1032                                      " at VLSN: " + nullTxn.getCommitVLSN() +
1033                                      " via null transaction:" + nullTxn.getId());
1034                     nullTxnVLSN = nullTxn.getCommitVLSN().getSequence();
1035                     persistedDTVLSN = dtvlsn;
1036                     stableDTVLSN = persistedDTVLSN;
1037                 } catch (Exception e) {
1038                     nullTxn.abort();
1039                     LoggerUtils.warning(logger, repNode.getRepImpl(),
1040                                "Failed to write null txn updating DTVLSN; " +
1041                                e.getMessage());
1042                 }
1043             }
1044         }
1045     }
