72     /**
73      * Dumps the data item count of each database in the specified environment.
74      */
75     private void dumpCount() {
76 
77         /*
78          * Initialize an environment configuration, and create an environment.
79          */
80         EnvironmentConfig envConfig = new EnvironmentConfig();
81         envConfig.setReadOnly(true);
82         envConfig.setAllowCreate(false);
83         Environment env = new Environment(envHome, envConfig);
84 
85         List<String> databaseNames = new LinkedList<String>();
86         databaseNames.addAll(env.getDatabaseNames());
87         for (String dbName : databaseNames) {
88 
89             DatabaseConfig dbCfg = new DatabaseConfig();
90             dbCfg.setAllowCreate(false);
91             dbCfg.setReadOnly(true);
92             DbInternal.setUseExistingConfig(dbCfg, true);
93 
94             Database db = env.openDatabase(null, dbName, dbCfg);
95             out.println("Database: " + dbName + ", Count: " + db.count());
96             db.close();
97         }
98 
99         env.close();
100     }
