74     /*
75      * Contains the values tracked by set() and reported by calculate().
76      *
77      * To clear the values, this field is assigned a new instance.  This
78      * prevents uninitialized values when set() and clear() run concurrently.
79      * Methods that access the values (set and calculate) should assign
80      * trackedValues to a local var and perform all access using the local var,
81      * so that clear() will not impact the computation.
82      *
83      * Concurrent access by set() and calculate() is handled differently.  The
84      * numOps and numRequests fields are incremented by set() last, and are
85      * checked first by calculate().  If numOps or numRequests is zero,
86      * calculate() will return an empty Latency object.  If numOps and
87      * numRequests are non-zero, calculate() may still return latency values
88      * that are inconsistent, when set() runs concurrently.  But at least
89      * calculate() won't return uninitialized latency values.  Without
90      * synchronizing set(), this is the best we can do.  Synchronizing set()
91      * might introduce contention during CRUD operations.
92      */
93     private volatile Values trackedValues;

144     /**
145      * Calculate may be called on a stat that is concurrently updating, so
146      * while it has to be thread safe, it's a bit inaccurate when there's
147      * concurrent activity. That tradeoff is made in order to avoid the cost of
148      * synchronization during the set() method.  See {@link #trackedValues}.
149      */
150     private synchronized Latency calculate(boolean clear) {
151 
152         /*
153          * Use a local var to support concurrent access.  See {@link
154          * #trackedValues}.
155          */
156         final Values values = clear ? clearInternal() : trackedValues;
157 
158         /*
159          * Check numOps and numReqests first and return an empty Latency if
160          * either one is zero.  This ensures that we don't report partially
161          * computed values when they are zero.  This works because the other
162          * values are calculated first by set(), and numOps and numRequests are
163          * incremented last.
164          */
165         final int totalOps = values.numOps.get();
166         final int totalRequests = values.numRequests.get();
167         if (totalOps == 0 || totalRequests == 0) {
168             return new Latency(maxTrackedLatencyMillis);
169         }
170 
171         final long totalNanos = values.totalNanos.get();
172         final int nOverflow = values.requestsOverflow.get();
173         final int maxIncludingOverflow = values.maxIncludingOverflow;
174         final int minIncludingOverflow = values.minIncludingOverflow;
175 
176         final float avgMs = (float) ((totalNanos * 1e-6) / totalRequests);
177 
178         /*
179          * The 95% and 99% values will be -1 if there are no recorded latencies
180          * in the histogram.
181          */
182         int percent95 = -1;
183         int percent99 = -1;
184 
185         /*
186          * Min/max can be inaccurate because of concurrent set() calls, i.e.,
187          * values may be from a mixture of different set() calls.  Bound the
188          * min/max to the average, so they are sensible.
189          */
190         final int avgMsInt = Math.round(avgMs);
191         int max = Math.max(avgMsInt, maxIncludingOverflow);
192         int min = Math.min(avgMsInt, minIncludingOverflow);
193 
194         final int percent95Count;
195         final int percent99Count;
196         final int nTrackedRequests = totalRequests - nOverflow;
197         if (nTrackedRequests == 1) {
198             /* For one request, always include it in the 95% and 99%. */
199             percent95Count = 1;
200             percent99Count = 1;
201         } else {
202             /* Otherwise truncate: never include the last/highest request. */
203             percent95Count = (int) (nTrackedRequests * .95);
204             percent99Count = (int) (nTrackedRequests * .99);
205         }
206 
207         final int histogramLength = values.histogram.length();
208         int numRequestsSeen = 0;
209         for (int latency = 0; latency < histogramLength; latency++) {
210 
211             final int count = values.histogram.get(latency);
212 
213             if (count == 0) {
214                 continue;
215             }
216 
217             if (min > latency) {
218                 min = latency;
219             }
220 
221             if (max < latency) {
222                 max = latency;
223             }
224 
225             if (numRequestsSeen < percent95Count) {
226                 percent95 = latency;
227             }
228 
229             if (numRequestsSeen < percent99Count) {
230                 percent99 = latency;
231             }
232 
233             numRequestsSeen += count;
234         }
235 
236         saveMax = max;
237         saveMin = min;
238         saveAvg = avgMs;
239         saveNumOps = totalOps;
240         saveNumRequests = totalRequests;
241         save95 = percent95;
242         save99 = percent99;
243         saveRequestsOverflow = nOverflow;
244 
245         return new Latency(maxTrackedLatencyMillis, saveMin, saveMax, saveAvg,
246                            saveNumOps, saveNumRequests, save95, save99,
247                            saveRequestsOverflow);
248     }

258     /**
259      * Record "numRecordedOps" (one or more) operations that took place in a
260      * single request of "nanoLatency" nanos.
261      */
262     public void set(int numRecordedOps, long nanoLatency) {
263 
264         /* ignore negative values [#22466] */
265         if (nanoLatency < 0) {
266             return;
267         }
268 
269         /*
270          * Use a local var to support concurrent access.  See {@link
271          * #trackedValues}.
272          */
273         final Values values = trackedValues;
274 
275         /* Round the latency to determine where to mark the histogram. */
276         final int millisRounded =
277             (int) ((nanoLatency + (1000000l / 2)) / 1000000l);
278 
279         /* Record this latency. */
280         if (millisRounded >= maxTrackedLatencyMillis) {
281             values.requestsOverflow.incrementAndGet();
282         } else {
283             values.histogram.incrementAndGet(millisRounded);
284         }
285 
286         /*
287          * Update the min/max latency if necessary.  This is not atomic, so we
288          * loop to account for lost updates.
289          */
290         while (values.maxIncludingOverflow < millisRounded) {
291             values.maxIncludingOverflow = millisRounded;
292         }
293         while (values.minIncludingOverflow > millisRounded) {
294             values.minIncludingOverflow = millisRounded;
295         }
296 
297         /*
298          * Keep a count of latency that is precise enough to record sub
299          * millisecond values.
300          */
301         values.totalNanos.addAndGet(nanoLatency);
302 
303         /*
304          * Increment numOps and numRequests last so that calculate() won't use
305          * other uninitialized values when numOps or numRequests is zero.
306          */
307         values.numOps.addAndGet(numRecordedOps);
308         values.numRequests.incrementAndGet();
309     }
