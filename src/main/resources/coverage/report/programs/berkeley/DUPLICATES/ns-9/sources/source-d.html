27 /**
28  * Iterator over files that should be migrated by cleaning them, even if
29  * they don't need to be cleaned for other reasons.
30  *
31  * Files are migrated either because they are named in the
32  * CLEANER_FORCE_CLEAN_FILES parameter or their log version is prior to the
33  * CLEANER_UPGRADE_TO_LOG_VERSION parameter.
34  *
35  * An iterator is used rather than finding the entire set at startup to
36  * avoid opening a large number of files to examine their log version.  For
37  * example, if all files are being migrated in a very large data set, this
38  * would involve opening a very large number of files in order to read
39  * their header.  This could significantly delay application startup.
40  *
41  * Because we don't have the entire set at startup, we can't select the
42  * lowest utilization file from the set to clean next.  Inteaad we iterate
43  * in file number order to increase the odds of cleaning lower utilization
44  * files first.
45  */
46 class FilesToMigrate {
47 
48     private final EnvironmentImpl env;
49 
50     /**
51      * An array of pairs of file numbers, where each pair is a range of
52      * files to be force cleaned.  Index i is the from value and i+1 is the
53      * to value, both inclusive.
54      */
55     private long[] forceCleanFiles;
56 
57     /** Log version to upgrade to, or zero if none. */
58     private int upgradeToVersion;
59 
60     /** Whether to continue checking the log version. */
61     private boolean checkLogVersion;
62 
63     /** Whether hasNext() has prepared a valid nextFile. */
64     private boolean nextAvailable;
65 
66     /** File to return; set by hasNext() and returned by next(). */
67     private long nextFile;
68 
69     FilesToMigrate(EnvironmentImpl env) {
70         this.env = env;
71         String forceCleanProp = env.getConfigManager().get
72             (EnvironmentParams.CLEANER_FORCE_CLEAN_FILES);
73         parseForceCleanFiles(forceCleanProp);
74 
75         upgradeToVersion = env.getConfigManager().getInt
76             (EnvironmentParams.CLEANER_UPGRADE_TO_LOG_VERSION);
77         if (upgradeToVersion == -1) {
78             upgradeToVersion = LogEntryType.LOG_VERSION;
79         }
80 
81         checkLogVersion = (upgradeToVersion != 0);
82         nextAvailable = false;
83         nextFile = -1;
84     }
85 
86     /**
87      * Returns whether there are more files to be migrated.  Must be called
88      * while synchronized on the UtilizationProfile.
89      */
90     boolean hasNext(SortedMap<Long, FileSummary> fileSummaryMap,
91                     Set<Long> inProgressFiles) {
92         if (nextAvailable) {
93             /* hasNext() has returned true since the last next(). */
94             return true;
95         }
96         long foundFile = -1;
97         for (long file :
98              fileSummaryMap.tailMap(nextFile + 1).keySet()) {
99             if (inProgressFiles.contains(file)) {
100                 continue;
101             }
102             if (isForceCleanFile(file)) {
103                 /* Found a file to force clean. */
104                 foundFile = file;
105                 break;
106             } else if (checkLogVersion) {
107                 try {
108                     int logVersion =
109                         env.getFileManager().getFileLogVersion(file);
110                     if (logVersion < upgradeToVersion) {
111                         /* Found a file to migrate. */
112                         foundFile = file;
113                         break;
114                     } else {
115 
116                         /*
117                          * All following files have a log version greater
118                          * or equal to this one; stop checking.
119                          */
120                         checkLogVersion = false;
121                     }
122                 } catch (RuntimeException e) {
123                     /* Throw exception but allow iterator to continue. */
124                     nextFile = file;
125                     throw e;
126                 }
127             }
128         }
129         if (foundFile != -1) {
130             nextFile = foundFile;
131             nextAvailable = true;
132             return true;
133         } else {
134             return false;
135         }
136     }
137 
138     /**
139      * Returns the next file file to be migrated.  Must be called while
140      * synchronized on the UtilizationProfile.
141      */
142     long next(SortedMap<Long, FileSummary> fileSummaryMap,
143               Set<Long> inProgressFiles)
144         throws NoSuchElementException {
145 
146         if (hasNext(fileSummaryMap, inProgressFiles)) {
147             nextAvailable = false;
148             return nextFile;
149         } else {
150             throw new NoSuchElementException();
151         }
152     }

154     /**
155      * Returns whether the given file is in the forceCleanFiles set.
156      */
157     private boolean isForceCleanFile(long file) {
158 
159         if (forceCleanFiles != null) {
160             for (int i = 0; i < forceCleanFiles.length; i += 2) {
161                 long from = forceCleanFiles[i];
162                 long to = forceCleanFiles[i + 1];
163                 if (file >= from && file <= to) {
164                     return true;
165                 }
166             }
167         }
168         return false;
169     }

171     /**
172      * Parses the je.cleaner.forceCleanFiles property value and initializes
173      * the forceCleanFiles field.
174      *
175      * @throws IllegalArgumentException via Environment ctor and
176      * setMutableConfig.
177      */
178     private void parseForceCleanFiles(String propValue)
179         throws IllegalArgumentException {
180 
181         if (propValue == null || propValue.length() == 0) {
182             forceCleanFiles = null;
183         } else {
184             String errPrefix = "Error in " +
185                 EnvironmentParams.CLEANER_FORCE_CLEAN_FILES.getName() +
186                 "=" + propValue + ": ";
187 
188             StringTokenizer tokens = new StringTokenizer
189                 (propValue, ",-", true /*returnDelims*/);
190 
191             /* Resulting list of Long file numbers. */
192             List<Long> list = new ArrayList<Long>();
193 
194             while (tokens.hasMoreTokens()) {
195 
196                 /* Get "from" file number. */
197                 String fromStr = tokens.nextToken();
198                 long fromNum;
199                 try {
200                     fromNum = Long.parseLong(fromStr, 16);
201                 } catch (NumberFormatException e) {
202                     throw new IllegalArgumentException
203                         (errPrefix + "Invalid hex file number: " +
204                          fromStr);
205                 }
206 
207                 long toNum = -1;
208                 if (tokens.hasMoreTokens()) {
209 
210                     /* Get delimiter. */
211                     String delim = tokens.nextToken();
212                     if (",".equals(delim)) {
213                         toNum = fromNum;
214                     } else if ("-".equals(delim)) {
215 
216                         /* Get "to" file number." */
217                         if (tokens.hasMoreTokens()) {
218                             String toStr = tokens.nextToken();
219                             try {
220                                 toNum = Long.parseLong(toStr, 16);
221                             } catch (NumberFormatException e) {
222                                 throw new IllegalArgumentException
223                                     (errPrefix +
224                                      "Invalid hex file number: " +
225                                      toStr);
226                             }
227                         } else {
228                             throw new IllegalArgumentException
229                                 (errPrefix + "Expected file number: " +
230                                  delim);
231                         }
232                     } else {
233                         throw new IllegalArgumentException
234                             (errPrefix + "Expected '-' or ',': " + delim);
235                     }
236                 } else {
237                     toNum = fromNum;
238                 }
239 
240                 assert toNum != -1;
241                 list.add(Long.valueOf(fromNum));
242                 list.add(Long.valueOf(toNum));
243             }
244 
245             forceCleanFiles = new long[list.size()];
246             for (int i = 0; i < forceCleanFiles.length; i += 1) {
247                 forceCleanFiles[i] = list.get(i).longValue();
248             }
249         }
250     }
