31 /**
32  * The default annotation-based entity model.  An <code>AnnotationModel</code>
33  * is based on annotations that are specified for entity classes and their key
34  * fields.
35  *
36  * <p>{@code AnnotationModel} objects are thread-safe.  Multiple threads may
37  * safely call the methods of a shared {@code AnnotationModel} object.</p>
38  *
39  * <p>The set of persistent classes in the annotation model is the set of all
40  * classes with the {@link Persistent} or {@link Entity} annotation.</p>
41  *
42  * <p>The annotations used to define persistent classes are: {@link Entity},
43  * {@link Persistent}, {@link PrimaryKey}, {@link SecondaryKey} and {@link
44  * KeyField}.  A good starting point is {@link Entity}.</p>
45  *
46  * @author Mark Hayes
47  */
48 public class AnnotationModel extends EntityModel {

103     @Override
104     public synchronized ClassMetadata getClassMetadata(String className) {
105         ClassMetadata metadata = classMap.get(className);
106         if (metadata == null) {
107             Class<?> type;
108             try {
109                 type = resolveClass(className);
110             } catch (ClassNotFoundException e) {
111                 return null;
112             }
113             
114             /* 
115              * Adds enum or array types to registeredSpecialClasses set, and
116              * does not create metadata for them. [#19377]
117              */
118             if (type.isEnum() ||
119                 type.isArray()) {
120                 registeredSpecialClasses.add(className);
121             }
122             
123             /* Get class annotation. */
124             Entity entity = type.getAnnotation(Entity.class);
125             Persistent persistent = type.getAnnotation(Persistent.class);
126             if (entity == null && persistent == null) {
127                 return null;
128             }
129             if (type.isEnum() ||
130                 type.isInterface() ||
131                 type.isPrimitive()) {
132                 throw new IllegalArgumentException
133                     ("@Entity and @Persistent not allowed for enum, " +
134                      "interface, or primitive type: " + type.getName());
135             }
136             if (entity != null && persistent != null) {
137                 throw new IllegalArgumentException
138                     ("Both @Entity and @Persistent are not allowed: " +
139                      type.getName());
140             }
141             boolean isEntity;
142             int version;
143             String proxiedClassName;
144             if (entity != null) {
145                 isEntity = true;
146                 version = entity.version();
147                 proxiedClassName = null;
148             } else {
149                 isEntity = false;
150                 version = persistent.version();
151                 Class proxiedClass = persistent.proxyFor();
152                 proxiedClassName = (proxiedClass != void.class) ?
153                                     proxiedClass.getName() : null;
154             }
155             /* Get instance fields. */
156             List<Field> fields = new ArrayList<Field>();
157             boolean nonDefaultRules = getInstanceFields(fields, type);
158             Collection<FieldMetadata> nonDefaultFields = null;
159             if (nonDefaultRules) {
160                 nonDefaultFields = new ArrayList<FieldMetadata>(fields.size());
161                 for (Field field : fields) {
162                     nonDefaultFields.add(new FieldMetadata
163                         (field.getName(), field.getType().getName(),
164                          type.getName()));
165                 }
166                 nonDefaultFields =
167                     Collections.unmodifiableCollection(nonDefaultFields);
168             }
169             /* Get the rest of the metadata and save it. */
170             metadata = new ClassMetadata
171                 (className, version, proxiedClassName, isEntity,
172                  getPrimaryKey(type, fields),
173                  getSecondaryKeys(type, fields),
174                  getCompositeKeyFields(type, fields),
175                  nonDefaultFields);
176             classMap.put(className, metadata);
177             /* Add any new information about entities. */
178             updateEntityInfo(metadata);
179         }
180         return metadata;
181     }

183     /**
184      * Fills in the fields array and returns true if the default rules for
185      * field persistence were overridden.
186      */
187     private boolean getInstanceFields(List<Field> fields, Class<?> type) {
188         boolean nonDefaultRules = false;
189         for (Field field : type.getDeclaredFields()) {
190             boolean notPersistent =
191                 (field.getAnnotation(NotPersistent.class) != null);
192             boolean notTransient = 
193                 (field.getAnnotation(NotTransient.class) != null);
194             if (notPersistent && notTransient) {
195                 throw new IllegalArgumentException
196                     ("Both @NotTransient and @NotPersistent not allowed");
197             }
198             if (notPersistent || notTransient) {
199                 nonDefaultRules = true;
200             }
201             int mods = field.getModifiers();
202 
203             if (!Modifier.isStatic(mods) &&
204                 !notPersistent &&
205                 (!Modifier.isTransient(mods) || notTransient)) {
206                 /* Field is DPL persistent. */
207                 fields.add(field);
208             } else {
209                 /* If non-persistent, no other annotations should be used. */
210                 if (field.getAnnotation(PrimaryKey.class) != null ||
211                     field.getAnnotation(SecondaryKey.class) != null ||
212                     field.getAnnotation(KeyField.class) != null) {
213                     throw new IllegalArgumentException
214                         ("@PrimaryKey, @SecondaryKey and @KeyField not " +
215                          "allowed on non-persistent field");
216                 }
217             }
218         }
219         return nonDefaultRules;
220     }

222     private PrimaryKeyMetadata getPrimaryKey(Class<?> type,
223                                              List<Field> fields) {
224         Field foundField = null;
225         String sequence = null;
226         for (Field field : fields) {
227             PrimaryKey priKey = field.getAnnotation(PrimaryKey.class);
228             if (priKey != null) {
229                 if (foundField != null) {
230                     throw new IllegalArgumentException
231                         ("Only one @PrimaryKey allowed: " + type.getName());
232                 } else {
233                     foundField = field;
234                     sequence = priKey.sequence();
235                     if (sequence.length() == 0) {
236                         sequence = null;
237                     }
238                 }
239             }
240         }
241         if (foundField != null) {
242             return new PrimaryKeyMetadata
243                 (foundField.getName(), foundField.getType().getName(),
244                  type.getName(), sequence);
245         } else {
246             return null;
247         }
248     }

250     private Map<String, SecondaryKeyMetadata>
251         getSecondaryKeys(Class<?> type, List<Field> fields) {
252 
253         Map<String, SecondaryKeyMetadata> map = null;
254         for (Field field : fields) {
255             SecondaryKey secKey = field.getAnnotation(SecondaryKey.class);
256             if (secKey != null) {
257                 Relationship rel = secKey.relate();
258                 String elemClassName = null;
259                 if (rel == Relationship.ONE_TO_MANY ||
260                     rel == Relationship.MANY_TO_MANY) {
261                     elemClassName = getElementClass(field);
262                 }
263                 String keyName = secKey.name();
264                 if (keyName.length() == 0) {
265                     keyName = field.getName();
266                 }
267                 Class<?> relatedClass = secKey.relatedEntity();
268                 String relatedEntity = (relatedClass != void.class) ?
269                                         relatedClass.getName() : null;
270                 DeleteAction deleteAction = (relatedEntity != null) ?
271                                         secKey.onRelatedEntityDelete() : null;
272                 SecondaryKeyMetadata metadata = new SecondaryKeyMetadata
273                     (field.getName(), field.getType().getName(),
274                      type.getName(), elemClassName, keyName, rel,
275                      relatedEntity, deleteAction);
276                 if (map == null) {
277                     map = new HashMap<String, SecondaryKeyMetadata>();
278                 }
279                 if (map.put(keyName, metadata) != null) {
280                     throw new IllegalArgumentException
281                         ("Only one @SecondaryKey with the same name allowed: "
282                          + type.getName() + '.' + keyName);
283                 }
284             }
285         }
286         if (map != null) {
287             map = Collections.unmodifiableMap(map);
288         }
289         return map;
290     }

292     private String getElementClass(Field field) {
293         Class cls = field.getType();
294         if (cls.isArray()) {
295             return cls.getComponentType().getName();
296         }
297         if (Collection.class.isAssignableFrom(cls)) {
298             Type[] typeArgs = null;
299             if (field.getGenericType() instanceof ParameterizedType) {
300                 typeArgs = ((ParameterizedType) field.getGenericType()).
301                     getActualTypeArguments();
302             }
303             if (typeArgs == null ||
304                 typeArgs.length != 1 ||
305                 !(typeArgs[0] instanceof Class)) {
306                 throw new IllegalArgumentException
307                     ("Collection typed secondary key field must have a" +
308                      " single generic type argument and a wildcard or" +
309                      " type bound is not allowed: " +
310                      field.getDeclaringClass().getName() + '.' +
311                      field.getName());
312             }
313             return ((Class) typeArgs[0]).getName();
314         }
315         throw new IllegalArgumentException
316             ("ONE_TO_MANY or MANY_TO_MANY secondary key field must have" +
317              " an array or Collection type: " +
318              field.getDeclaringClass().getName() + '.' + field.getName());
319     }

321     private List<FieldMetadata> getCompositeKeyFields(Class<?> type,
322                                                       List<Field> fields) {
323         List<FieldMetadata> list = null;
324         for (Field field : fields) {
325             KeyField keyField = field.getAnnotation(KeyField.class);
326             if (keyField != null) {
327                 int value = keyField.value();
328                 if (value < 1 || value > fields.size()) {
329                     throw new IllegalArgumentException
330                         ("Unreasonable @KeyField index value " + value +
331                          ": " + type.getName());
332                 }
333                 if (list == null) {
334                     list = new ArrayList<FieldMetadata>(fields.size());
335                 }
336                 if (value <= list.size() && list.get(value - 1) != null) {
337                     throw new IllegalArgumentException
338                         ("@KeyField index value " + value +
339                          " is used more than once: " + type.getName());
340                 }
341                 while (value > list.size()) {
342                     list.add(null);
343                 }
344                 FieldMetadata metadata = new FieldMetadata
345                     (field.getName(), field.getType().getName(),
346                      type.getName());
347                 list.set(value - 1, metadata);
348             }
349         }
350         if (list != null) {
351             if (list.size() < fields.size()) {
352                 throw new IllegalArgumentException
353                     ("@KeyField is missing on one or more instance fields: " +
354                      type.getName());
355             }
356             for (int i = 0; i < list.size(); i += 1) {
357                 if (list.get(i) == null) {
358                     throw new IllegalArgumentException
359                         ("@KeyField is missing for index value " + (i + 1) +
360                          ": " + type.getName());
361                 }
362             }
363         }
364         if (list != null) {
365             list = Collections.unmodifiableList(list);
366         }
367         return list;
368     }

371      * Add newly discovered metadata to our stash of entity info.  This info
372      * is maintained as it is discovered because it would be expensive to
373      * create it on demand -- all class metadata would have to be traversed.
374      */
375     private void updateEntityInfo(ClassMetadata metadata) {
376 
377         /*
378          * Find out whether this class or its superclass is an entity.  In the
379          * process, traverse all superclasses to load their metadata -- this
380          * will populate as much entity info as possible.
381          */
382         String entityClass = null;
383         PrimaryKeyMetadata priKey = null;
384         Map<String, SecondaryKeyMetadata> secKeys =
385             new HashMap<String, SecondaryKeyMetadata>();
386         for (ClassMetadata data = metadata; data != null;) {
387             if (data.isEntityClass()) {
388                 if (entityClass != null) {
389                     throw new IllegalArgumentException
390                         ("An entity class may not be derived from another" +
391                          " entity class: " + entityClass +
392                          ' ' + data.getClassName());
393                 }
394                 entityClass = data.getClassName();
395             }
396             /* Save first primary key encountered. */
397             if (priKey == null) {
398                 priKey = data.getPrimaryKey();
399             }
400             /* Save all secondary keys encountered by key name. */
401             Map<String, SecondaryKeyMetadata> classSecKeys =
402                 data.getSecondaryKeys();
403             if (classSecKeys != null) {
404                 for (SecondaryKeyMetadata secKey : classSecKeys.values()) {
405                     secKeys.put(secKey.getKeyName(), secKey);
406                 }
407             }
408             /* Load superclass metadata. */
409             Class cls;
410             try {
411                 cls = resolveClass(data.getClassName());
412             } catch (ClassNotFoundException e) {
413                 throw DbCompat.unexpectedException(e);
414             }
415             cls = cls.getSuperclass();
416             if (cls != Object.class) {
417                 data = getClassMetadata(cls.getName());
418                 if (data == null) {
419                     throw new IllegalArgumentException
420                         ("Persistent class has non-persistent superclass: " +
421                          cls.getName());
422                 }
423             } else {
424                 data = null;
425             }
426         }
427 
428         /* Add primary and secondary key entity info. */
429         if (entityClass != null) {
430             EntityInfo info = entityMap.get(entityClass);
431             if (info == null) {
432                 info = new EntityInfo();
433                 entityMap.put(entityClass, info);
434             }
435             if (priKey == null) {
436                 throw new IllegalArgumentException
437                     ("Entity class has no primary key: " + entityClass);
438             }
439             info.priKey = priKey;
440             info.secKeys.putAll(secKeys);
441         }
442     }
