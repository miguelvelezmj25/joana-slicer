49     /*
50      * We save a set number of passed transactions for debugging information.
51      * The list is limited in size in case we end up passing a large number
52      * of transactions on our hunt for a matchpoint. Note that we save both
53      * durable and non-durable transactions to aid with debugging.
54      */
55     private final List<PassedTxnInfo> passedTxns;

58     /*
59      * We need to keep the penultimate passed txn so we can readjust the
60      * passed transaction information when the matchpoint is found. For
61      * example, suppose the log contains:
62      *   txn A commit
63      *   txn B commit
64      *   txn C commit
65      * and txn A commit is the matchpoint. The way the loop works, we'll
66      * have earliestPassedTxn = txnA, and penultimate = txn B. If the
67      * matchpoint is txnA, the log will be truncated at txnB, and we
68      * should reset earliestPassedTxn = txnB.
69      *
70      * Note that numPassedDurableCommits is a subset of numPassedCommits
71      */
72     private PassedTxnInfo earliestPassedTxn;
73     private PassedTxnInfo penultimatePassedTxn;
74     private int numPassedCommits;
75     private int numPassedDurableCommits;

114     /**
115      * At the end of the search for a matchpoint, set the matchpointLSN and
116      * adjust the debugging list of passed transactions. The matchpoint entry
117      * is just before the truncation point, and does not get truncated.
118      */
119     void setMatchpoint(long match) {
120         matchpointLSN = match;
121         if ((earliestPassedTxn != null) &&
122             (earliestPassedTxn.lsn == matchpointLSN)) {
123             numPassedCommits--;
124 
125             if (passedTxns.size() > 0) {
126                int lastSavedTxn = passedTxns.size() - 1;
127                if (passedTxns.get(lastSavedTxn).lsn == match) {
128                   passedTxns.remove(lastSavedTxn);
129                }
130                earliestPassedTxn = penultimatePassedTxn;
131             }
132         }
133     }

135     /**
136      * The reader saw a transaction commit; record relevant information. Note
137      * that we record all transactions (replicated and non-replicated) to avoid
138      * cutting off a checkpoint. Note that the stream can be a combination of
139      * log records from pre and post-dtvlsn masters. Note that we are scanning
140      * the log backwards, so we are encountering commits in decreasing VLSN
141      * sequence.
142      *
143      * @param commit the commit being passed. Note that this object is reused
144      * and so should not be saved.
145      *
146      * @param commitVLSN the VLSN associated with the commit. It could be null
147      * if the commit was local and not HA
148      *
149      * @param commitLSN the LSN at which the commit is located
150      */
151     void notePassedCommits(TxnCommit commit,
152                            VLSN commitVLSN,
153                            long commitLSN) {
154         boolean durableCommit = false;
155         final Timestamp commitTime = commit.getTime();
156         final long txnId = commit.getId();
157         final long commitDTVLSN = commit.getDTVLSN();
158 
159         if ((commitVLSN != null) && !commitVLSN.isNull()) {
160             /* A replicated txn */
161             processDTVLSN(commitVLSN, commitDTVLSN);
162 
163             if (commit.hasLoggedEntries() &&
164                 ((dtvlsn.getSequence() == VLSN.UNINITIALIZED_VLSN_SEQUENCE) ||
165                  (commitVLSN.compareTo(dtvlsn) <= 0))) {
166                 numPassedDurableCommits++;
167                 durableCommit = true;
168             }
169         }
170 
171         numPassedCommits++;
172         if (earliestPassedTxn != null) {
173             penultimatePassedTxn = earliestPassedTxn;
174         }
175         earliestPassedTxn =
176             new PassedTxnInfo(commitTime, txnId, commitVLSN, commitLSN,
177                               durableCommit);
178 
179         /*
180          * Save only a limited number of passed txns, for displaying to the log
181          */
182         if (numPassedCommits <= passedTxnLimit) {
183             passedTxns.add(earliestPassedTxn);
184         }
185     }

187     private void processDTVLSN(VLSN txnEndVLSN, final long txnEndDTVLSN) {
188         if (dtvlsn.isNull()) {
189             /*
190              * The first commit/abort record, set the dtvlsn to a
191              * non null value
192              */
193             dtvlsn = new VLSN(txnEndDTVLSN);
194         } else {
195             /*
196              * Already set, verify sequencing. Make allowance for transitions
197              * from pre to post-DTVLSN stream segments. Note that we are going
198              * backwards in the log at this point.
199              */
200             if ((txnEndDTVLSN > dtvlsn.getSequence()) &&
201                 (dtvlsn.getSequence() != VLSN.UNINITIALIZED_VLSN_SEQUENCE)) {
202                 throw new IllegalStateException
203                     ("DTVLSNs should only decrease with decreasing VLSNs." +
204                      " prev:" + dtvlsn + " next:" + txnEndDTVLSN +
205                      " commit VLSN:" + txnEndVLSN);
206             }
207         }
208     }
