16 /**
17  * Unwraps nested exceptions by calling the {@link
18  * ExceptionWrapper#getCause()} method for exceptions that implement the
19  * {@link ExceptionWrapper} interface.  Does not currently support the Java 1.4
20  * <code>Throwable.getCause()</code> method.
21  *
22  * @author Mark Hayes
23  */
24 public class ExceptionUnwrapper {
25 
26     /**
27      * Unwraps an Exception and returns the underlying Exception, or throws an
28      * Error if the underlying Throwable is an Error.
29      *
30      * @param e is the Exception to unwrap.
31      *
32      * @return the underlying Exception.
33      *
34      * @throws Error if the underlying Throwable is an Error.
35      *
36      * @throws IllegalArgumentException if the underlying Throwable is not an
37      * Exception or an Error.
38      */
39     public static Exception unwrap(Exception e) {
40 
41         Throwable t = unwrapAny(e);
42         if (t instanceof Exception) {
43             return (Exception) t;
44         } else if (t instanceof Error) {
45             throw (Error) t;
46         } else {
47             throw new IllegalArgumentException("Not Exception or Error: " + t);
48         }
49     }
50 
51     /**
52      * Unwraps an Exception and returns the underlying Throwable.
53      *
54      * @param e is the Exception to unwrap.
55      *
56      * @return the underlying Throwable.
57      */
58     public static Throwable unwrapAny(Throwable e) {
59 
60         while (true) {
61             if (e instanceof ExceptionWrapper) {
62                 Throwable e2 = ((ExceptionWrapper) e).getCause();
63                 if (e2 == null) {
64                     return e;
65                 } else {
66                     e = e2;
67                 }
68             } else {
69                 return e;
70             }
71         }
72     }
