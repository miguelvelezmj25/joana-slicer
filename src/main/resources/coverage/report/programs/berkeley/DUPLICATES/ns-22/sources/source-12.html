175      * Negotiates a protocol that both the replica and feeder can support.
176      *
177      * @return the common protocol
178      *
179      * @throws IOException
180      */
181     private Protocol negotiateProtocol()
182         throws IOException {
183 
184         final Protocol defaultProtocol =
185             Protocol.getProtocol(repImpl, replicaNameIdPair, clock,
186                                  getCurrentProtocolVersion(),
187                                  groupFormatVersion);
188         /* Send over the latest version protocol this replica can support. */
189         defaultProtocol.write(defaultProtocol.new ReplicaProtocolVersion(),
190                               namedChannel);
191 
192         /*
193          * Returns the highest level the feeder can support, or the version we
194          * just sent, if it can support that version
195          */
196         Message message = defaultProtocol.read(namedChannel);
197         if (message instanceof DuplicateNodeReject) {
198             throw new EnvironmentFailureException
199                 (repImpl,
200                  EnvironmentFailureReason.HANDSHAKE_ERROR,
201                  "A replica with the name: " +  replicaNameIdPair +
202                  " is already active with the Feeder:" + feederNameIdPair);
203         }
204 
205         FeederProtocolVersion feederVersion =
206             ((FeederProtocolVersion) message);
207         feederNameIdPair = feederVersion.getNameIdPair();
208         Protocol configuredProtocol =
209             Protocol.get(repImpl, replicaNameIdPair,
210                          clock, feederVersion.getVersion(),
211                          getCurrentProtocolVersion(), groupFormatVersion);
212         LoggerUtils.fine(logger, repImpl,
213                          "Feeder id: " + feederVersion.getNameIdPair() +
214                          "Response message: " + feederVersion.getVersion());
215         namedChannel.setNameIdPair(feederNameIdPair);
216         LoggerUtils.fine(logger, repImpl,
217                          "Channel Mapping: " + feederNameIdPair + " is at " +
218                          namedChannel.getChannel());
219 
220         if (configuredProtocol == null) {
221             /* Include JE version information [#22541] */
222             throw new EnvironmentFailureException
223                 (repImpl,
224                  EnvironmentFailureReason.PROTOCOL_VERSION_MISMATCH,
225                  "Incompatible protocol versions. " +
226                  "Protocol version: " + feederVersion.getVersion() +
227                  " introduced in JE version: " +
228                  Protocol.getProtocolJEVersion(feederVersion.getVersion()) +
229                  " requested by the Feeder: " + feederNameIdPair +
230                  " is not supported by this Replica: " + replicaNameIdPair +
231                  " with protocol version: " + defaultProtocol.getVersion() +
232                  " introduced in JE version: " +
233                  Protocol.getProtocolJEVersion(defaultProtocol.getVersion()));
234         }
235         return configuredProtocol;
236     }

239      * Executes the replica side of the handshake.
240      * @throws ProtocolException
241      */
242     public Protocol execute()
243         throws IOException,
244                ProtocolException {
245 
246         LoggerUtils.info(logger, repImpl,
247                          "Replica-feeder handshake start");
248 
249         /* First negotiate the protocol, then use it. */
250         protocol = negotiateProtocol();
251 
252         /* Ensure that software versions are compatible. */
253         verifyVersions();
254 
255         /**
256          * Note whether log entries with later log versions need to be
257          * converted to log version 12 to work around [#25222].
258          */
259         if (feederJEVersions.getLogVersion() == LOG_VERSION_EXPIRE_INFO) {
260             protocol.setFixLogVersion12Entries(true);
261         }
262 
263         /*
264          * Now perform the membership information validation part of the
265          * handshake
266          */
267         verifyMembership();
268 
269         checkClockSkew();
270 
271         LoggerUtils.info(logger, repImpl,
272                          "Replica-feeder " + feederNameIdPair.getName() +
273                          " handshake completed.");
274         return protocol;
275     }

277     /**
278      * Checks software and log version compatibility.
279      */
280     private void verifyVersions()
281         throws IOException {
282 
283         protocol.write(protocol.new
284                        ReplicaJEVersions(getCurrentJEVersion(),
285                                          getCurrentLogVersion()),
286                        namedChannel);
287         Message message = protocol.read(namedChannel);
288         if (message instanceof JEVersionsReject) {
289             /* The software version is not compatible with the Feeder. */
290             throw new EnvironmentFailureException
291                 (repImpl,
292                  EnvironmentFailureReason.HANDSHAKE_ERROR,
293                  " Feeder: " + feederNameIdPair + ". " +
294                  ((JEVersionsReject) message).getErrorMessage());
295         }
296 
297         /*
298          * Save the version information in case we want to use it as the basis
299          * for further compatibility checking in future.
300          */
301         feederJEVersions = (FeederJEVersions) message;
302 
303         if (feederJEVersions.getLogVersion() > getCurrentLogVersion()) {
304             throw new EnvironmentFailureException(
305                 repImpl,
306                 EnvironmentFailureReason.HANDSHAKE_ERROR,
307                 " Feeder: " + feederNameIdPair + ". " +
308                 "Feeder log version " + feederJEVersions.getLogVersion() +
309                 " is not known to the replica, whose current log version is " +
310                 getCurrentLogVersion());
311         }

315      * Exchange membership information messages.
316      */
317     private void verifyMembership()
318         throws IOException {
319 
320         DbConfigManager configManager = repImpl.getConfigManager();
321         String groupName = configManager.get(GROUP_NAME);
322 
323         Message message = protocol.new
324             NodeGroupInfo(groupName,
325                           repGroup.getUUID(),
326                           replicaNameIdPair,
327                           repImpl.getHostName(),
328                           repImpl.getPort(),
329                           nodeType,
330                           repImpl.isDesignatedPrimary(),
331                           getCurrentJEVersion());
332         protocol.write(message, namedChannel);
333 
334         message = protocol.read(namedChannel);
335 
336         if (message instanceof NodeGroupInfoReject) {
337             NodeGroupInfoReject reject = (NodeGroupInfoReject) message;
338             throw new EnvironmentFailureException
339                 (repImpl,
340                  EnvironmentFailureReason.HANDSHAKE_ERROR,
341                  " Feeder: " + feederNameIdPair + ". " +
342                  reject.getErrorMessage());
343         }
344 
345         if (!(message instanceof NodeGroupInfoOK)) {
346             throw new EnvironmentFailureException
347                 (repImpl,
348                  EnvironmentFailureReason.HANDSHAKE_ERROR,
349                  " Feeder: " + feederNameIdPair + ". " +
350                  "Protocol error. Unexpected response " + message);
351         }
352         final NodeGroupInfoOK nodeGroupInfoOK = (NodeGroupInfoOK) message;
353         if (repGroup.hasUnknownUUID()) {
354             /* Correct the initial UUID */
355             repGroup.setUUID(nodeGroupInfoOK.getUUID());
356         }
357         if (nodeType.hasTransientId()) {
358             /* Update the transient node's ID */
359             replicaNameIdPair.update(nodeGroupInfoOK.getNameIdPair());
360         }
361     }

363     /**
364      * Checks for clock skew wrt the current feeder. It's important that the
365      * clock skew be within an acceptable range so that replica can meet any
366      * time based consistency requirements requested by transactions. The
367      * intent of this check is to draw the attention of the application or the
368      * administrators to the skew, not correct it.
369      * <p>
370      * The scheme implemented below is a variation on the scheme used by <a
371      * href="http://tools.ietf.org/html/rfc2030">SNTP</a> protocol. The Feeder
372      * plays the role of the SNTP server and the replica the role of the client
373      * in this situation. The mechanism used here is rough and does not
374      * guarantee the detection of a clock skew, especially since it's a one
375      * time check done each time a connection is re-established with the
376      * Feeder. The clocks could be in sync at the time of this check and drift
377      * apart over the lifetime of the connection. It's also for this reason
378      * that we do not store the skew value and make compensations using it when
379      * determining replica consistency.
380      * <p>
381      * Replications nodes should therefore ensure that they are using NTP or a
382      * similar time synchronization service to keep time on all the replication
383      * nodes in a group in sync.
384      * <p>
385      *
386      * @throws IOException
387      * @throws EnvironmentFailureException
388      * @throws ProtocolException
389      */
390     private void checkClockSkew()
391         throws IOException,
392                ProtocolException {
393 
394         boolean isLast = false;
395         int sampleCount = 0;
396         do {
397             if (checkClockSkew) {
398                 /* Iterate until we have a value that's good enough. */
399                 isLast = (++sampleCount >= CLOCK_SKEW_MAX_SAMPLE_SIZE) ||
400                          (clockDelay <= CLOCK_SKEW_MIN_DELAY_MS);
401             } else {
402                 isLast = true;
403             }
404 
405             protocol.write(protocol.new SNTPRequest(isLast), namedChannel);
406             SNTPResponse response = protocol.read(namedChannel,
407                                                   SNTPResponse.class);
408             if (response.getDelay() < clockDelay) {
409                 clockDelay = response.getDelay();
410                 clockDelta = response.getDelta();
411             }
412 
413         } while (!isLast);
414 
415         if (!checkClockSkew) {
416             return;
417         }
418 
419         LoggerUtils.logMsg
420             (logger, repImpl,
421              (Math.abs(clockDelta) >= maxClockDelta) ?
422              Level.SEVERE : Level.FINE,
423              "Round trip delay: " + clockDelay + " ms. " + "Clock delta: " +
424              clockDelta + " ms. " + "Max permissible delta: " +
425              maxClockDelta + " ms.");
426 
427         if (Math.abs(clockDelta) >= maxClockDelta) {
428             throw new EnvironmentFailureException
429                 (repImpl,
430                  EnvironmentFailureReason.HANDSHAKE_ERROR,
431                  "Clock delta: " + clockDelta + " ms. " +
432                  "between Feeder: " + feederNameIdPair.getName() +
433                  " and this Replica exceeds max permissible delta: " +
434                  maxClockDelta + " ms.");
435         }
436     }
