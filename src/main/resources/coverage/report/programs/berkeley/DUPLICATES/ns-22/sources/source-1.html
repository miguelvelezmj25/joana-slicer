57     public void addCommit(LogItem commitItem) {
58 
59         if (!queue.offer(commitItem)) {
60 
61             /*
62              * If the commit could not be added to the queue because
63              * the queue is filled. Try to remove an item
64              * and replace with the item with the higher VLSN.
65              * The Arbiter ack for the higher VLSN is sufficient
66              * for transactions with a lower commit VLSN.
67              */
68             nQueueFull.increment();
69             try {
70                 LogItem queuedItem = queue.remove();
71                 VLSN vlsn = commitItem.header.getVLSN();
72                 if (queuedItem.header.getVLSN().compareTo(vlsn) > 0) {
73 
74                     /*
75                      * The removed item has higher vlsn so use that one.
76                      */
77                     commitItem = queuedItem;
78                 }
79             } catch (NoSuchElementException noe) {
80                 /* Queue was empty so try to insert one last time. */
81             }
82 
83             /*
84              * Attempt to put the item on the queue. If another
85              * thread has inserted and the queue is full, we will
86              * skip this transaction for an Arbiter ack attempt. The
87              * transaction may still succeed in this case due to acks from
88              * Replicas or other Arbiter acked transactions with a higher
89              * VLSN.
90              */
91             queue.offer(commitItem);
92         }
93     }
