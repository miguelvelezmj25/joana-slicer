25 /**
26  * A consistency policy which ensures that the environment on a Replica node is
27  * at least as current as denoted by the specified {@link CommitToken}. This
28  * token represents a point in the serialized transaction schedule created by
29  * the master. In other words, this token is like a bookmark, representing a
30  * particular transaction commit in the replication stream. The Replica ensures
31  * that the commit identified by the {@link CommitToken} has been executed on
32  * this node before allowing the application's {@link
33  * com.sleepycat.je.Environment#beginTransaction(com.sleepycat.je.Transaction,
34  * com.sleepycat.je.TransactionConfig) Environment.beginTransaction()}
35  * operation on the Replica to proceed.
36  * <p>
37  * For example, suppose the application is a web application where a replicated
38  * group is implemented within a load balanced web server group. Each request
39  * to the web server consists of an update operation followed by read
40  * operations (say from the same client), The read operations naturally expect
41  * to see the data from the updates executed by the same request. However, the
42  * read operations might have been routed to a node that did not execute the
43  * update.
44  * <p>
45  * In such a case, the update request would generate a {@link CommitToken},
46  * which would be resubmitted by the browser, along with subsequent read
47  * requests. The read request could be directed at any one of the available web
48  * servers by a load balancer. The node which executes the read request would
49  * create a CommitPointConsistencyPolicy with that {@link CommitToken} and use
50  * it at transaction begin. If the environment at the web server was already
51  * current (wrt the commit token), it could immediately execute the transaction
52  * and satisfy the request. If not, the "transaction begin" would stall until
53  * the Replica replay had caught up and the change was available at that web
54  * server.
55  * <p>
56  * Consistency policies are specified at either a per-transaction level through
57  * {@link com.sleepycat.je.TransactionConfig#setConsistencyPolicy} or as an
58  * replication node wide default through {@link
59  * ReplicationConfig#setConsistencyPolicy}
60  *
61  * @see com.sleepycat.je.CommitToken
62  * @see <a href="{@docRoot}/../ReplicationGuide/consistency.html"
63  * target="_top">Managing Consistency</a>
64  */
65 public class CommitPointConsistencyPolicy implements ReplicaConsistencyPolicy {

85      * Defines how current a Replica needs to be in terms of a specific
86      * transaction that was committed on the Master. A transaction on the
87      * Replica that uses this consistency policy is allowed to start only
88      * after the transaction identified by the <code>commitToken</code> has
89      * been committed on the Replica. The {@link
90      * com.sleepycat.je.Environment#beginTransaction(
91      * com.sleepycat.je.Transaction, com.sleepycat.je.TransactionConfig)

93      * <code>timeout</code> for the Replica to catch up. If the Replica has
94      * not caught up in this period, the <code>beginTransaction()</code>
95      * method will throw a {@link ReplicaConsistencyException}.
96      *
97      * @param commitToken the token identifying the transaction
98      *
99      * @param timeout the maximum amount of time that the transaction start
100      * will wait to allow the Replica to catch up.
101      *
102      * @param timeoutUnit the {@code TimeUnit} for the timeout parameter.
103      *
104      * @throws IllegalArgumentException if the commitToken or timeoutUnit is
105      * null.
106      */
107     public CommitPointConsistencyPolicy(CommitToken commitToken,
108                                         long timeout,
109                                         TimeUnit timeoutUnit) {
110         if (commitToken == null) {
111             throw new IllegalArgumentException("commitToken must not be null");
112         }
113         this.commitToken = commitToken;
114         this.timeout = PropUtil.durationToMillis(timeout, timeoutUnit);
115     }

127      * @hidden
128      * For internal use only.
129      * Ensures that the replica has replayed the replication stream to the
130      * point identified by the commit token or past it. If it has not, the
131      * method waits until the constraint is satisfied, or the timeout period
132      * has expired, whichever event takes place first.
133      */
134     @Override
135     public void ensureConsistency(EnvironmentImpl envImpl)
136         throws InterruptedException,
137                ReplicaConsistencyException {
138 
139         /*
140          * Cast is done to preserve replication/non replication code
141          * boundaries.
142          */
143         RepImpl repImpl = (RepImpl) envImpl;
144         if (!commitToken.getRepenvUUID().equals
145                 (repImpl.getRepNode().getUUID())) {
146             throw new IllegalArgumentException
147                 ("Replication environment mismatch. " +
148                  "The UUID associated with the commit token is: " +
149                  commitToken.getRepenvUUID() +
150                  " but this replica environment has the UUID: " +
151                  repImpl.getRepNode().getUUID());
152         }
153         Replica replica = repImpl.getRepNode().replica();
154         replica.getConsistencyTracker().awaitVLSN
155             (commitToken.getVLSN(), this);
156     }
