106     /**
107      * Given a full version BIN, apply the deltas.
108      */
109     public void reconstituteBIN(DatabaseImpl dbImpl, BIN fullBIN) {
110 
111         fullBIN.setDatabase(dbImpl);
112         fullBIN.latch(CacheMode.UNCHANGED);
113         try {
114 
115             /*
116              * The BIN's lastFullLsn is set here, while its lastLoggedLsn is
117              * set by postFetchInit or postRecoveryInit.
118              */
119             fullBIN.setLastFullLsn(lastFullLsn);
120 
121             /* Process each delta. */
122             for (int i = 0; i < deltas.size(); i++) {
123                 final DeltaInfo info = deltas.get(i);
124                 fullBIN.applyDelta(
125                     info.getKey(), null/*data*/, info.getLsn(),
126                     info.getState(), 0 /*lastLoggedSize*/, 0 /*memId*/,
127                     VLSN.NULL_VLSN_SEQUENCE, null /*child*/,
128                     0 /*expiration*/, false /*expirationInHours*/);
129             }
130 
131             /*
132              * The applied deltas will leave some slots dirty, which is
133              * necessary as a record of changes that will be included in the
134              * next delta.  However, the BIN itself should not be dirty,
135              * because this delta is a persistent record of those changes.
136              */
137             fullBIN.setDirty(false);
138         } finally {
139             fullBIN.releaseLatch();
140         }
141     }

157     public void readFromLog(ByteBuffer itemBuffer, int entryVersion) {
158         dbId.readFromLog(itemBuffer, entryVersion);
159         lastFullLsn = LogUtils.readLong(itemBuffer, (entryVersion < 6));
160         if (entryVersion >= 8) {
161             prevDeltaLsn = LogUtils.readPackedLong(itemBuffer);
162         }
163         int numDeltas = LogUtils.readInt(itemBuffer, (entryVersion < 6));
164 
165         for (int i=0; i < numDeltas; i++) {
166             DeltaInfo info = new DeltaInfo();
167             info.readFromLog(itemBuffer, entryVersion);
168             deltas.add(info);
169         }
170 
171         /* Use minimum memory. */
172         deltas.trimToSize();
173     }
