28 /**
29  * Generates a unique sequence of ascending proposal numbers that is unique
30  * across all machines.
31  *
32  * Each proposal number is built as the catenation of the following components:
33  *
34  * ms time (8 bytes) | machineId (16 bytes) | locally unique Id (4 bytes)
35  *
36  * The ms time supplies the increasing number and the IP address is a number
37  * unique across machines.
38  *
39  * The machineId is generated as described below.
40  *
41  * The locally unique Id is used to allow for multiple unique proposal
42  * generators in a single process.
43  */
44 public class TimebasedProposalGenerator {
45 
46     /*
47      * A number that is unique for all instances of the TimeBasedGenerator on
48      * this machine.
49      */
50     private final int locallyUniqueId;
51     private static final AtomicInteger uniqueIdGenerator = new AtomicInteger(1);
52 
53     /*
54      * Tracks the time (in ms) used to generate the previous proposal
55      * preventing the creation of duplicate proposals.  Synchronize on this
56      * instance when accessing this field.
57      */
58     private long prevProposalTime = System.currentTimeMillis();
59 
60     /*
61      * A unique ID for this JVM, using a hex representation of the IP address
62      * XOR'ed with a random value. If the IP address cannot be determined,
63      * a secure random number is generated and used instead. The risk of
64      * collision is very low since the number of machines in a replication
65      * group is typically small, in the 10s at most.
66      */
67     private static final String machineId;
68 
69     /* Width of each field in the Proposal number in hex characters. */
70     final static int TIME_WIDTH = 16;
71 
72     /* Allow for 16 byte ipv6 addresses. */
73     final static int ADDRESS_WIDTH =32;
74     final static int UID_WIDTH = 8;
75 
76     /*
77      * Initialize machineId, do it just once to minimize latencies in the face
78      * of misbehaving networks that slow down calls to getLocalHost()
79      */
80     static  {
81 
82         InetAddress localHost;
83         try {
84             localHost = InetAddress.getLocalHost();
85         } catch (UnknownHostException e) {
86             /*
87              * Likely a misconfigured machine if it could not determine
88              * localhost.
89              */
90             localHost = null;
91         }
92         byte[] localAddress = null;
93         if (localHost != null) {
94             localAddress = localHost.getAddress();
95 
96             if (localHost.isLoopbackAddress()) {
97                 /* Linux platforms return a loopback address, examine the
98                  * interfaces individually for a suitable address.
99                  */
100                 localAddress = null;
101                 try {
102                     for (Enumeration<NetworkInterface> interfaces =
103                         NetworkInterface.getNetworkInterfaces();
104                         interfaces.hasMoreElements();) {
105                         for (Enumeration<InetAddress> addresses =
106                             interfaces.nextElement().getInetAddresses();
107                             addresses.hasMoreElements();) {
108                             InetAddress ia = addresses.nextElement();
109                             if (! (ia.isLoopbackAddress() ||
110                                    ia.isAnyLocalAddress() ||
111                                    ia.isMulticastAddress())) {
112                                 /* Found one, any one of these will do. */
113                                 localAddress = ia.getAddress();
114                                 break;
115                             }
116                         }
117                     }
118                 } catch (SocketException e) {
119                     /* Could not get the network interfaces, give up */
120                 }
121             }
122         }
123 
124         if (localAddress != null) {
125             /*
126              * Convert the address to a positive integer, XOR it with a
127              * random value of the right size, and format in hex
128              */
129             final BigInteger addrVal = new BigInteger(1, localAddress);
130             final BigInteger randVal =
131                 new BigInteger(ADDRESS_WIDTH * 4, new SecureRandom());
132             machineId = String.format("%0" + ADDRESS_WIDTH + "x",
133                                       addrVal.xor(randVal));
134         } else {
135             /*
136              * If the localAddress is null, this host is likely disconnected,
137              * or localHost is misconfigured, fall back to using just a secure
138              * random number.
139              */
140             final BigInteger randVal =
141                 new BigInteger(ADDRESS_WIDTH * 4, new SecureRandom());
142             machineId = String.format("%0" + ADDRESS_WIDTH + "x", randVal);
143         }
144     }

165     /**
166      * Returns the next Proposal greater than all previous proposals returned
167      * on this machine.
168      *
169      * @return the next unique proposal
170      */
171     public Proposal nextProposal() {
172         long proposalTime = System.currentTimeMillis();
173         synchronized (this) {
174             if (proposalTime <= prevProposalTime) {
175                 /* Proposals are moving faster than the clock. */
176                 proposalTime = ++prevProposalTime;
177             }
178             prevProposalTime = proposalTime;
179         }
180         return new StringProposal(String.format("%016x%s%08x", proposalTime,
181                                                 machineId, locallyUniqueId));
182     }
