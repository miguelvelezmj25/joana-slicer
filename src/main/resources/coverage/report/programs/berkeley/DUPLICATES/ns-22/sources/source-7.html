30 /**
31  * The FeederSyncupReader scans the log backwards for requested log entries.
32  * It uses the vlsnIndex to optimize its search, repositioning when a concrete
33  * vlsn->lsn mapping is available.
34  *
35  * The FeederSyncupReader is not thread safe, and can only be used serially. It
36  * will stop at the finishLsn, which should be set using the GlobalCBVLSN.
37  */
38 public class FeederSyncupReader extends VLSNReader {
39     /* The scanner is a cursor over the VLSNIndex. */
40     private final BackwardVLSNScanner scanner;
41 
42     public FeederSyncupReader(EnvironmentImpl envImpl,
43                               VLSNIndex vlsnIndex,
44                               long endOfLogLsn,
45                               int readBufferSize,
46                               VLSN startVLSN,
47                               long finishLsn)
48         throws IOException, DatabaseException {
49 
50         /*
51          * If we go backwards, endOfFileLsn and startLsn must not be null.
52          * Make them the same, so we always start at the same very end.
53          */
54         super(envImpl,
55               vlsnIndex,
56               false,           // forward
57               endOfLogLsn,
58               readBufferSize,
59               finishLsn);
60         scanner = new BackwardVLSNScanner(vlsnIndex);
61         initScan(startVLSN);
62     }
63 
64     /**
65      * Set up the FeederSyncupReader to start scanning from this VLSN. If we
66      * find a mapping for this VLSN, we'll start precisely at its LSN, else
67      * we'll have to start from an earlier location.
68      *
69      * @throws InterruptedException
70      * @throws IOException
71      * @throws DatabaseException
72      */
73     private void initScan(VLSN startVLSN)
74         throws DatabaseException, IOException {
75 
76         if (startVLSN.equals(VLSN.NULL_VLSN)) {
77             throw EnvironmentFailureException.unexpectedState
78                 ("FeederSyncupReader start can't be NULL_VLSN");
79         }
80 
81         VLSN startPoint = startVLSN;
82         startLsn = scanner.getStartingLsn(startPoint);
83         assert startLsn != NULL_LSN;
84 
85         /*
86          * Flush the log so that syncup can assume that all log entries that
87          * are represented in the VLSNIndex  are safely out of the log buffers
88          * and on disk. Simplifies this reader, so it can use the regular
89          * ReadWindow, which only works on a file.
90          */
91         envImpl.getLogManager().flushNoSync();
92 
93         window.initAtFileStart(startLsn);
94         currentEntryPrevOffset = window.getEndOffset();
95         currentEntryOffset = window.getEndOffset();
96         currentVLSN = startVLSN;
97     }

99     /**
100      * Backward scanning for records for the feeder's part in syncup.
101      * @throws ChecksumException 
102      * @throws FileNotFoundException 
103      */
104     public OutputWireRecord scanBackwards(VLSN vlsn)
105         throws FileNotFoundException, ChecksumException {
106 
107         VLSNRange range = vlsnIndex.getRange();
108         if (vlsn.compareTo(range.getFirst()) < 0) {
109             /*
110              * The requested VLSN is before the start of our range, we don't
111              * have this record.
112              */
113             return null;
114         }
115 
116         currentVLSN = vlsn;
117 
118         /*
119          * If repositionLsn is not NULL_LSN, the reader will seek to that
120          * position when calling readNextEntry instead of scanning.
121          * setPosition() is a noop if repositionLsn is null.
122          */
123         long repositionLsn = scanner.getPreciseLsn(vlsn);
124         setPosition(repositionLsn);
125 
126         if (readNextEntry()) {
127             return currentFeedRecord;
128         }
129 
130         return null;
131     }
132 
133     /**
134      * @throw an EnvironmentFailureException if we were scanning for a
135      * particular VLSN and we have passed it by.
136      */
137     private void checkForPassingTarget(int compareResult) {
138 
139         if (compareResult < 0) {
140             /* Hey, we passed the VLSN we wanted. */
141             throw EnvironmentFailureException.unexpectedState
142                 ("want to read " + currentVLSN + " but reader at " +
143                  currentEntryHeader.getVLSN());
144         }
145     }

147     @Override
148     protected boolean isTargetEntry()
149         throws DatabaseException {
150 
151         nScanned++;
152 
153         /* Skip invisible entries. */
154         if (currentEntryHeader.isInvisible()) {
155             return false;
156         }
157 
158         /* 
159          * Return true if this entry is replicated and its VLSN is currentVLSN.
160          */
161         if (entryIsReplicated()) {
162             VLSN entryVLSN = currentEntryHeader.getVLSN();
163             int compareResult = entryVLSN.compareTo(currentVLSN);
164             checkForPassingTarget(compareResult);
165 
166             /* return true if this is the entry we want. */
167             return (compareResult == 0);
168         }
169 
170         return false;
171     }

173     /**
174      * Instantiate a WireRecord to house this log entry.
175      */
176     @Override
177     protected boolean processEntry(ByteBuffer entryBuffer) {
178 
179         ByteBuffer buffer = entryBuffer.slice();
180         buffer.limit(currentEntryHeader.getItemSize());
181         currentFeedRecord =
182             new OutputWireRecord(envImpl, currentEntryHeader, buffer);
183 
184         entryBuffer.position(entryBuffer.position() +
185                              currentEntryHeader.getItemSize());
186         return true;
187     }
