55  * Establish where the replication stream should start for a replica and feeder
56  * pair. The replica compares what is in its log with what the feeder has, to
57  * determine the latest common log entry matchpoint
58  *
59  * - If the replica has applied log entries after that matchpoint, roll them
60  *   back

62  *   a network restore.

207     /**
208      * A matchpoint has been found. What happens next depends on the position
209      * of the matchpoint and its relationship to the last transaction end
210      * record.
211      *
212      * In following table,
213      *    M = some non-null matchpoint VLSN value,
214      *    T = some non-null last txn end value
215      *    S = some non-null last sync value
216      *
217      * Note that currently T == S, since a sync points is always a txn end.
218      *
219      * txn end                last sync   found        action
220      *  VLSN                  VLSN        matchpoint
221      * ----------             ---------   ---------    ------------------------
222      * NULL_VLSN              NULL_VLSN   NULL_VLSN    rollback everything
223      * NULL_VLSN              NULL_VLSN      M         can't occur
224      * NULL_VLSN                 S        NULL_VLSN    rollback everything
225      * NULL_VLSN                 S           M         rollback to M
226      *   T                    NULL_VLSN   NULL_VLSN    can't occur
227      *   T                    NULL_VLSN      M         can't occur
228      *   T                       S        NULL_VLSN    network restore, though
229      *                                                 could also do hard recov
230      *   T <= M                  S           M         rollback to matchpoint
231      *   T > M, truncate not ok  S           M         network restore
232      *   T > M, rollback limit
233      *          exceeded or      S           M         throw RollbackProhibited
234      *          disabled
235      *   T > M, truncate ok      S           M         hard recovery
236      *
237      * @throws IOException
238      * @throws HardRecoveryElectionException
239      */
240     private void verifyRollback(VLSNRange range)
241         throws RollbackException, InsufficientLogException,
242         HardRecoveryElectionException, IOException {
243         repImpl.setSyncupProgress(SyncupProgress.CHECK_FOR_ROLLBACK);
244         VLSN lastTxnEnd = range.getLastTxnEnd();
245         VLSN lastSync = range.getLastSync();
246 
247         LoggerUtils.finest(logger, repImpl, "verify rollback" +
248                            " vlsn range=" + range +
249                            " searchResults=" + searchResults);
250 
251         /* Test a rollback exception is thrown when sync up */
252         TestHookExecute.doHookIfSet(rollbackHook, this);
253 
254         /*
255          * If the lastTxnEnd VLSN is null, we don't have to worry about hard
256          * recovery. See truth table above.
257          */
258         if (lastTxnEnd.isNull()) {
259             if (range.getLastSync().isNull() && !matchpointVLSN.isNull()) {
260                 throw EnvironmentFailureException.unexpectedState
261                     (repNode.getRepImpl(), "Shouldn't be possible to find a "+
262                      "matchpoint of " + matchpointVLSN +
263                      " when the sync VLSN is null. Range=" + range);
264             }
265 
266             /* We'll be doing a normal rollback. */
267             LoggerUtils.fine(logger, repImpl, "normal rollback, no txn end");
268             return;
269         }
270 
271         if (lastSync.isNull()) {
272             throw EnvironmentFailureException.unexpectedState
273                 (repNode.getRepImpl(),
274                  "Shouldn't be possible to have a null sync VLSN when the "
275                  + " lastTxnVLSN " + lastTxnEnd + " is not null. Range=" +
276                  range);
277         }
278 
279         /*
280          * There is a non-null lastTxnEnd VLSN, so check if the found
281          * matchpoint precedes it. If it doesn't, we can't rollback.
282          */
283         if (matchpointVLSN.isNull()) {
284 
285             /*
286              * We could actually also try to do a hard recovery and truncate
287              * all committed txns, but for now, let's assume that it will cost
288              * less to copy the log files.
289              */
290             LoggerUtils.info(logger, repImpl,
291                              "This node had a txn end at vlsn = " + lastTxnEnd +
292                              "but no matchpoint found.");
293             throw setupLogRefresh(matchpointVLSN);
294         }
295 
296         /*
297          * The matchpoint is after or equal to the last txn end, no problem
298          * with doing a normal rollback.
299          */
300         if ((lastTxnEnd.compareTo(matchpointVLSN) <= 0) &&
301             (searchResults.getNumPassedCommits() == 0)) {
302             LoggerUtils.fine(logger, repImpl, "txn end vlsn of " + lastTxnEnd +
303                              "<= matchpointVLSN of " + matchpointVLSN +
304                              ", normal rollback");
305             return;
306         }
307 
308         /* Rolling back past a commit or abort. */
309 
310         if (hardRecoveryNeedsElection) {
311             throw new HardRecoveryElectionException
312                     (repNode.getMasterStatus().getNodeMasterNameId(),
313                      lastTxnEnd,  matchpointVLSN);
314         }
315 
316         /*
317          * We're planning on rolling back past a commit or abort. The more
318          * optimal course of action is to truncate the log and run a hard
319          * recovery, but if the matchpoint precedes a checkpoint that deleted
320          * log files, the truncation is not permissible because the resulting
321          * log might be missing needed files. Instead, we have to do a network
322          * restore.
323          *
324          * A checkpoint that resulted in deleted log files will have flushed
325          * references to entries migrated from the deleted file, and the
326          * original entries might still be referenced if we truncate the log
327          * prior to the CkptEnd. For example:
328          *
329          *    3/100 LN-A
330          *    4/100 BIN parent of LN-1 at 3/100
331          *    5/100 cleaning of file 3 begins
332          *    5/200 LN-A migrated from 3/100, parent BIN is dirtied
333          *    5/300 CkptStart
334          *    5/400 matchpoint
335          *    5/500 BIN parent of LN-A at 5/200
336          *    5/600 CkptEnd, after which file 3 is deleted
337          *
338          * If we truncate the log at the matchpoint, the BIN at 5/500 is
339          * truncated. The BIN at 4/100 will be used, which refers to an LN in a
340          * now deleted file. Note that the cleaner guarantees that a checkpoint
341          * occurs after migration and before file deletion.
342          *
343          * However, because files become reserved after a checkpoint and are
344          * not deleted immediately, we could trucate at a matchpoint that
345          * precedes such a checkpoint, when these reserved files have not yet
346          * been deleted. In the future we could enable such a feature by
347          * keeping track (persistently) of the earliest point at which the log
348          * can be truncated, taking into account the deletion of reserved
349          * files. Then we could use this information to validate a matchpoint.
350          */
351         if (searchResults.getPassedCheckpointEnd()) {
352             LoggerUtils.info(logger, repImpl, "matchpointVLSN of " +
353                              matchpointVLSN + " precedes a checkpoint end, " +
354                              "needs network restore.");
355             throw setupLogRefresh(matchpointVLSN);
356         }
357 
358         /*
359          * Likewise, if we skipped over a gap in the log files, we can't be
360          * sure if we passed a ckpt with deleted log files. Do a network
361          * restore rather than a hard recovery.
362          */
363         if (searchResults.getSkippedGap()) {
364             LoggerUtils.info(logger, repImpl, "matchpointVLSN of " +
365                              matchpointVLSN + " was found in a replica log " +
366                              "with gaps. Since we can't be sure if it " +
367                              "precedes a checkpoint end, do network restore.");
368             throw setupLogRefresh(matchpointVLSN);
369         }
370 
371         /*
372          * We're planning on rolling back past a commit or abort, and we know
373          * that we have not passed a barrier checkpoint. See if we have
374          * exceeded the number of rolledback commits limit.
375          */
376         EnvironmentImpl envImpl = repNode.getRepImpl();
377         DbConfigManager configMgr = envImpl.getConfigManager();
378         final int rollbackTxnLimit =
379             configMgr.getInt(RepParams.TXN_ROLLBACK_LIMIT);
380         final boolean rollbackDisabled =
381             configMgr.getBoolean(RepParams.TXN_ROLLBACK_DISABLED);
382 
383         final int numPassedDurableCommits =
384             searchResults.getNumPassedDurableCommits();
385         final int numPassedCommits =
386             searchResults.getNumPassedCommits();
387         final long dtvlsn = searchResults.getDTVLSN().getSequence();
388         LoggerUtils.info(logger, repImpl,
389                          String.format("Rollback info. " +
390                                        "Number of passed commits:%,d. " +
391                                        "(durable commits:%,d). " +
392                                        "Durable commit VLSN:%,d " +
393                                        "Rollback transaction limit:%,d",
394                                        numPassedCommits,
395                                        numPassedDurableCommits,
396                                        dtvlsn,
397                                        rollbackTxnLimit));
398 
399         if (numPassedDurableCommits > rollbackTxnLimit || rollbackDisabled) {
400 
401             LoggerUtils.severe(logger, repImpl,
402                                "Limited list of transactions that would " +
403                                " be truncated for hard recovery:\n" +
404                                searchResults.dumpPassedTxns());
405 
406             throw new RollbackProhibitedException(repNode.getRepImpl(),
407                                                   rollbackTxnLimit,
408                                                   rollbackDisabled,
409                                                   matchpointVLSN,
410                                                   searchResults);
411         }
412 
413         /*
414          * After passing all the earlier qualifications, do a truncation and
415          * hard recovery.
416          */
417         throw setupHardRecovery();
418     }

420     /**
421      * Find a matchpoint, which is a log entry in the replication stream which
422      * is the same on feeder and replica. Assign the matchpointVLSN field. The
423      * matchpoint log entry must be be tagged with an environment id. If no
424      * matching entry is found, the matchpoint is effectively the NULL_VLSN.
425      *
426      * To determine the matchpoint, exchange messages with the feeder and
427      * compare log entries. If the feeder does not have enough log entries,
428      * throw InsufficientLogException.
429      * @throws InsufficientLogException
430      */
431     private void findMatchpoint(VLSNRange range)
432         throws IOException,
433                InsufficientLogException {
434 
435         int matchCounter = 0;
436         repImpl.setSyncupProgress(SyncupProgress.FIND_MATCHPOINT,
437                                   matchCounter++, -1);
438         VLSN candidateMatchpoint = range.getLastSync();
439         if (candidateMatchpoint.equals(NULL_VLSN)) {
440 
441             /*
442              * If the replica has no sync-able log entries at all, the
443              * matchpoint is the NULL_VLSN, and we should start the replication
444              * stream at VLSN 1. Check if the feeder has the VLSN 1. If it
445              * doesn't, getFeederRecord() will throw a
446              * InsufficientLogException. We can assume that a non-cleaned
447              * feeder always has VLSN 1, because a ReplicatedEnvironment always
448              * creates a few replicated vlsns, such as the name db, at
449              * initial startup.
450              */
451             getFeederRecord(range, VLSN.FIRST_VLSN,
452                             false /*acceptAlternative*/);
453             return;
454         }
455 
456         /*
457          * CandidateMatchpoint is not null, so ask the feeder for the log
458          * record at that vlsn.
459          */
460         InputWireRecord feederRecord =
461             getFeederRecord(range, candidateMatchpoint,
462                             true /*acceptAlternative*/);
463 
464         /*
465          * The feeder may have suggested an alternative matchpoint, so reset
466          * candidate matchpoint.
467          */
468         candidateMatchpoint = feederRecord.getVLSN();
469         if (logger.isLoggable(Level.FINE)) {
470             LoggerUtils.fine(logger, repImpl,
471                              "first candidate matchpoint: " +
472                              candidateMatchpoint);
473         }
474         /*
475          * Start comparing feeder records to replica records. Instead of using
476          * the VLSNIndex to direct our search, we must scan from the end of the
477          * log, recording entries that have an impact on our ability to
478          * rollback, like checkpoints.
479          *
480          * Start by finding the candidate matchpoint in the Replica.
481          */
482         backwardsReader = setupBackwardsReader
483             (candidateMatchpoint,
484              repNode.getRepImpl().getFileManager().getLastUsedLsn());
485         OutputWireRecord replicaRecord = getReplicaRecord(candidateMatchpoint);
486 
487         while (!replicaRecord.match(feederRecord)) {
488             repImpl.setSyncupProgress(SyncupProgress.FIND_MATCHPOINT,
489                                       matchCounter++, -1);
490 
491             /*
492              * That first bid didn't match, now just keep looking at all
493              * potential matchpoints.
494              */
495             replicaRecord = scanMatchpointEntries();
496 
497             if (replicaRecord == null) {
498 
499                 /*
500                  * The search for the previous sync log entry went past our
501                  * available contiguous VLSN range, so there is no
502                  * matchpoint.
503                  */
504                 LoggerUtils.info(logger, repImpl,
505                                  "Looking at candidate matchpoint vlsn " +
506                                  candidateMatchpoint +
507                                  " but this node went past its available" +
508                                  " contiguous VLSN range, need network" +
509                                  " restore.");
510                 throw setupLogRefresh(candidateMatchpoint);
511             }
512 
513             /*
514              * Ask the feeder for the record. If the feeder doesn't have
515              * it, we'll throw out and do a network restore.
516              */
517             candidateMatchpoint = replicaRecord.getVLSN();
518             if (logger.isLoggable(Level.FINE)) {
519                 LoggerUtils.fine(logger, repImpl,
520                                  "Next candidate matchpoint: " +
521                                  candidateMatchpoint);
522             }
523             feederRecord = getFeederRecord(range, candidateMatchpoint,
524                                            false);
525         }
526 
527         /* We've found the matchpoint. */
528         matchedVLSNTime = replicaRecord.getTimeStamp();
529         matchpointVLSN = candidateMatchpoint;
530         searchResults.setMatchpoint(backwardsReader.getLastLsn());
531         LoggerUtils.finest(logger, repImpl,
532                            "after setting  matchpoint, searchResults=" +
533                            searchResults);
534     }

553     /**
554      * Search backwards for the replica's log record at this target VLSN. The
555      * target record is either the replica's first suggestion for a matchpoint,
556      * or feeder's counter offer. We have checked earlier that the counter
557      * offer is within the replica's vlsn range.
558      */
559     private OutputWireRecord getReplicaRecord(VLSN candidateMatchpoint) {
560 
561         OutputWireRecord replicaRecord = null;
562         do {
563             try {
564                 replicaRecord =
565                     backwardsReader.scanBackwards(candidateMatchpoint);
566 
567                 /*
568                  * We're hunting for a VLSN that should be in the VLSN range,
569                  * and it should exist.
570                  */
571                 if (replicaRecord == null) {
572                     throw EnvironmentFailureException.unexpectedState
573                         (repImpl,
574                          "Searching for candidate matchpoint " +
575                          candidateMatchpoint +
576                          " but got null record back ");
577                 }
578 
579                 /* We've found the record at candidateMatchpoint */
580                 return replicaRecord;
581             } catch (SkipGapException e) {
582                 /*
583                  * The ReplicaSyncupReader will throw a SkipGapException if it
584                  * encounters a cleaned files gap in the log. There can be
585                  * multiple gaps on its way toward finding the candidate
586                  * vlsn. The ReplicaSyncupReader is obliged to traverse the
587                  * log, in order to note checkpoints, rather than simply using
588                  * the vlsn index. When a gap is detected, the vlsn on the left
589                  * side of the gap is used to re-init a new reader. For
590                  * example, suppose the log looks like this:
591                  *
592                  * file 100 has vlsns 41-50
593                  * file 200 has vlsns 51-60
594                  * file 300 has vlsns 61-70
595                  *
596                  * and the candidate matchpoint is 45, the search will start at
597                  * vlsn 70.
598                  *  t1: SkipGapException thrown at gap between file 200 & 300,
599                  *      create new reader positioned at vlsn 60
600                  *  t2: SkipGapException thrown at gap between file 100 & 200,
601                  *      create new reader positioned at vlsn 50
602                  */
603                 VLSN gapRepositionVLSN = e.getVLSN();
604                 if (gapRepositionVLSN.compareTo(candidateMatchpoint) < 0) {
605                     throw EnvironmentFailureException.unexpectedState
606                         ("Gap reposition point of " + gapRepositionVLSN +
607                          " should always be >= candidate matchpoint VLSN of " +
608                          candidateMatchpoint);
609                 }
610 
611                 long startScanLsn = vlsnIndex.getGTELsn(gapRepositionVLSN);
612                 backwardsReader = setupBackwardsReader(candidateMatchpoint,
613                                                        startScanLsn);
614                 /*
615                  * If we skip a gap, there is a chance that we will have passed
616                  * a checkpoint which had deleted log files. This has no impact
617                  * if we are doing a soft rollback, but if we do a hard
618                  * recovery, it would prevent us from truncating the log. It
619                  * would require doing a network restore if we need to rollback
620                  * committed txns.
621                  */
622                 searchResults.noteSkippedGap();
623             }
624         } while (true);
625     }
626 
627     /**
628      * Search backwards for potential matchpoints in the replica log,
629      * accounting for potential gaps.
630      */
631     private OutputWireRecord scanMatchpointEntries() {
632         OutputWireRecord replicaRecord = null;
633         boolean firstAttempt = true;
634         do {
635             try {
636                 /*
637                  * The first time around, when firstAttempt is true, ask the
638                  * reader to search for the vlsn before the currentVLSN,
639                  * because we entered this method having searched to a given
640                  * target matchpoint. All subsequent times, we are in search of
641                  * the reader's currentVLSN, but haven't found it yet, because
642                  * we hit a gap, so leave the currentVLSN alone.
643                  */
644                 replicaRecord =
645                     backwardsReader.findPrevSyncEntry(firstAttempt);
646 
647                 /*
648                  * Either se've found a possible matchpoint, or we've come to
649                  * the end and the replicaRecord is null. One way or another,
650                  * return the results of the scan.
651                  */
652                 return replicaRecord;
653             } catch (SkipGapException e) {
654                 /*
655                  * The ReplicaSyncupReader will throw a SkipGapException if it
656                  * encounters a cleaned files gap in the log. There can be
657                  * multiple gaps on its way toward finding the next potential
658                  * matchpoint. The ReplicaSyncupReader is obliged to traverse
659                  * the log, in order to note checkpoints, rather than simply
660                  * using the vlsn index. When a gap is detected, the vlsn on
661                  * the left side of the gap is used to re-init a new
662                  * reader. For example, suppose the log looks like this and the
663                  * search starts at vlsn 70
664                  *
665                  * file 100 has vlsns 51-60
666                  * file 200 has no vlsns
667                  * file 300 has no vlsns
668                  * file 400 has vlsns 61-70
669                  *
670                  *  SkipGapException thrown at gap between file 300 & 400,
671                  *  when the reader's currentVLSN is 60. Create a new reader,
672                  *  positioned at vlsn 60, skipping over files 200 and 300.
673                  */
674 
675                 VLSN gapRepositionVLSN = e.getVLSN();
676                 backwardsReader = setupBackwardsReader
677                     (gapRepositionVLSN,
678                      vlsnIndex.getGTELsn(gapRepositionVLSN));
679                 firstAttempt = false;
680                 searchResults.noteSkippedGap();
681             }
682         } while(true);
683     }

686      * Ask the feeder for information to add to InsufficientLogException,
687      * and then throw the exception.
688      *
689      * The endVLSN marks the last VLSN that this node will want from
690      * the network restore. That information helps ensure that the restore
691      * source has enough vlsns to satisfy this replica.
692      *
693      * The replication node list identifies possible log provider members.
694      * @throws IOException
695      */
696     private InsufficientLogException setupLogRefresh(VLSN failedMatchpoint)
697         throws IOException {
698 
699         protocol.write(protocol.new RestoreRequest(failedMatchpoint),
700                        namedChannel);
701         RestoreResponse response =
702             (RestoreResponse) protocol.read(namedChannel);
703 
704         return new InsufficientLogException(
705             repNode,
706             new HashSet<>(Arrays.asList(response.getLogProviders())));
707     }

709     /**
710      * Hard recovery:  truncate the files, repeat recovery.
711      * If this hard recovery came about before the ReplicatedEnvironment was
712      * fully instantiated, we will recreate the environment under the
713      * covers. If this came while the replica was up and supporting existing
714      * Environment handles, we must invalidate the environment, and ask the
715      * application to reopen.
716      */
717     public RollbackException setupHardRecovery()
718         throws IOException {
719 
720         /* Creating the exception invalidates the environment. */
721         RollbackException r = new RollbackException(repImpl,
722                                                     matchpointVLSN,
723                                                     searchResults);
724         LoggerUtils.severe(logger, repImpl,
725                            "Limited list of transactions truncated for " +
726                            "hard recovery:\n" +
727                            searchResults.dumpPassedTxns());
728 
729         /*
730          * Truncate after the environment is invalidated, which happens
731          * when we instantiate RollbackException.
732          */
733         long matchpointLSN = searchResults.getMatchpointLSN();
734         repImpl.getFileManager().truncateLog
735             (DbLsn.getFileNumber(matchpointLSN),
736              DbLsn.getFileOffset(matchpointLSN));
737 
738         return r;
739     }

741     /**
742      * Request a log entry from the feeder at this VLSN. The Feeder will only
743      * return the log record or say that it isn't available.
744      *
745      * @throws InsufficientLogException
746      */
747     private InputWireRecord getFeederRecord(VLSNRange range,
748                                             VLSN requestVLSN,
749                                             boolean acceptAlternative)
750         throws IOException, InsufficientLogException {
751 
752         /* Ask the feeder for the matchpoint log record. */
753         protocol.write(protocol.new EntryRequest(requestVLSN), namedChannel);
754 
755         /*
756          * Expect
757          *  a) the requested log record
758          *  b) message that says this feeder doesn't have RequestVLSN
759          *  c) if acceptAlternative == true and the feeder didn't have
760          *     requestVLSN, but had an earlier entry, the feeder may send an
761          *      earlier, alternative matchpoint
762          */
763         Message message = protocol.read(namedChannel);
764         if (message instanceof Entry) {
765             Entry entry = (Entry) message;
766             return entry.getWireRecord();
767         }
768 
769         if (message instanceof EntryNotFound) {
770             LoggerUtils.info(logger, repImpl, "Requested " + requestVLSN +
771                              " from " + namedChannel.getNameIdPair() +
772                              " but that node did not have that vlsn.");
773             throw setupLogRefresh(requestVLSN);
774         }
775 
776         if ((acceptAlternative) &&
777             (message instanceof AlternateMatchpoint)) {
778 
779             AlternateMatchpoint alt = (AlternateMatchpoint) message;
780             InputWireRecord feederRecord = alt.getAlternateWireRecord();
781             VLSN altMatchpoint = feederRecord.getVLSN();
782             if (range.getFirst().compareTo(altMatchpoint) > 0) {
783 
784                 /*
785                  * The feeder suggest a different matchpoint, but it's outside
786                  * the replica's range. Give up and do a network restore.
787                  */
788                 throw setupLogRefresh(altMatchpoint);
789             }
790             return feederRecord;
791         }
792 
793         throw EnvironmentFailureException.unexpectedState
794             (repNode.getRepImpl(),
795              "Sent EntryRequest, got unexpected response of " + message);
796     }
