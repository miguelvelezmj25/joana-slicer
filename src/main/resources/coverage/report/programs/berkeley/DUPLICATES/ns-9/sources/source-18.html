63 /**
64  * The UP tracks utilization summary information for all log files.
65  *
66  * <p>Unlike the UtilizationTracker, the UP is not accessed under the log write
67  * latch and is instead synchronized on itself for protecting the cache.  It is
68  * not accessed during the primary data access path, except for when flushing
69  * (writing) file summary LNs.  This occurs in the following cases:
70  * <ol>
71  * <li>The summary information is flushed at the end of a checkpoint.  This
72  * allows tracking to occur in memory in between checkpoints, and replayed
73  * during recovery.</li>
74  * <li>When committing the truncateDatabase and removeDatabase operations, the
75  * summary information is flushed because detail tracking for those operations
76  * is not replayed during recovery</li>
77  * <li>The evictor will ask the UtilizationTracker to flush the largest summary
78  * if the memory taken by the tracker exeeds its budget.</li>
79  * </ol>
80  *
81  * <p>The cache is populated by the RecoveryManager just before performing the
82  * initial checkpoint.  The UP must be open and populated in order to respond
83  * to requests to flush summaries and to evict tracked detail, even if the
84  * cleaner is disabled.</p>
85  *
86  * <p>WARNING: While synchronized on this object, eviction is not permitted.
87  * If it were, this could cause deadlocks because the order of locking would be
88  * the UP object and then the evictor.  During normal eviction the order is to
89  * first lock the evictor and then the UP, when evicting tracked detail.</p>
90  *
91  * <p>The methods in this class synchronize to protect the cached summary
92  * information.  Some methods also access the UP database.  However, because
93  * eviction must not occur while synchronized, UP database access is not
94  * performed while synchronized except in one case: when inserting a new
95  * summary record.  In that case we disallow eviction during the database
96  * operation.</p>
97  */

426      * Removes a set of files from the DatabaseImpl utilization info, dirtying
427      * the MapLNs in the process. The MapLNs will be flushed by the
428      * checkpointer. This method performs eviction and is not synchronized.
429      *
430      * For a given file, this method must be called after inserting its
431      * reserved db record. If there is a crash and recovery finds the reserved
432      * db record in the checkpointer interval, recovery will redo this action.
433      *
434      * This method is optimally called for multiple files at a time, since this
435      * updates the MapLNs more efficiently than if it were done one file at a
436      * time.
437      */
438     private boolean removeDbMetadata(final Collection<Long> fileNums,
439                                      final Set<DatabaseId> databases) {
440 
441         final LogManager logManager = env.getLogManager();
442         final DbTree dbTree = env.getDbTree();
443         boolean anyRemoved = false;
444 
445         for (final DatabaseId dbId : databases) {
446             final DatabaseImpl db = dbTree.getDb(dbId);
447             if (db == null) {
448                 continue;
449             }
450             try {
451                 anyRemoved |= logManager.removeDbFileSummaries(db, fileNums);
452             } finally {
453                 dbTree.releaseDb(db);
454             }
455         }
456 
457         return anyRemoved;
458     }

527      * removeDbMetadata as described in {@link #populateCache}.

531 
532         Locker locker = null;
533         CursorImpl cursor = null;
534         try {
535             locker = BasicLocker.createBasicLocker(env, false /*noWait*/);
536             cursor = new CursorImpl(fileSummaryDb, locker);
537             /* Perform eviction in unsynchronized methods. */
538             cursor.setAllowEviction(true);
539 
540             DatabaseEntry keyEntry = new DatabaseEntry();
541             DatabaseEntry dataEntry = new DatabaseEntry();
542 
543             /* Do not return data to avoid a fetch of the existing LN. */
544             dataEntry.setPartial(0, 0, true);
545 
546             /* Search by file number. */
547             OperationResult result = null;
548             if (getFirstFSLN(

550                 result = DbInternal.DEFAULT_RESULT;
551             }
552

554             while (result != null &&
555                    fileNum ==
556                    FileSummaryLN.getFileNumber(keyEntry.getData())) {
557 
558                 /* Perform eviction once per operation. */
559                 env.daemonEviction(true /*backgroundIO*/);

563                  * fetch the LN.

568                     keyEntry, dataEntry, LockType.WRITE,

583         if (tfs != null) {

600     /**
601      * Updates and stores the FileSummary for a given tracked file.  This
602      * method is synchronized and may not perform eviction.
603      */
604     private synchronized PackedOffsets putFileSummary(TrackedFileSummary tfs)
605         throws DatabaseException {
606 
607         if (env.isReadOnly()) {
608             throw EnvironmentFailureException.unexpectedState
609                 ("Cannot write file summary in a read-only environment");
610         }
611 
612         if (tfs.isEmpty()) {
613             return null; // no delta
614         }
615 
616         if (!cachePopulated) {
617             /* Db does not exist and this is a read-only environment. */
618             return null;
619         }
620 
621         long fileNum = tfs.getFileNumber();
622         Long fileNumLong = fileNum;
623 
624         /* Get existing file summary or create an empty one. */
625         FileSummary summary = fileSummaryMap.get(fileNumLong);
626         if (summary == null) {
627 
628             /*
629              * An obsolete node may have been counted after its file was
630              * reserved or even deleted, for example, when compressing a BIN.
631              * Do not insert a new profile record if the file is reserved or
632              * if no corresponding log file exists.
633              */
634             if (!env.getFileProtector().isActiveOrNewFile(fileNumLong)) {
635 
636                 /*
637                  * File was deleted by the cleaner.  Remove it from the
638                  * UtilizationTracker and return.  Note that a file is normally
639                  * removed from the tracker by FileSummaryLN.writeToLog method
640                  * when it is called via insertFileSummary below. [#15512]
641                  */
642                 env.getLogManager().removeTrackedFile(tfs);
643                 return null;
644             }
645 
646             summary = new FileSummary();
647         }
648 
649         /*
650          * The key discriminator is a sequence that must be increasing over the
651          * life of the file.  We use the sum of all entries counted.  We must
652          * add the tracked and current summaries here to calculate the key.
653          */
654         FileSummary tmp = new FileSummary();
655         tmp.add(summary);
656         tmp.add(tfs);
657         int sequence = tmp.getEntriesCounted();
658 
659         /* Insert an LN with the existing and tracked summary info. */
660         FileSummaryLN ln = new FileSummaryLN(summary);
661         ln.setTrackedSummary(tfs);
662         insertFileSummary(ln, fileNum, sequence);
663 
664         /* Cache the updated summary object.  */
665         summary = ln.getBaseSummary();
666 
667         if (fileSummaryMap.put(fileNumLong, summary) == null) {
668             MemoryBudget mb = env.getMemoryBudget();
669             mb.updateAdminMemoryUsage(MemoryBudget.UTILIZATION_PROFILE_ENTRY);
670         }
671 
672         return ln.getObsoleteOffsets();

715     private long[] getObsoleteDetailInternal(Long fileNum, boolean logUpdate)
716         throws DatabaseException {
717 
718         assert cachePopulated;
719 
720         final long fileNumVal = fileNum;
721         final List<long[]> list = new ArrayList<>();
722 
723         /*
724          * Get a TrackedFileSummary that cannot be flushed (evicted) while we
725          * gather obsolete offsets.
726          */
727         final TrackedFileSummary tfs =
728             env.getLogManager().getUnflushableTrackedSummary(fileNumVal);
729         try {
730             /* Read the summary db. */
731             final Locker locker =
732                 BasicLocker.createBasicLocker(env, false /*noWait*/);
733             final CursorImpl cursor = new CursorImpl(fileSummaryDb, locker);
734             try {
735                 /* Perform eviction in unsynchronized methods. */
736                 cursor.setAllowEviction(true);
737 
738                 final DatabaseEntry keyEntry = new DatabaseEntry();
739                 final DatabaseEntry dataEntry = new DatabaseEntry();
740 
741                 /* Search by file number. */
742                 OperationResult result = null;
743                 if (getFirstFSLN(cursor, fileNumVal, keyEntry, dataEntry,
744                                   LockType.NONE)) {
745                     result = DbInternal.DEFAULT_RESULT;
746                 }
747 
748                 /* Read all LNs for this file number. */
749                 while (result != null) {
750 
751                     /* Perform eviction once per operation. */
752                     env.daemonEviction(true /*backgroundIO*/);
753 
754                     final FileSummaryLN ln = (FileSummaryLN)
755                         cursor.lockAndGetCurrentLN(LockType.NONE);
756 
757                     if (ln != null) {
758                         /* Stop if the file number changes. */
759                         if (fileNumVal !=
760                             FileSummaryLN.getFileNumber(keyEntry.getData())) {
761                             break;
762                         }
763 
764                         final PackedOffsets offsets = ln.getObsoleteOffsets();
765                         if (offsets != null) {
766                             list.add(offsets.toArray());
767                         }
768 
769                         /* Always evict after using a file summary LN. */
770                         cursor.evictLN();
771                     }
772 
773                     result = cursor.getNext(
774                         keyEntry, dataEntry, LockType.NONE,
775                         false /*dirtyReadAll*/, true /*forward*/,
776                         false /*isLatched*/, null /*rangeConstraint*/);
777                 }
778             } finally {
779                 cursor.close();
780                 locker.operationEnd();
781             }
782 
783             /*
784              * Write out tracked detail, if any, and add its offsets to the
785              * list.
786              */
787             if (!tfs.isEmpty()) {
788                 if (logUpdate) {
789                     final PackedOffsets offsets = putFileSummary(tfs);
790                     if (offsets != null) {
791                         list.add(offsets.toArray());
792                     }
793                 } else {
794                     final long[] offsetList = tfs.getObsoleteOffsets();
795                     if (offsetList != null) {
796                         list.add(offsetList);
797                     }
798                 }
799             }
800         } finally {
801             /* Allow flushing of TFS when all offsets have been gathered. */
802             tfs.setAllowFlush(true);
803         }
804 
805         /* Merge all offsets into a single array and pack the result. */
806         int size = 0;
807         for (final long[] a : list) {
808             size += a.length;
809         }
810         final long[] offsets = new long[size];
811         int index = 0;
812         for (long[] a : list) {
813             System.arraycopy(a, 0, offsets, index, a.length);
814             index += a.length;
815         }
816         assert index == offsets.length;
817         return offsets;
818     }

820     /**
821      * Populate the profile for file selection.  This method performs eviction
822      * and is not synchronized.  It must be called before recovery is complete
823      * so that synchronization is unnecessary.  It should be called before the
824      * recovery checkpoint so that the checkpoint can flush dirty metadata.
825      *
826      * After a file is cleaned, at checkpoint end it is moved to reserved
827      * status and these steps are taken to delete its metadata:
828      *  1. The file is cleaned and a checkpoint flushes the updated INs.
829      *  2. Utilization info in MapLNs referencing the file are updated in
830      *     cache and the MapLNs are dirtied. At some point later, the dirty
831      *     MapLNs are flushed by a normally scheduled checkpoint. This could
832      *     happen after any of the steps below.
833      *  3. A record is inserted for the file in the reserved file db.
834      *  4. All file summary db records for the file are deleted.
835      *
836      * When the reserved file is deleted, this is done in these steps.
837      *  5. The file itself is deleted. Note that because we do not flush the
838      *     log before deleting the file, step 3 and 4 may not be durable.
839      *  6. The files's record in the reserved file db is deleted.
840      *
841      * Data file deletion and file metadata deletion cannot be performed
842      * atomically. In fact none of the steps are grouped atomically into
843      * transactions, and no-sync logging is used when writing to the file
844      * summary db and the reserved file db. A crash can occur at any point, and
845      * we must handle that in some way, either in recovery or here (in the
846      * populateCache method).
847      *
848      *  - A crash prior to durable completion of step 3 will be handled
849      *    naturally by cleaning the file again, since it is present in the file
850      *    summary db but not in the reserved file db.
851      *
852      *  - If step 3 completes durably but a crash occurs before the dirty
853      *    MapLNs are flushed, this is indicated by the presence of a reserved
854      *    file record in the recovery interval, and step 2 will be repeated
855      *    here during recovery.
856      *
857      *  - If step 3 completes durably but not step 4 or 5, then step 4 is
858      *    completed here.
859      *
860      *  - If step 3 and 5 complete durably but not step 4 and 6, then steps 4
861      *    and 6 are completed here.
862      *
863      *  - If step 4 and 5 complete durably but not step 6, then step 6 is
864      *    completed here.
865      *
866      * Prior to log version 15, the reserved file db did not exist and reserved
867      * files were re-cleaned if they were not deleted before a crash. The old
868      * steps were:
869      *  A. The file is cleaned and a checkpoint flushes the updated INs.
870      *  B. If the file is reserved, this is tracked in memory but nothing is
871      *     logged.
872      *
873      * When deleting a file:
874      *  C. The file itself is deleted.
875      *  D. MapLNs are updated and flushed.
876      *  E. All file summary db records for the file are deleted.
877      *
878      * This is handled now as follows, which is the same as the old approach.
879      *
880      *  - A crash prior to step C results in re-cleaning the file.
881      *
882      *  - If step C completes but step E does not, step D and E are performed
883      *    here. MapLNs must be flushed immediately in this case, which is
884      *    inefficient but hopefully very infrequent.
885      */
886     public boolean populateCache(
887         final StartupTracker.Counter counter,
888         final RecoveryInfo recoveryInfo,
889         final Set<Long> recoveryReservedFiles,
890         final Set<DatabaseId> recoveryReservedFileDbs)
891         throws DatabaseException {
892 
893         assert !cachePopulated;
894

896         if (!openFileSummaryDatabase()) {
897             /* Db does not exist and this is a read-only environment. */
898             return false;

902         final Set<Long> reservedFileRecords = new HashSet<>();
903         final DatabaseEntry keyEntry = new DatabaseEntry();
904         final DatabaseEntry dataEntry = new DatabaseEntry();
905         final FileProtector fileProtector = env.getFileProtector();
906 
907         /*
908          * For each reserved file record in the recovery interval, ensure that
909          * the MapLN metadata is updated.
910          */

912             recoveryReservedFiles, recoveryReservedFileDbs);
913 
914         /*
915          * Open the reserved file db on first use. This may return false in a
916          * read-only old-version env where the file summary db exists but the
917          * reserved file db does not.
918          */
919         if (openReservedFilesDatabase()) {
920 
921             /*
922              * First read through the reserved file db and add reserved
923              * files to the file protector. It is possible that a file does
924              * not exist for a record in the db, and again we must clean up
925              * as described in method comments.
926              */
927             final Locker locker =
928                 BasicLocker.createBasicLocker(env, false /*noWait*/);
929             try {
930                 final ReadOptions options =
931                     new ReadOptions().setLockMode(LockMode.READ_UNCOMMITTED);
932 
933                 try (final Cursor cursor = DbInternal.makeCursor(
934                         reservedFilesDb, locker, null,
935                         false /*retainNonTxnLocks*/)) {
936 
937                     while (cursor.get(
938                         keyEntry, dataEntry, Get.NEXT, options) != null) {
939 
940                         counter.incNumRead();
941                         env.daemonEviction(false /*backgroundIO*/);
942 
943                         final Long file =
944                             ReservedFileInfo.entryToKey(keyEntry);
945 
946                         reservedFileRecords.add(file);
947 
948                         final ReservedFileInfo info =
949                             ReservedFileInfo.entryToObject(dataEntry);
950 
951                         if (Arrays.binarySearch(existingFiles, file) >= 0) {
952 
953                             counter.incNumProcessed();
954                             fileProtector.reserveFile(file, info.lastVLSN);
955 
956                         } else {
957                             counter.incNumDeleted();
958 
959                             if (!info.lastVLSN.isNull()) {
960                                 recoveryInfo.lastMissingFileNumber = file;
961                                 recoveryInfo.lastMissingFileVLSN =
962                                     info.lastVLSN;
963                             }
964 
965                             if (env.isReadOnly()) {
966                                 continue;
967                             }
968 
969                             cursor.delete();
970                         }
971                     }
972                 }
973             } finally {
974                 locker.operationEnd();
975             }
976         }
977 
978         final int oldMemorySize = fileSummaryMap.size() *
979             MemoryBudget.UTILIZATION_PROFILE_ENTRY;
980 
981         /*
982          * Next read through the file summary db and populate the profile.
983          * As above, it is possible that a file does not exist for a record in
984          * the db, and again we must clean up as described in method comments.
985          */
986         Locker locker = null;
987         CursorImpl cursor = null;
988         try {
989             locker = BasicLocker.createBasicLocker(env, false /*noWait*/);
990 
991             cursor = new CursorImpl(
992                 fileSummaryDb, locker, false /*retainNonTxnLocks*/,
993                 false /*isSecondaryCursor*/);
994 
995             /* Perform eviction in unsynchronized methods. */
996             cursor.setAllowEviction(true);
997 
998             if (cursor.positionFirstOrLast(true)) {
999 
1000                 /* Retrieve the first record. */
1001                 OperationResult result = cursor.lockAndGetCurrent(
1002                     keyEntry, dataEntry, LockType.NONE,
1003                     false /*dirtyReadAll*/,
1004                     true /*isLatched*/, true /*unlatch*/);
1005 
1006                 if (result == null) {
1007                     /* The record we're pointing at may be deleted. */
1008                     result = cursor.getNext(
1009                         keyEntry, dataEntry, LockType.NONE,
1010                         false /*dirtyReadAll*/, true /*forward*/,
1011                         false /*isLatched*/, null /*rangeConstraint*/);
1012                 }
1013 
1014                 while (result != null) {
1015                     counter.incNumRead();
1016 
1017                     /*
1018                      * Perform eviction once per operation.  Pass false for
1019                      * backgroundIO because this is done during recovery and
1020                      * there is no reason to sleep.
1021                      */
1022                     env.daemonEviction(false /*backgroundIO*/);
1023 
1024                     final FileSummaryLN ln = (FileSummaryLN)
1025                         cursor.lockAndGetCurrentLN(LockType.NONE);
1026 
1027                     if (ln == null) {
1028                         /* Advance past a cleaned record. */
1029                         result = cursor.getNext(
1030                             keyEntry, dataEntry, LockType.NONE,
1031                             false /*dirtyReadAll*/,
1032                             true /*forward*/, false /*isLatched*/,
1033                             null /*rangeConstraint*/);
1034                         continue;
1035                     }
1036 
1037                     final byte[] keyBytes = keyEntry.getData();
1038                     final boolean isOldVersion =
1039                         FileSummaryLN.hasStringKey(keyBytes);
1040                     final long fileNum = FileSummaryLN.getFileNumber(keyBytes);
1041                     final Long fileNumLong = fileNum;
1042 
1043                     if (!fileProtector.isReservedFile(fileNumLong) &&
1044                         Arrays.binarySearch(existingFiles, fileNumLong) >= 0) {
1045 
1046                         counter.incNumProcessed();
1047 
1048                         /* File is active, cache the FileSummaryLN. */
1049                         final FileSummary summary = ln.getBaseSummary();
1050                         fileSummaryMap.put(fileNumLong, summary);
1051 
1052                         /*
1053                          * Update old version records to the new version.  A
1054                          * zero sequence number is used to distinguish the
1055                          * converted records and to ensure that later records
1056                          * will have a greater sequence number.
1057                          */
1058                         if (isOldVersion && !env.isReadOnly()) {
1059                             insertFileSummary(ln, fileNum, 0);
1060                             cursor.deleteCurrentRecord(
1061                                 ReplicationContext.NO_REPLICATE);
1062                         } else {
1063                             /* Always evict after using a file summary LN. */
1064                             cursor.evictLN();
1065                         }
1066                     } else {
1067 
1068                         /*
1069                          * File does not exist or is a reserved file. Remove
1070                          * the summary from the map and delete all
1071                          * FileSummaryLN records. If the file has a reserved
1072                          * file record (even if it was deleted above) then we
1073                          * can rely on the reserved file record mechanism to
1074                          * update MapLNs (per-db metadata); otherwise we must
1075                          * update the MapLNs here.
1076                          */
1077                         counter.incNumDeleted();
1078 
1079                         fileSummaryMap.remove(fileNumLong);
1080 
1081                         if (!env.isReadOnly()) {
1082                             if (!reservedFileRecords.contains(fileNumLong)) {
1083                                 removeAndFlushDbMetadata(fileNumLong);
1084                             }
1085                             if (isOldVersion) {
1086                                 cursor.deleteCurrentRecord(
1087                                     ReplicationContext.NO_REPLICATE);
1088                             } else {
1089                                 deleteFileSummary(fileNumLong);
1090                             }
1091                         }
1092 
1093                         /*
1094                          * Do not evict after deleting since the compressor
1095                          * would have to fetch it again.
1096                          */
1097                     }
1098 
1099                     /* Go on to the next entry. */
1100                     if (isOldVersion) {
1101 
1102                         /* Advance past the single old version record. */
1103                         result = cursor.getNext(
1104                             keyEntry, dataEntry, LockType.NONE,
1105                             false /*dirtyReadAll*/,
1106                             true /*forward*/, false /*isLatched*/,
1107                             null /*rangeConstraint*/);
1108                     } else {
1109 
1110                         /*
1111                          * Skip over other records for this file by adding one
1112                          * to the file number and doing a range search.
1113                          */
1114                         if (!getFirstFSLN
1115                             (cursor,
1116                              fileNum + 1,
1117                              keyEntry, dataEntry,
1118                              LockType.NONE)) {
1119                             result = null;
1120                         }
1121                     }
1122                 }
1123             }
1124         } finally {
1125             if (cursor != null) {
1126                 /* positionFirstOrLast may leave BIN latched. */
1127                 cursor.close();
1128             }
1129             if (locker != null) {
1130                 locker.operationEnd();
1131             }
1132 
1133             final int newMemorySize = fileSummaryMap.size() *
1134                 MemoryBudget.UTILIZATION_PROFILE_ENTRY;
1135             final MemoryBudget mb = env.getMemoryBudget();
1136             mb.updateAdminMemoryUsage(newMemorySize - oldMemorySize);
1137         }
1138 
1139         cachePopulated = true;
1140         return forceCheckpoint;
1141     }
