30      * Converts the given duration (interval value plus unit) to milliseconds,
31      * ensuring that any given value greater than zero converts to at least one
32      * millisecond to avoid a zero millisecond result, since Object.wait(0)
33      * waits forever.
34      *
35      * @throws IllegalArgumentException if the duration argument is illegal.
36      * Thrown via API setter methods such as Transaction.setLockTimeout.
37      */
38     public static int durationToMillis(final long val, final TimeUnit unit) {
39         if (val == 0) {
40             /* Allow zero duration with null unit. */
41             return 0;
42         }
43         if (unit == null) {
44             throw new IllegalArgumentException
45                 ("Duration TimeUnit argument may not be null if interval " +
46                  "is non-zero");
47         }
48         if (val < 0) {
49             throw new IllegalArgumentException
50                 ("Duration argument may not be negative: " + val);
51         }
52         final long newVal = unit.toMillis(val);
53         if (newVal == 0) {
54             /* Input val is positive, so return at least one. */
55             return 1;
56         }
57         if (newVal > Integer.MAX_VALUE) {
58             throw new IllegalArgumentException
59                 ("Duration argument may not be greater than " +
60                  "Integer.MAX_VALUE milliseconds: " + newVal);
61         }
62         return (int) newVal;
63     }

106      * value in nanos.

109      * Thrown via the Environment ctor and setMutableConfig, and likewise for a

112     public static long parseDurationNS(final String property) {
113         StringTokenizer tokens =
114             new StringTokenizer(property.toUpperCase(java.util.Locale.ENGLISH),
115                                 " \t");
116         if (!tokens.hasMoreTokens()) {
117             throw new IllegalArgumentException("Duration argument is empty");
118         }
119         final long time;
120         try {
121             time = Long.parseLong(tokens.nextToken());
122         } catch (NumberFormatException e) {
123             throw new IllegalArgumentException
124                 ("Duration argument does not start with a long integer: " +
125                  property);
126         }
127         /* Convert time from specified unit to millis. */
128         long ns;
129         if (tokens.hasMoreTokens()) {
130             final String unitName = tokens.nextToken();
131             if (tokens.hasMoreTokens()) {
132                 throw new IllegalArgumentException
133                     ("Duration argument has extra characters after unit: " +
134                      property);
135             }
136             try {
137                 final TimeUnit unit = TimeUnit.valueOf(unitName);
138                 ns = TimeUnit.NANOSECONDS.convert(time, unit);
139             } catch (IllegalArgumentException e) {
140                 try {
141                     final IEEETimeUnit unit = IEEETimeUnit.valueOf(unitName);
142                     ns = unit.toNanos(time);
143                 } catch (IllegalArgumentException e2) {
144                     throw new IllegalArgumentException
145                         ("Duration argument has unknown unit name: " +
146                          property);
147                 }
148             }
149         } else {
150             /* Default unit is micros. */
151             ns = TimeUnit.NANOSECONDS.convert(time, TimeUnit.MICROSECONDS);
152         }
153         /* If input val is positive, return at least one. */
154         if (time > 0 && ns == 0) {
155             return 1;
156         }
157         return ns;
158     }
