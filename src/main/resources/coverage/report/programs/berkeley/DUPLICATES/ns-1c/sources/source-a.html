41 /**
42  * A Master Transfer operation.
43  * <p>
44  * Each Master Transfer operation uses a separate instance of this class.
45  * There is usually no more than one instance in the lifetime of a master node,
46  * because if the transfer succeeds, the old master node environment becomes
47  * invalid and must be closed.  However, if an operation times out, another
48  * operation can try again later.  Or, a second operation can "forcibly"
49  * supersede an existing operation in progress.
50  *
51  * @see ReplicatedEnvironment#transferMaster(Set, int, TimeUnit)
52  */

65      * and are now notifying everyone of the new (fake) election result.  Once
66      * we get to this point, we can't allow a new Master Transfer operation
67      * attempt to supersede us.

154     /**
155      * Performs the core processing of a Master Transfer operation.  We first
156      * wait for one of the candidate target replica nodes to become completely
157      * synchronized.  We then send a message to all nodes in the group
158      * (including ourselves) announcing which node is to become the new
159      * master.
160      * <p>
161      * If the operation fails we release any transaction commit/abort threads
162      * that may have been blocked during phase 2 of the wait.  However, in the
163      * success case the release of any such transaction threads is done as a
164      * natural by-product of the transition of the environment from master to
165      * replica status.
166      */
167     String transfer() {
168         try {
169             String result = chooseReplica();
170             if (result == null) {
171                 throw new MasterTransferFailureException(getTimeoutMsg());
172             }
173             done = true;
174             synchronized (this) {
175                 eventQueue = null;
176             }
177             annouceWinner(result);
178             return result;
179         } catch (MasterTransferFailureException e) {
180             LoggerUtils.warning(logger, repNode.getRepImpl(),
181                                 "Master Transfer operation failed: " + e);
182             throw e;
183         } catch (InterruptedException ie) {
184             throw new ThreadInterruptedException(repNode.getRepImpl(), ie);
185         } finally {
186             eventQueue = null;
187             if (!done && blocker != null) {
188                 blocker.countDown();
189             }
190         }
191     }

193     /**
194      * Prepares for a Master Transfer operation by waiting for one of the
195      * nominated candidate target replica nodes to catch up with the master,
196      * in two phases, as described in
197      * {@link ReplicatedEnvironment#transferMaster(Set, int, TimeUnit)}.
198      * <p>
199      * This method works by observing events generated by Feeder threads and
200      * passed to us via a queue.
201      *
202      * @return the node name of the first replica to complete phase 2 of the
203      * preparation, or {@code null} if the operation times out.
204      */
205     private String chooseReplica() throws InterruptedException {
206         final ExceptionAwareBlockingQueue<VLSNProgress> queue = getQueue();
207         if (queue == null) {
208             return null;
209         }
210         final FeederManager feederManager = repNode.feederManager();
211         final Map<String, Feeder> activeReplicas =
212             feederManager.activeReplicasMap();
213         for (String nodeName : replicas) {
214             final Feeder feeder = activeReplicas.get(nodeName);
215             if (feeder != null) {
216                 feeder.setMasterTransfer(this);
217             }
218         }
219 
220         /*
221          * Phase 1 could last a long time, if all of our candidate replicas are
222          * still catching up (or not even connected); so we allow new
223          * transactions to be written.  But once we get to phase 2 we block
224          * commit/abort operations for a final (quicker) catch-up.  Thus we can
225          * tell whether we're in phase 2 by whether we have a non-null blocker.
226          */
227         String result = null;
228         for (;;) {
229             final long pollTimeout =
230                 deadlineTimeMs - System.currentTimeMillis();
231             final VLSNProgress event =
232                 queue.pollOrException(pollTimeout, TimeUnit.MILLISECONDS);
233             if (event == null) {
234                 return null;
235             }
236             final VLSN endVLSN = repNode.getCurrentTxnEndVLSN();
237 
238             Level level = Level.INFO;
239             if (event.isFeederDeathEvent()) {
240                 readyReplicas.remove(event.replicaNodeName);
241                 if (blocker != null && readyReplicas.isEmpty()) {
242 
243                     /*
244                      * Must revert back to phase 1.  The latch will still
245                      * exist, because we've passed it to repImpl; and this is
246                      * exactly what we want, so that blocked txns can proceed,
247                      * and new ones won't get blocked for now.
248                      */
249                     blocker.countDown();
250                     blocker = null;
251                 }
252             } else if (blocker == null) {   /* phase 1 */
253                 assert readyReplicas.isEmpty();
254                 readyReplicas.put(event.replicaNodeName, event.vlsn);
255                 blocker = new CountDownLatch(1);
256                 repNode.getRepImpl().blockTxnCompletion(blocker);
257                 /*
258                  * >= comparison, here and below, since currentTxnEndVLSN can
259                  * lag the latest txnEndVLSN actually written to the log.
260                  */
261                 if (event.getVLSN().compareTo(endVLSN) >= 0) {
262                     result = event.replicaNodeName;
263                 }
264             } else {            /* phase 2 */
265                 if (event.getVLSN().compareTo(endVLSN) >= 0) {
266                     result = event.replicaNodeName;
267                 } else {
268 
269                     /*
270                      * The present VLSN does not match the ultimate target
271                      * VLSN, so we're not done yet.  Since there could be a few
272                      * events of this type, only log all of them at the
273                      * {@code FINE} level.
274                      */
275                     readyReplicas.put(event.replicaNodeName, event.vlsn);
276                     level = Level.FINE;
277                 }
278             }
279 
280             /* Emit log message after the fact */
281             LoggerUtils.logMsg(logger, repNode.getRepImpl(), level,
282                                "Master Transfer progress: " +
283                                event.replicaNodeName + ", " + event.vlsn +
284                                ", phase: " + (blocker == null ? 1 : 2) +
285                                ", endVLSN: " + endVLSN);
286             if (result != null) {
287                 return result;
288             }
289         }
290     }

293      * Broadcasts a fake election result message.  This does a couple things:
294      * (1) prods the chosen replica to become the new master; and (2) forces
295      * the old master to notice and shut down with a master-replica transition
296      * exception.
297      */
298     private void annouceWinner(String nodeName) {
299         final RepGroupImpl group = repNode.getGroup();
300         RepNodeImpl node = group.getNode(nodeName);
301         MasterValue newMaster = new MasterValue
302             (node.getSocketAddress().getHostName(),
303              node.getSocketAddress().getPort(),
304              node.getNameIdPair());
305         Proposal proposal =
306             new TimebasedProposalGenerator().nextProposal();
307         final Elections elections = repNode.getElections();
308         elections.getLearner();
309         Learner.informLearners
310             (group.getAllLearnerSockets(),
311              new WinningProposal(proposal, newMaster, null),
312              elections.getProtocol(),
313              elections.getThreadPool(),
314              elections.getLogger(),
315              repNode.getRepImpl(),
316              null);
317     }
