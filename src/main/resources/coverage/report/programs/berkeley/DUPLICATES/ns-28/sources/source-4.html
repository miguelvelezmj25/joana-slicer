333      * Builds a list of KeyManagers for incorporation into an SSLContext.
334      *
335      * @param ksInfo a KeyStoreInfo referencing the Keystore for which the
336      * the key manager list is to be built.
337      * @param ksAlias an optional KeyStore alias.  If set, the key manager
338      * for X509 certs will always select the certificate with the specified
339      * alias.
340      * @param clientMode set to true if this is for the client side of
341      * an SSL connection and fals otherwise.
342      */
343     private static KeyManager[] buildKeyManagerList(KeyStoreInfo ksInfo,
344                                                     String ksAlias,
345                                                     boolean clientMode) {
346 
347         /*
348          * Get a KeyManagerFactory
349          */
350         final KeyManagerFactory kmf;
351         try {
352             kmf = KeyManagerFactory.getInstance(X509_ALGO_NAME);
353         } catch (NoSuchAlgorithmException nsae) {
354             throw new IllegalStateException(
355                 "Unable to find a suitable KeyManagerFactory", nsae);
356         }
357 
358         /*
359          * Initialize the key manager factory
360          */
361         try {
362             kmf.init(ksInfo.ks, ksInfo.ksPwd);
363         } catch (KeyStoreException kse) {
364             throw new IllegalStateException(
365                 "Error processing keystore file " + ksInfo.ksFile,
366                 kse);
367         } catch (NoSuchAlgorithmException nsae) {
368             throw new IllegalStateException(
369                 "Unable to find appropriate algorithm for " +
370                 "keystore file " + ksInfo.ksFile, nsae);
371         } catch (UnrecoverableKeyException uke) {
372             throw new IllegalStateException(
373                 "Unable to recover key from keystore file " +
374                 ksInfo.ksFile, uke);
375         }
376 
377         /*
378          * Get the list of key managers used
379          */
380         KeyManager[] kmList = kmf.getKeyManagers();
381 
382         /*
383          * If an alias was specified, we need to construct an
384          * AliasKeyManager, which will delegate to the correct
385          * underlying KeyManager, which we need to locate.
386          */
387         if (ksAlias != null) {
388 
389             /*
390              * Locate the first appropriate keymanager in the list
391              */
392             X509ExtendedKeyManager x509KeyManager = null;
393             for (KeyManager km : kmList) {
394                 if (km instanceof X509ExtendedKeyManager) {
395                     x509KeyManager = (X509ExtendedKeyManager) km;
396                     break;
397                 }
398             }
399 
400             if (x509KeyManager == null) {
401                 throw new IllegalStateException(
402                     "Unable to locate an X509ExtendedKeyManager " +
403                     "corresponding to keyStore " + ksInfo.ksFile);
404             }
405 
406             kmList = new KeyManager[] {
407                 (clientMode ?
408                  new AliasKeyManager(x509KeyManager, null, ksAlias) :
409                  new AliasKeyManager(x509KeyManager, ksAlias, null)) };
410         }
411 
412         return kmList;
413     }

872     /**
873      * Load a keystore/truststore file into memory
874      * @param storeName the name of the store file
875      * @param storeFlavor a descriptive name of store type
876      * @param storeType JKS, etc
877      * @throws IllegalArgumentException if the specified parameters
878      * do now allow a store to be successfully loaded
879      */
880     private static KeyStore loadStore(String storeName,
881                                       char[] storePassword,
882                                       String storeFlavor,
883                                       String storeType)
884         throws IllegalArgumentException {
885 
886         if (storeType == null || storeType.isEmpty()) {
887             storeType = KeyStore.getDefaultType();
888         }
889 
890         final KeyStore ks;
891         try {
892             ks = KeyStore.getInstance(storeType);
893         } catch (KeyStoreException kse) {
894             throw new IllegalArgumentException(
895                 "Unable to find a " + storeFlavor + " instance of type " +
896                 storeType,
897                 kse);
898         }
899 
900         final FileInputStream fis;
901         try {
902             fis = new FileInputStream(storeName);
903         } catch (FileNotFoundException fnfe) {
904             throw new IllegalArgumentException(
905                 "Unable to locate specified " + storeFlavor +
906                 " " + storeName, fnfe);
907         }
908 
909         try {
910             ks.load(fis, storePassword);
911         } catch (IOException ioe) {
912             throw new IllegalArgumentException(
913                 "Error reading from " + storeFlavor + " file " + storeName,
914                 ioe);
915         } catch (NoSuchAlgorithmException nsae) {
916             throw new IllegalArgumentException(
917                 "Unable to check " + storeFlavor + " integrity: " + storeName,
918                 nsae);
919         } catch (CertificateException ce) {
920             throw new IllegalArgumentException(
921                 "Not all certificates could be loaded: " + storeName,
922                 ce);
923         } finally {
924             try {
925                 fis.close();
926             } catch (IOException ioe) {
927                 /* ignored */
928             }
929         }
930 
931         return ks;
932     }
