114     public long count()
115         throws DatabaseException {
116 
117         if (DbCompat.DATABASE_COUNT) {
118             return DbCompat.getDatabaseCount(db);
119         } else {
120             long count = 0;
121             DatabaseEntry key = NO_RETURN_ENTRY;
122             DatabaseEntry data = NO_RETURN_ENTRY;
123             CursorConfig cursorConfig = locking ?
124                 CursorConfig.READ_UNCOMMITTED : null;
125             Cursor cursor = db.openCursor(null, cursorConfig);
126             try {
127                 OperationStatus status = cursor.getFirst(key, data, null);
128                 while (status == OperationStatus.SUCCESS) {
129                     if (sortedDups) {
130                         count += cursor.count();
131                     } else {
132                         count += 1;
133                     }
134                     status = cursor.getNextNoDup(key, data, null);
135                 }
136             } finally {
137                 cursor.close();
138             }
139             return count;
140         }
141     }
