26 /**
27  * NamedChannelWithTimeout permits association of timeouts with a DataChannel.
28  * This mechanism is necessary, since the standard mechanism for associating
29  * timeouts with sockets using Socket.setSoTimeout is not supported by nio
30  * SocketChannels.
31  */
32 public class NamedChannelWithTimeout
33     extends NamedChannel {
34 
35     /*
36      * Denotes read activity associated with the channel. It's set each time a
37      * read is successfully executed on the channel. The presence of heartbeats
38      * is typically used to guarantee some minimum level of activity over the
39      * channel.
40      */
41     private volatile boolean readActivity;
42 
43     /*
44      * The timeout associated with the channel. A value of zero indicates no
45      * timeout.
46      */
47     private volatile int timeoutMs;
48 
49     /* Values to help with logging. */
50     private final EnvironmentImpl envImpl;
51     private final Logger logger;
52 
53     /* The "time" of the last check for read activity on the channel. */
54     private long lastCheckMs = 0l;
55 
56     public NamedChannelWithTimeout(RepNode repNode,
57                                    DataChannel channel,
58                                    int timeoutMs) {
59         this(repNode.getRepImpl(),
60              repNode.getLogger(),
61              repNode.getChannelTimeoutTask(),
62              channel,
63              timeoutMs);
64     }
65 
66     public NamedChannelWithTimeout(RepImpl repImpl,
67                                    Logger logger,
68                                    ChannelTimeoutTask channelTimeoutTask,
69                                    DataChannel channel,
70                                    int timeoutMs) {
71         super(channel);
72         this.timeoutMs = timeoutMs;
73         this.envImpl = repImpl;
74         this.logger = logger;
75         readActivity = true;
76         if (timeoutMs > 0) {
77             /* Only register with a timer, if a timeout is being requested. */
78             channelTimeoutTask.register(this);
79         }
80     }
81 
82     /**
83      * Used to modify the timeout associated with the channel.
84      *
85      * @param timeoutMs the new timeout value
86      */
87     public void setTimeoutMs(int timeoutMs) {
88         this.timeoutMs = timeoutMs;
89         /* Ensure that the next tick resets the time and counter. */
90         readActivity = true;
91     }
92 
93     /*
94      * Methods below supply the Protocol for byte channel. The implementations
95      * wrap read operations to track i/o activity by setting readActivity.
96      * getCurrentActivity() can be used to determine if there was any read
97      * activity since an earlier call to the method.
98      */
99     @Override
100     public int read(ByteBuffer dst)
101         throws IOException {
102 
103         final int bytes = channel.read(dst);
104         if (bytes > 0) {
105             readActivity = true;
106         }
107         return bytes;
108     }
109 
110     @Override
111     public void close()
112         throws IOException {
113 
114         channel.close();
115         readActivity = false;
116     }
117 
118     private void resetActivityCounter(long timeMs) {
119         lastCheckMs = timeMs;
120         readActivity = false;
121     }
122 
123     /**
124      * Method invoked by the time thread to check on the channel on a periodic
125      * basis. Note that the time that is passed in is a "pseudo" time that is
126      * only meaningful for calculating time differences.
127      *
128      * @param timeMs the pseudo time
129      *
130      * @return true if the channel is active, false if it isn't and has been
131      * closed
132      */
133     public boolean isActive(long timeMs) {
134 
135         if (!channel.isOpen()) {
136             /* some thread closed it. */
137             return false;
138         }
139 
140         if (!channel.getSocketChannel().isConnected()) {
141             /* Not yet connected, wait for it to be connected. */
142             return true;
143         }
144 
145         if (readActivity) {
146             resetActivityCounter(timeMs);
147             return true;
148         }
149 
150         if ((timeoutMs == 0) || (timeMs - lastCheckMs) < timeoutMs) {
151             return true;
152         }
153 
154         /*
155          * No activity, force the channel closed thus generating an
156          * AsynchronousCloseException in the read/write threads.
157          */
158         LoggerUtils.info(logger, envImpl,
159                          "Inactive channel: " + getNameIdPair() +
160                          " forced close. Timeout: " + timeoutMs + "ms.");
161         final long startTime = System.currentTimeMillis();
162         try {
163             channel.close();
164         } catch (IOException e) {
165             /* Ignore the exception. */
166         }
167         final long timeElapsed = System.currentTimeMillis() - startTime;
168         /* Log the unexpected closure time which is longer than 10 seconds */
169         if (timeElapsed > 10000) {
170             LoggerUtils.info(logger, envImpl,
171                              "Time to close inactive channel " +
172                              getNameIdPair() + ": " + timeElapsed + "ms");
173         }
174         return false;
175     }
