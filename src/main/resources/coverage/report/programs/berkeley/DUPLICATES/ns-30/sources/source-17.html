24      * Utility class to convert a sorted set of long values to a compact string
25      * suitable for printing. The representation is made compact by identifying
26      * ranges so that the sorted set can be represented as a sequence of hex
27      * ranges and singletons.
28      */
29     public static String asHexString(SortedSet<Long> set) {
30 
31         if (set.isEmpty()) {
32             return "";
33         }
34 
35         final StringBuilder sb = new StringBuilder();
36         java.util.Iterator<Long> i = set.iterator();
37         long rstart = i.next();
38         long rend = rstart;
39 
40         while (i.hasNext()) {
41             final long f= i.next();
42             if (f == (rend + 1)) {
43                 /* Continue the existing range. */
44                 rend++;
45                 continue;
46             }
47 
48             /* flush and start new range */
49             flushRange(sb, rstart, rend);
50             rstart = rend = f;
51         };
52 
53         flushRange(sb, rstart, rend);
54         return sb.toString();
55     }
56 
57     private static void flushRange(final StringBuilder sb,
58                                    long rstart,
59                                    long rend) {
60         if (rstart == -1) {

62         }
63

66         } else {
