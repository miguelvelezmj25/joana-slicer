119     /**
120      * Initialize starting position to the last file with a complete header
121      * with a valid checksum.
122      */
123     private void startAtLastGoodFile(Long singleFileNum)
124         throws ChecksumException {
125 
126         eof = false;
127         window.initAtFileStart(DbLsn.makeLsn(0, 0));
128 
129         /*
130          * Start at what seems like the last file. If it doesn't exist, we're
131          * done.
132          */
133         Long lastNum = ((singleFileNum != null) &&
134                         (singleFileNum.longValue() >= 0)) ?
135             singleFileNum :
136             fileManager.getLastFileNum();
137         FileHandle fileHandle = null;
138 
139         long fileLen = 0;
140         while ((fileHandle == null) && !eof) {
141             if (lastNum == null) {
142                 eof = true;
143             } else {
144                 try {
145                     try {
146                         window.initAtFileStart(DbLsn.makeLsn(lastNum, 0));
147                         fileHandle = fileManager.getFileHandle(lastNum);
148 
149                         /*
150                          * Check the size of this file. If it opened
151                          * successfully but only held a header or is 0 length,
152                          * backup to the next "last" file unless this is the
153                          * only file in the log. Note that an incomplete header
154                          * will end up throwing a checksum exception, but a 0
155                          * length file will open successfully in read only
156                          * mode.
157                          */
158                         fileLen = fileHandle.getFile().length();
159                         if (fileLen <= FileManager.firstLogEntryOffset()) {
160                             lastNum = fileManager.getFollowingFileNum
161                                 (lastNum, false);
162                             if (lastNum != null) {
163                                 fileHandle.release();
164                                 fileHandle = null;
165                             }
166                         }
167                     } catch (DatabaseException e) {
168                         lastNum = attemptToMoveBadFile(e);
169                         fileHandle = null;
170                     } catch (ChecksumException e) {
171                         lastNum = attemptToMoveBadFile(e);
172                         fileHandle = null;
173                     } finally {
174                         if (fileHandle != null) {
175                             fileHandle.release();
176                         }
177                     }
178                 } catch (IOException e) {
179                     throw new EnvironmentFailureException
180                         (envImpl, EnvironmentFailureReason.LOG_READ, e);
181                 }
182             }
183         }
184 
185         nextEntryOffset = 0;
186     }

188     /**
189      * Something is wrong with this file. If there is no data in this file (the
190      * header is <= the file header size) then move this last file aside and
191      * search the next "last" file. If the last file does have data in it,
192      * return null and throw an exception back to the application, since we're
193      * not sure what to do now.
194      *
195      * @param cause is a DatabaseException or ChecksumException.
196      */
197     private Long attemptToMoveBadFile(Exception cause)
198         throws IOException, ChecksumException, DatabaseException {
199 
200         String fileName = 
201             fileManager.getFullFileNames(window.currentFileNum())[0];
202         File problemFile = new File(fileName);
203 
204         if (problemFile.length() <= FileManager.firstLogEntryOffset()) {
205             fileManager.clear(); // close all existing files
206             /* Move this file aside. */
207             Long lastNum = fileManager.getFollowingFileNum
208                 (window.currentFileNum(), false);
209             if (!fileManager.renameFile(window.currentFileNum(),
210                                         FileManager.BAD_SUFFIX)) {
211                 throw EnvironmentFailureException.unexpectedState
212                     ("Could not rename file: 0x" +
213                       Long.toHexString(window.currentFileNum()));
214             }
215 
216             return lastNum;
217         }
218         /* There's data in this file, throw up to the app. */
219         if (cause instanceof DatabaseException) {
220             throw (DatabaseException) cause;
221         }
222         if (cause instanceof ChecksumException) {
223             throw (ChecksumException) cause;
224         }
225         throw EnvironmentFailureException.unexpectedException(cause);
226     }

275      * Validate the checksum on each entry, see if we should remember the LSN
276      * of this entry.
277      */
278     protected boolean processEntry(ByteBuffer entryBuffer) {
279 
280         /* 
281          * If we're supposed to remember this LSN, record it. Although LSN 
282          * recording is currently done for test purposes only, still do get a 
283          * valid logEntyType, so it can be used for reading in a log entry 
284          * further in this method.
285          */
286         entryType = LogEntryType.findType(currentEntryHeader.getType());
287         if (trackableEntries.contains(entryType)) {
288             lastOffsetSeen.put(entryType, Long.valueOf(currentEntryOffset));
289         }
290 
291         /*
292          * If this is a RestoreRequired entry, read it and
293          * deserialize. Otherwise, skip over the data, we're not doing anything
294          * with it.
295          */
296         if (entryType.equals(LogEntryType.LOG_RESTORE_REQUIRED)) {
297             LogEntry logEntry = entryType.getSharedLogEntry();
298             logEntry.readEntry(envImpl, currentEntryHeader, entryBuffer);
299             restoreRequired = (RestoreRequired) logEntry.getMainItem();
300         } else {
301             entryBuffer.position(entryBuffer.position() +
302                                  currentEntryHeader.getItemSize());
303         }
304 
305         return true;
306     }

309      * readNextEntry will stop at a bad entry.
310      * @return true if an element has been read.
311      */
312     @Override
313     public boolean readNextEntry() {
314 
315         boolean foundEntry = false;
316 
317         try {
318 
319             /*
320              * At this point,
321              *  currentEntryOffset is the entry we just read.
322              *  nextEntryOffset is the entry we're about to read.
323              *  currentEntryPrevOffset is 2 entries ago.
324              * Note that readNextEntry() moves all the offset pointers up.
325              */
326 
327             foundEntry = super.readNextEntryAllowExceptions();
328 
329             /*
330              * Note that initStartingPosition() makes sure that the file header
331              * entry is valid.  So by the time we get to this method, we know
332              * we're at a file with a valid file header entry.
333              */
334             lastValidOffset = currentEntryOffset;
335             nextUnprovenOffset = nextEntryOffset;
336         } catch (FileNotFoundException e) {
337             throw new EnvironmentFailureException
338                 (envImpl,
339                  EnvironmentFailureReason.LOG_FILE_NOT_FOUND, e);
340         } catch (ChecksumException e) {
341             LoggerUtils.fine
342                 (logger, envImpl,  
343                  "Found checksum exception while searching for end of log. " +
344                  "Last valid entry is at " + DbLsn.toString
345                  (DbLsn.makeLsn(window.currentFileNum(), lastValidOffset)) +
346                  " Bad entry is at " +
347                  DbLsn.makeLsn(window.currentFileNum(), nextUnprovenOffset));
348                  
349             DbConfigManager configManager = envImpl.getConfigManager();
350             boolean findCommitTxn = 
351                 configManager.getBoolean
352                 (EnvironmentParams.HALT_ON_COMMIT_AFTER_CHECKSUMEXCEPTION);
353                 
354             /* Find the committed transactions at the rest the log file. */
355             if (findCommitTxn) {
356                 boolean committedTxnFound = findCommittedTxn();  
357                 /* If we have found a committed txn. */
358                 if (committedTxnFound) {
359                     throw new EnvironmentFailureException
360                         (envImpl, EnvironmentFailureReason.FOUND_COMMITTED_TXN, 
361                          "Found committed txn after the corruption point");
362                 }
363             }
364         }
365         return foundEntry;
366     }

368     /* 
369      * [#18307] Find the committed transaction after the corrupted log entry 
370      * log file.
371      *
372      * Suppose we have LSN 1000, and the log entry there has a checksum 
373      * exception.
374      * Case 1. if the header at LSN 1000 is bad, findCommittedTxn() will not be
375      *         called, and just truncate the log.
376      *         Note: This case seems to not be handled. It seems we do call
377      *         findCommittedTxn when the header is bad. Perhaps this comment
378      *         is outdated.
379      * Case 2. if the header at LSN 1000 says the log entry size is N, we 
380      *         skip N bytes, which may let us go to the next log entry. If the 
381      *         next log entry also has a checksum exception (because size N is 
382      *         wrong, or because it really has a checksum problem), return 
383      *         false, and truncate the log.
384      * Case 3. if we manage to read past LSN 1000, but then hit a second 
385      *         checksum exception, return false and truncate the log at the 
386      *         first exception.
387      * Case 4. if we manage to read past LSN 1000, and do not see any checksum 
388      *         exceptions, and do not see any commits, return false and 
389      *         truncate the log.
390      * Case 5. if we manage to read past LSN 1000, and do not see any checksum 
391      *         exceptions, but do see a txn commit, return true and throw 
392      *         EnvironmentFailureException.
393      */
394     private boolean findCommittedTxn() {
395         try {
396         
397             /* 
398              * First we skip over the bad log entry, according to the item size
399              * we get from the log entry's header.
400              */
401             skipData(currentEntryHeader.getItemSize());
402             
403             /* 
404              * Begin searching for the committed txn from the next log entry to 
405              * the end of the log file.
406              */
407             while (super.readNextEntryAllowExceptions()) {
408                 /* Case 5. */
409                 if (LogEntryType.LOG_TXN_COMMIT.equals(entryType)) {
410                     return true;
411                 }
412             }
413         } catch (EOFException e) {
414         } catch (FileNotFoundException e) {
415             throw new EnvironmentFailureException
416                 (envImpl, EnvironmentFailureReason.LOG_FILE_NOT_FOUND, e);
417         } catch (ChecksumException e) {
418             /* Case 2 and 3. */
419             LoggerUtils.fine
420                 (logger, envImpl,  
421                  "Found checksum exception while searching for end of log. " +
422                  "Last valid entry is at " + DbLsn.toString
423                  (DbLsn.makeLsn(window.currentFileNum(), lastValidOffset)) + 
424                  " Bad entry is at " +
425                  DbLsn.makeLsn(window.currentFileNum(), nextUnprovenOffset));
426         }
427         
428         /* Case 4. */
429         return false;
430     }
