570     private boolean verifyIdentifierKeyInternal(IN in, DatabaseImpl dbImpl) {
571 
572         /*
573          * This check can only be done for full BIN, not upperIn and BIN-delta.
574          * Besides, if the slot number is 0, then we may also not check this.
575          */
576         if (in.isUpperIN() || in.isBINDelta() || in.getNEntries() == 0) {
577             return true;
578         }
579 
580         byte[] identifierKey = in.getIdentifierKey();
581         if (identifierKey == null) {
582             return false;
583         }
584

586          * There are two problematic cases about identifierKey which are caused
587          * by some errors in previous code:
588          *
589          * (1). The identifierKey is a prefix key due to the DupConvert bug.
590          *
591          *    When reading log files written by JE 4.1 or earlier, the
592          *    identifier key may be incorrect because DupConvert did not
593          *    convert it correctly. DupConvert converts the identifier key to
594          *    a prefix key, so it will not match the complete key in any slot.
595          *
596          *    We should probably fix DupConvert. But even if we fix it now,
597          *    it won't help users of JE 5.0 and above who have already upgraded
598          *    from JE 4.1 or earlier, because DupConvert is only used when
599          *    reading log files written by JE 4.1 or earlier.
600          *
601          *    This issue seems harmless, at least no user reports errors caused
602          *    by it. So we can choose to ignore this issue. Normally, we can
603          *    identify this issue by checking the end of the key for the
604          *    PREFIX_ONLY value. But unfortunately this will also ignore
605          *    identifier keys that happen to have the PREFIX_ONLY value at the
606          *    end of a complete key(in the user's data).
607          *
608          *    Considering the following second issue, we choose to not check
609          *    identifierKey for environments who is initially created with
610          *    LogEntryType.LOG_VERSION being LT 15, where 15 is just the new
611          *    log version of JE after we fix the following second issue.
612          *
613          * (2). The identifierKey is not in any slot due to the BIN-delta
614          * mutation bug.
615          * 
616          *     The fullBIN identifierKey may have changed when reconstituteBIN
617          *     called BIN.compress. The previous code forgot to reset it. Now
618          *     we fix this by reseting the identifier in BIN.mutateToFullBIN.
619          *
620          *     For the problematic identifierKey which is caused by the
621          *     BIN-delta mutation bug, we do not have good methods to correct
622          *     them. We can only detect them.
623          *
624          *     The problem with detecting them is that we know it is incorrect
625          *     in past releases, but even when it is incorrect, we don't know
626          *     the impact on the app in a particular case. It is possible that
627          *     the app is working OK, even though the identifier key is
628          *     incorrect. So if we detect it and the app stops working
629          *     (because we invalidate the env) then we may be making things
630          *     worse for the app -- this may not be what the user wants.
631          *
632          *  So combing above (1) and (2), we need to add a way to know the
633          *  earliest log version of the env. Then we can only validate the
634          *  identifierKey when this version is >= 15, where 15 is just the new
635          *  log version of JE after we fix (2). See DbTree.initialLogVersion
636          *  and LogEntryType.LOG_VERSION.
637          */
638         if (envImpl.getDbTree().getInitialLogVersion() < 15) {
639             return true;
640         }
641 
642         Comparator<byte[]> userCompareToFcn = dbImpl.getKeyComparator();
643 
644         for (int i = 0; i < in.getNEntries(); i++) {
645             byte[] key = in.getKey(i);
646             if (Key.compareKeys(identifierKey, key, userCompareToFcn) == 0) {
647                 return true;
648             }
649         }
650 
651         return false;
652     }

956         final DatabaseImpl databaseImpl = in.getDatabase();
957         /*

959          * embedded LN, we do not check the dangling LSN issue.
960          */

962             (in.isEmbeddedLN(index) || databaseImpl.getSortedDuplicates() ||

1140         DatabaseImpl dbImpl,
1141         Cursor cursor,
1142         DatabaseEntry lastKey,
1143         DatabaseEntry lastData) {
1144 
1145         DatabaseEntry usedKey = new DatabaseEntry();
1146         DatabaseEntry usedData = new DatabaseEntry();
1147         /* The first record of this db. */
1148         if (lastKey == null) {
1149             return cursor.get(
1150                 usedKey, usedData, Get.FIRST, NOLOCK_UNCHANGED) != null;
1151         }
1152 
1153         /* Find the first record according to (lastKey, lastData). */
1154         usedKey = new DatabaseEntry(
1155             lastKey.getData(), lastKey.getOffset(), lastKey.getSize());
1156         usedData = new DatabaseEntry(
1157             lastData.getData(), lastData.getOffset(), lastData.getSize());
1158 
1159         boolean isDuplicated = dbImpl.getSortedDuplicates();
1160         OperationResult result = null;
1161         if (isDuplicated) {
1162             result = cursor.get(
1163                 usedKey, usedData, Get.SEARCH_BOTH_GTE, NOLOCK_UNCHANGED);
1164             if (result != null) {
1165                 if (!usedData.equals(lastData)) {
1166                     /* Find next dup of lastKey. */
1167                     return true;
1168                 }
1169 
1170                 /*
1171                  * Find lastKey/lastData. Move to the next dup of lastKey or
1172                  * move to the first dup of next key.
1173                  */
1174                 return cursor.get(
1175                     usedKey, usedData, Get.NEXT, NOLOCK_UNCHANGED) != null;
1176             } else {
1177                 result = cursor.get(
1178                     usedKey, usedData, Get.SEARCH_GTE, NOLOCK_UNCHANGED);
1179                 if (result == null) {
1180                     /* No more records. */
1181                     return false;
1182                 }
1183 
1184                 if (!usedKey.equals(lastKey)) {
1185                     /* Find the first dup of next key. */
1186                     return true;
1187                 }
1188 
1189                 /*
1190                  * Find the first dup of lastKey. Skip over dups of lastKey
1191                  * to the first dup of next key. This may miss "phantoms" but
1192                  * that is OK -- see comments 26 and 28 in [#25960].
1193                  */
1194                 return cursor.get(
1195                     usedKey, usedData, Get.NEXT_NO_DUP,
1196                     NOLOCK_UNCHANGED) != null;
1197             }
1198         } else {
1199             result = cursor.get(
1200                 usedKey, usedData, Get.SEARCH_GTE, NOLOCK_UNCHANGED);
1201             if (result == null) {
1202                 /* No more records. */
1203                 return false;
1204             }
1205 
1206             if (!usedKey.equals(lastKey)) {
1207                 /* Find next key. */
1208                 return true;
1209             }
1210 
1211             /* Find lastKey. Move to next key. */
1212             return cursor.get(
1213                 usedKey, usedData, Get.NEXT, NOLOCK_UNCHANGED) != null;
1214         }
1215     }

1230         /* Traverse the database. */
1231         Tree tree = dbImpl.getTree();
1232         EnvironmentImpl.incThreadLocalReferenceCount();
1233         final Locker locker =
1234             LockerFactory.getInternalReadOperationLocker(envImpl);
1235         Cursor cursor = DbInternal.makeCursor(dbImpl, locker, null, false);
1236         CursorImpl cursorImpl = DbInternal.getCursorImpl(cursor);
1237         cursorImpl.setTreeStatsAccumulator(statsAcc);
1238         tree.setTreeStatsAccumulator(statsAcc);

1245         final Map<DatabaseId, DatabaseImpl> dbCache = new HashMap<>();

1252              * the data portion.

1254              * Data portion is needed only for the following two situations:
1255              * 1. Db is secondary and index needs to be checked
1256              * 2. Db is primary, verifySecondaries and verifyDataRecords are
1257              *    both true.
1258              *
1259              * Actually, now we have the following combinations:
1260              * verifySecondaries/verifyDataRecords   Meaning
1261              *
1262              *  No       No        Do not read the primary LN.  
1263              *                     Do not verify any secondaries.
1264              *
1265              *  Yes      No        Do not read the primary LN. 
1266              *                     Check that the secondary records refer to
1267              *                     existing primary records.
1268              *
1269              *  No       Yes       Read the LN as a basic check.
1270              *                     Do not verify any secondaries.
1271              *
1272              *  Yes      Yes       Read the LN as a basic check.
1273              *                     Check that the secondary records refer to
1274              *                     existing primary records.
1275              *                     Check that primary records refer to
1276              *                     existing secondary records.
1277              *
1278              * According to above combinations, only when verifySecondaries
1279              * and verifyDataRecords are both true, for a primary database,
1280              * we will check that primary records refer to existing secondary
1281              * records.
1282              *
1283              * But only if verifyDataRecords is true, for a primary database,
1284              * we need to check that the primary LN is valid, i.e. we need
1285              * to read data portion. This is why we do not use
1286              * verifyPrimaryDataRecords to replace (!isSecondaryDb &&
1287              * btreeVerifyConfig.getVerifyDataRecords()) when determining
1288              * whether we need to read the data portion.
1289              */
1290             boolean verifyPrimaryDataRecords =
1291                 priDb != null &&
1292                 btreeVerifyConfig.getVerifySecondaries() &&
1293                 btreeVerifyConfig.getVerifyDataRecords();
1294             boolean verifySecondary =
1295                 isSecondaryDb &&
1296                 btreeVerifyConfig.getVerifySecondaries();
1297             DatabaseEntry foundKey = new DatabaseEntry();
1298             DatabaseEntry foundData = new DatabaseEntry();
1299 
1300             if (!(verifySecondary ||
1301                 (priDb != null && btreeVerifyConfig.getVerifyDataRecords()))) {
1302 
1303                 foundData.setPartial(0, 0, true);
1304             }
1305 
1306             /* Whether the first record for this round check exists. */
1307             if (!findFirstRecord(dbImpl, cursor, lastKey, lastData)) {
1308                 return WalkDatabaseTreeResult.NO_MORE_RECORDS;
1309             }
1310 
1311             /*
1312              * The previous readPrimaryAfterGet implementation has a problem
1313              * when used in btree verification: it cannot detect
1314              * corruption when secDirtyRead is true and the primary record
1315              * is NOT_FOUND. In this situation, we don't have any locks,
1316              * so we don't know the true current state of either the primary or
1317              * secondary record.
1318              *
1319              * Therefore, for the index verification, we need to lock the
1320              * secondary first. And then use non-blocking lock to lock
1321              * primary record to avoid deadlock. If we cannot lock the primary
1322              * record, we can just skip the verification.
1323              * 
1324              * If verifyPrimaryDataRecords is true, we will first get the
1325              * record without acquiring a lock in this method and then try
1326              * to acquire a Read lock in verifyPrimaryData. So in
1327              * walkDatabaseTee we use READLOCK_UNCHANGED only when
1328              * verifySecondary is true.
1329              */
1330             int recordCount = 0;
1331             while (++recordCount <= batchSize) {
1332 
1333                 /* Stop the verification process asap. */
1334                 if (stopVerify) {
1335                     return WalkDatabaseTreeResult.NO_MORE_RECORDS;
1336                 }
1337 
1338                 try {
1339                     /*
1340                      * <1> For primary database:
1341                      * 1. The cursor.get(CURRENT, NEXT) used in this method
1342                      * all use lockMode NOLOCK_UNCHANGED. So there will not
1343                      * be a LockConflictException.
1344                      * 2. The (foundKey, foundData) will not be used in this
1345                      * method, so it is OK that their data array is null.
1346                      * The cursor.get(CURRENT, NEXT) in this method only aims
1347                      * to locate position. Note that, although we may verify
1348                      * primary record data, we will do that in
1349                      * verifyPrimaryData.
1350                      *
1351                      * <2> For secondary database (NOT verify secondary): the
1352                      * same with primary database.
1353                      *
1354                      * <3> For secondary database (verify secondary), a
1355                      * simple approach is problematic:
1356                      * 1. Before verifying the secondary record, we first need
1357                      * to READ lock the secondary record. So
1358                      * LockConflictException may be thrown.
1359                      * 2. The (foundKey, foundData) will be used to find the
1360                      * corresponding primary record. So foundData (priKey) can
1361                      * not be null.
1362                      * 3. We need to use nonSticky==true to avoid a deadlock
1363                      * when calling cursor.get(NEXT). But if cursor.get(NEXT)
1364                      * cannot succeed due to LockConflictException or
1365                      * something else, the cursorImpl will be reset, i.e. its
1366                      * previous location will be lost. This is not what we
1367                      * expect.
1368                      *
1369                      * The solution:
1370                      * 1. Use nonSticky==true
1371                      * 2. Use LockMode.READ_UNCOMMITTED when doing Get.NEXT.
1372                      *    This can resolve 3 above.
1373                      *    Because Get.NEXT will not acquire lock, if more
1374                      *    records exist, Get.NEXT can always succeed,
1375                      *    i.e. Get.NEXT can move to next record. So
1376                      *    'nonSticky==true' will not cause the cursorImpl to
1377                      *    move to an invalid position.
1378                      * 3. Use Get.CURRENT with LockMode.DEFAULT to
1379                      *    lock the record and read the record.
1380                      *    This can resolve 1 above.
1381                      *    This will acquire a READ lock on the record.
1382                      * 4. If Get.CURRENT in (3) returns null, i.e. the record
1383                      *    may have been deleted, then we will throw an internal
1384                      *    exception to cause the cursor to move to next slot.
1385                      *    This will resolve 2 above.
1386                      */
1387                     if (!isSecondaryDb || !verifySecondary) {
1388                         if (recordCount == 1) {
1389                             cursor.get(
1390                                 foundKey, foundData, Get.CURRENT,
1391                                 NOLOCK_UNCHANGED);
1392                         }
1393                     } else {
1394                         OperationResult result = cursor.get(
1395                             foundKey, foundData, Get.CURRENT,
1396                             READLOCK_UNCHANGED);
1397                         if (result == null) {
1398                             throw new MoveToNextRecordException();
1399                         }
1400                     }
1401 
1402                     /*
1403                      * Note that if we change this code to set nonSticky to be
1404                      * false for the cursor, then Get.NEXT will create a new
1405                      * CursorImpl, and we must refresh the CursorImpl variable.
1406                      */
1407                     cursorImpl.latchBIN();
1408                     BIN bin = cursorImpl.getBIN();
1409                     try {
1410                         verifyDanglingLSNAndObsoleteRecordsOneSlot(
1411                             cursorImpl.getIndex(), bin, true);
1412                     } finally {
1413                         cursorImpl.releaseBIN();
1414                     }
1415 
1416                     if (databaseOperDuringBatchCheckHook != null) {
1417                         if (priDb != null) {
1418                             databaseOperDuringBatchCheckHook.doHook(priDb);
1419                         } else {
1420                             databaseOperDuringBatchCheckHook.doHook(secDb);
1421                         }
1422                     }
1423 
1424                     /*
1425                      * When verifying index or foreign constraint, we
1426                      * first READ-lock the secondary record and then try
1427                      * to non-blocking READ-lock the primary record. Using
1428                      * non-blocking is to avoid deadlocks, since we are locking
1429                      * in the reverse of the usual order.
1430                      *
1431                      * If the non-blocking lock fails with
1432                      * LockNotAvailableException, we will not be able to detect
1433                      * corruption and we should ignore this exception and
1434                      * continue verification. In this case the primary record
1435                      * is write-locked and is being modified by another thread,
1436                      * so it is OK to skip this verification step in this case.
1437                      * This is a compromise.
1438                      */
1439                     if (verifySecondary) {
1440 
1441                         /*
1442                          * When isCorrupted returns true we should stop
1443                          * verifying this db, just like when
1444                          * SecondaryIntegrityException is thrown.
1445                          */
1446                         if (DbInternal.isCorrupted(secDb)) {
1447                             return WalkDatabaseTreeResult.NO_MORE_RECORDS;
1448                         }
1449 
1450                         /* For secondary database, check index integrity. */
1451                         verifyIndex(
1452                             dbImpl, secDb, cursor, foundKey, foundData);
1453 
1454                         /* For secondary database, check foreign constraint. */
1455                         verifyForeignConstraint(
1456                             secDb, cursor, foundKey, dbCache);
1457                     }
1458 
1459                     /* For a primary database, verify data. */
1460                     if (verifyPrimaryDataRecords) {
1461                         verifyPrimaryData(dbImpl, priDb, cursor, dbCache);
1462                     }
1463 
1464                     /*
1465                      * Even if we do not need the data part, for example, for
1466                      * a secondary database which does not need to check
1467                      * index issue, we may still need the data part to locate
1468                      * the first record of next batch. So for the last record
1469                      * of this batch, we need to get the data part.
1470                      */
1471                     if (recordCount == batchSize - 1) {
1472                         foundData = new DatabaseEntry();
1473                     }
1474 
1475                     /*
1476                      * For the last record of each batch, we should do all
1477                      * above check. But we can NOT continue to get NEXT
1478                      * record.
1479                      */
1480                     if (recordCount == batchSize) {
1481                         break;
1482                     }
1483 
1484                     if (cursor.get(
1485                             foundKey, foundData, Get.NEXT,
1486                             NOLOCK_UNCHANGED) == null) {
1487                         return WalkDatabaseTreeResult.NO_MORE_RECORDS;
1488                     }
1489 
1490                 } catch (StopDbVerificationException sve) {
1491                     /*
1492                      * StopDbVerificationException is thrown when
1493                      * 1. In verifyIndex, a SecondaryIntegrityException, which
1494                      *    is caused by index corruption, or a
1495                      *    IllegalStateException, which is caused by accessing
1496                      *    the closed primary database,  is caught.
1497                      * 2. In verifyForeignConstraint, the DatabaseImpl of the
1498                      *    foreign database can not be gotten or the
1499                      *    corresponding foreign record does not exist. 
1500                      * For both situations, we must stop verification of this
1501                      * db, but we should allow verification of other dbs to
1502                      * continue.
1503                      *
1504                      * No warning message needs to be logged here. For SIE,
1505                      * a message has already been logged when throwing SIE at
1506                      * the lower level.
1507                      */
1508 
1509                     return WalkDatabaseTreeResult.NO_MORE_RECORDS;
1510 
1511                 } catch (LockConflictException|MoveToNextRecordException e) {
1512                     /*
1513                      * LockConflictException can be thrown by
1514                      * Cursor.get(CURRENT) with READLOCK_UNCHANGED, which
1515                      * could be due to a normal timeout. Just move the cursor
1516                      * to next record.
1517                      *
1518                      * MoveToNextRecordException indicates that
1519                      * cursor.get(CURRENT) returns null because the record has
1520                      * been deleted. Just move the cursor to next record.
1521                      *
1522                      * These two exceptions should not prevent verification of
1523                      * other records in the same db, so we simply ignore it.
1524                      *
1525                      * If the cursor.get(NEXT, NOLOCK_UNCHANGED) here catches
1526                      * an exception, which will not be LockConflictException
1527                      * because NOLOCK_UNCHANGED is used, this is an unknown
1528                      * and unexpected exception, we just handle it in the same
1529                      * way as the following RuntimeException.
1530                      *
1531                      * TODO: A verification statistic is needed to find out
1532                      * how many times this happens. This should be returned
1533                      * and logged at the end of verification.
1534                      */
1535                     try {
1536                         if (cursor.get(
1537                             foundKey, foundData, Get.NEXT,
1538                             NOLOCK_UNCHANGED) == null) {
1539                             return WalkDatabaseTreeResult.NO_MORE_RECORDS;
1540                         }
1541                     } catch (RuntimeException re) {
1542                         LoggerUtils.logMsg(
1543                             logger, envImpl, Level.SEVERE,
1544                             "Exception aborted btree verification of db " +
1545                                 dbImpl.getDebugName() +
1546                                 ", verification of all dbs will stop. " + e);
1547 
1548                         setStopVerifyFlag(true);
1549                         return WalkDatabaseTreeResult.NO_MORE_RECORDS;
1550                     }
1551 
1552                 } catch (EnvironmentFailureException|
1553                          BtreeVerificationException e) {
1554                     throw e;
1555 
1556                 } catch (RuntimeException e) {
1557                     /*
1558                      * Consider all other exceptions. e.g. the
1559                      * OperationFailureException thrown by cursor.get which
1560                      * is not LockConflicExceptionException, to be fatal to
1561                      * the entire verification process, since we don't know
1562                      * what caused them.
1563                      */
1564                     LoggerUtils.logMsg(
1565                         logger, envImpl, Level.SEVERE,
1566                         "Exception aborted btree verification of db " +
1567                             dbImpl.getDebugName() +
1568                             ", verification of all dbs will stop. " + e);
1569 
1570                     setStopVerifyFlag(true);
1571                     return WalkDatabaseTreeResult.NO_MORE_RECORDS;
1572                 }
1573             }
1574 
1575             return new WalkDatabaseTreeResult(foundKey, foundData, false);
1576 
1577         } finally {
1578             cursorImpl.setTreeStatsAccumulator(null);
1579             tree.setTreeStatsAccumulator(null);
1580             EnvironmentImpl.decThreadLocalReferenceCount();
1581 
1582             cursor.close();
1583             locker.operationEnd();
1584 
1585             /* Release all cached DBs. */
1586             envImpl.getDbTree().releaseDbs(dbCache);
1587         }
1588     }

1590     private void verifyIndex(
1591         final DatabaseImpl dbImpl,
1592         final SecondaryDatabase secDb,
1593         final Cursor cursor,
1594         final DatabaseEntry key,
1595         final DatabaseEntry priKey)
1596         throws StopDbVerificationException {
1597 
1598         assert secDb != null;
1599 
1600         try {
1601             dbImpl.getEnv().getSecondaryAssociationLock().
1602                 readLock().lockInterruptibly();
1603         } catch (InterruptedException e) {
1604             throw new ThreadInterruptedException(dbImpl.getEnv(), e);
1605         }
1606 
1607         try {
1608             final SecondaryAssociation secAssoc =
1609                 DbInternal.getSecondaryAssociation(secDb);
1610             if (secAssoc.isEmpty()) {
1611                 return;
1612             }
1613 
1614             final Database priDb = secAssoc.getPrimary(priKey);
1615             if (priDb == null) {
1616                 return;
1617             }
1618             
1619             /*
1620              * We only need to check whether the primary record exists, we
1621              * do not need the data.
1622              */
1623             final DatabaseEntry priData = new DatabaseEntry();
1624             priData.setPartial(0, 0, true);
1625 
1626             /*
1627              * Currently the secondary record is locked. In order to avoid
1628              * deadlock, here we use the non-blocking lock. In order to
1629              * release the lock on the primary record, we create a new
1630              * Locker to acquire the lock and release the lock in finally
1631              * block.
1632              */
1633             final Locker locker =
1634                 LockerFactory.getInternalReadOperationLocker(envImpl);
1635             locker.setDefaultNoWait(true);
1636 
1637             try {
1638                 /*
1639                  * Cursor.readPrimaryAfterGet may return true or false, but for
1640                  * both cases, they do NOT indicate index corruption. Only
1641                  * throwing SecondaryIntegrityException means index corruption.
1642                  */
1643                 DbInternal.readPrimaryAfterGet(
1644                     cursor, priDb, key, priKey, priData, LockMode.DEFAULT,
1645                     false /*secDirtyRead*/, false /*lockPrimaryOnly*/,
1646                     true /*verifyOnly*/, locker, secDb, secAssoc);
1647             } catch (LockNotAvailableException e) {
1648                 /* Ignored -- see comment in walkDatabaseTree. */
1649             } finally {
1650                 /* Release primary record lock. */
1651                 locker.operationEnd();
1652             }
1653         } catch (SecondaryIntegrityException sie) {
1654             /*
1655              * Because currently the primary database is not marked as
1656              * CORRUPT, if we catch SIE here, it indicates that this SIE
1657              * was thrown by Cursor.readPrimaryAfterGet. Log related error
1658              * message here.
1659              */
1660             LoggerUtils.logMsg(
1661                 logger, envImpl, Level.WARNING,
1662                 "Secondary corruption is detected during btree " +
1663                     "verification. " + sie);
1664 
1665             throw new StopDbVerificationException();
1666 
1667         } catch (IllegalStateException ise) {
1668            /*
1669             * IllegalStateException is thrown when the primary database,
1670             * which is obtained via SecondaryAssociation.getPrimary, is
1671             * accessed after it is closed. For non-KVS apps, a secondary
1672             * database may only map to one unique primary database, and this
1673             * database will have already been closed. Therefore we just stop
1674             * the verification of the secondary database. In KVS, other primary
1675             * DBs (partitions) may still be open, but stopping verification of
1676             * the index is still acceptable.
1677             */
1678             throw new StopDbVerificationException();
1679         } finally {
1680             dbImpl.getEnv().getSecondaryAssociationLock().readLock().unlock();
1681         }
1682     }

1766     private void verifyPrimaryData(
1767         final DatabaseImpl dbImpl,
1768         final Database priDb,
1769         final Cursor cursor,
1770         final Map<DatabaseId, DatabaseImpl> dbCache) {
1771 
1772         assert priDb != null;
1773 
1774         try {
1775             dbImpl.getEnv().getSecondaryAssociationLock().
1776                 readLock().lockInterruptibly();
1777         } catch (InterruptedException e) {
1778             throw new ThreadInterruptedException(dbImpl.getEnv(), e);
1779         }
1780 
1781         try {
1782             final SecondaryAssociation secAssoc =
1783                 DbInternal.getSecondaryAssociation(priDb);
1784             if (secAssoc.isEmpty()) {
1785                 return;
1786             }
1787 
1788             DatabaseEntry key = new DatabaseEntry();
1789             DatabaseEntry data = new DatabaseEntry();
1790 
1791             /*
1792              * 1. Read the primary data portion with Read lock.
1793              * 2. If null is returned, this primary record is deleted. We
1794              *    just ignore it.
1795              * 3. If non-null is returned, the cursor, which is used by
1796              *    walkDatabaseTree, owns a read lock on the primary record.
1797              * 4. If LockConflictException is thrown, then this primary
1798              *    record is locked. Just return.
1799              */
1800             try {
1801                 if (cursor.get(
1802                     key, data, Get.CURRENT, READLOCK_UNCHANGED) == null) {
1803                     return;
1804                 }
1805             } catch (LockConflictException e) {
1806                 return;
1807             }
1808 
1809             /*
1810              * If checkSecondaryKeysExist cannot find the secondary record,
1811              * it will throw SIE. At that time, the cursor used in
1812              * checkSecondaryKeysExist is not at a meaningful slot, so we get
1813              * the expirationTime of the corresponding primary record here
1814              * and then pass it to checkSecondaryKeysExist.
1815              */
1816             for (final SecondaryDatabase secDb : secAssoc.getSecondaries(key)) {
1817                 /*
1818                  * If the primary database is removed from the
1819                  * SecondaryAssociation, then we will skip checking any
1820                  * secondary database.
1821                  *
1822                  * Besides, if the primary database is removed from the
1823                  * SecondaryAssociation, secAssoc.getPrimary may throw
1824                  * exception. 
1825                  */
1826                 try {
1827                     if (secAssoc.getPrimary(key) != priDb ) {
1828                         return;
1829                     }
1830                 } catch (Exception e) {
1831                     return;
1832                 }
1833                 
1834                 /*
1835                  * If the secondary database is in population phase, it
1836                  * may be reasonable that the BtreeVerifier can not find
1837                  * the corresponding secondary records for the checked
1838                  * primary record, because the primary record has not been
1839                  * populated to the secondary database.
1840                  */
1841                 if (secDb.isIncrementalPopulationEnabled()) {
1842                     continue;
1843                 }
1844 
1845                 checkSecondaryKeysExist(
1846                     priDb, secDb, key, data, dbCache, secAssoc,
1847                     DbInternal.getCursorImpl(cursor).getExpirationTime());
1848             }
1849         } finally {
1850             dbImpl.getEnv().getSecondaryAssociationLock().readLock().unlock();
1851         }
1852     }

1854     private void checkSecondaryKeysExist(

1857         DatabaseEntry priKey,
1858         DatabaseEntry priData,
1859         final Map<DatabaseId, DatabaseImpl> dbCache,
1860         final SecondaryAssociation secAssoc,
1861         final long expirationTime) {
1862 
1863         if (DbInternal.isCorrupted(secDb)) {
1864             /*
1865              * If the secondary database is set to be CORRUPT, then we will
1866              * not check this database. Just quick return.
1867              */
1868             return;
1869         }
1870 
1871         final SecondaryConfig secondaryConfig =
1872             DbInternal.getPrivateSecondaryConfig(secDb);

1874         final SecondaryMultiKeyCreator multiKeyCreator =
1875             secondaryConfig.getMultiKeyCreator();
