49  * @hidden
50  * Dumps the log using VLSNReaders. These are special FileReaders which use the
51  * vlsnIndex to optimize their traversal of the log. Most common use for this
52  * utility is as a debugging device for the VLSNReaders, because it mimics the
53  * way feeders and syncup search the log.
54  */

57     /**
58      * Dump a JE log into human readable form.
59      * @throws InterruptedException
60      */
61     private void dump(File envHome,
62                       String groupName,
63                       String nodeName,
64                       String host,
65                       boolean forward,
66                       long startLsn,
67                       VLSN startVLSN,
68                       @SuppressWarnings("unused") boolean verbose)
69         throws IOException,
70                EnvironmentNotFoundException,
71                EnvironmentLockedException,
72                DatabaseException,
73                InterruptedException,
74                ChecksumException {
75 
76         /* Create a single replicator */
77         Durability durability = new Durability(SyncPolicy.NO_SYNC,
78                                                SyncPolicy.NO_SYNC,
79                                                ReplicaAckPolicy.NONE);
80 
81         EnvironmentConfig envConfig = new EnvironmentConfig();
82         envConfig.setAllowCreate(false);
83         envConfig.setTransactional(true);
84         envConfig.setDurability(durability);
85 
86         ReplicationConfig repConfig = new ReplicationConfig();
87         repConfig.setConfigParam
88             (ReplicationConfig.ENV_CONSISTENCY_TIMEOUT, "1 min");
89         repConfig.setGroupName(groupName);
90         repConfig.setNodeName(nodeName);
91         int port = Integer.parseInt(RepParams.DEFAULT_PORT.getDefault());
92         String hostName = host + ":" + port;
93         repConfig.setNodeHostPort(hostName);
94         repConfig.setHelperHosts(hostName);
95         final ReplicatedEnvironment rep =
96             RepInternal.createDetachedEnv(envHome, repConfig, envConfig);
97         try {
98             ReplicatedEnvironment.State state = rep.getState();
99             if (state != ReplicatedEnvironment.State.DETACHED) {
100                 throw EnvironmentFailureException.unexpectedState
101                     ("joinState=" + state);
102             }
103 
104             EnvironmentImpl envImpl =
105                 DbInternal.getNonNullEnvImpl(rep);
106             VLSNIndex vlsnIndex =
107                 RepInternal.getNonNullRepImpl(rep).getVLSNIndex();
108             if (forward) {
109                 startLsn = DbLsn.NULL_LSN;
110             } else {
111                 startLsn = envImpl.getFileManager().getLastUsedLsn();
112             }
113 
114             int readBufferSize =
115                 envImpl.getConfigManager().getInt
116                 (EnvironmentParams.LOG_ITERATOR_READ_SIZE);
117             FeederReader feederReader = null;
118             FeederSyncupReader backwardsReader = null;
119 
120             VLSN scanVLSN = startVLSN;
121             if (startVLSN.equals(VLSN.NULL_VLSN)) {
122                 RepImpl repImpl = RepInternal.getNonNullRepImpl(rep);
123                 VLSNRange range = repImpl.getVLSNIndex().getRange();
124                 if (forward) {
125                     scanVLSN = range.getFirst();
126                 } else {
127                     scanVLSN = range.getLast();
128                 }
129             }
130             if (scanVLSN.equals(VLSN.NULL_VLSN)) {
131                 /* Give up, no VLSN entries in the index. */
132                 System.out.println("<DbPrintLog>");
133                 System.out.println("</DbPrintLog>");
134                 return;
135             }
136 
137             if (forward) {
138                 feederReader = new FeederReader(envImpl, vlsnIndex, startLsn,
139                                                 readBufferSize);
140                 feederReader.initScan(scanVLSN);
141             } else {
142                 backwardsReader = new FeederSyncupReader
143                     (envImpl, vlsnIndex,
144                      startLsn, readBufferSize,
145                      scanVLSN,
146                      DbLsn.NULL_LSN);
147             }
148 
149             OutputWireRecord record = null;
150             System.out.println("<DbPrintLog>");
151             long lastLsn = 0;
152             do  {
153                 if (forward) {
154                     record = feederReader.scanForwards(scanVLSN, 0);
155                     scanVLSN = scanVLSN.getNext();
156                     lastLsn = feederReader.getLastLsn();
157                 } else {
158                     record = backwardsReader.scanBackwards(scanVLSN);
159                     scanVLSN = scanVLSN.getPrev();
160                     lastLsn = backwardsReader.getLastLsn();
161                 }
162                 if (record != null) {
163                     System.out.println
164                     ("lsn=" + DbLsn.getNoFormatString(lastLsn) + " " + record);
165                 }
166             } while (record != null);
167 
168             System.out.println("</DbPrintLog>");
169         } finally {
170             rep.close();
171         }
172     }

174     public static void main(String[] argv) {
175         try {
176             int whichArg = 0;
177             String groupName = "repGroup";
178             String host = "localHost";
179             String nodeName = "Node1";
180             long startLsn = DbLsn.NULL_LSN;
181             VLSN startVLSN = VLSN.NULL_VLSN;
182             boolean verbose = true;
183             boolean forward = true;
184 
185             /* Default to looking in current directory. */
186             File envHome = new File(".");
187             Key.DUMP_TYPE = DumpType.BINARY;
188 
189             while (whichArg < argv.length) {
190                 String nextArg = argv[whichArg];
191                 if (nextArg.equals("-h")) {
192                     whichArg++;
193                     envHome = new File(CmdUtil.getArg(argv, whichArg));
194                 } else if (nextArg.equals("-s")) {
195                     whichArg++;
196                     String arg = CmdUtil.getArg(argv, whichArg);
197                     int slashOff = arg.indexOf("/");
198                     if (slashOff < 0) {
199                         long startFileNum = CmdUtil.readLongNumber(arg);
200                         startLsn = DbLsn.makeLsn(startFileNum, 0);
201                     } else {
202                         long startFileNum =
203                             CmdUtil.readLongNumber(arg.substring(0, slashOff));
204                         long startOffset = CmdUtil.readLongNumber
205                             (arg.substring(slashOff + 1));
206                         startLsn = DbLsn.makeLsn(startFileNum, startOffset);
207                     }
208                 } else if (nextArg.equals("-e")) {
209                     whichArg++;
210                     String arg = CmdUtil.getArg(argv, whichArg);
211                     int slashOff = arg.indexOf("/");
212                     /* SuppressWarnings because -e is not yet implemented.*/
213                     if (slashOff < 0) {
214                         @SuppressWarnings("unused")
215                         long endFileNum = CmdUtil.readLongNumber(arg);
216                     } else {
217                         @SuppressWarnings("unused")
218                         long endFileNum =
219                             CmdUtil.readLongNumber(arg.substring(0, slashOff));
220                         @SuppressWarnings("unused")
221                         long endOffset = CmdUtil.readLongNumber
222                             (arg.substring(slashOff + 1));
223                     }
224                 } else if (nextArg.equals("-q")) {
225                     verbose = false;
226                 } else if (nextArg.equals("-backward")) {
227                     forward = false;
228                 } else if (nextArg.equals("-vlsn")) {
229                     whichArg++;
230                     String arg = CmdUtil.getArg(argv, whichArg);
231                     startVLSN = new VLSN(CmdUtil.readLongNumber(arg));
232                 } else {
233                     System.err.println
234                         (nextArg + " is not a supported option.");
235                     usage();
236                     System.exit(-1);
237                 }
238                 whichArg++;
239             }
240 
241             DbFeederPrintLog printer = new DbFeederPrintLog();
242             printer.dump(envHome, groupName, nodeName, host, forward, startLsn,
243                          startVLSN, verbose);
244 
245         } catch (Throwable e) {
246             e.printStackTrace();
247             System.out.println(e.getMessage());
248             usage();
249             System.exit(1);
250         }
251     }
