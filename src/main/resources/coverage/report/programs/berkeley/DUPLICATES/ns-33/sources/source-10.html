261     private void initAndRetry(final String storePrefix,
262                               final EntityModel modelParam,
263                               final Mutations mutationsParam)
264         throws DatabaseException {
265 
266         for (int i = 0;; i += 1) {
267             Transaction txn = null;
268             if (transactional && DbCompat.getThreadTransaction(env) == null) {
269                 txn =
270                     env.beginTransaction(null, store.getAutoCommitTxnConfig());
271             }
272             boolean success = false;
273             try {
274                 init(txn, storePrefix, modelParam, mutationsParam);
275                 success = true;
276                 return;
277             } catch (LockConflictException e) {
278 
279                 /*
280                  * It is very unlikely that two threads opening the same
281                  * EntityStore will cause a lock conflict.  However, because we
282                  * read-modify-update the catalog record,
283                  * LockPreemptedException must be handled in a replicated JE
284                  * environment.  Since LockPreemptedException is a
285                  * LockConfictException, it is simplest to retry when any
286                  * LockConfictException occurs.
287                  */
288                 if (i >= MAX_TXN_RETRIES) {
289                     throw e;
290                 }
291                 continue;
292             } finally {
293 
294                 /*
295                  * If the catalog is read-only we abort rather than commit,
296                  * because a ReplicaWriteException may have occurred.
297                  * ReplicaWriteException invalidates the transaction, and there
298                  * are no writes to commit anyway. [#16655]
299                  */
300                 if (txn != null) {
301                     if (success && !isReadOnly()) {
302                         txn.commit();
303                     } else {
304                         txn.abort();
305                     }
306                 }
307             }
308         }
309     }

311     private void init(final Transaction txn,
312                       final String storePrefix,
313                       final EntityModel modelParam,
314                       final Mutations mutationsParam)
315         throws DatabaseException {
316 
317         try {
318             initData = readData(txn);
319             mutations = initData.mutations;
320             if (mutations == null) {
321                 mutations = new Mutations();
322             }
323 
324             /*
325              * When the beta version is detected, force a re-write of the
326              * catalog and disallow class changes.  This brings the catalog up
327              * to date so that evolution can proceed correctly from then on.
328              */
329             boolean betaVersion = (initData.version == BETA_VERSION);
330             boolean needWrite = betaVersion;
331             boolean disallowClassChanges = betaVersion;
332 
333             /*
334              * Store the given mutations if they are different from the stored
335              * mutations, and force evolution to apply the new mutations.
336              */
337             boolean forceEvolution = false;
338             if (mutationsParam != null &&
339                 !mutations.equals(mutationsParam)) {
340                 mutations = mutationsParam;
341                 needWrite = true;
342                 forceEvolution = true;
343             }
344 
345             final ClassLoader envClassLoader = DbCompat.getClassLoader(env);
346 
347             /* Get the existing format list, or copy it from SimpleCatalog. */
348             formatList = initData.formatList;
349             if (formatList == null) {
350                 formatList = SimpleCatalog.getAllSimpleFormats(envClassLoader);
351 
352                 /*
353                  * Special cases: Object and Number are predefined but are not
354                  * simple types.
355                  */
356                 Format format = new NonPersistentFormat(this, Object.class);
357                 format.setId(Format.ID_OBJECT);
358                 formatList.set(Format.ID_OBJECT, format);
359                 format = new NonPersistentFormat(this, Number.class);
360                 format.setId(Format.ID_NUMBER);
361                 formatList.set(Format.ID_NUMBER, format);
362             } else {
363                 /* Pick up any new predefined simple types. */
364                 if (SimpleCatalog.addMissingSimpleFormats(envClassLoader,
365                                                           formatList)) {
366                     needWrite = true;
367                 }
368                 nStoredFormats = formatList.size();
369             }
370 
371             /* Initialize transient catalog field before further use. */
372             for (Format format : formatList) {
373                 if (format != null) {
374                     format.initCatalog(this);
375                 }
376             }
377 
378             /* Special handling for JE 3.0.12 beta formats. */
379             if (betaVersion) {
380                 Map<String, Format> formatMap = new HashMap<String, Format>();
381                 for (Format format : formatList) {
382                     if (format != null) {
383                         formatMap.put(format.getClassName(), format);
384                     }
385                 }
386                 for (Format format : formatList) {
387                     if (format != null) {
388                         format.migrateFromBeta(formatMap);
389                     }
390                 }
391             }
392 
393             /*
394              * If we should not use the current model, initialize the stored
395              * model and return.
396              */
397             formatMap = new HashMap<String, Format>(formatList.size());
398             latestFormatMap = new HashMap<String, Format>(formatList.size());
399             if (rawAccess) {
400                 for (Format format : formatList) {
401                     if (format != null) {
402                         String name = format.getClassName();
403                         if (format.isCurrentVersion()) {
404                             formatMap.put(name, format);
405                         }
406                         if (format == format.getLatestVersion()) {
407                             latestFormatMap.put(name, format);
408                         }
409                     }
410                 }
411                 if (modelParam != null) {
412                     model = modelParam;
413                     storedModel = (StoredModel) modelParam;
414                 } else {
415                     storedModel = new StoredModel(this);
416                     model = storedModel;
417                 }
418                 ModelInternal.setClassLoader(model, envClassLoader);
419                 for (Format format : formatList) {
420                     if (format != null) {
421                         format.initializeIfNeeded(this, model);
422                     }
423                 }
424                 initModelAndMutations();
425                 return;
426             }
427 
428             /*
429              * We are opening a store that uses the current model. Default to
430              * the AnnotationModel if no model is specified.
431              */
432             if (modelParam != null) {
433                 model = modelParam;
434             } else {
435                 model = new AnnotationModel();
436             }
437             ModelInternal.setClassLoader(model, envClassLoader);
438             storedModel = null;
439 
440             /*
441              * Add all predefined (simple) formats to the format map.  The
442              * current version of other formats will be added below.
443              */
444             for (int i = 0; i <= Format.ID_PREDEFINED; i += 1) {
445                 Format simpleFormat = formatList.get(i);
446                 if (simpleFormat != null) {
447                     formatMap.put(simpleFormat.getClassName(), simpleFormat);
448                 }
449             }
450 
451             /*
452              * Known classes are those explicitly registered by the user via
453              * the model, plus the predefined proxy classes.
454              */
455             List<String> knownClasses =
456                 new ArrayList<String>(model.getKnownClasses());
457             /* Also adds the special classes, i.e., enum or array. [#19377] */
458             knownClasses.addAll(model.getKnownSpecialClasses());
459             addPredefinedProxies(knownClasses);
460 
461             /*
462              * Create a temporary map of proxied class name to proxy class
463              * name, using all known formats and classes.  This map is used to
464              * force proxy formats to be created prior to proxied formats.
465              * [#14665]
466              */
467             proxyClassMap = new HashMap<String, String>();
468             for (Format oldFormat : formatList) {
469                 if (oldFormat == null || Format.isPredefined(oldFormat)) {
470                     continue;
471                 }
472                 String oldName = oldFormat.getClassName();
473                 Renamer renamer = mutations.getRenamer
474                     (oldName, oldFormat.getVersion(), null);
475                 String newName =
476                     (renamer != null) ? renamer.getNewName() : oldName;
477                 addProxiedClass(newName, false /*isKnownClass*/);
478             }
479             for (String className : knownClasses) {
480                 addProxiedClass(className, true /*isKnownClass*/);
481             }
482 
483             /*
484              * Add known formats from the model and the predefined proxies.
485              * In general, classes will not be present in an AnnotationModel
486              * until an instance is stored, in which case an old format exists.
487              * However, registered proxy classes are an exception and must be
488              * added in advance.  And the user may choose to register new
489              * classes in advance.  The more formats we define in advance, the
490              * less times we have to write to the catalog database.
491              */
492             Map<String, Format> newFormats = new HashMap<String, Format>();
493             for (String className : knownClasses) {
494                 createFormat(className, newFormats);
495             }
496 
497             /*

499              * exception that contains the messages for all of the errors in
500              * mutations or in the definition of new classes.
501              */
502             initEvolver = new Evolver
503                 (this, storePrefix, mutations, newFormats, forceEvolution,
504                  disallowClassChanges);
505             for (Format oldFormat : formatList) {
506                 if (oldFormat == null || Format.isPredefined(oldFormat)) {
507                     continue;
508                 }
509                 if (oldFormat.isEntity()) {
510                     initEvolver.evolveFormat(oldFormat);
511                 } else {
512                     initEvolver.addNonEntityFormat(oldFormat);
513                 }
514             }
515             initEvolver.finishEvolution();
516             String errors = initEvolver.getErrors();
517             if (errors != null) {
518                 throw new IncompatibleClassException(errors);
519             }
520 
521             /*
522              * Add the new formats remaining.  New formats that are equal to
523              * old formats were removed from the newFormats map above.
524              */
525             for (Format newFormat : newFormats.values()) {
526                 addFormat(newFormat);
527             }
528 
529             /* Initialize all formats. */
530             for (Format format : formatList) {
531                 if (format != null) {
532                     format.initializeIfNeeded(this, model);
533                     if (format == format.getLatestVersion()) {
534                         latestFormatMap.put(format.getClassName(), format);
535                     }
536                 }
537             }
538 
539             final boolean formatsChanged =
540                  newFormats.size() > 0 ||
541                  initEvolver.areFormatsChanged();
542             needWrite |= formatsChanged;
543 
544             /* For unit testing. */
545             if (expectNoClassChanges && formatsChanged) {
546                 throw new IllegalStateException
547                     ("Unexpected changes " +
548                      " newFormats.size=" + newFormats.size() +
549                      " areFormatsChanged=" + initEvolver.areFormatsChanged());
550             }
551 
552             readOnly = db.getConfig().getReadOnly();
553 
554             /* Write the catalog if anything changed. */
555             if (needWrite && !readOnly) {
556 
557                 /* <!-- begin JE only --> */
558                 try {
559                 /* <!-- end JE only --> */
560 
561                     /*
562                      * Only rename/remove databases if we are going to update
563                      * the catalog to reflect those class changes.
564                      */
565                     initEvolver.renameAndRemoveDatabases(store, txn);
566 
567                     /*
568                      * Note that we use the Data object that was read above,
569                      * and the beta version determines whether to delete the
570                      * old mutations record.
571                      */
572                     initData.formatList = formatList;
573                     initData.mutations = mutations;
574                     writeData(txn, initData);
575                 /* <!-- begin JE only --> */
576                 } catch (ReplicaWriteException e) {
577                     readOnly = true;
578                 }
579                 /* <!-- end JE only --> */
580             }
581             initModelAndMutations();
582         } finally {
583 
584             /*
585              * Fields needed only for the duration of this ctor and which
586              * should be null afterwards.
587              */
588             proxyClassMap = null;
589             initData = null;
590             initEvolver = null;
591         }
592     }

616     private void addProxiedClass(String className, boolean isKnownClass) {
617         ClassMetadata metadata = model.getClassMetadata(className);
618         if (metadata != null) {
619             String proxiedClassName = metadata.getProxiedClassName();
620             if (proxiedClassName != null) {
621                 
622                 /* 
623                  * If the class is a registered known class, need to check if 
624                  * registering proxy class is allowed or not. Currently, only 
625                  * SimpleType is not allowed to register a proxy class.
626                  */
627                 if (isKnownClass) {
628                     try {
629                         Class type = resolveClass(proxiedClassName);
630                         
631                         /*
632                          * Check if the proxied class is allowed to register a 
633                          * proxy class. If not, IllegalArgumentException will 
634                          * be thrown.
635                          */
636                         if(!SimpleCatalog.allowRegisterProxy(type)) {
637                             throw new IllegalArgumentException
638                             ("Registering proxy is not allowed for " + 
639                              proxiedClassName + 
640                              ", which is a built-in simple type.");
641                         }
642                     } catch (ClassNotFoundException e) {
643                         throw DbCompat.unexpectedState
644                         ("Class does not exist: " + proxiedClassName);
645                     }
646                 }
647                 proxyClassMap.put(proxiedClassName, className);
648             }
649         }

681                 }
682                 subclass.add(format);
683             }
684         }
685         return subclassMap;

754     /**
755      * If the given class format is not already present in the given map and
756      * a format for this class name does not already exist, creates an
757      * uninitialized format, adds it to the map, and also collects related
758      * formats in the map.
759      */
760     public Format createFormat(Class type, Map<String, Format> newFormats) {
761         /* Return a new or existing format for this class. */
762         String className = type.getName();
763         Format format = getFormatFromMap(type, newFormats);
764         if (format != null) {
765             return format;
766         }
767         format = getFormatFromMap(type, formatMap);
768         if (format != null) {
769             return format;
770         }
771         /* Simple types are predefined. */
772         assert !SimpleCatalog.isSimpleType(type) : className;
773 
774         /*
775          * Although metadata is only needed for a complex type, call
776          * getClassMetadata for all types to support checks for illegal
777          * metadata on other types.
778          */
779         ClassMetadata metadata = model.getClassMetadata(className);
780         /* Create format of the appropriate type. */
781         String proxyClassName = null;
782         if (proxyClassMap != null) {
783             proxyClassName = proxyClassMap.get(className);
784         }
785         if (proxyClassName != null) {
786             format = new ProxiedFormat(this, type, proxyClassName);
787         } else if (type.isArray()) {
788             format = type.getComponentType().isPrimitive() ?
789                 (new PrimitiveArrayFormat(this, type)) :
790                 (new ObjectArrayFormat(this, type));
791         } else if (type.isEnum()) {
792             format = new EnumFormat(this, type);
793         } else if (type.getEnclosingClass() != null && 
794                    type.getEnclosingClass().isEnum()) {
795 
796             /* 
797              * If the type is an anonymous class of an enum class, the format
798              * which represents the enum class will be created. [#18357]
799              */
800             format = new EnumFormat(this, type.getEnclosingClass());
801         } else if (type == Object.class || type.isInterface()) {
802             format = new NonPersistentFormat(this, type);
803         } else {
804             if (metadata == null) {
805                 throw new IllegalArgumentException
806                     ("Class could not be loaded or is not persistent: " +
807                      className);
808             }
809             if (metadata.getCompositeKeyFields() != null &&
810                 (metadata.getPrimaryKey() != null ||
811                  metadata.getSecondaryKeys() != null)) {
812                 throw new IllegalArgumentException
813                     ("A composite key class may not have primary or" +
814                      " secondary key fields: " + type.getName());
815             }
816 
817             /*
818              * Check for inner class before default constructor, to give a
819              * specific error message for each.
820              */
821             if (type.getEnclosingClass() != null &&
822                 !Modifier.isStatic(type.getModifiers())) {
823                 throw new IllegalArgumentException
824                     ("Inner classes not allowed: " + type.getName());
825             }
826             try {
827                 type.getDeclaredConstructor();
828             } catch (NoSuchMethodException e) {
829                 throw new IllegalArgumentException
830                     ("No default constructor: " + type.getName(), e);
831             }
832             if (metadata.getCompositeKeyFields() != null) {
833                 format = new CompositeKeyFormat
834                     (this, type, metadata,
835                      metadata.getCompositeKeyFields());
836             } else {
837                 EntityMetadata entityMetadata =
838                     model.getEntityMetadata(className);
839                 format =
840                     new ComplexFormat(this, type, metadata, entityMetadata);
841             }
842         }
843         /* Collect new format along with any related new formats. */
844         newFormats.put(className, format);
845         format.collectRelatedFormats(this, newFormats);
846 
847         return format;
848     }

968 
969         if (formatId < 0) {
970             throw DbCompat.unexpectedState
971                 ("Format ID " + formatId + " is negative," +
972                  " may indicate data corruption.");
973         }
974 
975         /**
976          * If we're attempting to read a record containing a format ID that is
977          * greater than the maximum known stored format, then we refresh the
978          * formats from disk, expecting that the stored formats have been
979          * updated by the Master node.  Note that format IDs greater than
980          * nStoredFormats may exist in the formatList, if evolution took place
981          * on this Replica in a read-only mode.  Such formats are never written
982          * (Replicas do not write) and cannot be used for reading an existing

997         if (format == null) {
998             throw DbCompat.unexpectedState
999                 ("Format ID " + formatId + " has null format," +
1000                  " may indicate data corruption.");
1001         }
1002 
1003         /*
1004          * Currently we can't throw DeletedClassException because we should not
1005          * do this if we're being called during a Conversion, and we don't have
1006          * that state information available.
1007          */
1008         /*
1009         if (format.isDeleted()) {
1010             throw new DeletedClassException
1011                 ("Class " + format.getClassName() +
1012                  " was deleted with a Deleter muation, format ID " +
1013                  formatId + '.');
1014         }
1015         */

1021      * Get a format for a given class, creating it if it does not exist.
1022      *
1023      * <p>This method is called for top level entity instances by
1024      * PersistEntityBinding.  When a new entity subclass format is added we
1025      * call Store.checkEntitySubclassSecondaries to ensure that all secondary
1026      * databases have been opened, before storing the entity.  We do this here
1027      * while not holding a synchronization mutex, not in addNewFormat, to avoid
1028      * deadlocks. checkEntitySubclassSecondaries synchronizes on the Store.
1029      * [#16399]</p>
1030      *
1031      * <p>Historical note:  At one time we opened / created the secondary
1032      * databases rather than requiring the user to open them, see [#15247].
1033      * Later we found this to be problematic since a user txn may have locked
1034      * primary records, see [#16399].</p>
1035      */
1036     public Format getFormat(Class cls, boolean checkEntitySubclassIndexes)
1037         throws RefreshException {
1038 
1039         Format format = formatMap.get(cls.getName());
1040         if (format == null) {
1041             if (model != null) {
1042                 format = addNewFormat(cls);
1043                 /* Detect and handle new entity subclass. [#15247] */
1044                 if (checkEntitySubclassIndexes && store != null) {
1045                     Format entityFormat = format.getEntityFormat();
1046                     if (entityFormat != null && entityFormat != format) {
1047                         try {
1048                             store.checkEntitySubclassSecondaries
1049                                 (entityFormat.getEntityMetadata(),
1050                                  cls.getName());
1051                         } catch (DatabaseException e) {
1052                             throw RuntimeExceptionWrapper.wrapIfNeeded(e);
1053                         }
1054                     }
1055                 }
1056             }
1057             if (format == null) {
1058                 throw new IllegalArgumentException
1059                     ("Class is not persistent: " + cls.getName());
1060             }
1061         }
1062 
1063         return format;
1064     }

1128     /**
1129      * Adds a format for a new class.  Returns the format added for the given
1130      * class, or throws an exception if the given class is not persistent.
1131      *
1132      * <p>This method uses a copy-on-write technique to add new formats without
1133      * impacting other threads.</p>
1134      */
1135     private synchronized Format addNewFormat(Class cls)
1136         throws RefreshException {
1137 
1138         /*
1139          * After synchronizing, check whether another thread has added the
1140          * format needed.  Note that this is not the double-check technique
1141          * because the formatMap field is volatile and is not itself checked
1142          * for null.  (The double-check technique is known to be flawed in
1143          * Java.)
1144          */
1145         Format format = getFormatFromMap(cls, formatMap);
1146         if (format != null) {
1147             return format;
1148         }
1149 
1150         /* Copy the read-only format collections. */
1151         List<Format> newFormatList = new ArrayList<Format>(formatList);
1152         Map<String, Format> newFormatMap =
1153             new HashMap<String, Format>(formatMap);
1154         Map<String, Format> newLatestFormatMap =
1155             new HashMap<String, Format>(latestFormatMap);
1156 
1157         /* Add the new format and all related new formats. */
1158         Map<String, Format> newFormats = new HashMap<String, Format>();
1159         format = createFormat(cls, newFormats);
1160         for (Format newFormat : newFormats.values()) {
1161             addFormat(newFormat, newFormatList, newFormatMap);
1162         }
1163 
1164         /*
1165          * Initialize new formats using a read-only catalog because we can't
1166          * update this catalog until after we store it (below).
1167          */
1168         Catalog newFormatCatalog = new ReadOnlyCatalog
1169             (ModelInternal.getClassLoader(model), newFormatList, newFormatMap);
1170         for (Format newFormat : newFormats.values()) {
1171             newFormat.initializeIfNeeded(newFormatCatalog, model);
1172             newLatestFormatMap.put(newFormat.getClassName(), newFormat);
1173         }
1174 
1175         /*
1176          * Write the updated catalog using auto-commit, then assign the new
1177          * collections.  The database write must occur before the collections
1178          * are used, since a format must be persistent before it can be
1179          * referenced by a data record.
1180          *
1181          * In readOnly mode, which includes Replica upgrade mode, we should not
1182          * attempt to write since we could be elected Master and write stale
1183          * metadata.  If ReplicaWriteException occurs then we transition to
1184          * Replica upgrade mode in the same manner as in the init() method.
1185          * This can happen when no schema change is made except for one or more
1186          * new entity classes.  The new entity class will not be detected by
1187          * evolution (during init()) but will be detected here if the user
1188          * calls getPrimaryIndex.  [#16655]
1189          */
1190         if (!readOnly) {
1191             try {
1192                 Data newData = new Data();
1193                 newData.formatList = newFormatList;
1194                 newData.mutations = mutations;
1195                 writeDataCheckStale(newData);
1196             /* <!-- begin JE only --> */
1197             } catch (ReplicaWriteException e) {
1198                 readOnly = true;
1199             /* <!-- end JE only --> */
1200             } catch (DatabaseException e) {
1201                 throw RuntimeExceptionWrapper.wrapIfNeeded(e);
1202             }
1203         }
1204         formatList = newFormatList;
1205         formatMap = newFormatMap;
1206         latestFormatMap = newLatestFormatMap;
1207 
1208         return format;
1209     }

1264                 throw DbCompat.unexpectedException(e);
1265             } catch (IOException e) {
1266                 throw DbCompat.unexpectedException(e);
1267             }
1268         } else {
1269             oldData = new Data();
1270             oldData.version = Catalog.CURRENT_VERSION;
1271         }

1289     /**
1290      * Writes catalog Data after checking for stale metadata.
1291      */
1292     private void writeDataCheckStale(Data newData)
1293         throws DatabaseException, RefreshException {
1294 
1295         for (int i = 0;; i += 1) {
1296             Transaction txn = null;
1297             if (transactional && DbCompat.getThreadTransaction(env) == null) {
1298                 txn =
1299                     env.beginTransaction(null, store.getAutoCommitTxnConfig());
1300             }
1301             boolean success = false;
1302             try {
1303                 if (isMetadataStale(txn)) {
1304                     throw new RefreshException(store, this,  -1 /*formatId*/);
1305                 }
1306                 writeData(txn, newData);
1307                 success = true;
1308                 return;
1309             } catch (LockConflictException e) {
1310 
1311                 /*
1312                  * A lock conflict should not occur because writes to the
1313                  * catalog DB are in synchronized methods.  However, because we
1314                  * read-modify-update the catalog record,
1315                  * LockPreemptedException must be handled in a replicated JE
1316                  * environment.  Since LockPreemptedException is a
1317                  * LockConfictException, it is simplest to retry when any
1318                  * LockConfictException occurs.
1319                  */
1320                 if (i >= MAX_TXN_RETRIES) {
1321                     throw e;
1322                 }
1323                 continue;
