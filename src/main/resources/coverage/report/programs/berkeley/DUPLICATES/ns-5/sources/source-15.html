101     public final String readString()
102         throws IndexOutOfBoundsException, IllegalArgumentException  {
103 
104         byte[] myBuf = buf;
105         int myOff = off;
106         if (available() >= 2 &&
107             myBuf[myOff] == TupleOutput.NULL_STRING_UTF_VALUE &&
108             myBuf[myOff + 1] == 0) {
109             skip(2);
110             return null;
111         } else {
112             int byteLen = UtfOps.getZeroTerminatedByteLength(myBuf, myOff);
113             skip(byteLen + 1);
114             return UtfOps.bytesToString(myBuf, myOff, byteLen);
115         }
116     }

785      * Reads a sorted {@code BigDecimal} in normalized format with a single
786      * digit to the left of the decimal point.
787      */
788     private BigDecimal readSortedNormalizedBigDecimal() {
789 
790         StringBuilder valStr = new StringBuilder(32);
791         int subVal = readSortedPackedInt();
792         int sign = subVal < 0 ? -1 : 1;
793         
794         /* Read through the buf, until we meet the terminator byte. */
795         while (subVal != -1) {
796         
797             /* Adjust the sub-value back to the original. */
798             subVal = subVal < 0 ? subVal + 1 : subVal;
799             String groupDigits = Integer.toString(Math.abs(subVal));
800         
801             /* 
802              * subVal < 100000000 means some leading zeros have been removed,
803              * we have to add them back.
804              */ 
805             if (groupDigits.length() < 9) {
806                 final int insertLen = 9 - groupDigits.length();
807                 for (int i = 0; i < insertLen; i++) {
808                     valStr.append("0");
809                 }
810             }
811             valStr.append(groupDigits);
812             subVal = readSortedPackedInt();
813         }
814 
815         BigInteger digitsVal = new BigInteger(valStr.toString());
816         if (sign < 0) {
817             digitsVal = digitsVal.negate();
818         }
819         /* The normalized decimal has 1 digits in the int part. */
820         int scale = valStr.length() - 1;
821         
822         /* 
823          * Since we may pad trailing zeros for serialization, when doing 
824          * de-serialization, we need to delete the trailing zeros. 
825          */
826         return new BigDecimal(digitsVal, scale).stripTrailingZeros();
827     }
