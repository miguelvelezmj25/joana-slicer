35 /**
36  * <p>
37  * JEDiagnostics is a debugging mbean for a non replicated JE Environment.
38  * This is intended as a locus of field support functionality. While it may be
39  * used by the application developer, the primary use case is for a support
40  * situation. Currently much of this functionality is also available through
41  * the standard java.util.logging MBean.
42  * <p>
43  * It is a concrete MBean created by registering a JE Environment as an MBean
44  * through setting the JEDiagnostics system property. It only works on an
45  * active JE Environment, and one Environment can only have one JEDiagnostics
46  * instance. There are two attributes and one operation:
47  * <p>
48  * <u>Attributes:</u>
49  * <ul>
50  * <li> consoleHandlerLevel: which sets the console handler level.
51  * <li> fileHandlerLevel: which sets the file handler level.
52  * </ul>
53  * <u>Operations:</u>
54  * <ul>
55  * <li> resetLoggingLevel: which sets the level for the current loggers in
56  *      the LogManager.
57  * </ul>
58  * <p>
59  * We can use these attributes and operations to dynamically change the
60  * logging level for debugging purposes.
61  */

151         if (attribute == null) {

155         /* Sanity check parameters. */

171             Level level = Level.parse((String) value);

180             }
181         } catch (NullPointerException e) {
182             throw new InvalidAttributeValueException
183                 ("Setting value for attribute " + name +

185         } catch (SecurityException e) {

264         try {
265             EnvironmentImpl envImpl = DbInternal.getNonNullEnvImpl(env);
266 
267             if (actionName.equals(OP_RESET_LOGGING)) {
268                 if (params == null || params.length != 2) {
269                     return new IllegalArgumentException
270                         ("Parameter is not valid");
271                 }
272                 envImpl.resetLoggingLevel(((String) params[0]).trim(), 
273                                           Level.parse((String) params[1]));
274                 return null;
275             } 
276 
277             return new IllegalArgumentException
278                 ("ActionName: " + actionName + " is not valid.");
279         } catch (DatabaseException e) {
280 
281             /*
282              * Add the message for easiest deciphering of the problem. Since 
283              * the original exception cannot be transferred, send the exception
284              * stack.
285              */
286              throw new MBeanException(new RuntimeException
287                                       (e.getMessage() +
288                                        LoggerUtils.getStackTrace(e)));
289         } catch (NullPointerException e) {
290             throw new MBeanException(e, e.getMessage());
291         }
