28 /**
29  * A base class for JE stats that map String keys to component statistics, and
30  * that return results as formatted strings.
31  *
32  * @param <T> the value type of the individual statistics
33  * @param <C> the class of the individual statistics
34  */
35 public abstract class MapStat<T, C extends MapStatComponent<T, C>>
36         extends Stat<String> {
37     private static final long serialVersionUID = 1L;
38 
39     /**
40      * Maps keys to individual statistics.  Synchronize on the MapStat instance
41      * when accessing this field.
42      */
43     protected final Map<String, C> statMap =
44 
45         /* Use a sorted map so that the output is sorted */
46         new TreeMap<>();
47 
48     /**
49      * Creates an instance of this class.
50      *
51      * @param group the owning group
52      * @param definition the associated definition
53      */
54     protected MapStat(StatGroup group, StatDefinition definition) {
55         super(group, definition);
56     }
57 
58     /**
59      * Creates an instance of this class as a copy of another instance.  This
60      * instance should be a new instance to avoid lock ordering concerns.
61      *
62      * @param other the instance to copy
63      */
64     protected MapStat(MapStat<T, C> other) {
65         super(other.definition);
66         synchronized (this) {
67             synchronized (other) {
68                 for (final Entry<String, C> entry : other.statMap.entrySet()) {
69                     statMap.put(entry.getKey(), entry.getValue().copy());
70                 }
71             }
72         }
73     }

75     /**
76      * Removes the individual statistic associated with the specified key.
77      *
78      * @param key the key
79      */
80     public synchronized void removeStat(String key) {
81         assert key != null;
82         statMap.remove(key);
83     }

85     /**
86      * Returns a map from keys to values of individual statistics, ignoring
87      * individual statistics that are not set.
88      *
89      * @return map from keys to values of individual statistics
90      */
91     public synchronized SortedMap<String, T> getMap() {
92         SortedMap<String, T> ret = null;
93         for (final Entry<String, C> entry : statMap.entrySet()) {
94             final C stat = entry.getValue();
95             if (stat.isNotSet()) {
96                 continue;
97             }
98             if (ret == null) {
99                 ret = new TreeMap<>();
100             }
101             ret.put(entry.getKey(), stat.get());
102         }
103         if (ret == null) {
104             return emptySortedMap();
105         }
106         return ret;
107     }

136     private synchronized String getFormattedValue(boolean useCommas) {
137         final StringBuilder sb = new StringBuilder();
138         boolean first = true;
139         for (final Entry<String, C> entry : statMap.entrySet()) {
140             final C value = entry.getValue();
141             if (value.isNotSet()) {
142                 continue;
143             }
144             if (!first) {
145                 sb.append(';');
146             } else {
147                 first = false;
148             }
149             sb.append(entry.getKey()).append('=');
150             final String formattedValue =
151                 value.getFormattedValue(useCommas);
152             assert useCommas || (formattedValue.indexOf(',') == -1)
153                 : "Formatted value doesn't obey useCommas: " + formattedValue;
154             sb.append(formattedValue);
155         }
156         return sb.toString();
157     }
