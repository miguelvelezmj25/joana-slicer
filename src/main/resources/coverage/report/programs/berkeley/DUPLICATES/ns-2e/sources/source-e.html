258      * Currently true is always returned when this method returns normally,

260      *

263      * persistent corruption is detected,
264      * {@link EnvironmentFailureException#isCorrupted()} will return true.
265      *
266      * @deprecated as of 7.5, use {@link Environment#verify} or

268      * specifying all {@link VerifyConfig} properties.
269      */
270     public boolean verify(PrintStream out)
271         throws DatabaseException {
272 
273         final EnvironmentImpl envImpl = DbInternal.getNonNullEnvImpl(env);
274         final BtreeVerifier verifier = new BtreeVerifier(envImpl);
275         verifyConfig.setShowProgressStream(out);
276         verifier.setBtreeVerifyConfig(verifyConfig);
277 
278         if (dbName == null) {
279             verifier.verifyAll();
280         } else {
281             /* Get DB ID from name. */
282             BasicLocker locker =
283                 BasicLocker.createBasicLocker(envImpl, false /*noWait*/);
284             final DbTree dbTree = envImpl.getDbTree();
285             DatabaseImpl dbImpl = null;
286             DatabaseId dbId;
287 
288             try {
289                 dbImpl = dbTree.getDb(locker, dbName, null, false);
290                 if (dbImpl == null) {
291                     return true;
292                 }
293                 dbId = dbImpl.getId();
294             } finally {
295                 dbTree.releaseDb(dbImpl);
296                 locker.operationEnd();
297             }
298 
299             verifier.verifyDatabase(dbName, dbId);
300         }
301 
302         return true;
303     }
