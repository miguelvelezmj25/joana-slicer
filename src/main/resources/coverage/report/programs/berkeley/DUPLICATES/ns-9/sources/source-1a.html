66      * in the utilization profile. Assumes that the database and
67      * environment is quiescent, and that there is no current cleaner
68      * activity.
69      */
70     public static void checkLsns(DatabaseImpl dbImpl,
71                                  PrintStream out)
72         throws DatabaseException {
73 
74         /* Get all the LSNs in the database. */
75         GatherLSNs gatherLsns = new GatherLSNs();
76         long rootLsn = dbImpl.getTree().getRootLsn();
77         List<DatabaseException> savedExceptions =
78             new ArrayList<DatabaseException>();
79 
80         SortedLSNTreeWalker walker =
81             new SortedLSNTreeWalker(new DatabaseImpl[] { dbImpl },
82                                     false /*setDbState*/,
83                                     new long[] { rootLsn },
84                                     gatherLsns, savedExceptions, null);
85         walker.walk();
86 
87         /* Print out any exceptions seen during the walk. */
88         if (savedExceptions.size() > 0) {
89             out.println(savedExceptions.size() +
90                         " problems seen during tree walk for checkLsns");
91             Iterator<DatabaseException> iter = savedExceptions.iterator();
92             while (iter.hasNext()) {
93                 out.println("  " + iter.next());
94             }
95         }
96 
97         Set<Long> lsnsInTree = gatherLsns.getLsns();
98         if (rootLsn != DbLsn.NULL_LSN) {
99             lsnsInTree.add(rootLsn);
100         }
101 
102         /* Get all the files used by this database. */
103         Iterator<Long> iter = lsnsInTree.iterator();
104         Set<Long> fileNums = new HashSet<Long>();
105 
106         while (iter.hasNext()) {
107             long lsn = iter.next();
108             fileNums.add(DbLsn.getFileNumber(lsn));
109         }
110 
111         /* Gather up the obsolete LSNs in these file summary LNs. */
112         iter = fileNums.iterator();
113         Set<Long> obsoleteLsns = new HashSet<Long>();
114         EnvironmentImpl envImpl = dbImpl.getEnv();
115         UtilizationProfile profile = envImpl.getUtilizationProfile();
116 
117         while (iter.hasNext()) {
118             Long fileNum = iter.next();
119 
120             PackedOffsets obsoleteOffsets =
121                 profile.getObsoleteDetailPacked(fileNum, false /*logUpdate*/);
122             PackedOffsets.Iterator obsoleteIter = obsoleteOffsets.iterator();
123             while (obsoleteIter.hasNext()) {
124                 long offset = obsoleteIter.next();
125                 Long oneLsn = Long.valueOf(DbLsn.makeLsn(fileNum.longValue(),
126                                                          offset));
127                 obsoleteLsns.add(oneLsn);
128                 if (DEBUG) {
129                     out.println("Adding 0x" +
130                                 Long.toHexString(oneLsn.longValue()));
131                 }
132             }
133         }
134 
135         /* Check than none the LSNs in the tree is in the UP. */
136         boolean error = false;
137         iter = lsnsInTree.iterator();
138         while (iter.hasNext()) {
139             Long lsn = iter.next();
140             if (obsoleteLsns.contains(lsn)) {
141                 out.println("Obsolete LSN set contains valid LSN " +
142                             DbLsn.getNoFormatString(lsn.longValue()));
143                 error = true;
144             }
145         }
146 
147         /*
148          * Check that none of the LSNs in the file summary LN is in the
149          * tree.
150          */
151         iter = obsoleteLsns.iterator();
152         while (iter.hasNext()) {
153             Long lsn = iter.next();
154             if (lsnsInTree.contains(lsn)) {
155                 out.println("Tree contains obsolete LSN " +
156                             DbLsn.getNoFormatString(lsn.longValue()));
157                 error = true;
158             }
159         }
160 
161         if (error) {
162             throw new EnvironmentFailureException
163                 (envImpl, EnvironmentFailureReason.LOG_INTEGRITY,
164                  "Lsn mismatch");
165         }
166 
167         if (savedExceptions.size() > 0) {
168             throw new EnvironmentFailureException
169                 (envImpl, EnvironmentFailureReason.LOG_INTEGRITY,
170                  "Sorted LSN Walk problem");
171         }
172     }

373      * Adds up the per-file totals from the utilization information for each
374      * database to make a total per-file count.
375      *
376      * @return aggregation of per-file information.
377      */
378     private static Map<Long,DbFileSummary> calcDbDerivedUtilization
379                                                (EnvironmentImpl envImpl)
380         throws DatabaseException {
381 
382         final Map<Long,DbFileSummary> grandTotalsMap =
383             new HashMap<Long,DbFileSummary>();
384 
385         DbTree dbTree = envImpl.getDbTree();
386 
387         /* Add in the special id and name database. */
388         addDbDerivedTotals(dbTree.getDb(DbTree.ID_DB_ID), grandTotalsMap);
389         addDbDerivedTotals(dbTree.getDb(DbTree.NAME_DB_ID), grandTotalsMap);
390 
391         /* Walk through all the regular databases. */
392         CursorImpl.traverseDbWithCursor(dbTree.getDb(DbTree.ID_DB_ID),
393                                         LockType.NONE,
394                                         true /*allowEviction*/,
395                                         new CursorImpl.WithCursor() {
396             @Override
397             public boolean withCursor(CursorImpl cursor,
398                                       DatabaseEntry key,
399                                       DatabaseEntry data)
400                 throws DatabaseException {
401 
402                 MapLN mapLN = (MapLN)
403                     cursor.lockAndGetCurrentLN(LockType.NONE);
404 
405                 addDbDerivedTotals(mapLN.getDatabase(), grandTotalsMap);
406                 return true;
407             }
408         });
409         return grandTotalsMap;
410     }

414      * aggregate all the per-file/per db information into a single per-file
415      * grandTotals map.
416      */
417     private static void addDbDerivedTotals
418         (DatabaseImpl dbImpl,
419          Map<Long,DbFileSummary> grandTotalsMap) {
420 
421         Iterator<Map.Entry<Long,DbFileSummary>> entries =
422             dbImpl.getDbFileSummaries().entrySet().iterator();
423 
424         while (entries.hasNext()) {
425             Map.Entry<Long,DbFileSummary> entry = entries.next();
426             Long fileNum = entry.getKey();
427             DbFileSummary dbTotals = entry.getValue();
428             DbFileSummary grandTotals = grandTotalsMap.get(fileNum);
429             if (grandTotals == null) {
430                 grandTotals = new DbFileSummary();
431                 grandTotalsMap.put(fileNum, grandTotals);
432             }
433             grandTotals.add(dbTotals);
434         }
