44  * FeederTxns manages transactions that need acknowledgments.
45  *
46  * <p>The lastCommitVLSN, lastCommitTimestamp, and vlsnRate statistics provide
47  * general information about committed transactions on the master, but are also
48  * intended to be used programmatically along with other statistics for the
49  * feeder to provide information about how up-to-date the replicas are.  See
50  * the Feeder class for more details.
51  */

109     public void setupForAcks(MasterTxn txn) {
110         if (txn.getRequiredAckCount() == 0) {
111             /* No acks called for, no setup needed. */
112             return;
113         }
114         TxnInfo txnInfo = new TxnInfo(txn);

136     /**
137      * Notes that an acknowledgment was received from a replica.
138      *
139      * @param replica the replica node
140      * @param txnId the locally committed transaction that was acknowledged.
141      * @param isArbiterFeeder true if feeder is an Arbiter false otherwise.
142      *
143      * @return the TxnInfo associated with the txnId, if txnId needs an ack,
144      * null otherwise
145      */
146     public TxnInfo noteReplicaAck(final RepNodeImpl replica,
147                                   final long txnId) {
148         final DurabilityQuorum durabilityQuorum =
149             repImpl.getRepNode().getDurabilityQuorum();
150         if (!durabilityQuorum.replicaAcksQualify(replica)) {
151             return null;
152         }
153         final TxnInfo txnInfo = txnMap.get(txnId);
154         if (txnInfo == null) {
155             return null;
156         }
157         txnInfo.countDown();
158         return txnInfo;
159     }

161     /**
162      * Waits for the required number of replica acks to come through.
163      *
164      * @param txn identifies the transaction to wait for.
165      *
166      * @param timeoutMs the amount of time to wait for the acknowledgments
167      * before giving up.
168      *
169      * @throws InsufficientAcksException if the ack requirements were not met
170      */
171     public void awaitReplicaAcks(MasterTxn txn, int timeoutMs)
172         throws InterruptedException {
173 
174         /* Record master commit information even if no acks are needed */
175         final long vlsn = txn.getCommitVLSN().getSequence();
176         final long ackAwaitStartMs = System.currentTimeMillis();
177         lastCommitVLSN.set(vlsn);
178         lastCommitTimestamp.set(ackAwaitStartMs);
179         vlsnRate.add(vlsn, ackAwaitStartMs);
180 
181         TxnInfo txnInfo = txnMap.get(txn.getId());
182         if (txnInfo == null) {
183             return;
184         }
185         txnInfo.await(timeoutMs, ackAwaitStartMs);
186         txnMap.remove(txn.getId());
187         final RepNode repNode = repImpl.getRepNode();
188         if (repNode != null) {
189             repNode.getDurabilityQuorum().ensureSufficientAcks(
190                 txnInfo, timeoutMs);
191         }
192     }

194     /**
195      * Used to track the latch and the transaction information associated with
196      * a transaction needing an acknowledgment.
197      */
198     public class TxnInfo {
199         /* The latch used to track transaction acknowledgments. */
200         final private CountDownLatch latch;
201         final MasterTxn txn;
202 
203         private TxnInfo(MasterTxn txn) {
204             assert(txn != null);
205             final int numRequiredAcks = txn.getRequiredAckCount();
206             this.latch = (numRequiredAcks == 0) ?
207                 null :
208                 new CountDownLatch(numRequiredAcks);
209             this.txn = txn;
210         }
211 
212         /**
213          * Returns the VLSN associated with the committed txn, or null if the
214          * txn has not yet been committed.
215          */
216         public VLSN getCommitVLSN() {
217             return txn.getCommitVLSN();
218         }
219 
220         private final boolean await(int timeoutMs, long ackAwaitStartMs)
221             throws InterruptedException {
222 
223             boolean isZero = (latch == null) ||
224                 latch.await(timeoutMs, TimeUnit.MILLISECONDS);
225             if (isZero) {
226                 txnsAcked.increment();
227                 final long now = System.currentTimeMillis();
228                 ackWaitMs.add(now - ackAwaitStartMs);
229                 totalTxnMs.add(now - txn.getStartMs());
230             } else {
231                 txnsNotAcked.increment();
232             }
233             return isZero;
234         }
235 
236         public final void countDown() {
237             if (latch == null) {
238                 return;
239             }
240 
241             latch.countDown();
242         }
243 
244         public final int getPendingAcks() {
245             if (latch == null) {
246                 return 0;
247             }
248 
249             return (int) latch.getCount();
250         }
251 
252         public final MasterTxn getTxn() {
253             return txn;
254         }
255     }
