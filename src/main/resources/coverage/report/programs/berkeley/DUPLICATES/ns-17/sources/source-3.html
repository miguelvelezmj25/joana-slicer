74 /**
75  * The implementation of the Arbiter. The Arbiter is a participant in
76  * elections and may acknowledge transaction commits.
77  * <p>
78  * The Arbiter persists the Arbiter's replication group node identifier
79  * and the highest commit VLSN that has been acknowledged.
80  * Currently the Feeder sends commit acknowledgment requests
81  * to the Arbiter if the replication factor is two and the other
82  * Rep node is not available (RepImpl.preLogCommitHook).
83  * <p>
84  * The VLSN is used in response to an election promise request.
85  * The priority of the response is lower than
86  * a RepNode. This allows the RepNode to be selected if the VLSN is
87  * equal. An Arbiter may not "win" an election. If the Arbiter's VLSN
88  * is the highest in the election, the election result is ignored. A
89  * NULL nodeid in a promise response is used to identify a promise
90  * response is from an Arbiter.
91  * <p>
92  * Two pieces of information are persisted by the
93  * Arbiter.
94  * The replication group node identifier is persisted because
95  * this information is located in the group database and the Arbiter
96  * does not have a copy of the group database.
97  * The other is the high VLSN of a Arbiter ACKed commit.
98  * <p>
99  * In the future, the algorithm could be changed to request commit
100  * acknowledgments when the replication factor is greater than
101  * two. This would allow for better write availability when
102  * the replication factor is an even number.
103  */

283     @Override
284     public void run() {
285         /* Set to indicate an error-initiated shutdown. */
286         Error repNodeError = null;
287         try {
288             while (!isShutdownOrInvalid()) {
289                 queryGroupForMembership();
290                 masterStatus.sync();
291                 arbiterAcker = new ArbiterAcker(this, repImpl);
292                 arbiterAcker.runArbiterAckLoop();
293             }
294         } catch (InterruptedException e) {
295             LoggerUtils.fine(logger, repImpl,
296                              "Arbiter main thread interrupted - " +
297                              " forced shutdown.");
298         } catch (GroupShutdownException e) {
299             saveShutdownException(e);
300             LoggerUtils.fine(logger, repImpl,
301                              "Arbiter main thread sees group shutdown - " + e);
302         } catch (InsufficientLogException e) {
303             saveShutdownException(e);
304         } catch (RuntimeException e) {
305             LoggerUtils.fine(logger, repImpl,
306                              "Arbiter main thread sees runtime ex - " + e);
307             saveShutdownException(e);
308             throw e;
309         } catch (Error e) {
310             LoggerUtils.fine(logger, repImpl, e +
311                              " incurred during arbiter loop");
312             repNodeError = e;
313             repImpl.invalidate(e);
314         } finally {
315                 LoggerUtils.info(logger, repImpl,
316                                  "Arbiter main thread shutting down.");
317 
318                 if (repNodeError != null) {
319                     LoggerUtils.info(logger, repImpl,
320                                      "Node state at shutdown:\n"+
321                                      repImpl.dumpState());
322                     throw repNodeError;
323                 }
324                 Throwable exception = getSavedShutdownException();
325 
326                 if (exception == null) {
327                     LoggerUtils.fine(logger, repImpl,
328                                      "Node state at shutdown:\n"+
329                                      repImpl.dumpState());
330                 } else {
331                     LoggerUtils.info(logger, repImpl,
332                                      "Arbiter shutdown exception:\n" +
333                                      exception.getMessage() +
334                                      repImpl.dumpState());
335                 }
336 
337                 try {
338                     shutdown();
339                 } catch (DatabaseException e) {
340                     RepUtils.chainExceptionCause(e, exception);
341                     LoggerUtils.severe(logger, repImpl,
342                                        "Unexpected exception during shutdown" +
343                                        e);
344                     throw e;
345                 }
346             setState(ReplicatedEnvironment.State.DETACHED);
347             cleanup();
348         }
349     }

535     /**
536      * Communicates with existing nodes in the group in order figure out
537      * who is the master.
538      * In the case where the local node does not appear to be in
539      * the (local copy of the) GroupDB, typically because the node is starting
540      * up with an empty env directory.  It could be that this is a new node
541      * (never before been part of the group).  Or it could be a pre-existing
542      * group member that has lost its env dir contents and wants to be restored
543      * via a Network Restore operation.
544      * <p>
545      * We query the designated helpers for the Master
546      * information. The helpers are
547      * the ones that were identified via the node's configuration.
548      * <p>
549      * Returns normally when the master is found.
550      *
551      * @throws InterruptedException if the current thread is interrupted,
552      *         typically due to a shutdown
553      */
554     private void queryGroupForMembership()
555         throws InterruptedException {
556 
557         checkLoopbackAddresses();
558 
559         if (helperSockets.isEmpty()) {
560             throw EnvironmentFailureException.unexpectedState
561                 ("Need a helper to add a new node into the group");
562         }
563 
564         NameIdPair groupMasterNameId;
565         while (true) {
566             elections.getLearner().queryForMaster(helperSockets);
567             groupMasterNameId = masterStatus.getGroupMasterNameId();
568             if (!groupMasterNameId.hasNullId()) {
569                 /* A new, or pre-query, group master. */
570                 if (nameIdPair.hasNullId() &&
571                     groupMasterNameId.getName().equals(nameIdPair.getName())) {
572 
573                     /*
574                      * Residual obsolete information in replicas, ignore it.
575                      * Can't be master if we don't know our own id, but some
576                      * other node does! This state means that the node was a
577                      * master in the recent past, but has had its environment
578                      * deleted since that time.
579                      */
580                     try {
581                         Thread.sleep(MASTER_QUERY_INTERVAL);
582                     } catch (InterruptedException e) {
583                         throw EnvironmentFailureException.
584                             unexpectedException(e);
585                     }
586                     continue;
587                 }
588                 if (checkGroupMasterIsAlive(groupMasterNameId)) {
589                     /* Use the current group master if it's alive. */
590                     break;
591                 }
592             }
593             if (isShutdownOrInvalid()) {
594                 throw new InterruptedException("Arbiter node shutting down.");
595             }
596             Thread.sleep(MASTER_QUERY_INTERVAL);
597         }
598         LoggerUtils.fine(logger, repImpl, "New node " + nameIdPair.getName() +
599                          " located master: " + groupMasterNameId);
600     }

623 
624         final InetAddress myAddress = getSocket().getAddress();
625         final boolean isLoopback = myAddress.isLoopbackAddress();
626 
627         for (InetSocketAddress socketAddress : helperSockets) {
628             final InetAddress nodeAddress = socketAddress.getAddress();
629 
630             if (nodeAddress.isLoopbackAddress() == isLoopback) {
631                 continue;
632             }

637                 " of the address:" +
638                 socketAddress +
639                 (!isLoopback ? " which is a loopback address." :

641                 " Such mixing of addresses within a group is not allowed, " +

650      * alive and that we are not dealing with obsolete cached information.
651      *
652      * @return true if the master node could be contacted and was truly alive
653      *
654      * TODO: handle protocol version mismatch here and in DbPing, also
655      * consolidate code so that a single copy is shared.
656      */
657     private boolean checkGroupMasterIsAlive(NameIdPair groupMasterNameId) {
658 
659         DataChannel channel = null;
660 
661         try {
662             final InetSocketAddress masterSocket =
663                 masterStatus.getGroupMaster();
664 
665             final BinaryNodeStateProtocol protocol =
666                 new BinaryNodeStateProtocol(NameIdPair.NOCHECK, null);
667 
668             /* Build the connection. Set the parameter connectTimeout.*/
669             channel = repImpl.getChannelFactory().
670                 connect(masterSocket,
671                         new ConnectOptions().
672                         setTcpNoDelay(true).
673                         setOpenTimeout(5000).
674                         setReadTimeout(5000));
675             ServiceDispatcher.doServiceHandshake
676                 (channel, BinaryNodeStateService.SERVICE_NAME);
677             /* Send a NodeState request to the node. */
678             protocol.write(
679                 protocol.new BinaryNodeStateRequest(
680                     groupMasterNameId.getName(),
681                     repImpl.getConfigManager().get(RepParams.GROUP_NAME)),
682                  channel);
683             /* Get the response and return the NodeState. */
684             BinaryNodeStateResponse response =
685                 protocol.read(channel, BinaryNodeStateResponse.class);
686 
687             ReplicatedEnvironment.State state = response.getNodeState();
688            return (state != null) && state.isMaster();
689         } catch (Exception e) {
690             LoggerUtils.info(logger, repImpl,
691                              "Queried master:" + groupMasterNameId +
692                              " unavailable. Reason:" + e);
693             return false;
694         } finally {
695             if (channel != null) {
696                 try {
697                     channel.close();
698                 } catch (IOException ioe) {
699                     /* Ignore it */
700                 }
701             }
702         }
703     }
