30 /**
31  * Format for a composite key class.
32  *
33  * This class is similar to ComplexFormat in that a composite key class and
34  * other complex classes have fields, and the Accessor interface is used to
35  * access those fields.  Composite key classes are different in the following
36  * ways:
37  *
38  * - The superclass must be Object.  No inheritance is allowed.
39  *
40  * - All instance fields must be annotated with @KeyField, which determines
41  *   their order in the data bytes.
42  *
43  * - Although fields may be reference types (primitive wrappers or other simple
44  *   reference types), they are stored as if they were primitives.  No object
45  *   format ID is stored, and the class of the object must be the declared
46  *   classs of the field; i.e., no polymorphism is allowed for key fields.
47  *   In other words, a composite key is stored as an ordinary tuple as defined
48  *   in the com.sleepycat.bind.tuple package.  This keeps the key small and
49  *   gives it a well defined sort order.
50  *
51  * - If the key class implements Comparable, it is called by the Database
52  *   btree comparator.  It must therefore be available during JE recovery,
53  *   before the store and catalog have been opened.  To support this, this
54  *   format can be constructed during recovery.  A SimpleCatalog singleton
55  *   instance is used to provide a catalog of simple types that is used by
56  *   the composite key format.
57  *
58  * - When interacting with the Accessor, the composite key format treats the
59  *   Accessor's non-key fields as its key fields.  The Accessor's key fields
60  *   are secondary keys, while the composite format's key fields are the
61  *   component parts of a single key.
62  *
63  * @author Mark Hayes
64  */

104     private CompositeKeyFormat(Catalog catalog,
105                                Class cls,
106                                ClassMetadata metadata,
107                                String[] fieldNames) {
108         super(catalog, cls);
109         this.metadata = metadata;
110 
111         /* Check that the superclass is Object. */
112         Class superCls = cls.getSuperclass();
113         if (superCls != Object.class) {
114             throw new IllegalArgumentException
115                 ("Composite key class must be derived from Object: " +
116                  cls.getName());
117         }
118 
119         /* Populate fields list in fieldNames order. */
120         List<FieldInfo> instanceFields =
121             FieldInfo.getInstanceFields(cls, metadata);
122         fields = new ArrayList<FieldInfo>(instanceFields.size());
123         for (String fieldName : fieldNames) {
124             FieldInfo field = null;
125             for (FieldInfo tryField : instanceFields) {
126                 if (fieldName.equals(tryField.getName())) {
127                     field = tryField;
128                     break;
129                 }
130             }
131             if (field == null) {
132                 throw new IllegalArgumentException
133                     ("Composite key field is not an instance field: " +
134                      getClassName() + '.' + fieldName);
135             }
136             fields.add(field);
137             instanceFields.remove(field);
138             Class fieldCls = field.getFieldClass(getCatalog());
139             if (!SimpleCatalog.isSimpleType(fieldCls) &&
140                 !fieldCls.isEnum()) {
141                 throw new IllegalArgumentException
142                     ("Composite key field is not a simple type or enum: " +
143                      getClassName() + '.' + fieldName);
144             }
145         }
146         if (instanceFields.size() > 0) {
147             throw new IllegalArgumentException
148                 ("All composite key instance fields must be key fields: " +
149                  getClassName() + '.' + instanceFields.get(0).getName());
150         }
151     }

180 
181         /*
182          * Lazily create the raw type information.  Synchronization is not
183          * required since this object is immutable.  If by chance we create two
184          * maps when two threads execute this block, no harm is done.  But be

190             for (RawField field : fields) {

253     @Override
254     Object convertRawObject(Catalog catalog,
255                             boolean rawAccess,
256                             RawObject rawObject,
257                             IdentityHashMap converted)
258         throws RefreshException {
259 
260         /*
261          * Synchronization is not required since rawInputFields is immutable.
262          * If by chance we create duplicate values when two threads execute
263          * this block, no harm is done.  But be sure to assign the field only
264          * after the values are fully populated.
265          */
266         FieldInfo[] myFields = rawInputFields;
267         if (myFields == null) {
268             myFields = new FieldInfo[fields.size()];
269             fields.toArray(myFields);
270             rawInputFields = myFields;
271         }
272         if (rawObject.getSuper() != null) {
273             throw new IllegalArgumentException
274                 ("RawObject has too many superclasses: " +
275                  rawObject.getType().getClassName());
276         }
277         RawObject[] objects = new RawObject[myFields.length];
278         Arrays.fill(objects, rawObject);
279         EntityInput in = new RawComplexInput
280             (catalog, rawAccess, converted, myFields, objects);
281         Object o = newInstance(in, rawAccess);
282         converted.put(rawObject, o);
283         return readObject(o, in, rawAccess);
284     }

314     @Override
315     boolean evolve(Format newFormatParam, Evolver evolver) {
316 
317         /* Disallow evolution to a non-composite format. */
318         if (!(newFormatParam instanceof CompositeKeyFormat)) {
319             evolver.addEvolveError
320                 (this, newFormatParam, null,
321                  "A composite key class may not be changed to a different " +
322                  "type");
323             return false;
324         }
325         CompositeKeyFormat newFormat = (CompositeKeyFormat) newFormatParam;
326 
327         /* Check for added or removed key fields. */
328         if (fields.size() != newFormat.fields.size()) {
329             evolver.addEvolveError
330                 (this, newFormat,
331                  "Composite key class fields were added or removed ",
332                  "Old fields: " + fields +
333                  " new fields: " + newFormat.fields);
334             return false;
335         }
336 
337         /* Check for modified key fields. */
338         boolean newVersion = false;
339         for (int i = 0; i < fields.size(); i += 1) {
340             int result = evolver.evolveRequiredKeyField
341                 (this, newFormat, fields.get(i),
342                  newFormat.fields.get(i));
343             if (result == Evolver.EVOLVE_FAILURE) {
344                 return false;
345             }
346             if (result == Evolver.EVOLVE_NEEDED) {
347                 newVersion = true;
348             }
349         }
350 
351         /*
352          * We never need to use a custom reader because the physical key field
353          * formats never change.  But we do create a new evolved format when
354          * a type changes (primitive <-> primitive wrapper) so that the new
355          * type information is correct.
356          */
357         if (newVersion) {
358             evolver.useEvolvedFormat(this, newFormat, newFormat);
359         } else {
360             evolver.useOldFormat(this, newFormat);
361         }
362         return true;
363     }
