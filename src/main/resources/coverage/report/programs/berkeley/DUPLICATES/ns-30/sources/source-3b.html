84     private void tallyLevel(int levelArg, long[] nodesSeenByLevel) {
85         int level = levelArg;
86         if (level >= IN.MAIN_LEVEL) {

112     private void tallyEntries(BIN bin, long[] binEntriesHistogram) {
113         int nEntries = bin.getNEntries();
114         int nonDeletedEntries = 0;
115         for (int i = 0; i < nEntries; i++) {
116             /* KD and PD determine deletedness. */
117             if (!bin.isEntryPendingDeleted(i) &&
118                 !bin.isEntryKnownDeleted(i)) {
119                 nonDeletedEntries++;
120             }
121         }
122 
123         int bucket = (nonDeletedEntries * 100) / (bin.getMaxEntries() + 1);
124         bucket /= 10;
125         binEntriesHistogram[bucket]++;
126     }
