93         public IdentifierConfigParam(String configName,
94                                      String defaultValue,
95                                      boolean mutable,
96                                      boolean forReplication) {
97             super(configName, defaultValue, mutable, forReplication);
98         }
99 
100         @Override
101         public void validateValue(String value) {
102             if (Boolean.getBoolean(SKIP_NODENAME_VALIDATION)) {
103                 return;
104             }
105             if ((value == null) || (value.length() == 0)) {
106                 throw new IllegalArgumentException
107                     (DEBUG_NAME + ": a value is required");
108             }
109             for (char c : value.toCharArray()) {
110                 if (!isValid(c)) {
111                     throw new IllegalArgumentException
112                         (DEBUG_NAME + ": " + name + ", must consist of " +
113                          "letters, digits, hyphen, underscore, period.");
114                 }
115             }
116         }
117 
118         private boolean isValid(char c) {
119             if (Character.isLetterOrDigit(c) ||
120                 c == '-' ||
121                 c == '_' ||
122                 c == '.') {
123                 return true;
124             }
125             return false;
126         }

1055     /**
1056      * Verifies that the port is a reasonable number. The port must be outside
1057      * the range of "Well Known Ports" (zero through 1024).
1058      *
1059      * @param portString the string representing the port.
1060      */
1061     private static void validatePort(String portString)
1062         throws IllegalArgumentException {
1063 
1064         try {
1065             int port = Integer.parseInt(portString);
1066 
1067             if ((port <= 0) || (port > 0xffff)) {
1068                 throw new IllegalArgumentException
1069                     ("Invalid port number: " + portString);
1070             }
1071             if (port <= 1023) {
1072                 throw new IllegalArgumentException
1073                     ("Port number " + port +
1074                      " is invalid because the port must be "+
1075                      "outside the range of \"well known\" ports");
1076             }
1077         } catch (NumberFormatException e) {
1078             throw new IllegalArgumentException
1079                 ("Invalid port number: " + portString);
1080         }
1081     }

1083     /**
1084      * Validates that the hostPort is a string of the form:
1085      *
1086      * hostName[:port]
1087      *
1088      * @param hostAndPort
1089      * @param skipHostnameResolution if true, don't bother checking that the
1090      * hostname resolves
1091      * @throws IllegalArgumentException
1092      */
1093     private static void validateHostAndPort(String hostAndPort,
1094                                             boolean skipHostnameResolution)
1095         throws IllegalArgumentException {
1096 
1097         int colonToken = hostAndPort.indexOf(":");
1098         String hostName = (colonToken >= 0) ?
1099             hostAndPort.substring(0, colonToken) :
1100             hostAndPort;
1101         if ("".equals(hostName)) {
1102             throw new IllegalArgumentException("missing hostname");
1103         }
1104 
1105         if (!skipHostnameResolution) {
1106             try {
1107                 InetAddress.getByName(hostName);
1108             } catch (UnknownHostException e) {
1109                 throw new IllegalArgumentException
1110                     ("Invalid hostname: " + e.getMessage());
1111             }
1112         }
1113 
1114         if (colonToken >= 0) {
1115             validatePort(hostAndPort.substring(colonToken + 1));
1116         }
1117     }
