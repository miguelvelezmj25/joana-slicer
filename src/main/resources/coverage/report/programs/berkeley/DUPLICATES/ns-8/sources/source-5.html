21 /**
22  * Defines an opaque token that can be used to identify a specific transaction
23  * commit in a replicated environment. It's unique relative to its environment.
24  * <p>
25  * Since CommitTokens identify a point in the serialized transaction schedule
26  * created on the master, it's meaningful to compare commit tokens,
27  * as described in the {@link #compareTo(CommitToken)} method below.
28  * CommitTokens are obtained from {@link Transaction#getCommitToken()}
29  *
30  * @see com.sleepycat.je.rep.CommitPointConsistencyPolicy
31  */
32 public class CommitToken implements Serializable, Comparable<CommitToken> {
33

38     /**
39      * @hidden
40      * For internal use only.
41      * Creates a CommitToken suitable for use in a consistency policy.
42      *
43      * @param repenvUUID identifies the replicated environment associated with
44      * the <code>vlsn</code>
45      * @param vlsn the vlsn representing the state of the database.
46      */
47     public CommitToken(UUID repenvUUID, long vlsn) {

105     /**

107      * compare commit tokens across environments, since they represent

116     public int compareTo(CommitToken other) {
117         if (! repenvUUID.equals(other.repenvUUID)) {
118             throw new IllegalArgumentException

124         return (compare < 0) ? -1 : ((compare == 0) ? 0 : 1);
