26 /**
27  * NodeState encapsulates the current replicator state, and the ability to wait
28  * for state transition and fire state change notifications.
29  */
30 public class NodeState {
31     /* The rep impl whose state is being tracked. */
32     private final RepImpl repImpl;
33 
34     /* The application registered state change listener for this node. */
35     private StateChangeListener stateChangeListener = null;
36 
37     /* The state change event that resulted in the current state. */
38     private StateChangeEvent stateChangeEvent = null;
39     private final AtomicReference<ReplicatedEnvironment.State> currentState;
40     private final Logger logger;
41     private final NameIdPair nameIdPair;
42 
43     public NodeState(NameIdPair nameIdPair,
44                      RepImpl repImpl) {
45 
46         currentState = new AtomicReference<ReplicatedEnvironment.State>
47             (ReplicatedEnvironment.State.DETACHED);
48         this.nameIdPair = nameIdPair;
49         this.repImpl = repImpl;
50         logger = LoggerUtils.getLogger(getClass());
51     }
52 
53     synchronized public
54         void setChangeListener(StateChangeListener stateChangeListener){
55         this.stateChangeListener = stateChangeListener;
56     }
57 
58     synchronized public StateChangeListener getChangeListener() {
59         return stateChangeListener;
60     }
61 
62     /**
63      * Change to a new node state and release any threads waiting for a state
64      * transition.
65      */
66     synchronized public void changeAndNotify(ReplicatedEnvironment.State state,
67                                              NameIdPair masterNameId) {
68 
69         ReplicatedEnvironment.State newState = state;
70         ReplicatedEnvironment.State oldState = currentState.getAndSet(state);
71         stateChangeEvent = new StateChangeEvent(state, masterNameId);
72 
73         LoggerUtils.info(logger, repImpl,
74                          "node:" + masterNameId +
75                          " state change from " + oldState + " to " + newState);
76 
77         if (stateChangeListener != null) {
78             try {
79                 stateChangeListener.stateChange(stateChangeEvent);
80             } catch (Exception e) {
81                 LoggerUtils.severe(logger, repImpl,
82                                    "State Change listener exception" +
83                                    e.getMessage());
84                 throw new EnvironmentFailureException
85                     (repImpl, EnvironmentFailureReason.LISTENER_EXCEPTION, e);
86             }
87         }
88 
89         /* Make things obvious in thread dumps */
90         Thread.currentThread().setName(currentState + " " + nameIdPair);
91     }
92 
93     synchronized public ReplicatedEnvironment.State getRepEnvState() {

95     }
