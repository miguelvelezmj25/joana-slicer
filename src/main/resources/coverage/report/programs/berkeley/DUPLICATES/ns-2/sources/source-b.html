104      * Constructs a new {@link FieldWriter}.
105      * 
106      * @param cw
107      *            the class writer to which this field must be added.
108      * @param access
109      *            the field's access flags (see {@link Opcodes}).
110      * @param name
111      *            the field's name.
112      * @param desc
113      *            the field's descriptor (see {@link Type}).
114      * @param signature

117      *            the field's constant value. May be <tt>null</tt>.
118      */
119     FieldWriter(final ClassWriter cw, final int access, final String name,
120             final String desc, final String signature, final Object value) {
121         super(Opcodes.ASM5);
122         if (cw.firstField == null) {
123             cw.firstField = this;
124         } else {
125             cw.lastField.fv = this;
126         }
127         cw.lastField = this;
128         this.cw = cw;
129         this.access = access;
130         this.name = cw.newUTF8(name);
131         this.desc = cw.newUTF8(desc);
132         if (ClassReader.SIGNATURES && signature != null) {
133             this.signature = cw.newUTF8(signature);
134         }
135         if (value != null) {
136             this.value = cw.newConstItem(value).index;
137         }
138     }

144     @Override
145     public AnnotationVisitor visitAnnotation(final String desc,
146             final boolean visible) {
147         if (!ClassReader.ANNOTATIONS) {
148             return null;
149         }
150         ByteVector bv = new ByteVector();
151         // write type, and reserve space for values count
152         bv.putShort(cw.newUTF8(desc)).putShort(0);
153         AnnotationWriter aw = new AnnotationWriter(cw, true, bv, bv, 2);
154         if (visible) {
155             aw.next = anns;
156             anns = aw;
157         } else {
158             aw.next = ianns;
159             ianns = aw;
160         }
161         return aw;
162     }

164     @Override
165     public AnnotationVisitor visitTypeAnnotation(final int typeRef,
166             final TypePath typePath, final String desc, final boolean visible) {
167         if (!ClassReader.ANNOTATIONS) {
168             return null;
169         }
170         ByteVector bv = new ByteVector();
171         // write target_type and target_info
172         AnnotationWriter.putTarget(typeRef, typePath, bv);
173         // write type, and reserve space for values count
174         bv.putShort(cw.newUTF8(desc)).putShort(0);
175         AnnotationWriter aw = new AnnotationWriter(cw, true, bv, bv,
176                 bv.length - 2);
177         if (visible) {
178             aw.next = tanns;
179             tanns = aw;
180         } else {
181             aw.next = itanns;
182             itanns = aw;
183         }
184         return aw;
185     }

201     /**
202      * Returns the size of this field.
203      * 
204      * @return the size of this field.
205      */
206     int getSize() {
207         int size = 8;
208         if (value != 0) {
209             cw.newUTF8("ConstantValue");
210             size += 8;
211         }
212         if ((access & Opcodes.ACC_SYNTHETIC) != 0) {
213             if ((cw.version & 0xFFFF) < Opcodes.V1_5
214                     || (access & ClassWriter.ACC_SYNTHETIC_ATTRIBUTE) != 0) {
215                 cw.newUTF8("Synthetic");
216                 size += 6;
217             }
218         }
219         if ((access & Opcodes.ACC_DEPRECATED) != 0) {
220             cw.newUTF8("Deprecated");
221             size += 6;
222         }
223         if (ClassReader.SIGNATURES && signature != 0) {
224             cw.newUTF8("Signature");
225             size += 8;
226         }
227         if (ClassReader.ANNOTATIONS && anns != null) {
228             cw.newUTF8("RuntimeVisibleAnnotations");
229             size += 8 + anns.getSize();
230         }
231         if (ClassReader.ANNOTATIONS && ianns != null) {
232             cw.newUTF8("RuntimeInvisibleAnnotations");
233             size += 8 + ianns.getSize();
234         }
235         if (ClassReader.ANNOTATIONS && tanns != null) {
236             cw.newUTF8("RuntimeVisibleTypeAnnotations");
237             size += 8 + tanns.getSize();
238         }
239         if (ClassReader.ANNOTATIONS && itanns != null) {
240             cw.newUTF8("RuntimeInvisibleTypeAnnotations");
241             size += 8 + itanns.getSize();
242         }
243         if (attrs != null) {
244             size += attrs.getSize(cw, null, 0, -1, -1);
245         }
246         return size;
247     }

250      * Puts the content of this field into the given byte vector.
251      * 
252      * @param out
253      *            where the content of this field must be put.
254      */
255     void put(final ByteVector out) {
256         final int FACTOR = ClassWriter.TO_ACC_SYNTHETIC;
257         int mask = Opcodes.ACC_DEPRECATED | ClassWriter.ACC_SYNTHETIC_ATTRIBUTE
258                 | ((access & ClassWriter.ACC_SYNTHETIC_ATTRIBUTE) / FACTOR);
259         out.putShort(access & ~mask).putShort(name).putShort(desc);
260         int attributeCount = 0;
261         if (value != 0) {
262             ++attributeCount;
263         }
264         if ((access & Opcodes.ACC_SYNTHETIC) != 0) {
265             if ((cw.version & 0xFFFF) < Opcodes.V1_5
266                     || (access & ClassWriter.ACC_SYNTHETIC_ATTRIBUTE) != 0) {
267                 ++attributeCount;
268             }
269         }
270         if ((access & Opcodes.ACC_DEPRECATED) != 0) {
271             ++attributeCount;
272         }
273         if (ClassReader.SIGNATURES && signature != 0) {
274             ++attributeCount;
275         }
276         if (ClassReader.ANNOTATIONS && anns != null) {
277             ++attributeCount;
278         }
279         if (ClassReader.ANNOTATIONS && ianns != null) {
280             ++attributeCount;
281         }
282         if (ClassReader.ANNOTATIONS && tanns != null) {
283             ++attributeCount;
284         }
285         if (ClassReader.ANNOTATIONS && itanns != null) {
286             ++attributeCount;
287         }
288         if (attrs != null) {
289             attributeCount += attrs.getCount();
290         }
291         out.putShort(attributeCount);
292         if (value != 0) {
293             out.putShort(cw.newUTF8("ConstantValue"));
294             out.putInt(2).putShort(value);
295         }
296         if ((access & Opcodes.ACC_SYNTHETIC) != 0) {
297             if ((cw.version & 0xFFFF) < Opcodes.V1_5
298                     || (access & ClassWriter.ACC_SYNTHETIC_ATTRIBUTE) != 0) {
299                 out.putShort(cw.newUTF8("Synthetic")).putInt(0);
300             }
301         }
302         if ((access & Opcodes.ACC_DEPRECATED) != 0) {
303             out.putShort(cw.newUTF8("Deprecated")).putInt(0);
304         }
305         if (ClassReader.SIGNATURES && signature != 0) {
306             out.putShort(cw.newUTF8("Signature"));
307             out.putInt(2).putShort(signature);
308         }
309         if (ClassReader.ANNOTATIONS && anns != null) {
310             out.putShort(cw.newUTF8("RuntimeVisibleAnnotations"));
311             anns.put(out);
312         }
313         if (ClassReader.ANNOTATIONS && ianns != null) {
314             out.putShort(cw.newUTF8("RuntimeInvisibleAnnotations"));
315             ianns.put(out);
316         }
317         if (ClassReader.ANNOTATIONS && tanns != null) {
318             out.putShort(cw.newUTF8("RuntimeVisibleTypeAnnotations"));
319             tanns.put(out);
320         }
321         if (ClassReader.ANNOTATIONS && itanns != null) {
322             out.putShort(cw.newUTF8("RuntimeInvisibleTypeAnnotations"));
323             itanns.put(out);
324         }
325         if (attrs != null) {
326             attrs.put(cw, null, 0, -1, -1, out);
327         }
