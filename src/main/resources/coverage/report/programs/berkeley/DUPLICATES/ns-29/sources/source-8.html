47 /**
48  * See @link{VLSNIndex} for an overview of the mapping system. The VLSNTracker
49  * packages the VLSNRange and the cached, in-memory VLSNBuckets.
50  *
51  * The tracker has a notion of the "currentBucket", which is the one receiving
52  * updates. All other cached buckets are finished and are awaiting a write to
53  * the database. Those finished buckets will only be updated in special
54  * circumstances, such as log cleaning or replication stream truncation, when
55  * we can assume that there will no readers viewing the buckets.
56  */

204      * Return a bucket for reading a mapping for this VLSN. If vlsn is >
205      * lastOnDisk, a bucket is guaranteed to be returned.
206      */
207     synchronized VLSNBucket getGTEBucket(VLSN vlsn) {
208 
209         if (lastOnDiskVLSN.compareTo(vlsn) >= 0) {
210             /* The bucket is in the database, not the cache. */
211             return null;
212         }
213 
214         Long pivot = vlsn.getSequence() + 1;
215         SortedMap<Long, VLSNBucket> head = bucketCache.headMap(pivot);
216         VLSNBucket prevBucket = null;
217         if (head.size() > 0) {
218             prevBucket = head.get(head.lastKey());
219             if (prevBucket.owns(vlsn)) {
220                 return prevBucket;
221             }
222         }
223 
224         /*
225          * If the key is not in the headMap, we must return the next bucket
226          * with mappings that follow on.
227          */
228         SortedMap<Long, VLSNBucket> tail = bucketCache.tailMap(pivot);
229         if (tail.size() > 0) {
230             VLSNBucket bucket = tail.get(tail.firstKey());
231             assert (bucket.owns(vlsn) || bucket.follows(vlsn)) :
232                "VLSN " + vlsn + " got wrong bucket " + bucket;
233             return bucket;
234         }
235 
236         throw EnvironmentFailureException.unexpectedState
237             (envImpl, "VLSN " + vlsn + " should be held within this tracker. " +
238              this + " prevBucket=" + prevBucket);
239     }

242      * Get the bucket which holds a mapping for this VLSN. If there is no such
243      * bucket, get the one directly preceding it. If this VLSN is >=
244      * firstTrackedVLSN, then we should be able to guarantee that a bucket is
245      * returned.
246      */
247     synchronized VLSNBucket getLTEBucket(VLSN vlsn) {
248 
249         if (firstTrackedVLSN.equals(NULL_VLSN) ||
250             (firstTrackedVLSN.compareTo(vlsn) > 0)) {
251             return null;
252         }

255         SortedMap<Long, VLSNBucket> head = bucketCache.headMap(pivot);
256         if (head.size() > 0) {

265         VLSNBucket nextBucket = null;

274     /**
275      * Record a new VLSN->LSN mapping. We guarantee that the first and last
276      * VLSNs in the range have a mapping. If a VLSN comes out of order, we will
277      * discard it, and will not update the range.
278      */
279     synchronized void track(VLSN vlsn, long lsn, byte entryTypeNum) {
280 
281         VLSNBucket currentBucket;
282 
283         if (vlsn.compareTo(lastOnDiskVLSN) < 0) {
284 
285             /*
286              * This VLSN is a laggard. It belongs to a bucket that has already
287              * gone to disk. Since on disk buckets can't be modified, throw
288              * this mapping away. Do be sure to update the range in case
289              * lastSync or lastTxnEnd should be updated as a result of this
290              * mapping.
291              */
292             updateRange(vlsn, entryTypeNum);
293             return;
294         }
295 
296         if (bucketCache.size() == 0) {
297             /* Nothing in the tracker, add a new bucket. */
298             currentBucket = new VLSNBucket(DbLsn.getFileNumber(lsn), stride,
299                                            maxMappings, maxDistance, vlsn);
300             nBucketsCreated.increment();
301 
302             bucketCache.put(currentBucket.getFirst().getSequence(),
303                             currentBucket);
304         } else {
305             /* Find the last bucket. Only the last bucket is updateable. */
306             currentBucket =  bucketCache.get(bucketCache.lastKey());
307         }
308 
309         /*
310          * This VLSN is a laggard that was preceded by an earlier mapping which
311          * came unseasonably early. This VLSN can't fit into the current
312          * bucket, and since we only want to update the last bucket, we'll
313          * throw this mapping away.
314          */
315         if (currentBucket.follows(vlsn)) {
316             updateRange(vlsn, entryTypeNum);
317             return;
318         }
319 
320         if (!currentBucket.put(vlsn, lsn)) {
321 
322             /*
323              * Couldn't put the mapping in this bucket. Close it and move to a
324              * new one.
325              */
326             currentBucket =
327                 new VLSNBucket(DbLsn.getFileNumber(lsn), stride,
328                                maxMappings, maxDistance, vlsn);
329             nBucketsCreated.increment();
330             bucketCache.put(currentBucket.getFirst().getSequence(),
331                             currentBucket);
332             if (!currentBucket.put(vlsn, lsn)) {
333                 throw EnvironmentFailureException.unexpectedState
334                     (envImpl, "Couldn't put VLSN " + vlsn + " into " +
335                      currentBucket);
336             }
337         }
338 
339         updateRange(vlsn, entryTypeNum);
340 
341         /*
342          * Only update the firstTrackedVLSN if this mapping is really resident
343          * in a bucket. Don't update it if it's an out-of-order, skipped
344          * mapping.
345          */
346         if (firstTrackedVLSN == NULL_VLSN) {
347             firstTrackedVLSN = vlsn;
348         }
349     }

364     /**
365      * Flush the tracker cache to disk.
366      * Ideally, we'd allow concurrent VLSN put() calls while a flush to database
367      * is happening. To do that, we need a more sophisticated synchronization
368      * scheme that defines the immutable vs. mutable portion of the tracker
369      * cache. See SR [#17689]
370      */
371     synchronized void flushToDatabase(DatabaseImpl mappingDbImpl, Txn txn) {
372 
373     	VLSNRange flushRange = range;
374 
375         if (bucketCache.size() == 0) {
376             /*
377              * No mappings to write out, but it's possible that the range
378              * changed due to a truncation, in which case we must update the
379              * database. RangeTruncated is used to reduce the chance that
380              * we are doing unnecessary writes of the range record to the db.
381              */
382             if (rangeTruncated) {
383                 lastOnDiskVLSN = flushRange.writeToDatabase(envImpl,
384                 	                                    mappingDbImpl, txn);
385                 rangeTruncated = false;
386             }
387             return;
388         }
389 
390         /*
391          * Save information about the portion of the cache that we are trying
392          * to flush. Close off the last bucket, so the flush portion becomes
393          * immutable. In the past, this was in a synchronization block.
394          * This isn't strictly needed right now, since the method is currently
395          * fully synchronized, but is a good practice, and will be
396          * necessary if future changes make it possible to have concurrent
397          * puts.
398          */
399         VLSNBucket lastBucket = bucketCache.get(bucketCache.lastKey());
400         lastBucket.close();
401 
402         /*
403          * Write the flush portion to disk. Ideally, this portion would be done
404          * without synchronization on the tracker, to permit concurrent reads
405          * and writes to the tracker. Note that because there is no
406          * synchronization, the buckets must be flushed to disk in vlsn order,
407          * so that the vlsn index always looks consistent. There should be no
408          * gaps in the bucket range. This assumption permits concurrent access
409          * by VLSNIndex.getGTEBucketFromDatabase.
410          */
411         VLSN currentLastVLSN = lastOnDiskVLSN;
412         for (Long key : bucketCache.keySet()) {
413 
414             VLSNBucket target = bucketCache.get(key);
415 
416             /* Do some sanity checking before we write this bucket to disk. */
417             validateBeforeWrite(target,
418                                 flushRange,
419                                 currentLastVLSN,
420                                 target==lastBucket);
421             target.writeToDatabase(envImpl, mappingDbImpl, txn);
422             currentLastVLSN = target.getLast();
423         }
424 
425         lastOnDiskVLSN = flushRange.writeToDatabase(envImpl,
426         	                                    mappingDbImpl, txn);
427         rangeTruncated = false;
428 
429         /*
430          * Update the tracker to remove the parts that have been written to
431          * disk. Update firstTrackedVLSN, lastOnDiskVLSN, and clear the bucket
432          * cache.
433          *
434          * In the current implementation, bucketCache is guaranteed not to
435          * change, so we can just clear the cache. If we had written the
436          * buckets out without synchronization, the cache might have gained
437          * mappings, and we wouuld have to be careful to set the
438          * firstTrackedVLSN to be the first VLSN of the remaining buckets, and
439          * to set lastTrackedVLSN to be the last VLSN in flushRange.
440          *
441          * We would also have to explicitly call SortedMap.remove() to remove
442          * the written buckets. Do not use TreeMap.tailSet, which would
443          * inadvertently leave us with the semantics of a SubMap, which has
444          * mandated key ranges. With rollbacks, we may end up adding buckets
445          * that are "repeats" in the future.
446          */
447         bucketCache.clear();
448         firstTrackedVLSN = NULL_VLSN;
449     }

514     /**
515      * Try to advance the VLSNIndex ProtectedFileRange and truncate the head
516      * of the tracker's range, so that bytesNeeded can be freed by deleting
517      * files in this range.
518      *
519      * @return {deleteEnd, deleteFileNum} pair if the range was changed, or
520      * null if no change is currently needed/possible.
521      *  -- deleteEnd is the last VLSN to be truncated.
522      *  -- deleteFileNum the file having deleteEnd as its last VLSN.
523      */
524     synchronized Pair<VLSN, Long> tryTruncateFromHead(
525         final long bytesNeeded,
526         final LogItemCache logItemCache) {
527 
528         /* Do not allow the index to become empty. */
529         final long preserveVlsn = range.getLast().getSequence() -
530             envImpl.getConfigManager().getInt(RepParams.MIN_VLSN_INDEX_SIZE);
531 
532         if (preserveVlsn < 0) {
533             return null;
534         }
535 
536         /* Determine the vlsn/file after which to truncate the range. */
537         final Pair<VLSN, Long> truncateInfo =
538             envImpl.getFileProtector().checkVLSNIndexTruncation(
539                 bytesNeeded, new VLSN(preserveVlsn));
540 
541         if (truncateInfo == null) {
542             return null;
543         }
544 
545         /*
546          * It is possible that after checkVLSNIndexTruncation returns, a file
547          * in the range we expect to delete becomes protected. This is safe,
548          * because we know that a syncup will not begin while synchronized on
549          * the VLSNTracker, and feeders always advance their range (never
550          * retreat). It is also still beneficial to truncate the index because
551          * we expect the other protection for such a file to be removed soon,
552          * and then the file can be deleted quickly.
553          */
554         return truncateFromHead(
555             truncateInfo.first(), truncateInfo.second(), logItemCache) ?
556             truncateInfo : null;
557     }

559     /**
560      * Truncate the head of the tracker's range, and advance the
561      * ProtectedFileRange accordingly.
562      *
563      * @return true if range is changed, or false if no change is needed.
564      */
565     synchronized boolean truncateFromHead(
566         VLSN deleteEnd,
567         long deleteFileNum,
568         LogItemCache logItemCache) {
569 
570         /* The caller should pass a non-null VLSN, but check for safety. */
571         if (deleteEnd.equals(VLSN.NULL_VLSN)) {
572             return false;
573         }
574 
575         /*
576          * Check the VLSN found in the deleted file against the existing
577          * mappings. The range should not be empty, and doing the truncation
578          * should not remove the last sync point. We assume that once this
579          * environment has received any portion of the replication stream, it
580          * will maintain a minimum set of VLSNs.
581          */
582         VLSNRange oldRange = range;
583 
584         if (oldRange.getFirst().compareTo(deleteEnd) > 0) {
585             /* deleteEnd has already been cast out of the index. */
586             return false;
587         }
588 
589         if (oldRange.isEmpty()) {
590             throw EnvironmentFailureException.unexpectedState(
591                 envImpl, "Didn't expect current range to be empty. " +
592                 " End of delete range = " +  deleteEnd);
593         }
594 
595         if (!oldRange.getLastSync().equals(NULL_VLSN) &&
596             (deleteEnd.compareTo(oldRange.getLastSync()) > 0)) {
597 
598             throw EnvironmentFailureException.unexpectedState(
599                 envImpl, "Can't log clean away last matchpoint. DeleteEnd= " +
600                 deleteEnd + " lastSync=" + oldRange.getLastSync());
601         }
602 
603         /*
604          * Sanity checks are over and we are committed to changing the range.
605          *
606          * Clear the LogItemCache in case it contains entries with VLSNs being
607          * truncated. This seems extremely unlikely, since the cache is a small
608          * number of entries at the end of the VLSN range. But perhaps it is
609          * simplest to clear it in order to guarantee safely, and incur the
610          * infrequent cost of a few cache misses.
611          */
612         if (logItemCache != null) {
613             logItemCache.clear();
614         }
615 
616         /*
617          * Advance the ProtectedFileRange to allow files in the truncated
618          * portion to be deleted. deleteFileNum is the file having deleteEnd as
619          * its highest VLSN. The VLSN range will now start with deleteEnd+1, so
620          * now we only need to protect files starting at deleteFileNum+1.
621          */
622         protectedFileRange.advanceRange(deleteFileNum + 1);
623 
624         /*
625          * Update the in-memory, cached portion of the index.
626          *
627          * Update range first, to prevent others from attempting to read the
628          * truncated portions. Update the firstTrackedLSN last, after the
629          * buckets are removed; getLTEBucket relies on the firstTrackedLSN
630          * value.
631          */
632         range = oldRange.shortenFromHead(deleteEnd);
633 
634         /*
635          * Ensure that the range is written to the db even if the resulting
636          * number of buckets is 0.
637          */
638         rangeTruncated = true;
639 
640         /*
641          * afterDelete may not == range.getFirst, becayse range.getFirst may
642          * be NULL_VLSN if the entire vlsn index has been deleted.
643          */
644         VLSN afterDelete = deleteEnd.getNext();
645 
646         /*
647          * Check if the cached buckets needs to be modified. Suppose vlsns 1-8
648          * are on disk, and vlsns 12->40 are in the tracker cache, with an out
649          * of order gap from 9 - 11.
650          *
651          * case 1: deleteEnd <= 8, Do not need to truncate anything in the
652          *  bucket cacke, but we may have to make a ghost bucket to fill the
653          *  gap.
654          * case 2: deleteEnd == 11, we don't need to modify the cache or
655          *         create a ghost bucket.
656          * case 3: deleteEnd >8 and < 11 is in the gap, we need to create a
657          *         ghost bucket to hold the beginning of the range..
658          * case 4: deleteEnd >= 12, we need to modify the bucket cache.
659          */
660         if (!lastOnDiskVLSN.equals(VLSN.NULL_VLSN) &&
661             (lastOnDiskVLSN.compareTo(deleteEnd) >= 0)) {
662 
663             /*
664              * case 1:  the buckets in the tracker cache are unaffected, all
665              * the pertinent mappings are on disk.
666              */
667             if (lastOnDiskVLSN.equals(deleteEnd)) {
668                 if (firstTrackedVLSN.compareTo(afterDelete) > 0) {
669                     checkForGhostBucket(bucketCache, deleteFileNum);
670                 }
671                 lastOnDiskVLSN = NULL_VLSN;
672             }
673             return true;
674         }
675 
676         assert(!firstTrackedVLSN.equals(NULL_VLSN));
677 
678         if (firstTrackedVLSN.equals(afterDelete)) {
679             /* case 2: The tracker is lined up perfectly with the new range. */
680             lastOnDiskVLSN = NULL_VLSN;
681             return true;
682         }
683 
684         if (firstTrackedVLSN.compareTo(afterDelete) > 0) {
685             /*
686              * case 3: we have to make a ghost bucket.
687              */
688             checkForGhostBucket(bucketCache, deleteFileNum);
689             lastOnDiskVLSN = NULL_VLSN;
690             return true;
691         }
692 
693         /*
694          * case 4: we have to prune the buckets.
695          */
696         VLSNBucket owningBucket = getLTEBucket(deleteEnd);
697         Long retainBucketKey = owningBucket.getFirst().getNext().getSequence();
698         SortedMap<Long, VLSNBucket> tail;
699         try {
700             /*
701              * We need to chop off part of the bucket cache. Find the portion
702              * to retain.
703              */
704             tail = bucketCache.tailMap(retainBucketKey);
705             checkForGhostBucket(tail, deleteFileNum);
706             bucketCache = tail;
707         } catch (NoSuchElementException e) {
708             firstTrackedVLSN = NULL_VLSN;
709             bucketCache = new TreeMap<>();
710         }
711 
712         lastOnDiskVLSN = NULL_VLSN;
713         return true;
714     }

717                                      long deleteFileNum) {
718         /*
719          * The range has just been truncated. If the first bucket's first vlsn
720          * != the first in range, insert a GhostBucket.
721          */
722         Long firstKey = buckets.firstKey();
723         VLSNBucket firstBucket = buckets.get(firstKey);

738     /**
739      * Remove the mappings for VLSNs >= deleteStart.  This should only be used
740      * at syncup, or recovery. We can assume no new mappings are coming in, but
741      * the VLSNIndex may be read during this time; checkpoints continue to
742      * happen during rollbacks. Since syncup is always at a sync-able log
743      * entry, and since we don't roll back past a commit, we can assume that
744      * the lastSync can be changed to deleteStart.getPrev(), and lastTxnEnd is
745      * unchanged.  The VLSNRange itself is updated before the tracker and
746      * on-disk buckets are pruned.
747      *
748      * The VLSNIndex guarantees that the beginning and end vlsns in the range
749      * have mappings so LTE and GTE search operations work. When truncating the
750      * mappings in the tracker, check to see if the new end vlsn has a mapping,
751      * since that may not happen if the buckets are cut in the middle, or have
752      * a gap due to out of order mapping. For example, suppose mapping for vlsn
753      * 20 came before the mappings for vlsn 17, 18 and 19. The buckets would
754      * have this gap:
755      *     Bucket A: vlsns 10-16           
756      *     Bucket B: vlsns 20-25 
757      *     Bucket C: vlsns 26 - 30
758      * If the new end of vlsn range is 11-14, 17, 18, 19, 21 - 24, or 27-29,
759      * we have to end the vlsn tracker with a new mapping of 
760      * (vlsn (deleteStart-1) ->  prevLsn) to cap off the buckets.
761      *
762      * Cases:
763      * End of range is in the gap -- either 17, 18, 19
764      * 1. on disk buckets exist, but don't encompass the gap: bucket A on disk
765      * 2. all buckets are in the tracker cache: bucket A in tracker
766      * 3. on disk buckets encompass the gap: bucket A and B on disk.
767      */
768     synchronized void truncateFromTail(VLSN deleteStart, long prevLsn) {
769         
770         /* 
771          * Update the range first, it will define the range that the vlsnIndex
772          * covers. Then adjust any mappings held in the bucket cache. Don't
773          * update the lastOnDiskVLSN marker, which says what is on disk. Since
774          * the on-disk portion may also get truncated, the caller will update
775          * the lastOnDiskVLSN field when it inspects the database.
776          */
777         VLSNRange oldRange = range;
778         range = oldRange.shortenFromEnd(deleteStart);
779 
780         /*
781          * Ensure that the range is written to the db even if the resulting
782          * number of buckets is 0.
783          */
784         rangeTruncated = true;
785 
786         if (firstTrackedVLSN.equals(NULL_VLSN)) {
787             /* No mappings in this tracker */
788             return;
789         }
790 
791         if (firstTrackedVLSN.compareTo(deleteStart) >= 0) {
792 
793             /*
794              * Everything in this tracker should be removed. In addition, the
795              * caller may be removing some items from the database.
796              */
797             bucketCache.clear();
798             firstTrackedVLSN = NULL_VLSN;
799             return;
800         }
801 
802         /*
803          * We need to do some pruning of the bucket in the cache. Find the
804          * buckets in the bucketCache that have mappings >= deleteStart, and
805          * remove their mappings. First establish the headset of buckets that
806          * should be preserved.
807          */
808         VLSNBucket targetBucket = getGTEBucket(deleteStart);
809         Long targetKey = targetBucket.getFirst().getSequence();
810 
811         /* The newCache has buckets that should be preserved. */
812         SortedMap<Long, VLSNBucket> newCache =
813             new TreeMap<>(bucketCache.headMap(targetKey));
814 
815         /*
816          * Prune any mappings >= deleteStart out of targetBucket. If it has any
817          * mappings left, add it to newCache.
818          */
819         targetBucket.removeFromTail(deleteStart, prevLsn);
820         if (!targetBucket.empty()) {
821             newCache.put(targetBucket.getFirst().getSequence(),
822                          targetBucket);
823         }
824 
825         bucketCache = newCache;
826 
827         /*
828          * Now all truncated mappings have been removed from the index. Check
829          * that the end point of the vlsn range is represented in this
830          * tracker. Since vlsn mappings can  come out of order, it's
831          * possible that the buckets have a gap, and that truncation will
832          * cleave the bucket cache just at a point where a mapping is
833          * missing. For example, vlsn mappings come in this order:
834          *           vlsn 16, vlsn 18, vlsn 17
835          * causing the buckets to have this gap:
836          *           bucket N: vlsn 10 - 16
837          *           bucket N+1: vlsn 18 - 20
838          * and that the vlsn range is truncated with a deleteStart of vlsn 18.
839          * The new end range becomes 17, but the buckets do not have a
840          * mapping for vlsn 17. If so, create a mapping now.
841          *
842          * If we are at this point, we know that the tracker should contain
843          * a mapping for the last VLSN in the range. Fix it now, so the
844          * datastructure is as tidy as possible.
845          */
846         boolean needEndMapping;
847         if (bucketCache.isEmpty()) {
848             needEndMapping = true;
849         } else {
850             VLSNBucket lastBucket = bucketCache.get(bucketCache.lastKey());
851             needEndMapping = 
852                     (lastBucket.getLast().compareTo(range.getLast()) < 0);
853         }
854 
855         if (needEndMapping) {
856             addEndMapping(range.getLast(), prevLsn);
857         }
858     }

860     /**
861      * Called by the VLSNIndex to see if we have to add a bucket in the tracker
862      * after pruning the on-disk storage. The get{LTE,GTE}Bucket methods 
863      * assume that there are mappings for the first and last VLSN in the range.
864      * But since out of order vlsn mappings can cause non-contiguous buckets,
865      * the pruning may have lost the end mapping. [#23491]
866      */
867     synchronized void ensureRangeEndIsMapped(VLSN lastVLSN, long lastLsn) {
868 
869         /*
870          * if lastOnDiskVLSN < lastVLSN < firstTrackedVLSN or
871          *    lastOnDiskVLSN < lastVLSN and firstTrackedVLSN is null
872          * then we need to add a mapping for lastVLSN->lastLsn. Otherwise a
873          * mapping already exists.
874          */
875         if (lastOnDiskVLSN.compareTo(lastVLSN) < 0) {
876 
877             /* 
878              * The on-disk vlsn mappings aren't sufficient to cover the 
879              * lastVLSN. There needs to be a mapping in the tracker for the
880              * end point of the range.
881              */
882             if (!firstTrackedVLSN.equals(NULL_VLSN)) {
883 
884                 /* 
885                  * There are mappings in the tracker, so we can assume
886                  * that they cover the end point, because truncateFromTail() 
887                  * should have adjusted for that. But assert to be sure!
888                  */
889                 if (lastVLSN.compareTo(firstTrackedVLSN) < 0) { 
890                     throw EnvironmentFailureException.unexpectedState
891                         (envImpl,
892                          "Expected this tracker to cover vlsn " + lastVLSN +
893                          " after truncateFromTail. LastOnDiskVLSN= " +
894                          lastOnDiskVLSN + " tracker=" + this);
895                 }
896                 
897                 return;
898             }
899 
900             /* 
901              * There are no mappings in the tracker. The on disk mappings were
902              * pruned, and a gap was created between what was on disk and what
903              * is in the tracker cache. Add a mapping.
904              */
905             addEndMapping(lastVLSN, lastLsn);
906         }

922         nBucketsCreated.increment();

930      * Attempt to replace the mappings in this vlsnIndex for
931      * deleteStart->lastVLSN with those from the recovery mapper.
932      * Since we do not supply a prevLsn argument, we may not be able to "cap"
933      * the truncated vlsn bucket the same what that we can in
934      * truncateFromTail. Because of that, we may need to remove mappings that
935      * are < deleteStart.
936      *
937      * For example, suppose a bucket holds mappings
938      *   10 -> 101
939      *   15 -> 201
940      *   18 -> 301
941      * If deleteStart == 17, we will have to delete all the way to vlsn 15.
942      *
943      * The maintenance of VLSNRange.lastSync and lastTxnEnd are simplified
944      * because of assumptions we can make because we are about to append
945      * mappings found by the recovery tracker that begin at deleteStart. If
946      * lastSync and lastTxnEnd are <= deleteStart, we know that they will
947      * either stay the same, or be replaced by lastSync and lastTxnEnd from the
948      * recoveryTracker. Even we delete mappings that are < deleteStart, we know
949      * that we did not roll back past an abort or commit, so that we do not
950      * have to updated the lastSync or lastTxnEnd value.
951      */
952     void merge(VLSN prunedLastOnDiskVLSN, VLSNRecoveryTracker recoveryTracker) {
953 
954         VLSNRange oldRange = range;
955         range = oldRange.merge(recoveryTracker.range);
956         VLSN recoveryFirst = recoveryTracker.getRange().getFirst();
957 
958         lastOnDiskVLSN = prunedLastOnDiskVLSN;
959 
960         /*
961          * Find the buckets in the bucketCache that have mappings >=
962          * recoveryFirst, and remove their mappings. First establish the
963          * headset of buckets that should be preserved. At this point, we
964          * have already pruned the database, so the bucket set may not
965          * be fully contiguous -- we may have pruned away mappings that
966          * would normally be in the database.
967          */
968         if (bucketCache.size() == 0) {
969             /* Just take all the recovery tracker's mappings */
970             bucketCache = recoveryTracker.bucketCache;
971         } else {
972             VLSNBucket targetBucket = getGTEBucket(recoveryFirst);
973             Long targetKey = targetBucket.getFirst().getSequence();
974 
975             /* The newCache holds buckets that should be preserved. */
976             SortedMap<Long, VLSNBucket> newCache =
977                 new TreeMap<>(bucketCache.headMap(targetKey));
978 
979             /*
980              * Prune any mappings >= recoveryFirst out of targetBucket. If it
981              * has any mappings left, add it to newCache.
982              */
983             targetBucket.removeFromTail(recoveryFirst, DbLsn.NULL_LSN);
984             if (!targetBucket.empty()) {
985                 newCache.put(targetBucket.getFirst().getSequence(),
986                              targetBucket);
987             }
988 
989             newCache.putAll(recoveryTracker.bucketCache);
990             bucketCache = newCache;
991         }
992 
993         if (bucketCache.size() > 0) {
994             VLSNBucket firstBucket = bucketCache.get(bucketCache.firstKey());
995             firstTrackedVLSN = firstBucket.getFirst();
996         }
