128      * tallies up the results.
129      *
130      * @param proposal the new unique proposal.
131      *

133      * a promise from the majority of acceptors.

135      * @throws HigherNumberedProposal
136      */
137     private Phase1Result phase1(QuorumPolicy quorumPolicy, Proposal proposal)
138         throws HigherNumberedProposal {
139 
140         LoggerUtils.logMsg(logger, elections.getRepImpl(), formatter,
141                            Level.FINE, "Phase 1 proposal: " + proposal);
142 
143         /* Broadcast Propose requests. */
144         final Set<InetSocketAddress> acceptorSockets =
145             elections.getAcceptorSockets();
146         FutureTrackingCompService<MessageExchange> compService =
147             Utils.broadcastMessage

154         if (haveQuorum(quorumPolicy, result.promisories.size())) {

162     /**
163      * Tally the results from Phase 1. Terminate the tally if we see any
164      * rejection messages which indicate our proposal has been obsoleted by a
165      * more recent one. It also tracks the highest numbered proposal and the
166      * associated value that was accepted by the Acceptors.
167      *
168      * @param currentProposal the proposal for this round
169      * @param compService the futures resulting from Prepare messages
170      *
171      * @return results Summarized results from Phase 1
172      * @throws HigherNumberedProposal
173      */
174     private Phase1Result
175     tallyPhase1Results(Proposal currentProposal,
176                        final FutureTrackingCompService<MessageExchange>
177                        compService)
178         throws HigherNumberedProposal {
179 
180         final Phase1Result result = new Phase1Result();
181 
182         new Utils.WithFutureExceptionHandler<MessageExchange>
183                 (compService, 2 * elections.getProtocol().getReadTimeout(),
184                  TimeUnit.MILLISECONDS, logger, elections.getRepImpl(), null) {
185 
186             @Override
187             protected void processResponse(MessageExchange me) {
188 
189                 final MessageOp op = me.getResponseMessage().getOp();
190                 if (op == elections.getProtocol().REJECT) {
191 
192                     /*
193                      * The acceptor has already received a higher numbered
194                      * Prepare request. Abandon this round as an optimization.
195                      */
196                     compService.cancelFutures(true);
197                     phase1HigherProposal.increment();
198                     throw new HigherNumberedProposal(me.target);
199                 } else if (op == elections.getProtocol().PROMISE) {
200                     result.promisories.add(me.target);
201                     result.promisoryMessages.add(me);
202                     Promise accept = (Promise) me.getResponseMessage();
203                     Proposal acceptProposal = accept.getHighestProposal();
204                     if ((result.acceptProposal == null) ||
205                         ((acceptProposal != null) &&
206                             (acceptProposal.compareTo(result.acceptProposal) >
207                             0))) {
208                         result.acceptProposal = acceptProposal;
209                         result.acceptedValue = accept.getAcceptedValue();
210                     }
211                 } else if (op == elections.getProtocol().PROTOCOL_ERROR){
212                         ProtocolError protocolError =
213                             (ProtocolError) me.getResponseMessage();
214                         /* Possible protocol version, group, or config skew. */
215                         LoggerUtils.logMsg(logger, elections.getRepImpl(),
216                                            formatter, Level.WARNING,
217                                            "Proposer phase 1 protocol error: " +
218                                            protocolError.getMessage());
219                         return;
220                     } else {
221                         LoggerUtils.logMsg(logger, elections.getRepImpl(),
222                                            formatter, Level.WARNING,
223                                            "Proposer phase 1 unknown" +
224                                            " response: " + op);
225                         return;
226                     }
227                 }
228 
229             @Override
230             protected boolean isShutdown() {
231                 return shutdown;
232             }
233         }.execute();
234 
235         if ((result.promisories.size() > 0) &&
236             (result.acceptProposal == null)) {
237             result.acceptProposal = currentProposal;
238         }
239         promiseCount.set(result.promisories.size());
240 
241         return result;
242     }

280      * Tallies the results from Phase 2.
281      *
282      * @param compService the futures from phase2
283      * @return the Phase2Result
284      * @throws HigherNumberedProposal if a higher numbered proposal was
285      *         encountered
286      */
287     private Phase2Result
288         tallyPhase2Results(
289             final QuorumPolicy quorumPolicy,
290             final FutureTrackingCompService<MessageExchange> compService)
291         throws HigherNumberedProposal {
292 
293         final Phase2Result result = new Phase2Result();
294         try {
295             new Utils.WithFutureExceptionHandler<MessageExchange>
296             (compService, 2 * elections.getProtocol().getReadTimeout(),
297                 TimeUnit.MILLISECONDS, logger, elections.getRepImpl(), null) {
298 
299                 @Override
300                 protected void processResponse(MessageExchange me) {
301                     final ResponseMessage responseMessage =
302                         me.getResponseMessage();
303 
304                     final Protocol protocol = elections.getProtocol();
305                     final MessageOp op = responseMessage.getOp();
306                     if (op == protocol.REJECT) {
307                         compService.cancelFutures(true);
308                         phase2HigherProposal.increment();
309                         throw new HigherNumberedProposal(me.target);
310                     } else if (op == protocol.ACCEPTED) {
311                         result.accepts.add(me.target);
312                         if (haveQuorum(quorumPolicy,
313                                        result.accepts.size())) {
314                             compService.cancelFutures(true);
315                             throw new HaveQuorum();
316                         }
317                     } else if (op == protocol.PROTOCOL_ERROR) {
318                         final Protocol.ProtocolError errorMessage =
319                             (Protocol.ProtocolError)responseMessage;
320                         LoggerUtils.logMsg(logger, elections.getRepImpl(),
321                                            formatter, Level.WARNING,
322                                            "Proposer phase 2 " +
323                                                "protocol error:" +
324                                                errorMessage.getMessage());
325                     } else {
326                         LoggerUtils.logMsg(logger, elections.getRepImpl(),
327                                            formatter, Level.WARNING,
328                                            "Proposer Phase 2 " +
329                                                "unknown response: " + op);
330                     }
331                 }
332 
333                 @Override
334                 protected boolean isShutdown() {
335                     return shutdown;
336                 }
337             }.execute();
338 
339         } catch (HaveQuorum hq) {
340             /* Terminated phase2 early from an early quorum. */
341         }
342         return result;
343     }

363     /**
364      * Runs an election using the consensus algorithm. An election can be used
365      * either to determine an existing consensus value, or to force an election
366      * if the current value is not suitable.
367      *
368      * @param quorumPolicy the policy used to determine whether we have a
369      * quorum.
370      *
371      * @param retryPredicate the interface which determines whether a retry
372      * should be attempted.
373      *
374      * @return the WinningProposal encapsulating the proposal and chosen value
375      *
376      * @throws MaxRetriesException when a winning proposal could not be
377      * established.
378      * @throws InterruptedException
379      */
380     public WinningProposal issueProposal(QuorumPolicy quorumPolicy,
381                                          RetryPredicate retryPredicate)
382         throws MaxRetriesException, InterruptedException {
383 
384         statistics.clear();
385 
386         while (retryPredicate.retry()) {
387             try {
388                 final Proposal proposal = nextProposal();
389                 final Phase1Result result1 = phase1(quorumPolicy, proposal);
390                 if (result1 == null) {
391                     continue;
392                 }
393 
394                 final Value phase2Value =
395                     choosePhase2Value(result1.promisoryMessages);
396                 if (phase2Value == null) {
397                     continue;
398                 }
399 
400                 final Phase2Result result2 =
401                     phase2(quorumPolicy, proposal, phase2Value,
402                            result1.promisories);
403                 if (result2 == null) {
404                     continue;
405                 }
406 
407                 LoggerUtils.logMsg(logger, elections.getRepImpl(),
408                                    formatter, Level.INFO,
409                                    "Winning proposal: " + proposal +
410                                    " Value: " + phase2Value);
411                 return new WinningProposal(proposal, phase2Value, statistics);
412             } catch (HigherNumberedProposal eip) {
413                 continue;
414             }
415         }
416         throw new MaxRetriesException(statistics);
417     }
