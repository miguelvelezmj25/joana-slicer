25 /**
26  * An iterator that does not need closing because a cursor is not kept open
27  * across method calls.  A cursor is opened to read a block of records at a
28  * time and then closed before the method returns.
29  *
30  * @author Mark Hayes
31  */
32 class BlockIterator<E> extends BaseIterator<E> {
33 
34     private StoredCollection<E> coll;

37     /**
38      * Slots for a block of record keys and values.  The priKeys array is only
39      * used for secondary databases; otherwise it is set to the keys array.
40      */
41     private byte[][] keys;
42     private byte[][] priKeys;
43     private byte[][] values;
44 
45     /**
46      * The slot index of the record that would be returned by next().
47      * nextIndex is always greater or equal to zero.  If the next record is not
48      * available, then nextIndex is equal to keys.length or keys[nextIndex] is
49      * null.
50      *
51      * If the block is empty, then either the iterator is uninitialized or the
52      * key range is empty.  Either way, nextIndex will be the array length and
53      * all array values will be null.  This is the initial state set by the
54      * constructor.  If remove() is used to delete all records in the key
55      * range, it will restore the iterator to this initial state.  The block
56      * must never be allowed to become empty when the key range is non-empty,
57      * since then the iterator's position would be lost.  [#15858]
58      */
59     private int nextIndex;
60 
61     /**
62      * The slot index of the record last returned by next() or previous(), or
63      * the record inserted by add().  dataIndex is -1 if the data record is not
64      * available.  If greater or equal to zero, the slot at dataIndex is always
65      * non-null.
66      */
67     private int dataIndex;
68 
69     /**
70      * The iterator data last returned by next() or previous().  This value is
71      * set to null if dataIndex is -1, or if the state of the iterator is such
72      * that set() or remove() cannot be called.  For example, after add() this
73      * field is set to null, even though the dataIndex is still valid.

212     /**
213      * Inserts an added record at a given slot position and shifts other slots
214      * accordingly.  Also adjusts nextIndex and sets dataIndex to -1.
215      */
216     private void insertSlot(int i, DataCursor cursor) {
217 
218         if (i < keys.length) {
219             for (int j = keys.length - 1; j > i; j -= 1) {
220 
221                 /* Shift right. */
222                 keys[j] = keys[j - 1];
223                 priKeys[j] = priKeys[j - 1];
224                 values[j] = values[j - 1];
225 
226                 /* Bump key in recno-renumber database. */
227                 if (coll.view.recNumRenumber && keys[j] != null) {
228                     bumpRecordNumber(j);
229                 }
230             }
231             nextIndex += 1;
232         } else {
233             if (i != keys.length) {
234                 throw DbCompat.unexpectedState();
235             }
236             i -= 1;
237             for (int j = 0; j < i; j += 1) {
238                 /* Shift left. */
239                 keys[j] = keys[j + 1];
240                 priKeys[j] = priKeys[j + 1];
241                 values[j] = values[j + 1];
242             }
243         }
244         setSlot(i, cursor);
245         dataIndex = -1;
246     }

259     /**
260      * Deletes the given slot, adjusts nextIndex and sets dataIndex to -1.
261      */
262     private void deleteSlot(int i) {
263 
264         for (int j = i + 1; j < keys.length; j += 1) {
265             keys[j - 1] = keys[j];
266             priKeys[j - 1] = priKeys[j];
267             values[j - 1] = values[j];
268         }
269         int last = keys.length - 1;
270         keys[last] = null;
271         priKeys[last] = null;
272         values[last] = null;
273 
274         if (nextIndex > i) {
275             nextIndex -= 1;
276         }
277         dataIndex = -1;
278     }

292     public boolean hasNext() {
293 
294         if (isNextAvailable()) {
295             return true;
296         }
297         DataCursor cursor = null;
298         try {
299             cursor = new DataCursor(coll.view, writeAllowed);
300             int prev = nextIndex - 1;
301             boolean found = false;
302 
303             if (keys[prev] == null) {
304                 /* Get the first record for an uninitialized iterator. */
305                 OperationStatus status = cursor.getFirst(false);
306                 if (status == OperationStatus.SUCCESS) {
307                     found = true;
308                     nextIndex = 0;
309                 }
310             } else {
311                 /* Reposition to the last known key/data pair. */
312                 int repos = cursor.repositionRange
313                     (keys[prev], priKeys[prev], values[prev], false);
314 
315                 if (repos == DataCursor.REPOS_EXACT) {
316 
317                     /*
318                      * The last known key/data pair was found and will now be
319                      * in slot zero.
320                      */
321                     found = true;
322                     nextIndex = 1;
323 
324                     /* The data object is now in slot zero or not available. */
325                     if (dataIndex == prev) {
326                         dataIndex = 0;
327                     } else {
328                         dataIndex = -1;
329                         dataObject = null;
330                     }
331                 } else if (repos == DataCursor.REPOS_NEXT) {
332 
333                     /*
334                      * The last known key/data pair was not found, but the
335                      * following record was found and it will be in slot zero.
336                      */
337                     found = true;
338                     nextIndex = 0;
339 
340                     /* The data object is no longer available. */
341                     dataIndex = -1;
342                     dataObject = null;
343                 } else {
344                     if (repos != DataCursor.REPOS_EOF) {
345                         throw DbCompat.unexpectedState();
346                     }
347                 }
348             }
349 
350             if (found) {
351                 /* Clear all slots first in case an exception occurs below. */
352                 clearSlots();
353 
354                 /* Attempt to fill all slots with records. */
355                 int i = 0;
356                 boolean done = false;
357                 while (!done) {
358                     setSlot(i, cursor);
359                     i += 1;
360                     if (i < keys.length) {
361                         OperationStatus status = coll.iterateDuplicates() ?
362                                                  cursor.getNext(false) :
363                                                  cursor.getNextNoDup(false);
364                         if (status != OperationStatus.SUCCESS) {
365                             done = true;
366                         }
367                     } else {
368                         done = true;
369                     }
370                 }
371 
372             }
373 
374             /*
375              * If REPOS_EXACT was returned above, make sure we retrieved
376              * the following record.
377              */
378             return isNextAvailable();
379         } catch (Exception e) {
380             throw StoredContainer.convertException(e);
381         } finally {
382             coll.closeCursor(cursor);
383         }
384     }

386     public boolean hasPrevious() {
387 
388         if (isPrevAvailable()) {
389             return true;
390         }
391         if (!isNextAvailable()) {
392             return false;
393         }
394         DataCursor cursor = null;
395         try {
396             cursor = new DataCursor(coll.view, writeAllowed);
397             int last = keys.length - 1;
398             int next = nextIndex;
399             boolean found = false;
400 
401             /* Reposition to the last known key/data pair. */
402             int repos = cursor.repositionRange
403                 (keys[next], priKeys[next], values[next], false);
404 
405             if (repos == DataCursor.REPOS_EXACT ||
406                 repos == DataCursor.REPOS_NEXT) {
407 
408                 /*
409                  * The last known key/data pair, or the record following it,
410                  * was found and will now be in the last slot.
411                  */
412                 found = true;
413                 nextIndex = last;
414 
415                 /* The data object is now in the last slot or not available. */
416                 if (dataIndex == next && repos == DataCursor.REPOS_EXACT) {
417                     dataIndex = last;
418                 } else {
419                     dataIndex = -1;
420                     dataObject = null;
421                 }
422             } else {
423                 if (repos != DataCursor.REPOS_EOF) {
424                     throw DbCompat.unexpectedState();
425                 }
426             }
427 
428             if (found) {
429                 /* Clear all slots first in case an exception occurs below. */
430                 clearSlots();
431 
432                 /* Attempt to fill all slots with records. */
433                 int i = last;
434                 boolean done = false;
435                 while (!done) {
436                     setSlot(i, cursor);
437                     i -= 1;
438                     if (i >= 0) {
439                         OperationStatus status = coll.iterateDuplicates() ?
440                                                  cursor.getPrev(false) :
441                                                  cursor.getPrevNoDup(false);
442                         if (status != OperationStatus.SUCCESS) {
443                             done = true;
444                         }
445                     } else {
446                         done = true;
447                     }
448                 }
449             }
450 
451             /*
452              * Make sure we retrieved the preceding record after the reposition
453              * above.
454              */
455             return isPrevAvailable();
456         } catch (Exception e) {
457             throw StoredContainer.convertException(e);
458         } finally {
459             coll.closeCursor(cursor);
460         }
461     }

537     public void remove() {
538 
539         if (dataObject == null) {
540             throw new IllegalStateException();
541         }
542         DataCursor cursor = null;
543         boolean doAutoCommit = coll.beginAutoCommit();
544         try {
545             cursor = new DataCursor(coll.view, writeAllowed);
546             if (moveCursor(dataIndex, cursor)) {
547                 cursor.delete();
548                 deleteSlot(dataIndex);
549                 dataObject = null;
550 
551                 /*
552                  * Repopulate the block after removing all records, using the
553                  * cursor position of the deleted record as a starting point.
554                  * First try moving forward, since the user was moving forward.
555                  * (It is possible to delete all records in the block only by
556                  * moving forward, i.e, when nextIndex is greater than
557                  * dataIndex.)
558                  */
559                 if (nextIndex == 0 && keys[0] == null) {
560                     OperationStatus status;
561                     for (int i = 0; i < keys.length; i += 1) {
562                         status = coll.iterateDuplicates() ?
563                                  cursor.getNext(false) :
564                                  cursor.getNextNoDup(false);
565                         if (status == OperationStatus.SUCCESS) {
566                             setSlot(i, cursor);
567                         } else {
568                             break;
569                         }
570                     }
571 
572                     /*
573                      * If no records are found past the cursor position, try
574                      * moving backward.  If no records are found before the
575                      * cursor position, leave nextIndex set to keys.length,
576                      * which is the same as the initial iterator state and is
577                      * appropriate for an empty key range.
578                      */
579                     if (keys[0] == null) {
580                         nextIndex = keys.length;
581                         for (int i = nextIndex - 1; i >= 0; i -= 1) {
582                             status = coll.iterateDuplicates() ?
583                                      cursor.getPrev(false) :
584                                      cursor.getPrevNoDup(false);
585                             if (status == OperationStatus.SUCCESS) {
586                                 setSlot(i, cursor);
587                             } else {
588                                 break;
589                             }
590                         }
591                     }
592                 }
593                 coll.closeCursor(cursor);
594                 coll.commitAutoCommit(doAutoCommit);
595             } else {
596                 throw new IllegalStateException();
597             }
598         } catch (Exception e) {
599             coll.closeCursor(cursor);
600             throw coll.handleException(e, doAutoCommit);
601         }
602     }

604     public void add(E value) {
605 
606         /*
607          * The checkIterAddAllowed method ensures that one of the following two
608          * conditions holds and throws UnsupportedOperationException otherwise:
609          * 1- This is a list iterator for a recno-renumber database.
610          * 2- This is a collection iterator for a duplicates database.
611          */
612         coll.checkIterAddAllowed();
613         OperationStatus status = OperationStatus.SUCCESS;
614         DataCursor cursor = null;
615         boolean doAutoCommit = coll.beginAutoCommit();
616         try {
617             if (coll.view.keysRenumbered || !coll.areDuplicatesOrdered()) {
618 
619                 /*
620                  * This is a recno-renumber database or a btree/hash database
621                  * with unordered duplicates.
622                  */
623                 boolean hasPrev = hasPrevious();
624                 if (!hasPrev && !hasNext()) {
625 
626                     /* The collection is empty. */
627                     if (coll.view.keysRenumbered) {
628 
629                         /* Append to an empty recno-renumber database. */
630                         status = coll.view.append(value, null, null);
631 
632                     } else if (coll.view.dupsAllowed &&
633                                coll.view.range.isSingleKey()) {
634 
635                         /*
636                          * When inserting a duplicate into a single-key range,
637                          * the main key is fixed, so we can always insert into
638                          * an empty collection.
639                          */
640                         cursor = new DataCursor(coll.view, writeAllowed);
641                         cursor.useRangeKey();
642                         status = cursor.putNoDupData(null, value, null, true);
643                         coll.closeCursor(cursor);
644                         cursor = null;
645                     } else {
646                         throw new IllegalStateException
647                             ("Collection is empty, cannot add() duplicate");
648                     }
649 
650                     /*
651                      * Move past the record just inserted (the cursor should be
652                      * closed above to prevent multiple open cursors in certain
653                      * DB core modes).
654                      */
655                     if (status == OperationStatus.SUCCESS) {
656                         next();
657                         dataIndex = nextIndex - 1;
658                     }
659                 } else {
660 
661                     /*
662                      * The collection is non-empty.  If hasPrev is true then
663                      * the element at (nextIndex - 1) is available; otherwise
664                      * the element at nextIndex is available.
665                      */
666                     cursor = new DataCursor(coll.view, writeAllowed);
667                     int insertIndex = hasPrev ? (nextIndex - 1) : nextIndex;
668 
669                     if (!moveCursor(insertIndex, cursor)) {
670                         throw new IllegalStateException();
671                     }
672 
673                     /*
674                      * For a recno-renumber database or a database with
675                      * unsorted duplicates, insert before the iterator 'next'
676                      * position, or after the 'prev' position.  Then adjust
677                      * the slots to account for the inserted record.
678                      */
679                     status = hasPrev ? cursor.putAfter(value)
680                                      : cursor.putBefore(value);
681                     if (status == OperationStatus.SUCCESS) {
682                         insertSlot(nextIndex, cursor);
683                     }
684                 }
685             } else {
686                 /* This is a btree/hash database with ordered duplicates. */
687                 cursor = new DataCursor(coll.view, writeAllowed);
688 
689                 if (coll.view.range.isSingleKey()) {
690 
691                     /*
692                      * When inserting a duplicate into a single-key range,
693                      * the main key is fixed.
694                      */
695                     cursor.useRangeKey();
696                 } else {
697 
698                     /*
699                      * When inserting into a multi-key range, the main key
700                      * is the last dataIndex accessed by next(), previous()
701                      * or add().
702                      */
703                     if (dataIndex < 0 || !moveCursor(dataIndex, cursor)) {
704                         throw new IllegalStateException();
705                     }
706                 }
707 
708                 /*
709                  * For a hash/btree with duplicates, insert the duplicate,
710                  * put the new record in slot zero, and set the next index
711                  * to slot one (past the new record).
712                  */
713                 status = cursor.putNoDupData(null, value, null, true);
714                 if (status == OperationStatus.SUCCESS) {
715                     clearSlots();
716                     setSlot(0, cursor);
717                     dataIndex = 0;
718                     nextIndex = 1;
719                 }
720             }
721 
722             if (status == OperationStatus.KEYEXIST) {
723                 throw new IllegalArgumentException("Duplicate value");
724             } else if (status != OperationStatus.SUCCESS) {
725                 DbCompat.unexpectedState("Could not insert: " + status);
726             }
727 
728             /* Prevent subsequent set() or remove() call. */
729             dataObject = null;
730 
731             coll.closeCursor(cursor);
732             coll.commitAutoCommit(doAutoCommit);
733         } catch (Exception e) {
734             /* Catch RuntimeExceptions too. */
735             coll.closeCursor(cursor);
