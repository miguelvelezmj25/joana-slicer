232     public Object getAttribute(String attributeName)
233         throws AttributeNotFoundException,
234                MBeanException {
235 
236         if (attributeName == null) {
237             throw new AttributeNotFoundException
238                 ("Attribute name can't be null.");
239         }
240 
241         try {
242             EnvironmentConfig envConfig = env.getConfig();
243             if (attributeName.equals(ATT_ENV_HOME)) {
244                 return env.getHome().getCanonicalPath();
245             } else if (attributeName.equals(ATT_IS_READ_ONLY)) {
246                 return new Boolean(envConfig.getReadOnly());
247             } else if (attributeName.equals(ATT_IS_TRANSACTIONAL)) {
248                 return new Boolean(envConfig.getTransactional());
249             } else if (attributeName.equals(ATT_CACHE_SIZE)) {
250                 return new Long(envConfig.getCacheSize());
251             } else if (attributeName.equals(ATT_CACHE_PERCENT)) {
252                 return new Integer(envConfig.getCachePercent());
253             } else if (attributeName.equals(ATT_LOCK_TIMEOUT)) {
254                 return new Long(envConfig.getLockTimeout());
255             } else if (attributeName.equals(ATT_IS_SERIALIZABLE)) {
256                 return new
257                     Boolean(envConfig.getTxnSerializableIsolation());
258             } else if (attributeName.equals(ATT_TXN_TIMEOUT)) {
259                 return new Long(envConfig.getTxnTimeout());
260             } else {
261                 throw new AttributeNotFoundException
262                     ("Attribute " + attributeName + " is not valid.");
263             }
264         } catch (DatabaseException e) {
265             /* Do not pass JE exceptions to the mbean client. */
266             throw new MBeanException(new RuntimeException(e.getMessage()));
267         } catch (Exception e) {
268             /* Ok to pass general Java exception to the mbean client. */
269             throw new MBeanException(e, e.getMessage());

315         } catch (DatabaseException e) {
316             throw new InvalidAttributeValueException

382     /**
383      * @see DynamicMBean#invoke
384      */
385     public Object invoke(String actionName,
386                          Object[] params,
387                          String[] signature)
388         throws MBeanException {
389 
390         /* Sanity checking. */
391         if (actionName == null) {
392             throw new IllegalArgumentException("ActionName can't be null.");
393         }
394 
395         try {
396             if (actionName.equals(OP_CLEAN)) {
397                 int numFiles = env.cleanLog();
398                 return new Integer(numFiles);
399             } else if (actionName.equals(OP_EVICT)) {
400                 env.evictMemory();
401                 return null;
402             } else if (actionName.equals(OP_CHECKPOINT)) {
403                 CheckpointConfig ckptConfig = new CheckpointConfig();
404                 if ((params != null) && (params.length > 0)) {
405                     Boolean force = (Boolean) params[0];
406                     ckptConfig.setForce(force.booleanValue());
407                 }
408                 env.checkpoint(ckptConfig);
409                 return null;
410             } else if (actionName.equals(OP_SYNC)) {
411                 env.sync();
412                 return null;
413             } else if (actionName.equals(OP_ENV_STAT)) {
414                 return env.getStats(getStatsConfig(params)).toString();
415             } else if (actionName.equals(OP_TXN_STAT)) {
416                 return env.getTransactionStats
417                     (getStatsConfig(params)).toString();
418             } else if (actionName.equals(OP_DB_NAMES)) {
419                 return env.getDatabaseNames();
420             } else if (actionName.equals(OP_DB_STAT)) {
421                 DatabaseStats stats = getDatabaseStats(params);
422                 return stats != null ? stats.toString() : null;
423             } else if (actionName.equals(OP_GET_TIPS)) {
424                 return env.getStats
425                     (getStatsConfig(new Object[] {false, true})).getTips();
426             } else if (actionName.equals(OP_ENV_CONFIG)) {
427                 return env.getConfig().toString();
428             }
429 
430             return new IllegalArgumentException
431                 ("ActionName: " + actionName + " is not valid.");
432         } catch (DatabaseException e) {
433 
434             /*
435              * Add the message for easiest deciphering of the problem. Since 
436              * the original exception cannot be transferred, send the exception 
437              * stack.
438              */
439             throw new MBeanException(new RuntimeException
440                                      (e.getMessage() +
441                                       LoggerUtils.getStackTrace(e)));
442         }
443     }

446      * Helper to get statistics for a given database.
447      *
448      * @param params operation parameters
449      * @return DatabaseStats object
450      */
451     private DatabaseStats getDatabaseStats(Object[] params)
452         throws IllegalArgumentException, 
453                DatabaseException {
454 
455         if ((params == null) || (params.length < 3)) {
456             return null;
457         }
458 
459         String dbName = (String)params[2];
460 
461         Database db = null;
462         try {
463             DatabaseConfig dbConfig = new DatabaseConfig();
464             dbConfig.setReadOnly(true);
465             DbInternal.setUseExistingConfig(dbConfig, true);
466             db = env.openDatabase(null, dbName, dbConfig);
467 
468             return db.getStats(getStatsConfig(params));
469         } finally {
470             if (db != null) {
471                 db.close();
472             }
473         }
474     }

487 
488         if (env == null) {

491

493         for (int i = 0; i < COMMON_ATTR.length; i++) {
494             attrList.add(COMMON_ATTR[i]);
495         }
496 
497         /* Add attributes for an open Environment. */
498         for (int i = 0; i < OPEN_ATTR.length; i++) {
499             attrList.add(OPEN_ATTR[i]);
500         }
501 
502         /* Add attributes for an open, transactional Environment. */
503         try {
504             EnvironmentConfig config = env.getConfig();
505             if (config.getTransactional()) {
506                 for (int i = 0; i < TRANSACTIONAL_ATTR.length; i++) {
507                     attrList.add(TRANSACTIONAL_ATTR[i]);
508                 }
509             }
510         } catch (DatabaseException ignore) {
511             /* ignore */
512         }
513 
514         return attrList.toArray(new MBeanAttributeInfo[attrList.size()]);
515     }

531         try {
532             if (env.getConfig().getTransactional()) {
533                 operationList.add(OP_CHECKPOINT_INFO);
534                 operationList.add(OP_TXN_STAT_INFO);
535             } else {
536                 operationList.add(OP_SYNC_INFO);
537             }
538         } catch (DatabaseException e) {
539              /* Don't make any operations available. */
540              operationList = new ArrayList<MBeanOperationInfo>();
541              return;
