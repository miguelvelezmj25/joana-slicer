120 /**
121  * Replays log records from the replication stream, and manages the
122  * transactions for those records.
123  *
124  * The Replay module has a lifetime equivalent to the environment owned by
125  * this replicator. Its lifetime is longer than the feeder/replica stream.
126  * For example, suppose this is nodeX:
127  *
128  * t1 - Node X is a replica, node A is master. Replay X is alive
129  * t2 - Node X is a replica, node B takes over as master. X's Replay module
130  *      is still alive and has the same set of active txns. It doesn't matter
131  *      to X that the master has changed.
132  * t3 - Node X becomes the master. Now its Replay unit is cleared, because
133  *      anything managed by the Replay is defunct.
134  */

170     /**
171      * ActiveTxns is a collection of txn objects used for applying replicated
172      * transactions. This collection should be empty if the node is a master.
173      *
174      * Note that there is an interesting relationship between ActiveTxns and
175      * the txn collection managed by the environment TxnManager. ActiveTxns is
176      * effectively a subset of the set of txns held by the
177      * TxnManager. ReplayTxns must be sure to register and unregister
178      * themselves from ActiveTxns, just as all Txns must register and
179      * unregister with the TxnManager's set. One implementation alternative to
180      * having an ActiveTxns map here is to search the TxnManager set (which is
181      * a set, not a map) for a given ReplayTxn. Another is to subclass
182      * TxnManager so that replicated nodes have their own replayTxn map, just
183      * as XATxns have a XID->Txn map.
184      *
185      * Both alternatives seemed too costly in terms of performance or elaborate
186      * in terms of code to do for the current function. It seems clearer to
187      * make the ActiveTxns a map in the one place that it is currently
188      * used. This choice may change over time, and should be reevaluated if the
189      * implementation changes.
190      *
191      * The ActiveTxns key is the transaction id. These transactions are closed
192      * when:
193      * - the replay unit executes a commit received over the replication stream
194      * - the replay unit executes an abort received over the replication stream
195      * - the replication node realizes that it has just become the new master,
196      *   and was not previously the master.
197      *
198      * Note that the Replay class has a lifetime that is longer than that of a
199      * RepNode. This means in particular, that transactions may be left open,
200      * and will be resumed when a replica switches from one master to another,
201      * creating a new RepNode in the process. Because of that, part of the
202      * transaction may be implemented by the rep stream from one master and
203      * another part by another.
204      *
205      * The map is synchronized, so simple get/put operations do not require
206      * additional synchronization.  However, iteration requires synchronization
207      * and copyActiveTxns can be used in most cases.
208      */
209     private final SimpleTxnMap<ReplayTxn> activeTxns;

415     private void updateCommitStats(final boolean needsAck,
416                                    final SyncPolicy syncPolicy,
417                                    final long startTimeNanos,
418                                    final long masterCommitTimeMs,
419                                    final long replicaCommitTimeMs) {
420 
421         final long now = System.nanoTime();
422         final long commitNanos = now - startTimeNanos;
423 
424         if (commitNanos > ackTimeoutLogThresholdNs &&
425             logger.isLoggable(Level.INFO)) {
426             LoggerUtils.info
427                 (logger, repImpl,
428                  "Replay commit time: " + (commitNanos / 1000000) +
429                  " ms exceeded log threshold: " +
430                  (ackTimeoutLogThresholdNs / 1000000));
431         }
432 
433         nCommits.increment();
434 
435         if (needsAck) {
436             nCommitAcks.increment();
437         }
438 
439         if (syncPolicy == SyncPolicy.SYNC) {
440             nCommitSyncs.increment();
441         } else if (syncPolicy == SyncPolicy.NO_SYNC) {
442             nCommitNoSyncs.increment();
443         } else if (syncPolicy == SyncPolicy.WRITE_NO_SYNC) {
444             nCommitWriteNoSyncs.increment();
445         } else {
446             throw EnvironmentFailureException.unexpectedState
447                 ("Unknown sync policy: " + syncPolicy);
448         }
449 
450         totalCommitProcessingNanos.add(commitNanos);
451         minCommitProcessingNanos.setMin(commitNanos);
452         maxCommitProcessingNanos.setMax(commitNanos);
453 
454         /*
455          * Tally the lag between master and replica commits, even if clock skew
456          * makes the lag appear negative.  The documentation already warns that
457          * the value will be affected by clock skew, so users can adjust for
458          * that, but only if we don't throw the information way.
459          */
460         final long replicaLagMs = replicaCommitTimeMs - masterCommitTimeMs;
461         totalCommitLagMs.add(replicaLagMs);
462         latestCommitLagMs.set(replicaLagMs);
463     }

468     public void replayEntry(long startNs,
469                             Protocol.Entry entry)
470         throws DatabaseException,
471                IOException,
472                InterruptedException,
473                MasterSyncException {
474

476         final LogEntry logEntry = wireRecord.getLogEntry();
477 
478         /*
479          * Sanity check that the replication stream is in sequence. We want to
480          * forestall any possible corruption from replaying invalid entries.
481          */
482         if (!wireRecord.getVLSN().follows(lastReplayedVLSN)) {
483             throw EnvironmentFailureException.unexpectedState
484                 (repImpl,
485                  "Rep stream not sequential. Current VLSN: " +
486                  lastReplayedVLSN +
487                  " next log entry VLSN: " + wireRecord.getVLSN());
488         }
489 
490         if (logger.isLoggable(Level.FINEST)) {
491             LoggerUtils.finest(logger, repImpl, "Replaying " + wireRecord);
492         }
493 
494         final ReplayTxn repTxn = getReplayTxn(logEntry.getTransactionId(), true);
495         updateReplicaSequences(logEntry);
496         final byte entryType = wireRecord.getEntryType();
497 
498         lastReplayedVLSN = wireRecord.getVLSN();
499 
500         try {
501             final long txnId = repTxn.getId();
502 
503             if (LOG_TXN_COMMIT.equalsType(entryType)) {
504                 Protocol.Commit commitEntry = (Protocol.Commit) entry;
505 
506                 final boolean needsAck = commitEntry.getNeedsAck();
507                 final SyncPolicy txnSyncPolicy =
508                     commitEntry.getReplicaSyncPolicy();
509                 final SyncPolicy implSyncPolicy =
510                     needsAck ?
511                     groupCommit.getImplSyncPolicy(txnSyncPolicy) :
512                     noAckSyncPolicy;
513 
514                 logReplay(repTxn, needsAck, implSyncPolicy);
515 
516                 final TxnCommit commit = (TxnCommit) logEntry.getMainItem();
517                 final long dtvlsn = updateDTVLSN(commit);
518 
519                 if (needsAck) {
520 
521                     /*
522                      * Only wait if the replica is not lagging and the
523                      * durability requires it.
524                      */
525                     repImpl.getRepNode().getVLSNFreezeLatch().awaitThaw();
526                     repImpl.getRepNode().getMasterStatus().assertSync();
527                 }
528 
529                 repTxn.commit(implSyncPolicy,
530                               new ReplicationContext(lastReplayedVLSN),
531                               commit.getMasterNodeId(),
532                               dtvlsn);
533 
534                 final long masterCommitTimeMs = commit.getTime().getTime();
535                 lastReplayedTxn = new TxnInfo(lastReplayedVLSN,
536                                               masterCommitTimeMs);
537 
538                 updateCommitStats(needsAck, implSyncPolicy, startNs,
539                                   masterCommitTimeMs, repTxn.getEndTime());
540 
541                 /* Respond to the feeder. */
542                 if (needsAck) {
543                     /*
544                      * Need an ack, either buffer it, for sync group commit, or
545                      * queue it.
546                      */
547                     if (!groupCommit.bufferAck(startNs, repTxn,
548                                                txnSyncPolicy)) {
549                         queueAck(txnId);
550                     }
551                 }
552 
553                 /*
554                  * The group refresh and recalculation can be expensive, since
555                  * it may require a database read. Do it after the ack.
556                  */
557                 if (repTxn.getRepGroupDbChange() && canRefreshGroup(repTxn)) {
558                     repImpl.getRepNode().refreshCachedGroup();
559                     repImpl.getRepNode().recalculateGlobalCBVLSN();
560                 }
561 
562                 nElapsedTxnTime.add(repTxn.elapsedTime());
563 
564             } else if (LOG_TXN_ABORT.equalsType(entryType)) {
565 
566                 nAborts.increment();
567                 final TxnAbort abort = (TxnAbort) logEntry.getMainItem();
568                 final ReplicationContext abortContext =
569                     new ReplicationContext(wireRecord.getVLSN());
570                 if (logger.isLoggable(Level.FINEST)) {
571                     LoggerUtils.finest(logger, repImpl,
572                                        "abort called for " + txnId +
573                                        " masterId=" +
574                                        abort.getMasterNodeId() +
575                                        " repContext=" + abortContext);
576                 }
577 
578                 long dtvlsn = updateDTVLSN(abort);
579                 repTxn.abort(abortContext, abort.getMasterNodeId(), dtvlsn);
580                 lastReplayedTxn = new TxnInfo(lastReplayedVLSN,
581                                               abort.getTime().getTime());
582                 if (repTxn.getRepGroupDbChange() && canRefreshGroup(repTxn)) {
583 
584                     /*
585                      * Refresh is the safe thing to do on an abort, since a
586                      * refresh may have been held back from an earlier commit
587                      * due to this active transaction.
588                      */
589                     repImpl.getRepNode().refreshCachedGroup();
590                 }
591                 nElapsedTxnTime.add(repTxn.elapsedTime());
592 
593             } else if (LOG_NAMELN_TRANSACTIONAL.equalsType(entryType)) {
594 
595                 repImpl.getRepNode().getReplica().clearDbTreeCache();
596                 nNameLNs.increment();
597                 applyNameLN(repTxn, wireRecord);
598 
599             } else {
600                 nLNs.increment();
601                 /* A data operation. */
602                 assert wireRecord.getLogEntry() instanceof LNLogEntry;
603                 applyLN(repTxn, wireRecord);
604             }
605 
606             /* Remember the last VLSN applied by this txn. */
607             repTxn.setLastAppliedVLSN(lastReplayedVLSN);
608 
609         } catch (DatabaseException e) {
610             e.addErrorMessage("Problem seen replaying entry " + wireRecord);
611             throw e;
612         } finally {
613             final long elapsedNs = System.nanoTime() - startNs;
614             if (elapsedNs > replayLoggingThresholdNs) {
615                 LoggerUtils.info(logger, repImpl,
616                                  "Replay time for entry type:" +
617                                  LogEntryType.findType(entryType) + " " +
618                                  NANOSECONDS.toMillis(elapsedNs) + "ms " +
619                                  "exceeded threshold:" +
620                                  NANOSECONDS.
621                                      toMillis(replayLoggingThresholdNs) +
622                                  "ms");
623             }
624         }
625     }

627     /**
628      * Update the replica's in-memory DTVLSN using the value in the
629      * commit/abort entry.
630      *
631      * In the normal course of events, DTVLSNs should not decrease. However,
632      * there is just one exception: if the rep stream transitions from a post
633      * to a pre-dtvlsn stream, it will transition from a positive to the
634      * UNINITIALIZED_VLSN_SEQUENCE.
635      *
636      * A transition from a pre to a post-dtvlsn transition (from zero to some
637      * positive value), observes the "DTVLSNs should not decrease" rule
638      * automatically.
639      *
640      * @return the DTVLSN entry in the log txnEnd record so it can be used in
641      * the commit/abort operation for replay
642      */
643     private long updateDTVLSN(final TxnEnd txnEnd) {
644         final long txnDTVLSN = txnEnd.getDTVLSN();
645 
646         if (txnDTVLSN == VLSN.UNINITIALIZED_VLSN_SEQUENCE) {
647             /*
648              * A pre DTVLSN format entry, simply set it as the in-memory DTVLSN
649              */
650             final long prevDTVLSN = repImpl.getRepNode().setDTVLSN(txnDTVLSN);
651             if (prevDTVLSN != VLSN.UNINITIALIZED_VLSN_SEQUENCE) {
652                 LoggerUtils.info(logger, repImpl,
653                                  "Transitioned to pre DTVLSN stream." +
654                                  " DTVLSN:" + prevDTVLSN +
655                                  " at VLSN:" + lastReplayedVLSN);
656 
657             }
658             lastReplayedDTVLSN = txnDTVLSN;
659             return txnDTVLSN;
660         }
661 
662         /* Sanity check. */
663         if (txnDTVLSN < lastReplayedDTVLSN) {
664             String msg = "DTVLSNs must be in ascending order in the stream. " +
665                 " prev DTVLSN:" + lastReplayedDTVLSN +
666                 " next DTVLSN:" + txnDTVLSN + " at VLSN: " +
667                 lastReplayedVLSN.getSequence();
668           throw EnvironmentFailureException.unexpectedState(repImpl, msg);
669         }
670 
671         if ((lastReplayedDTVLSN == VLSN.UNINITIALIZED_VLSN_SEQUENCE) &&
672             (txnDTVLSN > 0)) {
673             LoggerUtils.info(logger, repImpl,
674                              "Transitioned to post DTVLSN stream." +
675                              " DTVLSN:" + txnDTVLSN +
676                              " at VLSN:" + lastReplayedVLSN);
677         }
678 
679         lastReplayedDTVLSN = txnDTVLSN;
680         repImpl.getRepNode().setDTVLSN(txnDTVLSN);
681 
682         return txnDTVLSN;
683     }

813     /**
814      * Replays the NameLN.
815      *
816      * Note that the operations: remove, rename and truncate need to establish
817      * write locks on the database. Any open handles are closed by this
818      * operation by virtue of the ReplayTxn's importunate property.  The
819      * application will receive a LockPreemptedException if it subsequently
820      * accesses the database handle.
821      */
822     private void applyNameLN(ReplayTxn repTxn,
823                              InputWireRecord wireRecord)
824         throws DatabaseException {
825 
826         NameLNLogEntry nameLNEntry = (NameLNLogEntry) wireRecord.getLogEntry();
827         final NameLN nameLN = (NameLN) nameLNEntry.getLN();
828 
829         String databaseName = StringUtils.fromUTF8(nameLNEntry.getKey());
830 
831         final DbOpReplicationContext repContext =
832             new DbOpReplicationContext(wireRecord.getVLSN(), nameLNEntry);
833 
834         DbOperationType opType = repContext.getDbOperationType();
835         DatabaseImpl dbImpl = null;
836         try {
837             switch (opType) {
838                 case CREATE:
839                 {
840                     DatabaseConfig dbConfig =
841                         repContext.getCreateConfig().getReplicaConfig(repImpl);
842 
843                     dbImpl = repImpl.getDbTree().createReplicaDb
844                       (repTxn, databaseName, dbConfig, nameLN, repContext);
845 
846                     /*
847                      * We rely on the RepGroupDB.DB_ID value, so make sure
848                      * it's what we expect for this internal replicated
849                      * database.
850                      */
851                     if ((dbImpl.getId().getId() == RepGroupDB.DB_ID) &&
852                         !DbType.REP_GROUP.getInternalName().equals
853                         (databaseName)) {
854                         throw EnvironmentFailureException.unexpectedState
855                             ("Database: " +
856                              DbType.REP_GROUP.getInternalName() +
857                              " is associated with id: " +
858                              dbImpl.getId().getId() +
859                              " and not the reserved database id: " +
860                              RepGroupDB.DB_ID);
861                     }
862 
863                     TriggerManager.runOpenTriggers(repTxn, dbImpl, true);
864                     break;
865                 }
866 
867                 case REMOVE: {
868                     dbImpl = repImpl.getDbTree().getDb(nameLN.getId());
869                     try {
870                         repImpl.getDbTree().removeReplicaDb
871                             (repTxn, databaseName, nameLN.getId(), repContext);
872                         TriggerManager.runRemoveTriggers(repTxn, dbImpl);
873                     } catch (DatabaseNotFoundException e) {
874                         throw EnvironmentFailureException.unexpectedState
875                             ("Database: " + dbImpl.getName() +
876                              " Id: " + nameLN.getId() +
877                              " not found on the Replica.");
878                     }
879                     break;
880                 }
881 
882                 case TRUNCATE: {
883                     dbImpl = repImpl.getDbTree().getDb
884                         (repContext.getTruncateOldDbId());
885                     try {
886                         TruncateDbResult result =
887                         repImpl.getDbTree().truncateReplicaDb
888                             (repTxn, databaseName, false, nameLN, repContext);
889                         TriggerManager.runTruncateTriggers(repTxn, result.newDb);
890                     } catch (DatabaseNotFoundException e) {
891                         throw EnvironmentFailureException.unexpectedState
892                             ("Database: " + dbImpl.getName() +
893                              " Id: " + nameLN.getId() +
894                              " not found on the Replica.");
895                     }
896 
897                     break;
898                 }
899 
900                 case RENAME: {
901                     dbImpl = repImpl.getDbTree().getDb(nameLN.getId());
902                     try {
903                         dbImpl =
904                         repImpl.getDbTree().renameReplicaDb
905                             (repTxn, dbImpl.getName(), databaseName, nameLN,
906                              repContext);
907                         TriggerManager.runRenameTriggers(repTxn, dbImpl,
908                                                          databaseName);
909                     } catch (DatabaseNotFoundException e) {
910                         throw EnvironmentFailureException.unexpectedState
911                             ("Database rename from: " + dbImpl.getName() +
912                              " to " + databaseName +
913                              " failed, name not found on the Replica.");
914                     }
915                     break;
916                 }
917 
918                 case UPDATE_CONFIG: {
919                     /* Get the replicated database configurations. */
920                     DatabaseConfig dbConfig =
921                         repContext.getCreateConfig().getReplicaConfig(repImpl);
922 
923                     /* Update the NameLN and write it to the log. */
924                     dbImpl = repImpl.getDbTree().getDb(nameLN.getId());
925                     final String dbName = dbImpl.getName();
926                     repImpl.getDbTree().updateNameLN
927                         (repTxn, dbName, repContext);
928 
929                     /* Set the new configurations to DatabaseImpl. */
930                     dbImpl.setConfigProperties
931                         (repTxn, dbName, dbConfig, repImpl);
932 
933                     repImpl.getDbTree().modifyDbRoot(dbImpl);
934 
935                     break;
936                 }
937 
938                 default:
939                     throw EnvironmentFailureException.unexpectedState
940                         ("Illegal database op type of " + opType.toString() +
941                          " from " + wireRecord + " database=" + databaseName);
942             }
943         } finally {
944             if (dbImpl != null) {
945                 repImpl.getDbTree().releaseDb(dbImpl);
946             }
947         }
948     }

950     private void applyLN(
951         final ReplayTxn repTxn,
952         final InputWireRecord wireRecord)
953         throws DatabaseException {
954 
955         final LNLogEntry<?> lnEntry = (LNLogEntry<?>) wireRecord.getLogEntry();
956         final DatabaseId dbId = lnEntry.getDbId();
957 
958         /*
959          * If this is a change to the rep group db, remember at commit time,
960          * and refresh this node's group metadata.
961          */
962         if (dbId.getId() == RepGroupDB.DB_ID) {
963             repTxn.noteRepGroupDbChange();
964         }
965 
966         /*
967          * Note that we don't have to worry about serializable isolation when
968          * applying a replicated txn; serializable isolation in only an issue
969          * for txns that take read locks, and a replicated txn consists only of
970          * write operations.
971          */
972         final DatabaseImpl dbImpl =
973             repImpl.getRepNode().getReplica().getDbCache().get(dbId, repTxn);
974 
975         lnEntry.postFetchInit(dbImpl);
976 
977         final ReplicationContext repContext =
978             new ReplicationContext(wireRecord.getVLSN());
979 
980         try (final Cursor cursor = DbInternal.makeCursor(
981                 dbImpl, repTxn, null /*cursorConfig*/)) {
982 
983             OperationResult result;
984             final LN ln = lnEntry.getLN();
985 
986             /* In a dup DB, do not expect embedded LNs or non-empty data. */
987             if (dbImpl.getSortedDuplicates() &&
988                 (lnEntry.isEmbeddedLN() ||
989                  (ln.getData() != null && ln.getData().length > 0))) {
990 
991                 throw EnvironmentFailureException.unexpectedState(
992                     dbImpl.getEnv(),
993                     "[#25288] emb=" + lnEntry.isEmbeddedLN() +
994                     " key=" + Key.getNoFormatString(lnEntry.getKey()) +
995                     " data=" + Key.getNoFormatString(ln.getData()) +
996                     " vlsn=" + ln.getVLSNSequence());
997             }
998 
999             if (ln.isDeleted()) {
1000 
1001                 /*
1002                  * Perform an exact search by key. Use a partial data entry
1003                  * (delDataEntry) to avoid reading old data.
1004                  */
1005                 replayKeyEntry.setData(lnEntry.getKey());
1006 
1007                 result = DbInternal.searchForReplay(
1008                     cursor, replayKeyEntry, delDataEntry,
1009                     LockMode.RMW, SearchMode.SET);
1010 
1011                 if (result != null) {
1012                     result = DbInternal.deleteForReplay(cursor, repContext);
1013                 }
1014             } else {
1015                 replayKeyEntry.setData(lnEntry.getKey());
1016                 replayDataEntry.setData(ln.getData());
1017 
1018                 result = DbInternal.putForReplay(
1019                     cursor, replayKeyEntry, replayDataEntry, ln,
1020                     lnEntry.getExpiration(), lnEntry.isExpirationInHours(),
1021                     PutMode.OVERWRITE, repContext);
1022             }
1023 
1024             if (result == null) {
1025                 throw new EnvironmentFailureException(
1026                     repImpl,
1027                     EnvironmentFailureReason.LOG_INCOMPLETE,
1028                     "Replicated operation could  not be applied. " +
1029                     wireRecord);
1030             }
1031         }
1032     }

1034     /**
1035      * Go through all active txns and rollback up to but not including the log
1036      * entry represented by the matchpoint VLSN.
1037      *
1038      * Effectively truncate these rolled back log entries by making them
1039      * invisible. Flush the log first, to make sure these log entries are out
1040      * of the log buffers and are on disk, so we can reliably find them through
1041      * the FileManager.
1042      *
1043      * Rollback steps are described in
1044      * https://sleepycat.oracle.com/trac/wiki/Logging#Recoverysteps. In
1045      * summary,
1046      *
1047      * 1. Log and fsync a new RollbackStart record
1048      * 2. Do the rollback in memory. There is no need to explicitly
1049      *    log INs made dirty by the rollback operation.
1050      * 3. Do invisibility masking by overwriting LNs.
1051      * 4. Fsync all overwritten log files at this point.
1052      * 5. Write a RollbackEnd record, for ease of debugging
1053      *
1054      * Note that application read txns  can continue to run during syncup.
1055      * Reader txns cannot access records that are being rolled back, because
1056      * they are in txns that are not committed, i.e, they are write locked.
1057      * The rollback interval never includes committed txns, and we do a hard
1058      * recovery if it would include them.
1059      */
1060     public void rollback(VLSN matchpointVLSN, long matchpointLsn) {
1061 
1062         String rollbackStatus = RBSTATUS_START;
1063 
1064         final Map<Long, ReplayTxn> localActiveTxns = copyActiveTxns();
1065         try {
1066             if (localActiveTxns.size() == 0) {
1067                 /* no live read/write txns, nothing to do. */
1068                 rollbackStatus = RBSTATUS_NO_ACTIVE;
1069                 return;
1070             }
1071 
1072             VLSNRange range = repImpl.getVLSNIndex().getRange();
1073             if (range.getLast().equals(matchpointVLSN)) {
1074                 /* nothing to roll back. */
1075                 rollbackStatus = RBSTATUS_RANGE_EQUALS;
1076                 return;
1077             }
1078 
1079             repImpl.setSyncupProgress(SyncupProgress.DO_ROLLBACK);
1080 
1081             /*
1082              * Stop the log file backup service, since the files will be in an
1083              * inconsistent state while the rollback is in progress.
1084              */
1085             repImpl.getRepNode().shutdownNetworkBackup();
1086 
1087             /*
1088              * Set repImpl's isRollingBack to true, and invalidate all the in
1089              * progress DbBackup.
1090              */
1091             repImpl.setBackupProhibited(true);
1092             repImpl.invalidateBackups(DbLsn.getFileNumber(matchpointLsn));
1093 
1094             /*
1095              * 1. Log RollbackStart. The fsync guarantees that this marker will
1096              * be present in the log for recovery. It also ensures that all log
1097              * entries will be flushed to disk and the TxnChain will not have
1098              * to worry about entries that are in log buffers when constructing
1099              * the rollback information.
1100              */
1101             LogManager logManager = repImpl.getLogManager();
1102             LogEntry rollbackStart = SingleItemEntry.create(
1103                 LogEntryType.LOG_ROLLBACK_START,
1104                 new RollbackStart(
1105                     matchpointVLSN, matchpointLsn, localActiveTxns.keySet()));
1106             long rollbackStartLsn =
1107                 logManager.logForceFlush(rollbackStart,
1108                                          true, // fsyncRequired,
1109                                          ReplicationContext.NO_REPLICATE);
1110             rollbackStatus = RBSTATUS_LOG_RBSTART;
1111 
1112             /*
1113              * 2. Do rollback in memory. Undo any operations that were logged
1114              * after the matchpointLsn, and save the LSNs for those log
1115              * entries.. There should be something to undo, because we checked
1116              * earlier that there were log entries after the matchpoint.
1117              */
1118             List<Long> rollbackLsns = new ArrayList<Long>();
1119             for (ReplayTxn replayTxn : localActiveTxns.values()) {
1120                 Collection<Long> txnRollbackLsns =
1121                     replayTxn.rollback(matchpointLsn);
1122 
1123                 /*
1124                  * Txns that were entirely rolled back should have been removed
1125                  * from the activeTxns map.
1126                  */
1127                 assert checkRemoved(replayTxn) :
1128                     "Should have removed " + replayTxn;
1129 
1130                 rollbackLsns.addAll(txnRollbackLsns);
1131             }
1132             rollbackStatus = RBSTATUS_MEM_ROLLBACK;
1133             assert rollbackLsns.size() != 0 : dumpActiveTxns(matchpointLsn);
1134 
1135             /*
1136              * 3 & 4 - Mark the rolled back log entries as invisible.  But
1137              * before doing so, invoke any registered rewrite listeners, so the
1138              * application knows that existing log files will be modified.
1139              *
1140              * After all are done, fsync the set of files. By waiting, some may
1141              * have made it out on their own.
1142              */

1159                  true, // fsyncRequired
1160                  ReplicationContext.NO_REPLICATE);
1161

1164              * rollback were successful.
1165              */

1168             rollbackStatus = RBSTATUS_FINISH;
1169         } finally {

1172             lastReplayedVLSN = matchpointVLSN;

1174                              "Rollback to matchpoint " + matchpointVLSN +
1175                              " at " + DbLsn.getNoFormatString(matchpointLsn) +

1298     /**
1299      * Implements group commit. It's really a substructure of Replay and exists
1300      * mainly for modularity reasons.
1301      * <p>
1302      * Since replay is single threaded, the group commit mechanism works
1303      * differently in the replica than in the master. In the replica, SYNC
1304      * transactions are converted into NO_SYNC transactions and executed
1305      * immediately, but their acknowledgments are delayed until after either
1306      * the REPLICA_GROUP_COMMIT_INTERVAL (the max amount the first transaction
1307      * in the group is delayed) has expired, or the size of the group (as
1308      * specified by REPLICA_MAX_GROUP_COMMIT) has been exceeded.
1309      */
1310     private class GroupCommit {
1311 
1312         /* Size determines max fsync commits that can be grouped. */
1313         private final long pendingCommitAcks[];
1314 
1315         /* Number of entries currently in pendingCommitAcks */
1316         private int nPendingAcks;
1317 
1318         /*
1319          * If this time limit is reached, the group will be forced to commit.
1320          * Invariant: nPendingAcks > 0 ==> limitGroupCommitNs > 0
1321          */
1322         private long limitGroupCommitNs = 0;
1323 
1324         /* The time interval that an open group is held back. */
1325         private final long groupCommitIntervalNs;
1326 
1327         private final LongStat nGroupCommitTimeouts;
1328         private final LongStat nGroupCommitMaxExceeded;
1329         private final LongStat nGroupCommits;
1330         private final LongStat nGroupCommitTxns;
1331 
1332         private GroupCommit(DbConfigManager configManager) {
1333             pendingCommitAcks = new long[configManager.
1334                 getInt(RepParams.REPLICA_MAX_GROUP_COMMIT)];
1335 
1336             nPendingAcks = 0;
1337 
1338             final long groupCommitIntervalMs = configManager.
1339                 getDuration(RepParams.REPLICA_GROUP_COMMIT_INTERVAL);
1340 
1341             groupCommitIntervalNs =
1342                 NANOSECONDS.convert(groupCommitIntervalMs, MILLISECONDS);
1343             nGroupCommitTimeouts =
1344                 new LongStat(statistics, N_GROUP_COMMIT_TIMEOUTS);
1345 
1346             nGroupCommitMaxExceeded =
1347                 new LongStat(statistics, N_GROUP_COMMIT_MAX_EXCEEDED);
1348 
1349             nGroupCommitTxns =
1350                 new LongStat(statistics, N_GROUP_COMMIT_TXNS);
1351 
1352             nGroupCommits =
1353                 new LongStat(statistics, N_GROUP_COMMITS);
1354         }

1356         /**
1357          * Returns true if group commits are enabled at the replica.
1358          */
1359         private boolean isEnabled() {
1360             return pendingCommitAcks.length > 0;
1361         }
1362 
1363         /**
1364          * The interval used to poll for incoming log entries. The time is
1365          * lowered from the defaultNs time, if there are pending
1366          * acknowledgments.
1367          *
1368          * @param defaultNs the default poll interval
1369          *
1370          * @return the actual poll interval
1371          */
1372         private long getPollIntervalNs(long defaultNs) {
1373             if (nPendingAcks == 0) {
1374                 return defaultNs;
1375             }
1376             final long now = System.nanoTime();
1377 
1378             final long interval = limitGroupCommitNs - now;
1379             return Math.min(interval, defaultNs);
1380         }
1381 
1382         /**
1383          * Returns the sync policy to be implemented at the replica. If
1384          * group commit is active, and SYNC is requested it will return
1385          * NO_SYNC instead to delay the fsync.
1386          *
1387          * @param txnSyncPolicy the sync policy as stated in the txn
1388          *
1389          * @return the sync policy to be implemented by the replica
1390          */
1391         private SyncPolicy getImplSyncPolicy(SyncPolicy txnSyncPolicy) {
1392             return ((txnSyncPolicy ==  SyncPolicy.SYNC) && isEnabled()) ?
1393                    SyncPolicy.NO_SYNC : txnSyncPolicy;
1394         }
1395 
1396         /**
1397          * Buffers the acknowledgment if the commit calls for a sync, or if
1398          * there are pending acknowledgments to ensure that acks are sent
1399          * in order.
1400          *
1401          * @param nowNs the current time
1402          * @param ackTxn the txn associated with the ack
1403          * @param txnSyncPolicy the sync policy as request by the committing
1404          * txn
1405          *
1406          * @return true if the ack has been buffered
1407          */
1408         private final boolean bufferAck(long nowNs,
1409                                         ReplayTxn ackTxn,
1410                                         SyncPolicy txnSyncPolicy)
1411             throws IOException {
1412 
1413             if (!isEnabled() ||
1414                 !((txnSyncPolicy == SyncPolicy.SYNC) || (nPendingAcks > 0))) {
1415                 return false;
1416             }
1417 
1418             pendingCommitAcks[nPendingAcks++] = ackTxn.getId();
1419 
1420             if (nPendingAcks == 1) {
1421                 /* First txn in group, start the clock. */
1422                 limitGroupCommitNs = nowNs + groupCommitIntervalNs;
1423             } else {
1424                 flushPendingAcks(nowNs);
1425             }
1426             return true;
1427         }
1428 
1429         /**
1430          * Flush if there are pending acks and either the buffer limit or the
1431          * group interval has been reached.
1432          *
1433          * @param nowNs the current time (passed in to minimize system calls)
1434          */
1435         private final void flushPendingAcks(long nowNs)
1436             throws IOException {
1437 
1438             if ((nPendingAcks == 0) ||
1439                 ((nPendingAcks != pendingCommitAcks.length) &&
1440                  (NanoTimeUtil.compare(nowNs, limitGroupCommitNs) < 0))) {
1441 
1442                 return;
1443             }
1444 
1445             /* Update statistics. */
1446             nGroupCommits.increment();
1447             nGroupCommitTxns.add(nPendingAcks);
1448             if (NanoTimeUtil.compare(nowNs, limitGroupCommitNs) >= 0) {
1449                 nGroupCommitTimeouts.increment();
1450             } else if (nPendingAcks >= pendingCommitAcks.length) {
1451                 nGroupCommitMaxExceeded.increment();
1452             }
1453 
1454             /* flush log buffer and fsync to disk */
1455             repImpl.getLogManager().flushSync();
1456 
1457             /* commits are on disk, send out acknowledgments on the network. */
1458             for (int i=0; i < nPendingAcks; i++) {
1459                 queueAck(pendingCommitAcks[i]);
1460                 pendingCommitAcks[i] = 0;
1461             }
1462 
1463             nPendingAcks = 0;
1464             limitGroupCommitNs = 0;
1465         }
1466     }
