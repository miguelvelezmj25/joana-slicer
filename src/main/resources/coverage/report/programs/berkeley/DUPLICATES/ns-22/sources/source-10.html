128      * entry will only be created once, at most.
129      *

134     public synchronized ReplicableLogEntry instantiateEntry()
135         throws DatabaseException {
136 
137         if (logEntry != null) {
138             return logEntry;
139         }
140         if (logItem != null) {
141             logEntry = logItem.cachedEntry;
142             if (logEntry != null) {
143                 return logEntry;
144             }
145         }
146         final LogEntry entry = instantiateEntry(envImpl, entryBuffer);
147         if (!(entry instanceof ReplicableLogEntry)) {
148             throw EnvironmentFailureException.unexpectedState(
149                 "Log entry type does not support replication: " +
150                 entry.getClass().getName());
151         }
152         logEntry = (ReplicableLogEntry) entry;
153         if (logItem != null) {
154             logItem.cachedEntry = logEntry;
155         }
156         return logEntry;

265     /**
266      * Returns whether the log entry will be re-serialized when written, due to
267      * a required format change or because an optimized replication format
268      * should be used.
269      *
270      * This method caches its result, so that it can be called twice (from
271      * getEntrySize and writeToWire) and will not repeat the calculation or
272      * return different results. If true is returned, reSerializedSize will
273      * also be set.
274      */
275     private boolean willReSerialize(final int logVersion) {
276 
277         if (reserialize != null) {
278             return reserialize;
279         }
280 
281         int newSize = -1;
282 
283         if (header.getVersion() < 8) {
284             /*
285              * Before version 8, duplicates LN format conversion is necessary
286              * and we don't have the DatabaseImpl needed to do that, so we
287              * cannot convert to a later version. Conversion to an earlier
288              * version is not needed, because we can only convert to versions
289              * greater than 8. So re-serialization is not possible or needed.
290              */
291             reserialize = false;
292 
293         } else if (isOldFormatRequired(logVersion)) {
294             /* Re-serialization is mandatory. */
295             reserialize = true;
296 
297         } else {
298             /* Determine whether re-serialization is worthwhile. */
299             if (logEntry != null) {
300                 /*
301                  * If we have the entry, then using the optimized replication
302                  * format is worthwhile simply if the entry has an optimized
303                  * format and its size is smaller, since the cost of
304                  * re-serialization is fairly low.
305                  */
306                 if (logEntry.hasReplicationFormat()) {
307                     newSize = logEntry.getSize(
308                         logVersion, true /*forReplication*/);
309                     reserialize = header.getItemSize() > newSize;
310                 } else {
311                     reserialize = false;
312                 }
313             } else {
314                 /*
315                  * If we must materialize the entry in order to re-serialize
316                  * it, then we must make a best guess about whether this is
317                  * worthwhile by examining the entry in serialized format.
318                  */
319                 reserialize = getSharedEntry().isReplicationFormatWorthwhile(
320                     entryBuffer, header.getVersion(), logVersion);
321             }
322         }
323 
324         if (reserialize) {
325             if (newSize == -1) {
326                 newSize = instantiateEntry().getSize(
327                     logVersion, true /*forReplication*/);
328             }
329             reSerializedSize = newSize;
330         }
331 
332         assert reserialize != null;
333         return reserialize;
334     }

337      * Returns whether the format of the entry needs to be changed in order to
338      * be read by a replica that only understands versions no later than {@code
339      * logVersion}.
340      */
341     private boolean isOldFormatRequired(final int logVersion) {
342 
343         if (oldFormatRequired != null) {
344             return oldFormatRequired;
345         }
346 
347         oldFormatRequired =
348             /* The requested version is older than the current version, */
349             logVersion < LogEntryType.LOG_VERSION &&
350             /* it is older than the entry version, */
351             logVersion < header.getVersion() &&
352             /* and it is older than the entry class's last format change */
353             logVersion < (getSharedEntry().getLastFormatChange());
354 
355         return oldFormatRequired;
356     }

358     /**
359      * Write the log header and entry associated with this instance to the
360      * specified buffer using the format for the specified log version.
361      *
362      * @param messageBuffer the destination buffer
363      * @param logVersion the log version of the format
364      * @return whether the data format was changed to support an old version.
365      */
366     boolean writeToWire(final ByteBuffer messageBuffer,
367                         final int logVersion) {
368 
369         messageBuffer.put(header.getType());
370         if (willReSerialize(logVersion)) {
371             final ReplicableLogEntry entry = instantiateEntry();
372             LogUtils.writeInt(messageBuffer, logVersion);
373             LogUtils.writeInt(messageBuffer, reSerializedSize);
374             LogUtils.writeLong(messageBuffer, header.getVLSN().getSequence());
375             entryBuffer.mark();
376             entry.writeEntry(
377                 messageBuffer, logVersion, true /*forReplication*/);
378         } else {
379             LogUtils.writeInt(messageBuffer, header.getVersion());
380             LogUtils.writeInt(messageBuffer, header.getItemSize());
381             LogUtils.writeLong(messageBuffer, header.getVLSN().getSequence());
382             entryBuffer.mark();
383             messageBuffer.put(entryBuffer);
384         }
385         entryBuffer.reset();
386         return isOldFormatRequired(logVersion);
387     }

480      *
481      * It seems a little fragile to test this here, using instanceof to decide
482      * what to test. It would be cleaner to put this validity check as part of

484      * we want to keep the core code as independent as possible. The check is
485      * here rather than in some other test code because it needs to access the
486      * deserialized log entry. We don't want to provide a method which returns
487      * a logEntry, because in general an OutputWireRecord should not
488      * instantiate the log entry.
489      *
490      * @throws RuntimeException if there are any sequences that are not
491      * negative.
492      */
493     public boolean verifyNegativeSequences(final String debugTag) {
494 
495         LogEntry entry = null;
496         try {
497             entry = instantiateEntry();
498         } catch (DatabaseException e) {
499             throw EnvironmentFailureException.unexpectedException(e);
500         }
501 
502         if (entry.getTransactionId() >= 0) {
503             throw EnvironmentFailureException.unexpectedState
504                 (debugTag + " txn id should be negative: " + entry);
505         }
506 
507         if (entry instanceof LNLogEntry) {
508             if (LogEntryType.LOG_NAMELN_TRANSACTIONAL.equalsType
509                 (getEntryType())) {
510                 final LNLogEntry<?> lnEntry = (LNLogEntry<?>) entry;
511                 lnEntry.postFetchInit(false /*isDupDb*/);
512                 final NameLN nameLN = (NameLN) lnEntry.getLN();

514                     throw EnvironmentFailureException.unexpectedState
515                         (debugTag + " db id should be negative: " + entry);
516                 }
517             } else {
518                 if (entry.getDbId().getId() >= 0) {
519                     throw EnvironmentFailureException.unexpectedState
520                         (debugTag + " db id should be negative: " + entry);
521                 }
522             }
