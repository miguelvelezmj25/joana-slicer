26 /**
27  * A specialized <code>ObjectOutputStream</code> that stores class description
28  * information in a <code>ClassCatalog</code>.  It is used by
29  * <code>SerialBinding</code>.
30  *
31  * <p>This class is used instead of an {@link ObjectOutputStream}, which it
32  * extends, to write a compact object stream.  For writing objects to a
33  * database normally one of the serial binding classes is used.  {@link
34  * SerialOutput} is used when an {@link ObjectOutputStream} is needed along
35  * with compact storage.  A {@link ClassCatalog} must be supplied, however, to
36  * stored shared class descriptions.</p>
37  *
38  * <p>The {@link ClassCatalog} is used to store class definitions rather than
39  * embedding these into the stream.  Instead, a class format identifier is
40  * embedded into the stream.  This identifier is then used by {@link
41  * SerialInput} to load the class format to deserialize the object.</p>
42  *
43  * @see <a href="SerialBinding.html#evolution">Class Evolution</a>
44  *
45  * @author Mark Hayes
46  */
47 public class SerialOutput extends ObjectOutputStream {
48 
49     /*
50      * Serialization version constants. Instead of hardcoding these we get them
51      * by creating a SerialOutput, which itself guarantees that we'll always
52      * use a PROTOCOL_VERSION_2 header.
53      */
54     private final static byte[] STREAM_HEADER;
55     static {
56         ByteArrayOutputStream baos = new ByteArrayOutputStream();
57         try {
58             new SerialOutput(baos, null);
59         } catch (IOException e) {
60             throw RuntimeExceptionWrapper.wrapIfNeeded(e);
61         }
62         STREAM_HEADER = baos.toByteArray();
63     }

69      *
70      * @param out is the output stream to which the compact serialized objects
71      * will be written.
72      *
73      * @param classCatalog is the catalog to which the class descriptions for
74      * the serialized objects will be written.
75      *
76      * @throws IOException if an I/O error occurs while writing stream header.
77      */

89     // javadoc is inherited
90     protected void writeClassDescriptor(ObjectStreamClass classdesc)
91         throws IOException {
92 
93         try {
94             byte[] id = classCatalog.getClassID(classdesc);
95             writeByte(id.length);
96             write(id);
97         } catch (DatabaseException e) {
98 
99             /*
100              * Do not throw IOException from here since ObjectOutputStream
101              * will write the exception to the stream, which causes another
102              * call here, etc.
103              */
104             throw RuntimeExceptionWrapper.wrapIfNeeded(e);
105         } catch (ClassNotFoundException e) {
106             throw RuntimeExceptionWrapper.wrapIfNeeded(e);
107         }
108     }
109

112      * PROTOCOL_VERSION_2 format.  To save space this header can be removed and
113      * serialized streams before storage and inserted before deserializing.
