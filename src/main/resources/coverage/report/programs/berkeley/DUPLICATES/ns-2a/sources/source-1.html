52     public EnvStatsLogger(EnvironmentImpl env) {
53 
54         File statdirf;
55 
56         this.env = env;
57         logger = LoggerUtils.getLogger(getClass());
58         String statdir = env.getConfigManager().get(
59                 EnvironmentParams.STATS_FILE_DIRECTORY);
60         if (statdir == null || statdir.equals("")) {
61             statdirf = env.getEnvironmentHome();
62         } else {
63             statdirf = new File(statdir);
64         }
65 
66         try {
67             stlog = new StatLogger(statdirf,
68                                    STATFILENAME,
69                                    STATFILEEXT,
70                                    MAXFILECOUNT,
71                                    MAXROWCOUNT);
72         } catch (IOException e) {
73             throw new IllegalStateException(
74                 "Error accessing statistics capture file "+
75                 STATFILENAME + "." + STATFILEEXT +
76                 " IO Exception: " + e);
77         }
78     }

80     public void log() {
81         SortedMap<String, String> envConfigMap = new TreeMap<String, String>();
82         EnvironmentConfig mc = env.cloneConfig();
83         for (String colname :
84              EnvironmentParams.SUPPORTED_PARAMS.keySet()) {
85             envConfigMap.put("envcfg:" + colname, mc.getConfigParam(colname));
86         }
87         addSystemStats(envConfigMap);
88         sb.setLength(0);
89         valb.setLength(0);
90         sb.append("time");
91         valb.append(StatUtils.getDate(System.currentTimeMillis()));
92         for (Entry<String, String> e : envConfigMap.entrySet()) {
93             if (sb.length() != 0) {
94                 sb.append(DELIMITER);
95                 valb.append(DELIMITER);
96             }
97             sb.append(e.getKey());
98             valb.append(QUOTE + e.getValue() + QUOTE);
99         }
100         try {
101             stlog.setHeader(sb.toString());
102             stlog.logDelta(valb.toString());
103         } catch (IOException e) {
104             LoggerUtils.warning(logger, env,
105                 " Error accessing environment statistics file " +
106                 STATFILENAME + "." + STATFILEEXT +
107                 " IO Exception: " + e);
108         }
109         sb.setLength(0);
110         valb.setLength(0);
111     }

129         statmap.put("os:name", osbean.getName());
130         statmap.put("os:version", osbean.getVersion());
131         statmap.put("mc:arch", osbean.getArch());
132         statmap.put("mc:processors",
133                     Integer.toString(osbean.getAvailableProcessors()));
134         statmap.put("java:minMemory" ,
135                     Long.toString(memoryBean.getHeapMemoryUsage().getInit()));
136         statmap.put("java:maxMemory" ,
137                 Long.toString(memoryBean.getHeapMemoryUsage().getMax()));
138         List<String> args =
139                      ManagementFactory.getRuntimeMXBean().getInputArguments();
140         sb.setLength(0);

143         }
144         statmap.put("java:args", sb.toString());
145         sb.setLength(0);
