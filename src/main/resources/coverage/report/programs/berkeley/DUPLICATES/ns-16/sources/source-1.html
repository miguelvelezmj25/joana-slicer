37 /**
38  * Provides a mechanism to allow write availability for the Replication
39  * group even when the number of replication nodes is less than majority.
40  * The main use of an Arbiter is when the replication group consists of
41  * two nodes. The addition of an Arbiter to the replication group
42  * allows for one node to fail and provide write availability with ACK
43  * durability of SIMPLE_MAJORITY. The Arbiter acknowledges the transaction,
44  * but does not retain a copy of the data. The Arbiter persists a
45  * small amount of state to insure that only the Replication nodes that
46  * contain the Arbiter acknowledged transactions may become a Master.
47  * <p>
48  * The Arbiter node participates in elections and may acknowledge transaction
49  * commits.
50  * <p>
51  * The Arbiter state is as follows:
52  * UNKNOWN [ UNKNOWN | REPLICA]+ DETACHED
53  */

63     /**
64      * An Arbiter used in elections and transaction acknowledgments.
65      * This method returns when a connection to the current master
66      * replication node is made. The Arbiter.shutdown() method is
67      * used to shutdown the threads that run as part of the Arbiter.
68      *
69      * @param arbiterConfig Configuration parameters for the Arbiter.
70      *
71      * @throws EnvironmentNotFoundException if the environment does not exist
72      *
73      * @throws EnvironmentLockedException when an environment cannot be opened
74      * because another Arbiter has the environment open.
75      *
76      * @throws DatabaseException problem establishing connection to the master.
77      *
78      * @throws IllegalArgumentException if an invalid parameter is specified,
79      * for example, an invalid {@code ArbiterConfig} parameter.
80      */
81     public Arbiter(ArbiterConfig arbiterConfig)
82         throws EnvironmentNotFoundException,
83                EnvironmentLockedException,
84                DatabaseException,
85                IllegalArgumentException {
86 
87         ac = arbiterConfig.clone();
88         verifyParameters(ac);
89         File envHome = new File(ac.getArbiterHome());
90         if (!envHome.exists()) {
91             throw new IllegalArgumentException(
92                 "The specified environment directory " +
93                 envHome.getAbsolutePath() +
94                 " does not exist.");
95         }
96         Properties allProps = ac.getProps();
97         EnvironmentConfig envConfig =
98             new EnvironmentConfig(getEnvProps(allProps));
99         envConfig.setReadOnly(true);
100         envConfig.setTransactional(true);
101         envConfig.setConfigParam(
102             EnvironmentParams.ENV_RECOVERY.getName(), "false");
103         envConfig.setConfigParam(
104             EnvironmentParams.ENV_SETUP_LOGGER.getName(), "true");
105         envConfig.setConfigParam(
106             EnvironmentParams.LOG_USE_WRITE_QUEUE.getName(), "false");
107         envConfig.setConfigParam(
108             EnvironmentParams.LOG_WRITE_QUEUE_SIZE.getName(), "4096");
109         if (ac.getLoggingHandler() != null) {
110             envConfig.setLoggingHandler(ac.getLoggingHandler());
111         }
112 
113         ReplicationConfig repConfig =
114             new ReplicationConfig(getRepEnvProps(allProps));
115         repConfig.setConfigParam(RepParams.ARBITER_USE.getName(), "true");
116         repConfig.setRepNetConfig(ac.getRepNetConfig());
117 
118         repEnv = RepInternal.createInternalEnvHandle(envHome,
119                                                  repConfig,
120                                                  envConfig);
121         try {
122             ai = new ArbiterImpl(
123                 envHome, RepInternal.getNonNullRepImpl(repEnv));
124             ai.runArbiter();
125         } catch (Throwable t) {
126             shutdown();
127             throw t;
128         }
129     }

217         Iterator<Entry<Object, Object>> iter = props.entrySet().iterator();
218         while (iter.hasNext()) {
219             Entry<Object, Object> m = iter.next();
