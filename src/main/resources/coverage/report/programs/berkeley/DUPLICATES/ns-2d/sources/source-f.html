570     /**
571      * Called when an LN is logged by an operation that will not hold the lock
572      * such as eviction/checkpoint deferred-write logging or cleaner LN
573      * migration.  We must acquire a lock on the new LSN on behalf of every
574      * locker that currently holds a lock on the old LSN.
575      *
576      * Lock is non-blocking because no contention is possible on the new LSN.
577      *
578      * Because this locker is being used by multiple threads, this method may
579      * be called for a locker that has been closed or for which the lock on the
580      * old LSN has been released.  Unlike other locking methods, in this case
581      * we simply return rather than report an error.
582      */
583     public synchronized void lockAfterLsnChange(long oldLsn,
584                                                 long newLsn,
585                                                 DatabaseImpl dbImpl) {
586         if (!isValid()) {
587             /* Locker was recently closed, made abort-only, etc. */
588             return;
589         }
590 
591         final LockType lockType = lockManager.getOwnedLockType(oldLsn, this);
592         if (lockType == null) {
593             /* Lock was recently released. */
594             return;
595         }
596 
597         final LockResult lockResult = nonBlockingLock
598             (newLsn, lockType, false /*jumpAheadOfWaiters*/, dbImpl);
599 
600         if (lockResult.getLockGrant() == LockGrantType.DENIED) {
601             throw EnvironmentFailureException.unexpectedState
602                 ("No contention is possible on new LSN: " +
603                  DbLsn.getNoFormatString(newLsn) +
604                  " old LSN: " + DbLsn.getNoFormatString(oldLsn) +
605                  " LockType: " + lockType);
606         }
607     }
