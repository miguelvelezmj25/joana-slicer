33 /**
34  * Creates a Map that Threads can conveniently wait on to contain a specific
35  * number of entries, where the values optionally match a predicate. The wait
36  * functionality is provided by the sizeAwait() method defined by this
37  * class. Map values must not be null.
38  */

95     /**
96      * Causes the requesting thread to wait until the map reaches the specified
97      * size or the thread is interrupted.
98      *
99      * @param thresholdSize the size to wait for.
100      *
101      * @return true if the threshold was reached, false, if the wait timed out.
102      *
103      * @throws InterruptedException for the usual reasons, or if the map
104      * was cleared and the size threshold was not actually reached.
105      *
106      */
107     public boolean sizeAwait(int thresholdSize,
108                              long timeout,
109                              TimeUnit unit)
110         throws InterruptedException {
111 
112         assert(thresholdSize >= 0);
113         ExceptionAwareCountDownLatch l = null;
114         synchronized (this) {
115             if (thresholdSize <= count) {
116                 nNoWaits.increment();
117                 return true;
118             }
119             l = thresholdLatches.get(thresholdSize);
120             if (l == null) {
121                 l = new ExceptionAwareCountDownLatch(envImpl, 1);
122                 thresholdLatches.put(thresholdSize, l);
123             }
124         }
125         nRealWaits.increment();
126         long startTime = System.currentTimeMillis();
127         try {
128             return l.awaitOrException(timeout, unit);
129         } finally {
130             nWaitTime.add((System.currentTimeMillis() - startTime));
131         }
132     }

142     /**
143      * Notes the addition of a new value and counts down any latches that were
144      * assigned to that threshold.
145      */
146     @Override
147     public synchronized V put(final K key, final V value) {
148         if (value == null) {
149             throw new IllegalArgumentException("Value must not be null");
150         }
151         int countDelta = checkPredicate(value) ? 1 : 0;
152         final V oldValue = map.put(key, value);
153         if ((oldValue != null) && checkPredicate(oldValue)) {
154             countDelta--;
155         }
156         count += countDelta;
157         if (countDelta > 0) {
158             /* Incremented count */
159             final CountDownLatch l = thresholdLatches.remove(count);
160             if (l != null) {
161                 l.countDown();
162             }
163         }
164         return oldValue;
165     }
166 
167     /** Checks if the value matches the predicate. */
168     private boolean checkPredicate(final V value) {
169         return (predicate == null) || predicate.match(value);
170     }
171 
172     @Override
173     public synchronized V remove(Object key) {
174         final V oldValue = map.remove(key);
175         if ((oldValue != null) && checkPredicate(oldValue)) {
176             count--;
177         }
178         return oldValue;
179     }

191      * Clears the underlying map and the latch map, after first counting them
192      * down, thus permitting any waiting threads to make progress.
193      *
194      * @cause the value is non-null if the map is being cleared in response to
195      * an exception and results in the exception being thrown in the waiting
196      * threads. It's null if the map is being cleared as part of a normal
197      * shutdown, in which case no exception is thrown.
198      */
199     public synchronized void clear(Exception cause) {
200         for (ExceptionAwareCountDownLatch l : thresholdLatches.values()) {
201             l.releaseAwait(cause);
202         }
203         thresholdLatches.clear();
204         map.clear();
205         count = 0;
206     }
