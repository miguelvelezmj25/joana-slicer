688                 } catch (LockConflictException e) {

837      * Open a secondary database, setting AllowCreate, ExclusiveCreate and
838      * AllowPopulate appropriately.  We either set all three of these params to
839      * true or all to false.  This ensures that we only populate a database
840      * when it is created, never if it just happens to be empty.  [#16399]
841      *
842      * We also handle correction of a bug in duplicate ordering.  See
843      * ComplexFormat.incorrectlyOrderedSecKeys.
844      *
845      * @param doNotCreate is true when StoreConfig.getSecondaryBulkLoad is true
846      * and we are opening a secondary as a side effect of opening a primary,
847      * i.e., getSecondaryIndex is not being called.  If doNotCreate is true and
848      * the database does not exist, we silently ignore the failure to create
849      * the DB and return null.  When getSecondaryIndex is subsequently called,
850      * the secondary database will be created and populated from the primary --
851      * a bulk load.
852      */
853     private SecondaryDatabase
854         openSecondaryDatabase(final Transaction txn,
855                               final String[] fileAndDbNames,
856                               final PrimaryIndex priIndex,
857                               final String keyName,
858                               final SecondaryConfig config,
859                               final boolean doNotCreate)
860         throws DatabaseException {
861 
862         assert config.getAllowPopulate();
863         assert !config.getExclusiveCreate();
864         final Database priDb = priIndex.getDatabase();
865         final ComplexFormat entityFormat = (ComplexFormat)
866             ((PersistEntityBinding) priIndex.getEntityBinding()).entityFormat;
867         final boolean saveAllowCreate = config.getAllowCreate();
868         /* <!-- begin JE only --> */
869         final boolean saveOverrideDuplicateComparator = 
870             config.getOverrideDuplicateComparator();
871         /* <!-- end JE only --> */
872         final Comparator<byte[]> saveDupComparator = 
873             config.getDuplicateComparator();
874         try {
875             if (doNotCreate) {
876                 config.setAllowCreate(false);
877             }
878             /* First try creating a new database, populate if needed. */
879             if (config.getAllowCreate()) {
880                 config.setExclusiveCreate(true);
881                 /* AllowPopulate is true; comparators are set. */
882                 final SecondaryDatabase db = DbCompat.openSecondaryDatabase
883                     (env, txn, fileAndDbNames[0], fileAndDbNames[1], priDb,
884                      config);
885                 if (db != null) {
886                     /* For unit testing. */ 
887                     boolean doFlush = false;
888                     /* Update dup ordering bug info. [#17252] */
889                     if (config.getDuplicateComparator() != null &&
890                         entityFormat.setSecKeyCorrectlyOrdered(keyName)) {
891                         catalog.flush(txn);
892                         doFlush = true;
893                     }
894                     
895                     /* 
896                      * expectFlush is false except when set by
897                      * SecondaryDupOrderTest.
898                      */
899                     assert !expectFlush || doFlush;
900                     
901                     return db;
902                 }
903             }
904             /* Next try opening an existing database. */
905             config.setAllowCreate(false);
906             config.setAllowPopulate(false);
907             config.setExclusiveCreate(false);
908             
909             /* Account for dup ordering bug. [#17252] */
910             if (config.getDuplicateComparator() != null &&
911                 entityFormat.isSecKeyIncorrectlyOrdered(keyName)) {
912                 /* <!-- begin JE only --> */
913                 config.setOverrideDuplicateComparator(false);
914                 /* <!-- end JE only --> */
915                 config.setDuplicateComparator((Comparator<byte[]>) null);
916             }
917             final SecondaryDatabase db = DbCompat.openSecondaryDatabase
918                 (env, txn, fileAndDbNames[0], fileAndDbNames[1], priDb,
919                  config);
920             return db;
921         } finally {
922             config.setAllowPopulate(true);
923             config.setExclusiveCreate(false);
924             config.setAllowCreate(saveAllowCreate);
925             /* <!-- begin JE only --> */
926             config.setOverrideBtreeComparator(saveOverrideDuplicateComparator);
927             /* <!-- end JE only --> */
928             config.setDuplicateComparator(saveDupComparator);
929         }
930     }

1738     /**
1739      * Called via RefreshException.refresh when handling the RefreshException
1740      * in the binding methods, when a Replica detects that its in-memory
1741      * metadata is stale.
1742      *
1743      * During refresh, objects that are visible to the user must not be
1744      * re-created, since the user may have a reference to them.  The
1745      * PersistCatalog is re-created by this method, and the additional objects
1746      * listed below are refreshed without creating a new instance.  The
1747      * refresh() method of non-indented classes is called, and these methods
1748      * forward the call to indented classes.
1749      *
1750      *  PersistCatalog
1751      *      EntityModel
1752      *  PrimaryIndex
1753      *      PersistEntityBinding
1754      *          PersistKeyAssigner
1755      *  SecondaryIndex
1756      *      PersistKeyCreator
1757      *  PersistKeyBinding
1758      *
1759      * These objects have volatile catalog and format fields.  When a refresh
1760      * in one thread changes these fields, other threads should notice the
1761      * changes ASAP.  However, it is not necessary that all access to these
1762      * fields is synchronized.  It is Ok for a mix of old and new fields to be
1763      * used at any point in time.  If an old object is used after a refresh,
1764      * the need for a refresh may be detected, causing another call to this
1765      * method.  In most cases the redundant refresh will be avoided (see check
1766      * below), but in some cases an extra unnecessary refresh may be performed.
1767      * This is undesirable, but is not dangerous.  Synchronization must be
1768      * avoided to prevent blocking during read/write operations.
1769      *
1770      * [#16655]
1771      */
1772     synchronized PersistCatalog refresh(final PersistCatalog oldCatalog,
1773                                         final int errorFormatId,
1774                                         final RefreshException cause) {
1775 
1776         /*
1777          * While synchronized, check to see whether metadata has already been
1778          * refreshed.
1779          */
1780         if (oldCatalog != catalog) {
1781             /* Another thread refreshed the metadata -- nothing to do. */
1782             return catalog;
1783         }
1784 
1785         /*
1786          * First refresh the catalog information, then check that the new
1787          * metadata contains the format ID we're interested in using.
1788          */
1789         try {
1790             catalog = new PersistCatalog(oldCatalog, storePrefix);
1791         } catch (DatabaseException e) {
1792             throw RuntimeExceptionWrapper.wrapIfNeeded(e);
1793         }
1794 
1795         if (errorFormatId >= catalog.getNFormats()) {
1796             /* Even with current metadata, the format is out of range. */
1797             throw DbCompat.unexpectedException
1798                 ("Catalog could not be refreshed, may indicate corruption, " +
1799                  "errorFormatId=" + errorFormatId + " nFormats=" +
1800                  catalog.getNFormats() + ", .", cause);
1801         }
1802

1804          * Finally refresh all other objects that directly reference catalog
1805          * and format objects.
1806          */
1807         for (InternalPrimaryIndex index : priIndexMap.values()) {

1813         for (PersistKeyBinding binding : keyBindingMap.values()) {
1814             binding.refresh(catalog);

1816         for (SecondaryConfig config : secConfigMap.values()) {
1817             PersistKeyCreator keyCreator = getKeyCreator(config);

1819         }
1820 
1821         return catalog;
