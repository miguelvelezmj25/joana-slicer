98     public static FutureTrackingCompService<MessageExchange>
99         broadcastMessage(Set<InetSocketAddress> targets,
100                          String serviceName,
101                          RequestMessage requestMessage,
102                          ExecutorService threadPool) {
103 
104         final FutureTrackingCompService<MessageExchange> compService =
105             new FutureTrackingCompService<>(threadPool);
106 
107         for (InetSocketAddress socketAddress : targets) {
108             MessageExchange me = requestMessage.getProtocol()
109                 .new MessageExchange(socketAddress, serviceName, requestMessage);
110             try {
111                 compService.submit(me, me);
112             } catch (RejectedExecutionException ree) {
113                 if (threadPool.isTerminated()) {
114                     /*
115                      * The thread pool has been shutdown asynchronously as
116                      * part of a general elections shutdown. Discard submitted
117                      * and running tasks.
118                      */
119                     compService.cancelFutures(true);
120                     return compService;
121                 }
122 
123                 /*
124                  * Unexpected, rethrow so it can be reported at a higher
125                  * level.
126                  */
127                 throw ree;
128             }
129         }
130         return compService;
131     }

248         /**
249          * Processes futures in the order in which they complete, as
250          * determined by the completion service, to minimize unnecessary
251          * waiting.
252          */
253         public final void execute() {
254             final long limitTimeMs =
255                 System.currentTimeMillis() + unit.toMillis(completionTimeout);
256             try {
257                 for (int count = completionService.getFutures().size();
258                      count > 0; count--) {
259                     final long pollTimeoutMs =
260                         limitTimeMs - System.currentTimeMillis();
261                     if (pollTimeoutMs <= 0) {
262                         /* Timed out. */
263                         LoggerUtils.
264                         logMsg(logger, envImpl, formatter, Level.INFO,
265                                "Election messages timed out after " +
266                                unit.toMillis(completionTimeout) + "ms.");
267                         return;
268                     }
269 
270                     /* Wait for the next task that is ready. */
271                     final Future<T> f =
272                         completionService.poll(pollTimeoutMs,
273                                                TimeUnit.MILLISECONDS);
274 
275                     if (f == null) {
276                         /* Timed out. */
277                         LoggerUtils.
278                         logMsg(logger, envImpl, formatter, Level.INFO,
279                                "Election messages timed out after " +
280                                unit.toMillis(completionTimeout) + "ms.");
281                         return;
282                     }
283 
284                     if (isShutdown()) {
285                         LoggerUtils.
286                         logMsg(logger, envImpl, formatter, Level.INFO,
287                                "Election messages terminated." +
288                                " Environment being shutdown." );
289                         /* Simply exit. */
290                         return;
291                     }
292 
293                     if (f.isCancelled()) {
294                         continue;
295                     }
296 
297                     final long futureTimeoutMs =
298                         limitTimeMs - System.currentTimeMillis();
299                     if (futureTimeoutMs <= 0) {
300                         return;
301                     }
302 
303                     assert(f.isDone());
304                     executeInternal(f);
305                 }
306 
307             } catch (InterruptedException e) {
308 
309                 if (envImpl != null) {
310                     final RepNode rn = envImpl.getRepNode();
311                     if ((rn != null) && rn.isShutdown()) {
312 
313                         /*
314                          * Interrupt for shutdown, it's likely part of a
315                          * "hard" stoppable thread shutdown, ignore it.
316                          */
317                         LoggerUtils.
318                         logMsg(logger, envImpl, formatter, Level.INFO,
319                                "Election operation interrupted." +
320                                " Environment being shutdown." );
321                         /* Simply exit. */
322                         return;
323                     }
324                 }
325 
326                 throw EnvironmentFailureException.unexpectedException(e);
327             } finally {
328                 /*
329                  * Clean up all tasks, in case we exited before they were all
330                  * done.
331                  */
332                 completionService.cancelFutures(true);
333             }
334         }

336         /**
337          * The method represents the future process code being wrapped. It
338          * processes generic future-specific exceptions, where the impact of
339          * the fault can be localized to the future.
340          *
341          * @param future the "done" future
342          */
343         private final void executeInternal(Future<T> future)
344             throws InterruptedException {
345 
346             try {
347                 /* We don't expect to wait, since the future is "done". */
348                 assert future.isDone();
349                 T result = future.get();
350                 if (result.getResponseMessage() == null) {
351                     processNullResponse(result);
352                 } else {
353                     processResponse(result);
354                 }
355             } catch (CancellationException ce) {
356                 /* Ignore. */
357             } catch (ExecutionException e) {
358                 /* Get the true cause, unwrap the intermediate wrappers */
359                 Exception cause = (Exception)e.getCause();
360                 while (cause instanceof RuntimeException) {
361                     Throwable t = ((RuntimeException)cause).getCause();
362                     if ((t != null) && (t instanceof Exception)) {
363                         cause = (Exception)t;
364                     } else {
365                         break;
366                     }
367                 }
368                 if ((cause instanceof ConnectException) ||
369                     (cause instanceof SocketException) ||
370                     (cause instanceof SocketTimeoutException) ||
371                     (cause instanceof ServiceConnectFailedException)){
372                     // Network exceptions are expected, log it and keep moving
373                     LoggerUtils.logMsg(logger, envImpl, formatter, Level.FINE,
374                                        "Election connection failure " +
375                                        cause.getMessage());
376                     return;
377                 }
378                 /* Unanticipated exception, higher level will handle it */
379                 throw EnvironmentFailureException.unexpectedException(e);
380             }
381         }

385      * @hidden
386      *
387      * A subclass of ExecutorCompletionService that tracks submitted tasks, so
388      * that futures associated with tasks can be cancelled in a modular way,
389      * without the need for maintaining distinct state about the futures that
390      * were created.
391      */
392     public static class FutureTrackingCompService<V> extends
393         ExecutorCompletionService<V> {
394 
395         /* The list of futures resulting from submitted tasks. */
396         private final List<Future<V>> futures = new LinkedList<>();
397 
398         public FutureTrackingCompService(Executor executor) {
399             super(executor);
400         }
401 
402         public List<Future<V>> getFutures() {
403             return futures;
404         }
405 
406         /**
407          * Wrapper around submit() method to track futures resulting from
408          * submitted tasks.
409          */
410         @Override
411         public Future<V> submit(Runnable task, V result) {
412             final Future<V> f = super.submit(task, result);
413             futures.add(f);
414             return f;
415         }
416 
417         /**
418          * Wrapper around submit() method to track futures resulting from
419          * submitted tasks.
420          */
421         @Override
422         public Future<V> submit(Callable<V> task) {
423             final Future<V> f = super.submit(task);
424             futures.add(f);
425             return f;
426         }
427 
428         public void cancelFutures(boolean mayInterruptIfRunning) {
429             for (Future<V> f : futures) {
430                 if (!f.isCancelled()) {
431                     f.cancel(mayInterruptIfRunning);
432                 }
433             }
434         }
