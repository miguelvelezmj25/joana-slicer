21  * A mutation for converting an old version of an object value to conform to
22  * the current class or field definition.  For example:
23  *
24  * <pre class="code">
25  *  package my.package;
26  *
27  *  // The old class.  Version 0 is implied.
28  *  //
29  *  {@literal @Entity}
30  *  class Person {
31  *      // ...
32  *  }
33  *
34  *  // The new class.  A new version number must be assigned.
35  *  //
36  *  {@literal @Entity(version=1)}
37  *  class Person {
38  *      // Incompatible changes were made here...
39  *  }
40  *
41  *  // Add a converter mutation.
42  *  //
43  *  Mutations mutations = new Mutations();
44  *
45  *  mutations.addConverter(new Converter(Person.class.getName(), 0,
46  *                                       new MyConversion()));
47  *
48  *  // Configure the mutations as described {@link Mutations here}.</pre>

50  * <p>See {@link Conversion} for more information.</p>

86     public Converter(String declaringClassName,

92 
93         /* Require explicit implementation of the equals method. */
94         Class cls = conversion.getClass();
95         try {
96             Method m = cls.getMethod("equals", Object.class);
97             if (m.getDeclaringClass() == Object.class) {
98                 throw new IllegalArgumentException
99                     ("Conversion class does not implement the equals method " +
100                      "explicitly (Object.equals is not sufficient): " +
101                      cls.getName());
102             }
103         } catch (NoSuchMethodException e) {

105         }
106     }
