24 /**
25  * Warm-up the file system cache during startup, for some portion of the log
26  * that is not being read by recovery.
27  *
28  * See EnvironmentConfig.LOG_FILE_WARM_UP_SIZE (until we publish it, this is
29  * in EnvironmentParams).
30  */
31 public class FileCacheWarmer extends Thread {
32 
33     private final EnvironmentImpl envImpl;
34     private final long recoveryStartLsn;
35     private final long endOfLogLsn;
36     private final int warmUpSize;
37     private final int bufSize;
38     private volatile boolean stop;
39 
40     FileCacheWarmer(final EnvironmentImpl envImpl,
41                     final long recoveryStartLsn,
42                     final long endOfLogLsn,
43                     final int warmUpSize,
44                     final int bufSize) {
45         this.envImpl = envImpl;
46         this.recoveryStartLsn = recoveryStartLsn;
47         this.endOfLogLsn = endOfLogLsn;
48         this.warmUpSize = warmUpSize;
49         this.bufSize = bufSize;
50         stop = false;
51     }
52 
53     /**
54      * Stops this thread. At most one read will occur after calling this
55      * method, and then the thread will exit.
56      */
57     void shutdown() {
58         stop = true;
59     }
60 
61     @Override
62     public void run() {
63         try {
64             doRun();
65         } catch (Throwable e) {
66 
67             /*
68              * Log error as SEVERE but do not invalidate environment since it
69              * perfectly usable.
70              */
71             LoggerUtils.traceAndLogException(
72                 envImpl, FileCacheWarmer.class.getName(), "run",
73                 "Unable to warm file system cache due to exception", e);
74 
75         } finally {
76             /* Ensure that this thread can be GC'd after it stops. */
77             envImpl.getFileManager().clearFileCacheWarmer();
78         }
79     }
80 
81     private void doRun()
82         throws Throwable {
83 
84         final FileManager fm = envImpl.getFileManager();
85 
86         final long ONE_MB = 1L << 20;
87 
88         long remaining = (warmUpSize * ONE_MB) -
89             DbLsn.getTrueDistance(recoveryStartLsn, endOfLogLsn, fm);
90 
91         if (remaining <= 0) {
92             return;
93         }
94 
95         // System.out.println("FileCacheWarmer start " + remaining);
96 
97         final byte[] buf = new byte[bufSize];
98 
99         long fileNum = DbLsn.getFileNumber(recoveryStartLsn);
100         long fileOff = DbLsn.getFileOffset(recoveryStartLsn);
101 
102         String filePath = fm.getFullFileName(fileNum);
103         File file = new File(filePath);
104         RandomAccessFile raf = null;
105 
106         try {
107             raf = new RandomAccessFile(file, "r");
108 
109             while (!stop && remaining > 0) {
110 
111                 if (fileOff <= 0) {
112                     raf.close();
113                     raf = null;
114 
115                     while (!stop) {
116                         final Long nextFileNum = fm.getFollowingFileNum(
117                             fileNum, false /*forward*/);
118 
119                         if (nextFileNum == null) {
120                             return;
121                         }
122 
123                         fileNum = nextFileNum;
124                         filePath = fm.getFullFileName(fileNum);
125                         file = new File(filePath);
126                         try {
127                             raf = new RandomAccessFile(file, "r");
128                         } catch (FileNotFoundException e) {
129                             continue;
130                         }
131                         fileOff = raf.length();
132                         break;
133                     }
134                 }
135 
136                 final long pos = Math.max(0L, fileOff - bufSize);
137                 raf.seek(pos);
138 
139                 final int bytes = (int) (fileOff - pos);
140                 final int read = raf.read(buf, 0, bytes);
141 
142                 if (read != bytes) {
143                     throw new IllegalStateException(
144                         "Requested " + bytes + " bytes but read " + read);
145                 }
146 
147                 remaining -= bytes;
148                 fileOff = pos;
149             }
150 
151             raf.close();
152             raf = null;
153 
154         } finally {
155 
156             // System.out.println(
157             // "FileCacheWarmer finish " + remaining + " " + stop);
158 
159             if (raf != null) {
160                 try {
161                     raf.close();
162                 } catch (Exception e) {
163                     /* Ignore this. Another exception is in flight. */
164                 }
165             }
166         }
167     }
168 }
