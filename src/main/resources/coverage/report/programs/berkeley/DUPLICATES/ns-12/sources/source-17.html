20  * a portion of a LogBuffer.

22  */
23 class LogBufferSegment {

33      * Copies the data into the underlying LogBuffer
34      * and decrements the LogBuffer pin count.

36      *        LogBuffer.
37      */
38     public void put(ByteBuffer dataToCopy) {
39 
40         /*
41          * The acquisition of the log buffer latch is
42          * done to guarantee the java happens-before
43          * semantic. There is no other reason to take the
44          * latch here.
45          */
46         logBuffer.latchForWrite();
47         data.put(dataToCopy);
