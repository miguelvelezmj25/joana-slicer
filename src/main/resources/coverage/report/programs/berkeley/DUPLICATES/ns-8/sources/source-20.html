26 /**
27  * Specifies the environment attributes that may be changed after the
28  * environment has been opened.  EnvironmentMutableConfig is a parameter to
29  * {@link Environment#setMutableConfig} and is returned by {@link
30  * Environment#getMutableConfig}.
31  *
32  * <p>There are two types of mutable environment properties: per-environment
33  * handle properties, and environment wide properties.</p>
34  *
35  * <h4>Per-Environment Handle Properties</h4>
36  *
37  * <p>Per-environment handle properties apply only to a single Environment
38  * instance.  For example, to change the default transaction commit behavior
39  * for a single environment handle, do this:</p>
40  *
41  * <blockquote><pre>
42  *     // Specify no-sync behavior for a given handle.
43  *     EnvironmentMutableConfig mutableConfig = env.getMutableConfig();
44  *     mutableConfig.setDurability(Durability.COMMIT_NO_SYNC);
45  *     env.setMutableConfig(mutableConfig);
46  * </pre></blockquote>
47  *
48  * <p>The per-environment handle properties are listed below.  These properties
49  * are accessed using the setter and getter methods listed, as shown in the
50  * example above.</p>
51  *
52  * <ul>
53  * <li>{@link #setDurability}, {@link #getDurability}</li>
54  * <li>{@link #setTxnNoSync}, {@link #getTxnNoSync} <em>deprecated</em></li>
55  * <li>{@link #setTxnWriteNoSync}, {@link #getTxnWriteNoSync} <em>deprecated</em></li>
56  * </ul>
57  *
58  * <h4>Environment-Wide Mutable Properties</h4>
59  *
60  * <p>Environment-wide mutable properties are those that can be changed for an
61  * environment as a whole, irrespective of which environment instance (for the
62  * same physical environment) is used.  For example, to stop the cleaner daemon
63  * thread, do this:</p>
64  *
65  * <blockquote><pre>
66  *     // Stop the cleaner daemon threads for the environment.
67  *     EnvironmentMutableConfig mutableConfig = env.getMutableConfig();
68  *     mutableConfig.setConfigParam(EnvironmentConfig.ENV_RUN_CLEANER, "false");
69  *     env.setMutableConfig(mutableConfig);
70  * </pre></blockquote>
71  *
72  * <p>The environment-wide mutable properties are documented as such for each
73  * EnvironmentConfig String constant.</p>
74  *
75  * <h4>Getting the Current Environment Properties</h4>
76  *
77  * To get the current "live" properties of an environment after constructing it
78  * or changing its properties, you must call {@link Environment#getConfig} or
79  * {@link Environment#getMutableConfig}.  The original EnvironmentConfig or
80  * EnvironmentMutableConfig object used to set the properties is not kept up to
81  * date as properties are changed, and does not reflect property validation or
82  * properties that are computed.
83  *
84  * @see EnvironmentConfig
85  */

604      * Checks that the immutable values in the environment config used to open
605      * an environment match those in the config object saved by the underlying
606      * shared EnvironmentImpl.
607      * @param handleConfigProps are the config property values that were
608      * specified by configuration object from the Environment.
609      */
610     void checkImmutablePropsForEquality(Properties handleConfigProps)
611         throws IllegalArgumentException {
612 
613         Iterator<String> iter =
614             EnvironmentParams.SUPPORTED_PARAMS.keySet().iterator();
615         while (iter.hasNext()) {
616             String paramName = iter.next();
617             ConfigParam param =
618                 EnvironmentParams.SUPPORTED_PARAMS.get(paramName);
619             assert param != null;
620             if (!param.isMutable() && !param.isForReplication()) {
621                 String paramVal = props.getProperty(paramName);
622                 String useParamVal = handleConfigProps.getProperty(paramName);
623                 if ((paramVal != null) ?
624                     (!paramVal.equals(useParamVal)) :
625                     (useParamVal != null)) {
626                     throw new IllegalArgumentException
627                         (paramName + " is set to " +
628                          useParamVal +
629                          " in the config parameter" +
630                          " which is incompatible" +
631                          " with the value of " +
632                          paramVal + " in the" +
633                          " underlying environment");
634                 }
635             }
636         }
637     }

681      * Unchecked suppress here because Properties don't play well with
682      * generics in Java 1.5

684     @SuppressWarnings("unchecked")
685     void copyMutablePropsTo(EnvironmentMutableConfig toConfig) {
686 
687         Properties toProps = toConfig.props;
688         Enumeration propNames = props.propertyNames();
689         while (propNames.hasMoreElements()) {
690             String paramName = (String) propNames.nextElement();
691             ConfigParam param =
692                 EnvironmentParams.SUPPORTED_PARAMS.get(paramName);
693             assert param != null;
694             if (param.isMutable()) {
695                 String newVal = props.getProperty(paramName);
696                 toProps.setProperty(paramName, newVal);
697             }
698         }
699         toConfig.exceptionListener = this.exceptionListener;
700         toConfig.cacheMode = this.cacheMode;
701     }
