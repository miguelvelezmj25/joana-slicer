71     FileDeletionDetector(final File dbEnvHome,
72                          final File[] dbEnvDataDirs,
73                          final EnvironmentImpl envImpl) {
74         this.envImpl = envImpl;
75 
76         filesDeletedByJE = new HashSet<>();
77         fileDeletionWatchKeys = new HashMap<>();
78 
79         /*
80          * Create the WatchService which monitors the root env
81          * home or the sub-directories, such as data00N.
82          */
83         try {
84             fileDeletionWatcher =
85                 FileSystems.getDefault().newWatchService();
86 
87             /* 
88              * Register the root env home or the sub-directories.
89              * 
90              * If there exist sub-directories, then only sub-
91              * directories contain .jdb files, so we only register
92              * sub-directories. Otherwise, we only register the
93              * root env home.
94              *
95              * Here, we do not use Files.walkFileTree(
96              * Path, FileVisitor). This is because this method will
97              * check every entry(file or directory) under the root
98              * path. The following scenario may happen(
99              * com.sleepycat.je.MultiProcessWriteTest.
100              * testMultiEnvWrite):
101              * 1. One thread is closing env
102              * 2. The second thread wants to register the env home dir
103              * 3. In the second thread, Files.walkFileTree() finds
104              *    that je.info.0.lck, which is actually used by
105              *    the first thread
106              * 4. When Files.walkFileTree() wants to further check
107              *    the attribute of je.info.0.lck, je.info.0.lck can
108              *    not be found because the first thread have closed
109              *    the env.
110              *
111              * In a word, Files.walkFileTree may check more un-related
112              * files to cause unexpected Exception.
113              * 
114              * On AIX, the deletion detect has an about 4
115              * seconds delay, i.e. there is a 4 seconds delay from
116              * file deletion to deletion detect. We use
117              * SensitivityWatchEventModifier.HIGH to resolve this.
118              * 
119              * In addition, when directly deleting a directory which
120              * contains some files on AIX, WatchService can only
121              * detect directory-deletion event, but ignore the
122              * file-deletion event. So for AIX, if a directory is
123              * unexpected deleted, the current method can not handle
124              * it.
125              */
126             if (dbEnvDataDirs != null) {
127                 for (File f : dbEnvDataDirs) {
128                     final WatchKey key = f.toPath().register(
129                         fileDeletionWatcher,
130                         new WatchEvent.Kind[]{ENTRY_DELETE},
131                         SensitivityWatchEventModifier.HIGH);
132                     fileDeletionWatchKeys.put(key, f);
133                 }
134             } else {
135                 final WatchKey key = dbEnvHome.toPath().register(
136                     fileDeletionWatcher,
137                     new WatchEvent.Kind[]{ENTRY_DELETE},
138                     SensitivityWatchEventModifier.HIGH);
139                 fileDeletionWatchKeys.put(key, dbEnvHome);
140             }
141         } catch (IOException ie) {
142             throw new EnvironmentFailureException(
143                 envImpl,
144                 EnvironmentFailureReason.UNEXPECTED_EXCEPTION,
145                 "Can not register " +  dbEnvHome.toString() +
146                 " or its sub-directories to WatchService.",
147                 ie);
148         }
149 
150         DbConfigManager configManager = envImpl.getConfigManager();
151         final int interval = configManager.getDuration(
152             EnvironmentParams.LOG_DETECT_FILE_DELETE_INTERVAL);
153         /* Periodically detect unexpected log file deletion. */
154         fileDeletionTimer = new Timer(
155             envImpl.makeDaemonThreadName(
156                 Environment.FILE_DELETION_DETECTOR_NAME),
157             true);
158         fileDeletionTask = new FileDeleteDetectTask();
159         fileDeletionTimer.schedule(fileDeletionTask, 0, interval);
160     }

162     private class FileDeleteDetectTask extends TimerTask {
163         public void run() {
164             boolean success = false;
165             try {
166                 processLogFileDeleteEvents();
167                 success = true;
168             } catch (EnvironmentFailureException e) {
169                 /*
170                  * Log file is deleted unexpectedly. We have already
171                  * invalidated the environment. Here we just close
172                  * the Timer, TimerTask and the WatchService. Do this
173                  * in finally block.
174                  */
175             } catch (Exception e) {
176                 /*
177                  * It is possible that processLogFileDeleteEvents is doing
178                  * something on WatchService when FileManager.close call
179                  * WatchService.close. Then processLogFileDeleteEvents may
180                  * throw Exception, such as IOException or
181                  * java.nio.file.ClosedWatchServiceException. For this
182                  * situation, we just ignore the Exception and close
183                  * the Timer, TimerTask and the WatchService in finally
184                  * block. The close may be redundant, but for these close
185                  * method, the second and subsequent calls have no effect.
186                  *
187                  * If the Exception is not caused by the WatchService.close
188                  * in FileManager.close, then some ugly things happen. We
189                  * should handle the Exception by method which is similar
190                  * to StoppableThread.handleUncaughtException.
191                  *
192                  * !envImpl.isClosing represents that FileManager is not
193                  * closing. requestShutdownDaemons sets envImpl.isClosing
194                  * to be true and requestShutdownDaemons happens before
195                  * FileManager.close.
196                  *
197                  * envImpl.isValid represents that the current env is valid.
198                  * If the evn is invalid, then some other place has already
199                  * invalidate the env, so here we do not need to invalidate
200                  * the env again. 
201                  */
202                 if (envImpl.isValid() && !envImpl.isClosing()) {
203                     handleUnexpectedThrowable(Thread.currentThread(), e);
204                 }
205             } catch (Error e) {
206                 /*
207                  * Some ugly things happen.
208                  */
209                 handleUnexpectedThrowable(Thread.currentThread(), e);
210             } finally {
211                 if (!success) {
212                     try {
213                         close();
214                     } catch (IOException ie) {
215                         handleUnexpectedThrowable(Thread.currentThread(), ie);
216                     }
217                 }
218             }
219         }
220     }

224             envImpl.getLogger(), envImpl, t, e);

227     /*
228      * Detect unexpected log file deletion.
229      */
230     private void processLogFileDeleteEvents() throws Exception{
231         /*
232          * We may register multi-directories to the WatchService,
233          * there may be multi WatchKey for this WatchService. It is
234          * possible that file deletion happen in these multi directories
235          * simultaneously, i.e. multi WatchKey may be non-null.
236          * We should handle them all.
237          */
238         while (true) {
239             final WatchKey key = fileDeletionWatcher.poll();
240             if (key == null) {
241                 /*
242                  * If no event is detected, we check whether the directories
243                  * corresponding to the WatchKeys still exist.
244                  */
245                 for (final File file : fileDeletionWatchKeys.values()) {
246                     if (!file.exists()) {
247                         final String dir = file.getCanonicalPath();
248                         throw new IOException(
249                             "Directory " + dir + " does not exist now. " +
250                             "Something abnormal may happen.");
251                     }
252                 }
253                 break;
254             }
255  
256             for (final WatchEvent<?> event: key.pollEvents()) {
257                 final WatchEvent.Kind kind = event.kind();
258                 if (kind == OVERFLOW) {
259                     continue;
260                 }
261 
262                 /* Get the file name from the context. */
263                 final WatchEvent<Path> ev = cast(event);
264                 final String fileName = ev.context().toString();
265                 if (fileName.endsWith(FileManager.JE_SUFFIX)) {
266                     synchronized (filesDeletedByJE) {
267                         if (!filesDeletedByJE.contains(fileName)) {
268                             /* TimerTask.run will handle this exception. */
269                             throw new EnvironmentFailureException(
270                                 envImpl,
271                                 EnvironmentFailureReason.
272                                     LOG_UNEXPECTED_FILE_DELETION,
273                                 "Log file " + fileName +
274                                 " was deleted unexpectedly.");
275                         }
276                         filesDeletedByJE.remove(fileName);
277                     }
278                 }
279             }
280 
281             if (!key.reset()) {
282                 /*
283                  * If key.reset returns false, it indicates that the key
284                  * is no longer valid. Invalid state happens when one of
285                  * the following events occurs:
286                  * 1. The process explicitly cancels the key by using the
287                  *    cancel method. --- The JE code never does this.
288                  * 2. The directory becomes inaccessible.  -- This indicates
289                  *    an abnormal situation.
290                  * 3. The watch service is closed. -- The close may be
291                  *    expected, e.g. caused by WatchService.close. The
292                  *    exception caused by this can be handled in
293                  *    FileDeleteDetectTask.run.
294                  */
295                 final String dir =
296                     fileDeletionWatchKeys.get(key).getCanonicalPath();
297                 throw new IOException(
298                     "Watch Key corresponding to " + dir + " return false " +
299                     "when reset. Something abnormal may happen.");
300             }
301         }
302     }
