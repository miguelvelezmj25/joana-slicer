27 /**
28  * A RestoreMarker is a file that indicates that a normal recovery is not
29  * possible, because the log files are physically or semantically inconsistent
30  * in some way.
31  *
32  * One example is an interrupted, incomplete network restore.  The network
33  * restore copies log files from a source to destination node. If it's halted,
34  * while the destination node may contain files that are readable,
35  * checksum-able and seem correct, the set as a whole may not have a complete
36  * and coherent copy of the log. In such a case, recovery should not be run on
37  * this environment's log until some action is taken to make it consistent. For
38  * a network restore, this curative action is to restart the copy process, and
39  * to complete the copy of a set of logs from some node to this node. The
40  * network restore creates a marker file just before it does any form of change
41  * to the log that would make the log inconsistent.
42  *
43  * The restore marker file is named <MAXINT>.jdb, and holds a normal log file
44  * header and a RestoreRequired log entry. The RestoreRequired entry indicates
45  * the type of error of the initial cause, and information needed to repair the
46  * environment. The mechanism depends on the fact that the very first step of
47  * recovery is to read backwards from the last file in the
48  * environment. Recovery will start at this file, and will fail when it reads
49  * the RestoreRequired log entry, throwing an exception that contains
50  * prescriptive information for how the environment can be repaired.
51  *
52  * Note that createMarkerFile() is idempotent and can be safely called multiple
53  * times. It's done that way to make it simpler for the caller.
54  *
55  * The handler that repairs the log must also delete the marker file so future
56  * recoveries can succeed.
57  */
58 public class RestoreMarker {
59 
60     /**
61      * Internal exception used to distinguish marker file creation from other
62      * IOErrors.
63      */
64     public static class FileCreationException extends Exception {
65         FileCreationException(String msg, Throwable cause) {
66             super(msg, cause);
67         }
68     }

70     private final LogManager logManager;
71     private final File lastFile;
72 
73     public RestoreMarker(FileManager fileManager,
74                          LogManager logManager) {
75 
76         this.logManager = logManager;
77         String lastFileName = fileManager.getFullFileName(Integer.MAX_VALUE);
78         lastFile = new File(lastFileName);
79     }
80 
81     public static String getMarkerFileName() {
82         return FileManager.getFileName(Integer.MAX_VALUE);
83     }
84 
85     /**
86      * Remove the marker file. Use FileManager.delete so this file works with
87      * the FileDeletionDetector.
88      * @throws IOException if the file won't delete.
89      */
90     public void removeMarkerFile(FileManager fileManager)
91         throws IOException {
92 
93         if (lastFile.exists()) {
94             fileManager.deleteFile(Integer.MAX_VALUE);
95         }
96     }
97 
98     /**
99      * Create the restore marker file.
100      *
101      * The method may be called repeatedly, but will not re-create the marker
102      * file if there's already a non-zero length file.
103      *
104      * @param failureType the failure type that should be recorded in the
105      * RestoreRequired log entry.
106      * @param props will be serialized to store information about how to handle
107      * the failure type.
108      * @throws FileCreationException if the marker file can't be created.
109      */
110     public void createMarkerFile(RestoreRequired.FailureType failureType,
111                                  Properties props)
112         throws FileCreationException {
113 
114         /* Don't overwrite the file if it already exists. */
115         if (lastFile.exists() && lastFile.length() > 0) {
116             return;
117         }
118 
119         try {
120             lastFile.createNewFile();
121     
122             /*
123              * The file will have two log entries:
124              * - a manufactured file header. Note that the file header usually
125              * has a previous offset that points to the previous log entry. In
126              * this case, it's set to 0 because we will never scan backwards
127              * from this file.
128              * - a RestoreRequired log entry
129              */
130             FileHeader header = new FileHeader(Integer.MAX_VALUE, 0);
131             LogEntry headerLogEntry =
132                     new FileHeaderEntry(LogEntryType.LOG_FILE_HEADER, header);
133             ByteBuffer buf1 = logManager.putIntoBuffer(headerLogEntry,
134                                                        0); // prevLogEntryOffset
135     
136             RestoreRequired rr = new RestoreRequired(failureType, props);
137     
138             LogEntry marker =
139                 SingleItemEntry.create(LogEntryType.LOG_RESTORE_REQUIRED, rr);
140             ByteBuffer buf2 = logManager.putIntoBuffer(marker, 0);
141 
142             try (FileOutputStream stream = new FileOutputStream(lastFile);
143                  FileChannel channel = stream.getChannel()) {
144                 channel.write(buf1);
145                 channel.write(buf2);
146             } catch (IOException e) {
147                 /* the stream and channel will be closed */
148                 throw e;
149             }
150         } catch (IOException ioe) {
151             throw new FileCreationException(
152                 "Marker file creation failed for: " + failureType +
153                 " " + ioe.toString(), ioe);
154         }
155     }
156 }
