23 /**
24  * Tracks the number of operations begun, as a way of measuring level of
25  * activity. Is capable of displaying thread dumps if the activity level
26  * reaches a specified ceiling
27  */
28 public class ActivityCounter {
29 
30     private final AtomicInteger activeCount;
31     private final AtomicBoolean threadDumpInProgress;
32     private volatile long lastThreadDumpTime;
33     private volatile int numCompletedDumps;
34     private final int activeThreshold;
35     private final int  maxNumDumps;
36     private final AtomicInteger maxActivity;
37 
38     /*
39      * Thread dumps can only happen this many milliseconds apart, to avoid
40      * overwhelming the system.
41      */
42     private final long requiredIntervalMillis;
43 
44     private final Logger logger;
45 
46     private final ExecutorService dumper;
47 
48     public ActivityCounter(int activeThreshold,
49                            long requiredIntervalMillis,
50                            int maxNumDumps,
51                            Logger logger) {
52 
53         activeCount = new AtomicInteger(0);
54         threadDumpInProgress = new AtomicBoolean(false);
55         maxActivity = new AtomicInteger(0);
56 
57         this.activeThreshold = activeThreshold;
58         this.requiredIntervalMillis = requiredIntervalMillis;
59         this.maxNumDumps = maxNumDumps;
60         this.logger = logger;
61 
62         dumper = Executors.newSingleThreadExecutor();
63     }
64 
65     /* An operation has started. */
66     public void start() {
67         int numActive = activeCount.incrementAndGet();
68         int max = maxActivity.get();
69         if (numActive > max) {
70             maxActivity.compareAndSet(max, numActive);
71         }
72         check(numActive);
73     }
74 
75     /* An operation has finished. */
76     public void finish() {
77         activeCount.decrementAndGet();
78     }
79 
80     public int getAndClearMaxActivity() {
81         return maxActivity.getAndSet(0);
82     }
83 
84     private boolean intervalIsTooShort() {
85         /* Don't do a thread dump if the last dump was too recent */
86         long interval = System.currentTimeMillis() - lastThreadDumpTime;
87         return interval < requiredIntervalMillis;
88     }
89 
90     /**
91      * If the activity level is above a threshold, there is no other thread
92      * that is dumping now, and a dump hasn't happened for a while, dump
93      * thread stack traces.
94      */
95     private void check(int numActive) {
96 
97         /* Activity is low, no need to do any dumps. */
98         if (numActive <= activeThreshold) {
99             return;
100         }
101 
102         if (numCompletedDumps >= maxNumDumps) {
103             return;
104         }
105 
106         /* Don't do a thread dump if the last dump was too recent */
107         if (intervalIsTooShort()) {
108             return;
109         }
110 
111         /* There's one in progress. */
112         if (threadDumpInProgress.get()) {
113             return;
114         }
115 
116         /*
117          * Let's do a dump. The ExecutorServices guarantees that all activity
118          * executes in a single thread, so further serialization and
119          * synchronization is handled there.
120          */
121         dumper.execute(new GetStackTraces());
122     }
123 
124     /**
125      * For unit test support.
126      */
127     public int getNumCompletedDumps() {
128         return numCompletedDumps;
129     }
130 
131     private class GetStackTraces implements Runnable {
132 
133         public void run() {
134 
135             if (intervalIsTooShort()) {
136                 return;
137             }
138 
139             if (!threadDumpInProgress.compareAndSet(false, true)) {
140                 logger.warning("Unexpected: ActivityCounter stack trace " +
141                                "dumper saw threadDumpInProgress flag set.");
142                 return;
143             }
144 
145             try {
146                 lastThreadDumpTime = System.currentTimeMillis();
147                 dumpThreads();
148                 numCompletedDumps++;
149             } finally {
150                 boolean reset = threadDumpInProgress.compareAndSet(true, false);
151                 assert reset : "ThreadDump should have been in progress";
152             }
153         }
154 
155         private void dumpThreads() {
156 
157             int whichDump = numCompletedDumps;
158 
159             logger.info("[Dump " + whichDump +
160                         " --Dumping stack traces for all threads]");
161 
162             Map<Thread, StackTraceElement[]> stackTraces =
163                 Thread.getAllStackTraces();
164 
165             for (Map.Entry<Thread, StackTraceElement[]> stme :
166                      stackTraces.entrySet()) {
167                 if (stme.getKey() == Thread.currentThread()) {
168                     continue;
169                 }
170                 logger.info(stme.getKey().toString());
171                 for (StackTraceElement ste : stme.getValue()) {
172                     logger.info("     " + ste);
173                 }
174             }
175 
176             logger.info("[Dump " + whichDump + " --Thread dump completed]");
177         }
178     }
