26 /**
27  * A concrete MBean for monitoring a replicated open JE Environment.
28  *
29  * It not only has the same attributes and operations as the standalone

31  */

56          statParams, "java.lang.String", MBeanOperationInfo.INFO);

82     @Override
83     public Object invoke(String actionName,
84                          Object[] params,
85                          String[] signature)
86         throws MBeanException {
87 
88         if (actionName == null) {
89             throw new IllegalArgumentException("ActionName can't be null.");
90         }
91 
92         try {
93             if (actionName.equals(OP_DUMP_REPSTATS)) {
94                 return ((ReplicatedEnvironment) env).
95                     getRepStats(getStatsConfig(params)).toString();
96             } else if (actionName.equals(OP_GET_REP_TIPS)) {
97                 return ((ReplicatedEnvironment) env).getRepStats
98                     (getStatsConfig(new Object[] {false, true})).getTips();
99             } else if (actionName.equals(OP_DUMP_STATE)) {
100                 return RepInternal.getNonNullRepImpl
101                     ((ReplicatedEnvironment) env).dumpState();
102             }
103         } catch (DatabaseException e) {
104             throw new MBeanException(new RuntimeException(e.getMessage()));
105         }
106 
107         return super.invoke(actionName, params, signature);
108     }

115     }
