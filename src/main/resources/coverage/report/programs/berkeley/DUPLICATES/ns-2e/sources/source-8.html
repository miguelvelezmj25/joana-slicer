68     private void dump(File envHome,
69                      String entryTypes,
70                      String dbIds,
71                      String txnIds,
72                      long startLsn,
73                      long endLsn,
74                      boolean verbose,
75                      boolean stats,
76                      boolean repEntriesOnly,
77                      boolean csvFormat,
78                      boolean forwards,
79                      boolean vlsnDistribution,
80                      String customDumpReaderClass) {
81         EnvironmentImpl env =
82             CmdUtil.makeUtilityEnvironment(envHome, true);
83         FileManager fileManager = env.getFileManager();
84         fileManager.setIncludeDeletedFiles(true);
85         int readBufferSize =
86             env.getConfigManager().getInt
87             (EnvironmentParams.LOG_ITERATOR_READ_SIZE);
88 
89         /* Configure the startLsn and endOfFileLsn if reading backwards. */
90         long endOfFileLsn = DbLsn.NULL_LSN;
91         if (startLsn == DbLsn.NULL_LSN && endLsn == DbLsn.NULL_LSN &&
92             !forwards) {
93             LastFileReader fileReader =
94                 new LastFileReader(env, readBufferSize);
95             while (fileReader.readNextEntry()) {
96             }
97             startLsn = fileReader.getLastValidLsn();
98             endOfFileLsn = fileReader.getEndOfLog();
99         }
100 
101         try {
102 
103             /* 
104              * Make a reader. First see if a custom debug class is available,
105              * else use the default versions. 
106              */
107             DumpFileReader reader = null;
108             if (customDumpReaderClass != null) {
109 
110                 reader = getDebugReader(
111                     customDumpReaderClass, env, readBufferSize,
112                     startLsn, endLsn, endOfFileLsn, entryTypes, txnIds,
113                     verbose, repEntriesOnly, forwards);
114 
115             } else {
116                 if (stats) {
117 
118                     reader = new StatsFileReader(
119                         env, readBufferSize, startLsn, endLsn, endOfFileLsn,
120                         entryTypes, dbIds, txnIds, verbose, repEntriesOnly,
121                         forwards);
122 
123                 } else if (vlsnDistribution) {
124 
125                     reader = new VLSNDistributionReader(
126                         env, readBufferSize, startLsn, endLsn, endOfFileLsn,
127                         verbose, forwards);
128 
129                 } else {
130 
131                     reader = new PrintFileReader(
132                         env, readBufferSize, startLsn, endLsn, endOfFileLsn,
133                         entryTypes, dbIds, txnIds, verbose, repEntriesOnly,
134                         forwards);
135                 }
136             }
137 
138             /* Enclose the output in a tag to keep proper XML syntax. */
139             if (!csvFormat) {
140                 System.out.println("<DbPrintLog>");
141             }
142 
143             while (reader.readNextEntry()) {
144             }
145 
146             reader.summarize(csvFormat);
147             if (!csvFormat) {
148                 System.out.println("</DbPrintLog>");
149             }
150         } finally {
151             env.close();
152         }
153     }

270             }

318      * If a custom dump reader class is specified, we'll use that for 
319      * DbPrintLog instead of the regular DumpFileReader. The custom reader must
320      * have DumpFileReader as a superclass ancestor. Its constructor must have
321      * this signature:
322      *
323      *  public class FooReader extends DumpFileReader {
324      *
325      *      public FooReader(EnvironmentImpl env,
326      *                       Integer readBufferSize, 
327      *                       Long startLsn,
328      *                       Long finishLsn,
329      *                       Long endOfFileLsn,
330      *                       String entryTypes,
331      *                       String txnIds,
332      *                       Boolean verbose,
333      *                       Boolean repEntriesOnly,
334      *                       Boolean forwards) 
335      *          super(env, readBufferSize, startLsn, finishLsn, endOfFileLsn,
336      *                entryTypes, txnIds, verbose, repEntriesOnly, forwards);
337      *
338      * See com.sleepycat.je.util.TestDumper, on the test side, for an example.
339      */
340     private DumpFileReader getDebugReader(String customDumpReaderClass,
341                                           EnvironmentImpl env,
342                                           int readBufferSize,
343                                           long startLsn,
344                                           long finishLsn,
345                                           long endOfFileLsn,
346                                           String entryTypes,
347                                           String txnIds,
348                                           boolean verbose,
349                                           boolean repEntriesOnly,
350                                           boolean forwards) {
351         Class<?> debugClass = null;
352         try {
353             debugClass = Class.forName(customDumpReaderClass);
354         } catch (Exception e) {
355             throw new IllegalArgumentException
356                 ("-c was specified, but couldn't load " +
357                  customDumpReaderClass + " ", e);
358         }
359 
360         Class<?> args[] = { EnvironmentImpl.class,
361                             Integer.class,     // readBufferSize
362                             Long.class,        // startLsn
363                             Long.class,        // finishLsn
364                             Long.class,        // endOfFileLsn
365                             String.class,      // entryTypes
366                             String.class,      // txnIds
367                             Boolean.class,     // verbose
368                             Boolean.class,     // repEntriesOnly
369                             Boolean.class };   // forwards
370 
371         DumpFileReader debugReader = null;
372         try {
373             Constructor<?> con = 
374                 debugClass.getConstructor(args);
375             debugReader = (DumpFileReader) con.newInstance(env,
376                                                            readBufferSize,
377                                                            startLsn,
378                                                            finishLsn,
379                                                            endOfFileLsn,
380                                                            entryTypes,
381                                                            txnIds,
382                                                            verbose,
383                                                            repEntriesOnly,
384                                                            forwards);
385         } catch (Exception e) {
386             throw new IllegalStateException
387                 ("-c was specified, but couldn't instantiate " + 
388                  customDumpReaderClass + " ", e);
389         }
390 
391         return debugReader;
392     }
