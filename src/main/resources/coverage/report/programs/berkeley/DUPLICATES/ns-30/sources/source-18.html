16 import java.util.logging.Formatter;

20 /**
21  * Redirect the ConsoleHandler to use a specific Formatter. This is the
22  * same redirect approach used in
23  * com.sleepycat.je.utilint.ConsoleRedirectHandler, but in this case, an
24  * environment (and its associated stored console handler) is not available.
25  * In order to still have prefixed logging output, we incur the higher level
26  * cost of resetting the formatter.
27  */
28 public class FormatterRedirectHandler 
29     extends java.util.logging.ConsoleHandler {
30     
31     /*
32      * We want console logging to be determined by the level for
33      * com.sleepycat.je.util.ConsoleHandler. Check that handler's level and use
34      * it to set FormatterRedirectHandler explicitly.
35      */
36     private static final String HANDLER_LEVEL = 
37         com.sleepycat.je.util.ConsoleHandler.class.getName() + ".level";
38 
39     public FormatterRedirectHandler() {
40         super();
41         
42         String level = LoggerUtils.getLoggerProperty(HANDLER_LEVEL);
43         setLevel((level == null) ? Level.OFF : Level.parse(level));
44     }
45 
46     @Override
47     public void publish(LogRecord record) {
48         Formatter formatter = 
49             LoggerUtils.formatterMap.get(Thread.currentThread());
50         if (formatter != null) {
51             setFormatter(formatter);
52         }
53         super.publish(record);
54     }
55 }
