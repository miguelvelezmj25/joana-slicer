24 /**
25  * A concrete <code>EntryBinding</code> that treats a key or data entry as
26  * a serialized object.
27  *
28  * <p>This binding stores objects in serialized object format.  The
29  * deserialized objects are returned by the binding, and their
30  * <code>Class</code> must implement the <code>Serializable</code>
31  * interface.</p>
32  *
33  * <p>For key bindings, a tuple binding is usually a better choice than a
34  * serial binding.  A tuple binding gives a reasonable sort order, and works
35  * with comparators in all cases -- see below.</p>
36  *
37  * <p><em>WARNING:</em> SerialBinding should not be used with Berkeley DB Java
38  * Edition for key bindings, when a custom comparator is used.  In JE,
39  * comparators are instantiated and called internally at times when databases
40  * are not accessible.  Because serial bindings depend on the class catalog
41  * database, a serial binding cannot be used during these times.  An attempt
42  * to use a serial binding with a custom comparator will result in a
43  * NullPointerException during environment open or close.</p>
44  *
45  * <p><a name="evolution"><strong>Class Evolution</strong></a></p>
46  *
47  * <p>{@code SerialBinding} and other classes in this package use standard Java
48  * serialization and all rules of Java serialization apply.  This includes the
49  * rules for class evolution.  Once an instance of a class is stored, the class
50  * must maintain its {@code serialVersionUID} and follow the rules defined in
51  * the Java specification.  To use a new incompatible version of a class, a
52  * different {@link ClassCatalog} must be used or the class catalog database
53  * must be truncated.</p>
54  *
55  * <p>If more advanced class evolution features are required, consider using
56  * the {@link com.sleepycat.persist.evolve Direct Persistence Layer}.</p>
57  *
58  * @author Mark Hayes
59  */

96      * Returns the class loader to be used during deserialization, or null if a

102      *

104      * class loader.  For example, <code>getBaseClass().getClassLoader()</code>

123     /**
124      * Deserialize an object from an entry buffer.  May only be called for data
125      * that was serialized using {@link #objectToEntry}, since the fixed
126      * serialization header is assumed to not be included in the input data.
127      * {@link SerialInput} is used to deserialize the object.
128      *
129      * @param entry is the input serialized entry.
130      *
131      * @return the output deserialized object.
132      */
133     public E entryToObject(DatabaseEntry entry) {
134 
135         int length = entry.getSize();
136         byte[] hdr = SerialOutput.getStreamHeader();
137         byte[] bufWithHeader = new byte[length + hdr.length];
138 
139         System.arraycopy(hdr, 0, bufWithHeader, 0, hdr.length);
140         System.arraycopy(entry.getData(), entry.getOffset(),
141                          bufWithHeader, hdr.length, length);
142 
143         try {
144             SerialInput jin = new SerialInput(
145                 new FastInputStream(bufWithHeader, 0, bufWithHeader.length),
146                 classCatalog,
147                 getClassLoader());
148             return (E) jin.readObject();
149         } catch (IOException e) {
150             throw RuntimeExceptionWrapper.wrapIfNeeded(e);
151         } catch (ClassNotFoundException e) {
152             throw RuntimeExceptionWrapper.wrapIfNeeded(e);
153         }
154     }

156     /**
157      * Serialize an object into an entry buffer.  The fixed serialization
158      * header is not included in the output data to save space, and therefore
159      * to deserialize the data the complementary {@link #entryToObject} method
160      * must be used.  {@link SerialOutput} is used to serialize the object.
161      *
162      * <p>Note that this method sets the DatabaseEntry offset property to a
163      * non-zero value and the size property to a value less than the length of
164      * the byte array.</p>
165      *
166      * @param object is the input deserialized object.
167      *
168      * @param entry is the output serialized entry.
169      *
170      * @throws IllegalArgumentException if the object is not an instance of the
171      * base class for this binding, including if the object is null and a
172      * non-null base class was specified.
173      */
174     public void objectToEntry(E object, DatabaseEntry entry) {
175 
176         if (baseClass != null && !baseClass.isInstance(object)) {
177             throw new IllegalArgumentException
178                 (((object != null) ?
179                   ("Data object class (" + object.getClass() + ')') :
180                   "Null value") +
181                  " is not an instance of binding's base class (" +
182                  baseClass + ')');
183         }
184         FastOutputStream fo = getSerialOutput(object);
185         try {
186             SerialOutput jos = new SerialOutput(fo, classCatalog);
187             jos.writeObject(object);
188         } catch (IOException e) {
189             throw RuntimeExceptionWrapper.wrapIfNeeded(e);
190         }
191 
192         byte[] hdr = SerialOutput.getStreamHeader();
193         entry.setData(fo.getBufferBytes(), hdr.length,
194                      fo.getBufferLength() - hdr.length);
195     }
