148     public void run() {
149 
150         LoggerUtils.info(logger, envImpl,
151                          "Start subscription from VLSN " + reqVLSN +
152                          " from feeder at " +
153                          config.getFeederHost() + ":" + config.getFeederPort());
154 
155         try {
156             final int maxRetry = config.getMaxConnectRetries();
157             boolean auxThreadCreated = false;
158             int numRetry = 0;
159 
160             while (!isShutdown()) {
161                 try {
162                     initializeConnection();
163                     if (!auxThreadCreated) {
164                         LoggerUtils.fine(logger, envImpl,
165                                          "Create auxiliary msg processing " +
166                                          "and output threads");
167 
168                         auxThreadCreated = createAuxThread();
169                         if (auxThreadCreated) {
170                             /* subscription succeed, start streaming data */
171                             status = SubscriptionStatus.SUCCESS;
172                             loopInternal();
173                         } else {
174                             status = SubscriptionStatus.UNKNOWN_ERROR;
175                         }
176                     }
177                     break;
178                 } catch (ConnectionException e) {
179                     if (numRetry == maxRetry) {
180                         LoggerUtils.info(logger, envImpl,
181                                          "Reaching the max retry " + maxRetry +
182                                          " to connect feeder " +
183                                          config.getFeederHost() +
184                                          ", shut down subscription" +
185                                          "\n" + LoggerUtils.getStackTrace(e));
186                         storedException = e;
187                         status = SubscriptionStatus.CONNECTION_ERROR;
188                         break;
189                     } else {
190                         numRetry++;
191                         LoggerUtils.fine(logger, envImpl,
192                                          "Fail to connect feeder at " +
193                                          config.getFeederHost() +
194                                          " sleep for " + e.getRetrySleepMs() +
195                                          " ms and re-connect again");
196                         Thread.sleep(e.getRetrySleepMs());
197                     }
198                 }
199             }
200         } catch (ReplicationSecurityException ure) {
201             storedException = ure;
202             LoggerUtils.warning(logger, envImpl,
203                                 "Subscription exited due to security check " +
204                                 "failure: " + ure.getMessage());
205             status = SubscriptionStatus.SECURITY_CHECK_ERROR;
206         } catch (GroupShutdownException e) {
207             if (messageProcThread.isAlive()) {
208                 try {
209                     /* let message processing thread finish up */
210                     messageProcThread.join();
211                 } catch (InterruptedException ie) {
212                     /* ignore since we will shut down, just log */
213                     LoggerUtils.fine(logger, envImpl,
214                                      "exception in shutting down msg proc " +
215                                      "thread " + ie.getMessage() +
216                                      "\n" + LoggerUtils.getStackTrace(ie));
217                 }
218             }
219             storedException = e;
220             LoggerUtils.info(logger, envImpl,
221                              "received group shutdown " + e.getMessage() +
222                              "\n" + LoggerUtils.getStackTrace(e));
223             status = SubscriptionStatus.GRP_SHUTDOWN;
224         } catch (InsufficientLogException e) {
225             storedException = e;
226             LoggerUtils.info(logger, envImpl,
227                              "unable to subscribe from requested VLSN " +
228                              reqVLSN +
229                              "\n" + LoggerUtils.getStackTrace(e));
230             status = SubscriptionStatus.VLSN_NOT_AVAILABLE;
231         } catch (EnvironmentFailureException e) {
232             storedException = e;
233             LoggerUtils.warning(logger, envImpl,
234                                 "unable to sync up with feeder due to EFE " +
235                                 e.getMessage() +
236                                 "\n" + LoggerUtils.getStackTrace(e));
237             status = SubscriptionStatus.UNKNOWN_ERROR;
238         } catch (InterruptedException e) {
239             storedException = e;
240             LoggerUtils.warning(logger, envImpl,
241                                 "interrupted exception " + e.getMessage() +
242                                 "\n" + LoggerUtils.getStackTrace(e));
243             status = SubscriptionStatus.UNKNOWN_ERROR;
244         } catch (InternalException e) {
245             storedException = e;
246             LoggerUtils.warning(logger, envImpl,
247                                 "internal exception " + e.getMessage() +
248                                 "\n" + LoggerUtils.getStackTrace(e));
249             status = SubscriptionStatus.UNKNOWN_ERROR;
250         } finally {
251             shutdown();
252         }
253     }

278     void shutdown() {
279 
280         /* Note start of shutdown and return if already requested */
281         if (shutdownDone(logger)) {
282             return;
283         }
284 
285         /* shutdown aux threads */
286         if (messageProcThread != null) {
287             try {
288                 messageProcThread.shutdownThread(logger);
289                 LoggerUtils.info(logger, envImpl,
290                                  "message processing thread has shut down.");
291             } catch (Exception e) {
292                 /* Ignore so shutdown can continue */
293                 LoggerUtils.warning(logger, envImpl,
294                                     "error in shutdown msg proc thread: " +
295                                     e.getMessage() + ", continue shutdown the" +
296                                     " subscription thread.");
297             } finally {
298                 messageProcThread = null;
299             }
300         }
301         if (outputThread != null) {
302             try {
303                 outputThread.shutdownThread(logger);
304                 LoggerUtils.info(logger, envImpl,
305                                  "output thread has shut down.");
306 
307             } catch (Exception e) {
308                 /* Ignore we will clean up via killing IO channel anyway. */
309                 LoggerUtils.warning(logger, envImpl,
310                                     "error in shutdown output thread: " +
311                                     e.getMessage() + ", continue shutdown " +
312                                     "subscription thread.");
313             } finally {
314                 outputThread = null;
315             }
316         }
317 
318         inputQueue.clear();
319         outputQueue.clear();
320         RepUtils.shutdownChannel(namedChannel);
321         if (channelTimeoutTask != null) {
322             channelTimeoutTask.cancel();
323         }
324 
325         shutdownThread(logger);
326 
327         LoggerUtils.info(logger, envImpl,
328                          "queues cleared and channel closed, subscription " +
329                          "thread has completely shut down");
330     }

342

344

346                !inputQueue.offer(message,
347                                  SubscriptionConfig.QUEUE_POLL_INTERVAL_MS,
348                                  TimeUnit.MILLISECONDS)) {
349             /* Offer timed out. */
350             if (!messageProcThread.isAlive()) {
351                 LoggerUtils.info(logger, repImpl,
352                                  "Thread consuming input queue is gone, start" +
353                                  " shutdown process");
354                 throw new GroupShutdownException(logger, repImpl,
355                                                  config.getFeederHost(),
356                                                  stats.getHighVLSN(), 0);
357             } else {
358                 /* count the overflow and retry */
359                 stats.getNumReplayQueueOverflow().increment();
360             }
361         }

372      * @throws ReplicationSecurityException if authentication failure

375         EnvironmentFailureException, ConnectionException,
376         ReplicationSecurityException {
377 
378         /* open a channel to feeder */
379         LoggerUtils.fine(logger, envImpl,
380                          "Subscription " + config.getSubNodeName() +
381                          " start open channel and handshake with feeder");
382 
383         try {
384

386             ReplicaFeederHandshake handshake =
387                 new ReplicaFeederHandshake(new SubFeederHandshakeConfig

389

391 
392             /* check if negociated protocol version is high enough */
393             final int minReqVersion = config.getMinProtocolVersion();
394             if (protocol.getVersion() < minReqVersion) {
395                 throw new BinaryProtocol.ProtocolException(
396                     "HA protocol version (" + protocol.getVersion() + ") is " +
397                     "lower than minimal required version (" + minReqVersion +
398                     ")");
399             }

402                              "subscription " + config.getSubNodeName() +

408                                            config.getStreamMode(),

411             LoggerUtils.fine(logger, envImpl,
412                              "sync-up with feeder done, start vlsn: " +
413                              startVLSN);
414

416 
417                 stats.setStartVLSN(startVLSN);
418 
419                 /* read heartbeat and respond */
420                 protocol.read(namedChannel.getChannel(),
421                               Protocol.Heartbeat.class);
422                 queueAck(ReplicaOutputThread.HEARTBEAT_ACK);
423 
424                 LoggerUtils.info(logger, envImpl,
425                                  "Subscription " + config.getSubNodeName() +
426                                  " successfully connect to feeder at " +
427                                  config.getFeederHost() + ":" +
428                                  config.getFeederPort() +
429                                  ", reqVLSN: " + reqVLSN +
430                                  ", start VLSN: " + startVLSN);
431             } else {
432                 throw new InsufficientLogException((RepImpl) envImpl, reqVLSN);
433             }
434         } catch (IOException e) {
435             throw new ConnectionException("Unable to connect due to " +
436                                           e.getMessage() +
437                                           ",  will retry later.",
438                                           config.getSleepBeforeRetryMs(),
439                                           e);
440         } catch (EnvironmentFailureException e) {
441             logger.warning("Fail to handshake with feeder: " +
442                            e.getMessage());
443             throw e;
444         } catch (BinaryProtocol.ProtocolException e) {
445             final String msg = ("Unable to connect to feeder " +
446                                 config.getFeederHost() +
447                                 " due to protocol exception " +
448                                 e.getMessage());
449             LoggerUtils.warning(logger, envImpl, msg);
450             throw new InternalException(msg, e);
451         }
452     }

511

513

516         }
517

520                 new DataChannelFactory

522                     .setTcpNoDelay(config.TCP_NO_DELAY)
523                     .setReceiveBufferSize(config.getReceiveBufferSize())
524                     .setOpenTimeout((int) config

526                     .setBlocking(config.BLOCKING_MODE_CHANNEL);
527 
528             final DataChannel channel =
529                 repImpl.getChannelFactory()

531

536                              "channel opened to service " +
537                              FeederManager.FEEDER_SERVICE + "@" +
538                              config.getFeederHost() +
539                              "[address: " + config.getFeederHostAddr() +
540                              " port: " + config.getFeederPort() + "]");
541

543                 getDuration(RepParams.PRE_HEARTBEAT_TIMEOUT);
544

546             namedChannel =
547                 new NamedChannelWithTimeout(repImpl, logger, channelTimeoutTask,
548                                             channel, timeoutMs);
549         } catch (IOException cause) {
550             /* retry if unable to connect to feeder */
551             throw new ConnectionException("Fail to open channel to feeder " +

556         } catch (ServiceDispatcher.ServiceConnectFailedException cause) {

560              * as yet. For example, the transition to the master may not have
561              * been completed.
562              */
563             if (cause.getResponse() ==
564                 ServiceDispatcher.Response.UNKNOWN_SERVICE) {
565                 throw new ConnectionException("Service exception: " +
566                                               cause.getMessage() +
567                                               ", wait longer and will retry " +
568                                               "later",
569                                               config.getSleepBeforeRetryMs(),
570                                               cause);
571             }
572

575                 throw new ReplicationSecurityException(

577                     config.getSubNodeName(),
578                     cause);
579             }

583                                         "failed to handshake for service " +
584                                         FeederManager.FEEDER_SERVICE +
585                                         " with feeder " +
586                                         config.getFeederHost(),
587                                         cause);
588         }
589         LoggerUtils.info(logger, envImpl,
590                          "Subscription " + config.getSubNodeName() +
591                          " has successfully created a channel to feeder at " +
592                          config.getFeederHost() + ":" + config.getFeederPort());
593

595     }

597     /**
598      * Internal loop to dequeue message from channel to the feeder,
599      * process shutdown and heartbeat messages, and relay data operations to
600      * the input queue to be consumed by input thread.
601      *
602      * @throws InternalException if error in reading messages from channel or
603      *                           enqueue message into input queue
604      * @throws GroupShutdownException if receive shutdown message from feeder
605      * @throws ReplicationSecurityException if output thread exits due to
606      * security check failure. In this case the main subscription need to
607      * exit without retry.
608      */
609     private void loopInternal() throws InternalException,
610         GroupShutdownException, ReplicationSecurityException {
611 
612         RepImpl repImpl = (RepImpl)envImpl;
613 
614         try {
615 
616             LoggerUtils.info(logger, envImpl,
617                              "Start reading messages from feeder " +
618                              config.getFeederHost() + ":" +
619                              config.getFeederPort());
620             while (!isShutdown()) {
621 
622                 checkOutputThread();
623 
624                 BinaryProtocol.Message message = protocol.read(namedChannel);
625 
626                 if ((message == null)) {
627                     LoggerUtils.info(logger, envImpl,
628                                      "Subscription " + config.getSubNodeName() +
629                                      " has nothing stream, exit loop.");
630                     return;
631                 }
632 
633                 assert TestHookExecute.doHookIfSet(exceptionHandlingTestHook,
634                                                    this);
635 
636                 stats.getNumMsgReceived().increment();
637 
638                 BinaryProtocol.MessageOp messageOp = message.getOp();
639                 if (messageOp == Protocol.HEARTBEAT) {
640                     LoggerUtils.finest(logger, envImpl,
641                                        "receive heartbeat from " +
642                                        namedChannel.getNameIdPair());
643                     queueAck(ReplicaOutputThread.HEARTBEAT_ACK);
644                 } else if (messageOp == Protocol.SHUTDOWN_REQUEST) {
645 
646                     LoggerUtils.info(logger, envImpl,
647                                      "Receive shutdown request from feeder " +
648                                      config.getFeederHost() +
649                                      ", shutdown subscriber");
650                     
651                     /*
652                      * create a shutdown request, make it in the queue so
653                      * client is able to see that in callback, and throw an
654                      * exception.
655                      *
656                      * The message processing thread will exit when seeing a
657                      * GroupShutdownException
658                      */
659                     Protocol.ShutdownRequest req =
660                         (Protocol.ShutdownRequest) message;
661                     Exception exp =
662                         new GroupShutdownException(logger, repImpl,
663                                                    config.getFeederHost(),
664                                                    stats.getHighVLSN(),
665                                                    req.getShutdownTimeMs());
666                     offer(exp);
667                     throw exp;
668                 } else {
669                     /* a regular data entry message */
670                     offer(message);
671 
672                     final long pending = inputQueue.size();
673                     if (pending > stats.getMaxPendingInput().get()) {
674                         stats.getMaxPendingInput().set(pending);
675                         LoggerUtils.finest(logger, envImpl,
676                                            "Max pending request log items:" +
677                                            pending);
678                     }
679                 }
680 
681             }
682         } catch (GroupShutdownException | ReplicationSecurityException exp) {
683             /* throw to caller, let caller deal with it */
684             throw exp;
685         } catch (Exception e) {
686             /* other exception is thrown as IE */
687             throw new InternalException(e.getMessage(), e);
688         }
689     }

691     /**
692      * Checks status of output thread and propagates RSE to main
693      * loop. If output thread exited due to RSE, the main thread need to
694      * capture it to set the subscription status correctly. For other
695      * exceptions, output thread uses the traditional mechanism to notify the
696      * main subscription thread: simply shut down channel.
697      */
698     private void checkOutputThread()
699         throws InternalException, ReplicationSecurityException {
700 
701         /*
702          * output thread can be shutdown and set to null anytime, thus
703          * use a cached copy to avoid NPE after the first check
704          */
705         final SubscriptionOutputThread cachedOutputThread = outputThread;
706 
707         /* output thread already gone */
708         if (cachedOutputThread == null) {
709             /*
710              * if output thread is set to null only when subscription thread
711              * shut down. If we reach here, it means the subscription thread
712              * is shut down right after isShutdown check in loopInternal().
713              * We simply return and subscription thread would detect the shut
714              * down in next check of isShutdown in loopInternal().
715              */
716             LoggerUtils.fine(logger, envImpl,
717                              "output thread no longer exists");
718             return;
719         }
720 
721         if (cachedOutputThread.getException() instanceof
722                 ReplicationSecurityException) {
723             final ReplicationSecurityException rse =
724                 (ReplicationSecurityException) cachedOutputThread.getException();
725             LoggerUtils.warning(logger, envImpl,
726                                 "Output thread exited due to security check " +
727                                 "failure: " + rse.getMessage());
728             throw rse;
729         }
730     }

732     /**
733      * Enqueue an ack message in output queue
734      *
735      * @param xid txn id to enqueue
736      *
737      * @throws IOException if fail to queue the msg
738      */
739     private void queueAck(Long xid) throws IOException {
740 
741         try {
742             outputQueue.put(xid);
743         } catch (InterruptedException ie) {
744 
745             /*
746              * If interrupted while waiting, have the higher levels treat
747              * it like an IOE and exit the thread.
748              */
749             throw new IOException("Ack I/O interrupted", ie);
750         }
751     }
