37 /**
38  * TxnChain supports "txn rollback", which undoes the write operations for a
39  * given txn to an arbitrary point. Txn rollback (and TxnChain construction)
40  * is done in 2 occasions:
41  * 1. During normal processing, when an ongoing txn must be rolled-back due to
42  *    a syncup operation (see rep/txn/ReplayTxn.java).
43  * 2. During recovery, to process a "rollback period" (see RollbackTracker.java)
44  *
45  * In the JE log, the logrecs that make up a txn are chained, but each logrec
46  * contains undo info that refers to the pre-txn version of the associated
47  * record, which may not be the immediately previous version, if the txn writes
48  * the same record multiple times. For example, a log looks like this:
49  *
50  * lsn       key   data         abortlsn
51  * 100       A      10          null_lsn (first instance of record A)
52  * 150       B      100         null_lsn (first instance of record B)
53  *  .....  txn begins .....
54  * 200       A      20          100
55  * 300       A    deleted       100
56  * 400       B     200          150
57  * 500       A      30          100
58  * 600       C      10          null_lsn
59  *
60  * When reading the log, we can find all the records in the transaction. This
61  * chain exists:
62  * 500->400->300->200->null_lsn
63  *
64  * To rollback to an arbitrary entry in the transaction, we need a chain of all
65  * the records that occupied a given BIN slot during the transaction.
66  * chain. The key, data, and comparators are used to determine which records
67  * hash to the same slot, mimicking the btree itself.
68  *
69  *
70  *   300      400      500     600
71  *    |        |        |       |
72  *   \ /      \ /      \ /     \ /
73  *    
74  *   200      150      300    null_lsn   revertToLsn
75  *                             true       revertKD
76  */

118     /*
119      * Find the previous version for all entries in this transaction.
120      * DatabaseImpls used during txn chain creation are taken from the
121      * transaction's undoDatabases cache.
122      */
123     public TxnChain(
124         long lastLoggedLsn,
125         long txnId,
126         long matchpoint,
127         Map<DatabaseId, DatabaseImpl> undoDatabases,
128         EnvironmentImpl envImpl)
129         throws DatabaseException {
130 
131         LogManager logManager = envImpl.getLogManager();
132 
133         this.envImpl = envImpl;
134         this.undoDatabases = undoDatabases;
135 
136         remainingLockedNodes = new HashSet<Long>();
137 
138         /*
139          * A map that stores for each record R the revert info for the
140          * latest R-logrec seen during the backwards traversal of the
141          * txn chain done below.
142          */
143         TreeMap<CompareSlot, RevertInfo> recordsMap =
144             new TreeMap<CompareSlot, RevertInfo>();
145 
146         revertList = new LinkedList<RevertInfo>();
147 
148         /*
149          * Traverse this txn's entire logrec chain and record revert info
150          * for each logrec in the chain. Start the traversal with the last
151          * logrec generated by this txn and move backwards.
152          */
153         long currLsn = lastLoggedLsn;
154 
155         try {
156             lastValidVLSN = VLSN.NULL_VLSN;
157 
158             while (currLsn != DbLsn.NULL_LSN) {
159 
160                 WholeEntry wholeEntry =
161                     logManager.getLogEntryAllowInvisible(currLsn);
162 
163                 LNLogEntry<?> currLogrec =
164                     (LNLogEntry<?>) wholeEntry.getEntry();
165 
166                 DatabaseImpl dbImpl = getDatabaseImpl(currLogrec.getDbId());
167 
168                 if (dbImpl == null) {
169 
170                     if (undoDatabases != null) {
171                         throw EnvironmentFailureException.unexpectedState(
172                             envImpl, // fatal error, this is a corruption
173                             "DB missing during non-recovery rollback, dbId=" +
174                             currLogrec.getDbId() + " txnId=" + txnId);
175                     }
176 
177                     /*
178                      * For recovery rollback, simply skip this entry when the
179                      * DB has been deleted.  This has no impact on the chain
180                      * for other LNs.  This LN will not be processed by
181                      * recovery rollback, since RollbackTracker.rollback
182                      * ignores LNs in deleted DBs.  [#22071] [#22052]
183                      */
184                     currLsn = currLogrec.getUserTxn().getLastLsn();
185                     continue;
186                 }
187 
188                 currLogrec.postFetchInit(dbImpl);
189 
190                 try {
191                     /*
192                      * Let L be the current logrec, and let R and T be the
193                      * record and txn associated with L. If T wrote R again
194                      * after L, let Ln be the 1st R-logrec by T after L. 
195                      */
196                     CompareSlot recId = new CompareSlot(dbImpl, currLogrec);
197 
198                     RevertInfo ri = recordsMap.get(recId);
199 
200                     /*
201                      * If Ln exists, update the RevertInfo created earlier for
202                      * Ln so that it now refers to the L version of R.
203                       */
204                     if (ri != null) {
205                         ri.revertLsn = currLsn;
206                         ri.revertKD = false;
207                         ri.revertPD = currLogrec.isDeleted();
208 
209                         ri.revertKey = 
210                             (dbImpl.allowsKeyUpdates() ?
211                              currLogrec.getKey() : null);
212 
213                         ri.revertData =
214                             (currLogrec.isEmbeddedLN() ?
215                              currLogrec.getData() : null);
216 
217                         ri.revertVLSN =
218                             (currLogrec.isEmbeddedLN() ?
219                              currLogrec.getLN().getVLSNSequence() :
220                              VLSN.NULL_VLSN_SEQUENCE);
221 
222                         ri.revertExpiration = currLogrec.getExpiration();
223 
224                         ri.revertExpirationInHours =
225                             currLogrec.isExpirationInHours();
226                     }
227 
228                     /*
229                      * If L will be rolled back, assume that it is the 1st
230                      * R-logrec by T and thus set its revert info to refer
231                      * to the pre-T version of R. 
232                      */
233                     if (DbLsn.compareTo(currLsn, matchpoint) > 0) {
234 
235                         ri = new RevertInfo(
236                             currLogrec.getAbortLsn(),
237                             currLogrec.getAbortKnownDeleted(),
238                             currLogrec.getAbortKey(),
239                             currLogrec.getAbortData(),
240                             currLogrec.getAbortVLSN(),
241                             currLogrec.getAbortExpiration(),
242                             currLogrec.isAbortExpirationInHours());
243 
244                         revertList.add(ri);
245                         recordsMap.put(recId, ri);
246 
247                     } else {
248 
249                         /*
250                          * We are done with record R, so remove it from the
251                          * map, if it is still there.
252                          */
253                         if (ri != null) {
254                             recordsMap.remove(recId);
255                         }
256 
257                         remainingLockedNodes.add(currLsn);
258 
259                         if (lastValidVLSN != null &&
260                             lastValidVLSN.isNull() &&
261                             wholeEntry.getHeader().getVLSN() != null &&
262                             !wholeEntry.getHeader().getVLSN().isNull()) {
263 
264                             lastValidVLSN = wholeEntry.getHeader().getVLSN();
265                         }
266                     }
267 
268                     /* Move on to the previous logrec for this txn. */
269                     currLsn = currLogrec.getUserTxn().getLastLsn();
270 
271                 } finally {
272                     releaseDatabaseImpl(dbImpl);
273                 }
274             }
275         } catch (FileNotFoundException e) {
276             throw EnvironmentFailureException.promote(
277                 envImpl, EnvironmentFailureReason.LOG_INTEGRITY,
278                 "Problem finding intermediates for txn " + txnId +
279                 " at lsn " + DbLsn.getNoFormatString(currLsn), e);
280         }
281     }

384      * Compare two keys using the appropriate comparator. Keys from different
385      * databases should never be equal.
386      */
387     public static class CompareSlot implements Comparable<CompareSlot> {
388 
389         private final DatabaseImpl dbImpl;
390         private final byte[] key;
391 
392         public CompareSlot(DatabaseImpl dbImpl, LNLogEntry<?> undoEntry) {
393             this(dbImpl, undoEntry.getKey());
394         }
395 
396         private CompareSlot(DatabaseImpl dbImpl, byte[] key) {
397             this.dbImpl = dbImpl;
398             this.key = key;
399         }
400 
401         public int compareTo(CompareSlot other) {
402             int dbCompare = dbImpl.getId().compareTo(other.dbImpl.getId());
403             if (dbCompare != 0) {
404                 /* LNs are from different databases. */
405                 return dbCompare;
406             }
407 
408             /* Compare keys. */
409             return Key.compareKeys(key, other.key, dbImpl.getKeyComparator());
410         }
411 
412         @Override
413         public boolean equals(Object other) {
414             if (!(other instanceof CompareSlot)) {
415                 return false;
416             }
417             return compareTo((CompareSlot) other) == 0;
418         }
419 
420         @Override
421         public int hashCode() {
422 
423             /*
424              * Disallow use of HashSet/HashMap/etc.  TreeSet/TreeMap/etc should
425              * be used instead when a CompareSlot is used as a key.
426              *
427              * Because a comparator may be configured that compares only a part
428              * of the key, a hash code cannot take into account the key or
429              * data, because hashCode() must return the same value for two
430              * objects whenever equals() returns true.  We could hash the DB ID
431              * alone, but that would not produce an efficient hash table.
432              */
433             throw EnvironmentFailureException.unexpectedState
434                 ("Hashing not supported");
435         }
