15 import java.lang.ref.SoftReference;
16 import java.util.concurrent.atomic.AtomicReference;
17 
18 import com.sleepycat.je.log.LogItem;
19 import com.sleepycat.je.utilint.LongStat;
20 import com.sleepycat.je.utilint.StatGroup;

23 /**
24  * A no-wait cache used to retain the most recent VLSNs. The Feeders check this
25  * cache first for log entries to send out to the Replicas. Feeders that are
26  * feeding at the most up to date portion of the replication stream will likely
27  * hit in the cache, preventing a lookup in the log buffers or log files.
28  *
29  * The log item cache is made up of weak references so there is never any
30  * guarantee that even the most recent 32 entries are in there.
31  */
32 class LogItemCache {
33 
34     /*
35      * Must be a power of 2 and small, typically around 32 entries. Increasing
36      * sizes typically yield diminishing returns.
37      */
38     private final int cacheSize;
39     private final int sizeMask;
40 
41     /*
42      * Soft reference to array, so that the LogItems can be released when
43      * under GC pressure.
44      */
45     private volatile SoftReference<AtomicReference<LogItem>[]>
46         cacheReference =
47         new SoftReference<AtomicReference<LogItem>[]>(null);
48 
49     private final LongStat nHits;
50     private final LongStat nMisses;
51 
52     /**
53      * Creates a log item size of the specified size.
54      *
55      * @param cacheSize it must be a power of two
56      * @param statGroup the statsGroup to which this cache adds its stats
57      * @throws IllegalArgumentException via ReplicatedEnvironment ctor.
58      */
59     LogItemCache(int cacheSize, StatGroup statGroup) {
60         if (Integer.bitCount(cacheSize) != 1) {
61             throw new IllegalArgumentException
62                 ("Bad cache size: " + cacheSize + "; it must be a power of 2");
63         }
64         this.cacheSize = cacheSize;
65         sizeMask = cacheSize - 1;
66         nHits = new LongStat(statGroup, VLSNIndexStatDefinition.N_HITS);
67         nMisses = new LongStat(statGroup, VLSNIndexStatDefinition.N_MISSES);
68     }
69 
70     void put(VLSN vlsn, LogItem logItem) {
71         getArray()[(int)vlsn.getSequence() & sizeMask].set(logItem);
72     }
73 
74     LogItem get(VLSN vlsn) {
75         final LogItem item =
76             getArray()[(int)vlsn.getSequence() & sizeMask].get();
77         if ((item != null) && item.header.getVLSN().equals(vlsn)) {
78             nHits.increment();
79             return item;
80         }
81 
82         nMisses.increment();
83         return null;
84     }
85 
86     /**
87      * For explicit release of references.
88      */
89     void clear() {
90         for (AtomicReference<LogItem> element : getArray()) {
91             element.set(null);
92         }
93     }
94 
95     /**
96      * Returns the cache array, creating a new one, if the GC had cleared the
97      * reference to the earlier one.
98      * <p>
99      *
100      * Note that there may be a slight inefficiency if getArray is called
101      * concurrently, and it had been cleared earlier, since it would be
102      * allocated twice and introduce a cache miss. This occurrence is
103      * infrequent enough that it's not worth the overhead of a sync mechanism.
104      *
105      * @return the underlying array, allocating a new one, if the previous one
106      * had been GC'd
107      */
108     @SuppressWarnings("unchecked")
109     private final AtomicReference<LogItem>[] getArray() {
110         AtomicReference<LogItem>[] array = cacheReference.get();
111         if (array == null) {
112             array = new AtomicReference[cacheSize];
113             for (int i=0; i < array.length; i++) {
114                 array[i] = new AtomicReference<LogItem>();
115             }
116             cacheReference =
117                 new SoftReference<AtomicReference<LogItem>[]>(array);
118         }
119         return array;
120     }
121 }
