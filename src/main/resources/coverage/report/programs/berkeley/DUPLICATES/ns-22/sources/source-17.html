25 /**
26  * Format for log entries sent across the wire for replication. Instead of
27  * sending a direct copy of the log entry as it is stored on the JE log files
28  * (LogEntryHeader + LogEntry), select parts of the header are sent.
29  *
30  * @see InputWireRecord
31  * @see OutputWireRecord
32  */
33 abstract class WireRecord {
34 
35     final LogEntryHeader header;
36 
37     WireRecord(final LogEntryHeader header) {
38         this.header = header;
39     }
40 
41     /**
42      * Returns the log entry type for this record.
43      */
44     LogEntryType getLogEntryType()
45         throws DatabaseException {
46 
47         final LogEntryType type = LogEntryType.findType(header.getType());
48         if (type == null) {
49             throw EnvironmentFailureException.unexpectedState(
50                 "Unknown header type:" + header.getType());
51         }
52         return type;
53     }
54 
55     /**
56      * Instantiates the log entry for this wire record using the specified
57      * environment and data.
58      */
59     LogEntry instantiateEntry(final EnvironmentImpl envImpl,
60                               final ByteBuffer buffer)
61         throws DatabaseException {
62 
63         final LogEntry entry = getLogEntryType().getNewLogEntry();
64         buffer.mark();
65         entry.readEntry(envImpl, header, buffer);
66         buffer.reset();
67         return entry;
68     }
