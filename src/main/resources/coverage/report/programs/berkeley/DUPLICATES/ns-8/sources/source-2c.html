25 /**
26  * A specialized join cursor for use in performing equality or natural joins on
27  * secondary indices.
28  *
29  * <p>A join cursor is returned when calling {@link Database#join
30  * Database.join}.</p>
31  *
32  * <p>To open a join cursor using two secondary cursors:</p>
33  *
34  * <pre>
35  *     Transaction txn = ...
36  *     Database primaryDb = ...
37  *     SecondaryDatabase secondaryDb1 = ...
38  *     SecondaryDatabase secondaryDb2 = ...
39  *     <p>
40  *     SecondaryCursor cursor1 = null;
41  *     SecondaryCursor cursor2 = null;
42  *     JoinCursor joinCursor = null;
43  *     try {
44  *         DatabaseEntry key = new DatabaseEntry();
45  *         DatabaseEntry data = new DatabaseEntry();
46  *         <p>
47  *         cursor1 = secondaryDb1.openSecondaryCursor(txn, null);
48  *         cursor2 = secondaryDb2.openSecondaryCursor(txn, null);
49  *         <p>
50  *         key.setData(...); // initialize key for secondary index 1
51  *         OperationStatus status1 =
52  *         cursor1.getSearchKey(key, data, LockMode.DEFAULT);
53  *         key.setData(...); // initialize key for secondary index 2
54  *         OperationStatus status2 =
55  *         cursor2.getSearchKey(key, data, LockMode.DEFAULT);
56  *         <p>
57  *         if (status1 == OperationStatus.SUCCESS &amp;&amp;
58  *                 status2 == OperationStatus.SUCCESS) {
59  *             <p>
60  *             SecondaryCursor[] cursors = {cursor1, cursor2};
61  *             joinCursor = primaryDb.join(cursors, null);
62  *             <p>
63  *             while (true) {
64  *                 OperationStatus joinStatus = joinCursor.getNext(key, data,
65  *                     LockMode.DEFAULT);
66  *                 if (joinStatus == OperationStatus.SUCCESS) {
67  *                      // Do something with the key and data.
68  *                 } else {
69  *                     break;
70  *                 }
71  *             }
72  *         }
73  *     } finally {
74  *         if (cursor1 != null) {
75  *             cursor1.close();
76  *         }
77  *         if (cursor2 != null) {
78  *             cursor2.close();
79  *         }
80  *         if (joinCursor != null) {
81  *             joinCursor.close();
82  *         }
83  *     }
84  * </pre>
85  *
86  * <p>The join algorithm is described here so that its cost can be estimated and
87  * compared to other approaches for performing a query.  Say that N cursors are
88  * provided for the join operation. According to the order they appear in the
89  * array the cursors are labeled C(1) through C(n), and the keys at each cursor
90  * position are labeled K(1) through K(n).</p>
91  *
92  * <ol>
93  *
94  * <li>Using C(1), the join algorithm iterates sequentially through all records
95  * having K(1).  This iteration is equivalent to a {@link Cursor#getNextDup
96  * Cursor.getNextDup} operation on the secondary index.  The primary key of a
97  * candidate record is determined in this manner.  The primary record itself is
98  * not retrieved and the primary database is not accessed.</li>
99  *
100  * <li>For each candidate primary key found in step 1, a Btree lookup is
101  * performed using C(2) through C(n), in that order.  The Btree lookups are
102  * exact searches to determine whether the candidate record also contains
103  * secondary keys K(2) through K(n).  The lookups are equivalent to a {@link
104  * Cursor#getSearchBoth Cursor.getSearchBoth} operation on the secondary index.
105  * The primary record itself is not retrieved and the primary database is not
106  * accessed.</li>
107  *
108  * <li>If any lookup in step 2 fails, the algorithm advances to the next
109  * candidate record using C(1).  Lookups are performed in the order of the
110  * cursor array, and the algorithm proceeds to the next C(1) candidate key as
111  * soon as a single lookup fails.</li>
112  *
113  * <li>If all lookups in step 2 succeed, then the matching key and/or data is
114  * returned by the {@code getNext} method.  If the {@link
115  * #getNext(DatabaseEntry,DatabaseEntry,LockMode)} method signature is used,
116  * then the primary database is read to obtain the record data, as if {@link
117  * Cursor#getSearchKey Cursor.getSearchKey} were called for the primary
118  * database.  If the {@link #getNext(DatabaseEntry,LockMode)} method signature
119  * is used, then only the primary key is returned and the primary database is
120  * not accessed.</li>
121  *
122  * <li>The algorithm ends when C(1) has no more candidate records with K(1),
123  * and the {@code getNext} method will then return {@link
124  * com.sleepycat.je.OperationStatus#NOTFOUND OperationStatus.NOTFOUND}.</li>
125  *
126  * </ol>
127  */

141     JoinCursor(final Database primaryDb,
142                final Cursor[] cursors,
143                final JoinConfig configParam)
144         throws DatabaseException {
145 
146         priDb = primaryDb;
147         config = (configParam != null) ? configParam.clone()
148                                        : JoinConfig.DEFAULT;
149         scratchEntry = new DatabaseEntry();
150         firstSecKey = new DatabaseEntry();
151         cursorScratchEntries = new DatabaseEntry[cursors.length];
152         for (int i = 0; i < cursors.length; i += 1) {
153             cursorScratchEntries[i] = new DatabaseEntry();
154         }
155         cursorFetchedFirst = new boolean[cursors.length];
156         Cursor[] sortedCursors = new Cursor[cursors.length];
157         System.arraycopy(cursors, 0, sortedCursors, 0, cursors.length);
158 
159         if (!config.getNoSort()) {
160 
161             /*
162              * Sort ascending by duplicate count.  Collect counts before
163              * sorting so that countEstimate is called only once per cursor.
164              */
165             final long[] counts = new long[cursors.length];
166             for (int i = 0; i < cursors.length; i += 1) {
167                 counts[i] = cursors[i].countEstimateInternal();
168                 assert counts[i] >= 0;
169             }
170             Arrays.sort(sortedCursors, new Comparator<Cursor>() {
171                 public int compare(Cursor o1, Cursor o2) {
172                     long count1 = -1;
173                     long count2 = -1;
174 
175                     /*
176                      * Scan for objects in cursors not sortedCursors since
177                      * sortedCursors is being sorted in place.
178                      */
179                     for (int i = 0; i < cursors.length &&
180                                     (count1 < 0 || count2 < 0); i += 1) {
181                         if (cursors[i] == o1) {
182                             count1 = counts[i];
183                         } else if (cursors[i] == o2) {
184                             count2 = counts[i];
185                         }
186                     }
187                     assert count1 >= 0 && count2 >= 0;
188                     long cmp = count1 - count2;
189                     return (cmp < 0) ? (-1) : ((cmp > 0) ? 1 : 0);
190                 }
191             });
192         }
193 
194         /*
195          * Dup cursors last.  If an error occurs before the constructor is
196          * complete, close them and ignore exceptions during close.
197          */
198         try {
199             secCursors = new Cursor[cursors.length];
200             for (int i = 0; i < cursors.length; i += 1) {
201                 secCursors[i] = sortedCursors[i].dup(true);
202             }
203         } catch (DatabaseException e) {
204             close(e); /* will throw e */
205         }

385     /**
386      * Internal version of getNext(), with an optional data param.
387      * <p>
388      * Since duplicates are always sorted and duplicate-duplicates are not
389      * allowed, a natural join can be implemented by simply traversing through
390      * the duplicates of the first cursor to find candidate keys, and then
391      * looking for each candidate key in the duplicate set of the other
392      * cursors, without ever reseting a cursor to the beginning of the
393      * duplicate set.
394      * <p>
395      * This only works when the same duplicate comparison method is used for
396      * all cursors.  We don't check for that, we just assume the user won't
397      * violate that rule.
398      * <p>
399      * A future optimization would be to add a SearchMode.BOTH_DUPS operation
400      * and use it instead of using SearchMode.BOTH.  This would be the
401      * equivalent of the undocumented DB_GET_BOTHC operation used by DB core's
402      * join() implementation.
403      */
404     private OperationResult retrieveNext(final DatabaseEntry keyParam,
405                                          final DatabaseEntry dataParam,
406                                          final LockMode lockMode,
407                                          final CacheMode cacheMode) {
408         boolean readUncommitted =
409             secCursors[0].isReadUncommittedMode(lockMode);
410 
411         outerLoop: while (true) {
412 
413             /* Process the first cursor to get a candidate key. */
414             Cursor secCursor = secCursors[0];
415             DatabaseEntry candidateKey = cursorScratchEntries[0];
416             OperationResult result;
417             if (!cursorFetchedFirst[0]) {
418                 /* Get first duplicate at initial cursor position. */
419                 result = secCursor.getCurrentInternal(
420                     firstSecKey, candidateKey, lockMode, cacheMode);
421                 if (readUncommitted && result == null) {
422                     /* Deleted underneath read-uncommitted cursor; skip it. */
423                     cursorFetchedFirst[0] = true;
424                     continue;
425                 }
426                 cursorFetchedFirst[0] = true;
427             } else {
428                 /* Already initialized, move to the next candidate key. */
429                 result = secCursor.retrieveNext(
430                     firstSecKey, candidateKey, lockMode, cacheMode,
431                     GetMode.NEXT_DUP);
432             }
433             if (result == null) {
434                 /* No more candidate keys. */
435                 return null;
436             }
437 
438             /* Process the second and following cursors. */
439             for (int i = 1; i < secCursors.length; i += 1) {
440                 secCursor = secCursors[i];
441                 DatabaseEntry secKey = cursorScratchEntries[i];
442                 if (!cursorFetchedFirst[i]) {
443                     result = secCursor.getCurrentInternal(
444                         secKey, scratchEntry, lockMode, cacheMode);
445                     if (readUncommitted &&
446                         result == null) {
447                         /* Deleted underneath read-uncommitted; skip it. */
448                         result = secCursor.retrieveNext(
449                             secKey, scratchEntry, lockMode, cacheMode,
450                             GetMode.NEXT_DUP);
451                         if (result == null) {
452                             /* All keys were deleted; no possible match. */
453                             return null;
454                         }
455                     }
456                     cursorFetchedFirst[i] = true;
457                 }
458                 scratchEntry.setData(secKey.getData(), secKey.getOffset(),
459                                      secKey.getSize());
460                 result = secCursor.search(
461                     scratchEntry, candidateKey, lockMode, cacheMode,
462                     SearchMode.BOTH, true);
463                 if (result == null) {
464                     /* No match, get another candidate key. */
465                     continue outerLoop;
466                 }
467             }
468 
469             /* The candidate key was found for all cursors. */
470             if (dataParam != null) {
471                 if (!secCursors[0].readPrimaryAfterGet(
472                         priDb, firstSecKey, candidateKey, dataParam, lockMode,
473                         readUncommitted, false /*lockPrimaryOnly*/,
474                         false /*verifyPrimary*/,
475                         secCursors[0].getCursorImpl().getLocker(),
476                         secCursors[0].getDatabase(), null)) {
477                     /* Deleted underneath read-uncommitted cursor; skip it. */
478                     continue;
479                 }
480 
481                 /*
482                  * Copy primary info to all secondary cursors. The 0th cursor
483                  * was updated above with the primary info.
484                  */
485                 final CursorImpl firstSecCursor = secCursors[0].cursorImpl;
486                 for (int i = 1; i < secCursors.length; i += 1) {
487                     secCursors[i].cursorImpl.setPriInfo(firstSecCursor);
488                 }
489             }
490             keyParam.setData(candidateKey.getData(), candidateKey.getOffset(),
491                              candidateKey.getSize());
492             return result;
493         }
494     }
