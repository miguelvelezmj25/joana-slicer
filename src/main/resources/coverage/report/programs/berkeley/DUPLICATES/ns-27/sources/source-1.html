427      * Fills a dedicated empty buffer with bytes read from the channel. It
428      * flips the buffer after it has been filled, so it's ready for reading.
429      *
430      * @param channel the channel to be read
431      * @param buffer the buffer to be filled
432      * @throws IOException if the errors were encountered while reading from
433      * the channel or the buffer could not be filled with the expected number
434      * of bytes
435      */
436     private void fillBuffer(ReadableByteChannel channel,
437                             ByteBuffer buffer)
438         throws IOException {
439 
440         final long start = System.nanoTime();
441         assert(buffer.position() == 0);
442         while (buffer.position() < buffer.limit()) {
443             int numRead = channel.read(buffer);
444             if (numRead <= 0) {
445                 throw new IOException("Expected bytes: " + buffer.limit() +
446                                       " read bytes: " + buffer.position());
447             }
448         }
449         nReadNanos.add(System.nanoTime() - start);
450         buffer.flip();
451     }

453     /**
454      * Read and parse an incoming message, specifying the incoming version.
455      *
456      * @param channel the channel to read from. Declared as a
457      * ReadableByteChannel rather than the more obvious SocketChannel to
458      * facilitate unit testing.
459      *
460      * @throws IOException
461      */
462     public Message read(ReadableByteChannel channel)
463         throws IOException {
464 
465         /* Get the message header. */
466         fillBuffer(channel, header);
467 
468         /* Use the type value to determine the message type. */
469         MessageOp op = getOp(header);
470         try {
471             Constructor<? extends Message> cons = op.getConstructor();
472 
473             /* Read the size to determine the body of the message. */
474             int messageBodySize = LogUtils.readInt(header);
475             nBytesRead.add(MESSAGE_HEADER_SIZE + messageBodySize);
476             nMessagesRead.increment();
477             if (messageBodySize > 0) {
478                 if (messageBodySize > maxMessageSize) {
479                     throw EnvironmentFailureException.unexpectedState
480                         ("Message op: " + op + " Body size: " +
481                          messageBodySize + " is too large.  maxSizeAllowed: " +
482                          maxMessageSize +
483                          "\nIf a larger value is needed, set the " +
484                          "'je.rep.maxMessageSize' parameter.");
485                 }
486 
487                 ByteBuffer body = allocateReadBuffer(messageBodySize);
488                 fillBuffer(channel, body);
489                 return cons.newInstance(this, body);
490             }
491 
492             if (messageBodySize < 0) {
493                 throw EnvironmentFailureException.unexpectedState
494                     ("Message op: " + op + " Body size: " + messageBodySize);
495             }
496             /* No body */
497             return cons.newInstance(this, null);
498         } catch (InstantiationException e) {
499             throw EnvironmentFailureException.unexpectedException(e);
500         } catch (IllegalAccessException e) {
501             throw EnvironmentFailureException.unexpectedException(e);
502         } catch (SecurityException e) {
503             throw EnvironmentFailureException.unexpectedException(e);
504         } catch (InvocationTargetException e) {
505             throw EnvironmentFailureException.unexpectedException(e);
506         } finally {
507             /* The header buffer will be reused, so clear it. */
508             header.clear();
509         }
510     }

581     /**
582      * Buffers the serialized form of the message (if possible) for later
583      * writes to the network.
584      *
585      * If buffering the message would result in an overflow, the current
586      * contents of the buffer are flushed before the message is added to the
587      * buffer. If the size of the message exceeds the size of the buffer, the
588      * message is flushed directly to the network.
589      *
590      * @param channel the channel to which the buffer is flushed on buffer
591      * overflows.
592      *
593      * @param batchWriteBuffer the buffer accumulating the serialized messages
594      * It's best for performance if this is a direct byte buffer, to avoid
595      * another copy when the buffer is finally flushed.
596      *
597      * @param nMessages the number of messages currently in the buffer,
598      * including this message, that is, nMessages is always > 0 upon entry.
599      *
600      * @param message the message to be buffered.
601      *
602      * @return the number of messages currently in the buffer after accounting
603      * for potential buffer flushes.
604      */
605     public int bufferWrite(WritableByteChannel channel,
606                            ByteBuffer batchWriteBuffer,
607                            int nMessages,
608                            Message message)
609         throws IOException {
610 
611         assert nMessages > 0 ;
612 
613         final ByteBuffer messageBuffer = message.wireFormat();
614 
615         if (batchWriteBuffer.remaining() < messageBuffer.limit()) {
616 
617             flushBufferedWrites(channel, batchWriteBuffer, nMessages - 1);
618             /* 1 for the message we add below. */
619             nMessages = 1;
620 
621             if (batchWriteBuffer.remaining() < messageBuffer.limit()) {
622                 /*
623                  * Buffer is too small for message, so write it directly.
624                  * This write must always be preceded by a buffer flush.
625                  */
626                 assert batchWriteBuffer.position() == 0 ;
627                 nMessagesWritten.increment();
628                 flushBuffer(channel, messageBuffer);
629                 nMessages = 0;
630                 return nMessages;
631             }
632         }
633 
634         batchWriteBuffer.put(messageBuffer);
635         return nMessages;
636     }

666     /**
667      * Writes the entire contents of the buffer to the blocking channel.
668      */
669     private void flushBuffer(WritableByteChannel channel,
670                              ByteBuffer bb)
671         throws IOException {
672 
673         assert bb.position() == 0;
674 
675         if (bb.limit() == 0) {
676             return;
677         }
678 
679         final long start = System.nanoTime();
680 
681         /**
682          * Even though the underlying channel is a blocking channel, there are
683          * circumstances where the channel.write() returns without writing all
684          * of the buffer, in seeming contradiction to the javadoc. One such
685          * example is when the peer fails in the middle of a write; in this
686          * case there is a "successful" partial write and the next write will
687          * result in an I/O exception. There may be other such examples as
688          * well, hence the defensive write loop below which tries to continue
689          * writes to their logical conclusion or to an IOE.
690          */
691         while (bb.remaining() > 0) {
692             int bytes = channel.write(bb);
693             nBytesWritten.add(bytes);
694             if (bytes == 0) {
695 
696                 /*
697                  * This should not happen since it's a blocking channel, but
698                  * the java doc is vague on this subject, so yield control if
699                  * we are not making progress.
700                  */
701                 Thread.yield();
702             }
703         }
704 
705         nWriteNanos.add(System.nanoTime() - start);
706     }

767      * utmost importance and reflection can be used to simplify message

769      */
770     protected abstract class SimpleMessage extends Message {
771 
772         /**
773          * Assembles a sequence of arguments into its byte based wire format.
774          * The method does the serialization in two steps. In step 1, it
775          * calculates the length of the buffer. In step 2, it assembles the
776          * bytes into the allocated buffer. The interpretive approach used here
777          * is suitable for the low performance requirements of the handshake
778          * protocol, but not for the core data stream itself. It's for this
779          * reason that the method is associated with the Handshake message
780          * class and not the Message class.
781          *
782          * @param arguments the arguments to be passed in the message
783          *
784          * @return a byte buffer containing the serialized form of the message
785          */
786         protected ByteBuffer wireFormat(Object... arguments) {
787             int size = 0;
788             for (final Object obj : arguments) {
789                 size += wireFormatSize(obj);
790             }
791 
792             /* Allocate the buffer and fill it up */
793             final ByteBuffer buffer = allocateInitializedBuffer(size);
794 
795             for (final Object obj : arguments) {
796                 putWireFormat(buffer, obj);
797             }
798             buffer.flip();
799             return buffer;
800         }
801 
802         /**
803          * Put the bytes of the wire format for the object into the current
804          * position in the buffer.
805          *
806          * @param buffer the buffer
807          * @param obj the object
808          * @throws EnvironmentFailureException if the object is not supported
809          */
810         protected void putWireFormat(final ByteBuffer buffer,
811                                      final Object obj) {
812             final Class<?> cl = obj.getClass();
813             if (cl == Long.class) {
814                 LogUtils.writeLong(buffer, ((Long)obj).longValue());
815             } else if (cl == Integer.class) {
816                 LogUtils.writeInt(buffer, ((Integer)obj).intValue());
817             } else if (cl == Short.class) {
818                 LogUtils.writeShort(buffer, ((Short)obj).shortValue());
819             } else if (cl == Byte.class) {
820                 buffer.put(((Byte)obj).byteValue());
821             }  else if (cl == Boolean.class) {
822                 buffer.put(((Boolean)obj).booleanValue() ?
823                             (byte)1 :
824                             (byte)0);
825             } else if (cl == VLSN.class) {
826                 LogUtils.writeLong(buffer, ((VLSN)obj).getSequence());
827             } else if (Enum.class.isAssignableFrom(cl)) {
828                 /* An enum is stored as it's identifier string. */
829                 Enum<?> e = (Enum<?>)obj;
830                 putString(e.name(), buffer);
831             } else if (cl == String.class) {
832 
833                 /*
834                  * A string is stored with its length followed by its
835                  * contents.
836                  */
837                 putString((String)obj, buffer);
838             } else if (cl == Double.class) {
839                 /* Treat a Double as a String. */
840                 putString(((Double) obj).toString(), buffer);
841             } else if (cl == String[].class) {
842                 String sa[] = (String[])obj;
843                 LogUtils.writeInt(buffer, sa.length);
844                 for (String element : sa) {
845                     putString(element, buffer);
846                 }
847             } else if (cl == byte[].class) {
848                 putByteArray(buffer, (byte[])obj);
849             } else if (obj instanceof NameIdPair) {
850                 /* instanceof used to accomodate ReadOnlyNameIdPair. */
851                 ((NameIdPair) obj).serialize(buffer, BinaryProtocol.this);
852             } else {
853                 throw EnvironmentFailureException.unexpectedState(
854                     "Unknown type: " + cl);
855             }
856         }
857 
858         /**
859          * Put the bytes for the wire format of the specified byte array into
860          * the current position in the buffer.
861          *
862          * @param buffer the buffer
863          * @param ba the byte array
864          */
865         protected void putByteArray(ByteBuffer buffer, byte[] ba) {
866             LogUtils.writeInt(buffer, ba.length);
867             buffer.put(ba);
868         }

871          * Return the wire format size of the specified object.
872          *
873          * @param obj the object
874          * @return the size
875          * @throws EnvironmentFailureException if the object is not supported
876          */
877         protected int wireFormatSize(final Object obj) {
878             final Class<?> cl = obj.getClass();
879             if (cl == Long.class) {
880                 return 8;
881             } else if (cl == Integer.class) {
882                 return 4;
883             } else if (cl == Short.class) {
884                 return 2;
885             } else if (cl == Byte.class) {
886                 return 1;
887             } else if (cl == Boolean.class) {
888                 return 1;
889             } else if (cl == VLSN.class) {
890                 return 8;
891             } else if (Enum.class.isAssignableFrom(cl)) {
892                 return stringSize(((Enum<?>)obj).name());
893             } else if (cl == String.class) {
894                return stringSize((String)obj);
895             } else if (cl == Double.class) {
896                return stringSize(((Double) obj).toString());
897             } else if (cl == String[].class) {
898                 int size = 4; /* array length */
899                 final String sa[] = (String[])obj;
900                 for (String element : sa) {
901                     size += stringSize(element);
902                 }
903                 return size;
904             } else if (cl == byte[].class) {
905                 return 4 + ((byte[])obj).length;
906             } else if (obj instanceof NameIdPair) {
907                 /* instanceof used to accomodate ReadOnlyNameIdPair. */
908                 return ((NameIdPair) obj).serializedSize(BinaryProtocol.this);
909             } else {
910                 throw EnvironmentFailureException.unexpectedState
911                     ("Unknown type: " + cl);
912             }
913         }
