111 /**
112  * The Cleaner is responsible for effectively garbage collecting the JE log.
113  * It selects the least utilized log file for cleaning (see FileSelector),
114  * reads through the log file (FileProcessor) and determines whether each entry
115  * is obsolete (no longer relevant) or active (referenced by the Btree).
116  * Entries that are active are migrated (copied) to the end of the log, and
117  * finally the cleaned file is deleted.
118  *
119  * The migration of active entries is a multi-step process that can be
120  * configured to operate in different ways.  Eviction and checkpointing, as
121  * well as the cleaner threads (FileProcessor instances) are participants in
122  * this process.  Migration may be immediate or lazy.
123  *
124  * Active INs are always migrated lazily, which means that they are marked
125  * dirty by the FileProcessor, and then logged later by an eviction or
126  * checkpoint.  Active LNs are always migrated immediately by the FileProcessor
127  * by logging them.
128  *
129  * When the FileProcessor is finished with a file, all lazy migration for that
130  * file is normally completed by the end of the next checkpoint, if not sooner
131  * via eviction.  The checkpoint/recovery mechanism will ensure that obsolete
132  * entries will not be referenced by the Btree.  At the end of the checkpoint,
133  * it is therefore safe to delete the log file.
134  *
135  * There is one exception to the above paragraph.  When attempting to migrate
136  * an LN, if the LN cannot be locked then we must retry the migration at a
137  * later time.  Also, if a database removal is in progress, we consider all
138  * entries in the database obsolete but cannot delete the log file until
139  * database removal is complete.  Such "pending" LNs and databases are queued
140  * and processed periodically during file processing and at the start of a
141  * checkpoint; see processPending().  In this case, we may have to wait for
142  * more than one checkpoint to occur before the log file can be deleted.  See
143  * FileSelector and the use of the pendingLNs and pendingDBs collections.
144  */

741     /**
742      * Updates log size stats and deletes unprotected reserved files in order
743      * to stay within disk limits.
744      *
745      * This method must be called frequently enough to maintain disk usage
746      * safely below the limits. For an HA env this is particularly important
747      * since we retain all reserved files until we approach the disk limits.
748      * For this, calling this method at least every CLEANER_BYTES_INTERVAL
749      * should suffice.
750      *
751      * It is also important to call this method based on a time interval
752      * when writing stops, to retry deletions when files are protected or the
753      * env is locked by read-only processes. For this, calling this method at
754      * least every CLEANER_WAKEUP_INTERVAL should suffice.
755      */
756     public void manageDiskUsage() {
757 
758         /* Fail loudly if the environment is invalid. */
759         env.checkIfInvalid();
760 
761         if (env.isMemOnly() || env.mayNotWrite() || !fileDeletionEnabled) {
762             return;
763         }
764 
765         /*
766          * Only one thread at a time can truncate the VLSNIndex head, request
767          * the environment lock, and update stats. This is a periodic action,
768          * so probe the lock to avoid blocking other cleaner threads.
769          */
770         if (!manageDiskUsageLock.tryLock()) {
771             return;
772         }
773 
774         try {
775             /* Periodically update the stats. */
776             freshenLogSizeStats();
777 
778             if (fileProtector.getNReservedFiles() > 0) {
779 
780                 boolean freshenStats = false;
781 
782                 if (env.isReplicated()) {
783                     /*
784                      * Reserved files are retained until we approach a disk
785                      * limit. Determine how many bytes we need to reclaim by
786                      * deleting reserved files. Add a reasonable value to stay
787                      * safely below the limits between cleaner wakeups. Note
788                      * that max(overage,shortage) may be negative, since
789                      * overage and shortage may be negative.
790                      */
791                     final long origBytesNeeded =
792                         ((maxDiskLimit > 0) ?
793                             Math.max(maxDiskOverage, freeDiskShortage) :
794                             freeDiskShortage) +
795                         Math.max(
796                             1L << 20,
797                             3 * cleanerBytesInterval);
798 
799                     /*
800                      * First try deleting files without truncating the
801                      * VLSNIndex head.
802                      */
803                     long bytesNeeded = origBytesNeeded;
804                     if (bytesNeeded > 0) {
805                         bytesNeeded = deleteUnprotectedFiles(bytesNeeded);
806                     }
807 
808                     /*
809                      * If we still need space, try truncating the VLSNIndex
810                      * and then deleting files. See FileProtector for details.
811                      */
812                     if (bytesNeeded > 0 &&
813                         env.tryVlsnHeadTruncate(bytesNeeded)) {
814 
815                         bytesNeeded = deleteUnprotectedFiles(bytesNeeded);
816                         freshenStats = true;
817                     }
818 
819                     if (bytesNeeded < origBytesNeeded) {
820                         freshenStats = true;
821                     }
822                 } else {
823                     /*
824                      * For a non-HA env, simply try to delete all the reserved
825                      * files.
826                      */
827                     final long bytesNeeded =
828                         deleteUnprotectedFiles(Long.MAX_VALUE);
829 
830                     if (bytesNeeded < Long.MAX_VALUE) {
831                         freshenStats = true;
832                     }
833                 }
834 
835                 /*
836                  * Freshen the stats if any files are deleted, so write
837                  * operations can occur ASAP if they were previously
838                  * prohibited. Also freshen the stats if we truncated the
839                  * VLSNIndex, so that the stats reflect the current factors
840                  * gating file deletion.
841                  */
842                 if (freshenStats) {
843                     freshenLogSizeStats();
844                 }
845             }
846 
847             /*
848              * If there is still a violation, and we have not logged it since
849              * the violation status changed, then log it now. We do not expect
850              * the violation status to change frequently.
851              */
852             final String violation = diskUsageViolationMessage;
853             if (violation != null) {
854                 if (!loggedDiskLimitViolation) {
855                     LoggerUtils.logMsg(logger, env, Level.SEVERE, violation);
856                     loggedDiskLimitViolation = true;
857                 }
858             } else {
859                 loggedDiskLimitViolation = false;
860             }
861 
862         } catch (EnvLockedException e) {
863 
864             LoggerUtils.logMsg(
865                 logger, env, Level.SEVERE,
866                 "Could not delete files due to read-only processes. " +
867                     diskUsageMessage);
868 
869         } finally {
870             manageDiskUsageLock.unlock();
871         }
872     }

875     private static class EnvLockedException extends Exception {}
876 
877     /**
878      * Deletes unprotected reserved files in an attempt to free bytesNeeded.
879      * In a non-HA env, attempts to delete all reserved files, irrespective of
880      * bytesNeeded.
881      *
882      * <p>An exclusive environment is held while deleting the files to lock
883      * out read-only processes. The lock is held while deleting the reserved
884      * file records as well, but this is inexpensive and should not cause long
885      * delays for read-only processes.</p>
886      *
887      * @param bytesNeeded the amount of space we need to reclaim to stay within
888      * disk limits.
889      *
890      * @return number of bytes we could not reclaim due to protected files, or
891      * zero if we deleted files totaling bytesNeeded or more.
892      *
893      * @throws EnvLockedException if we can't get an exclusive environment
894      * lock because the env is locked by read-only processes, and therefore no
895      * files can be deleted.
896      */
897     private long deleteUnprotectedFiles(long bytesNeeded)
898         throws EnvLockedException {
899 
900         final FileManager fileManager = env.getFileManager();
901         final SortedSet<Long> deletedFiles = new TreeSet<>();
902 
903         if (!fileManager.lockEnvironment(false, true)) {
904             throw new EnvLockedException();
905         }
906         try {
907             long file = -1;
908 
909             while (bytesNeeded > 0 || !env.isReplicated()) {
910 
911                 final Pair<Long, Long> pair =
912                     fileProtector.takeCondemnedFile(file + 1);
913 
914                 if (pair == null) {
915                     break;
916                 }
917 
918                 file = pair.first();
919                 final long size = pair.second();
920 
921                 if (!deleteFile(file)) {
922                     /* Sometimes files cannot be deleted on Windows. */
923                     fileProtector.putBackCondemnedFile(file, size);
924                     continue;
925                 }
926 
927                 bytesNeeded = Math.max(0, bytesNeeded - size);
928                 profile.deleteReservedFileRecord(file);
929                 nCleanerDeletions.increment();
930                 deletedFiles.add(file);
931             }
932 
933         } finally {
934             fileManager.releaseExclusiveLock();
935 
936             if (!deletedFiles.isEmpty()) {
937 
938                 final StringBuilder sb = new StringBuilder(
939                     "Cleaner deleted files:");
940 
941                 for (final Long file : deletedFiles) {
942                     sb.append(" 0x");
943                     sb.append(Long.toHexString(file));
944                 }
945 
946                 LoggerUtils.traceAndLog(
947                     logger, env, Level.INFO, sb.toString());
948             }
949         }
950 
951         return bytesNeeded;
952     }

954     /**
955      * Attempts to delete the file and returns whether it has been deleted.
956      */
957     private boolean deleteFile(final Long file) {
958         final FileManager fileManager = env.getFileManager();
959 
960         final String expungeLabel = expunge ? "delete" : "rename";
961         final String expungedLabel = expungeLabel + "d";
962 
963         try {
964             if (expunge) {
965                 if (fileManager.deleteFile(file)) {
966                     return true;
967                 }
968             } else {
969                 /* See EnvironmentConfig.CLEANER_EXPUNGE. */
970 
971                 final File newFile = fileManager.renameFile(
972                     file, FileManager.DEL_SUFFIX,
973                     useDeletedDir ? DELETED_SUBDIR : null);
974 
975                 if (newFile != null) {
976                     newFile.setLastModified(System.currentTimeMillis());
977                     return true;
978                 }
979             }
980         } catch (IOException e) {
981             throw new EnvironmentFailureException(
982                 env, EnvironmentFailureReason.LOG_WRITE,
983                 "Unable to " + expungeLabel + " " + file, e);
984         }
985 
986         /*
987          * If the file is not valid (missing) then the file was previously
988          * deleted. This can occur on Windows when we retry deletion (see
989          * below).
990          */
991         if (!fileManager.isFileValid(file)) {
992             return true;
993         }
994 
995         /*
996          * Log a message and return false to retry the deletion later. The
997          * deletion is known to fail on Windows, and probably this occurs whe
998          * the file was recently closed.
999          */
1000         LoggerUtils.traceAndLog(
1001             logger, env, Level.WARNING,
1002             "Log file 0x" + Long.toHexString(file) + " could not be " +
1003                 expungedLabel + ". The deletion will be retried later.");
1004 
1005         return false;
1006     }
