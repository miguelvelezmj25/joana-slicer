30 /**
31  * Provides Dispatcher service handshake logic, including identification of the
32  * service to be accessed and authentication for access to the service.
33  *
34  * The service initialization protocol looks like one of these flows.  In the
35  * flows, the notation byte(X) indicates that X is a 1-byte network field and
36  * 2byte(X) indicates that X is a 2-byte network field.
37  *
38  * If No authentication is configured:
39  *   Client                                         Dispatcher
40  *       ->  "Service:" + byte(len) + <service name>
41  *       <-  byte(OK|BUSY|UNKNOWN_SERVICE)
42  *
43  * If authentication is configured:
44  *   Client                                         Dispatcher
45  *       ->  "Service:" + byte(len) + <service name>
46  *       <-  byte(AUTHENTICATE)
47  *       ->  "Authenticate:" + byte(<len>) + <mechanism list>
48  *       <-  "Mechanism:" + 2byte(<len>) + <server-selected mech name> +
49  *                        ":" + <auth mech params>
50  *       ->  <Mechanism-specific payload>
51  *       <-  byte(OK|BUSY|UNKNOWN_SERVICE|INVALID)
52  *
53  * The code here is organized into client-side and server-side functions.
54  * Client-side code is presumed to operated in blocking mode and server-side
55  * code is presumed to operate in non-blocking mode.
56  */

129          * Process the handshake incrementally, as channel is presumed to be
130          * in non-blocking mode.
131          *
132          * This function returns one of the following:
133          *    InitResult.READ: handshake is not complete; more data needed
134          *    InitResult.FAIL: handshake was rejected or otherwise failed
135          *      the underlying data channel has been closed.
136          *    InitResult.DONE: handshake is complete, and any authentication
137          *      has been performed.  A Response still needs to be sent to
138          *      the client.  Caller must do this after any additional validation
139          *      that may be required.
140          */
141         InitResult process() throws IOException {
142             final InitResult result = currentOp.processOp(channel);
143             if (result != InitResult.DONE) {
144                 return result;
145             }
146 
147             if (channel.isTrustCapable()) {
148                 if (channel.isTrusted()) {
149                     /*
150                      * Authentication of the requester has been handled at the
151                      * DataChannel level, so no need to worry further about
152                      * authentication.
153                      */
154                     return InitResult.DONE;
155                 }
156                 logMsg(Level.WARNING,
157                        false, // noteError
158                        "DataChannel is trust-capable but is not trusted");
159 
160                 /*
161                  * Defer rejecting the connection until the
162                  * RequireAuthenticateOp step, in case there is an alternate
163                  * authentication mechanism.
164                  */
165             }
166 
167             if (currentOp instanceof RequireAuthenticateOp ||
168                 authInfo == null) {
169                 /*
170                  * Either we've just successfully completed authentication or
171                  * no authentication is required.
172                  */
173                 return InitResult.DONE;
174             }
175 
176             /* Initiate the authentication step. */
177             currentOp = new RequireAuthenticateOp(this, authInfo);
178             return currentOp.processOp(channel);
179         }

243          *
244          * This function returns one of the following:
245          *    InitResult.READ: operation is not complete; more data needed
246          *    InitResult.FAIL: operation was rejected or otherwise failed

253         /**
254          * Helper function to read until the buffer is full.
255          */
256         protected InitResult fillBuffer(DataChannel channel, ByteBuffer buffer)
257             throws IOException {
258 
259             while (buffer.remaining() > 0) {
260                 final int count = channel.read(buffer);
261                 if (count < 0) {
262                     /* Premature EOF */
263                     initState.logMsg(Level.WARNING,
264                                      true, // noteError
265                                      "Premature EOF on channel: " + channel +
266                                      ", service: " +
267                                      initState.getServiceName() +
268                                      " - read() returned: " + count);
269                     closeChannel(channel);
270                     return InitResult.FAIL;
271                 }
272                 if (count == 0) {
273                     return InitResult.READ;
274                 }
275             }
276             return InitResult.DONE;
277         }
278 
279         /**
280          * Helper function to write the contents of the buffer.
281          * We make the simplifying assumption here that we will be able
282          * to write as much as we want (within reason).
283          */
284         protected InitResult sendBuffer(DataChannel channel, ByteBuffer buffer)
285             throws IOException {
286             int tryCount = 0;
287             while (buffer.remaining() > 0) {
288                 final int count = channel.write(buffer);
289                 if (count < 0) {
290                     /* Premature EOF */
291                     initState.logMsg(Level.WARNING,
292                                      true, // noteError
293                                      "Premature EOF on channel: " +
294                                      channel + " write() returned: " +
295                                      count);
296                     closeChannel(channel);
297                     return InitResult.FAIL;
298                 } else if (count == 0) {
299                     tryCount++;
300                     if (tryCount > CHANNEL_WRITE_ATTEMPT_LIMIT) {
301                         initState.logMsg(Level.WARNING,
302                                          true, // noteError
303                                          "Failed to write to channel. " +
304                                          "Send buffer size: " +
305                                          channel.getSocketChannel().socket().
306                                          getSendBufferSize());
307                         throw EnvironmentFailureException.unexpectedState(
308                             "Failed to write to channel");
309                     }
310                 } else {
311                     tryCount = 0;
312                 }
313             }
314             return InitResult.DONE;
315         }

380      * Operations to communicate the requested service name.  This is the
381      * initial portion of the message and if we have a connection from a
382      * foreign entity, this is most likely to detect an error.

385     /* The prefix for a service request. */
386     private static final String REQUEST_PREFIX = "Service:";
387     private static final byte[] REQUEST_PREFIX_BYTES =
388         StringUtils.toASCII(REQUEST_PREFIX);
389 
390     /**
391      * Server-side: expect client to provide a service name.
392      * Expected data format:
393      *    Literal: "Service:" in ASCII encoding
394      *    Length: 1 byte
395      *    Service Name: <Length> bytes in ASCII encoding
396      */
397     static class ReceiveNameOp extends ServerInitOp {
398 
399         /*
400          * The initial size is the prefix plus the byte that holds the
401          * length of the service name.
402          */
403         private final int INITIAL_BUFFER_SIZE = REQUEST_PREFIX_BYTES.length+1;
404 
405         private ByteBuffer buffer;
406 
407         ReceiveNameOp(ServerHandshake initState) {
408             super(initState);
409             this.buffer = ByteBuffer.allocate(INITIAL_BUFFER_SIZE);
410         }
411 
412         @Override
413         protected InitResult processOp(DataChannel channel) throws IOException {
414             InitResult readResult = fillBuffer(channel, buffer);
415             if (readResult != InitResult.DONE) {
416                 return readResult;
417             }
418             buffer.flip();
419             if (buffer.capacity() == INITIAL_BUFFER_SIZE) {
420                 /*
421                  * We've received exactly enough data to contain:
422                  * "Service:"<Length>
423                  * Make sure that it has the right format.
424                  */
425                 final String prefix = StringUtils.fromASCII
426                     (buffer.array(), 0, REQUEST_PREFIX.length());
427                 if (!prefix.equals(REQUEST_PREFIX)) {
428                     initState.logMsg
429                         (Level.WARNING,
430                          true, // noteError
431                          "Malformed service request: " + prefix);
432                     channel.write(Response.FORMAT_ERROR.byteBuffer());
433                     closeChannel(channel);
434                     return InitResult.FAIL;
435                 }
436 
437                 /* Enlarge the buffer to read the service name as well */
438                 final int nameLength = buffer.get(INITIAL_BUFFER_SIZE-1);
439                 if (nameLength <= 0) {
440                     initState.logMsg
441                         (Level.WARNING,
442                          true, // noteError
443                          "Bad service service name length: " + nameLength);
444                     channel.write(Response.FORMAT_ERROR.byteBuffer());
445                     closeChannel(channel);
446                     return InitResult.FAIL;
447                 }
448                 final ByteBuffer newBuffer =
449                     ByteBuffer.allocate(INITIAL_BUFFER_SIZE + nameLength);
450                 newBuffer.put(buffer);
451                 buffer = newBuffer;
452 
453                 /*
454                  * Recursive call to get the service name
455                  */
456                 return processOp(channel);
457             }
458 
459             /*
460              * If we made it here, we have a complete service request
461              * message. Extract the service name from the buffer
462              */
463             final String request = StringUtils.fromASCII(buffer.array());
464             initState.setServiceName(
465                 request.substring(REQUEST_PREFIX.length()+1));
466             return InitResult.DONE;
467         }
468     }

557     /**
558      * Server side operation to communicate the need for authentication
559      * to the client, negotiate an authentication mechanism and then
560      * complete the authentication.
561      */
562     static class RequireAuthenticateOp extends ServerInitOp {
563 
564         private final AuthenticationMethod[] authInfo;
565         private ExpectAuthRequestOp expectRequestOp;
566         private ServerInitOp authOp;
567 
568         RequireAuthenticateOp(ServerHandshake initState,
569                               AuthenticationMethod[] authInfo) {
570             super(initState);
571             this.authInfo = authInfo;
572         }
573 
574         @Override
575         protected InitResult processOp(DataChannel channel) throws IOException {
576 
577             if (expectRequestOp == null) {
578                 /* Tell the other end that they need to authenticate */
579                 Response response = Response.AUTHENTICATE;
580                 InitResult writeResult = sendBuffer(channel,
581                                                     response.byteBuffer());
582                 if (writeResult != InitResult.DONE) {
583                     return writeResult;
584                 }
585                 expectRequestOp =
586                     new ExpectAuthRequestOp(initState, authInfo);
587             }
588 
589             if (authOp == null) {
590                 /* Waiting for the authentication initiation request */
591                 final InitResult readResult =
592                     expectRequestOp.processOp(channel);
593                 if (readResult != InitResult.DONE) {
594                     return readResult;
595                 }
596 
597                 AuthenticationMethod selectedAuth =
598                     expectRequestOp.getSelectedAuthentication();
599 
600                 if (selectedAuth == null) {
601                     selectedAuth = new NoMatchAuthentication();
602                 }
603                 authOp = selectedAuth.getServerOp(initState);
604 
605                 /* Prepare the response message */
606                 final String authResponseStr = selectedAuth.getMechanismName() +
607                     ":" + selectedAuth.getServerParams();
608                 final byte[] authResponseBytes =
609                     StringUtils.toASCII(authResponseStr);
610                 final int length = 1 + AUTH_MECH_PREFIX_BYTES.length +
611                     2 + authResponseBytes.length;
612 
613                 final ByteBuffer buffer = ByteBuffer.allocate(length);
614                 buffer.put(Response.PROCEED.byteBuffer());
615                 buffer.put(AUTH_MECH_PREFIX_BYTES);
616                 putShort(buffer, (short) authResponseBytes.length);
617                 buffer.put(authResponseBytes);
618 
619                 buffer.flip();
620 
621                 /* Send the response message */
622                 final InitResult writeResult = sendBuffer(channel, buffer);
623                 if (writeResult != InitResult.DONE) {
624                     return writeResult;
625                 }
626             }
627 
628             return authOp.processOp(channel);
629         }

634      * in response to a message from the server notifying us that authentication
635      * is required.  The implementation of processOp() aggregates functionality
636      * from SendRequestOp, which tells the server what authentication we
637      * can provide, and then when we've agreed with the server on how to
638      * communicate, another ClientOp is created that provides the actual
639      * authentication implementation.

641     static class DoAuthenticateOp extends ClientInitOp {
642 
643         private final AuthenticationMethod[] authInfo;
644 
645         DoAuthenticateOp(ClientHandshake initState,
646                          AuthenticationMethod[] authInfo) {
647             super(initState);
648             this.authInfo = authInfo;
649         }
650 
651         @Override

653 
654             /*
655              * Tell the server that we want to authenticate ourselves, and
656              * negotiate with the server how that will happen.
657              */
658             final SendRequestOp sendOp = new SendRequestOp(initState, authInfo);
659             final InitResult sendResult = sendOp.processOp(ioAdapter);
660             if (sendResult != InitResult.DONE) {
661                 return sendResult;
662             }
663

666              * actual authentication steps.
667              */
668             final ClientInitOp authOp = sendOp.getAuthOp();
669             final InitResult authResult = authOp.processOp(ioAdapter);
670             if (authResult == InitResult.DONE) {
671                 setResponse(authOp.getResponse());
672             }
673             return authResult;
674         }

714         @Override
715         protected InitResult processOp(DataChannel channel) throws IOException {
716             InitResult readResult = fillBuffer(channel, buffer);
717             if (readResult != InitResult.DONE) {
718                 return readResult;
719             }
720 
721             if (phase == WAIT_FOR_REQUEST) {
722                 final String prefix = StringUtils.fromASCII
723                     (buffer.array(), 0, AUTH_PREFIX.length());
724                 if (!prefix.equals(AUTH_PREFIX)) {
725                     initState.logMsg(Level.WARNING,
726                                      true, // noteError
727                                      "Malformed authentication request: " +
728                                      prefix);
729                     sendBuffer(channel, Response.FORMAT_ERROR.byteBuffer());
730                     closeChannel(channel);
731                     return InitResult.FAIL;
732                 }
733 
734                 /* Get the length of auth mechanism list */
735                 buffer.clear();
736                 buffer.limit(1);
737                 phase = WAIT_FOR_LIST_SIZE;
738 
739                 readResult = fillBuffer(channel, buffer);
740                 if (readResult != InitResult.DONE) {
741                     return readResult;
742                 }
743             }
744 
745             if (phase == WAIT_FOR_LIST_SIZE) {
746                 buffer.flip();
747                 final int mechListSize = buffer.get();
748                 if (mechListSize < 0) {
749                     initState.logMsg(Level.WARNING,
750                                      true, // noteError
751                                      "Negative mechanism list size received: " +
752                                      mechListSize);
753                     sendBuffer(channel, Response.FORMAT_ERROR.byteBuffer());
754                     closeChannel(channel);
755                     return InitResult.FAIL;
756                 }
757 
758                 /* Now get the list itself */
759                 buffer = ByteBuffer.allocate(mechListSize);
760                 phase = WAIT_FOR_LIST;
761 
762                 readResult = fillBuffer(channel, buffer);
763                 if (readResult != InitResult.DONE) {
764                     return readResult;
765                 }
766             }
767 
768             if (phase != WAIT_FOR_LIST) {
769                 throw EnvironmentFailureException.unexpectedState(
770                     "Unexpected state: + phase");
771             }
772 
773             /*
774              * Get the mechanism list in string form and then split into
775              * constituent components.
776              */
777             final String mechListStr = StringUtils.fromASCII
778                 (buffer.array(), 0, buffer.capacity());
779             final String[] mechList = mechListStr.split(",");
780 
781             /* Find the first available match */
782             selectedAuth = findMatch(mechList, availableAuth);
783 
784             if (selectedAuth == null) {
785                 /* No acceptable mechanism found */
786                 initState.logMsg
787                     (Level.WARNING,
788                      true, // noteError
789                      "No acceptable authentication mechanism in list: " +
790                      mechListStr);
791                 sendBuffer(channel, Response.INVALID.byteBuffer());
792                 closeChannel(channel);
793                 return InitResult.FAIL;
794             }
795 
796             return InitResult.DONE;
797         }

816

818

821              * including reporting our available authentication mechanisms.
822              */
823             ioAdapter.write(serviceAuthenticateMessage());

826              * Wait for a response indicating whether to proceed with
827              * authentication
828              */
829             int result = ioAdapter.read(responseByte);
830             if (result < 0) {
831                 throw new IOException(
832                     "No service authenticate response byte: " + result);
833             }
834 
835             final Response response = Response.get(responseByte[0]);
836             setResponse(response);
837             if (response == null) {
838                 throw new IOException("Unexpected read response byte: " +
839                                       responseByte[0]);
840             }
841             if (response != Response.PROCEED) {
842                 return InitResult.FAIL;
843             }
844 
845             /*
846              * response is PROCEED, so find out what mechanism we are
847              * to use.
848              */
849             final byte[] mechPrefix =
850                 new byte[AUTH_MECH_PREFIX_BYTES.length + 2];
851             result = ioAdapter.read(mechPrefix);
852             if (result < 0) {
853                 throw new IOException(

855             }

857                              AUTH_MECH_PREFIX_BYTES.length)) {
858                 throw new IOException(
859                     "Unexpected service authenticate response: " +
860                     encodeBytes(mechPrefix));
861             }

863                                          AUTH_MECH_PREFIX_BYTES.length);
864             if (mechLen < 0) {
865                 throw new IOException(

867             }

870             result = ioAdapter.read(mechBytes);

872             if (result < 0) {
873                 throw new IOException(
874                     "EOF reading service authenticate mechanism: " +
875                     result);
876             }
877

880              * mechanism parameters, if required, separated by a ':'.
881              */
882             final String mechStr =
883                 StringUtils.fromASCII(mechBytes, 0, mechBytes.length);
884             final String[] mechList = mechStr.split(":");
885 
886             /*
887              * Get the authentication mechanism object based on the name.
888              */
889

891                 findMatch(new String[] { mechList[0] },
892                           authInfo);
893             if (selectedAuth == null) {
894                 throw new IOException(
895                     "Requested authentication mechanism not supported; " +
896                     mechList[0]);
897             }
898

900              * Then get the client-side operation from the authentication
901              * mechanism object.
902              */
903             authOp = selectedAuth.getClientOp(
904                 initState,
905                 mechList.length > 1 ? mechList[1] : "");
906 
907             return InitResult.DONE;
908         }

918          *
919          * @return the byte encoding of the authentication request message

922             final byte[] mechListBytes =
