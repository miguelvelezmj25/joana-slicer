28  * A rolling window of key/data pairs used by the ldiff algorithm.
29  */
30 public class Window {
31     private final Cursor cursor;
32     private List<byte[]> window;

43     /**
44      * Create a window of the given size, starting at the next record pointed
45      * at by the Cursor.
46      * 
47      * @param cursor an open cursor on the database being diff'd
48      * @param windowSize the number of records to include in the window
49      * @throws Exception
50      */
51     public Window(Cursor cursor, int windowSize)
52         throws Exception {
53 
54         this.cursor = cursor;
55         this.windowSize = windowSize;
56         /* To compute a MD5 hash for each block. */
57         try {
58             md = MessageDigest.getInstance("MD5");
59         } catch (NoSuchAlgorithmException e) {
60             e.printStackTrace();
61             throw new Exception("MD5 hashes are required for ldiff.");
62         }
63 
64         nextWindow();
65     }
66 
67     /**
68      * Roll a window forward by one key. The rolling checksum is adjusted to
69      * account for this move past one key/value pair. Note that the md5 hash
70      * associated with the window is computed on demand and is not updated
71      * here.
72      */
73     public void rollWindow()
74         throws Exception {
75 
76         DatabaseEntry key = new DatabaseEntry();
77         DatabaseEntry data = new DatabaseEntry();
78         if (cursor.getNext(key, data, LockMode.DEFAULT) ==
79             OperationStatus.SUCCESS) {
80             byte[] keyValue = LDiffUtil.concatByteArray(key.getData(),
81                                                         data.getData());
82             int removeXi = LDiffUtil.getXi(window.remove(0));
83             window.add(keyValue);
84             int addXi = LDiffUtil.getXi(keyValue);
85             rollChecksum(removeXi, addXi);
86         } else {
87             chksum = 0;
88         }
89         diffSize++;
90     }
91 
92     /**
93      * Advance the window to the next block of records and update the rolling
94      * checksum.
95      */
96     public void nextWindow() {
97         /* DatabaseEntry represents the key and data of each record. */
98         DatabaseEntry key = new DatabaseEntry();
99         DatabaseEntry data = new DatabaseEntry();
100         int i = 0;
101         window = new ArrayList<byte[]>();
102         diffSize = 0;
103 
104         /* Please pay attention to the check order in while loop. */
105         while ((i < windowSize) &&
106                 (cursor.getNext(key, data, LockMode.DEFAULT) ==
107                  OperationStatus.SUCCESS)) {
108             if (i == 0) {
109                 beginKey = key.getData();
110                 beginData = data.getData();
111             }
112             window.add(LDiffUtil.concatByteArray(key.getData(),
113                                                  data.getData()));
114             i++;
115         }
116 
117         setChecksum();
118     }

168     /**
169      * We use the rsync rolling checksum algorithm with the following changes:
170      * 
171      * 1. Each byte (Xi in the tech report) is replaced by a 32 bit Adler
172      * checksum of the bytes representing the concatenation of the key/value
173      * pair.
174      * 
175      * 2. The value for M is 64 instead of 32 to reduce the chances of false
176      * collisions on the rolling checksum, given our adaptation of the original
177      * algorithm to logically use 32 bit bytes.
178      */
179     private void setChecksum() {
180 
181         /* Adler32 to compute the rolling checksum of key/data pair. */
182         Adler32 adler32 = new Adler32();
183 
184         int a = 0, b = 0;
185         for (int i = 0; i < size(); i++) {
186             byte[] element = window.get(i);
187             adler32.reset();
188             adler32.update(element, 0, element.length);
189             final int xi = (int) adler32.getValue(); /* It's really 32 bits */
190             a += xi;
191             b += (xi * (size() - i));
192         }
193         chksum = (a & LDiffUtil.MASK_32BIT) | ((long) b << 32);
194     }
195 
196     /**
197      * Update the checksum by removing removeXi and adding addXi, according to
198      * the rsync algorithm.
199      * 
200      * @param removeXi
201      *            the value to remove from the checksum
202      * @param addXi
203      *            the value to add to the checksum
204      */
205     private void rollChecksum(int removeXi, int addXi) {
206         final int a = (int) chksum - removeXi + addXi;
207         final int b = (int) (chksum >> 32) - (removeXi * size()) + a;
208         chksum = (a & LDiffUtil.MASK_32BIT) | ((long) b << 32);
209     }
