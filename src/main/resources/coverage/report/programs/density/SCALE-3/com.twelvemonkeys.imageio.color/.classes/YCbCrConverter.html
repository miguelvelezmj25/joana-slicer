44  private final static int MAXJSAMPLE = 255; 45  private final static int CENTERJSAMPLE = 128; 46  private final static int ONE_HALF = 1 << (SCALEBITS - 1); 47  48  private final static int[] Cr_R_LUT = new int[MAXJSAMPLE + 1]; 52  53  /** 54  * Initializes tables for YCC->RGB color space conversion. 55  */ 56  private static void buildYCCtoRGBtable() { 57  if (ColorSpaces.DEBUG) { 58  System.err.println("Building YCC conversion table"); 59  } 60  61  for (int i = 0, x = -CENTERJSAMPLE; i <= MAXJSAMPLE; i++, x++) { 62  // i is the actual input pixel value, in the range 0..MAXJSAMPLE 63  // The Cb or Cr value we are thinking of is x = i - CENTERJSAMPLE 64  // Cr=>R value is nearest int to 1.40200 * x 65  Cr_R_LUT[i] = (int) ((1.40200 * (1 << SCALEBITS) + 0.5) * x + ONE_HALF) >> SCALEBITS;
